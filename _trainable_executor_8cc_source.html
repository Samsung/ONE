31TrainableExecutor::TrainableExecutor(
32 std::unique_ptr<compiler::train::LoweredTrainableGraph> lowered_graph,
33 backend::train::TrainableBackendContexts &&backend_contexts,
34 const compiler::train::TensorRegistries &tensor_regs,
35 compiler::train::TrainableCodeMap &&code_map,
36 const std::vector<ir::OperationIndex> &forward_order,
37 const std::vector<ir::OperationIndex> &backward_order, const util::TracingCtx *tracing_ctx,
38 const ir::train::LossInfo &loss_info)
39 : _code_map{std::move(code_map)}, _forward_order{std::move(forward_order)},
40 _backward_order{std::move(backward_order)}, _lowered_graph{std::move(lowered_graph)},
41 _backend_contexts{std::move(backend_contexts)},
42 _trainable_graph{_lowered_graph->trainable_graph()}, _tensor_regs{std::move(tensor_regs)},
43 _mutex(), _tracing_ctx(tracing_ctx), _loss_info(loss_info)
44{
45 auto build_tensor_list = [&](const auto &ind_seq, auto &tensors) {
46 assert(tensors.empty());
47 for (auto &&ind : ind_seq)
48 {
49 backend::ITensor *tensor = _tensor_regs.getITensor(ind);
50 assert(tensor != nullptr);
51 auto io_tensor = nnfw::misc::polymorphic_downcast<backend::builtin::IOTensor *>(tensor);
52 tensors.push_back(io_tensor);
53 }
54 };
55 build_tensor_list(_trainable_graph.getInputs(), _input_tensors);
56 build_tensor_list(_trainable_graph.getOutputs(), _output_tensors);
57}

65void TrainableExecutor::forward(const IODescription &desc, bool training)
66{
67 // For thread-safe, use mutex
68 // TODO: if all used backends on this executor are thread-safe,
69 // do not need to use mutex (otherwise, use mutex)
70 std::lock_guard<std::mutex> lock(_mutex);
71
72 // TODO Update IO tensors if desc has dynamic input
73 // Set input(s)
74 assert(_input_tensors.size() == desc.inputs.size());
75 for (uint32_t i = 0; i < _input_tensors.size(); ++i)
76 {
77 auto tensor = _input_tensors[i];
78
79 // TODO Check if (desc.inputs[i] == nullptr)
80 // TODO Better design for ITensor? (we need const_cast as ITensor is writable)
81 tensor->setUserTensor(static_cast<uint8_t *>(const_cast<void *>(desc.inputs[i]->buffer)),
82 desc.inputs[i]->size);
83 }
84
85 // Set output(s)
86 assert(_output_tensors.size() == desc.outputs.size());
87 for (uint32_t i = 0; i < _output_tensors.size(); ++i)
88 {
89 auto tensor = _output_tensors[i];
90
91 assert(desc.outputs[i] != nullptr);
92 if (desc.outputs[i]->buffer != nullptr && desc.outputs[i]->size != 0)
93 tensor->setUserTensor(static_cast<uint8_t *>(desc.outputs[i]->buffer), desc.outputs[i]->size);
94 }
95
96 forwardImpl(training);
97
98 // TODO Update output(s) desc if desc has dynamic input
99}

101void TrainableExecutor::forwardImpl(bool training)
102{
103 if (_tracing_ctx)
104 {
105 auto profiling_subg_index = _tracing_ctx->getSubgraphIndex(&_trainable_graph.graph());
106
107 _subject.notifySubgraphBegin(profiling_subg_index);
108 for (auto &&index : _forward_order)
109 {
110 const auto &code = _code_map.at(index);
111 const auto backend = code.lower_info->backend();
112// TODO : Move ruy profiler into ExecutionObserver
113#ifdef RUY_PROFILER
114 ruy::profiler::ScopeLabel label(code.op->name());
115#endif
116 _subject.notifyJobBegin(this, profiling_subg_index, code.op_ind, backend);
117
118 auto &tn_seq = code.tn_seq;
119 tn_seq->forward(training && code.op->isRequiredForBackward());
120
121 _subject.notifyJobEnd(this, profiling_subg_index, code.op_ind, backend);
122 }
123 _subject.notifySubgraphEnd(profiling_subg_index);
124 }
125 else
126 {
127 for (auto &&index : _forward_order)
128 {
129 const auto &code = _code_map.at(index);
130// TODO : Move ruy profiler into ExecutionObserver
131#ifdef RUY_PROFILER
132 ruy::profiler::ScopeLabel label(code.op->name());
133#endif
134 auto &tn_seq = code.tn_seq;
135 tn_seq->forward(training);
136 }
137 }
138}

150void TrainableExecutor::backwardImpl(uint32_t training_step)
151{
152 if (_tracing_ctx)
153 {
154 auto profiling_subg_index = _tracing_ctx->getSubgraphIndex(&_trainable_graph.graph());
155
156 _subject.notifySubgraphBegin(profiling_subg_index);
157 for (auto &&index : _backward_order)
158 {
159 const auto &code = _code_map.at(index);
160 if (!code.op->isRequiredForBackward())
161 {
162 continue;
163 }
164 const auto backend = code.lower_info->backend();
165// TODO : Move ruy profiler into ExecutionObserver
166#ifdef RUY_PROFILER
167 ruy::profiler::ScopeLabel label(code.op->name());
168#endif
169 _subject.notifyJobBegin(this, profiling_subg_index, code.op_ind, backend);
170
171 auto &tn_seq = code.tn_seq;
172 tn_seq->backward(training_step, code.op->isWeightsUpdateEnabled());
173
174 _subject.notifyJobEnd(this, profiling_subg_index, code.op_ind, backend);
175 }
176 _subject.notifySubgraphEnd(profiling_subg_index);
177 }
178 else
179 {
180 for (auto &&index : _backward_order)
181 {
182 const auto &code = _code_map.at(index);
183// TODO : Move ruy profiler into ExecutionObserver
184#ifdef RUY_PROFILER
185 ruy::profiler::ScopeLabel label(code.op->name());
186#endif
187 auto &tn_seq = code.tn_seq;
188 tn_seq->backward(training_step, code.op->isWeightsUpdateEnabled());
189 }
190 }

193float TrainableExecutor::getLoss(const ir::IOIndex &pred_io_ind) const
194{
195 const auto &loss_ind = _trainable_graph.getLossIndex(pred_io_ind);
196 if (loss_ind.undefined())
197 throw std::runtime_error{"Loss " + std::to_string(loss_ind.value()) + " is not defined."};
198 backend::ITensor *tensor = _tensor_regs.getITensor(loss_ind);
199 long double sum = 0;
200 for (uint64_t i = 0; i < tensor->getShape().num_elements(); ++i)
201 {
202 sum += reinterpret_cast<float *>(tensor->buffer())[i];
203 }
204 if (_loss_info.reduction_type == ir::train::LossReductionType::SumOverBatchSize)
205 {
206 sum /= tensor->getShape().num_elements();
207 }
208 return static_cast<float>(sum);
209}
