50 def _verify_cfg(self, cfgparser):

59 def _is_available(self, driver):

61 return (self.cfgparser.has_option('onecc', driver) and self.cfgparser.getboolean(
62 'onecc', driver)) or (self.cfgparser.has_option('one-build', driver)
63 and self.cfgparser.getboolean('one-build', driver))

85 def run(self, working_dir, verbose=False):
86 # set environment
87 CFG_ENV_SECTION = 'Environment'
88 if self.cfgparser.has_section(CFG_ENV_SECTION):
89 for key in self.cfgparser[CFG_ENV_SECTION]:
90 os.environ[key] = self.cfgparser[CFG_ENV_SECTION][key]
91
92 section_to_run = []
93 for d in self.import_drivers + self.driver_sequence:
94 if self._is_available(d):
95 section_to_run.append(d)
96
97 for section in section_to_run:
98 options = ['--config', self.path, '--section', section]
99 if section == 'one-optimize' and self.optparser:
100 options += ['-O', self.opt]
101 if verbose:
102 options.append('--verbose')
103 driver_path = os.path.join(working_dir, section)
104 cmd = [driver_path] + options
105 oneutils.run(cmd)
