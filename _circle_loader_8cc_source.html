125 std::unique_ptr<ir::Graph> loadSubgraph(const circle::SubGraph *circle_subg) override
126 {
127 auto subg = std::make_unique<ir::Graph>();
128 // Load tensors
129 _tensor_to_operand.resize(circle_subg->tensors()->size());
130 for (flatbuffers::uoffset_t i = 0; i < circle_subg->tensors()->size(); ++i)
131 {
132 _tensor_to_operand[i] = loadOperand(circle_subg->tensors()->Get(i), *subg);
133 subg->operands().at(_tensor_to_operand[i]).setOriginIndex(ir::OriginIndex(i));
134 }
135 // Set inputs
136 for (const std::int32_t input_ind : *circle_subg->inputs())
137 {
138 subg->addInput(tensorIdxToOperandIdx(input_ind),
139 _tensor_names.at(_tensor_to_operand[input_ind]));
140 }
141 // Set outputs
142 for (const std::int32_t output_ind : *circle_subg->outputs())
143 {
144 subg->addOutput(tensorIdxToOperandIdx(output_ind),
145 _tensor_names.at(_tensor_to_operand[output_ind]));
146 }
147 // Create operations
148 for (const auto *op : *circle_subg->operators())
149 {
150 CircleLoader::loadOperation(op, *subg);
151 }
152
153 subg->verify();
154
155 return subg;
156 }

158 void loadOperation(const circle::Operator *op, ir::Graph &subg)
159 {
160 auto const builtin_op = getBuiltinOperator(op);
161
162 switch (builtin_op)
163 {
164 case circle::BuiltinOperator::BuiltinOperator_BATCH_MATMUL:
165 loadBatchMatMul(op, subg);
166 return;
167 case circle::BuiltinOperator::BuiltinOperator_INSTANCE_NORM:
168 loadInstanceNorm(op, subg);
169 return;
170 case circle::BuiltinOperator::BuiltinOperator_BCQ_FULLY_CONNECTED:
171 loadBCQFullyConnected(op, subg);
172 return;
173 case circle::BuiltinOperator::BuiltinOperator_BCQ_GATHER:
174 loadBCQGather(op, subg);
175 return;
176 case circle::BuiltinOperator::BuiltinOperator_RMS_NORM:
177 loadRmsNorm(op, subg);
178 return;
179 case circle::BuiltinOperator::BuiltinOperator_ROPE:
180 loadRoPE(op, subg);
181 return;
182 case circle::BuiltinOperator::BuiltinOperator_CALL:
183 loadCall(op, subg);
184 return;
185 case circle::BuiltinOperator::BuiltinOperator_RUN_MODEL:
186 loadRunModel(op, subg);
187 return;
188 case circle::BuiltinOperator::BuiltinOperator_CUSTOM:
189 loadCustom(op, subg);
190 return;
191 case circle::BuiltinOperator::BuiltinOperator_ATTENTION:
192 loadAttention(op, subg);
193 return;
194 default:
195 BaseLoader::loadOperation(op, subg);
196 return;
197 }
198 }

201void CircleLoader::loadBatchMatMul(const Operator *op, ir::Graph &subg)
202{
203 ir::OperandIndexSequence inputs;
204 ir::OperandIndexSequence outputs;
205
206 loadOperationIO(op, inputs, outputs);
207
208 ir::operation::BatchMatMul::Param param;
209 const auto *options = op->builtin_options_as_BatchMatMulOptions();
210
211 param.adj_x = options->adjoint_lhs();
212 param.adj_y = options->adjoint_rhs();
213
214 std::unique_ptr<ir::Operation> new_op(new ir::operation::BatchMatMul(inputs, outputs, param));
215 subg.addOperation(std::move(new_op));
216}

319void CircleLoader::loadRunModel(const Operator *op, ir::Graph &subg)
320{
321 ir::OperandIndexSequence inputs;
322 ir::OperandIndexSequence outputs;
323
324 loadOperationIO(op, inputs, outputs);
325
326 // As workaround, we assume that only tvn type model is used for RUN_MODEL
327 // So subgraph will have only one OP: Bulk
328 // We copy this OP into this subgraph (like model inlining)
329 //
330 // TODO Introduce RunModel operator type to support various model
331 // TODO Introduce ModelIndexManager to handle various case
332 // - ModelIndex mapping for nnpackage's multimodel
333 // - ModelIndex mapping for RUN_MODEL
334 // - Lazy loading for model index for RUN_MODEL
335 // - Model inlining on optimize phase (compile)
336 assert(!_file_path.empty());
337 auto model_base_path = std::filesystem::path(_file_path).parent_path();
338 auto *options = op->builtin_options_as_RunModelOptions();
339 auto location = options->location()->str();
340 // Multiple files can be specified as ';' separated string
341 auto model_path = (location.find(';') == std::string::npos) ? model_base_path / location
342 : std::filesystem::path(location);
343 auto extension_path = model_path.extension();
344 if (extension_path.empty())
345 {
346 throw std::runtime_error("Model path has no extension: " + model_path.string());
347 }
348 auto extension = extension_path.string();
349 auto type = extension.substr(1); // remove dot
350
351 auto model = onert::loader::loadModel(model_path.string(), type);
352 assert(model);
353
354 // Get 1st OP in 1st subgraph
355 auto &op_bulk = model->primary_subgraph()->operations().at(ir::OperationIndex{0});
356 // Check if it is Bulk operation
357 assert(op_bulk.opcode() == ir::OpCode::Bulk);
358 const auto bulk_op = dynamic_cast<const ir::operation::Bulk *>(&op_bulk);
359
360 std::unique_ptr<ir::Operation> new_op(new ir::operation::Bulk(inputs, outputs, bulk_op->param()));
361 subg.addOperation(std::move(new_op));
362}

364void CircleLoader::loadBCQUnembedding(const Operator *op, ir::Graph &subg)
365{
366 ir::OperandIndexSequence inputs;
367 ir::OperandIndexSequence outputs;
368
369 loadOperationIO(op, inputs, outputs);
370
371 ir::operation::BCQUnembedding::Param param;
372 if (op->custom_options() == nullptr)
373 {
374 throw std::runtime_error{"BCQUnembedding: empty option"};
375 }
376 else
377 {
378 const auto attr_map = getCustomOpAttrMap(op);
379 param.weights_hidden_size = attr_map["weights_hidden_size"].AsUInt32();
380 param.lsh_type = attr_map["lsh_type"].AsString().str();
381 param.lsh_choices = attr_map["lsh_choices"].AsInt32();
382 }
383
384 const auto fbn = loadOperationTo<ir::operation::BCQUnembedding>(op, subg, param);
385
386 if (fbn->getInputs().size() != 5)
387 {
388 throw std::runtime_error{"BCQUnembedding: NYI input - only support five inputs"};
389 }
390}

392void CircleLoader::loadCustom(const Operator *op, ir::Graph &subg)
393{
394 ir::OperandIndexSequence inputs;
395 ir::OperandIndexSequence outputs;
396
397 assert(op->custom_options_format() == CustomOptionsFormat::CustomOptionsFormat_FLEXBUFFERS &&
398 "Unsupported custom operation options format");
399
400 auto *op_code = _domain_model->operator_codes()->Get(op->opcode_index());
401 auto custom_op_name = op_code->custom_code()->str();
402
403 enum class BuiltinOP
404 {
405 BCQUnembedding,
406 };
407
408 // Mapping from custom op name string to BuiltinOP enum
409 std::map<std::string, BuiltinOP> builtin_map = {
410 {"BCQUnembedding", BuiltinOP::BCQUnembedding},
411 };
412
413 // If unknown circle custom op, pass to BaseLoader
414 if (builtin_map.find(custom_op_name) == builtin_map.end())
415 {
416 BaseLoader::loadOperation(op, subg);
417 return;
418 }
419
420 auto custom_op_id = builtin_map.at(custom_op_name);
421 switch (custom_op_id)
422 {
423 case BuiltinOP::BCQUnembedding:
424 loadBCQUnembedding(op, subg);
425 break;
426 default:
427 throw std::runtime_error{"CircleLoader: Circle Custom OP map is defined but operation loader "
428 "function is not defined"};
429 }
430
431 return;
432}
