122 std::unique_ptr<ir::Graph> loadSubgraph(const circle::SubGraph *circle_subg) override
123 {
124 auto subg = std::make_unique<ir::Graph>();
125 // Load tensors
126 _tensor_to_operand.resize(circle_subg->tensors()->size());
127 for (flatbuffers::uoffset_t i = 0; i < circle_subg->tensors()->size(); ++i)
128 {
129 _tensor_to_operand[i] = loadOperand(circle_subg->tensors()->Get(i), *subg);
130 subg->operands().at(_tensor_to_operand[i]).setOriginIndex(ir::OriginIndex(i));
131 }
132 // Set inputs
133 for (const std::int32_t input_ind : *circle_subg->inputs())
134 {
135 subg->addInput(tensorIdxToOperandIdx(input_ind),
136 _tensor_names.at(_tensor_to_operand[input_ind]));
137 }
138 // Set outputs
139 for (const std::int32_t output_ind : *circle_subg->outputs())
140 {
141 subg->addOutput(tensorIdxToOperandIdx(output_ind),
142 _tensor_names.at(_tensor_to_operand[output_ind]));
143 }
144 // Create operations
145 for (const auto *op : *circle_subg->operators())
146 {
147 CircleLoader::loadOperation(op, *subg);
148 }
149
150 subg->verify();
151
152 return subg;
153 }

155 void loadOperation(const circle::Operator *op, ir::Graph &subg)
156 {
157 auto const builtin_op = getBuiltinOperator(op);
158
159 switch (builtin_op)
160 {
161 case circle::BuiltinOperator::BuiltinOperator_BATCH_MATMUL:
162 loadBatchMatMul(op, subg);
163 return;
164 case circle::BuiltinOperator::BuiltinOperator_INSTANCE_NORM:
165 loadInstanceNorm(op, subg);
166 return;
167 case circle::BuiltinOperator::BuiltinOperator_BCQ_FULLY_CONNECTED:
168 loadBCQFullyConnected(op, subg);
169 return;
170 case circle::BuiltinOperator::BuiltinOperator_BCQ_GATHER:
171 loadBCQGather(op, subg);
172 return;
173 case circle::BuiltinOperator::BuiltinOperator_RMS_NORM:
174 loadRmsNorm(op, subg);
175 return;
176 case circle::BuiltinOperator::BuiltinOperator_ROPE:
177 loadRoPE(op, subg);
178 return;
179 case circle::BuiltinOperator::BuiltinOperator_CALL:
180 loadCall(op, subg);
181 return;
182 case circle::BuiltinOperator::BuiltinOperator_CUSTOM:
183 loadCustom(op, subg);
184 return;
185 default:
186 BaseLoader::loadOperation(op, subg);
187 return;
188 }
189 }

192void CircleLoader::loadBatchMatMul(const Operator *op, ir::Graph &subg)
193{
194 ir::OperandIndexSequence inputs;
195 ir::OperandIndexSequence outputs;
196
197 loadOperationIO(op, inputs, outputs);
198
199 ir::operation::BatchMatMul::Param param;
200 const auto *options = op->builtin_options_as_BatchMatMulOptions();
201
202 param.adj_x = options->adjoint_lhs();
203 param.adj_y = options->adjoint_rhs();
204
205 std::unique_ptr<ir::Operation> new_op(new ir::operation::BatchMatMul(inputs, outputs, param));
206 subg.addOperation(std::move(new_op));
207}

310void CircleLoader::loadRunModel(const Operator *op, ir::Graph &subg)
311{
312 ir::OperandIndexSequence inputs;
313 ir::OperandIndexSequence outputs;
314
315 loadOperationIO(op, inputs, outputs);
316
317 // As workaround, we assume that only tvn type model is used for RUN_MODEL
318 // So subgraph will have only one OP: Bulk
319 // We copy this OP into this subgraph (like model inlining)
320 //
321 // TODO Introduce RunModel operator type to support various model
322 // TODO Introduce ModelIndexManager to handle various case
323 // - ModelIndex mapping for nnpackage's multimodel
324 // - ModelIndex mapping for RUN_MODEL
325 // - Lazy loading for model index for RUN_MODEL
326 // - Model inlining on optimize phase (compile)
327 assert(!_file_path.empty());
328 auto model_base_path = std::filesystem::path(_file_path).parent_path();
329 // TODO Get location from BuiltinOptions instead of custom OP option
330 auto location_obj = getCustomOpAttrMap(op)["location"];
331 if (location_obj.IsNull())
332 throw std::runtime_error("Cannot find 'location'");
333
334 auto model_path = model_base_path / location_obj.ToString();
335 auto extension_path = model_path.extension();
336 if (extension_path.empty())
337 {
338 throw std::runtime_error("Model path has no extension: " + model_path.string());
339 }
340 auto extension = extension_path.string();
341 auto type = extension.substr(1); // remove dot
342
343 auto model = onert::loader::loadModel(model_path.string(), type);
344 assert(model);
345
346 // Get 1st OP in 1st subgraph
347 auto &op_bulk = model->primary_subgraph()->operations().at(ir::OperationIndex{0});
348 // Check if it is Bulk operation
349 assert(op_bulk.opcode() == ir::OpCode::Bulk);
350 const auto bulk_op = dynamic_cast<const ir::operation::Bulk *>(&op_bulk);
351
352 std::unique_ptr<ir::Operation> new_op(new ir::operation::Bulk(inputs, outputs, bulk_op->param()));
353 subg.addOperation(std::move(new_op));
354}

356void CircleLoader::loadCustom(const Operator *op, ir::Graph &subg)
357{
358 ir::OperandIndexSequence inputs;
359 ir::OperandIndexSequence outputs;
360
361 assert(op->custom_options_format() == CustomOptionsFormat::CustomOptionsFormat_FLEXBUFFERS &&
362 "Unsupported custom operation options format");
363
364 auto *op_code = _domain_model->operator_codes()->Get(op->opcode_index());
365 auto custom_op_name = op_code->custom_code()->str();
366
367 if (custom_op_name == "RunModel")
368 loadRunModel(op, subg);
369 else
370 BaseLoader::loadOperation(op, subg);
371}
