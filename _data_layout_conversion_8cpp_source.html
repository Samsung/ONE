37coco::Copy *make_copy(coco::FeatureObject *from, coco::FeatureObject *into)
38{
39 auto m = from->module();
40 assert(m != nullptr);
41 assert(from->module() == into->module());
42
43 auto copy = m->entity()->instr()->create<coco::Copy>();
44
45 copy->from(from);
46 copy->into(into);
47
48 return copy;
49}

51coco::FeatureObject *clone_feature(const coco::FeatureObject *oldobj)
52{
53 auto module = oldobj->module();
54 auto newobj = module->entity()->object()->create<coco::FeatureObject>();
55 newobj->layout(coco::FeatureLayouts::BHWC::create(oldobj->shape()));
56
57 if (oldobj->bag() != nullptr)
58 {
59 using nncc::core::ADT::feature::num_elements;
60
61 // NOTE The size of bag should be at least "BxHxWxC" as "newobj" uses BHWC layout
62 const uint32_t batch = newobj->layout()->batch();
63 const uint32_t count = num_elements(newobj->layout()->shape());
64 const uint32_t bag_size = batch * count;
65
66 // Clone bag only when there is a backing bag for a given feature object
67 auto newbag = module->entity()->bag()->create(bag_size);
68 newobj->bag(newbag);
69 }
70
71 return newobj;
72}

79void insert_copy_before_load(coco::Load *load)
80{
81 assert(load->parent() != nullptr);
82 assert(load->parent()->parent() != nullptr);
83
84 if (auto obj = load->object())
85 {
86 if (auto ifm = obj->asFeature())
87 {
88 if (ifm->layout()->id() != coco::FeatureLayouts::BHWC::uid())
89 {
90 auto oldobj = ifm;
91 auto newobj = clone_feature(oldobj);
92
93 load->object(newobj);
94
95 auto copy = make_copy(oldobj, newobj);
96 copy->insertBefore(load->parent());
97 }
98 }
99 }
100}

185void change_conv2d_kernel_layout(coco::Conv2D *conv)
186{
187 auto m = conv->module();
188 assert(m != nullptr);
189 auto d = enco::data(enco::session(m));
190 assert(d != nullptr);
191
192 auto old_obj = conv->ker();
193 assert(old_obj != nullptr);
194 auto old_bag = old_obj->bag();
195 assert(old_bag != nullptr);
196
197 if (old_obj->layout()->id() == coco::KernelLayouts::NHWC::uid())
198 {
199 // Skip if kernel already uses NHWC layout
200 return;
201 }
202
203 const auto &ker_shape = old_obj->shape();
204
205 assert(d->allocated(old_bag));
206
207 auto new_bag = m->entity()->bag()->create(old_bag->size());
208 auto new_obj = m->entity()->object()->create<coco::KernelObject>();
209
210 new_obj->bag(new_bag);
211 new_obj->layout(coco::KernelLayouts::NHWC::create(ker_shape));
212
213 d->f32()->allocate(new_bag);
214
215 auto src = d->f32()->read(old_obj);
216 auto dst = d->f32()->access(new_obj);
217
218 const auto ker_N = ker_shape.count();
219 const auto ker_C = ker_shape.depth();
220 const auto ker_H = ker_shape.height();
221 const auto ker_W = ker_shape.width();
222
223 for (uint32_t n = 0; n < ker_N; ++n)
224 {
225 for (uint32_t ch = 0; ch < ker_C; ++ch)
226 {
227 for (uint32_t row = 0; row < ker_H; ++row)
228 {
229 for (uint32_t col = 0; col < ker_W; ++col)
230 {
231 dst->at(n, ch, row, col) = src->at(n, ch, row, col);
232 }
233 }
234 }
235 }
236
237 conv->ker(new_obj);
238 d->release(old_bag);
239}

251std::set<coco::Load *> loads(coco::Module *m)
252{
253 std::set<coco::Load *> res;
254
255 for (uint32_t n = 0; n < m->entity()->op()->size(); ++n)
256 {
257 auto op = m->entity()->op()->at(n);
258
259 // Skip if this op is dangling
260 if (op->parent() == nullptr)
261 {
262 continue;
263 }
264
265 // Skip if eval instruction of this op is dangling
266 if (op->parent()->parent() == nullptr)
267 {
268 continue;
269 }
270
271 if (auto load = m->entity()->op()->at(n)->asLoad())
272 {
273 res.insert(load);
274 }
275 }
276
277 return res;
278}

343void NormalizePass::runOnModule(coco::Module *m) const
344{
345 // Insert Copy before all Load Op (if necessary)
346 for (auto load : loads(m))
347 {
348 insert_copy_before_load(load);
349 }
350
351 // Insert Copy after all Eval Instr (if necessary)
352 for (auto eval : evals(m))
353 {
354 insert_copy_after_eval(eval);
355 }
356
357 // Change Kernel Layout of Conv2D opertion (if necessary)
358 for (auto conv : convs(m))
359 {
360 change_conv2d_kernel_layout(conv);
361 }
362
363 // Insert Copy (for Layout Conversion) before/after ANNDepthConcatF instructions (if necessary)
364 for (auto depth_concat : depth_concats(m))
365 {
366 convert_data_layout(depth_concat);
367 }
368}
