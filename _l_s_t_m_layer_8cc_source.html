31void KernelGenerator::visit(const ir::operation::LSTM &node)
32{
33 const auto scratch_buffer_index{
34 node.getOutputs().at(ir::operation::LSTM::Output::SCRATCH_BUFFER)};
35 const auto output_state_out_index{
36 node.getOutputs().at(ir::operation::LSTM::Output::OUTPUT_STATE_OUT)};
37 const auto cell_state_out_index{
38 node.getOutputs().at(ir::operation::LSTM::Output::CELL_STATE_OUT)};
39 const auto output_index{node.getOutputs().at(ir::operation::LSTM::Output::OUTPUT)};
40
41 const auto input_index{node.getInputs().at(ir::operation::LSTM::Input::INPUT)};
42 const auto input_to_input_weights_index{
43 node.getInputs().at(ir::operation::LSTM::Input::INPUT_TO_INPUT_WEIGHTS)}; // optional
44 const auto input_to_forget_weights_index{
45 node.getInputs().at(ir::operation::LSTM::Input::INPUT_TO_FORGET_WEIGHTS)};
46 const auto input_to_cell_weights_index{
47 node.getInputs().at(ir::operation::LSTM::Input::INPUT_TO_CELL_WEIGHTS)};
48 const auto input_to_output_weights_index{
49 node.getInputs().at(ir::operation::LSTM::Input::INPUT_TO_OUTPUT_WEIGHTS)};
50 const auto recurrent_to_input_weights_index{
51 node.getInputs().at(ir::operation::LSTM::Input::RECURRENT_TO_INPUT_WEIGHTS)}; // optional
52 const auto recurrent_to_forget_weights_index{
53 node.getInputs().at(ir::operation::LSTM::Input::RECURRENT_TO_FORGET_WEIGHTS)};
54 const auto recurrent_to_cell_weights_index{
55 node.getInputs().at(ir::operation::LSTM::Input::RECURRENT_TO_CELL_WEIGHTS)};
56 const auto recurrent_to_output_weights_index{
57 node.getInputs().at(ir::operation::LSTM::Input::RECURRENT_TO_OUTPUT_WEIGHTS)};
58 const auto cell_to_input_weights_index{
59 node.getInputs().at(ir::operation::LSTM::Input::CELL_TO_INPUT_WEIGHTS)}; // optional
60 const auto cell_to_forget_weights_index{
61 node.getInputs().at(ir::operation::LSTM::Input::CELL_TO_FORGET_WEIGHTS)}; // optional
62 const auto cell_to_output_weights_index{
63 node.getInputs().at(ir::operation::LSTM::Input::CELL_TO_OUTPUT_WEIGHTS)}; // optional
64 const auto input_gate_bias_index{
65 node.getInputs().at(ir::operation::LSTM::Input::INPUT_GATE_BIAS)};
66 const auto forget_gate_bias_index{
67 node.getInputs().at(ir::operation::LSTM::Input::FORGET_GATE_BIAS)};
68 const auto cell_gate_bias_index{node.getInputs().at(ir::operation::LSTM::Input::CELL_BIAS)};
69 const auto output_gate_bias_index{
70 node.getInputs().at(ir::operation::LSTM::Input::OUTPUT_GATE_BIAS)};
71 const auto projection_weights_index{
72 node.getInputs().at(ir::operation::LSTM::Input::PROJECTION_WEIGHTS)}; // optional
73 const auto projection_bias_index{
74 node.getInputs().at(ir::operation::LSTM::Input::PROJECTION_BIAS)}; // optional
75 const auto output_state_in_index{
76 node.getInputs().at(ir::operation::LSTM::Input::OUTPUT_STATE_IN)};
77 const auto cell_state_in_index{node.getInputs().at(ir::operation::LSTM::Input::CELL_STATE_IN)};
78 const auto time_major = node.param().time_major;
79
80 // NOTE The input_to_input_weights and the recurrent_to_input_weights do not exist in CIFG.
81 // has_input_to_input_weights && has_recurrent_to_input_weights: no CIFG
82 // !(has_input_to_input_weights && has_recurrent_to_input_weights): CIFG
83 // NOTE The cell_to_input_weights does not exist in non-peephole although regular LSTM(non-CIFG).
84 bool has_input_to_input_weights = _ctx.exist(input_to_input_weights_index) &&
85 (_ctx.at(input_to_input_weights_index).shape().dim(0) != 0 &&
86 _ctx.at(input_to_input_weights_index).shape().dim(1) != 0);
87 bool has_recurrent_to_input_weights =
88 _ctx.exist(recurrent_to_input_weights_index) &&
89 (_ctx.at(recurrent_to_input_weights_index).shape().dim(0) != 0 &&
90 _ctx.at(recurrent_to_input_weights_index).shape().dim(1) != 0);
91
92 // NOTE The cell_to_forget_weights and the cell_to_output_weights exist in peephole.
93 // But the cell_to_input_weights does not exist in regular CIFG although peephole.
94 // has_cell_to_forget_weights && has_cell_to_output_weights: peephole
95 // !(has_cell_to_forget_weights && has_cell_to_output_weights): no peephole
96 bool has_cell_to_forget_weights = _ctx.exist(cell_to_forget_weights_index) &&
97 _ctx.at(cell_to_forget_weights_index).shape().dim(0) != 0;
98 bool has_cell_to_output_weights = _ctx.exist(cell_to_output_weights_index) &&
99 _ctx.at(cell_to_output_weights_index).shape().dim(0) != 0;
100
101 bool has_input_gate_bias =
102 _ctx.exist(input_gate_bias_index) && _ctx.at(input_gate_bias_index).shape().dim(0);
103
104 bool has_projection_weights = _ctx.exist(projection_weights_index) &&
105 (_ctx.at(projection_weights_index).shape().dim(0) != 0 &&
106 _ctx.at(projection_weights_index).shape().dim(1) != 0);
107 bool has_projection_bias =
108 _ctx.exist(projection_bias_index) && _ctx.at(projection_bias_index).shape().dim(0);
109
110 auto scratch_buffer_tensor = _ctx.exist(scratch_buffer_index)
111 ? _tensor_reg->getPortableTensor(scratch_buffer_index)
112 : nullptr; // optional
113 auto output_state_out_tensor = _ctx.exist(output_state_out_index)
114 ? _tensor_reg->getPortableTensor(output_state_out_index)
115 : nullptr; // optional
116 auto cell_state_out_tensor = _ctx.exist(cell_state_out_index)
117 ? _tensor_reg->getPortableTensor(cell_state_out_index)
118 : nullptr; // optional
119 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
120
121 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
122
123 auto input_to_input_weights_tensor =
124 has_input_to_input_weights ? _tensor_reg->getPortableTensor(input_to_input_weights_index)
125 : nullptr; // optional
126 auto input_to_forget_weights_tensor =
127 _tensor_reg->getPortableTensor(input_to_forget_weights_index);
128 auto input_to_cell_weights_tensor = _tensor_reg->getPortableTensor(input_to_cell_weights_index);
129 auto input_to_output_weights_tensor =
130 _tensor_reg->getPortableTensor(input_to_output_weights_index);
131 auto recurrent_to_input_weights_tensor =
132 has_recurrent_to_input_weights
133 ? _tensor_reg->getPortableTensor(recurrent_to_input_weights_index)
134 : nullptr; // optional
135 auto recurrent_to_forget_weights_tensor =
136 _tensor_reg->getPortableTensor(recurrent_to_forget_weights_index);
137 auto recurrent_to_cell_weights_tensor =
138 _tensor_reg->getPortableTensor(recurrent_to_cell_weights_index);
139 auto recurrent_to_output_weights_tensor =
140 _tensor_reg->getPortableTensor(recurrent_to_output_weights_index);
141
142 auto cell_to_input_weights_tensor = _tensor_reg->getPortableTensor(cell_to_input_weights_index);
143 auto cell_to_forget_weights_tensor =
144 has_cell_to_forget_weights ? _tensor_reg->getPortableTensor(cell_to_forget_weights_index)
145 : nullptr; // optional
146 auto cell_to_output_weights_tensor =
147 has_cell_to_output_weights ? _tensor_reg->getPortableTensor(cell_to_output_weights_index)
148 : nullptr; // optional
149
150 auto input_gate_bias_tensor =
151 has_input_gate_bias ? _tensor_reg->getPortableTensor(input_gate_bias_index) : nullptr;
152 auto forget_gate_bias_tensor = _tensor_reg->getPortableTensor(forget_gate_bias_index);
153 auto cell_gate_bias_tensor = _tensor_reg->getPortableTensor(cell_gate_bias_index);
154 auto output_gate_bias_tensor = _tensor_reg->getPortableTensor(output_gate_bias_index);
155 auto output_state_in_tensor = _tensor_reg->getPortableTensor(output_state_in_index);
156 auto cell_state_in_tensor = _tensor_reg->getPortableTensor(cell_state_in_index);
157
158 auto projection_weights_tensor = has_projection_weights
159 ? _tensor_reg->getPortableTensor(projection_weights_index)
160 : nullptr; // optional
161 auto projection_bias_tensor = has_projection_bias
162 ? _tensor_reg->getPortableTensor(projection_bias_index)
163 : nullptr; // optional
164
165 IPortableTensor *input_layer_norm_weights_tensor = nullptr;
166 IPortableTensor *forget_layer_norm_weights_tensor = nullptr;
167 IPortableTensor *cell_layer_norm_weights_tensor = nullptr;
168 IPortableTensor *output_layer_norm_weights_tensor = nullptr;
169 if (node.getInputs().size() == 24)
170 {
171 const auto input_layer_norm_weights_index{
172 node.getInputs().at(ir::operation::LSTM::Input::INPUT_LAYER_NORMALIZATION_WEIGHTS)};
173 const auto forget_layer_norm_weights_index{
174 node.getInputs().at(ir::operation::LSTM::Input::FORGET_LAYER_NORMALIZATION_WEIGHTS)};
175 const auto cell_layer_norm_weights_index{
176 node.getInputs().at(ir::operation::LSTM::Input::CELL_LAYER_NORMALIZATION_WEIGHTS)};
177 const auto output_layer_norm_weights_index{
178 node.getInputs().at(ir::operation::LSTM::Input::OUTPUT_LAYER_NORMALIZATION_WEIGHTS)};
179
180 input_layer_norm_weights_tensor =
181 _tensor_reg->getPortableTensor(input_layer_norm_weights_index);
182 forget_layer_norm_weights_tensor =
183 _tensor_reg->getPortableTensor(forget_layer_norm_weights_index);
184 cell_layer_norm_weights_tensor = _tensor_reg->getPortableTensor(cell_layer_norm_weights_index);
185 output_layer_norm_weights_tensor =
186 _tensor_reg->getPortableTensor(output_layer_norm_weights_index);
187 }
188
189 auto fn = std::make_unique<ops::LSTMLayer>();
190
191 fn->configure(
192 input_tensor, input_to_input_weights_tensor, input_to_forget_weights_tensor,
193 input_to_cell_weights_tensor, input_to_output_weights_tensor, recurrent_to_input_weights_tensor,
194 recurrent_to_forget_weights_tensor, recurrent_to_cell_weights_tensor,
195 recurrent_to_output_weights_tensor, cell_to_input_weights_tensor, cell_to_forget_weights_tensor,
196 cell_to_output_weights_tensor, input_layer_norm_weights_tensor,
197 forget_layer_norm_weights_tensor, cell_layer_norm_weights_tensor,
198 output_layer_norm_weights_tensor,
199 /*aux_input=*/nullptr,
200 /*aux_input_to_input_weights=*/nullptr,
201 /*aux_input_to_forget_weights=*/nullptr,
202 /*aux_input_to_cell_weights=*/nullptr,
203 /*aux_input_to_output_weights=*/nullptr, input_gate_bias_tensor, forget_gate_bias_tensor,
204 cell_gate_bias_tensor, output_gate_bias_tensor, projection_weights_tensor,
205 projection_bias_tensor, output_state_in_tensor, cell_state_in_tensor, node.param(),
206 /*forward_sequence=*/true, time_major,
207 /*output_offset=*/0, scratch_buffer_tensor, output_state_out_tensor, cell_state_out_tensor,
208 output_tensor,
209 !_ctx.at(output_state_in_index).info().isVariable() /* means empty buffer on frontend now */,
210 !_ctx.at(cell_state_in_index).info().isVariable());
211
212 _return_fn = std::move(fn);
213}

222template <typename T>
223T *getOptionalOutputBuffer(onert::backend::IPortableTensor *tensor, std::vector<uint8_t> *temp_vec,
224 size_t total_size)
225{
226 if (tensor == nullptr)
227 {
228 temp_vec->reserve(total_size);
229 return reinterpret_cast<T *>(temp_vec->data());
230 }
231 else
232 {
233 assert(tensor->total_size() == total_size);
234 return getBuffer<T>(tensor);
235 }
236}

238inline void initializeStateBuffer(const onert::backend::IPortableTensor *tensor_in, void *buffer,
239 bool needs_memcpy)
240{
241 assert(tensor_in != nullptr);
242 assert(buffer != nullptr);
243 if (needs_memcpy)
244 memcpy(buffer, tensor_in->buffer(), tensor_in->total_size());
245 else
246 memset(buffer, 0, tensor_in->total_size());
247}

250void LSTMLayer::LSTMFloat()
251{
252 auto in_shape = _input->getShape();
253 assert(in_shape.rank() >= 2 && in_shape.rank() <= 3);
254 int max_time, n_batch;
255 if (in_shape.rank() == 3)
256 {
257 max_time = (_time_major) ? in_shape.dim(0) : in_shape.dim(1);
258 n_batch = (_time_major) ? in_shape.dim(1) : in_shape.dim(0);
259 }
260 else
261 {
262 max_time = 1;
263 n_batch = in_shape.dim(0);
264 }
265 const int n_input = in_shape.dim(_input->getShape().rank() - 1);
266 const int aux_input_size = 0;
267
268 // n_cell and n_output will be the same size when there is no projection.
269 const int n_cell = _input_to_output_weights->getShape().dim(0);
270 const int n_output = _recurrent_to_output_weights->getShape().dim(1);
271
272 // Since we have already checked that weights are all there or none, we can
273 // check the existence of only one to the get the condition.
274 const bool use_cifg = (_input_to_input_weights == nullptr);
275
276 // Optional outputs
277 float *output_state_buf = getOptionalOutputBuffer<float>(_output_state, &_output_state_vec,
278 _output_state_in->total_size());
279 float *cell_state_buf =
280 getOptionalOutputBuffer<float>(_cell_state, &_cell_state_vec, _cell_state_in->total_size());
281
282 initializeStateBuffer(_output_state_in, output_state_buf, _has_output_state_data);
283 initializeStateBuffer(_cell_state_in, cell_state_buf, _has_cell_state_data);
284
285 // Index the scratch buffers pointers to the global scratch buffer.
286 float *scratch_buffer_buf = getOptionalOutputBuffer<float>(
287 _scratch_buffer, &_scratch_vec, n_batch * n_cell * (use_cifg ? 3 : 4) * sizeof(float));
288 float *input_gate_scratch = nullptr;
289 float *cell_gate_scratch = nullptr;
290 float *forget_gate_scratch = nullptr;
291 float *output_gate_scratch = nullptr;
292 if (use_cifg)
293 {
294 cell_gate_scratch = scratch_buffer_buf;
295 forget_gate_scratch = scratch_buffer_buf + n_cell * n_batch;
296 output_gate_scratch = scratch_buffer_buf + 2 * n_cell * n_batch;
297 }
298 else
299 {
300 input_gate_scratch = scratch_buffer_buf;
301 cell_gate_scratch = scratch_buffer_buf + n_cell * n_batch;
302 forget_gate_scratch = scratch_buffer_buf + 2 * n_cell * n_batch;
303 output_gate_scratch = scratch_buffer_buf + 3 * n_cell * n_batch;
304 }
305
306 auto optional_tensor_ptr = [](const IPortableTensor *tensor) {
307 // If tensor is not given or the tensor size is 0, consider it was not given
308 return (tensor && tensor->total_size() > 0) ? getBuffer<float>(tensor) : nullptr;
309 };
310 // Optional inputs
311 const float *input_to_input_weights_ptr = optional_tensor_ptr(_input_to_input_weights);
312 const float *recurrent_to_input_weights_ptr = optional_tensor_ptr(_recurrent_to_input_weights);
313 const float *cell_to_input_weights_ptr = optional_tensor_ptr(_cell_to_input_weights);
314 const float *cell_to_forget_weights_ptr = optional_tensor_ptr(_cell_to_forget_weights);
315 const float *cell_to_output_weights_ptr = optional_tensor_ptr(_cell_to_output_weights);
316 const float *input_gate_bias_ptr = optional_tensor_ptr(_input_gate_bias);
317 const float *projection_weights_ptr = optional_tensor_ptr(_projection_weights);
318 const float *projection_bias_ptr = optional_tensor_ptr(_projection_bias);
319 const float *input_layer_norm_coefficients_ptr =
320 optional_tensor_ptr(_input_layer_norm_coefficients);
321 const float *forget_layer_norm_coefficients_ptr =
322 optional_tensor_ptr(_forget_layer_norm_coefficients);
323 const float *cell_layer_norm_coefficients_ptr =
324 optional_tensor_ptr(_cell_layer_norm_coefficients);
325 const float *output_layer_norm_coefficients_ptr =
326 optional_tensor_ptr(_output_layer_norm_coefficients);
327
328 // Copy out the LSTM specific params so they can be passed in the function.
329 nnfw::cker::LSTMParams lstm_params;
330 lstm_params.activation = convertActivationType(_params.activation);
331 lstm_params.cell_clip = _params.cell_threshold;
332 lstm_params.proj_clip = _params.projection_threshold;
333
334 auto out_shape = _output->getShape();
335 const int output_batch_leading_dim = out_shape.dim(out_shape.rank() - 1);
336 if (_time_major)
337 {
338 // Loop through the sequence.
339 const int input_step = n_batch * n_input;
340 const int output_step = n_batch * output_batch_leading_dim;
341 for (int t = 0; t < max_time; t++)
342 {
343 // If this is the forward_sequence, step forward, otherwise step
344 // backwards.
345 const int t_rel = _forward_sequence ? t : max_time - t - 1;
346 const float *input_ptr = getBuffer<float>(_input) + t_rel * input_step;
347 const float *aux_input_ptr = nullptr;
348 if (_aux_input)
349 {
350 aux_input_ptr = getBuffer<float>(_aux_input) + t_rel * input_step;
351 }
352 float *output_ptr = getBuffer<float>(_output) + t_rel * output_step + _output_offset;
353
354 LstmStepFloat(
355 input_ptr, input_to_input_weights_ptr, getBuffer<float>(_input_to_forget_weights),
356 getBuffer<float>(_input_to_cell_weights), getBuffer<float>(_input_to_output_weights),
357 aux_input_ptr,
358 /*aux_input_to_input_weights=*/nullptr,
359 /*aux_input_to_forget_weights=*/nullptr,
360 /*aux_input_to_cell_weights=*/nullptr,
361 /*aux_input_to_output_weights=*/nullptr, recurrent_to_input_weights_ptr,
362 getBuffer<float>(_recurrent_to_forget_weights),
363 getBuffer<float>(_recurrent_to_cell_weights),
364 getBuffer<float>(_recurrent_to_output_weights), cell_to_input_weights_ptr,
365 cell_to_forget_weights_ptr, cell_to_output_weights_ptr, input_layer_norm_coefficients_ptr,
366 forget_layer_norm_coefficients_ptr, cell_layer_norm_coefficients_ptr,
367 output_layer_norm_coefficients_ptr, input_gate_bias_ptr,
368 getBuffer<float>(_forget_gate_bias), getBuffer<float>(_cell_gate_bias),
369 getBuffer<float>(_output_gate_bias), projection_weights_ptr, projection_bias_ptr,
370 &lstm_params, n_batch, n_cell, n_input, aux_input_size, n_output, output_batch_leading_dim,
371 output_state_buf, cell_state_buf, input_gate_scratch, forget_gate_scratch,
372 cell_gate_scratch, output_gate_scratch, output_ptr);
373 }
374 }
375 else
376 {
377 for (int b = 0; b < n_batch; b++)
378 {
379 const int input_step = n_input;
380 const int output_step = output_batch_leading_dim;
381 for (int t = 0; t < max_time; t++)
382 {
383 // If this is the forward_sequence, step forward, otherwise step
384 // backwards.
385 const int t_rel = _forward_sequence ? t : max_time - t - 1;
386 const int time_offset = b * max_time + t_rel;
387 const float *input_ptr = getBuffer<float>(_input) + time_offset * input_step;
388 const float *aux_input_ptr = nullptr;
389 if (_aux_input)
390 {
391 aux_input_ptr = getBuffer<float>(_aux_input) + time_offset * input_step;
392 }
393 float *output_ptr = getBuffer<float>(_output) + time_offset * output_step + _output_offset;
394
395 // Offset the {output,cell}_state pointers to the right batch.
396 float *output_state_ptr = output_state_buf + b * output_batch_leading_dim;
397 float *cell_state_ptr = cell_state_buf + b * n_cell;
398 // Offset the scratch pointers to the right batch.
399 float *input_gate_scratch_ptr =
400 input_gate_scratch ? input_gate_scratch + b * n_cell : nullptr;
401 float *forget_gate_scratch_ptr = forget_gate_scratch + b * n_cell;
402 float *cell_gate_scratch_ptr = cell_gate_scratch + b * n_cell;
403 float *output_gate_scratch_ptr = output_gate_scratch + b * n_cell;
404
405 LstmStepFloat(
406 input_ptr, input_to_input_weights_ptr, getBuffer<float>(_input_to_forget_weights),
407 getBuffer<float>(_input_to_cell_weights), getBuffer<float>(_input_to_output_weights),
408 aux_input_ptr,
409 /*aux_input_to_input_weights=*/nullptr,
410 /*aux_input_to_forget_weights=*/nullptr,
411 /*aux_input_to_cell_weights=*/nullptr,
412 /*aux_input_to_output_weights=*/nullptr, recurrent_to_input_weights_ptr,
413 getBuffer<float>(_recurrent_to_forget_weights),
414 getBuffer<float>(_recurrent_to_cell_weights),
415 getBuffer<float>(_recurrent_to_output_weights), cell_to_input_weights_ptr,
416 cell_to_forget_weights_ptr, cell_to_output_weights_ptr, input_layer_norm_coefficients_ptr,
417 forget_layer_norm_coefficients_ptr, cell_layer_norm_coefficients_ptr,
418 output_layer_norm_coefficients_ptr, input_gate_bias_ptr,
419 getBuffer<float>(_forget_gate_bias), getBuffer<float>(_cell_gate_bias),
420 getBuffer<float>(_output_gate_bias), projection_weights_ptr, projection_bias_ptr,
421 &lstm_params, /*n_batch=*/1, n_cell, n_input, aux_input_size, n_output,
422 output_batch_leading_dim, output_state_ptr, cell_state_ptr, input_gate_scratch_ptr,
423 forget_gate_scratch_ptr, cell_gate_scratch_ptr, output_gate_scratch_ptr, output_ptr);
424 }
425 }
426 }
427}

429void LSTMLayer::configure(
430 const IPortableTensor *input, const IPortableTensor *input_to_input_weights,
431 const IPortableTensor *input_to_forget_weights, const IPortableTensor *input_to_cell_weights,
432 const IPortableTensor *input_to_output_weights, const IPortableTensor *recurrent_to_input_weights,
433 const IPortableTensor *recurrent_to_forget_weights,
434 const IPortableTensor *recurrent_to_cell_weights,
435 const IPortableTensor *recurrent_to_output_weights, const IPortableTensor *cell_to_input_weights,
436 const IPortableTensor *cell_to_forget_weights, const IPortableTensor *cell_to_output_weights,
437 const IPortableTensor *input_layer_norm_weights, const IPortableTensor *forget_layer_norm_weights,
438 const IPortableTensor *cell_layer_norm_weights, const IPortableTensor *output_layer_norm_weights,
439 const IPortableTensor *aux_input, const IPortableTensor *aux_input_to_input_weights,
440 const IPortableTensor *aux_input_to_forget_weights,
441 const IPortableTensor *aux_input_to_cell_weights,
442 const IPortableTensor *aux_input_to_output_weights, const IPortableTensor *input_gate_bias,
443 const IPortableTensor *forget_gate_bias, const IPortableTensor *cell_gate_bias,
444 const IPortableTensor *output_gate_bias, const IPortableTensor *projection_weights,
445 const IPortableTensor *projection_bias, const IPortableTensor *output_state_in,
446 const IPortableTensor *cell_state_in, const ir::operation::LSTM::Param &params,
447 bool forward_sequence, bool time_major, int output_offset, IPortableTensor *scratch_buffer,
448 IPortableTensor *output_state, IPortableTensor *cell_state, IPortableTensor *output,
449 bool has_output_state_data, bool has_cell_state_data)
450{
451 _input = input;
452 _input_to_input_weights = input_to_input_weights;
453 _input_to_forget_weights = input_to_forget_weights;
454 _input_to_cell_weights = input_to_cell_weights;
