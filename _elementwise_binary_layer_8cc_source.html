37template <typename T>
38void FloorDivGeneric(const IPortableTensor *lhs, const IPortableTensor *rhs,
39 IPortableTensor *output)
40{
41 if (!HaveSameShapes(lhs, rhs))
42 {
43 nnfw::cker::FloorDivBroadcast<T>(getShape(lhs), getBuffer<T>(lhs), getShape(rhs),
44 getBuffer<T>(rhs), getShape(output), getBuffer<T>(output));
45 }
46 else
47 {
48 nnfw::cker::FloorDivElementwise<T>(getShape(lhs), getBuffer<T>(lhs), getBuffer<T>(rhs),
49 getBuffer<T>(output));

99bool haveSameQauntInfo(const IPortableTensor *lhs, const IPortableTensor *rhs,
100 const IPortableTensor *output)
101{
102 return (lhs->data_scale() == rhs->data_scale() && lhs->data_scale() == output->data_scale()) &&
103 (lhs->data_zero_point() == rhs->data_zero_point() &&
104 lhs->data_zero_point() == output->data_zero_point());

108void ElementwiseBinaryLayer::configure(const IPortableTensor *lhs, const IPortableTensor *rhs,
109 IPortableTensor *output, const ElementwiseBinaryType op_type)
110{
111 assert(lhs != nullptr);
112 assert(rhs != nullptr);
113 assert(output != nullptr);
114
115 _lhs = lhs;
116 _rhs = rhs;
117 _output = output;
118
119 switch (op_type)
120 {
121 case ElementwiseBinaryType::kFloorDiv:
122 if (_lhs->data_type() == OperandType::FLOAT32)
123 {
124 _kernel = FloorDivGeneric<float>;
125 }
126 else if (_lhs->data_type() == OperandType::INT32)
127 {
128 _kernel = FloorDivGeneric<int32_t>;
129 }
130 else
131 {
132 throw std::runtime_error{"Max: unsupported data type"};
133 }
134 break;
135 case ElementwiseBinaryType::kLogicalAnd:
136 if ((_lhs->data_type() == OperandType::BOOL8) && (_rhs->data_type() == OperandType::BOOL8))
137 {
138 _kernel = logicalAndGeneric<bool>;
139 }
140 else
141 {
142 throw std::runtime_error{"LogicalOr: Unsupported data type"};
143 }
144 break;
145 case ElementwiseBinaryType::kLogicalOr:
146 if ((_lhs->data_type() == OperandType::BOOL8) && (_rhs->data_type() == OperandType::BOOL8))
147 {
148 _kernel = logicalOrGeneric<bool>;
149 }
150 else
151 {
152 throw std::runtime_error{"LogicalOr: Unsupported data type"};
153 }
154 break;
155 case ElementwiseBinaryType::kMax:
156 if (_lhs->data_type() == OperandType::QUANT_UINT8_ASYMM)
157 {
158 if (!haveSameQauntInfo(_lhs, _rhs, _output))
159 {
160 throw std::runtime_error("Max NYI for quantized");
161 }
162 _kernel = maximumGeneric<uint8_t>;
163 }
164 else if (_lhs->data_type() == OperandType::FLOAT32)
165 {
166 _kernel = maximumGeneric<float>;
167 }
168 else
169 {
170 throw std::runtime_error{"Max: unsupported data type"};
171 }
172 break;
173 case ElementwiseBinaryType::kMin:
174 if (_lhs->data_type() == OperandType::QUANT_UINT8_ASYMM)
175 {
176 if (!haveSameQauntInfo(_lhs, _rhs, _output))
177 {
178 throw std::runtime_error("Min NYI for quantized");
179 }
180 _kernel = minimumGeneric<uint8_t>;
181 }
182 else if (_lhs->data_type() == OperandType::INT32)
183 {
184 _kernel = minimumGeneric<int32_t>;
185 }
186 else if (_lhs->data_type() == OperandType::FLOAT32)
187 {
188 _kernel = minimumGeneric<float>;
189 }
190 else
191 {
192 throw std::runtime_error{"Min: unsupported data type"};
193 }
194 break;
195 default:
196 throw std::runtime_error{"ElementwiseBinary: Unsupported ElementwiseBinary type"};
197 }
198}
