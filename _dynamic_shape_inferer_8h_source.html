107 // in case of output tensor of an op, it is possible that
108 // the output became dynamic although it had been static before.
109 // Once a tensor becomes dynamic, it will lost memory allocated for static.
110 // Therefore once output is dynamic, it should be treated as dynamic tensor. (memory should be
111 // allocated at runtime) `previously` means `dynamic` or `static` has been set in previous loop in
112 // WHILE of previous call of `nnfw_run()`
113 bool previously_static(backend::ITensor *op_output) { return !op_output->is_dynamic(); }
114
115 // helper function that check if op's input is static
116 // Note that input of n'th op has been set to static or dynamic by (n-1)th op.
117 // That's why it is called `currently_static`
118 bool currently_static(backend::ITensor *op_input) { return !op_input->is_dynamic(); }
