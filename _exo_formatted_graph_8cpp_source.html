35const char *to_str(locoex::FusedActFunc fused)
36{
37 switch (fused)
38 {
39 case locoex::FusedActFunc::NONE:
40 return "NONE";
41 case locoex::FusedActFunc::RELU:
42 return "RELU";
43 case locoex::FusedActFunc::RELU6:
44 return "RELU6";
45 default:
46 return "Error";
47 }
48}

156bool TFLNodeSummaryBuilderBase::build(const loco::Node *node, locop::NodeSummary &s) const
157{
158 if (node->dialect() != locoex::TFLDialect::get())
159 return false;
160
161#define TFL_NODE(OPCODE, CLASS) \
162 if (dynamic_cast<const CLASS *>(node)) \
163 { \
164 s.opname(tfl_opname(node->opnum())); \
165 return summary(dynamic_cast<const CLASS *>(node), s); \
166 }
167#include "Dialect/IR/TFLNodes.lst"
168#undef TFL_NODE
169
170 return false;
171}

501{
