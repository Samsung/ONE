39 Conv() : _im2col_shape(4), _need_im2col(false), _prepared(false) {}

41 void prepare(const Shape &input_shape, const Shape &kernel_shape, const Shape &output_shape,
42 uint32_t stride_width, uint32_t stride_height, uint32_t dilation_width_factor,
43 uint32_t dilation_height_factor)
44 {
45 if (!_prepared)
46 {
47 IsRequiredIm2col(input_shape, kernel_shape, output_shape, stride_width, stride_height,
48 dilation_width_factor, dilation_height_factor);
49 _prepared = true;
50 }
51 }

53 void operator()(const ConvParams &params, const Shape &input_shape, const float *input_data,
54 const Shape &filter_shape, const float *filter_data, const Shape &bias_shape,
55 const float *bias_data, const Shape &output_shape, float *output_data,
56 ::ruy::Context *ruy_context)
57 {
58 if (!_prepared)
59 {
60 // This means that input or output are dynamic or filter is not constant
61 IsRequiredIm2col(input_shape, filter_shape, output_shape, params.stride_width,
62 params.stride_height, params.dilation_width_factor,
63 params.dilation_height_factor);
64 _prepared = true;
65 }
66
67 int im2col_size = _need_im2col ? _im2col_shape.FlatSize() : 0;
68
69 // Use heap if size is larger than 8MB
70 if (im2col_size > 2 * 1024 * 1024)
71 {
72 std::unique_ptr<float[]> im2col_data = std::make_unique<float[]>(im2col_size);
73 ConvFloat(params, input_shape, input_data, filter_shape, filter_data, bias_shape, bias_data,
74 output_shape, output_data, _im2col_shape, im2col_data.get(), ruy_context);
75 }
76 else if (im2col_size > 0)
77 {
78 float im2col_data[im2col_size];
79 ConvFloat(params, input_shape, input_data, filter_shape, filter_data, bias_shape, bias_data,
80 output_shape, output_data, _im2col_shape, im2col_data, ruy_context);
81 }
82 else
83 {
84 ConvFloat(params, input_shape, input_data, filter_shape, filter_data, bias_shape, bias_data,
85 output_shape, output_data, _im2col_shape, nullptr, ruy_context);
86 }
87 }
88
89private:
90 void ConvFloat(const ConvParams &params, const Shape &input_shape, const float *input_data,
91 const Shape &filter_shape, const float *filter_data, const Shape &bias_shape,
92 const float *bias_data, const Shape &output_shape, float *output_data,
93 const Shape &im2col_shape, float *im2col_data, ::ruy::Context *ruy_context)
94 {
95 UNUSED_RELEASE(bias_shape);
96 const int stride_width = params.stride_width;
97 const int stride_height = params.stride_height;
98 const int dilation_width_factor = params.dilation_width_factor;
99 const int dilation_height_factor = params.dilation_height_factor;
100 const float output_activation_min = params.float_activation_min;
101 const float output_activation_max = params.float_activation_max;
102 assert(input_shape.DimensionsCount() == 4);
103 assert(filter_shape.DimensionsCount() == 4);
104 assert(output_shape.DimensionsCount() == 4);
105
106 // NB: the float 0.0f value is represented by all zero bytes.
107 const uint8_t float_zero_byte = 0x00;
108 const float *gemm_input_data = nullptr;
109 const Shape *gemm_input_shape = nullptr;
110 const int filter_width = filter_shape.Dims(2);
111 const int filter_height = filter_shape.Dims(1);
112 const bool need_dilated_im2col = dilation_width_factor != 1 || dilation_height_factor != 1;
113 const bool need_im2col =
114 stride_width != 1 || stride_height != 1 || filter_width != 1 || filter_height != 1;
115 if (need_dilated_im2col)
116 {
117 DilatedIm2col(params, float_zero_byte, input_shape, input_data, filter_shape, output_shape,
118 im2col_data);
119 gemm_input_data = im2col_data;
120 gemm_input_shape = &im2col_shape;
121 }
122 else if (need_im2col)
123 {
124 assert(im2col_data);
125 Im2col(params, filter_height, filter_width, float_zero_byte, input_shape, input_data,
126 im2col_shape, im2col_data);
127 gemm_input_data = im2col_data;
128 gemm_input_shape = &im2col_shape;
129 }
130 else
131 {
132 // TODO(aselle): We need to make sure to not send im2col if it is not
133 // needed.
134 assert(!im2col_data);
135 gemm_input_data = input_data;
136 gemm_input_shape = &input_shape;
137 }
138
139 const int gemm_input_dims = gemm_input_shape->DimensionsCount();
140 int m = FlatSizeSkipDim(*gemm_input_shape, gemm_input_dims - 1);
141 int n = output_shape.Dims(3);
142 int k = gemm_input_shape->Dims(gemm_input_dims - 1);
143
144 // When an optimized CBLAS implementation is not available, fall back
145 // to using cpu_backend_gemm.
146 MatrixParams<float> lhs_params;
147 lhs_params.order = Order::kRowMajor;
148 lhs_params.rows = n;
149 lhs_params.cols = k;
150 MatrixParams<float> rhs_params;
151 rhs_params.order = Order::kColMajor;
152 rhs_params.rows = k;
153 rhs_params.cols = m;
154 MatrixParams<float> dst_params;
155 dst_params.order = Order::kColMajor;
156 dst_params.rows = n;
157 dst_params.cols = m;
158 GemmParams<float, float> gemm_params;
159 gemm_params.bias = bias_data;
160 gemm_params.clamp_min = output_activation_min;
161 gemm_params.clamp_max = output_activation_max;
162
163 // Below code is from tflite::cpu_backend_gemm::detail::GemmImplUsingRuy
164 ::ruy::Matrix<float> ruy_lhs;
165 ::ruy::Matrix<float> ruy_rhs;
166 ::ruy::Matrix<float> ruy_dst;
167 // Note that cache is always enabled for input and weight tensors
168 ruy_support::MakeRuyMatrix(lhs_params, filter_data, &ruy_lhs, true);
169 ruy_support::MakeRuyMatrix(rhs_params, gemm_input_data, &ruy_rhs, true);
170 ruy_support::MakeRuyMatrix(dst_params, output_data, &ruy_dst);
171
172 ::ruy::MulParams<float, float> ruy_mul_params;
173 ruy_support::MakeRuyMulParams(gemm_params, &ruy_mul_params);
174
175 ::ruy::Mul(ruy_lhs, ruy_rhs, ruy_mul_params, ruy_context, &ruy_dst);
176 }

178 void IsRequiredIm2col(const Shape &input_shape, const Shape &kernel_shape,
179 const Shape &output_shape, uint32_t stride_width, uint32_t stride_height,
180 uint32_t dilation_width_factor, uint32_t dilation_height_factor)
181 {
182 const bool need_dilated_im2col = dilation_width_factor != 1 || dilation_height_factor != 1;
183 const bool need_non_dilated_im2col = stride_width != 1 || stride_height != 1 ||
184 kernel_shape.Dims(1) != 1 || kernel_shape.Dims(2) != 1;
185
186 _need_im2col = need_dilated_im2col || need_non_dilated_im2col;
187
188 if (_need_im2col)
189 {
190 _im2col_shape.SetDim(0, output_shape.Dims(0));
191 _im2col_shape.SetDim(1, output_shape.Dims(1));
192 _im2col_shape.SetDim(2, output_shape.Dims(2));
193 _im2col_shape.SetDim(3, input_shape.Dims(3) * kernel_shape.Dims(1) * kernel_shape.Dims(2));
194 }
195 }
196
197private:
198 Shape _im2col_shape;
199 bool _need_im2col;
200 bool _prepared;
201};
