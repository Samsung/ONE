48def to_nhwc(tensor):
49 if (tensor.ndim == 4):
50 return np.transpose(tensor, (0, 2, 3, 1))

54class OnnxRunner:
55 def __init__(self, filepath):
56 self.filepath = filepath
57 self.session = None
58 self.inputs = None
59 self.inputs_size = None
60 self.inputs_data = None
61 self.outputs = None
62 self.outputs_size = None
63
64 def load(self):
65 model = onnx.load(self.filepath)
66 onnx.checker.check_model(model)
67 self.session = ort.InferenceSession(self.filepath)
68
69 def feed_random_inputs(self):
70 self.inputs = self.session.get_inputs()
71 self.inputs_size = len(self.inputs)
72 # reset input dictionary
73 self.inputs_data = {}
74 for in_idx in range(self.inputs_size):
75 input_shape = self.inputs[in_idx].shape
76 input_type = self.inputs[in_idx].type
77 if input_type == 'tensor(float)':
78 torch_type = torch.float32
79 else:
80 # TODO support other dtype
81 raise SystemExit("Unsupported input dtype")
82
83 x = torch.randn(input_shape, dtype=torch_type)
84 input_npa = to_numpy(x)
85 self.inputs_data.update({self.inputs[in_idx].name: input_npa})
86
87 # save NHWC form of input for luci-interpreter
88 input_npa_nhwc = to_nhwc(input_npa)
89 input_npa_nhwc.tofile(circle_filepath + ".input" + str(in_idx))
90
91 def run(self):
92 self.outs = self.session.run(None, self.inputs_data)
93
94 def get_outputs(self):
95 self.outputs = self.session.get_outputs()
96 self.outputs_size = len(self.outputs)

117# Compare results
118rtolerance = 1e-03
119atolerance = 1e-04
120result_compare = True
121for idx in range(onnx_runner.outputs_size):
122 output_shape = onnx_runner.outputs[idx].shape
123 output_type = onnx_runner.outputs[idx].type
124 if output_type == 'tensor(float)':
125 output_np_type = np.float32
126 else:
127 # TODO support other dtype
128 raise SystemExit("Unsupported output dtype")
129
130 # output of luci-interpreter
131 output_data = np.fromfile(circle_filepath + ".output" + str(idx), output_np_type)
132 shape_file = open(circle_filepath + ".output" + str(idx) + ".shape", 'r')
133 output_shape = [int(i) for i in shape_file.read().split(',')]
134 luci_output_data = np.reshape(output_data, output_shape)
135
136 # output of onnx runtime
137 output_nchw = onnx_runner.outs[idx]
138 output_nhwc = to_nhwc(output_nchw)
139
140 # diff has tensor of boolean for each values within tolerance or not
141 diff = np.isclose(output_nhwc, luci_output_data, rtol=rtolerance, atol=atolerance)
142 # get one boolean if all are True then True
143 result_compare_one = np.all(diff)
144 print("Compare", idx, result_compare_one)
145 if (not result_compare_one):
146 diff_val = np.subtract(output_nhwc, luci_output_data)
147 print("ONNX Result", output_nhwc)
148 print("Diff", diff_val)
149 print("Diff Max", np.ndarray.max(diff_val))
150
151 result_compare = result_compare and result_compare_one
152
153if (not result_compare):
154 exit(-1)
