53std::unique_ptr<exec::train::TrainableFnSequence> KernelGenerator::generate(ir::OperationIndex idx)
54{
55 // TODO Generate TrainableFnSequence that can go backward as well
56 auto ret = std::make_unique<exec::train::TrainableFnSequence>();
57
58 const auto &op = _tgraph.operation(idx);
59 op.accept(*this);
60 assert(_return_fn);
61 ret->append(std::move(_return_fn));
62
63 for (auto &&ind : (op.getInputs() | ir::Remove::UNDEFINED) + op.getOutputs())
64 {
65 auto portable_tensor = _tensor_reg->getPortableTensor(ind);
66 if (portable_tensor)
67 {
68 assert(portable_tensor->layout() == ir::Layout::NHWC);
69 }
70 auto tensor = _tensor_reg->getNativeTensor(ind);
71 if (tensor)
72 {
73 tensor->increase_ref();
74 }
75 }
76 return ret;
77}
