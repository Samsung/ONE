85std::unique_ptr<exec::train::TrainableFnSequence> KernelGenerator::generate(ir::OperationIndex idx)
86{
87 auto ret = std::make_unique<exec::train::TrainableFnSequence>();
88
89 const auto &op = _tgraph.operation(idx);
90 op.accept(*this);
91 assert(_return_fn);
92 ret->append(std::move(_return_fn));
93
94 for (auto &&update_fn : _update_funcs)
95 ret->append(std::move(update_fn));
96 _update_funcs.clear();
97
98 for (auto &&ind : (op.getInputs() | ir::Remove::UNDEFINED) + op.getOutputs())
99 {
100 auto portable_tensor = _tensor_reg->getPortableTensor(ind);
101 if (portable_tensor)
102 {
103 assert(portable_tensor->layout() == ir::Layout::NHWC);
104 }
105 auto tensor = _tensor_reg->getNonConstTensor(ind);
106 if (tensor)
107 {
108 tensor->increase_ref();
109 }
110 }
111 return ret;
112}

125void KernelGenerator::visit(const ir::train::operation::BinaryArithmetic &node)
126{
127 using ir::train::operation::BinaryArithmetic;
128
129 const auto output_index{node.getOutputs().at(0)};
130 const auto lhs_index{node.getInputs().at(BinaryArithmetic::Input::LHS)};
131 const auto rhs_index{node.getInputs().at(BinaryArithmetic::Input::RHS)};
132
133 const auto arithmetic_type = node.param().arithmetic_type;
134 const auto activation = node.param().activation;
135
136 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
137 auto lhs_tensor = _tensor_reg->getPortableTensor(lhs_index);
138 auto rhs_tensor = _tensor_reg->getPortableTensor(rhs_index);
139
140 auto back_prop_output_tensor = _tensor_reg->getBackPropTensor(output_index);
141 auto back_prop_lhs_tensor = _tensor_reg->getBackPropTensor(lhs_index);
142 auto back_prop_rhs_tensor = _tensor_reg->getBackPropTensor(rhs_index);
143
144 auto fn = std::make_unique<ops::BinaryArithmeticLayer>();
145 fn->configure(lhs_tensor, rhs_tensor, output_tensor, back_prop_lhs_tensor, back_prop_rhs_tensor,
146 back_prop_output_tensor, activation,
147 static_cast<train::ops::ArithmeticType>(arithmetic_type));
148 _return_fn = std::move(fn);
149}

151void KernelGenerator::visit(const ir::train::operation::Conv2D &node)
152{
153 using ir::train::operation::Conv2D;
154
155 const auto out_index{node.getOutputs().at(0)};
156 const auto in_index{node.getInputs().at(Conv2D::Input::INPUT)};
157 const auto ker_index{node.getInputs().at(Conv2D::Input::KERNEL)};
158 const auto bias_index{node.getInputs().at(Conv2D::Input::BIAS)};
159
160 auto out_tensor = _tensor_reg->getPortableTensor(out_index);
161 auto in_tensor = _tensor_reg->getPortableTensor(in_index);
162 auto ker_tensor = _tensor_reg->getTrainableTensor(ker_index);
163 auto bias_tensor = _tensor_reg->getTrainableTensor(bias_index);
164
165 auto out_back_prop_tensor = _tensor_reg->getBackPropTensor(out_index);
166 auto in_back_prop_tensor = _tensor_reg->getBackPropTensor(in_index);
167 auto ker_grad_tensor = _tensor_reg->getGradientTensor(ker_index);
168 auto bias_grad_tensor = _tensor_reg->getGradientTensor(bias_index);
169
170 // Generate kernel
171 const auto stride = node.param().stride;
172 const auto activation = node.param().activation;
173 const auto &param_padding = node.param().padding;
174 const auto dilation = node.param().dilation;
175 auto fn = std::make_unique<ops::ConvolutionLayer>();
176
177 auto &operands = _tgraph.operands();
178 const auto ifm_shape = operands.at(in_index).shape().asFeature(_current_layout);
179 const auto ofm_shape = operands.at(out_index).shape().asFeature(_current_layout);
180 // Kernel format is [depth_out, kernel_height, kernel_width, depth_in].
181 const auto &ker_shape = operands.at(ker_index).shape();
182 const auto ker_height = ker_shape.dim(1);
183 const auto ker_width = ker_shape.dim(2);
184
185 const auto padding =
186 ir::calculatePadding(param_padding, ifm_shape, ofm_shape, stride, ker_width, ker_height,
187 dilation.width_factor, dilation.height_factor);
188
189 fn->configure(in_tensor, ker_tensor, bias_tensor, out_tensor, in_back_prop_tensor,
190 ker_grad_tensor, bias_grad_tensor, out_back_prop_tensor, param_padding.type,
191 padding.left, padding.right, padding.top, padding.bottom, stride.horizontal,
192 stride.vertical, dilation.width_factor, dilation.height_factor, activation);
193
194 _return_fn = std::move(fn);
195
196 // Generate GradientApplier
197 if (bias_tensor)
198 _update_funcs.emplace_back(generateGradientApplier(_optimizer, bias_grad_tensor, bias_tensor));
199 _update_funcs.emplace_back(generateGradientApplier(_optimizer, ker_grad_tensor, ker_tensor));
200}

202void KernelGenerator::visit(const ir::train::operation::DepthwiseConv2D &node)
203{
204 using ir::train::operation::DepthwiseConv2D;
205
206 const auto ofm_index{node.getOutputs().at(0)};
207 const auto ifm_index{node.getInputs().at(DepthwiseConv2D::Input::INPUT)};
208 const auto ker_index{node.getInputs().at(DepthwiseConv2D::Input::KERNEL)};
209 const auto bias_index{node.getInputs().at(DepthwiseConv2D::Input::BIAS)};
210
211 auto ofm_tensor = _tensor_reg->getPortableTensor(ofm_index);
212 auto ifm_tensor = _tensor_reg->getPortableTensor(ifm_index);
213 auto ker_tensor = _tensor_reg->getTrainableTensor(ker_index);
214 auto bias_tensor = _tensor_reg->getTrainableTensor(bias_index);
215
216 auto ofm_back_prop_tensor = _tensor_reg->getBackPropTensor(ofm_index);
217 auto ifm_back_prop_tensor = _tensor_reg->getBackPropTensor(ifm_index);
218 auto ker_grad_tensor = _tensor_reg->getGradientTensor(ker_index);
219 auto bias_grad_tensor = _tensor_reg->getGradientTensor(bias_index);
220
221 const auto stride = node.param().stride;
222 const auto &operands = _tgraph.operands();
223 const auto ofm_shape = operands.at(ofm_index).shape().asFeature(_current_layout);
224 const auto ifm_shape = operands.at(ifm_index).shape().asFeature(_current_layout);
225 // Kernel format is [1, kernel_height, kernel_width, depth_out].
226 const auto &ker_shape = operands.at(ker_index).shape();
227 const auto ker_height = ker_shape.dim(1);
228 const auto ker_width = ker_shape.dim(2);
229 const auto dilation_width = node.param().dilation.width_factor;
230 const auto dilation_height = node.param().dilation.height_factor;
231 const auto padding = ir::calculatePadding(node.param().padding, ifm_shape, ofm_shape, stride,
232 ker_width, ker_height, dilation_width, dilation_height);
233 const auto multiplier = node.param().multiplier;
234 const auto activation = node.param().activation;
235
236 auto fn = std::make_unique<ops::DepthwiseConvolutionLayer>();
237
238 fn->configure(ifm_tensor, ker_tensor, bias_tensor, ofm_tensor, ifm_back_prop_tensor,
239 ker_grad_tensor, bias_grad_tensor, ofm_back_prop_tensor, padding.left,
240 padding.right, padding.top, padding.bottom, stride.horizontal, stride.vertical,
241 multiplier, dilation_width, dilation_height, activation, _external_context);
242
243 _return_fn = std::move(fn);
244
245 // Generate GradientApplier
246 if (bias_tensor)
247 _update_funcs.emplace_back(generateGradientApplier(_optimizer, bias_grad_tensor, bias_tensor));
248 _update_funcs.emplace_back(generateGradientApplier(_optimizer, ker_grad_tensor, ker_tensor));
249}

311void KernelGenerator::visit(const ir::train::operation::Loss &node)
312{
313 using ir::train::operation::Loss;
314
315 const auto output_index{node.getOutputs().at(0)};
316 const auto y_pred_index{node.getInputs().at(Loss::Y_PRED)};
317 const auto y_true_index{node.getInputs().at(Loss::Y_TRUE)};
318
319 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
320 auto y_pred_tensor = _tensor_reg->getPortableTensor(y_pred_index);
321 auto y_true_tensor = _tensor_reg->getPortableTensor(y_true_index);
322
323 auto back_prop_y_pred_tensor = _tensor_reg->getBackPropTensor(y_pred_index);
324
325 auto loss_code = node.param().loss_code;
326 auto loss_param = node.param().loss_param;
327
328 switch (loss_code)
329 {
330 case ir::train::LossCode::MeanSquaredError:
331 {
332 auto fn = std::make_unique<ops::LossMeanSquaredErrorLayer>();
333 fn->configure(y_pred_tensor, y_true_tensor, output_tensor, back_prop_y_pred_tensor);
334 _return_fn = std::move(fn);
335 break;
336 }
337 case ir::train::LossCode::CategoricalCrossentropy:
338 {
339 auto fn = std::make_unique<ops::LossCategoricalCrossentropyLayer>();
340 fn->configure(y_pred_tensor, y_true_tensor, output_tensor, back_prop_y_pred_tensor,
341 loss_param.cce.axis, loss_param.cce.label_smoothing);
342 _return_fn = std::move(fn);
343 break;
344 }
345 default:
346 throw std::runtime_error("LossLayer: unsupported loss type");
347 break;
348 }
349}

371 auto in_back_prop_tensor = _tensor_reg->getBackPropTensor(input_index);
372
373 fn->configure(input, pad, value, output, in_back_prop_tensor, out_back_prop_tensor);
374 _return_fn = std::move(fn);

377void KernelGenerator::visit(const ir::train::operation::Pool2D &node)
378{
379 using ir::train::operation::Pool2D;
380
381 const auto output_index{node.getOutputs().at(0)};
382 const auto input_index{node.getInputs().at(0)};
383
384 const auto &operands = _tgraph.operands();
385 const auto &ofm_shape = operands.at(output_index).shape();
386 const auto &ifm_shape = operands.at(input_index).shape();
387
388 if (ifm_shape.rank() != 4)
389 {
390 std::runtime_error(node.name() + " only supports 4D tensor as input");
391 }
392
393 // calcualate padding
394 const auto stride = node.param().stride;
395 const auto kh = node.param().kh;
396 const auto kw = node.param().kw;
397 const auto padding =
398 ir::calculatePadding(node.param().padding, ifm_shape.asFeature(_current_layout),
399 ofm_shape.asFeature(_current_layout), stride, kw, kh);
400
401 auto out_tensor = _tensor_reg->getPortableTensor(output_index);
402 auto in_tensor = _tensor_reg->getPortableTensor(input_index);
403
404 auto out_back_prop_tensor = _tensor_reg->getBackPropTensor(output_index);
405 auto in_back_prop_tensor = _tensor_reg->getBackPropTensor(input_index);
406
407 const auto activation = node.param().activation;
408 const auto pool_type = convertPoolType(node.param().op_type);
409
410 auto fn = std::make_unique<ops::PoolLayer>();
411
412 fn->configure(in_tensor, padding.left, padding.right, padding.top, padding.bottom,
413 stride.horizontal, stride.vertical, kw, kh, activation, out_tensor, pool_type,
414 in_back_prop_tensor, out_back_prop_tensor);
415
416 _return_fn = std::move(fn);
417}
