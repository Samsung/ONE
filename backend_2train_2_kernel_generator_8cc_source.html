91std::unique_ptr<exec::train::TrainableFnSequence> KernelGenerator::generate(ir::OperationIndex idx)
92{
93 auto ret = std::make_unique<exec::train::TrainableFnSequence>();
94
95 const auto &op = _tgraph.operation(idx);
96 op.accept(*this);
97 assert(_return_fn);
98 ret->append(std::move(_return_fn));
99
100 for (auto &&update_fn : _update_funcs)
101 ret->append(std::move(update_fn));
102 _update_funcs.clear();
103
104 for (auto &&ind : (op.getInputs() | ir::Remove::UNDEFINED) + op.getOutputs())
105 {
106 auto portable_tensor = _tensor_reg->getPortableTensor(ind);
107 if (portable_tensor)
108 {
109 assert(portable_tensor->layout() == ir::Layout::NHWC);
110 }
111 auto tensor = _tensor_reg->getNonConstTensor(ind);
112 if (tensor)
113 {
114 tensor->increase_ref();
115 }
116 }
117 return ret;
118}

170void KernelGenerator::visit(const ir::train::operation::FullyConnected &node)
171{
172 using ir::train::operation::FullyConnected;
173
174 const auto out_index{node.getOutputs().at(0)};
175 const auto in_index{node.getInputs().at(FullyConnected::Input::INPUT)};
176 const auto weights_index{node.getInputs().at(FullyConnected::Input::WEIGHT)};
177 const auto bias_index{node.getInputs().at(FullyConnected::Input::BIAS)};
178
179 auto out_tensor = _tensor_reg->getPortableTensor(out_index);
180 auto in_tensor = _tensor_reg->getPortableTensor(in_index);
181 auto weights_tensor = _tensor_reg->getTrainableTensor(weights_index);
182 auto bias_tensor = _tensor_reg->getTrainableTensor(bias_index);
183
184 auto out_back_prop_tensor = _tensor_reg->getBackPropTensor(out_index);
185 auto in_back_prop_tensor = _tensor_reg->getBackPropTensor(in_index);
186 auto weights_grad_tensor = _tensor_reg->getGradientTensor(weights_index);
187 auto bias_grad_tensor = _tensor_reg->getGradientTensor(bias_index);
188
189 // Generate kernel
190 const auto activation = node.param().activation;
191 const auto weights_format = node.param().weights_format;
192
193 auto fn = std::make_unique<ops::FullyConnectedLayer>();
194
195 fn->configure(in_tensor, weights_tensor, bias_tensor, out_tensor, in_back_prop_tensor,
196 weights_grad_tensor, bias_grad_tensor, out_back_prop_tensor, activation,
197 weights_format, _external_context);
198
199 _return_fn = std::move(fn);
200
201 // Generate GradientAppliers
202 if (bias_tensor)
203 _update_funcs.emplace_back(generateGradientApplier(_optimizer, bias_grad_tensor, bias_tensor));
204 _update_funcs.emplace_back(
205 generateGradientApplier(_optimizer, weights_grad_tensor, weights_tensor));
206}

231void KernelGenerator::visit(const ir::train::operation::Pool2D &node)
232{
233 using ir::train::operation::Pool2D;
234
235 const auto output_index{node.getOutputs().at(0)};
236 const auto input_index{node.getInputs().at(0)};
237
238 const auto operands = _tgraph.operands();
239 const auto ofm_shape = operands.at(output_index).shape();
240 const auto ifm_shape = operands.at(input_index).shape();
241
242 if (ifm_shape.rank() != 4)
243 {
244 std::runtime_error(node.name() + " only supports 4D tensor as input");
245 }
246
247 // calcualate padding
248 const auto stride = node.param().stride;
249 const auto kh = node.param().kh;
250 const auto kw = node.param().kw;
251 const auto padding =
252 ir::calculatePadding(node.param().padding, ifm_shape.asFeature(_current_layout),
253 ofm_shape.asFeature(_current_layout), stride, kw, kh);
254
255 auto out_tensor = _tensor_reg->getPortableTensor(output_index);
256 auto in_tensor = _tensor_reg->getPortableTensor(input_index);
257
258 auto out_back_prop_tensor = _tensor_reg->getBackPropTensor(output_index);
259 auto in_back_prop_tensor = _tensor_reg->getBackPropTensor(input_index);
260
261 const auto activation = node.param().activation;
262 const auto pool_type = convertPoolType(node.param().op_type);
263
264 auto fn = std::make_unique<ops::PoolLayer>();
265
266 fn->configure(in_tensor, padding.left, padding.right, padding.top, padding.bottom,
267 stride.horizontal, stride.vertical, kw, kh, activation, out_tensor, pool_type,
268 in_back_prop_tensor, out_back_prop_tensor);
269
270 _return_fn = std::move(fn);
271}
