66std::unique_ptr<exec::FunctionSequence> KernelGenerator::generate(ir::OperationIndex ind)
67{
68 // TODO Generate FunctionSequence for backwarding as well
69 auto ret = std::make_unique<exec::FunctionSequence>();
70 ret->enableDynamicShapeInferer(false);
71
72 const auto &op = _graph.operations().at(ind);
73 op.accept(*this);
74 ret->append(releaseFunction());
75
76 for (auto ind : (op.getInputs() | ir::Remove::UNDEFINED) + op.getOutputs())
77 {
78 auto portable_tensor = _tensor_reg->getPortableTensor(ind);
79 if (portable_tensor)
80 {
81 assert(portable_tensor->layout() == ir::Layout::NHWC);
82 }
83
84 auto tensor = _tensor_reg->getNativeTensor(ind);
85 if (tensor)
86 {
87 tensor->increase_ref();
88 }
89 }
90 return ret;
91}

93void KernelGenerator::visit(const ir::operation::Conv2D &node)
94{
95 using ir::operation::Conv2D;
96
97 const auto ofm_index{node.getOutputs().at(0)};
98 const auto ifm_index{node.getInputs().at(Conv2D::Input::INPUT)};
99 const auto ker_index{node.getInputs().at(Conv2D::Input::KERNEL)};
100 const auto bias_index{node.getInputs().at(Conv2D::Input::BIAS)};
101
102 auto ofm_tensor = _tensor_reg->getPortableTensor(ofm_index);
103 auto ifm_tensor = _tensor_reg->getPortableTensor(ifm_index);
104 auto ker_tensor = _tensor_reg->getPortableTensor(ker_index);
105 auto bias_tensor = _tensor_reg->getPortableTensor(bias_index);
106
107 const auto stride = node.param().stride;
108 const auto activation = node.param().activation;
109 const auto param_padding = node.param().padding;
110 const auto dilation = node.param().dilation;
111 auto fn = std::make_unique<ops::ConvolutionLayer>();
112
113 if (_ctx.at(ifm_index).info().isDynamic() || _ctx.at(ker_index).info().isDynamic())
114 {
115 fn->configure(ifm_tensor, ker_tensor, bias_tensor, param_padding.type, param_padding.param.left,
116 param_padding.param.right, param_padding.param.top, param_padding.param.bottom,
117 stride.horizontal, stride.vertical, dilation.width_factor, dilation.height_factor,
118 activation, ofm_tensor);
119
120 _return_fn = std::move(fn);
121 return;
122 }
123 const auto ifm_shape = _ctx.at(ifm_index).shape().asFeature(_current_layout);
124 const auto ofm_shape = _ctx.at(ofm_index).shape().asFeature(_current_layout);
125 // Kernel format is [depth_out, kernel_height, kernel_width, depth_in].
126 const auto &ker_shape = _ctx.at(ker_index).shape();
127 const auto ker_height = ker_shape.dim(1);
128 const auto ker_width = ker_shape.dim(2);
129
130 const auto padding =
131 ir::calculatePadding(param_padding, ifm_shape, ofm_shape, stride, ker_width, ker_height,
132 dilation.width_factor, dilation.height_factor);
133
134 fn->configure(ifm_tensor, ker_tensor, bias_tensor, param_padding.type, padding.left,
135 padding.right, padding.top, padding.bottom, stride.horizontal, stride.vertical,
136 dilation.width_factor, dilation.height_factor, activation, ofm_tensor);
137
138 _return_fn = std::move(fn);
139}

141void KernelGenerator::visit(const ir::operation::Pool2D &node)
142{
143 const auto ofm_index{node.getOutputs().at(0)};
144 const auto ifm_index{node.getInputs().at(ir::operation::Pool2D::Input::INPUT)};
145
146 const auto kh = node.param().kh;
147 const auto kw = node.param().kw;
148 const auto stride = node.param().stride;
149 const auto ifm_shape = _ctx.at(ifm_index).shape().asFeature(_current_layout);
150 const auto ofm_shape = _ctx.at(ofm_index).shape().asFeature(_current_layout);
151 const auto padding =
152 ir::calculatePadding(node.param().padding, ifm_shape, ofm_shape, stride, kw, kh);
153 const auto activation = node.param().activation;
154
155 auto ofm_tensor = _tensor_reg->getPortableTensor(ofm_index);
156 auto ifm_tensor = _tensor_reg->getPortableTensor(ifm_index);
157
158 auto fn = std::make_unique<ops::PoolLayer>();
159
160 fn->configure(ifm_tensor, padding.left, padding.right, padding.top, padding.bottom,
161 stride.horizontal, stride.vertical, kw, kh, activation, ofm_tensor,
162 convertPoolType(node.param().op_type));
163
164 _return_fn = std::move(fn);
165}
