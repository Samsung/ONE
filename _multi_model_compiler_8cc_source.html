51std::shared_ptr<CompilerArtifact> MultiModelCompiler::compile(void)
52{
53 /***************************************************
54 * Prepare compilation phase
55 ***************************************************/
56 for (auto options : _voptions)
57 {
58 if (!options)
59 throw std::runtime_error{"Empty compile option"};
60
61 // Compilable check
62 // TODO: Support hybrid execution -
63 // execution between interpreter and compiled executor (including control flow)
64 if (options->disable_compile)
65 throw std::runtime_error{"NYI: Disable compilation for multi model is not supported yet"};
66
67 // Mode check
68 // TODO handle option for each model
69 if (options->he_profiling_mode)
70 throw std::runtime_error("NYI: Profiling mode for multiple model is not supported yet");
71
72 options->forceInternalOptions();
73 options->verboseOptions();
74 }
75
76 // NYI: allow one model compilation
77 auto const model_count = _nnpkg->model_count();
78 if (model_count != _voptions.size())
79 throw std::runtime_error{"Model count and option vector size mismatch"};
80
81 for (uint16_t i = 0; i < model_count; i++)
82 {
83 _nnpkg->model(ir::ModelIndex{i})->iterate([&](const ir::SubgraphIndex &, ir::Graph &subg) {
84 // Mandatory passes
85 pass::PassRunner{}
86 .append(std::make_unique<pass::ConstantOutputPass>(subg))
87 .append(std::make_unique<pass::OddOutputPass>(subg))
88 .run();
89
90 // Optimizations
91 pass::PassRunner{}.append(std::make_unique<pass::UnusedOperandEliminationPass>(subg)).run();
92 });
93 }
94
95 /***************************************************
96 * Backend independent analysis & optimization phase
97 ***************************************************/
98 // TODO Handle dump level for each model
99 auto dump_level = static_cast<dumper::dot::DotDumper::Level>(_voptions[0]->graph_dump_level);
100 onert::dumper::dot::DotDumper dot_dumper(dump_level);
101
102 // Tracing context
103 // TODO Support tracing_ctx for multiple model
104 std::unique_ptr<util::TracingCtx> tracing_ctx = nullptr;
105
106 // Model edge context: copy model edge context
107 auto model_edges = std::make_unique<ir::ModelEdges>(_nnpkg->model_edges());
108
109 // Lower: Assign backend
110 std::unordered_map<ir::ModelIndex,
111 std::unordered_map<ir::SubgraphIndex, std::unique_ptr<compiler::LoweredGraph>>>
112 lowered_subgs;
113
114 for (uint16_t i = 0; i < model_count; i++)
115 {
116 auto const model_index = ir::ModelIndex{i};
117 auto model = _nnpkg->model(model_index);
118
119 model->iterate([&](const ir::SubgraphIndex &subg_index, ir::Graph &subg) {
120 dot_dumper.dump(subg,
121 nnfw::misc::str("before_lower_model-", i, "-subg-", subg_index.value()));
122 // Lower: Assign backend
123 lowered_subgs[model_index][subg_index] =
124 std::make_unique<compiler::LoweredGraph>(subg, *_voptions[i]);
125 // Set tracing_ctx for copied graph
126 if (tracing_ctx != nullptr)
127 tracing_ctx->setSubgraphIndex(&(lowered_subgs[model_index][subg_index]->graph()),
128 subg_index.value());
129 });
130 }
131
132 _nnpkg.reset();
133
134 for (auto &pair : lowered_subgs)
135 {
136 const auto &model_index = pair.first;
137 auto &model_lsubg = pair.second;
138
139 for (auto &pair_inner : model_lsubg)
140 {
141 const auto &subg_index = pair_inner.first;
142 auto &lowered_subg = pair_inner.second;
143 dot_dumper.dump(*lowered_subg, nnfw::misc::str("after_lower_model-", model_index.value(),
144 "-subg-", subg_index.value()));
145 }
146 }
147
148 // Shape inference.
149 for (auto &pair : lowered_subgs)
150 {
151 auto &model_lsubgs = pair.second;
152 // Run the StaticShapeInfer of primary subg. All child StaticShapeInferers are called
153 // recursively
154 std::unordered_map<ir::SubgraphIndex, std::unique_ptr<StaticShapeInferer>> inferers =
155 StaticShapeInferer::createStaticShapeInferers(model_lsubgs);
156
157 const auto primary_subg_idx = ir::SubgraphIndex{0};
158 inferers.at(primary_subg_idx)->infer();
159
160 for (const auto &pair_inferer : inferers)
161 {
162 const auto inferer = pair_inferer.second.get();
163 inferer->dump();
164 }
165 }
166
167 // Shape validation
168 // TODO Move shape independent feature check from ShapeValidator to OperationValidator
169 // TODO Move ShapeValidator into shape inference
170 // - Check input tensor shape validation
171 // - Check parameter value validation which valid value is depend on input tensor shape
172 // - Output tensor shape validation check is needless because
173 // static/dynamic shape inferer will make valid output shape
174 for (auto &pair : lowered_subgs)
175 {
176 auto &model_lsubgs = pair.second;
177
178 for (auto &pair_inner : model_lsubgs)
179 {
180 auto &lowered_subg = pair_inner.second;
181 compiler::ShapeValidator{lowered_subg->graph()}();
182 }
183 }
184
185 /*************************************************************
186 * Backend independent analysis & optimization phase finished
187 *************************************************************/
188 auto executors = std::make_shared<exec::Executors>(std::move(model_edges));
189 for (auto &pair : lowered_subgs)
190 {
191 auto const &model_index = pair.first;
192 auto &model_lsubgs = pair.second;
193
194 for (auto &pair_inner : model_lsubgs)
195 {
196 auto const subg_index = pair_inner.first;
197 auto &lowered_subg = pair_inner.second;
198 auto const indexed_ranks = lowered_subg->indexed_ranks();
199
200 ir::OperationDumper dumper("Executor generation of Subgraph " +
201 std::to_string(subg_index.value()));
202 lowered_subg->graph().operations().iterate(
203 [&](const ir::OperationIndex &, const ir::Operation &op) { op.accept(dumper); });
204
205 auto &options = *_voptions[model_index.value()];
206 auto executor = std::unique_ptr<exec::IExecutor>{ExecutorFactory::get().create(
207 std::move(lowered_subg), tracing_ctx.get(), options, executors, model_index)};
208 executor->setIndexedRanks(indexed_ranks);
209 executors->emplace(model_index, subg_index, std::move(executor));
210 }
211 }
212
213 /********************************
214 * Code generation phase finished
215 ********************************/
216 return std::make_shared<CompilerArtifact>(executors, std::move(tracing_ctx));
217}
