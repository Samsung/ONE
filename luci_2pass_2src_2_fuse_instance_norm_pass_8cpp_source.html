32bool is_1D_with_dummy_dim(luci::CircleConst *node, uint32_t depth)
33{
34 auto rank = node->rank();
35 uint32_t axis;
36 for (axis = 0; axis < rank - 1; ++axis)
37 {
38 if (node->dim(axis).value() != 1)
39 return false;
40 }
41 return node->dim(axis).value() == depth;
42}

44bool is_instance_mean_v1(luci::CircleMean *mean)
45{
46 //
47 // CHECK 1) input is rank 4
48 //
49 auto input = loco::must_cast<luci::CircleNode *>(mean->input());
50 if (input->shape_status() != luci::ShapeStatus::VALID)
51 return false;
52 if (input->rank() != 4)
53 return false;
54
55 //
56 // CHECK 2) 'reduction indices' is CircleConst of value [1,2], that is HW of NHWC
57 //
58 // TODO Support equivalent case, like [-3,-2]
59 // TODO Support non-Const case?
60 // TODO What if input is NCHW format in Circle?
61 auto red_indices = dynamic_cast<luci::CircleConst *>(mean->reduction_indices());
62 if (not red_indices)
63 return false;
64 if (red_indices->rank() != 1)
65 return false;
66 std::set<int32_t> red_indices_set;
67 {
68 // TODO Currently only support S32, support other types
69 assert(red_indices->dtype() == loco::DataType::S32);
70 for (uint32_t i = 0; i < red_indices->dim(0).value(); ++i)
71 red_indices_set.insert(red_indices->at<loco::DataType::S32>(i));
72 }
73 if (red_indices_set.size() != 2)
74 return false;
75 if (red_indices_set.find(1) == red_indices_set.end())
76 return false;
77 if (red_indices_set.find(2) == red_indices_set.end())
78 return false;
79
80 //
81 // CHECK 3) keep_dims == true (?)
82 //
83 // We only have case of 'keep_dims == true' so far, but it might be okay with 'keep_dims == false'
84 // TODO Check this fact, and if true, return true regardless of keep_dims
85 return mean->keep_dims();
86}

89bool is_1D_float32_const(const luci::CircleConst *node, uint32_t channel_size)
90{
91 if (node->rank() != 1)
92 return false;
93
94 if (node->dim(0).value() != channel_size)
95 return false;
96
97 if (node->dtype() != loco::DataType::FLOAT32)
98 return false;
99
100 if (node->size<loco::DataType::FLOAT32>() != channel_size)
101 return false;
102
103 return true;
104}

371#define CHECK_OR_FALSE(condition) \
372 if (not(condition)) \

375bool InstanceNormPattern::condition_common_1_5(uint32_t ifm_channel_depth)
376{
377 add_as_variance = dynamic_cast<luci::CircleAdd *>(rsqrt->x());
378 CHECK_OR_FALSE(add_as_variance);
379
380 CHECK_OR_FALSE(
381 luci::fill(&mean_as_variance, &const_as_epsilon).with_commutative_args_of(add_as_variance));
382
383 CHECK_OR_FALSE(const_as_epsilon->dtype() == loco::DataType::FLOAT32);
384 // TODO Support regarding broadcast
385 CHECK_OR_FALSE(const_as_epsilon->size<loco::DataType::FLOAT32>() == 1);
386
387 CHECK_OR_FALSE(is_instance_mean_v1(mean_as_variance));
388
389 sqdiff = dynamic_cast<luci::CircleSquaredDifference *>(mean_as_variance->input());
390 CHECK_OR_FALSE(sqdiff);
391
392 loco::Node *ifm_should_be = nullptr;
393 CHECK_OR_FALSE(luci::fill(&ifm_should_be, &mean_of_ifm).with_commutative_args_of(sqdiff));
394 CHECK_OR_FALSE(ifm == ifm_should_be);
395 CHECK_OR_FALSE(is_instance_mean_v1(mean_of_ifm));
396 CHECK_OR_FALSE(ifm == mean_of_ifm->input());
397
398 const_as_beta = dynamic_cast<luci::CircleConst *>(sub->x());
399 CHECK_OR_FALSE(const_as_beta);
400 CHECK_OR_FALSE(is_1D_with_dummy_dim(const_as_beta, ifm_channel_depth));
401
402 return true;
403}

405bool InstanceNormPattern::condition_common_3_4()
406{
407 // check left sub
408 ifm = sub->x();
409 CHECK_OR_FALSE(ifm);
410
411 luci::CircleNode *ifm_node = loco::must_cast<luci::CircleNode *>(ifm);
412 CHECK_OR_FALSE(ifm_node->rank() == 4);
413 CHECK_OR_FALSE(ifm_node->dim(3).known());
414
415 mean_of_ifm = dynamic_cast<luci::CircleMean *>(sub->y());
416 CHECK_OR_FALSE(mean_of_ifm);
417 CHECK_OR_FALSE(ifm == mean_of_ifm->input());
418
419 // continue search from add_as_variance
420 CHECK_OR_FALSE(luci::fill(&sqrt, &const_as_epsilon).with_commutative_args_of(add_as_variance));
421 CHECK_OR_FALSE(const_as_epsilon->dtype() == loco::DataType::FLOAT32);
422 // TODO Support regarding broadcast
423 CHECK_OR_FALSE(const_as_epsilon->size<loco::DataType::FLOAT32>() == 1);
424
425 mean_as_variance = dynamic_cast<luci::CircleMean *>(sqrt->x());
426 CHECK_OR_FALSE(mean_as_variance);
427
428 square = dynamic_cast<luci::CircleSquare *>(mean_as_variance->input());
429 CHECK_OR_FALSE(square);
430
431 sub_2 = dynamic_cast<luci::CircleSub *>(square->x());
432 CHECK_OR_FALSE(sub_2);
433 CHECK_OR_FALSE(ifm == sub_2->x());
434
435 mean_of_ifm_2 = dynamic_cast<luci::CircleMean *>(sub_2->y());
436 CHECK_OR_FALSE(mean_of_ifm_2);
437 CHECK_OR_FALSE(ifm == mean_of_ifm_2->input());
438
439 loco::Node *ifm_should_be = nullptr;
440 luci::CircleMean *mean_of_ifm_2_should_be = nullptr;
441 CHECK_OR_FALSE(
442 luci::fill(&ifm_should_be, &mean_of_ifm_2_should_be).with_commutative_args_of(sub_2));
443 CHECK_OR_FALSE(ifm == ifm_should_be);
444 CHECK_OR_FALSE(mean_of_ifm_2 == mean_of_ifm_2_should_be);
445
446 return true;
447}

449template <> bool InstanceNormPattern::match<InstanceNormPattern::PatternVersion::Version_1>()
450{
451 CHECK_OR_FALSE(luci::fill(&mul_as_scaled_ifm, &sub).with_commutative_args_of(add_as_terminal));
452 CHECK_OR_FALSE(luci::fill(&ifm, &mul_gamma).with_commutative_args_of(mul_as_scaled_ifm));
453
454 auto ifm_circle = loco::must_cast<luci::CircleNode *>(ifm);
455 CHECK_OR_FALSE(ifm_circle->shape_status() == luci::ShapeStatus::VALID);
456 CHECK_OR_FALSE(ifm_circle->rank() == 4);
457 CHECK_OR_FALSE(ifm_circle->dim(3).known());
458 uint32_t ifm_channel_depth = ifm_circle->dim(3).value();
459
460 CHECK_OR_FALSE(luci::fill(&rsqrt, &const_as_gamma).with_commutative_args_of(mul_gamma));
461
462 CHECK_OR_FALSE(is_1D_with_dummy_dim(const_as_gamma, ifm_channel_depth));
463
464 CHECK_OR_FALSE(condition_common_1_5(ifm_channel_depth));
465
466 luci::CircleMul *mul_gamma_should_be = nullptr;
467 luci::CircleMean *mean_of_ifm_should_be = nullptr;
468
469 mul_as_scaled_mean = dynamic_cast<luci::CircleMul *>(sub->y());
470 CHECK_OR_FALSE(mul_as_scaled_mean);
471 CHECK_OR_FALSE(luci::fill(&mul_gamma_should_be, &mean_of_ifm_should_be)
472 .with_commutative_args_of(mul_as_scaled_mean));
473 CHECK_OR_FALSE(mul_gamma == mul_gamma_should_be);
474 CHECK_OR_FALSE(mean_of_ifm == mean_of_ifm_should_be);
475
476 _matched = true;
477 return true;
478}

480template <> bool InstanceNormPattern::match<InstanceNormPattern::PatternVersion::Version_2>()
481{
482 CHECK_OR_FALSE(luci::fill(&mul_gamma, &const_as_beta).with_commutative_args_of(add_as_terminal));
483 CHECK_OR_FALSE(luci::fill(&div, &const_as_gamma).with_commutative_args_of(mul_gamma));
484
485 sub = dynamic_cast<luci::CircleSub *>(div->x());
486 CHECK_OR_FALSE(sub);
487
488 ifm = sub->x();
489 CHECK_OR_FALSE(ifm);
490
491 luci::CircleNode *ifm_node = loco::must_cast<luci::CircleNode *>(ifm);
492 CHECK_OR_FALSE(ifm_node->rank() == 4);
493 CHECK_OR_FALSE(ifm_node->dim(3).known());
494 uint32_t ifm_channel_depth = ifm_node->dim(3).value();
495
496 mean_of_ifm = dynamic_cast<luci::CircleMean *>(sub->y());
497 CHECK_OR_FALSE(mean_of_ifm);
498
499 CHECK_OR_FALSE(ifm == mean_of_ifm->input());
500
501 pow = dynamic_cast<luci::CirclePow *>(div->y());
502 CHECK_OR_FALSE(pow);
503
504 add_as_variance = dynamic_cast<luci::CircleAdd *>(pow->x());
505 CHECK_OR_FALSE(add_as_variance);
506
507 luci::CircleConst *zero_point_five = dynamic_cast<luci::CircleConst *>(pow->y());
508 CHECK_OR_FALSE(zero_point_five);
509 CHECK_OR_FALSE(zero_point_five->dtype() == loco::DataType::FLOAT32);
510 // TODO Support regarding broadcast
511 CHECK_OR_FALSE(zero_point_five->size<loco::DataType::FLOAT32>() == 1);
512 CHECK_OR_FALSE(zero_point_five->at<loco::DataType::FLOAT32>(0) == 0.5);
513
514 CHECK_OR_FALSE(
515 luci::fill(&mean_as_variance, &const_as_epsilon).with_commutative_args_of(add_as_variance));
516 CHECK_OR_FALSE(const_as_epsilon->dtype() == loco::DataType::FLOAT32);
517 // TODO Support regarding broadcast
518 CHECK_OR_FALSE(const_as_epsilon->size<loco::DataType::FLOAT32>() == 1);
519
520 CHECK_OR_FALSE(is_instance_mean_v1(mean_as_variance));
521
522 sqdiff = dynamic_cast<luci::CircleSquaredDifference *>(mean_as_variance->input());
523 CHECK_OR_FALSE(sqdiff);
524
525 loco::Node *ifm_should_be = nullptr;
526 luci::CircleMean *mean_of_ifm_should_be = nullptr;
527 CHECK_OR_FALSE(
528 luci::fill(&ifm_should_be, &mean_of_ifm_should_be).with_commutative_args_of(sqdiff));
529 CHECK_OR_FALSE(ifm == ifm_should_be);
530 CHECK_OR_FALSE(mean_of_ifm == mean_of_ifm_should_be);
531
532 // Check for channel size
533 CHECK_OR_FALSE(is_1D_float32_const(const_as_gamma, ifm_channel_depth));
534 CHECK_OR_FALSE(is_1D_float32_const(const_as_beta, ifm_channel_depth));
535
536 _matched = true;
537 return true;
538}

540template <> bool InstanceNormPattern::match<InstanceNormPattern::PatternVersion::Version_3>()
541{
542 CHECK_OR_FALSE(luci::fill(&mul_gamma, &const_as_beta).with_commutative_args_of(add_as_terminal));
543 CHECK_OR_FALSE(luci::fill(&div, &const_as_gamma).with_commutative_args_of(mul_gamma));
544 CHECK_OR_FALSE(luci::fill(&sub, &add_as_variance).with_commutative_args_of(div));
545
546 CHECK_OR_FALSE(condition_common_3_4());
547
548 _matched = true;
549 return true;

552luci::CircleConst *make_const_one(loco::Graph *graph, float value)
553{
554 auto const_one = graph->nodes()->create<luci::CircleConst>();
555 const_one->dtype(loco::DataType::FLOAT32);
556 const_one->rank(1);
557 const_one->size<loco::DataType::FLOAT32>(1);
558 const_one->at<loco::DataType::FLOAT32>(0) = value;
559 return const_one;

562template <> bool InstanceNormPattern::match<InstanceNormPattern::PatternVersion::Version_4>()
563{
564 CHECK_OR_FALSE(div);
565 CHECK_OR_FALSE(luci::fill(&sub, &add_as_variance).with_commutative_args_of(div));
566
567 CHECK_OR_FALSE(condition_common_3_4());
568
569 assert(const_as_gamma == nullptr);
570 assert(const_as_beta == nullptr);
571 assert(mul_gamma == nullptr);
572 assert(add_as_terminal == nullptr);
573
574 // create 1.0 gamma and 0.0 beta
575 auto graph = div->graph();
576 const_as_gamma = make_const_one(graph, 1.0f);
577 const_as_beta = make_const_one(graph, 0.0f);
578 const_as_gamma->name(div->name() + "/gamma");
579 const_as_beta->name(div->name() + "/beta");
580
581 _matched = true;
582 return true;
583}

585template <> bool InstanceNormPattern::match<InstanceNormPattern::PatternVersion::Version_5>()
586{
587 CHECK_OR_FALSE(luci::fill(&mul_as_scaled_ifm, &sub).with_commutative_args_of(add_as_terminal));
588 CHECK_OR_FALSE(luci::fill(&ifm, &rsqrt).with_commutative_args_of(mul_as_scaled_ifm));
589
590 auto ifm_circle = loco::must_cast<luci::CircleNode *>(ifm);
591 CHECK_OR_FALSE(ifm_circle->shape_status() == luci::ShapeStatus::VALID);
592 CHECK_OR_FALSE(ifm_circle->rank() == 4);
593 CHECK_OR_FALSE(ifm_circle->dim(3).known());
594 uint32_t ifm_channel_depth = ifm_circle->dim(3).value();
595
596 CHECK_OR_FALSE(condition_common_1_5(ifm_channel_depth));
597
598 luci::CircleRsqrt *rsqrt_should_be = nullptr;
599 luci::CircleMean *mean_of_ifm_should_be = nullptr;
600
601 mul_as_scaled_mean = dynamic_cast<luci::CircleMul *>(sub->y());
602 CHECK_OR_FALSE(mul_as_scaled_mean);
603 CHECK_OR_FALSE(luci::fill(&rsqrt_should_be, &mean_of_ifm_should_be)
604 .with_commutative_args_of(mul_as_scaled_mean));
605 CHECK_OR_FALSE(rsqrt == rsqrt_should_be);
606 CHECK_OR_FALSE(mean_of_ifm == mean_of_ifm_should_be);
607
608 // mul_gamma is absent
609 // const_as_gamma assume to be 1.0
610 auto graph = add_as_terminal->graph();
611 const_as_gamma = make_const_one(graph, 1.0f);
612 const_as_gamma->name(add_as_terminal->name() + "/gamma");
613
614 _matched = true;
615 return true;
616}

618bool InstanceNormPattern::matched()
619{
620 if (_matched)
621 return true;
622
623 // Check order is DFS
624
625 switch (_pv)
626 {
627 case PatternVersion::Version_1:
628 return match<PatternVersion::Version_1>();
629 case PatternVersion::Version_2:
630 return match<PatternVersion::Version_2>();
631 case PatternVersion::Version_3:
632 return match<PatternVersion::Version_3>();
633 case PatternVersion::Version_4:
634 return match<PatternVersion::Version_4>();
635 case PatternVersion::Version_5:
636 return match<PatternVersion::Version_5>();
637
638 default:
639 break;
640 }
641
642 throw std::runtime_error("Invalid InstanceNorm PatternVersion.");
643}

682void FuseInstanceNorm::reshape_gamma_beta()
683{
684 // Version 1 and 3 need to reshape
685 {
686 _p.const_as_gamma->rank(1);
687 _p.const_as_gamma->dim(0).set(_p.const_as_gamma->size<loco::DataType::FLOAT32>());
688 _p.const_as_beta->rank(1);
689 _p.const_as_beta->dim(0).set(_p.const_as_beta->size<loco::DataType::FLOAT32>());
690
691 _p.const_as_gamma->shape_status(luci::ShapeStatus::UNDEFINED);
692 _p.const_as_beta->shape_status(luci::ShapeStatus::UNDEFINED);
693 }
694}

696luci::CircleInstanceNorm *FuseInstanceNorm::create_inst_norm(loco::Graph *graph)
697{
698 // Make Instance Norm to replace
699 auto instance_norm = graph->nodes()->create<luci::CircleInstanceNorm>();
700 instance_norm->input(_p.ifm);
701 instance_norm->gamma(_p.const_as_gamma);
702 instance_norm->beta(_p.const_as_beta);
703 float epsilon = _p.const_as_epsilon->at<loco::DataType::FLOAT32>(0);
704 instance_norm->epsilon(epsilon);
705 if (_p.add_as_terminal != nullptr)
706 {
707 instance_norm->fusedActivationFunction(_p.add_as_terminal->fusedActivationFunction());
708 // NOTE unique name should be assigned in export
709 instance_norm->name("FusedInstanceNorm/" + _p.add_as_terminal->name());
710 }
711 else
712 {
713 // VERSION_4
714 assert(_p.div != nullptr);
715 instance_norm->fusedActivationFunction(_p.div->fusedActivationFunction());
716 instance_norm->name("FusedInstanceNorm/" + _p.div->name());
717 }
718
719 return instance_norm;
720}

722template <> void FuseInstanceNorm::apply<InstanceNormPattern::PatternVersion::Version_1>()
723{
724 auto graph = _p.add_as_terminal->graph();
725
726 reshape_gamma_beta();
727
728 auto instance_norm = create_inst_norm(graph);
729
730 // set origin
731 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> origin_vec{
732 luci::get_origin(_p.mean_of_ifm),
733 luci::get_origin(_p.sqdiff),
734 luci::get_origin(_p.mean_as_variance),
735 luci::get_origin(_p.add_as_variance),
736 luci::get_origin(_p.rsqrt),
737 luci::get_origin(_p.mul_gamma),
738 luci::get_origin(_p.mul_as_scaled_ifm),
739 luci::get_origin(_p.mul_as_scaled_mean),
740 luci::get_origin(_p.sub),
741 luci::get_origin(_p.add_as_terminal)};
742
743 luci::add_origin(instance_norm, luci::composite_origin(origin_vec));
744
745 replace(_p.add_as_terminal).with(instance_norm);
746}

901{

920{

922
923 auto input_chn = input_const->dim(0).value();

1009bool fuse_instance_norm(luci::CircleAdd *add)
1010{
1011 InstanceNormPattern::PatternVersion pv = InstanceNormPattern::PatternVersion::Version_1;
1012
1013 if (is_add_input_mul_const(add))
1014 pv = InstanceNormPattern::PatternVersion::Version_2;
1015
1016 InstanceNormPattern pattern(add, pv);
1017 if (pattern.matched())
1018 {
1019 FuseInstanceNorm fuse(pattern);
1020 fuse.apply();
1021 return true;
1022 }
1023
1024 if (pv == InstanceNormPattern::PatternVersion::Version_1)
1025 {
1026 // if Version_1 failed, try with Version_5
1027 pv = InstanceNormPattern::PatternVersion::Version_5;
1028 InstanceNormPattern pattern(add, pv);
1029 if (pattern.matched())
1030 {
1031 FuseInstanceNorm fuse(pattern);
1032 fuse.apply();
1033 return true;
1034 }
1035 }
1036 else if (pv == InstanceNormPattern::PatternVersion::Version_2)
1037 {
1038 // if Version_2 failed, try with Version_3

1041 if (pattern.matched())

1043 FuseInstanceNorm fuse(pattern);

1079bool FuseInstanceNormPass::run(loco::Graph *g)
1080{
1081 bool changed = false;
1082
1083 // Check Version_1, Version_2, Version_3, Version_5
1084 for (auto node : loco::active_nodes(loco::output_nodes(g)))
1085 {
1086 auto add = dynamic_cast<luci::CircleAdd *>(node);
1087 if (not add)
1088 continue;
1089
1090 if (fuse_instance_norm(add))
1091 changed = true;
1092 }
1093
1094 // Check Version_4(from DIV) if MUL-ADD pattern is not found
1095 for (auto node : loco::active_nodes(loco::output_nodes(g)))
1096 {
1097 auto div = dynamic_cast<luci::CircleDiv *>(node);
1098 if (not div)
1099 continue;
1100
1101 if (fuse_instance_norm(div))
1102 changed = true;
1103 }
1104
1105 // Post processing of FuseInstanceNorm
1106 for (auto node : loco::active_nodes(loco::output_nodes(g)))
1107 {
1108 auto inst_norm = dynamic_cast<luci::CircleInstanceNorm *>(node);
1109 if (not inst_norm)
1110 continue;
1111
1112 if (post_fusion(inst_norm))
1113 changed = true;
1114 }
1115
1116 return changed;
1117}
