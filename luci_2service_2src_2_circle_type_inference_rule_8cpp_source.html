37 loco::DataType visit(const luci::CircleAddN *node) final
38 {
39 auto dtype = luci::dtype_get(node->inputs(0));

558 loco::DataType visit(const luci::CircleOutput *node) final
559 {
560 auto graph_outputs = node->graph()->outputs();
561 auto graph_output = graph_outputs->at(node->index());
562 auto output_dtype = graph_output->dtype();
563
564 if (dynamic_cast<luci::CircleOutputDummy *>(node->from()) == nullptr &&
565 dynamic_cast<luci::CircleOutputExclude *>(node->from()) == nullptr)
566 {
567 // We don't care for the type if from() is CircleOutputDummy or CircleOutputExclude
568 // from() type should match that of CircleOutput
569 assert(output_dtype == luci::dtype_get(node->from()));
570 }
571 return output_dtype;
572 }

630 {
631 return luci::dtype_get(node->input());

642
643 loco::DataType visit(const luci::CircleWhileOut *node) final
644 {
649 auto circle_while = dynamic_cast<const luci::CircleWhile *>(node->input());
650 if (circle_while == nullptr)
651 {
652 INTERNAL_EXN("CircleWhile IR is not configured correctly");
653 }
654
655 auto index = node->index();
656 auto cond_graph = circle_while->cond_graph();
657 assert(cond_graph != nullptr);
658
659 // Assumption: the index of CircleWhileOut matches with the index of input nodes returned by
660 // loco::input_nodes
661 auto cond_inputs = loco::input_nodes(cond_graph);
662 auto cond_in = loco::must_cast<luci::CircleInput *>(cond_inputs.at(index));
663
664 auto cond_graph_inputs = cond_graph->inputs();
665 auto cond_graph_input = cond_graph_inputs->at(cond_in->index());
666
667 return cond_graph_input->dtype();
668 }

676bool CircleTypeInferenceRule::recognize(const loco::Dialect *d) const
677{
678 return CircleDialect::get() == d;

680
681bool CircleTypeInferenceRule::infer(const loco::Node *node, loco::DataType &dtype) const
682{
683 assert(node->dialect() == CircleDialect::get());
684
685 TypeInferenceAlgorithm alg;
686
687 auto circle_node = loco::must_cast<const CircleNode *>(node);
688 dtype = circle_node->accept(&alg);
689 assert(dtype != loco::DataType::Unknown);
690
691 return true;
692}
