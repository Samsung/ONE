25class QErrorComputer:
26 def __init__(self, fp32_dir, fq_dir):
27 self._fp32_dir = fp32_dir
28 self._fq_dir = fq_dir
29 self.qerror_map = defaultdict(float)
30 self._num_processed_data = 0
31
32 def collect_data_path(self, fp32_dir, fq_dir):
33 # Assumption: FM data are saved as follows
34 #
35 # fp32_dir/
36 # tensors.txt
37 # <DATA_INDEX>/
38 # <TENSOR_NAME>.npy
39 #
40 # fq_dir/
41 # tensors.txt
42 # <DATA_INDEX>/
43 # <TENSOR_NAME>.npy
44 self._num_data = len(list(filter(os.path.isdir, glob.glob(fp32_dir + '/*'))))
45 if self._num_data != len(list(filter(os.path.isdir, glob.glob(fq_dir + '/*')))):
46 raise RuntimeError("Number of data mistmatches")
47
48 self._num_processed_data += self._num_data
49
50 self._filename_to_tensor = dict()
51 with open(Path(fp32_dir) / 'tensors.txt') as f:
52 tensors = set([line.rstrip() for line in f])
53 for tensor in tensors:
54 # Check if filename is unique
55 # Fix name finding logic unless
56 assert to_filename(tensor) not in self._filename_to_tensor
57 self._filename_to_tensor[to_filename(tensor)] = tensor
58
59 # Save paths to fp32 data and fq data for each tensor
60 # dict
61 # {
62 # <tensor_name>: (fp32_path, fq_path),
63 # <tensor_name>: (fp32_path, fq_path),
64 # ...
65 # }
66 data_paths = dict()
67 for data_idx in range(self._num_data):
68 fp32_results = glob.glob(fp32_dir + '/' + str(data_idx) + '/*.npy')
69 for fp32_data_path in fp32_results:
70 fp32_path = Path(fp32_data_path)
71 fq_data_path = fq_dir + '/' + str(data_idx) + '/' + fp32_path.with_suffix(
72 '.npy').name
73 fq_path = Path(fq_data_path)
74 filename = fp32_path.stem
75 tensor_name = self._filename_to_tensor[filename]
76
77 # Only save the tensors which have both fp32 data and fq data
78 if fq_path.is_file() and fp32_path.is_file():
79 if tensor_name in data_paths:
80 data_paths[tensor_name].append((fp32_data_path, fq_data_path))
81 else:
82 data_paths[tensor_name] = [(fp32_data_path, fq_data_path)]
83
84 return data_paths

87 '''Return qerror map (dict: tensor_name(string) -> qerror(float)).'''
88 raise NotImplementedError # Child must implement this

91class MPEIRComputer(QErrorComputer):
92 def __init__(self, fp32_dir, fq_dir):
93 super().__init__(fp32_dir, fq_dir)
94
95 # Incrementally compute Qerror while traversing all data in fp32_dir and fq_dir
96 def advance_on(self, fp32_dir, fq_dir):
97 data_paths = self.collect_data_path(fp32_dir, fq_dir)
98 for tensor_name, data_path in data_paths.items():
99 for (fp32_data_path, fq_data_path) in data_path:
100 fp32_data = np.load(fp32_data_path)
101 fq_data = np.load(fq_data_path)
102
103 diff = np.absolute(fp32_data - fq_data).reshape(-1)
104
105 fp32_min = np.min(fp32_data.reshape(-1))
106 fp32_max = np.max(fp32_data.reshape(-1))
107
108 # Peak Error-to-Interval Ratio (PEIR)
109 # NOTE: PEIR is an analogue of PSNR (Peak Signal to Noise Ratio)
110 PEAK_ERROR = np.max(diff)
111 INTERVAL = fp32_max - fp32_min
112
113 # If INTERVAL is 0, PEIR becomes NaN.
114 # To prevent this, relaxed PEIR with epsilon(10^(-6)) is used.
115 rPEIR = PEAK_ERROR / (INTERVAL + 0.000001)
116
117 self.qerror_map[tensor_name] += rPEIR
118
119 # Return
120 # qerror_map (dict: tensor_name(string) -> qerror(float))
121 # qerror_min (float)
122 # qerror_max (float)
123 def get_final_result(self):
124 qerror_map = dict()
125 for tensor_name, acc in self.qerror_map.items():
126 qerror_map[tensor_name] = acc / self._num_processed_data
127
128 # Fixed qerror_min (0), qerror_max (1)
129 return qerror_map, 0.0, 1.0
130
131 def run(self):
132 self.advance_on(self._fp32_dir, self._fq_dir)
133 return self.get_final_result()

209# Scaled Root Mean Square Error (SRMSE)
210# SRMSE = sqrt(MSE) / scale
211class SRMSEComputer(QErrorComputer):
212 def __init__(self, fp32_dir, fq_dir):
213 super().__init__(fp32_dir, fq_dir)
214 if fq_dir != None:
215 self.scale_file = Path(fq_dir) / 'scales.txt'
216
217 # Incrementally compute Qerror while traversing all data in fp32_dir and fq_dir
218 def advance_on(self, fp32_dir, fq_dir):
219 if fq_dir != None:
220 self.scale_file = Path(fq_dir) / 'scales.txt'
221 self._fq_dir_fq_dir = fq_dir
222
223 data_paths = self.collect_data_path(fp32_dir, fq_dir)
224
225 for tensor_name, data_path in data_paths.items():
226 for (fp32_data_path, fq_data_path) in data_path:
227 fp32_data = np.load(fp32_data_path)
228 fq_data = np.load(fq_data_path)
229
230 MSE = np.square(fp32_data - fq_data).mean()
231
232 self.qerror_map[tensor_name] += MSE
233
234 # Return
235 # qerror_map (dict: tensor_name(string) -> qerror(float))
236 # qerror_min (float)
237 # qerror_max (float)
238 def get_final_result(self):
239 with open(self.scale_file) as f:
240 # scale_map: {tensor_name(str) -> scale(float)}
241 scale_map = json.load(f)
242
243 qerror_max = 0.0
244 qerror_map = dict()
245 for tensor_name, acc in self.qerror_map.items():
246 MSE = acc / self._num_processed_data
247 SRMSE = np.sqrt(MSE) / scale_map[tensor_name]
248 qerror_map[tensor_name] = SRMSE
249 qerror_max = max(SRMSE, qerror_max)
250
251 return qerror_map, 0.0, qerror_max
252
253 def run(self):
254 self.advance_on(self._fp32_dir, self._fq_dir_fq_dir)
255 return self.get_final_result()
