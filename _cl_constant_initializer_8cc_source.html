57void ClConstantInitializer::registerCopyInitializer(const ir::OperandIndex &index,
58 const ir::Operand &obj)
59{
60 // For only CONSTANTS
61 // TODO Add to check if tensor has been allocated
62 if (!obj.isConstant())
63 return;
64
65 const auto type = obj.typeInfo().type();
66 using ir::DataType;
67
68 switch (type)
69 {
70 case DataType::FLOAT32:
71 _init_map[index] = copyInit<float>;
72 break;
73 default:
74 throw std::runtime_error("Not supported, yet");
75 break;
76 }
77}

79void ClConstantInitializer::registerPermuteInitializer(const ir::OperandIndex &index,
80 const ir::Operand &obj)
81{
82 // For only CONSTANTS
83 // TODO Add to check if tensor has been allocated
84 if (!obj.isConstant())
85 return;
86
87 const auto type = obj.typeInfo().type();
88 using ir::DataType;
89 using namespace std::placeholders;
90
91 switch (type)
92 {
93 case DataType::FLOAT32:
94 _init_map[index] = std::bind(permuteInit<float>, _1, _2, _current_layout);
95 break;
96 case DataType::INT32:
97 _init_map[index] = std::bind(permuteInit<int32_t>, _1, _2, _current_layout);
98 break;
99 default:
100 throw std::runtime_error("Not supported, yet");
101 break;
102 }
103}
