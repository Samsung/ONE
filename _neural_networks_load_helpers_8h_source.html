102inline void* loadFunction(const char* name) {
103 void* fn = nullptr;
104 if (getLibraryHandle() != nullptr) {
105 fn = dlsym(getLibraryHandle(), name);
106 }
107 if (fn == nullptr) {
108 NNAPI_LOG("nnapi error: unable to open function %s", name);
109 abort();
110 }
111 else {
112#ifdef _GNU_SOURCE
113 Dl_info info;
114 if (dladdr(fn, &info))
115 {
116 NNAPI_LOG("nnapi function '%s' is loaded from '%s' ", name, info.dli_fname);
117 }
118 else
119 {
120 NNAPI_LOG("nnapi function '%s' is failed to load", name);
121 }
122#endif // _GNU_SOURCE
123 }
124 return fn;
125}
