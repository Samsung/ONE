45void KernelGenerator::visit(const ir::operation::Pool2D &node)
46{
47 const auto ofm_index{node.getOutputs().at(0)};
48 const auto ifm_index{node.getInputs().at(ir::operation::Pool2D::Input::INPUT)};
49
50 const auto kh = node.param().kh;
51 const auto kw = node.param().kw;
52 const auto stride = node.param().stride;
53 const auto ifm_shape = _ctx.at(ifm_index).shape().asFeature();
54 const auto ofm_shape = _ctx.at(ofm_index).shape().asFeature();
55 const auto padding =
56 ir::calculatePadding(node.param().padding, ifm_shape, ofm_shape, stride, kw, kh);
57 const auto activation = node.param().activation;
58
59 auto ofm_tensor = _tensor_reg->getPortableTensor(ofm_index);
60 auto ifm_tensor = _tensor_reg->getPortableTensor(ifm_index);
61
62 auto fn = std::make_unique<ops::PoolLayer>();
63
64 fn->configure(ifm_tensor, padding.left, padding.right, padding.top, padding.bottom,
65 stride.horizontal, stride.vertical, kw, kh, activation, ofm_tensor,
66 convertPoolType(node.param().op_type));
67
68 _return_fn = std::move(fn);
69}

94template <typename T>
95std::function<void(const IPortableTensor *, IPortableTensor *)>
96generateKernelGeneric(const nnfw::cker::PoolParams &params, PoolType op_type)
97{
98 if (op_type == PoolType::kAvg)
99 {
100 return std::bind(&avgPool2D<T>, params, std::placeholders::_1, std::placeholders::_2);
101 }
102 else if (op_type == PoolType::kMax)
103 {
104 return std::bind(&maxPool2D<T>, params, std::placeholders::_1, std::placeholders::_2);
105 }
106 else
107 {
108 throw std::runtime_error{"Pool: unsupported pool type"};
109 }
110}

131void PoolLayer::configure(const IPortableTensor *input, const uint32_t paddingLeft, const uint32_t,
132 const uint32_t paddingTop, const uint32_t, const uint32_t strideWidth,
133 const uint32_t strideHeight, const uint32_t kernelWidth,
134 const uint32_t kernelHeight, const ir::Activation activation,
135 IPortableTensor *output, const PoolType op_type)
136{
137 assert(input != nullptr);
138 assert(output != nullptr);
139
140 _input = input;
141 _output = output;
142
143 POOLING_PARAMETERS
144
145 switch (_input->data_type())
146 {
147 case OperandType::FLOAT32:
148 {
149 float output_activation_min = 0;
150 float output_activation_max = 0;
151 CalculateActivationRange<float>(activation, &output_activation_min, &output_activation_max);
152 op_params.float_activation_min = output_activation_min;
153 op_params.float_activation_max = output_activation_max;
154
155 _kernel = generateKernelGeneric<float>(op_params, op_type);
156 break;
157 }
158 case OperandType::QUANT_UINT8_ASYMM:
159 {
160 int32_t output_activation_min = 0;
161 int32_t output_activation_max = 0;
162 CalculateActivationRangeQuantized(activation, _output, &output_activation_min,
163 &output_activation_max);
164 op_params.quantized_activation_min = output_activation_min;
165 op_params.quantized_activation_max = output_activation_max;
166 _kernel = generateKernelGeneric<uint8_t>(op_params, op_type);
167 break;
168 }
169 case OperandType::QUANT_INT8_ASYMM:
170 {
171 int32_t output_activation_min = 0;
172 int32_t output_activation_max = 0;
173 CalculateActivationRangeQuantized(activation, _output, &output_activation_min,
174 &output_activation_max);
175 op_params.quantized_activation_min = output_activation_min;
176 op_params.quantized_activation_max = output_activation_max;
177 _kernel = generateKernelGeneric<int8_t>(op_params, op_type);
178 break;
179 }
180 default:
181 throw std::runtime_error{"Pool: unsupported data type"};
182 }
183}
