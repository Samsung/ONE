30void set_filter_enc(loco::FilterEncode *filter_enc)
31{
32 auto enc = std::make_unique<loco::PermutingEncoder<loco::Domain::Filter>>();
33
34 // In TensorFlow, conv2d filter is a 4-D tensor of following shape:
35 // [filter_height, filter_width, in_channels, out_channels] -> HWIO (HWCN)
36 enc->perm()->axis(loco::FilterAxis::Height) = 0;
37 enc->perm()->axis(loco::FilterAxis::Width) = 1;
38 enc->perm()->axis(loco::FilterAxis::Depth) = 2;
39 enc->perm()->axis(loco::FilterAxis::Count) = 3;
40
41 filter_enc->encoder(std::move(enc));
42}

97 auto input_plane_shape = moco::make_plane_shape(input_feature_shape);

100 *conv2d->stride() = node_stride;
101
102 // update graph
103 auto node_A = node->input();
104 auto node_B = node->filter();

108 filter_enc->input(node_B);
109 conv2d->ifm(feature_enc);
110 conv2d->ker(filter_enc);
