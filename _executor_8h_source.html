28// Information we maintain about each operand during execution that
29// may change during execution.
30struct RunTimeOperandInfo
31{
32 // TODO Storing the type here is redundant, as it won't change during execution.
33 OperandType type;
34
35 // The type and dimensions of the operand. The dimensions can
36 // change at runtime. We include the type because it's useful
37 // to pass together with the dimension to the functions implementing
38 // the operators.
39 //
40 // Q: Is it possible??
41 std::vector<uint32_t> dimensions;
42 float scale;
43 int32_t zeroPoint;
44
45 // Where the operand's data is stored. Check the corresponding
46 // location information in the model to figure out if this points
47 // to memory we have allocated for an temporary operand.
48 uint8_t *buffer;
49 // The length of the buffer.
50 uint32_t length;
51
52 // Whether this is a temporary variable, a model input, a constant, etc.
53 OperandLifeTime lifetime;
54
55 // Keeps track of how many operations have yet to make use
56 // of this temporary variable. When the count is decremented to 0,
57 // we free the buffer. For non-temporary variables, this count is
58 // always 0.
59 uint32_t numberOfUsesLeft;
60
61 Shape shape() const
62 {
63 return Shape{.type = type, .dimensions = dimensions, .scale = scale, .offset = zeroPoint};
64 }
65};
