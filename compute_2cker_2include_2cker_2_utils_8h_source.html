45inline void QuantizeMultiplier(double double_multiplier, int32_t *quantized_multiplier, int *shift)
46{
47 if (double_multiplier == 0.)
48 {
49 *quantized_multiplier = 0;
50 *shift = 0;
51 return;
52 }
53
54 const double q = std::frexp(double_multiplier, shift);
55 auto q_fixed = static_cast<int64_t>(round(q * (1ll << 31)));
56
57 assert(q_fixed <= (1ll << 31));
58 if (q_fixed == (1ll << 31))
59 {
60 q_fixed /= 2;
61 ++*shift;
62 }
63 assert(q_fixed <= std::numeric_limits<int32_t>::max());
64 // A shift amount smaller than -31 would cause all bits to be shifted out
65 // and thus all results would be zero. We implement that instead with
66 // q_fixed==0, so as to avoid hitting issues with right-shift
67 // operations with shift amounts greater than 31. Note that this happens
68 // roughly when abs(double_multiplier) < 2^-31 and the present handling means
69 // that we're effectively flushing tiny double_multiplier's to zero.
70 // We could conceivably handle values in the range (roughly) [32, 63]
71 // as 'denormals' i.e. (shift==0, q_fixed < 2^30). In that point of view
72 // the present handling is just doing 'flush denormals to zero'. We could
73 // reconsider and actually generate nonzero denormals if a need arises.
74 if (*shift < -31)
75 {
76 *shift = 0;
77 q_fixed = 0;
78 }
79 *quantized_multiplier = static_cast<int32_t>(q_fixed);
80}

82inline void QuantizeMultiplierSmallerThanOneExp(double double_multiplier,
83 int32_t *quantized_multiplier, int *left_shift)

89 assert(shift <= 0);

93inline int32_t MultiplyByQuantizedMultiplier(int32_t x, int32_t quantized_multiplier, int shift)
94{
95 int left_shift = shift > 0 ? shift : 0;
96 int right_shift = shift > 0 ? 0 : -shift;
97 return gemmlowp::RoundingDivideByPOT(
98 gemmlowp::SaturatingRoundingDoublingHighMul(x * (1 << left_shift), quantized_multiplier),
99 right_shift);
100}

117inline int32x4x4_t MultiplyByQuantizedMultiplier4Rows(int32x4x4_t input_val,
118 int32_t quantized_multiplier, int32_t shift)
119{
120 const int left_shift = std::max(shift, 0);
121 const int right_shift = std::min(shift, 0);
122 int32x4x4_t result;
123
124 int32x4_t multiplier_dup = vdupq_n_s32(quantized_multiplier);
125 int32x4_t left_shift_dup = vdupq_n_s32(left_shift);
126 int32x4_t right_shift_dup = vdupq_n_s32(right_shift);
127
128 result.val[0] = vrshlq_s32(
129 vqrdmulhq_s32(vshlq_s32(input_val.val[0], left_shift_dup), multiplier_dup), right_shift_dup);
130
131 result.val[1] = vrshlq_s32(
132 vqrdmulhq_s32(vshlq_s32(input_val.val[1], left_shift_dup), multiplier_dup), right_shift_dup);
133
134 result.val[2] = vrshlq_s32(
135 vqrdmulhq_s32(vshlq_s32(input_val.val[2], left_shift_dup), multiplier_dup), right_shift_dup);
136
137 result.val[3] = vrshlq_s32(
138 vqrdmulhq_s32(vshlq_s32(input_val.val[3], left_shift_dup), multiplier_dup), right_shift_dup);
139
140 return result;
141}

149inline int CountLeadingZeros(uint32_t integer_input)
150{
151 const uint32_t one_in_leading_positive = 1U << 31;
152 int leading_zeros = 0;
153 while (integer_input < one_in_leading_positive)
154 {
155 integer_input <<= 1;
156 ++leading_zeros;
157 }
158 return leading_zeros;
159}

161inline void GetInvSqrtQuantizedMultiplierExp(int32_t input, int reverse_shift,
162 int32_t *output_inv_sqrt, int *output_shift)
163{
164 assert(input >= 0);
165 if (input <= 1)
166 {
167 // Handle the input value 1 separately to avoid overflow in that case
168 // in the general computation below (b/143972021). Also handle 0 as if it
169 // were a 1. 0 is an invalid input here (divide by zero) and 1 is a valid
170 // but rare/unrealistic input value. We can expect both to occur in some
171 // incompletely trained models, but probably not in fully trained models.
172 *output_inv_sqrt = std::numeric_limits<std::int32_t>::max();
173 *output_shift = 0;
174 return;
175 }
176 assert(input > 1);
177 *output_shift = 11;
178 while (input >= (1 << 29))
179 {
180 input /= 4;
181 ++*output_shift;
182 }
183 const unsigned max_left_shift_bits = CountLeadingZeros(static_cast<uint32_t>(input)) - 1;
184 const unsigned max_left_shift_bit_pairs = max_left_shift_bits / 2;
185 const unsigned left_shift_bit_pairs = max_left_shift_bit_pairs - 1;
186 *output_shift -= left_shift_bit_pairs;
187 input <<= 2 * left_shift_bit_pairs;
188 assert(input >= (1 << 27));
189 assert(input < (1 << 29));
190 using gemmlowp::FixedPoint;
191 using gemmlowp::Rescale;
192 using gemmlowp::SaturatingRoundingMultiplyByPOT;
193 // Using 3 integer bits gives us enough room for the internal arithmetic in
194 // this Newton-Raphson iteration.
195 using F3 = FixedPoint<int32_t, 3>;
196 using F0 = FixedPoint<int32_t, 0>;
197 const F3 fixedpoint_input = F3::FromRaw(input >> 1);
198 const F3 fixedpoint_half_input = SaturatingRoundingMultiplyByPOT<-1>(fixedpoint_input);
199 const F3 fixedpoint_half_three =
200 GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F3, (1 << 28) + (1 << 27), 1.5);
201 // Newton-Raphson iteration
202 // Naive unoptimized starting guess: x = 1
203 F3 x = F3::One();
204 // Naive unoptimized number of iterations: 5
205 for (int i = 0; i < 5; i++)
206 {
207 const F3 x3 = Rescale<3>(x * x * x);
208 x = Rescale<3>(fixedpoint_half_three * x - fixedpoint_half_input * x3);
209 }
210 const F0 fixedpoint_half_sqrt_2 =
211 GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F0, 1518500250, std::sqrt(2.) / 2.);
212 x = x * fixedpoint_half_sqrt_2;
213 *output_inv_sqrt = x.raw();
214 if (*output_shift < 0)
215 {
216 *output_inv_sqrt <<= -*output_shift;
217 *output_shift = 0;
218 }
219 // Convert right shift (right is positive) to left shift.
220 *output_shift *= reverse_shift;
221}

223// Comment from tensorflow lite:
224//
225// DO NOT USE THIS STRUCT FOR NEW FUNCTIONALITY BEYOND IMPLEMENTING
226// BROADCASTING.
227//
228// NdArrayDesc<N> describes the shape and memory layout of an N-dimensional
229// rectangular array of numbers.
230//
231// NdArrayDesc<N> is basically identical to Dims<N> defined in types.h.
232// However, as Dims<N> is to be deprecated, this class exists as an adaptor
233// to enable simple unoptimized implementations of element-wise broadcasting
234// operations.
235template <int N> struct NdArrayDesc
236{
237 // The "extent" of each dimension. Indices along dimension d must be in the
238 // half-open interval [0, extents[d]).
239 int extents[N];
240
241 // The number of *elements* (not bytes) between consecutive indices of each
242 // dimension.
243 int strides[N];
244};

251// Same as Offset(), except takes as NdArrayDesc<N> instead of Dims<N>.
252inline int SubscriptToIndex(const NdArrayDesc<4> &desc, int i0, int i1, int i2, int i3)
253{
254 assert(i0 >= 0 && i0 < desc.extents[0]);
255 assert(i1 >= 0 && i1 < desc.extents[1]);
256 assert(i2 >= 0 && i2 < desc.extents[2]);
257 assert(i3 >= 0 && i3 < desc.extents[3]);
258 return i0 * desc.strides[0] + i1 * desc.strides[1] + i2 * desc.strides[2] + i3 * desc.strides[3];
259}

261template <int N> inline int SubscriptToIndexGeneric(const NdArrayDesc<N> *desc, int *iter)
262{
263 int ret_indx = 0;
264 for (size_t idx = 0; idx < static_cast<size_t>(N); idx++)
265 {
266 assert(iter[idx] >= 0 && iter[idx] < desc->extents[idx]);
267 ret_indx += iter[idx] * desc->strides[idx];
268 }
269
270 return ret_indx;
271}

273// Copies dims to desc, calculating strides.
274template <int N> inline void CopyDimsToDesc(const Shape &input_shape, NdArrayDesc<N> *desc_out)
275{
276 int desc_stride = 1;
277 for (int i = N - 1; i >= 0; --i)
278 {
279 desc_out->extents[i] = input_shape.Dims(i);
280 desc_out->strides[i] = desc_stride;
281 desc_stride *= input_shape.Dims(i);
282 }
283}

285template <int N>
286inline void
287NdArrayDescsForElementwiseBroadcast(const Shape &input0_shape, const Shape &input1_shape,
288 NdArrayDesc<N> *desc0_out, NdArrayDesc<N> *desc1_out)
289{
290 assert(desc0_out != nullptr);
291 assert(desc1_out != nullptr);
292
293 auto extended_input0_shape = Shape::ExtendedShape(N, input0_shape);
294 auto extended_input1_shape = Shape::ExtendedShape(N, input1_shape);
295
296 // Copy dims to desc, calculating strides.
297 CopyDimsToDesc<N>(extended_input0_shape, desc0_out);
298 CopyDimsToDesc<N>(extended_input1_shape, desc1_out);
299
300 // Walk over each dimension. If the extents are equal do nothing.
301 // Otherwise, set the desc with extent 1 to have extent equal to the other and
302 // stride 0.
303 for (int i = 0; i < N; ++i)
304 {
305 const int extent0 = extended_input0_shape.Dims(i);
306 const int extent1 = extended_input1_shape.Dims(i);
307 if (extent0 != extent1)
308 {
309 if (extent0 == 1)
310 {
311 desc0_out->strides[i] = 0;
312 desc0_out->extents[i] = extent1;
313 }
314 else
315 {
316 assert(extent1 == 1);
317 desc1_out->strides[i] = 0;
318 desc1_out->extents[i] = extent0;
319 }
320 }
321 }
322}

324template <int N>
325inline void
326NdArrayDescsForElementwiseBroadcast(const Shape &input0_shape, const Shape &input1_shape,
327 const Shape &input2_shape, NdArrayDesc<N> *desc0_out,
328 NdArrayDesc<N> *desc1_out, NdArrayDesc<N> *desc2_out)
329{
330 assert(desc0_out != nullptr);
331 assert(desc1_out != nullptr);
332 assert(desc2_out != nullptr);
333
334 auto extended_input0_shape = Shape::ExtendedShape(N, input0_shape);
335 auto extended_input1_shape = Shape::ExtendedShape(N, input1_shape);
336 auto extended_input2_shape = Shape::ExtendedShape(N, input2_shape);
337
338 // Copy dims to desc, calculating strides.
339 CopyDimsToDesc<N>(extended_input0_shape, desc0_out);
340 CopyDimsToDesc<N>(extended_input1_shape, desc1_out);
341 CopyDimsToDesc<N>(extended_input2_shape, desc2_out);
342
343 // Walk over each dimension. If the extents are equal do nothing.
344 // Otherwise, set the desc with extent 1 to have extent equal to the other and
345 // stride 0.
346 for (int i = 0; i < N; ++i)
347 {
348 const int extent0 = extended_input0_shape.Dims(i);
349 const int extent1 = extended_input1_shape.Dims(i);
350 const int extent2 = extended_input2_shape.Dims(i);
351
352 int extent = extent0;
353 if (extent1 != 1)
354 extent = extent1;
355 if (extent2 != 1)
356 extent = extent2;
357
358 assert(extent0 == 1 || extent0 == extent);
359 assert(extent1 == 1 || extent1 == extent);
360 assert(extent2 == 1 || extent2 == extent);
361
362 if (!(extent0 == extent1 && extent1 == extent2))
363 {
364 if (extent0 == 1)
365 {
366 desc0_out->strides[i] = 0;
367 desc0_out->extents[i] = extent;
368 }
369 if (extent1 == 1)
370 {
371 desc1_out->strides[i] = 0;
372 desc1_out->extents[i] = extent;
373 }
374 if (extent2 == 1)
375 {
376 desc2_out->strides[i] = 0;
377 desc2_out->extents[i] = extent;
378 }
379 }
380 }

383// Gets next index to iterate through a multidimensional array.
384inline bool NextIndex(const int num_dims, const int *dims, int *current)
385{
386 if (num_dims == 0)
387 {
388 return false;
389 }
390 assert(dims != nullptr);
391 assert(current != nullptr);
392 int carry = 1;
393 for (int idx = num_dims - 1; idx >= 0; --idx)
394 {
395 int current_val = current[idx] + carry;
396 assert(dims[idx] >= current_val);
397 if (dims[idx] == current_val)
398 {
399 current[idx] = 0;
400 }
401 else
402 {
403 current[idx] = current_val;
404 carry = 0;
405 break;
406 }
407 }
408 return (carry == 0);
409}

411// Gets offset of index if reducing on axis. When reducing, the flattened offset
412// will not change, if the input index changes on the given axis. For example,
413// if you have a 3D tensor and you are reducing to 2D by eliminating axis 0,
414// then index (0, 1, 2) and index (1, 1, 2) will map to the same flattened
415// offset.
416// TODO(kanlig): uses Dims to represent dimensions.
417inline size_t ReducedOutputOffset(const int num_dims, const int *dims, const int *index,
418 const int num_axis, const int *axis)
419{
420 if (num_dims == 0)
421 {
422 return 0;
423 }
424
425 assert(dims != nullptr);
426 assert(index != nullptr);
427
428 size_t offset = 0;
429 for (int idx = 0; idx < num_dims; ++idx)
430 {
431 // if we need to skip this axis
432 bool is_axis = false;
433 if (axis != nullptr)
434 {
435 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx)
436 {
437 if (idx == axis[axis_idx])
438 {
439 is_axis = true;
440 break;
441 }
442 }
443 }
444 if (!is_axis)
445 {
446 offset = offset * static_cast<size_t>(dims[idx]) + static_cast<size_t>(index[idx]);
447 }
448 }
449 return offset;
450}
