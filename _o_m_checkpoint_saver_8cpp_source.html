31bool isTrainableWeights(const circle::OperatorCode *opcode)
32{
33 switch (opcode->builtin_code())
34 {
35 case circle::BuiltinOperator_FULLY_CONNECTED:
36 case circle::BuiltinOperator_CONV_2D:
37 return true;
38 default:
39 return false;
40 }
41}

51size_t OMCheckpointSaver::calculateFileSize(core::OMRuntimeContext &context,
52 OMTrainingStorage &train_storage,
53 const OMConfig &config)
54{
55 size_t result = 0;
56
57 // 2 bytes for Magic Number
58 result += 2;
59
60 // 1 byte for Schema version
61 result += 1;
62
63 // 1 byte for Reserved field
64 result += 1;
65
66 // 4 bytes for Adam's state m buffers offset value
67 result += 4;
68
69 // 4 bytes for Adam's state v buffers offset value
70 result += 4;
71
72 // 4 bytes for others parameters offset value
73 result += 4;
74
75 // 4 bytes for number of tensors
76 result += 4;
77
78 auto tensors = context.getCircleTensors();
79
80 auto tensors_size = tensors->size();
81
82 // tensors_size * 4 bytes for buffers offsets
83 result += tensors_size * 4;
84
85 const auto *operators = context.getCircleOperators();
86 const auto num_kernels = operators->size();
87 uint32_t num_train_layers = config.training_context.num_of_train_layers == 0
88 ? num_kernels
89 : config.training_context.num_of_train_layers;
90 uint32_t last_node_pos = std::min(num_kernels, num_train_layers);
91 // Goes among trainable ops
92 size_t buffer_size = 0;
93 const auto *op_codes = context.getCircleOpcodes();
94 for (uint32_t index = 0; index < last_node_pos; ++index)
95 {
96 uint32_t cur_op_index = num_kernels - index - 1;
97 auto *cur_op = operators->operator[](cur_op_index);
98
99 auto input_tensors = cur_op->inputs();
100
101 for (uint32_t i = 0; i < input_tensors->size(); ++i)
102 {
103 const auto input_tensor_index = input_tensors->operator[](i);
104 // Check is const
105 if (not context.isConstTensor(input_tensor_index))
106 continue;
107
108 uint32_t cur_opcode_index = cur_op->opcode_index();
109
110 assert(cur_opcode_index < op_codes->size());
111
112 const auto opcode = op_codes->operator[](cur_opcode_index);
113
114 // Check it is trainable const
115 if (not isTrainableWeights(opcode))
116 continue;
117
118 const auto tensor = context.getTensorByIndex(input_tensor_index);
119 OMRuntimeShape shape(tensor);
120
121 auto type_size = sizeof(OMDataType(tensor->type()));
122
123 buffer_size += type_size * shape.flatSize();
124 }
125 }
126
127 // If we use Adam optimizer then need to add Adam specific buffers
128 assert(config.training_context.optimizer == SGD or
129 config.training_context.optimizer == ADAM && "Unsupported type");
130 if (config.training_context.optimizer == ADAM)
131 {
132 // Check is it save any state
133 if (not train_storage.getAdam()->isReset())
134 {
135 // If yes - then just buffer_size = buffer_size * 3 (original weights and two buffers from
136 // Adam state)
137 buffer_size *= 3;
138
139 // Add offsets for m
140 result += tensors_size * 4;
141 // Add offsets for v
142 result += tensors_size * 4;
143 }
144 }
145
146 // Add buffer size
147 result += buffer_size;
148
149 // 4 bytes to save information about current step
150 result += 4;
151
152 // 4 bytes to save information about current epoch
153 result += 4;
154
155 return result;
156}

158OMStatus OMCheckpointSaver::writeOffsetsAndBuffers(core::OMRuntimeContext &context,
159 OMTrainingStorage &train_storage,
160 const OMConfig &config, std::vector<char> &data)
161{
162 // Point to start of the buffer
163 char *cur_ptr = data.data();
164
165 // Set to the n_buffers field
166 cur_ptr += 16;
167
168 auto tensors = context.getCircleTensors();
169 auto tensors_size = tensors->size();
170
171 // Write number of buffers
172 std::memcpy(cur_ptr, &tensors_size, sizeof(tensors_size));
173
174 // Calculate buffers offsets, set all to zeros
175 // Zero value means that there is no buffer for this tensor
176 std::vector<uint32_t> offsets(tensors_size, 0);
177 // Start offset for buffers
178 uint32_t cur_offset = 20 + tensors_size * 4;
179
180 // To calculate sum of saved buffers sizes
181 uint32_t acc_buffer_size = 0;
182
183 const auto *operators = context.getCircleOperators();
184 const auto num_kernels = operators->size();
185 uint32_t num_train_layers = config.training_context.num_of_train_layers == 0
186 ? num_kernels
187 : config.training_context.num_of_train_layers;
188 uint32_t last_node_pos = std::min(num_kernels, num_train_layers);
189 // Goes among trainable ops
190
191 const auto *op_codes = context.getCircleOpcodes();
192
193 for (uint32_t index = 0; index < last_node_pos; ++index)
194 {
195 uint32_t cur_op_index = num_kernels - index - 1;
196 auto *cur_op = operators->operator[](cur_op_index);
197
198 auto input_tensors = cur_op->inputs();
199
200 for (uint32_t i = 0; i < input_tensors->size(); ++i)
201 {
202 const auto input_tensor_index = input_tensors->operator[](i);
203 // Check is const
204 if (not context.isConstTensor(input_tensor_index))
205 continue;
206
207 uint32_t cur_opcode_index = cur_op->opcode_index();
208
209 assert(cur_opcode_index < op_codes->size());
210
211 const auto opcode = op_codes->operator[](cur_opcode_index);
212
213 // Check it is trainable const
214 if (not isTrainableWeights(opcode))
215 continue;
216
217 // Found trainable weight tensor, lets calculate its size and save offset for current buffer
218 const auto tensor = context.getTensorByIndex(input_tensor_index);
219 OMRuntimeShape shape(tensor);
220
221 auto type_size = sizeof(OMDataType(tensor->type()));
222
223 size_t buffer_size = type_size * shape.flatSize();
224 // Save for current tensor index its offset
225 offsets[input_tensor_index] = cur_offset;
226 // Get buffer data
227 uint8_t *tensor_data;
228 OMStatus status = context.getConstDataByTensorIndex(&tensor_data, input_tensor_index);
229 assert(status == Ok);
230 if (status != Ok)
231 return status;
232 assert(tensor_data != nullptr);
233
234 // Write buffer data into vector
235 cur_ptr = data.data() + cur_offset;
236 std::memcpy(cur_ptr, tensor_data, buffer_size);
237 // Move offset
238 cur_offset += buffer_size;
239 // Save buffers size
240 acc_buffer_size += buffer_size;
241 }
242 }
243
244 // Now cur_offset points to last position after adding all buffers
245 // Let's handle with Adam buffers offsets
246 assert(config.training_context.optimizer == ADAM or
247 config.training_context.optimizer == SGD && "Unsupported type");
248 if (config.training_context.optimizer == ADAM and not train_storage.getAdam()->isReset())
249 {
250 // Move pointer to m offset field
251 cur_ptr = data.data() + 4;
252 // Save offset for the m offset field
253 uint32_t m_offset = cur_offset;
254 std::memcpy(cur_ptr, &m_offset, sizeof(m_offset));
255 // Move pointer to the v offset field
256 cur_ptr += 4;
257 // Calculate offset for the v offset field
258 uint32_t v_offset = m_offset + acc_buffer_size + tensors_size * 4;
259 // Save it
260 std::memcpy(cur_ptr, &v_offset, sizeof(v_offset));
261
262 // Let's write offsets and buffers
263 const auto adam_opt = train_storage.getAdam();
264 assert(adam_opt != nullptr);
265
266 // Move m and v to buffers offsets
267 uint32_t m_buffer_offset = m_offset + tensors_size * 4;
268 uint32_t v_buffer_offset = v_offset + tensors_size * 4;
269
270 // Adam buffers
271 std::vector<uint32_t> m_offsets(tensors_size, 0);
272 std::vector<uint32_t> v_offsets(tensors_size, 0);
273
274 // Goes among trainable ops
275 for (uint32_t index = 0; index < last_node_pos; ++index)
276 {
277 uint32_t cur_op_index = num_kernels - index - 1;
278 auto *cur_op = operators->operator[](cur_op_index);
279
280 auto input_tensors = cur_op->inputs();
281
282 for (uint32_t i = 0; i < input_tensors->size(); ++i)
283 {
284 const auto input_tensor_index = input_tensors->operator[](i);
285 // Check is const
286 if (not context.isConstTensor(input_tensor_index))
287 continue;
288
289 uint32_t cur_opcode_index = cur_op->opcode_index();
290
291 assert(cur_opcode_index < op_codes->size());
292
293 const auto opcode = op_codes->operator[](cur_opcode_index);
294
295 // Check it is trainable const
296 if (not isTrainableWeights(opcode))
297 continue;
298
299 // Found trainable weight tensor, lets calculate its size and save offset for current buffer
300 const auto tensor = context.getTensorByIndex(input_tensor_index);
301 OMRuntimeShape shape(tensor);
302
303 auto type_size = sizeof(OMDataType(tensor->type()));
304
305 size_t buffer_size = type_size * shape.flatSize();
306 // Save for current tensor index its offset
307 m_offsets[input_tensor_index] = m_buffer_offset;
308 v_offsets[input_tensor_index] = v_buffer_offset;
309
310 // Obtain m and v data from train storage
311 uint8_t *m_data = adam_opt->getExponentAvgDataByTensorIndex(input_tensor_index);
312 assert(m_data != nullptr);
313 uint8_t *v_data = adam_opt->getExponentAvgSquaresDataByTensorIndex(input_tensor_index);
314 assert(v_data != nullptr);
315
316 // Write m data
317 cur_ptr = data.data() + m_buffer_offset;
318 std::memcpy(cur_ptr, m_data, buffer_size);
319 // Write v data
320 cur_ptr = data.data() + v_buffer_offset;
321 std::memcpy(cur_ptr, v_data, buffer_size);
322
323 // Move m and v buffers offsets
324 m_buffer_offset += buffer_size;
325 v_buffer_offset += buffer_size;
326 }
327 }
328
329 // Set pointer to the m offset field start
330 char *m_offset_start = data.data() + m_offset;
331 // Set pointer to the v offset field start
332 char *v_offset_start = data.data() + v_offset;
333 // Write m and v offsets
334 for (uint32_t i = 0; i < tensors_size; ++i)
335 {
336 // Save m buffer offset
337 uint32_t cur_m_offset = m_offsets[i];
338 std::memcpy(m_offset_start, &cur_m_offset, sizeof(cur_m_offset));
339 m_offset_start += 4;
340
341 // Save v buffer offset
342 uint32_t cur_v_offset = v_offsets[i];
343 std::memcpy(v_offset_start, &cur_v_offset, sizeof(cur_v_offset));
344 v_offset_start += 4;
345 }
346 }
347 else
348 {
349 // Note: offset = 0 - means there are no such buffers
350 // Move pointer to m offset field
351 cur_ptr = data.data() + 4;
352 // Save offset for the m offset field
353 uint32_t m_offset = 0;
354 std::memcpy(cur_ptr, &m_offset, sizeof(m_offset));
355
356 // Move pointer to the v offset field
357 cur_ptr += 4;
358 // Save offset for the v offset field
359 uint32_t v_offset = 0;
360 std::memcpy(cur_ptr, &v_offset, sizeof(v_offset));
361 }
362
363 // Move cur_ptr to the start of the offsets field
364 cur_ptr = data.data() + 20;
365 // Write offsets
366 for (uint32_t i = 0; i < tensors_size; ++i)
367 {
368 uint32_t offset = offsets[i];
369 std::memcpy(cur_ptr, &offset, sizeof(offset));
370 cur_ptr += 4;
371 }
372
373 // Save other parameters offset: 20 initial bytes + tensors_size * 4 bytes for buffer offsets +
374 // buffer size
375 uint32_t other_parameters_offset = 20 + tensors_size * 4 + acc_buffer_size;
376 // Adam case need add two more acc_buffer_size
377 if (config.training_context.optimizer == ADAM and not train_storage.getAdam()->isReset())
378 {
379 other_parameters_offset += acc_buffer_size * 2;
380 other_parameters_offset += tensors_size * 4 * 2;
381 }
382
383 // Write this offset
384 cur_ptr = data.data() + 12;
385 std::memcpy(cur_ptr, &other_parameters_offset, sizeof(other_parameters_offset));
386
387 // Move pointer to other parameters offset
388 cur_ptr = data.data() + other_parameters_offset;
389
390 // Write current step
391 std::memcpy(cur_ptr, &config.training_context.num_step, sizeof(config.training_context.num_step));
392
393 cur_ptr += 4;
394 // Write current epoch
395 std::memcpy(cur_ptr, &config.training_context.num_epoch,
396 sizeof(config.training_context.num_epoch));
397
398 return Ok;
399}

401// To check checkpoint file format please see https://github.com/Samsung/ONE/discussions/13037
402OMStatus OMCheckpointSaver::createCheckpointData(core::OMRuntimeContext &context,
403 OMTrainingStorage &train_storage,
404 std::vector<char> &data, const OMConfig &config)
405{
406 // Clear data
407 data.clear();
408
409 // Obtain file size and resize vector
410 const size_t data_size = calculateFileSize(context, train_storage, config);
411 data.resize(data_size);
412
413 // Point to start of the buffer
414 char *cur_ptr = data.data();
415
416 // Write MAGIC_NUMBER
417 std::memcpy(cur_ptr, &MAGIC_NUMBER, sizeof(MAGIC_NUMBER));
418 cur_ptr += 2;
419
420 // Write SCHEMA_VERSION
421 std::memcpy(cur_ptr, &SCHEMA_VERSION, sizeof(SCHEMA_VERSION));
422 cur_ptr += 1;
423
424 // Miss RESERVED field
425 cur_ptr += 1;
426
427 // Writes buffers and offsets
428 OMStatus status = writeOffsetsAndBuffers(context, train_storage, config, data);
429
430 assert(status == Ok);
431
432 return Ok;
433}
