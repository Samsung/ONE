64loco::TensorShape tensor_shape(const TFConst *node)
65{
66 assert(node != nullptr);
67
68 loco::TensorShape shape;
69
70 uint32_t rank = node->rank();
71 shape.rank(rank);
72 for (uint32_t index = 0; index < rank; ++index)
73 {
74 assert(node->dim(index).known());
75 shape.dim(index) = node->dim(index).value();
76 }
77
78 return shape;
79}

81uint32_t num_elements(const TFConst *tfconst)
82{
83 assert(tfconst != nullptr);
84
85 uint32_t num_elements = 1;
86 for (uint32_t index = 0; index < tfconst->rank(); ++index)
87 {
88 assert(tfconst->dim(index).known());
89 uint32_t dim = tfconst->dim(index).value();
90 num_elements = num_elements * dim;

95bool same_shape(const TFConst *lhs, const TFConst *rhs)
96{
97 assert(lhs != nullptr);
98 assert(rhs != nullptr);
99
100 if (lhs->rank() != rhs->rank())
101 return false;
102
103 for (uint32_t index = 0; index < lhs->rank(); ++index)
104 {
105 assert(lhs->dim(index).known());
106 assert(rhs->dim(index).known());
107 if (lhs->dim(index).value() != rhs->dim(index).value())
108 return false;
109 }
110 return true;
111}
