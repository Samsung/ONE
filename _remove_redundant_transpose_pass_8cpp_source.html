26bool check_perm(const luci::CircleConst *first_perm, const luci::CircleConst *second_perm)
27{
28 assert(first_perm->rank() == 1);
29 assert(second_perm->rank() == 1);
30 assert(second_perm->size<loco::DataType::S32>() == first_perm->size<loco::DataType::S32>());
31 for (int32_t i = 0; i < static_cast<int32_t>(first_perm->size<loco::DataType::S32>()); i++)
32 {
33 if (first_perm->at<loco::DataType::S32>(second_perm->at<loco::DataType::S32>(i)) != i)
34 return false;
35 }
36 return true;
37}

39bool remove_consecutive_transpose_function(luci::CircleTranspose *target_node)
40{
41 auto pred_node = dynamic_cast<luci::CircleTranspose *>(target_node->a());
42 if (pred_node == nullptr)
43 return false;
44
45 auto target_perm = dynamic_cast<luci::CircleConst *>(target_node->perm());
46 if (target_perm == nullptr)
47 return false;
48
49 auto pred_perm = dynamic_cast<luci::CircleConst *>(pred_node->perm());
50 if (pred_perm == nullptr)
51 return false;
52
53 auto main_node = loco::must_cast<luci::CircleNode *>(pred_node->a());
54 if (check_perm(target_perm, pred_perm))
55 {
56 replace(target_node).with(main_node);
57 }
58 else
59 {
60 auto name = target_node->name();
61 assert(name.length() > 0);
62
63 auto g = pred_perm->graph();
64 auto new_const_node = g->nodes()->create<luci::CircleConst>();
65
66 new_const_node->dtype(loco::DataType::S32);
67 new_const_node->rank(1);
68 new_const_node->dim(0) = pred_perm->dim(0);
69 new_const_node->size<loco::DataType::S32>(pred_perm->dim(0).value());
70 new_const_node->shape_status(luci::ShapeStatus::VALID);
71 for (uint32_t i = 0; i < pred_perm->size<loco::DataType::S32>(); i++)
72 {
73 new_const_node->at<loco::DataType::S32>(i) =
74 pred_perm->at<loco::DataType::S32>(target_perm->at<loco::DataType::S32>(i));
75 }
76 new_const_node->name(name + "/Transpose/perm");
77
78 // Create New Transpose Node
79 auto new_transpose_node = g->nodes()->create<luci::CircleTranspose>();
80 new_transpose_node->dtype(target_node->dtype());
81 new_transpose_node->a(main_node);
82 new_transpose_node->perm(new_const_node);
83 new_transpose_node->name(name + "/Transpose");
84 luci::add_origin(new_transpose_node, luci::get_origin(target_node));
85
86 replace(target_node).with(new_transpose_node);
87 }
88 return true;
89}

116bool RemoveRedundantTransposePass::run(loco::Graph *g)
117{
118 bool changed = false;
119 for (auto node : loco::active_nodes(loco::output_nodes(g)))
120 {
121 if (auto transpose = dynamic_cast<luci::CircleTranspose *>(node))
122 {
123 if (remove_consecutive_transpose_function(transpose))
124 changed = true;
125 }
126 }
127 return changed;
128}
