45class KernelBuiltinConfigureRegistry
46{
47public:
48 constexpr KernelBuiltinConfigureRegistry() : _operator_configure()
49 {
50#define REGISTER_KERNEL(builtin_operator, name) \
51 registerKernelConfigure(core::OMBuilderID::BuiltinOperator_##builtin_operator, \
52 configure_kernel_Circle##name);
53
54#include "KernelsToBuild.lst"
55
56#undef REGISTER_KERNEL
57 }
58
59public:
60 OMStatus getKernelConfigureFunc(core::OMBuilderID builderID,
61 KernelConfigureFunc **configure_func) const
62 {
63 const auto builder_id_opcode = size_t(builderID);
64 assert(builder_id_opcode < size_t(core::OMBuilderID::BuiltinOperatorsSize));
65 if (builder_id_opcode >= size_t(core::OMBuilderID::BuiltinOperatorsSize))
66 {
67 *configure_func = nullptr;
68 return UnknownError;
69 }
70 *configure_func = _operator_configure[builder_id_opcode];
71 return Ok;
72 }
73
74private:
75 constexpr void registerKernelConfigure(core::OMBuilderID id, KernelConfigureFunc *func)
76 {
77 assert(size_t(id) < size_t(core::OMBuilderID::BuiltinOperatorsSize));
78 _operator_configure[size_t(id)] = func;
79 }
80
81private:
82 KernelConfigureFunc *_operator_configure[size_t(core::OMBuilderID::BuiltinOperatorsSize)];
83};

85class KernelCustomConfigureRegistry
86{
87public:
88 constexpr KernelCustomConfigureRegistry() : _operator_configure()
89 {
90#define REGISTER_CUSTOM_KERNEL(name, string_name) \
91 registerKernelConfigure(core::OMBuilderID::CUSTOM_##name, configure_kernel_Circle##name);
92
93#include "CustomKernelsToBuild.lst"
94
95#undef REGISTER_CUSTOM_KERNEL
96 }
97
98public:
99 OMStatus getKernelConfigureFunc(core::OMBuilderID builderID,
100 KernelConfigureFunc **configure_func) const
101 {
102 auto builder_id_opcode = size_t(builderID);
103 if (builder_id_opcode >= size_t(core::OMBuilderID::Size))
104 {
105 *configure_func = nullptr;
106 return UnknownError;
107 }
108 const auto builder_id_offset = size_t(core::OMBuilderID::BuiltinOperatorsSize);
109 builder_id_opcode -= builder_id_offset - 1;
110 if (builder_id_opcode < 0)
111 {
112 *configure_func = nullptr;
113 return UnknownError;
114 }
115 *configure_func = _operator_configure[builder_id_opcode];
116 return Ok;
117 }
118
119private:
120 constexpr void registerKernelConfigure(core::OMBuilderID id, KernelConfigureFunc *func)
121 {
122 auto builder_id_opcode = size_t(id);
123 const auto builder_id_offset = size_t(core::OMBuilderID::BuiltinOperatorsSize);
124 builder_id_opcode = builder_id_opcode - builder_id_offset - 1;
125 _operator_configure[builder_id_opcode] = func;
126 }
127
128private:
129 KernelConfigureFunc *_operator_configure[size_t(core::OMBuilderID::Size) -
130 size_t(core::OMBuilderID::BuiltinOperatorsSize) - 1];
131};
