54int32_t unknown_dim_count(luci::CircleNode *node)
55{
56 int32_t count = 0;
57
58 for (uint32_t i = 0; i < node->rank(); ++i)
59 if (!node->dim(i).known())
60 ++count;
61
62 return count;
63}

68int32_t value_from_circle_const(const luci::CircleConst *node, uint32_t idx)
69{
70 // Scalar case: rank 0, only one element in CircleConst
71 if (node->rank() == 0)
72 {
73 if (node->dtype() == loco::DataType::S64)
74 {
75 assert(node->size<loco::DataType::S64>() == 1); // FIX_ME_UNLESS
76 return static_cast<int32_t>(node->at<loco::DataType::S64>(0));
77 }
78 else if (node->dtype() == loco::DataType::S32)
79 {
80 assert(node->size<loco::DataType::S32>() == 1); // FIX_ME_UNLESS
81 return node->at<loco::DataType::S32>(0);
82 }
83 else
84 {
85 throw std::runtime_error("Unsupported dtype");
86 }
87 }
88
89 assert(node->rank() == 1 && node->dim(0).value() > idx);
90 assert(node->dtype() == loco::DataType::S64 || node->dtype() == loco::DataType::S32);
91
92 if (node->dtype() == loco::DataType::S64)
93 return static_cast<int32_t>(node->at<loco::DataType::S64>(idx));
94 return node->at<loco::DataType::S32>(idx);
95}

97bool substitute_expand_dims_to_reshape(luci::CircleNode *node)
98{
99 auto target_node = dynamic_cast<luci::CircleExpandDims *>(node);
100 if (target_node == nullptr)
101 return false;
102 auto input_node = loco::must_cast<luci::CircleNode *>(target_node->input());
103 if (input_node->rank() <= 0)
104 return false;
105 auto axis_node = dynamic_cast<luci::CircleConst *>(target_node->axis());
106 if (axis_node == nullptr)
107 return false;
108
109 if (axis_node->dtype() != loco::DataType::S64 && axis_node->dtype() != loco::DataType::S32)
110 {
111 // Abnormal model with unexpected dtype in axis
112 return false;
113 }
114
115 int32_t axis = value_from_circle_const(axis_node, 0);
116 if (axis < 0)
117 // WHY ADD +1?
118 // As minux index is calculated from the last index,
119 // it must expect the new expanded(+1) rank.
120 axis = axis + static_cast<int32_t>(input_node->rank()) + 1;
121
122 auto name = node->name();
123 assert(name.length() > 0);
124
125 auto graph = target_node->graph();
126 auto reshape_node = graph->nodes()->create<luci::CircleReshape>();
127 reshape_node->tensor(input_node);
128 reshape_node->name(name + "/Reshape");
129 luci::add_origin(reshape_node, luci::get_origin(node));
130
131 auto const_node = graph->nodes()->create<luci::CircleConst>();
132 const_node->dtype(loco::DataType::S32);
133 const_node->size<loco::DataType::S32>(input_node->rank() + 1);
134 const_node->shape_status(luci::ShapeStatus::VALID);
135 const_node->rank(1);
136 const_node->dim(0).set(input_node->rank() + 1);
137 luci::add_origin(const_node, luci::get_origin(node));
138
139 for (int32_t i = 0; i < static_cast<int32_t>(input_node->rank()) + 1; i++)
140 {
141 if (i == axis)
142 {
143 const_node->at<loco::DataType::S32>(i) = 1;
144 }
145 else if (i < axis)
146 {
147 const_node->at<loco::DataType::S32>(i) = input_node->dim(i).value();
148 }
149 else
150 {
151 const_node->at<loco::DataType::S32>(i) = input_node->dim(i - 1).value();
152 }
153 }
154 const_node->name(name + "/Reshape/shape");
155 reshape_node->shape(const_node);
156 replace(target_node).with(reshape_node);
157 return true;
158}

165bool SubstituteExpandDimsToReshapePass::run(loco::Graph *g)
166{
167 bool changed = false;
168 for (auto node : loco::active_nodes(loco::output_nodes(g)))
169 {
170 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
171 if (unknown_dim_count(circle_node) == 0 && substitute_expand_dims_to_reshape(circle_node))
172 {
173 changed = true;
174 }
175 }
176 return changed;
177}
