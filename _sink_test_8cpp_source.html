50/* This tests swapping relu and transpose */
51TEST(OptPass, sinkTrReLU)
52{
53 mir::Graph g;
54 /*
55 * Create graph:
56 * [input]
57 * ||
58 * [Transpose]
59 * ||
60 * [relu]
61 * ||
62 * [tanh]
63 */
64 mir::TensorType input_type{mir::DataType::FLOAT32, Shape{1, 2, 3}};
65 Operation *input = g.create<ops::InputOp>(input_type);
66 Operation *tr1 = g.create<ops::TransposeOp>(input->getOutput(0), vector<size_t>{1, 0, 2});
67 Operation *relu = g.create<ops::ReluOp>(tr1->getOutput(0));
68 Operation *tanh = g.create<ops::TanhOp>(relu->getOutput(0));
69 Operation *out = g.create<ops::OutputOp>(tanh->getOutput(0));
70 (void)out;
71
72 // Check that layout is desired
73 SinkTranspose pass;
74 pass.run(&g);
75
76 // Assert transposes are removed
77 ASSERT_EQ(g.getInputs()[0]->getType(), mir::Operation::Type::input);
78 ASSERT_EQ(getPrev(g.getOutputs()[0])->getType(), mir::Operation::Type::tanh);
79 ASSERT_EQ(getNext(g.getInputs()[0])->getType(), mir::Operation::Type::ReLU);
80 ASSERT_EQ(getPrev(tanh)->getType(), mir::Operation::Type::transpose);
81}

84TEST(OptPass, sinkTrConcat)
85{
86 mir::Graph g;
87 /*
88 * Create graph:
89 * [input] [input2]
90 * || ||
91 * [Transpose 1] [Transpose 2]
92 * \\ //
93 * [Concat]
94 * ||
95 * [TanH]
96 */
97
98 mir::TensorType in1_type{mir::DataType::FLOAT32, Shape{1, 1, 2, 3}};
99 Operation *in1 = g.create<ops::InputOp>(in1_type);
100
101 mir::TensorType in2_type{mir::DataType::FLOAT32, Shape{1, 1, 2, 3}};
102 Operation *in2 = g.create<ops::InputOp>(in2_type);
103 Operation *tr1 = g.create<ops::TransposeOp>(in1->getOutput(0), vector<size_t>{0, 3, 1, 2});
104 Operation *tr2 = g.create<ops::TransposeOp>(in2->getOutput(0), vector<size_t>{0, 3, 1, 2});
105 Operation *conc =
106 g.create<ops::ConcatOp>(vector<Operation::Output *>{tr1->getOutput(0), tr2->getOutput(0)}, 1);
107 Operation *tanh = g.create<ops::TanhOp>(conc->getOutput(0));
108 Operation *out = g.create<ops::OutputOp>(tanh->getOutput(0));
109 (void)out;
110 // Check that layout is as desired
111 SinkTranspose pass;
112 pass.run(&g);
113
114 ASSERT_EQ(getPrev(getPrev(g.getOutputs()[0]))->getType(), Operation::Type::transpose);
115 ASSERT_TRUE(static_cast<ops::TransposeOp *>(getPrev(tanh))->getAxisOrder() ==
116 vector<size_t>({0, 3, 1, 2}));
117 /* Expected Result:
118 * TanH(Transpose(Concat(inp1,inp2)))
119 */
120}

160TEST(OptPass, sinkPoolReLU)
161{
162 mir::Graph g;
163 /*
164 * Create graph:
165 * [input]
166 * ||
167 * [relu]
168 * ||
169 * [MaxPool]
170 * ||
171 * [tanh]
172 */
173 mir::TensorType input_type{mir::DataType::FLOAT32, Shape{1, 4, 4, 3}};
174 Operation *input = g.create<ops::InputOp>(input_type);
175 Operation *relu = g.create<ops::ReluOp>(input->getOutput(0));
176 mir::MaxPool2DOpAttributes attributes;
177 attributes.window = {2, 2};
178 attributes.strides = {2, 2};
179 Operation *mp = g.create<ops::MaxPool2DOp>(relu->getOutput(0), attributes);
180 Operation *tanh = g.create<ops::TanhOp>(mp->getOutput(0));
181 Operation *out = g.create<ops::OutputOp>(tanh->getOutput(0));
182 (void)out;
183
184 SinkRelu pass;
185 pass.run(&g);
186 stringstream ss;
187 DumpVisitor d{ss};
188 g.accept(&d);
189
190 // tanh(relu(pool(input)))
191 ASSERT_EQ(getNext(g.getInputs()[0])->getType(), mir::Operation::Type::maxPool2D);
192 ASSERT_EQ(getPrev(g.getOutputs()[0])->getType(), mir::Operation::Type::tanh);
193 ASSERT_EQ("i_0.p_5.r_6.th_3.", ss.str());
194}
