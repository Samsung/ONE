42void prepare(ExecEnv *env, const ir::Operation &node)
43{
44 const auto input_index = node.getInputs().at(0);
45 const auto output_index = node.getOutputs().at(0);
46
47 const auto input_tensor = env->tensorAt(input_index);
48
49 const auto output_info = env->graph().operands().at(output_index).info();
50 if (output_info.total_size() == 0)
51 {
52 // Output's shape and type is same with input
53 auto input_info = input_tensor->tensorInfo();
54 // We can handle already allocated (ex. model output)
55 env->allocateIfNeeded(output_index, input_info);
56 }
57 else
58 {
59 env->allocateIfNeeded(output_index, output_info);
60 }
61
62 const auto output_tensor = env->tensorAt(output_index);
63 // Check shape and type lhs is same with output
64 // TODO Util function to compare TensorInfo
65 if (input_tensor->data_type() != output_tensor->data_type())
66 {
67 throw std::runtime_error{"Interp(ElementwiseActivation): Invalid output type"};
68 }
69}

71template <ActivationType act_type>
72void evalFloat(const float *input_ptr, float *output_ptr, uint64_t num_elements, float alpha,
73 float beta)
74{
75 std::function<float(const float &)> fn = [](const float &) { return std::nanf(""); };
76 switch (act_type)
77 {
78 case ActivationType::ReLU:
79 fn = [alpha, beta](const float &in) { return std::min(std::max(beta, in), alpha); };
80 break;
81 case ActivationType::Tanh:
82 fn = [](const float &in) { return std::tanh(in); };
83 break;
84 default:
85 throw std::runtime_error{"Interp(ElementwiseActivation): NYI - Unsupported activation"};
86 break;
87 }
88
89 const float *input_end = input_ptr + num_elements;
90 for (; input_ptr < input_end; input_ptr++, output_ptr++)
91 {
92 *output_ptr = fn(*input_ptr);
93 }
94}

96template <ActivationType act_type> void invoke(const ExecEnv *env, const ir::Operation &node)
97{
98 const auto input_index = node.getInputs().at(0);
99 const auto output_index = node.getOutputs().at(0);
100
101 // Check lhs shape is same with rhs (with broadcast)
102 const auto input_tensor = env->tensorAt(input_index);
103 const auto output_tensor = env->tensorAt(output_index);
104
105 const auto data_type = input_tensor->data_type();
106 if (data_type == ir::DataType::FLOAT32)
107 {
108 uint64_t elements = input_tensor->num_elements();
109 const float *input_start = reinterpret_cast<const float *>(input_tensor->bufferRO());
110 float *out = reinterpret_cast<float *>(output_tensor->buffer());
111 if (act_type == ActivationType::Logistic)
112 {
113 const auto cker_input_shape = convertShape(input_tensor->tensorInfo().shape());
114 const auto cker_output_shape = convertShape(output_tensor->tensorInfo().shape());
115 nnfw::cker::Logistic(cker_input_shape, input_start, cker_output_shape, out);
116 }
117 else
118 {
119 const auto &act_node =
120 nnfw::misc::polymorphic_downcast<const ir::operation::ElementwiseActivation &>(node);
121 evalFloat<act_type>(input_start, out, elements, act_node.param().alpha,
122 act_node.param().beta);
123 }
124 }
125 else
126 {
127 throw std::runtime_error{"Interp(" + node.name() + "): NYI - Support float only"};
128 }
129}

131void invokeElementwiseActivation(const ExecEnv *env, const ir::Operation &node)
132{
133 const auto &act_node =
134 nnfw::misc::polymorphic_downcast<const ir::operation::ElementwiseActivation &>(node);
135 switch (act_node.param().op_type)
136 {
137 case ir::operation::ElementwiseActivation::Type::LOGISTIC:
138 invoke<ActivationType::Logistic>(env, node);
139 break;
140 case ir::operation::ElementwiseActivation::Type::RELU:
141 invoke<ActivationType::ReLU>(env, node);
142 break;
143 case ir::operation::ElementwiseActivation::Type::TANH:
144 invoke<ActivationType::Tanh>(env, node);
145 break;
146 default:
147 throw std::runtime_error("Interp(" + node.name() + "): NYI - Unsupported activation");
148 }
149}
