44// Throw exception if virtual Op (Scale/Shift) exists
45void check_no_scale_shift(loco::Graph *g)
46{
47 for (auto node : loco::active_nodes(loco::output_nodes(g)))
48 {
49 const auto custom = dynamic_cast<const luci::CircleCustom *>(node);
50 if (custom == nullptr)
51 continue;
52
53 if (custom->numInputs() != 2)
54 continue;
55
56 const auto code = custom->custom_code();
57 if (code == "PreScale" or code == "PreShift" or code == "PostScale" or code == "PostShift")
58 throw std::runtime_error("Virtual node(" + code + ") remains. " + custom->name());
59 }
60}

67void FMEqualizer::equalize(loco::Graph *g, const std::vector<EqualizePattern> &p)
68{
69 THROW_UNLESS(g != nullptr, "Invalid argument g");
70
71 // Check the patterns are valid on the graph
72 check_patterns_valid(g, p);
73
74 // Insert Scale/Shift based on patterns
75 InsertScaleShift issp(p);
76 issp.run(g);
77
78 logo::Phase phase;
79
80 // Default passes
81 phase.emplace_back(std::make_unique<logo::RemoveDeadNodeWithQueryPass>());
82 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
83 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
84
85 // Forward PreScale/PreShift
86 phase.emplace_back(std::make_unique<fme_apply::ForwardPreScalePass>());
87 phase.emplace_back(std::make_unique<fme_apply::ForwardPreShiftPass>());
88
89 // Fuse Pre/Post Scale/Shift
90 phase.emplace_back(std::make_unique<fme_apply::FusePreScalePass>());
91 phase.emplace_back(std::make_unique<fme_apply::FusePostScalePass>());
92 phase.emplace_back(std::make_unique<fme_apply::FusePreShiftPass>());
93 phase.emplace_back(std::make_unique<fme_apply::FusePostShiftPass>());
94
95 ProgressReporter prog(g, logo::PhaseStrategy::Restart);
96 logo::PhaseRunner<logo::PhaseStrategy::Restart> phase_runner{g};
97 phase_runner.attach(&prog);
98 phase_runner.run(phase);
99
100 // Check if all Scale/Shift nodes are removed
101 check_no_scale_shift(g);
102}
103
104} // namespace fme_apply
