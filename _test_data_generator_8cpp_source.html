63template <typename T> void generate_random_data(std::mt19937 &gen, void *data, uint32_t size)
64{
65 std::normal_distribution<float> distrib(0, 2); // mean(0), stddev(2)
66 for (uint32_t i = 0; i < size; i++)
67 {
68 static_cast<T *>(data)[i] = static_cast<T>(distrib(gen));
69 }
70}

72template <> void generate_random_data<bool>(std::mt19937 &gen, void *data, uint32_t size)
73{
74 std::normal_distribution<float> distrib(0, 2); // mean(0), stddev(2)
75 for (uint32_t i = 0; i < size; i++)
76 {
77 static_cast<bool *>(data)[i] = distrib(gen) >= 0 ? true : false;
78 }
79}

81template <typename T>
82void generate_random_range(void *data, uint32_t size, int32_t range_min, int32_t range_max)
83{
84 assert(range_min <= range_max);
85
86 for (uint32_t i = 0; i < size; i++)
87 {
88 // +1 will make value of [range_min, range_max]
89 int32_t range = range_max - range_min + 1;
90 int32_t value = (rand() % range) + range_min;
91 static_cast<T *>(data)[i] = static_cast<T>(value);
92 }
93}

95void fill_random_data(void *data, uint32_t size, loco::DataType dtype, uint32_t seed)
96{
97 std::mt19937 gen(seed); // standard mersenne_twister_engine seeded with rd()
98
99 switch (dtype)
100 {
101 case loco::DataType::U8:
102 generate_random_data<uint8_t>(gen, data, size);
103 break;
104 case loco::DataType::S16:
105 generate_random_data<int16_t>(gen, data, size);
106 break;
107 case loco::DataType::S32:
108 generate_random_data<int32_t>(gen, data, size);
109 break;
110 case loco::DataType::S64:
111 generate_random_data<int64_t>(gen, data, size);
112 break;
113 case loco::DataType::FLOAT32:
114 generate_random_data<float>(gen, data, size);
115 break;
116 case loco::DataType::BOOL:
117 generate_random_data<bool>(gen, data, size);
118 break;
119 default:
120 throw std::runtime_error("NYI data type.");
121 }
122}

142int entry(int argc, char **argv)
143{
144 arser::Arser arser;
145 arser.add_argument("circle").help("Circle file you want to test");
146 arser.add_argument("--input_data").required(true).help("Path to generate input data h5 file");
147 arser.add_argument("--expected_data")
148 .required(true)
149 .help("Path to generate expected data h5 file");
150 arser.add_argument("--fixed_seed")
151 .nargs(0)
152 .help("Put a fixed seed into the random number generator");
153 arser.add_argument("--input_range")
154 .nargs(3)
155 .type(arser::DataType::STR_VEC)
156 .help("Set random number range [min max] for the input as 'name min max'");
157
158 try
159 {
160 arser.parse(argc, argv);
161 }
162 catch (const std::runtime_error &err)
163 {
164 std::cout << err.what() << std::endl;
165 std::cout << arser;
166 return 255;
167 }
168
169 std::string circle_file = arser.get<std::string>("circle");
170
171 // load circle file
172 foder::FileLoader file_loader{circle_file};
173 std::vector<char> model_data = file_loader.load();
174 const circle::Model *circle_model = circle::GetModel(model_data.data());
175 if (circle_model == nullptr)
176 {
177 std::cerr << "ERROR: Failed to load circle '" << circle_file << "'" << std::endl;
178 return EXIT_FAILURE;
179 }
180
181 // load luci module
182 std::unique_ptr<luci::Module> module = luci::Importer().importModule(circle_model);
183 luci_interpreter::Interpreter interpreter(module.get());
184
200 // create random data and dump into hdf5 file
201 H5::H5File input_file{arser.get<std::string>("--input_data"), H5F_ACC_TRUNC};
202 std::unique_ptr<H5::Group> input_name_group =
203 std::make_unique<H5::Group>(input_file.createGroup("name"));
204 std::unique_ptr<H5::Group> input_value_group =
205 std::make_unique<H5::Group>(input_file.createGroup("value"));
206
207 H5::H5File output_file{arser.get<std::string>("--expected_data"), H5F_ACC_TRUNC};
208 std::unique_ptr<H5::Group> output_name_group =
209 std::make_unique<H5::Group>(output_file.createGroup("name"));
210 std::unique_ptr<H5::Group> output_value_group =
211 std::make_unique<H5::Group>(output_file.createGroup("value"));
212
213 std::string range_name;
214 int32_t range_min = 0;
215 int32_t range_max = 0;
216 bool range_check = false;
217 bool range_input_found = false;
218 if (arser["--input_range"])
219 {
220 // NOTE limitation: we can only set one input range
221 // TODO expand this for multiple inputs
222 std::vector<std::string> values = arser.get<std::vector<std::string>>("--input_range");
223 assert(values.size() == 3);
224 range_name = values.at(0);
225 // TODO add check for valid numbers
226 range_min = std::atoi(values.at(1).c_str());
227 range_max = std::atoi(values.at(2).c_str());
228 range_check = true;
229 }
230
231 std::random_device rd; // used to obtain a seed for the random number engine
232 uint32_t input_index = 0;
233 // TODO remove indentation
234 {
235 // NOTE we only need to prepare data for main graph (subgraph 0) as
236 // other subgraphs are invoked by the main graph
237 const auto input_nodes = loco::input_nodes(module->graph(0));
238 for (const auto &node : input_nodes)
239 {
240 const auto *input_node = dynamic_cast<const luci::CircleInput *>(node);
241 std::string name = input_node->name();
242 assert(not name.empty());
243 if (name.find(":") == std::string::npos)
244 name += ":0";
245
246 // create attribute
247 H5::DataSpace name_dataspace(H5S_SCALAR);
248 H5::StrType name_datatype(H5::PredType::C_S1, name.size());
249
250 auto name_attr = input_name_group->createAttribute(std::to_string(input_index), name_datatype,
251 name_dataspace);
252
253 name_attr.write(name_datatype, name);
254
255 // create value
256 std::vector<hsize_t> dims(input_node->rank());
257 for (uint32_t d = 0; d < input_node->rank(); d++)
258 {
259 dims.at(d) = input_node->dim(d).value();
260 assert(dims.at(d) >= 0);
261 }
262 auto dataspace = std::make_unique<H5::DataSpace>(dims.size(), dims.data());
263 auto dtype = hdf5_dtype_cast(input_node->dtype());
264 auto dataset = std::make_unique<H5::DataSet>(
265 input_file.createDataSet("value/" + std::to_string(input_index), dtype, *dataspace));
266
267 auto data_size = ::element_num(dims);
268 auto dtype_size = loco::size(input_node->dtype());
269 auto byte_size = dtype_size * data_size;
270 std::vector<int8_t> data(byte_size);
271
272 // generate random data
273 if (range_name == input_node->name())
274 {
275 fill_random_range(data.data(), data_size, input_node->dtype(), range_min, range_max);
276 range_input_found = true;
277 }
278 else if (arser["--fixed_seed"])
279 fill_random_data(data.data(), data_size, input_node->dtype(), 0);
280 else
281 fill_random_data(data.data(), data_size, input_node->dtype(), rd());
282
283 dataset->write(data.data(), dtype);
284
285 interpreter.writeInputTensor(input_node, data.data(), byte_size);
286
287 input_index++;
288 }
289 }
290
291 if (range_check && not range_input_found)
292 {
293 std::cerr << "ERROR: input_range for input [" << range_name << "] not found." << std::endl;
294 return EXIT_FAILURE;
295 }
296
297 interpreter.interpret();
298
299 // dump output data into hdf5 file
300 uint32_t output_index = 0;
301 // TODO remove indentation
302 {
303 const auto output_nodes = loco::output_nodes(module->graph(0));
304 for (const auto &node : output_nodes)
305 {
306 const auto *output_node = dynamic_cast<const luci::CircleOutput *>(node);
307 std::string name = output_node->name();
308 if (name.find(":") == std::string::npos)
309 name += ":0";
310
311 // create attribute
312 H5::DataSpace name_dataspace(H5S_SCALAR);
313 H5::StrType name_datatype(H5::PredType::C_S1, name.size());
314
315 auto name_attr = output_name_group->createAttribute(std::to_string(output_index),
316 name_datatype, name_dataspace);
317
318 name_attr.write(name_datatype, name);
319
320 // create value
321 std::vector<hsize_t> dims(output_node->rank());
322 for (uint32_t d = 0; d < output_node->rank(); d++)
323 {
324 dims.at(d) = output_node->dim(d).value();
325 assert(dims.at(d) >= 0);
326 }
327 auto dataspace = std::make_unique<H5::DataSpace>(dims.size(), dims.data());
328 auto dtype = hdf5_dtype_cast(output_node->dtype());
329 auto dataset = std::make_unique<H5::DataSet>(
330 output_file.createDataSet("value/" + std::to_string(output_index), dtype, *dataspace));
331
332 uint32_t tensor_bytesize = loco::size(output_node->dtype());
333 tensor_bytesize *= ::element_num(dims);
334 std::vector<int8_t> output_data(tensor_bytesize);
335 interpreter.readOutputTensor(output_node, output_data.data(), output_data.size());
336
337 dataset->write(output_data.data(), dtype);
338
339 output_index++;
340 }
341 }
342
343 return EXIT_SUCCESS;
