43void Div::execute() const
44{
45 switch (input1()->element_type())
46 {
47 case DataType::FLOAT32:
48 evalFloat();
49 break;
50 case DataType::S64:
51 evalInteger<int64_t>();
52 break;
53 case DataType::S32:
54 evalInteger<int32_t>();
55 break;
56 case DataType::U8:
57 evalQuantized();
58 break;
59 default:
60 assert(false && "Unsupported type.");
61 }
62}

108void Div::evalQuantized() const
109{
110 const auto input1_scale = static_cast<double>(input1()->scale());
111 const auto input2_scale = static_cast<double>(input2()->scale());
112 const auto output_scale = static_cast<double>(output()->scale());
113
114 const double real_output_multiplier = input1_scale / (input2_scale * output_scale);
115
116 int32_t output_multiplier{};
117 int output_shift{};
118
119 quantizeMultiplier(real_output_multiplier, &output_multiplier, &output_shift);
120
121 int32_t activation_min{};
122 int32_t activation_max{};
123 calculateActivationRangeQuantized(_params.activation, output(), &activation_min, &activation_max);
124
125 tflite::ArithmeticParams params{};
126
127 params.input1_offset = -input1()->zero_point(); // Note the '-'.
128 params.input2_offset = -input2()->zero_point(); // Note the '-'.
129 params.output_offset = output()->zero_point();
130 params.output_multiplier = output_multiplier;
131 params.output_shift = output_shift;
132 params.quantized_activation_min = activation_min;
133 params.quantized_activation_max = activation_max;
134
135 const bool need_broadcast = tflite::reference_ops::ProcessBroadcastShapes(
136 getTensorShape(input1()), getTensorShape(input2()), &params);
137
138 if (need_broadcast)
139 {
140 tflite::reference_ops::BroadcastDivSlow(
141 params, getTensorShape(input1()), getTensorData<uint8_t>(input1()), getTensorShape(input2()),
142 getTensorData<uint8_t>(input2()), getTensorShape(output()), getTensorData<uint8_t>(output()));
143 }
144 else
145 {
146 tflite::reference_ops::Div(params, getTensorShape(input1()), getTensorData<uint8_t>(input1()),
147 getTensorShape(input2()), getTensorData<uint8_t>(input2()),
148 getTensorShape(output()), getTensorData<uint8_t>(output()));
149 }
