29template <typename T>
30void setValue(nnfw::misc::RandomGenerator &randgen, const TfLiteTensor *tensor)
31{
32 auto tensor_view = nnfw::tflite::TensorView<T>::make(tensor);
33
34 nnfw::misc::tensor::iterate(tensor_view.shape())
35 << [&](const nnfw::misc::tensor::Index &ind) { tensor_view.at(ind) = randgen.generate<T>(); };
36}

40void RandomInputInitializer::run(TfLiteInterpreter &interp)
41{
42 const auto input_count = TfLiteInterpreterGetInputTensorCount(&interp);
43 for (int32_t idx = 0; idx < input_count; idx++)
44 {
45 auto tensor = TfLiteInterpreterGetInputTensor(&interp, idx);
46 auto const tensor_type = TfLiteTensorType(tensor);
47 switch (tensor_type)
48 {
49 case kTfLiteFloat32:
50 setValue<float>(_randgen, tensor);
51 break;
52 case kTfLiteInt32:
53 setValue<int32_t>(_randgen, tensor);
54 break;
55 case kTfLiteUInt8:
56 setValue<uint8_t>(_randgen, tensor);
57 break;
58 case kTfLiteBool:
59 setValue<bool>(_randgen, tensor);
60 break;
61 case kTfLiteInt8:
62 setValue<int8_t>(_randgen, tensor);
63 break;
64 default:
65 throw std::runtime_error{"Not supported input type"};
66 }
67 }
68}
