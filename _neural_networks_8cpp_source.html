29int ANeuralNetworksMemory_createFromFd(size_t size, int prot, int fd, size_t offset,
30 ANeuralNetworksMemory **memory)
31{
32 *memory = nullptr;
33 auto m = std::make_unique<MappedMemory>();
34 if (m == nullptr)
35 {
36 return ANEURALNETWORKS_OUT_OF_MEMORY;
37 }
38 int n = m->set(size, prot, fd, offset);
39 if (n != ANEURALNETWORKS_NO_ERROR)
40 {
41 return n;
42 }
43 *memory = reinterpret_cast<ANeuralNetworksMemory *>(m.release());
44 return ANEURALNETWORKS_NO_ERROR;
45}

290int ANeuralNetworksExecution_startCompute(ANeuralNetworksExecution *execution,
291 ANeuralNetworksEvent **event)
292{
293 if (!execution || !event)
294 {
295 LOG(ERROR) << "ANeuralNetworksExecution_startCompute passed a nullptr";
296 return ANEURALNETWORKS_UNEXPECTED_NULL;
297 }
298 // TODO validate the rest
299
300 ExecutionBuilder *r = reinterpret_cast<ExecutionBuilder *>(execution);
301
302 // Dynamically allocate an sp to wrap an ExecutionCallback, seen in the NN
303 // API as an abstract event object. The sp<ExecutionCallback> object is
304 // returned when the execution has been successfully launched, otherwise a
305 // nullptr is returned. The sp is used for ref-counting purposes. Without
306 // it, the HIDL service could attempt to communicate with a dead callback
307 // object.
308 *event = nullptr;
309
310 int n = r->startCompute();
311 if (n != ANEURALNETWORKS_NO_ERROR)
312 {
313 return n;
314 }
315 *event = reinterpret_cast<ANeuralNetworksEvent *>(new int);
316 return ANEURALNETWORKS_NO_ERROR;
317}
