34template <typename T> void fill_tensor_with_values(Tensor *state_tensor, T val)
35{
36 switch (state_tensor->element_type())
37 {
38 case DataType::FLOAT32:
39 {
40 auto state_data = state_tensor->data<float>();
41 std::fill_n(state_data, state_tensor->shape().num_elements(), val);
42 break;
43 }
44 case DataType::S32:
45 {
46 auto state_data = state_tensor->data<int32_t>();
47 std::fill_n(state_data, state_tensor->shape().num_elements(), val);
48 break;
49 }
50 case DataType::S16:
51 {
52 auto state_data = state_tensor->data<int16_t>();
53 std::fill_n(state_data, state_tensor->shape().num_elements(), val);
54 break;
55 }
56 case DataType::S8:
57 {
58 auto state_data = state_tensor->data<int8_t>();
59 std::fill_n(state_data, state_tensor->shape().num_elements(), val);
60 break;
61 }
62 case DataType::U8:
63 {
64 auto state_data = state_tensor->data<uint8_t>();
65 std::fill_n(state_data, state_tensor->shape().num_elements(), val);
66 break;
67 }
68 default:
69 assert(false && "Unsupported type.");
70 }
71}

76void UpdateLstmCellFloat(int n_batch, int n_cell, float *cell_state, const float *input_gate,
77 float *forget_gate, const float *cell_gate, bool use_cifg, float clip)
78{
79// NOTE tflite source is as is but will fail build with gcc-8 and above
80// TODO remove #pragma
81#pragma GCC diagnostic ignored "-Wrestrict"
82 tensor_utils::VectorVectorCwiseProduct(forget_gate, cell_state, n_batch * n_cell, cell_state);
83
84 if (use_cifg)
85 {
86 // With CIFG, input_gate = 1-forget_gate. Use the forget_gate array as
87 // scratch, as input_gate array is not allocated in this case. (Be careful
88 // not to write to the scratch before reading the forget gate data.)
89 float *scratch = forget_gate;
90 tensor_utils::Sub1Vector(forget_gate, n_batch * n_cell, scratch);
91 tensor_utils::VectorVectorCwiseProductAccumulate(cell_gate, scratch, n_batch * n_cell,
92 cell_state);
93 }
94 else
95 {
96 tensor_utils::VectorVectorCwiseProductAccumulate(cell_gate, input_gate, n_batch * n_cell,
97 cell_state);
98 }
99 if (clip > 0.0f)
100 {
101 tensor_utils::CwiseClipping(cell_state, n_batch * n_cell, clip);
102 }

105void CalculateLstmOutputFloat(int n_batch, int n_cell, int n_output, const float *cell_state,
106 const float *output_gate, TfLiteFusedActivation activation,
107 const float *projection_weights, const float *projection_bias,
108 const float proj_clip, float *output_state, float *scratch)
109{
110 luci_interpreter_pal::ApplyActivationToVector(cell_state, n_batch * n_cell, activation, scratch);
111 tensor_utils::VectorVectorCwiseProduct(output_gate, scratch, n_batch * n_cell, scratch);
112
113 const bool use_projection = (projection_weights != nullptr);
114 const bool use_projection_bias = (projection_bias != nullptr);
115
116 if (use_projection)
117 {
118 if (use_projection_bias)
119 {
120 tensor_utils::VectorBatchVectorAssign(projection_bias, n_output, n_batch, output_state);
121 }
122 else
123 {
124 std::fill_n(output_state, n_batch * n_output, 0.0f);
125 }
126 tensor_utils::MatrixBatchVectorMultiplyAccumulate(projection_weights, n_output, n_cell, scratch,
127 n_batch, output_state);
128 if (proj_clip > 0.0f)
129 {
130 tensor_utils::CwiseClipping(output_state, n_batch * n_output, proj_clip);
131 }
132 }
133 else
134 {
135 std::copy_n(scratch, n_batch * n_output, output_state);
136 }
137}

139inline void CalculateLstmGateFloat(const float *input, const float *input_to_gate_weights,
140 const float *aux_input, const float *aux_input_to_gate_weights,
141 const float *output_state,
142 const float *recurrent_to_gate_weights, const float *cell_state,
143 const float *cell_to_gate_weights,
144 const float *layer_norm_coefficients, const float *gate_bias,
145 const int n_batch, const int n_input, const int n_aux_input,
146 const int n_output, const int n_cell,
147 const TfLiteFusedActivation activation, float *gate,
148 const bool is_input_all_zeros, const bool is_aux_input_all_zeros)
149{
150 const bool use_peephole = (cell_to_gate_weights != nullptr);
151 const bool use_layer_norm = (layer_norm_coefficients != nullptr);
152
153 // Initialize scratch buffers with bias for regular lstm or initialize with
154 // zero for layer norm lstm.
155 if (use_layer_norm)
156 {
157 std::fill_n(gate, n_cell * n_batch, 0.0f);
158 }
159 else
160 {
161 tensor_utils::VectorBatchVectorAssign(gate_bias, n_cell, n_batch, gate);
162 }
163 // For each batch and cell: compute input_weight * input.
164 // Skip if input is all zeros.
165 if (!is_input_all_zeros)
166 {
167 tensor_utils::MatrixBatchVectorMultiplyAccumulate(input_to_gate_weights, n_cell, n_input, input,
168 n_batch, gate);
169 }
170 // For each batch and cell: compute aux_input_weight * aux_input.
171 // Skip if auxiliary input is not available or all zeros.
172 if (!is_aux_input_all_zeros)
173 {
174 tensor_utils::MatrixBatchVectorMultiplyAccumulate(aux_input_to_gate_weights, n_cell,
175 n_aux_input, aux_input, n_batch, gate);
176 }
177 // For each batch and cell: compute recurrent_weight * output_state.
178 tensor_utils::MatrixBatchVectorMultiplyAccumulate(recurrent_to_gate_weights, n_cell, n_output,
179 output_state, n_batch, gate);
180 // For each batch and cell: compute cell_weight .* cell_state (peephole LSTM)
181 if (use_peephole)
182 {
183 tensor_utils::VectorBatchVectorCwiseProductAccumulate(cell_to_gate_weights, n_cell, cell_state,
184 n_batch, gate);
185 }
186 // Do layer normalization (if layer norm LSTM)
187 if (use_layer_norm)
188 {
189 tensor_utils::MeanStddevNormalization(gate, gate, n_cell, n_batch);
190 tensor_utils::VectorBatchVectorCwiseProduct(layer_norm_coefficients, n_cell, gate, n_batch,
191 gate);
192 tensor_utils::VectorBatchVectorAdd(gate_bias, n_cell, n_batch, gate);
193 }
194 // Apply activation
195 luci_interpreter_pal::ApplyActivationToVector(gate, n_batch * n_cell, activation, gate);
196}

198inline void LstmStepFloat(
199 const float *input_ptr, const float *input_to_input_weights_ptr,
200 const float *input_to_forget_weights_ptr, const float *input_to_cell_weights_ptr,
201 const float *input_to_output_weights_ptr, const float *aux_input_ptr,
202 const float *aux_input_to_input_weights_ptr, const float *aux_input_to_forget_weights_ptr,
203 const float *aux_input_to_cell_weights_ptr, const float *aux_input_to_output_weights_ptr,
204 const float *recurrent_to_input_weights_ptr, const float *recurrent_to_forget_weights_ptr,
205 const float *recurrent_to_cell_weights_ptr, const float *recurrent_to_output_weights_ptr,
206 const float *cell_to_input_weights_ptr, const float *cell_to_forget_weights_ptr,
207 const float *cell_to_output_weights_ptr, const float *input_layer_norm_coefficients_ptr,
208 const float *forget_layer_norm_coefficients_ptr, const float *cell_layer_norm_coefficients_ptr,
209 const float *output_layer_norm_coefficients_ptr, const float *input_gate_bias_ptr,
210 const float *forget_gate_bias_ptr, const float *cell_gate_bias_ptr,
211 const float *output_gate_bias_ptr, const float *projection_weights_ptr,
212 const float *projection_bias_ptr, const TfLiteLSTMParams *params, int n_batch, int n_cell,
213 int n_input, int n_aux_input, int n_output, int output_batch_leading_dim, float *output_state_ptr,
214 float *cell_state_ptr, float *scratch0, float *scratch1, float *scratch2, float *scratch3,
215 float *output_ptr)
216{
217 // Since we have already checked that weights are all there or none, we can
218 // check the existence of only one to the get the condition.
219 const bool use_cifg = (input_to_input_weights_ptr == nullptr);
220
221 // Make named scratch buffers.
222 float *input_gate_scratch = scratch0;
223 float *forget_gate_scratch = scratch1;
224 float *cell_gate_scratch = scratch2;
225 float *output_gate_scratch = scratch3;
226
227 // Check if inputs are all zeros so we can skip some computations.
228 const bool is_input_all_zeros = tensor_utils::IsZeroVector(input_ptr, n_batch * n_input);
229 const bool is_aux_input_all_zeros =
230 (aux_input_ptr == nullptr || tensor_utils::IsZeroVector(aux_input_ptr, n_batch * n_aux_input));
231 if (!use_cifg)
232 {
233 // Calculate the input gate. (If not CIFG.)
234 CalculateLstmGateFloat(input_ptr, input_to_input_weights_ptr, aux_input_ptr,
235 aux_input_to_input_weights_ptr, output_state_ptr,
236 recurrent_to_input_weights_ptr, cell_state_ptr,
237 cell_to_input_weights_ptr, input_layer_norm_coefficients_ptr,
238 input_gate_bias_ptr, n_batch, n_input, n_aux_input, n_output, n_cell,
239 /*activation=*/kTfLiteActSigmoid, input_gate_scratch, is_input_all_zeros,
240 is_aux_input_all_zeros);
241 }
242 // Calculate the forget gate.
243 CalculateLstmGateFloat(input_ptr, input_to_forget_weights_ptr, aux_input_ptr,
244 aux_input_to_forget_weights_ptr, output_state_ptr,
245 recurrent_to_forget_weights_ptr, cell_state_ptr,
246 cell_to_forget_weights_ptr, forget_layer_norm_coefficients_ptr,
247 forget_gate_bias_ptr, n_batch, n_input, n_aux_input, n_output, n_cell,
248 /*activation=*/kTfLiteActSigmoid, forget_gate_scratch, is_input_all_zeros,
249 is_aux_input_all_zeros);
250 // Calculate the cell update gate.
251 CalculateLstmGateFloat(
252 input_ptr, input_to_cell_weights_ptr, aux_input_ptr, aux_input_to_cell_weights_ptr,
253 output_state_ptr, recurrent_to_cell_weights_ptr, /*cell_state=*/nullptr,
254 /*cell_to_gate_weights=*/nullptr, cell_layer_norm_coefficients_ptr, cell_gate_bias_ptr, n_batch,
255 n_input, n_aux_input, n_output, n_cell, params->activation, cell_gate_scratch,
256 is_input_all_zeros, is_aux_input_all_zeros);
257 // Update the cell state.
258 UpdateLstmCellFloat(n_batch, n_cell, cell_state_ptr, input_gate_scratch, forget_gate_scratch,
259 cell_gate_scratch, use_cifg, params->cell_clip);
260 // Calculate output gate.
261 CalculateLstmGateFloat(input_ptr, input_to_output_weights_ptr, aux_input_ptr,
262 aux_input_to_output_weights_ptr, output_state_ptr,
263 recurrent_to_output_weights_ptr, cell_state_ptr,
264 cell_to_output_weights_ptr, output_layer_norm_coefficients_ptr,
265 output_gate_bias_ptr, n_batch, n_input, n_aux_input, n_output, n_cell,
266 /*activation=*/kTfLiteActSigmoid, output_gate_scratch, is_input_all_zeros,
267 is_aux_input_all_zeros);
268 // Update the output state.
269 CalculateLstmOutputFloat(n_batch, n_cell, n_output, cell_state_ptr, output_gate_scratch,
270 params->activation, projection_weights_ptr, projection_bias_ptr,
271 params->proj_clip, output_state_ptr, scratch2);
272 // Copy output state to the output. Note that the output's rows may not be
273 // contiguous (output_batch_leading_dim != n_output).
274 for (int b = 0; b < n_batch; b++)
275 {
276 std::copy_n(output_state_ptr + b * n_output, n_output,
277 output_ptr + b * output_batch_leading_dim);
278 }
279}

283void EvalFloat(const Tensor *input,
284
285 const Tensor *input_to_input_weights, const Tensor *input_to_forget_weights,
286 const Tensor *input_to_cell_weights, const Tensor *input_to_output_weights,
287
288 const Tensor *recurrent_to_input_weights, const Tensor *recurrent_to_forget_weights,
289 const Tensor *recurrent_to_cell_weights, const Tensor *recurrent_to_output_weights,
290
291 const Tensor *cell_to_input_weights, const Tensor *cell_to_forget_weights,
292 const Tensor *cell_to_output_weights,
293
294 const Tensor *input_layer_norm_coefficients,
295 const Tensor *forget_layer_norm_coefficients,
296 const Tensor *cell_layer_norm_coefficients,
297 const Tensor *output_layer_norm_coefficients,
298
299 const Tensor *aux_input, const Tensor *aux_input_to_input_weights,
300 const Tensor *aux_input_to_forget_weights, const Tensor *aux_input_to_cell_weights,
301 const Tensor *aux_input_to_output_weights,
302
303 const Tensor *input_gate_bias, const Tensor *forget_gate_bias,
304 const Tensor *cell_gate_bias, const Tensor *output_gate_bias,
305
306 const Tensor *projection_weights, const Tensor *projection_bias,
307 const TfLiteLSTMParams *params,
308
309 bool forward_sequence, bool time_major, int output_offset,
310
311 Tensor *scratch_buffer, Tensor *output_state, Tensor *cell_state, Tensor *output)
312{
313 const Shape &input_shape = input->shape();
314 assert(input_shape.num_dims() >= 2 && input_shape.num_dims() <= 3);
315 int max_time, n_batch;
316 if (input_shape.num_dims() == 3)
317 {
318 max_time = (time_major) ? input_shape.dim(0) : input_shape.dim(1);
319 n_batch = (time_major) ? input_shape.dim(1) : input_shape.dim(0);
320 }
321 else
322 {
323 max_time = 1;
324 n_batch = input_shape.dim(0);
325 }
326 const int n_input = input_shape.dim(input_shape.num_dims() - 1);
327
328 int aux_input_temp = 0;
329 if (aux_input)
330 {
331 const Shape &aux_input_shape = aux_input->shape();
332 aux_input_temp = aux_input_shape.dim(aux_input_shape.num_dims() - 1);
333 }
334 const int aux_input_size = aux_input_temp;
335
336 // n_cell and n_output will be the same size when there is no projection.
337 const Shape &input_to_output_weights_shape = input_to_output_weights->shape();
338 const Shape &recurrent_to_output_weights_shape = recurrent_to_output_weights->shape();
339 const int n_cell = input_to_output_weights_shape.dim(0);
340 const int n_output = recurrent_to_output_weights_shape.dim(1);
341
342 // Since we have already checked that weights are all there or none, we can
343 // check the existence of only one to the get the condition.
344 const bool use_cifg = (input_to_input_weights == nullptr);
345
346 // Index the scratch buffers pointers to the global scratch buffer.
347 float *scratch_buffer_ptr = getTensorData<float>(scratch_buffer);
348 float *input_gate_scratch = nullptr;
349 float *cell_gate_scratch = nullptr;
350 float *forget_gate_scratch = nullptr;
351 float *output_gate_scratch = nullptr;
352 if (use_cifg)
353 {
354 cell_gate_scratch = scratch_buffer_ptr;
355 forget_gate_scratch = scratch_buffer_ptr + n_cell * n_batch;
356 output_gate_scratch = scratch_buffer_ptr + 2 * n_cell * n_batch;
357 }
358 else
359 {
360 input_gate_scratch = scratch_buffer_ptr;
361 cell_gate_scratch = scratch_buffer_ptr + n_cell * n_batch;
362 forget_gate_scratch = scratch_buffer_ptr + 2 * n_cell * n_batch;
363 output_gate_scratch = scratch_buffer_ptr + 3 * n_cell * n_batch;
364 }
365
366 const Shape &output_shape = output->shape();
367 const int output_batch_leading_dim = output_shape.dim(output_shape.num_dims() - 1);
368 if (time_major)
369 {
370 // Loop through the sequence.
371 const int input_step = n_batch * n_input;
372 const int output_step = n_batch * output_batch_leading_dim;
373 for (int t = 0; t < max_time; t++)
374 {
375 // If this is the forward_sequence, step forward, otherwise step
376 // backwards.
377 const int t_rel = forward_sequence ? t : max_time - t - 1;
378 const float *input_ptr = getTensorData<float>(input) + t_rel * input_step;
379 const float *aux_input_ptr = nullptr;
380 if (aux_input)
381 {
382 aux_input_ptr = getTensorData<float>(aux_input) + t_rel * input_step;
383 }
384 float *output_ptr = getTensorData<float>(output) + t_rel * output_step + output_offset;
385
386 LstmStepFloat(
387 input_ptr, getTensorData<float>(input_to_input_weights),
388 getTensorData<float>(input_to_forget_weights), getTensorData<float>(input_to_cell_weights),
389 getTensorData<float>(input_to_output_weights), aux_input_ptr,
390 getTensorData<float>(aux_input_to_input_weights),
391 getTensorData<float>(aux_input_to_forget_weights),
392 getTensorData<float>(aux_input_to_cell_weights),
393 getTensorData<float>(aux_input_to_output_weights),
394 getTensorData<float>(recurrent_to_input_weights),
395 getTensorData<float>(recurrent_to_forget_weights),
396 getTensorData<float>(recurrent_to_cell_weights),
397 getTensorData<float>(recurrent_to_output_weights),
398 getTensorData<float>(cell_to_input_weights), getTensorData<float>(cell_to_forget_weights),
399 getTensorData<float>(cell_to_output_weights),
400 getTensorData<float>(input_layer_norm_coefficients),
401 getTensorData<float>(forget_layer_norm_coefficients),
402 getTensorData<float>(cell_layer_norm_coefficients),
403 getTensorData<float>(output_layer_norm_coefficients), getTensorData<float>(input_gate_bias),
404 getTensorData<float>(forget_gate_bias), getTensorData<float>(cell_gate_bias),
405 getTensorData<float>(output_gate_bias), getTensorData<float>(projection_weights),
406 getTensorData<float>(projection_bias), params, n_batch, n_cell, n_input, aux_input_size,
407 n_output, output_batch_leading_dim, getTensorData<float>(output_state),
408 getTensorData<float>(cell_state), input_gate_scratch, forget_gate_scratch,
409 cell_gate_scratch, output_gate_scratch, output_ptr);
410 }
411 }
412 else
413 {
414 for (int b = 0; b < n_batch; b++)
415 {
416 const int input_step = n_input;
417 const int output_step = output_batch_leading_dim;
418 for (int t = 0; t < max_time; t++)
419 {
420 // If this is the forward_sequence, step forward, otherwise step
421 // backwards.
422 const int t_rel = forward_sequence ? t : max_time - t - 1;
423 const int time_offset = b * max_time + t_rel;
424 const float *input_ptr = getTensorData<float>(input) + time_offset * input_step;
425 const float *aux_input_ptr = nullptr;
426 if (aux_input)
427 {
428 aux_input_ptr = getTensorData<float>(aux_input) + time_offset * input_step;
429 }
430 float *output_ptr =
431 getTensorData<float>(output) + time_offset * output_step + output_offset;
432
433 // Offset the {output,cell}_state pointers to the right batch.
434 float *output_state_ptr = getTensorData<float>(output_state) + b * output_batch_leading_dim;
435 float *cell_state_ptr = getTensorData<float>(cell_state) + b * n_cell;
436 // Offset the scratch pointers to the right batch.
437 float *input_gate_scratch_ptr =
438 input_gate_scratch ? input_gate_scratch + b * n_cell : nullptr;
439 float *forget_gate_scratch_ptr = forget_gate_scratch + b * n_cell;
440 float *cell_gate_scratch_ptr = cell_gate_scratch + b * n_cell;
441 float *output_gate_scratch_ptr = output_gate_scratch + b * n_cell;
442
443 LstmStepFloat(
444 input_ptr, getTensorData<float>(input_to_input_weights),
445 getTensorData<float>(input_to_forget_weights),
446 getTensorData<float>(input_to_cell_weights),
447 getTensorData<float>(input_to_output_weights), aux_input_ptr,
448 getTensorData<float>(aux_input_to_input_weights),
449 getTensorData<float>(aux_input_to_forget_weights),
450 getTensorData<float>(aux_input_to_cell_weights),
451 getTensorData<float>(aux_input_to_output_weights),
452 getTensorData<float>(recurrent_to_input_weights),
453 getTensorData<float>(recurrent_to_forget_weights),
454 getTensorData<float>(recurrent_to_cell_weights),
455 getTensorData<float>(recurrent_to_output_weights),
456 getTensorData<float>(cell_to_input_weights), getTensorData<float>(cell_to_forget_weights),
457 getTensorData<float>(cell_to_output_weights),
458 getTensorData<float>(input_layer_norm_coefficients),
459 getTensorData<float>(forget_layer_norm_coefficients),
460 getTensorData<float>(cell_layer_norm_coefficients),
461 getTensorData<float>(output_layer_norm_coefficients),
462 getTensorData<float>(input_gate_bias), getTensorData<float>(forget_gate_bias),
463 getTensorData<float>(cell_gate_bias), getTensorData<float>(output_gate_bias),
464 getTensorData<float>(projection_weights), getTensorData<float>(projection_bias), params,
465 /*n_batch=*/1, n_cell, n_input, aux_input_size, n_output, output_batch_leading_dim,
466 output_state_ptr, cell_state_ptr, input_gate_scratch_ptr, forget_gate_scratch_ptr,
467 cell_gate_scratch_ptr, output_gate_scratch_ptr, output_ptr);
468 }
469 }

535void UnidirectionalSequenceLSTM::populate_quantized_lstm_params()
536{
537 const float cell_clip = params().cell_clip;
538 const float proj_clip = params().proj_clip;
539
540 LUCI_INTERPRETER_CHECK(cell_state() != nullptr);
541 LUCI_INTERPRETER_CHECK(cell_state()->scales().size() > 0 and
542 cell_state()->zero_points().size() > 0);
543
544 LUCI_INTERPRETER_CHECK(output()->scales().size() > 0 and output()->zero_points().size() > 0);
545
546 if (cell_clip > 0.0f)
547 {
548 integer_lstm_params.quantized_cell_clip = static_cast<int16_t>(
549 std::min(std::max(cell_clip / cell_state()->scales()[0], -32768.0f), 32767.0f));
550 }
551 else
552 {
553 integer_lstm_params.quantized_cell_clip = 0;
554 }
555
556 if (proj_clip > 0.0f)
557 {
558 integer_lstm_params.quantized_proj_clip =
559 static_cast<int8_t>(std::min(std::max(proj_clip / output()->scales()[0], -128.0f), 127.0f));
560 }
561 else
562 {
563 integer_lstm_params.quantized_proj_clip = 0;
564 }
565
566 // Calculate effective scales.
567 bool use_layer_norm = (forget_layer_norm_coefficients() != nullptr);
568
569 // Since we have already checked that weights are all there or none, we can
570 // check the existence of only one to get the condition.
571 const bool use_cifg = (input_to_input_weights() == nullptr);
572 const bool use_peephole = (cell_to_output_weights() != nullptr);
573 const bool use_projection = (projection_weights() != nullptr);
574
575 // Get intermediate scales and zero points.
576 float intermediate_scale[5];
577 int32_t intermediate_zp[5];
578
579 for (int i = 0; i < 4; ++i)
580 {
581 if (use_layer_norm)
582 {
583 assert(false);
584 // TODO: support it
585 }
586 else
587 {
588 intermediate_scale[i] = std::pow(2.0f, -12.0f);
589 intermediate_zp[i] = 0;
590 }
591 }
592 intermediate_scale[4] = params().intermediate_affine_quant[0]->scale[0];
593 intermediate_zp[4] = params().intermediate_affine_quant[0]->zero_point[0];
594
595 // Scales.
596 const float default_scale = 1.0;
597 float input_scale = default_scale;
598 float input_to_input_weight_scale = default_scale;
599 float recurrent_to_input_weight_scale = default_scale;
600 float cell_to_input_weight_scale = default_scale;
601 float input_to_forget_weight_scale = default_scale;
602 float recurrent_to_forget_weight_scale = default_scale;
603 float cell_to_forget_weight_scale = default_scale;
604 float input_to_cell_weight_scale = default_scale;
605 float recurrent_to_cell_weight_scale = default_scale;
606 float input_to_output_weight_scale = default_scale;
607 float recurrent_to_output_weight_scale = default_scale;
608 float cell_to_output_weight_scale = default_scale;
609 float projection_weight_scale = default_scale;
610 float layer_norm_input_scale = default_scale;
611 float layer_norm_forget_scale = default_scale;
612 float layer_norm_cell_scale = default_scale;
613 float layer_norm_output_scale = default_scale;
614 float output_state_scale = default_scale;
615 int cell_scale = 1;
616
617 // Effective scales.
618 float effective_input_to_input_scale = default_scale;
619 float effective_recurrent_to_input_scale = default_scale;
620 float effective_cell_to_input_scale = default_scale;
621 float effective_input_to_forget_scale = default_scale;
622 float effective_recurrent_to_forget_scale = default_scale;
623 float effective_cell_to_forget_scale = default_scale;
624 float effective_input_to_cell_scale = default_scale;
625 float effective_recurrent_to_cell_scale = default_scale;
626 float effective_input_to_output_scale = default_scale;
627 float effective_recurrent_to_output_scale = default_scale;
628 float effective_cell_to_output_scale = default_scale;
629 float effective_proj_scale = default_scale;
630 float effective_hidden_scale = default_scale;
631
632 // Populate scales.
633 if (!use_cifg)
634 {
635 input_to_input_weight_scale = input_to_input_weights()->scale();
636 recurrent_to_input_weight_scale = recurrent_to_input_weights()->scale();
637 }
638
639 if (use_peephole)
640 {
641 if (!use_cifg)
642 {
643 cell_to_input_weight_scale = cell_to_input_weights()->scale();
644 }
645 cell_to_forget_weight_scale = cell_to_forget_weights()->scale();
646 cell_to_output_weight_scale = cell_to_output_weights()->scale();
647 }
648
649 if (use_layer_norm)
650 {
651 if (!use_cifg)
652 {
653 layer_norm_input_scale = input_layer_norm_coefficients()->scale();
654 }
655 layer_norm_forget_scale = forget_layer_norm_coefficients()->scale();
656 layer_norm_cell_scale = cell_layer_norm_coefficients()->scale();
657 layer_norm_output_scale = output_layer_norm_coefficients()->scale();
658 }
659
660 if (use_projection)
661 {
662 projection_weight_scale = projection_weights()->scale();
663 }
664 output_state_scale = output_state()->scale();
665
666 input_to_forget_weight_scale = input_to_forget_weights()->scale();
667 input_to_cell_weight_scale = input_to_cell_weights()->scale();
668 input_to_output_weight_scale = input_to_output_weights()->scale();
669 recurrent_to_forget_weight_scale = recurrent_to_forget_weights()->scale();
670 recurrent_to_cell_weight_scale = recurrent_to_cell_weights()->scale();
671 recurrent_to_output_weight_scale = recurrent_to_output_weights()->scale();
672
673 // Check cell state (already used above)
674 {
675 const float x_log2 = std::log(cell_state()->scale()) * (1.0f / std::log(2.0f));
676 const float x_log2_rounded = std::round(x_log2);
677 const float x_log2_fracpart = x_log2 - x_log2_rounded;
678 cell_scale = static_cast<int>(x_log2_rounded);
679
680 LUCI_INTERPRETER_CHECK(std::abs(x_log2_fracpart) < 1e-3f);
681 }
682 integer_lstm_params.cell_scale = cell_scale;
683 input_scale = input()->scale();
684
685 // Calculate effective scales.
686 if (!use_cifg)
687 {
688 effective_input_to_input_scale =
689 input_to_input_weight_scale * input_scale / intermediate_scale[0];
690 effective_recurrent_to_input_scale =
691 recurrent_to_input_weight_scale * output_state_scale / intermediate_scale[0];
692 }
693 effective_input_to_forget_scale =
694 input_to_forget_weight_scale * input_scale / intermediate_scale[1];
695 effective_recurrent_to_forget_scale =
696 recurrent_to_forget_weight_scale * output_state_scale / intermediate_scale[1];
697
698 effective_input_to_cell_scale = input_to_cell_weight_scale * input_scale / intermediate_scale[2];
699 effective_recurrent_to_cell_scale =
700 recurrent_to_cell_weight_scale * output_state_scale / intermediate_scale[2];
701
702 effective_input_to_output_scale =
703 input_to_output_weight_scale * input_scale / intermediate_scale[3];
704 effective_recurrent_to_output_scale =
705 recurrent_to_output_weight_scale * output_state_scale / intermediate_scale[3];
706
707 effective_hidden_scale = std::pow(2.0f, -15.0f) / intermediate_scale[4] * std::pow(2.0f, -15.0f);
708
709 effective_proj_scale = projection_weight_scale * intermediate_scale[4] / output_state_scale;
710
711 if (use_peephole)
712 {
713 if (!use_cifg)
714 {
715 effective_cell_to_input_scale = std::pow(2.0f, static_cast<float>(cell_scale)) *
716 cell_to_input_weight_scale / intermediate_scale[0];
717 }
718 effective_cell_to_forget_scale = std::pow(2.0f, static_cast<float>(cell_scale)) *
719 cell_to_forget_weight_scale / intermediate_scale[1];
720 effective_cell_to_output_scale = std::pow(2.0f, static_cast<float>(cell_scale)) *
721 cell_to_output_weight_scale / intermediate_scale[3];
722 }
723
724 // Decompose scales.
725 int shift_output = 0;
726
727 quantizeMultiplier(static_cast<double>(effective_input_to_input_scale),
728 &integer_lstm_params.effective_input_to_input_scale_a, &shift_output);
729 integer_lstm_params.effective_input_to_input_scale_b = static_cast<int32_t>(shift_output);
730
731 quantizeMultiplier(static_cast<double>(effective_recurrent_to_input_scale),
732 &integer_lstm_params.effective_recurrent_to_input_scale_a, &shift_output);
733 integer_lstm_params.effective_recurrent_to_input_scale_b = static_cast<int32_t>(shift_output);
734
735 quantizeMultiplier(static_cast<double>(effective_cell_to_input_scale),
736 &integer_lstm_params.effective_cell_to_input_scale_a, &shift_output);
737 integer_lstm_params.effective_cell_to_input_scale_b = static_cast<int32_t>(shift_output);
738
739 quantizeMultiplier(static_cast<double>(effective_input_to_forget_scale),
740 &integer_lstm_params.effective_input_to_forget_scale_a, &shift_output);
741 integer_lstm_params.effective_input_to_forget_scale_b = static_cast<int32_t>(shift_output);
742
743 quantizeMultiplier(static_cast<double>(effective_recurrent_to_forget_scale),
744 &integer_lstm_params.effective_recurrent_to_forget_scale_a, &shift_output);
745 integer_lstm_params.effective_recurrent_to_forget_scale_b = static_cast<int32_t>(shift_output);
746
747 quantizeMultiplier(static_cast<double>(effective_cell_to_forget_scale),
748 &integer_lstm_params.effective_cell_to_forget_scale_a, &shift_output);
749 integer_lstm_params.effective_cell_to_forget_scale_b = static_cast<int32_t>(shift_output);
750 quantizeMultiplier(static_cast<double>(effective_input_to_cell_scale),
751 &integer_lstm_params.effective_input_to_cell_scale_a, &shift_output);
752 integer_lstm_params.effective_input_to_cell_scale_b = static_cast<int32_t>(shift_output);
753
754 quantizeMultiplier(static_cast<double>(effective_recurrent_to_cell_scale),
755 &integer_lstm_params.effective_recurrent_to_cell_scale_a, &shift_output);
756 integer_lstm_params.effective_recurrent_to_cell_scale_b = static_cast<int32_t>(shift_output);
757
758 quantizeMultiplier(static_cast<double>(effective_input_to_output_scale),
759 &integer_lstm_params.effective_input_to_output_scale_a, &shift_output);
760 integer_lstm_params.effective_input_to_output_scale_b = static_cast<int32_t>(shift_output);
761
762 quantizeMultiplier(static_cast<double>(effective_recurrent_to_output_scale),
763 &integer_lstm_params.effective_recurrent_to_output_scale_a, &shift_output);
764 integer_lstm_params.effective_recurrent_to_output_scale_b = static_cast<int32_t>(shift_output);
765
766 quantizeMultiplier(static_cast<double>(effective_cell_to_output_scale),
767 &integer_lstm_params.effective_cell_to_output_scale_a, &shift_output);
768 integer_lstm_params.effective_cell_to_output_scale_b = static_cast<int32_t>(shift_output);
769
770 quantizeMultiplier(static_cast<double>(effective_proj_scale),
771 &integer_lstm_params.effective_proj_scale_a, &shift_output);
772 integer_lstm_params.effective_proj_scale_b = static_cast<int32_t>(shift_output);
773
774 quantizeMultiplier(static_cast<double>(effective_hidden_scale),
775 &integer_lstm_params.effective_hidden_scale_a, &shift_output);
776
777 integer_lstm_params.effective_hidden_scale_b = static_cast<int32_t>(shift_output);
778
779 quantizeMultiplier(static_cast<double>(layer_norm_input_scale),
780 &integer_lstm_params.layer_norm_input_scale_a, &shift_output);
781 integer_lstm_params.layer_norm_input_scale_b = static_cast<int32_t>(shift_output);
782
783 quantizeMultiplier(static_cast<double>(layer_norm_forget_scale),
784 &integer_lstm_params.layer_norm_forget_scale_a, &shift_output);
785 integer_lstm_params.layer_norm_forget_scale_b = static_cast<int32_t>(shift_output);
786
787 quantizeMultiplier(static_cast<double>(layer_norm_cell_scale),
788 &integer_lstm_params.layer_norm_cell_scale_a, &shift_output);
789 integer_lstm_params.layer_norm_cell_scale_b = static_cast<int32_t>(shift_output);
790
791 quantizeMultiplier(static_cast<double>(layer_norm_output_scale),
792 &integer_lstm_params.layer_norm_output_scale_a, &shift_output);
793 integer_lstm_params.layer_norm_output_scale_b = static_cast<int32_t>(shift_output);
794
795 integer_lstm_params.hidden_zp = intermediate_zp[4];
796
797 // 10000 is used to make sure the kernel logic does not overflow.
798 if (!use_cifg)
799 {
800 integer_lstm_params.input_variance_guard =
801 std::max(1, static_cast<int>(10000 * layer_norm_input_scale));
802 }
803 integer_lstm_params.forget_variance_guard =
804 std::max(1, static_cast<int>(10000 * layer_norm_forget_scale));
805 integer_lstm_params.cell_variance_guard =
806 std::max(1, static_cast<int>(10000 * layer_norm_cell_scale));
807 integer_lstm_params.output_variance_guard =
808 std::max(1, static_cast<int>(10000 * layer_norm_output_scale));
809}

811void UnidirectionalSequenceLSTM::populate_precomputed_zp_times_weight_with_bias()
812{
813 const int32_t input_zero_point = -input()->zero_point();
814 const int32_t output_state_zero_point = -output_state()->zero_point();
815
816 const int32_t hidden_zp = params().intermediate_affine_quant[0]->zero_point[0];
817
818 // Get bias and perform zero point calculation.
819 // When there is layer normalization, the gate bias does not apply to matmul
820 // directly:
821 // y = ln(w * x + w * r + w * c) + b.
822
823 bool is_layer_norm = (forget_layer_norm_coefficients() != nullptr);
824
825 const auto forget_gate_bias_tmp = is_layer_norm ? nullptr : forget_gate_bias();
826
827 precompute_zero_point_times_weight_with_bias(input_zero_point, input_to_forget_weights(),
828 forget_gate_bias_tmp,
829 integer_lstm_params.input_to_forget_effective_bias);
830
831 precompute_zero_point_times_weight_with_bias(
832 output_state_zero_point, recurrent_to_forget_weights(), nullptr,
833 integer_lstm_params.recurrent_to_forget_effective_bias);
834
835 // Modulation gate.
836 const auto cell_gate_bias_tmp = is_layer_norm ? nullptr : cell_gate_bias();
837
838 precompute_zero_point_times_weight_with_bias(input_zero_point, input_to_cell_weights(),
839 cell_gate_bias_tmp,
840 integer_lstm_params.input_to_cell_effective_bias);
841 precompute_zero_point_times_weight_with_bias(
842 output_state_zero_point, recurrent_to_cell_weights(), nullptr,
843 integer_lstm_params.recurrent_to_cell_effective_bias);
844
845 // Output gate.
846 const auto output_gate_bias_tmp = is_layer_norm ? nullptr : output_gate_bias();
847
848 precompute_zero_point_times_weight_with_bias(input_zero_point, input_to_output_weights(),
849 output_gate_bias_tmp,
850 integer_lstm_params.input_to_output_effective_bias);
851 precompute_zero_point_times_weight_with_bias(
852 output_state_zero_point, recurrent_to_output_weights(), nullptr,
853 integer_lstm_params.recurrent_to_output_effective_bias);
854
855 // Input gate. The calculation is only meaningful for non-cifg case.
856 const auto input_gate_bias_tmp = is_layer_norm ? nullptr : input_gate_bias();
857
858 precompute_zero_point_times_weight_with_bias(input_zero_point, input_to_input_weights(),
859 input_gate_bias_tmp,
860 integer_lstm_params.input_to_input_effective_bias);
861 precompute_zero_point_times_weight_with_bias(
862 output_state_zero_point, recurrent_to_input_weights(), nullptr,
863 integer_lstm_params.recurrent_to_input_effective_bias);
864
865 // Projection bias. The calculation is only meaningful for with projection.
866 precompute_zero_point_times_weight_with_bias(hidden_zp, projection_weights(), projection_bias(),
867 integer_lstm_params.projection_effective_bias);
868}

870void UnidirectionalSequenceLSTM::precompute_zero_point_times_weight_with_bias(
871 int32_t zero_point, const Tensor *weight_tensor, const Tensor *bias_tensor,
872 std::vector<int32_t> &output)
873{
874 if (weight_tensor == nullptr)
875 return;
876
877 const auto weight_shape = weight_tensor->shape();
878
879 LUCI_INTERPRETER_CHECK(weight_shape.num_dims() == 2);
880 const int row = weight_shape.dim(0);
881 const int col = weight_shape.dim(1);
882
883 output.resize(row);
884
885 if (bias_tensor == nullptr)
886 {
887 memset(output.data(), 0, row * sizeof(int32_t));
888 }
889 else
890 {
891 const auto *bias = getTensorData<int32_t>(bias_tensor);
892 std::memcpy(output.data(), bias, row * sizeof(int32_t));
893 }
894 if (zero_point != 0)
895 {
896 const auto *weight = getTensorData<int8_t>(weight_tensor);
897 matrixScalarMultiplyAccumulate(weight, zero_point, row, col, output.data());
898 }
899}

1003 LUCI_INTERPRETER_CHECK(input_gate_bias_shape.dim(0) == n_cell);

1015 LUCI_INTERPRETER_CHECK(forget_gate_bias_shape.num_dims() == 1);
1016 LUCI_INTERPRETER_CHECK(forget_gate_bias_shape.dim(0) == n_cell);
1017 if (is_integer)
1018 {
1019 LUCI_INTERPRETER_CHECK(forget_gate_bias()->element_type() == DataType::S32);
1020 }
1021 else
1022 {
1023 LUCI_INTERPRETER_CHECK(forget_gate_bias()->element_type() == DataType::FLOAT32);

1027 LUCI_INTERPRETER_CHECK(cell_gate_bias_shape.num_dims() == 1);
1028 LUCI_INTERPRETER_CHECK(cell_gate_bias_shape.dim(0) == n_cell);
1029 if (is_integer)
1030 {
1031 LUCI_INTERPRETER_CHECK(cell_gate_bias()->element_type() == DataType::S32);

1035 LUCI_INTERPRETER_CHECK(cell_gate_bias()->element_type() == DataType::FLOAT32);

1159 // number of cells from the input tensors.

1161 LUCI_INTERPRETER_CHECK(input_shape.num_dims() > 1);
1162 const bool time_major = params().time_major;
1163 const int n_batch = time_major ? input_shape.dim(1) : input_shape.dim(0);
1164 // NOTE as dim(2) is accessed, we need to check this is valid

1169 const int n_cell = input_to_output_weights_shape.dim(0);
1170 LUCI_INTERPRETER_CHECK(input_to_output_weights_shape.num_dims() == 2);
1171 LUCI_INTERPRETER_CHECK(input_to_output_weights_shape.dim(1) == n_input);

1174 LUCI_INTERPRETER_CHECK(recurrent_to_output_weights_shape.num_dims() == 2);
1175 LUCI_INTERPRETER_CHECK(recurrent_to_output_weights_shape.dim(0) == n_cell);
1176
1177 const int n_output = recurrent_to_output_weights_shape.dim(1);
1178
1179 // Check that input tensor dimensions matches with each other.
1180 check_input_tensor_dimensions(n_input, n_output, n_cell, use_layer_norm, is_integer);
1181
1182 // Check the shape of input state tensors.
1183 // These tensor may be 1D or 2D. It's fine as long as the total size is
1184 // correct.
1185 const Shape &output_state_shape = output_state()->shape();
1186 const Shape &cell_state_shape = cell_state()->shape();
1187 LUCI_INTERPRETER_CHECK(output_state_shape.num_elements() == n_batch * n_output);
1188 LUCI_INTERPRETER_CHECK(cell_state_shape.num_elements() == n_batch * n_cell);

1190 // Resize the output tensors.
1191 Shape output_shape = Shape(input_shape.num_dims());

1193 {
1194 output_shape.dim(i) = input_shape.dim(i);
1195 }

1199 const bool use_cifg = (input_to_input_weights() == nullptr);
1200

1214 input()->element_type() == DataType::FLOAT32)
1215 {
1216 // TODO support hybrid

1218 }
1219 }
