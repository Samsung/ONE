29bool canonicalize_biasadd(loco::Graph *graph, moco::TFBiasAdd *node)
30{
31 LOGGER(l);
32
54 INFO(l) << "TFNodeCanonicalize TFBiasAdd begin";
55
56 // tensorflow data_format: one of NHWC or NCHW.
57 auto data_layout = plier::tf::as_data_layout(node->data_layout());
58
59 // creating loco nodes
60 auto bias_enc = graph->nodes()->create<loco::BiasEncode>();
61
62 auto bias_add = graph->nodes()->create<loco::TensorBiasAdd>();
63 {
64 if (data_layout == DataLayout::NHWC)
65 {
66 INFO(l) << "TFNodeCanonicalize TFBiasAdd axis 3";
67 bias_add->axis(3);
68 }
69 else if (data_layout == DataLayout::NCHW)
70 {
71 INFO(l) << "TFNodeCanonicalize TFBiasAdd axis 1";
72 bias_add->axis(1); // Channel
73 // Note: the following descrition of TF 1.13 at
74 // https://www.tensorflow.org/api_docs/python/tf/nn/bias_add seems wrong:
75 // "bias: A 1-D Tensor with size matching the last dimension of value."
76 // because providing the size of W (last dimension) to bias throws an error with TensorFlow
77 }
78 }
79
80 auto node_A = node->value();
81 auto node_B = node->bias();
82
83 // update connections
84 bias_add->value(node_A);
85 bias_add->bias(bias_enc);
86 bias_enc->input(node_B);
87
88 // replace old with new : about C in above note
89 replace(node).with(bias_add);
90
91 INFO(l) << "TFNodeCanonicalize TFBiasAdd done";
92
93 return true;
94}
