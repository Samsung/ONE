43class KernelBuiltinExecuteRegistry
44{
45public:
46 constexpr KernelBuiltinExecuteRegistry() : _operator_execute()
47 {
48#define REGISTER_KERNEL(builtin_operator, name) \
49 registerKernelExecute(core::OMBuilderID::BuiltinOperator_##builtin_operator, \
50 execute_kernel_Circle##name);
51
52#include "KernelsToBuild.lst"
53
54#undef REGISTER_KERNEL
55 }
56
57public:
58 OMStatus getKernelExecuteFunc(core::OMBuilderID builderID, KernelExecuteFunc **execute_func) const
59 {
60 const auto builder_id_opcode = size_t(builderID);
61 assert(builder_id_opcode < size_t(core::OMBuilderID::BuiltinOperatorsSize));
62 if (builder_id_opcode >= size_t(core::OMBuilderID::BuiltinOperatorsSize))
63 {
64 *execute_func = nullptr;
65 return UnknownError;
66 }
67 *execute_func = _operator_execute[builder_id_opcode];
68 return Ok;
69 }
70
71private:
72 constexpr void registerKernelExecute(core::OMBuilderID id, KernelExecuteFunc *func)
73 {
74 assert(size_t(id) < size_t(core::OMBuilderID::BuiltinOperatorsSize));
75 _operator_execute[size_t(id)] = func;
76 }
77
78private:
79 KernelExecuteFunc *_operator_execute[size_t(core::OMBuilderID::BuiltinOperatorsSize)];
80};

82class KernelCustomExecuteRegistry
83{
84public:
85 constexpr KernelCustomExecuteRegistry() : _operator_execute()
86 {
87#define REGISTER_CUSTOM_KERNEL(name, string_name) \
88 registerKernelExecute(core::OMBuilderID::CUSTOM_##name, execute_kernel_Circle##name);
89
90#include "CustomKernelsToBuild.lst"
91
92#undef REGISTER_CUSTOM_KERNEL
93 }
94
95public:
96 OMStatus getKernelExecuteFunc(core::OMBuilderID builderID, KernelExecuteFunc **execute_func) const
97 {
98 auto builder_id_opcode = size_t(builderID);
99 if (builder_id_opcode >= size_t(core::OMBuilderID::Size))
100 {
101 *execute_func = nullptr;
102 return UnknownError;
103 }
104 const auto builder_id_offset = size_t(core::OMBuilderID::BuiltinOperatorsSize);
105 builder_id_opcode -= builder_id_offset - 1;
106 if (builder_id_opcode < 0)
107 {
108 *execute_func = nullptr;
109 return UnknownError;
110 }
111 *execute_func = _operator_execute[builder_id_opcode];
112 return Ok;
113 }
114
115private:
116 constexpr void registerKernelExecute(core::OMBuilderID id, KernelExecuteFunc *func)
117 {
118 auto builder_id_opcode = size_t(id);
119 const auto builder_id_offset = size_t(core::OMBuilderID::BuiltinOperatorsSize);
120 builder_id_opcode = builder_id_opcode - builder_id_offset - 1;
121 _operator_execute[builder_id_opcode] = func;
122 }
123
124private:
125 KernelExecuteFunc *_operator_execute[size_t(core::OMBuilderID::Size) -
126 size_t(core::OMBuilderID::BuiltinOperatorsSize) - 1];
127};
