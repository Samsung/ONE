60/*
61 * OMTrainingContext contains training specific parameters
62 * batch_size - batch size for training (Note: 1 - default one)
63 * num_of_train_layers - number of trainable last layers (Note: 0 - all layers will be trained)
64 * optimizer - optimizer which onert-micro training will be used (Note: SGD - default one)
65 * loss - loss which onert-micro training will be used (Note: CROSS_ENTROPY - default one)
66 * learning_rate - used by all optimizers
67 * beta - used by ADAM optimizer
68 * beta_squares - used by ADAM optimizer
69 * epsilon - used by ADAM optimizer
70 * num_Step - used by ADAM optimizer
71 * training_config_info_data - pointer to the training config data, to store training specific
72 * scenario (default null)
73 */
74struct OMTrainingContext
75{
76 uint32_t batch_size = 1;
77 uint32_t num_of_train_layers = 0;
78 OMTrainOptimizer optimizer = SGD;
79 OMLoss loss = MSE;
80 float learning_rate = 0.001f;
81 float beta = 0.9f;
82 float beta_squares = 0.9f;
83 float epsilon = 10e-8;
84 uint32_t num_step = 0;
85 uint32_t num_epoch = 0;
86 uint32_t epochs = 0;
87
88 char *training_config_info_data = nullptr;
89};

91/*
92 * OMConfig - main config to store information for runtime and for training runtime
93 * keep_input - will the allocated memory be saved for the input data, or can it be deleted after
94 * use cmsis_nn - will CMSIS NN kernels be used or not (needed to some internal settings) wof_ptr -
95 * a pointer to the data that stores weights separate from the model train_mode - a flag to indicate
96 * whether we are currently in training mode or not
97 */
98struct OMConfig

102 // For case with divided weights and circle file
103 char *wof_ptr = nullptr;
104 bool train_mode = false;
105 char *model_ptr = nullptr;
106 size_t model_size = 0;
107 OMTrainingContext training_context = {};
108};
