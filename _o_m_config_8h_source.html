57/*
58 * OMTrainingContext contains training specific parameters
59 * batch_size - batch size for training (Note: 1 - default one)
60 * num_of_train_layers - number of trainable last layers (Note: 0 - all layers will be trained)
61 * optimizer - optimizer which onert-micro training will be used (Note: SGD - default one)
62 * loss - loss which onert-micro training will be used (Note: CROSS_ENTROPY - default one)
63 * learning_rate - used by all optimizers
64 * beta - used by ADAM optimizer
65 * beta_squares - used by ADAM optimizer
66 * epsilon - used by ADAM optimizer
67 * num_Step - used by ADAM optimizer
68 */
69struct OMTrainingContext
70{
71 uint32_t batch_size = 1;
72 uint32_t num_of_train_layers = 0;
73 OMTrainOptimizer optimizer = SGD;
74 OMLoss loss = MSE;
75 float learning_rate = 0.001f;
76 float beta = 0.9f;
77 float beta_squares = 0.9f;
78 float epsilon = 10e-8;
79 uint32_t num_step = 0;
80 uint32_t num_epoch = 0;
81};

83/*
84 * OMConfig - main config to store information for runtime and for training runtime
85 * keep_input - will the allocated memory be saved for the input data, or can it be deleted after
86 * use cmsis_nn - will CMSIS NN kernels be used or not (needed to some internal settings) wof_ptr -
87 * a pointer to the data that stores weights separate from the model train_mode - a flag to indicate
88 * whether we are currently in training mode or not
89 */
90struct OMConfig
91{
92 bool keep_input = false;
93 bool cmsis_nn = false;
94 // For case with divided weights and circle file
95 char *wof_ptr = nullptr;
96 bool train_mode = false;
97 char *model_ptr = nullptr;
98 size_t model_size = 0;
99 OMTrainingContext training_context = {};
100};
