62class ExecObservers
63{
64public:
65 void add(std::unique_ptr<IExecutionObserver> &&observer)
66 {
67 _observers.emplace(observer->type(), std::move(observer));
68 }
69
70 IExecutionObserver *get(ObserverType type) const
71 {
72 if (_observers.find(type) != _observers.end())
73 return _observers.at(type).get();
74
75 return nullptr;
76 }
77
78private:
79 std::unordered_map<ObserverType, std::unique_ptr<IExecutionObserver>> _observers;
80};
