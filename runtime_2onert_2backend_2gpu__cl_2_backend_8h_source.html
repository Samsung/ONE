48 std::unique_ptr<onert::backend::BackendContext> newContext(ContextData &&data) const override
49 {
50 const auto &graph = *data.graph;
51 const auto &operands = data.graph->operands();
52 auto context = std::make_unique<gpu_cl::BackendContext>(this, std::move(data));
53
54 auto environment = std::make_shared<tflite::gpu::cl::Environment>();
55 if (!CreateEnvironment(environment.get()).ok())
56 {
57 return nullptr;
58 }
59
60 tflite::gpu::CreateGpuModelInfo create_info;
61 create_info.precision = tflite::gpu::CalculationsPrecision::F32;
62 create_info.storage_type =
63 tflite::gpu::cl::GetStorageTypeWithMinimalMemoryConsumption(environment->device().GetInfo());
64 create_info.hints.Add(tflite::gpu::ModelHints::kFastestInference);
65
66 auto tm = createTensorManager(&environment->context(), create_info, environment);
67
68 auto tr = std::make_shared<TensorRegistry>(tm);
69
70 auto cc = std::make_shared<tflite::gpu::cl::CreationContext>();
71 cc->device = environment->GetDevicePtr();
72 cc->context = &environment->context();
73 cc->queue = environment->queue();
74 cc->cache = environment->program_cache();
75
76 auto tb = std::make_shared<TensorBuilder>(operands, tm);
77 context->tensor_registry = tr;
78 context->tensor_builder = tb;
79
80 context->kernel_gen = std::make_shared<KernelGenerator>(graph, tb, tr, cc);
81 context->constant_initializer = std::make_shared<ConstantInitializer>(operands, tr);
82 return context;
83 }

85private:
