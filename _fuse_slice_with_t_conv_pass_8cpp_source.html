31inline int32_t compute_input_size(luci::Padding padding, int32_t image_size, int32_t filter_size,
32 int32_t stride)
33{
34 switch (padding)
35 {
36 case luci::Padding::SAME:
37 return (image_size + stride - 1) / stride;
38 case luci::Padding::VALID:
39 return (image_size + stride - filter_size) / stride;
40 default:
41 throw std::runtime_error("Unsupported padding");
42 }
43}

45inline int32_t extract_pad_value(int32_t stride, int32_t in_size, int32_t filter_size,
46 int32_t out_size)
47{
48 const int32_t padding = ((in_size - 1) * stride + filter_size - out_size) / 2;
49 return padding > 0 ? padding : 0;
50}

52inline uint32_t cal_offset(const luci::CircleConst *node, const uint32_t *indices)
53{
54 return indices[0] * node->dim(1).value() * node->dim(2).value() * node->dim(3).value() +
55 indices[1] * node->dim(2).value() * node->dim(3).value() +
56 indices[2] * node->dim(3).value() + indices[3];
57}

59luci::Padding compute_padding(const luci::CircleTransposeConv *tconv, int32_t out_height,
60 int32_t out_width, int32_t pad_top, int32_t pad_left)
61{
62 auto const filter = dynamic_cast<luci::CircleConst *>(tconv->filter());
63 if (!filter)
64 return luci::Padding::UNDEFINED;
65
66 auto tconv_shape = dynamic_cast<luci::CircleConst *>(tconv->inputSizes());
67 if (!tconv_shape)
68 return luci::Padding::UNDEFINED;
69
70 luci::Padding padding = luci::Padding::UNDEFINED;
71 std::initializer_list<luci::Padding> paddings_to_check = {luci::Padding::VALID,
72 luci::Padding::SAME};
73
74 auto const filter_height = filter->dim(1).value();
75 auto const filter_width = filter->dim(2).value();
76 auto const stride_height = tconv->stride()->h();
77 auto const stride_width = tconv->stride()->w();
78
79 for (auto padding_to_check : paddings_to_check)
80 {
81 auto const in_height =
82 compute_input_size(padding_to_check, out_height, filter_height, stride_height);
83 auto const pad_top_virtual =
84 extract_pad_value(stride_height, in_height, filter_height, out_height);
85 if (pad_top_virtual != pad_top)
86 continue;
87
88 auto const in_width =
89 compute_input_size(padding_to_check, out_width, filter_width, stride_width);
90 auto const pad_left_virtual =
91 extract_pad_value(stride_width, in_width, filter_width, out_width);
92 if (pad_left_virtual == pad_left)
93 {
94 padding = padding_to_check; // correct padding is found
95 break;
96 }
97 }
98
99 return padding;
100}

127bool fuse_slice_with_tconv(luci::CircleSlice *slice)
128{
129 // NOTE: assume NHWC layout
130 auto tconv = dynamic_cast<luci::CircleTransposeConv *>(slice->input());
131 RETURN_FALSE_UNLESS(tconv != nullptr);
132
133 // offset
134 auto begin = dynamic_cast<luci::CircleConst *>(slice->begin());
135 // sanity check
136 RETURN_FALSE_UNLESS(begin != nullptr && begin->dtype() == loco::DataType::S32 &&
137 begin->rank() == 1);
138
139 // output shape
140 auto out_shape = dynamic_cast<luci::CircleConst *>(slice->size());
141 // sanity check
142 RETURN_FALSE_UNLESS(out_shape != nullptr && out_shape->dtype() == loco::DataType::S32 &&
143 out_shape->rank() == 1);
144
145 // output shape of tconv
146 auto tconv_shape = dynamic_cast<luci::CircleConst *>(tconv->inputSizes());
147 // sanity check
148 RETURN_FALSE_UNLESS(tconv_shape != nullptr && tconv_shape->dtype() == loco::DataType::S32 &&
149 tconv_shape->rank() == 1);
150
151 // no update if batch dimension is processed in slice
152 RETURN_FALSE_UNLESS(begin->at<loco::DataType::S32>(0) == 0 &&
153 out_shape->at<loco::DataType::S32>(0) ==
154 tconv_shape->at<loco::DataType::S32>(0));
155
156 // filter
157 auto const tconv_filter = dynamic_cast<luci::CircleConst *>(tconv->filter());
158 // sanity check
159 RETURN_FALSE_UNLESS(tconv_filter != nullptr && tconv_filter->rank() == 4 &&
160 tconv_filter->dtype() == loco::DataType::FLOAT32);
161
162 // bias
163 auto const tconv_bias = dynamic_cast<luci::CircleConst *>(tconv->bias());
164 // Only support const bias
165 // TODO Support non-const bias
166 RETURN_FALSE_UNLESS(tconv_bias != nullptr && tconv_bias->rank() == 1 &&
167 tconv_bias->dtype() == loco::DataType::FLOAT32);
168
169 auto const out_height = out_shape->at<loco::DataType::S32>(1);
170 auto const out_width = out_shape->at<loco::DataType::S32>(2);
171
172 auto const pad_top = begin->at<loco::DataType::S32>(1);
173 auto const pad_left = begin->at<loco::DataType::S32>(2);
174
175 // As there is no option to set numerical values of pad explicitly for CircleTransposeConv
176 // we need to be sure that interpretation of PADDING + OUTPUT_SHAPE will produce
177 // the pad values, defined by slice. If possible compute_padding will return correct
178 // padding value, otherwise it will return UNDEFINED
179 auto const padding = compute_padding(tconv, out_height, out_width, pad_top, pad_left);
180 if (padding == luci::Padding::UNDEFINED)
181 return false; // impossible to fuse
182
183 auto const out_channels = out_shape->at<loco::DataType::S32>(3);
184 // update filter and bias in case it's needed
185 loco::Node *fused_filter = tconv->filter();
186 loco::Node *fused_bias = tconv->bias();
187 // Channel-direction slice
188 // Corresponding weights/bias of TConv is sliced.
189 if (begin->at<loco::DataType::S32>(3) != 0 ||
190 out_channels != tconv_shape->at<loco::DataType::S32>(3))
191 {
192 // fused filter
193 auto const in_channels = tconv_filter->dim(3).value();
194
195 luci::CircleConst *fused_tconv_filter = luci::clone(tconv_filter);
196 fused_tconv_filter->dim(0).set(out_channels); // out_channels
197 // update size due to channels change
198 fused_tconv_filter->size<loco::DataType::FLOAT32>(out_channels * tconv_filter->dim(1).value() *
199 tconv_filter->dim(2).value() * in_channels);
200 auto const ch_offset = begin->at<loco::DataType::S32>(3);
201 // set reduced filter values
202 for (uint32_t out_chan = 0; out_chan < fused_tconv_filter->dim(0).value(); out_chan++)
203 {
204 for (uint32_t out_height = 0; out_height < fused_tconv_filter->dim(1).value(); out_height++)
205 {
206 for (uint32_t out_width = 0; out_width < fused_tconv_filter->dim(2).value(); out_width++)
207 {
208 for (uint32_t in_chan = 0; in_chan < fused_tconv_filter->dim(3).value(); in_chan++)
209 {
210 uint32_t indices[4] = {out_chan, out_height, out_width, in_chan};
211 uint32_t old_indices[4] = {out_chan + ch_offset, out_height, out_width, in_chan};
212 auto const data =
213 tconv_filter->at<loco::DataType::FLOAT32>(cal_offset(tconv_filter, old_indices));
214 fused_tconv_filter->at<loco::DataType::FLOAT32>(
215 cal_offset(fused_tconv_filter, indices)) = data;
216 }
217 }
218 }
219 }
220 fused_tconv_filter->name(tconv_filter->name() + "/FusedSlice");
221 luci::add_origin(fused_tconv_filter, luci::get_origin(tconv_shape));
222 fused_filter = fused_tconv_filter;
223
224 // fused bias
225 luci::CircleConst *fused_tconv_bias = luci::clone(tconv_bias);
226 fused_tconv_bias->size<loco::DataType::FLOAT32>(out_channels);
227 fused_tconv_bias->dim(0).set(out_channels); // out_channels
228 // set reduced bias values
229 for (int32_t c = 0; c < out_channels; c++)
230 {
231 auto const data = tconv_bias->at<loco::DataType::FLOAT32>(c + ch_offset);
232 fused_tconv_bias->at<loco::DataType::FLOAT32>(c) = data;
233 }
234
235 fused_tconv_bias->name(tconv_bias->name() + "/FusedSlice");
236 luci::add_origin(fused_tconv_bias, luci::get_origin(tconv_bias));
237 fused_bias = fused_tconv_bias;
238 }
239
240 auto *fused_tconv_shape = luci::clone(tconv_shape);
241 // spatial dimensions
242 fused_tconv_shape->at<loco::DataType::S32>(1) = out_height;
243 fused_tconv_shape->at<loco::DataType::S32>(2) = out_width;
244 // channels
245 fused_tconv_shape->at<loco::DataType::S32>(3) = out_channels;
246 fused_tconv_shape->name(tconv_shape->name() + "/FusedSlice");
247 luci::add_origin(fused_tconv_shape, luci::get_origin(tconv_shape));
248
249 // Configure new CircleTransposeConv operation.
250 auto *fused_tconv =
251 loco::must_cast<luci::CircleTransposeConv *>(luci::clone_node(tconv, slice->graph()));
252 fused_tconv->inputSizes(fused_tconv_shape);
253 fused_tconv->outBackprop(tconv->outBackprop());
254 fused_tconv->filter(fused_filter);
255 fused_tconv->bias(fused_bias);
256 fused_tconv->padding(padding);
257 fused_tconv->name(tconv->name() + "/FusedSlice");
258 luci::add_origin(fused_tconv,
259 luci::composite_origin({luci::get_origin(tconv), luci::get_origin(slice)}));
260
261 // Replace old slice operation with new fused_tconv with merged pad values
262 replace(slice).with(fused_tconv);
263
264 return true;
265}

272bool FuseSliceWithTConvPass::run(loco::Graph *g)
273{
274 bool changed = false;
275 for (auto node : loco::active_nodes(loco::output_nodes(g)))
276 {
277 auto slice = dynamic_cast<luci::CircleSlice *>(node);
278 if (not slice)
279 continue;
280
281 if (fuse_slice_with_tconv(slice))
282 changed = true;
283 }
284
285 return changed;
286}
