68std::string opname(const loco::Node *node)
69{
70 if (node->dialect() == loco::CanonicalDialect::get())
71 {
72 auto canonical_node = loco::must_cast<const loco::CanonicalNode *>(node);
73
74 switch (canonical_node->opcode())
75 {
76#define CANONICAL_NODE(OPCODE, CLASS) \
77 case loco::CanonicalOpcode::OPCODE: \
78 return "canonical." #OPCODE;
79#include "loco/IR/CanonicalNodes.lst"
80#undef CANONICAL_NODE
81 default:
82 break;
83 };
84
85 return "canonical."
86 "Invalid";
87 }
88
89 return "unknown."
90 "Unknown";
91}

114NodeDesc default_node_desc(const SymbolTable &tbl, const loco::Node *node)
115{
116 NodeDesc res{opname(node)};
117
118 for (uint32_t n = 0; n < node->arity(); ++n)
119 {
120 res.arg(std::string{"arg"} + std::to_string(n), tbl.lookup(node->arg(n)));
121 }
122 res.state(NodeDesc::State::PartiallyKnown);
123
124 return res;
125}

279bool CanonicalNodeSummaryBuilder::build(const loco::Node *node, locop::NodeSummary &out) const
280{
281 // Skip if a given node does not belong to loco.canonical
282 if (node->dialect() != loco::CanonicalDialect::get())
283 {
284 return false;
285 }
286
287 auto canonical_node = loco::must_cast<const loco::CanonicalNode *>(node);
288 out = canonical_node_desc(*_tbl, canonical_node);
289 return true;
290}
