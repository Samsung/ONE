47void RuntimeGraph::buildAllocDeallocPlan(bool dealloc_input)
48{
49 assert(_reader->get_current_subgraph_index() == _subgraph_index);
50 invalidate();
51 using Lifetime = std::pair<int32_t, int32_t>;
52 std::map<const circle::Tensor *, Lifetime> lifetimes;
53 const size_t num_kernels = _reader->operators().size();
54
55 if (dealloc_input)
56 {
57 for (const auto input_ind : _reader->inputs())
58 {
59 const auto raw_tensor = _reader->tensors()[input_ind];
60
61 assert(lifetimes.count(raw_tensor) == 0);
62 lifetimes[raw_tensor] = Lifetime(-1, 0);
63 }
64 }
65
66 for (int32_t index = 0; index < num_kernels; ++index)
67 {
68 const auto kernel = _reader->operators().at(index);
69 assert(kernel != nullptr);
70
71 for (int32_t j = 0; j < kernel->inputs()->size(); ++j)
72 {
73 const auto input_index = kernel->inputs()->operator[](j);
74
75 if (input_index == -1)
76 continue;
77
78 const auto raw_tensor = _reader->tensors()[input_index];
79
80 // Pass constant tensors
81 if (Tensor::is_constant_tensor(_reader, raw_tensor))
82 continue;
83
84 if (lifetimes.count(raw_tensor) > 0)
85 {
86 if (_inplace_op_indexes.find(index) != _inplace_op_indexes.end())
87 lifetimes.at(raw_tensor).second = -1;
88 else
89 lifetimes.at(raw_tensor).second = index;
90 }
91 }
92
93 for (int32_t j = 0; j < kernel->outputs()->size(); ++j)
94 {
95 const auto output_index = kernel->outputs()->operator[](j);
96 const auto raw_tensor = _reader->tensors()[output_index];
97
98 assert(lifetimes.count(raw_tensor) == 0);
99 if (_inplace_op_indexes.find(index) != _inplace_op_indexes.end())
100 lifetimes[raw_tensor] = Lifetime(-1, index);
101 else
102 lifetimes[raw_tensor] = Lifetime(index, index);
103 }
104 }
105
106 for (const auto output_ind : _reader->outputs())
107 {
108 const auto raw_tensor = _reader->tensors()[output_ind];
109
110 if (lifetimes.count(raw_tensor) > 0)
111 lifetimes.at(raw_tensor).second = num_kernels;
112 }
113
114 _alloc_plan.assign(num_kernels, std::vector<const circle::Tensor *>());
115 _dealloc_plan.assign(num_kernels + 1, std::vector<const circle::Tensor *>());
116 for (const auto &item : lifetimes)
117 {
118 if (item.second.first != -1)
119 _alloc_plan[item.second.first].push_back(item.first);
120 if (item.second.second != -1)
121 _dealloc_plan[item.second.second].push_back(item.first);
122 }
123 _is_valid = true;
124}

126void RuntimeGraph::allocate(size_t kernel_index)
127{
128 assert(_reader->get_current_subgraph_index() == _subgraph_index);
129 assert(_is_valid && kernel_index < _alloc_plan.size());
130 for (const circle::Tensor *tensor : _alloc_plan[kernel_index])
131 {
132 if (_tensor_to_data.find(tensor) != _tensor_to_data.end())
133 {
134 auto *data = _tensor_to_data.at(tensor);
135 _memory_manager->release_memory(data);
136 }
137 auto *data = _memory_manager->allocate_memory(tensor);
138 _tensor_to_data[tensor] = data;
139 }
140}

199void RuntimeGraph::resetOutputTensorsData()
200{
201 assert(_reader->get_current_subgraph_index() == _subgraph_index);
202 const auto graph_inputs = _reader->inputs();
203 for (int i = 0; i < _reader->outputs().size(); ++i)
204 {
205 const auto tensor_index = _reader->outputs()[i];
206 assert(tensor_index != -1);
207
208 if (std::find(graph_inputs.begin(), graph_inputs.end(), tensor_index) != graph_inputs.end())
209 return;
210
211 const auto tensor = _reader->tensors()[tensor_index];
212 assert(tensor != nullptr);
213
214 auto tensor_it = _tensor_to_data.find(tensor);
215 if (tensor_it != _tensor_to_data.end())
216 {
217 auto *data = _tensor_to_data.at(tensor);
218 _memory_manager->release_memory(data);
219 _tensor_to_data.erase(tensor_it);
220 }
221 }

240{

344void RuntimeGraph::makeInplaceOperation(const circle::Tensor *removing_tensor,
345 const circle::Tensor *dst_tensor)
346{
347 assert(_reader->get_current_subgraph_index() == _subgraph_index);
348 assert(removing_tensor != nullptr);
349
350 auto src_it = _tensor_to_data.find(removing_tensor);
351
352 if (src_it == _tensor_to_data.end())
353 return;
354
355 auto *data = _tensor_to_data[removing_tensor];
356
357 _tensor_to_data.erase(src_it);
358
359 if (dst_tensor == nullptr)
360 {
361 delete[] data;
362 return;
363 }
364
365 assert(_tensor_to_data.find(dst_tensor) == _tensor_to_data.end() &&
366 "Failed makeInplaceOperation");
367 _tensor_to_data[dst_tensor] = data;
368}

392void RuntimeGraph::configure(bool dealloc_input)
393{
394 selectOwnSubgraph();
395
396 KernelConfigureRegistry kernel_configure;
397
398 for (uint32_t i = 0; i < _reader->operators().size(); ++i)
399 {
400 const auto op = _reader->operators().at(i);
401 assert(op != nullptr);
402
403 const auto opcode = _reader->builtin_code(op);
404
405 kernel_configure.configure_kernel(op, opcode, this);
406 }
407
408 if (not _is_valid)
409 buildAllocDeallocPlan(dealloc_input);
410
411 _is_valid = true;
412}

419void RuntimeGraph::execute()
420{
421 selectOwnSubgraph();
422
423 if (not _is_valid)
424 configure(true);
425
426 KernelExecuteRegistry kernel_executor;
427
428 for (uint32_t i = 0; i < _reader->operators().size(); ++i)
429 {
430 const auto op = _reader->operators().at(i);
431 assert(op != nullptr);
432
433 const auto opcode = _reader->builtin_code(op);
434
435 allocate(i);
436
437 bool is_inplace = false;
438
439 if (_inplace_op_indexes.find(i) != _inplace_op_indexes.end())
440 is_inplace = true;
441
442 kernel_executor.execute_kernel(op, opcode, this, is_inplace);
443
444 deallocate(i);
445 }
