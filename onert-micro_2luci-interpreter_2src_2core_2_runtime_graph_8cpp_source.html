87void RuntimeGraph::buildAllocDeallocPlan()
88{
89 invalidate();
90 using Lifetime = std::pair<size_t, size_t>;
91 std::map<Tensor *, Lifetime> lifetimes;
92 const size_t num_kernels = _kernels.size();
93 for (size_t index = 0; index < num_kernels; ++index)
94 {
95 const auto &kernel = _kernels[index];
96 for (const Tensor *tensor : kernel->getInputTensors())
97 {
98 auto nc_tensor = const_cast<Tensor *>(tensor);
99 if (lifetimes.count(nc_tensor) > 0)
100 {
101 if (kernel->getInplaceValue())
102 lifetimes.at(nc_tensor).second = -1;
103 else
104 lifetimes.at(nc_tensor).second = index;
105 }
106 }
107 for (Tensor *tensor : kernel->getOutputTensors())
108 {
109 assert(lifetimes.count(tensor) == 0);
110 if (kernel->getInplaceValue())
111 lifetimes[tensor] = Lifetime(-1, index);
112 else
113 lifetimes[tensor] = Lifetime(index, index);
114 }
115 }
116 for (const Tensor *tensor : getOutputTensors())
117 {
118 auto nc_tensor = const_cast<Tensor *>(tensor);
119 if (lifetimes.count(nc_tensor) > 0)
120 lifetimes.at(nc_tensor).second = num_kernels;
121 }
122 _alloc_plan.assign(num_kernels, std::vector<Tensor *>());
123 _dealloc_plan.assign(num_kernels + 1, std::vector<Tensor *>());
124 for (const auto &item : lifetimes)
125 {
126 if (item.second.first != -1)
127 _alloc_plan[item.second.first].push_back(item.first);
128 if (item.second.second != -1)
129 _dealloc_plan[item.second.second].push_back(item.first);
130 }
131 _is_valid = true;
132}

165void RuntimeGraph::execute()
166{
167 if (not _is_valid)
168 configure();
169
170 for (size_t index = 0; index < _kernels.size(); ++index)
171 {
172 const auto &kernel = _kernels[index];
173
174 // TODO: add kernel->configure for methods with dynamic shapes
175
176 // Preallocate outputs in advance instead of relying on automatic allocation
177 allocate(index);
178
179 kernel->execute();
180
181 deallocate(index);
182 }

199void StaticRuntimeGraph::configure()
200{
201 // Allocate memory for intermediate computing buffer and for output buffer.
202
203 _memory_manager->allocate_computing_buf();
204 _memory_manager->allocate_output_buf();
205
206 // Set tensor's data pointer for intermediate tensors
207 for (auto &kernel : _kernels)
208 {
209 const auto output_tensors = kernel->getOutputTensors();
210
211 for (auto tensor : output_tensors)
212 _memory_manager->allocate_memory(*tensor);
213 }
214
215 // Set tensor's data pointer for output tensors
216 for (const auto output_tensor : _output_tensors)
217 _memory_manager->allocate_memory_for_output(*output_tensor);
218
219 _is_valid = true;
220}
