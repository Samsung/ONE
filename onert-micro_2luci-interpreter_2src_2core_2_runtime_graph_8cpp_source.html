81void RuntimeGraph::buildAllocDeallocPlan()
82{
83 invalidate();
84 using Lifetime = std::pair<size_t, size_t>;
85 std::map<Tensor *, Lifetime> lifetimes;
86 const size_t num_kernels = _kernels.size();
87 for (size_t index = 0; index < num_kernels; ++index)
88 {
89 const auto &kernel = _kernels[index];
90 for (const Tensor *tensor : kernel->getInputTensors())
91 {
92 auto nc_tensor = const_cast<Tensor *>(tensor);
93 if (lifetimes.count(nc_tensor) > 0)
94 {
95 if (kernel->getInplaceValue())
96 lifetimes.at(nc_tensor).second = -1;
97 else
98 lifetimes.at(nc_tensor).second = index;
99 }
100 }
101 for (Tensor *tensor : kernel->getOutputTensors())
102 {
103 assert(lifetimes.count(tensor) == 0);
104 if (kernel->getInplaceValue())
105 lifetimes[tensor] = Lifetime(-1, index);
106 else
107 lifetimes[tensor] = Lifetime(index, index);
108 }
109 }
110 for (const Tensor *tensor : getOutputTensors())
111 {
112 auto nc_tensor = const_cast<Tensor *>(tensor);
113 if (lifetimes.count(nc_tensor) > 0)
114 lifetimes.at(nc_tensor).second = num_kernels;
115 }
116 _alloc_plan.assign(num_kernels, std::vector<Tensor *>());
117 _dealloc_plan.assign(num_kernels + 1, std::vector<Tensor *>());
118 for (const auto &item : lifetimes)
119 {
120 if (item.second.first != -1)
121 _alloc_plan[item.second.first].push_back(item.first);
122 if (item.second.second != -1)
123 _dealloc_plan[item.second.second].push_back(item.first);
124 }
125 _is_valid = true;
126}

193void StaticRuntimeGraph::configure()
194{
195 // Allocate memory for intermediate computing buffer and for output buffer.
196
197 _memory_manager->allocate_computing_buf();
198 _memory_manager->allocate_output_buf();
199
200 // Set tensor's data pointer for intermediate tensors
201 for (auto &kernel : _kernels)
202 {
203 const auto output_tensors = kernel->getOutputTensors();
204
205 for (auto tensor : output_tensors)
206 _memory_manager->allocate_memory(*tensor);
207 }
208
209 // Set tensor's data pointer for output tensors
210 for (const auto output_tensor : _output_tensors)
211 _memory_manager->allocate_memory_for_output(*output_tensor);
212
213 _is_valid = true;
214}
