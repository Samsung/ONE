31inline void DepthwiseConvPerChannel(const DepthwiseConvParams &params,
32 const int32_t *output_multiplier, const int32_t *output_shift,
33 const Shape &input_shape, const uint8_t *input_data,
34 const Shape &filter_shape, const uint8_t *filter_data,
35 const int32_t *filter_zeropoint, const Shape &bias_shape,
36 const int32_t *bias_data, const Shape &output_shape,
37 uint8_t *output_data)
38{
39 // Get parameters.
40 // TODO(b/141565753): Re-introduce ScopedProfilingLabel on Micro.
41 const int stride_width = params.stride_width;
42 const int stride_height = params.stride_height;
43 const int dilation_width_factor = params.dilation_width_factor;
44 const int dilation_height_factor = params.dilation_height_factor;
45 const int pad_width = params.padding_values.width;
46 const int pad_height = params.padding_values.height;
47 const int depth_multiplier = params.depth_multiplier;
48 const int32_t input_offset = params.input_offset;
49 const int32_t output_offset = params.output_offset;
50 const int32_t output_activation_min = params.quantized_activation_min;
51 const int32_t output_activation_max = params.quantized_activation_max;
52
53 // Check dimensions of the tensors.
54 assert(input_shape.DimensionsCount() == 4);
55 assert(filter_shape.DimensionsCount() == 4);
56 assert(output_shape.DimensionsCount() == 4);
57
58 assert(output_activation_min <= output_activation_max);
59 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
60 const int output_depth = MatchingDim(filter_shape, 3, output_shape, 3);
61 const int input_height = input_shape.Dims(1);
62 const int input_width = input_shape.Dims(2);
63 const int input_depth = input_shape.Dims(3);
64 const int filter_height = filter_shape.Dims(1);
65 const int filter_width = filter_shape.Dims(2);
66 const int output_height = output_shape.Dims(1);
67 const int output_width = output_shape.Dims(2);
68 UNUSED_RELEASE(output_depth);
69 UNUSED_RELEASE(bias_shape);
70 assert(output_depth == input_depth * depth_multiplier);
71 assert(bias_shape.FlatSize() == output_depth);
72
73 for (int batch = 0; batch < batches; ++batch)
74 {
75 for (int out_y = 0; out_y < output_height; ++out_y)
76 {
77 for (int out_x = 0; out_x < output_width; ++out_x)
78 {
79 for (int in_channel = 0; in_channel < input_depth; ++in_channel)
80 {
81 for (int m = 0; m < depth_multiplier; ++m)
82 {
83 const int output_channel = m + in_channel * depth_multiplier;
84 const int in_x_origin = (out_x * stride_width) - pad_width;
85 const int in_y_origin = (out_y * stride_height) - pad_height;
86 int32_t acc = 0;
87 for (int filter_y = 0; filter_y < filter_height; ++filter_y)
88 {
89 for (int filter_x = 0; filter_x < filter_width; ++filter_x)
90 {
91 const int in_x = in_x_origin + dilation_width_factor * filter_x;
92 const int in_y = in_y_origin + dilation_height_factor * filter_y;
93 // Zero padding by omitting the areas outside the image.
94 const bool is_point_inside_image =
95 (in_x >= 0) && (in_x < input_width) && (in_y >= 0) && (in_y < input_height);
96 if (is_point_inside_image)
97 {
98 uint8_t input_val =
99 input_data[Offset(input_shape, batch, in_y, in_x, in_channel)];
100 uint8_t filter_val =
101 filter_data[Offset(filter_shape, 0, filter_y, filter_x, output_channel)];
102
103 // { for per-channel
104 // NOTE: The following comment is copied from tflite int8 implementation
105 // It may not be 100% true for uint8 per-channel.
106 //
107 // Accumulate with 32 bits accumulator.
108 // In the nudging process during model quantization, we force
109 // real value of 0.0 be represented by a quantized value. This
110 // guarantees that the input_offset is a int8, even though it
111 // is represented using int32_t.
112 // int32 += int8 * (int8 - int8) so the highest value we can
113 // get from each accumulation is [-127, 127] * ([-128, 127] -
114 // [-128, 127]), which is [-32512, 32512]. log2(32512)
115 // = 14.98, which means we can accumulate at least 2^16
116 // multiplications without overflow. The accumulator is
117 // applied to a filter so the accumulation logic will hold as
118 // long as the filter size (filter_y * filter_x * in_channel)
119 // does not exceed 2^16, which is the case in all the models
120 // we have seen so far.
121 // TODO(jianlijianli): Add a check to make sure the
122 // accumulator depth is smaller than 2^16.
123 const int32_t filter_offset = -filter_zeropoint[output_channel];
124 acc += (filter_val + filter_offset) * (input_val + input_offset);
125 // } for per-channel
126 }
127 }
128 }
129 if (bias_data)
130 {
131 acc += bias_data[output_channel];
132 }
133 acc = MultiplyByQuantizedMultiplier(acc, output_multiplier[output_channel],
134 output_shift[output_channel]);
135 acc += output_offset;
136 acc = std::max(acc, output_activation_min);
137 acc = std::min(acc, output_activation_max);
138 // For q8u per-channel, int8_t -> uint8_t
139 output_data[Offset(output_shape, batch, out_y, out_x, output_channel)] =
140 static_cast<uint8_t>(acc);
141 }
142 }
143 }
144 }
145 }
146}
