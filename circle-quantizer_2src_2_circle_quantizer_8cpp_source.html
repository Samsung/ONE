40std::vector<std::shared_ptr<LayerParam>> read_layer_params(std::string &filename)
41{
42 Json::Value root;
43 std::ifstream ifs(filename);
44
45 // Failed to open cfg file
46 if (not ifs.is_open())
47 throw std::runtime_error("Cannot open config file. " + filename);
48
49 Json::CharReaderBuilder builder;
50 JSONCPP_STRING errs;
51
52 // Failed to parse
53 if (not parseFromStream(builder, ifs, &root, &errs))
54 throw std::runtime_error("Cannot parse config file (json format). " + errs);
55
56 auto layers = root["layers"];
57 std::vector<std::shared_ptr<LayerParam>> p;
58 for (auto layer : layers)
59 {
60 if (layer.isMember("name"))
61 {
62 auto l = std::make_shared<LayerParam>();
63 {
64 l->name = layer["name"].asString();
65 l->dtype = layer["dtype"].asString();
66 l->granularity = layer["granularity"].asString();
67 }
68 p.emplace_back(l);
69 }
70
71 // Multiple names with the same dtype & granularity
72 if (layer.isMember("names"))
73 {
74 for (auto name : layer["names"])
75 {
76 auto l = std::make_shared<LayerParam>();
77 {
78 l->name = name.asString();
79 l->dtype = layer["dtype"].asString();
80 l->granularity = layer["granularity"].asString();
81 }
82 p.emplace_back(l);
83 }
84 }
85 }
86
87 return p;
88}

105int entry(int argc, char **argv)
106{
107 luci::CircleQuantizer quantizer;
108
109 auto options = quantizer.options();
110 auto settings = luci::UserSettings::settings();
111
112 const std::string qdqw = "--quantize_dequantize_weights";
113 const std::string qwmm = "--quantize_with_minmax";
114 const std::string rq = "--requantize";
115 const std::string fq = "--force_quantparam";
116 const std::string cq = "--copy_quantparam";
117 const std::string fake_quant = "--fake_quantize";
118 const std::string cfg = "--config";
119
120 const std::string tf_maxpool = "--TF-style_maxpool";
121
122 const std::string gpd = "--generate_profile_data";
123
124 arser::Arser arser("circle-quantizer provides circle model quantization");
125
126 arser::Helper::add_version(arser, print_version);
127 arser::Helper::add_verbose(arser);
128
129 arser.add_argument(qdqw)
130 .nargs(3)
131 .type(arser::DataType::STR_VEC)
132 .help("Quantize-dequantize weight values required action before quantization. "
133 "Three arguments required: input_model_dtype(float32) "
134 "output_model_dtype(uint8) granularity(layer, channel)");
135
136 arser.add_argument(qwmm)
137 .nargs(3)
138 .type(arser::DataType::STR_VEC)
139 .help("Quantize with min/max values. "
140 "Three arguments required: input_model_dtype(float32) "
141 "output_model_dtype(uint8) granularity(layer, channel)");
142
143 arser.add_argument(tf_maxpool)
144 .nargs(0)
145 .default_value(false)
146 .help("Force MaxPool Op to have the same input/output quantparams. NOTE: This feature can "
147 "degrade accuracy of some models");
148
149 arser.add_argument(fake_quant)
150 .nargs(0)
151 .help("Convert a quantized model to a fake-quantized model. NOTE: This feature will "
152 "generate an fp32 model.");
153
154 arser.add_argument(rq)
155 .nargs(2)
156 .type(arser::DataType::STR_VEC)
157 .help("Requantize a quantized model. "
158 "Two arguments required: input_model_dtype(int8) "
159 "output_model_dtype(uint8)");
160
161 arser.add_argument(fq)
162 .nargs(3)
163 .type(arser::DataType::STR_VEC)
164 .accumulated(true)
165 .help("Write quantization parameters to the specified tensor. "
166 "Three arguments required: tensor_name(string), "
167 "scale(float) zero_point(int)");
168
169 arser.add_argument(cq)
170 .nargs(2)
171 .type(arser::DataType::STR_VEC)
172 .accumulated(true)
173 .help("Copy quantization parameter from a tensor to another tensor."
174 "Two arguments required: source_tensor_name(string), "
175 "destination_tensor_name(string)");
176
177 arser.add_argument("--input_type")
178 .help("Input type of quantized model (uint8, int16, or float32)");
179
180 arser.add_argument("--output_type")
181 .help("Output type of quantized model (uint8, int16, or float32)");
182
183 arser.add_argument(cfg).help("Path to the quantization configuration file");
184
185 arser.add_argument("input").help("Input circle model");
186 arser.add_argument("output").help("Output circle model");
187
188 arser.add_argument(gpd).nargs(0).required(false).default_value(false).help(
189 "This will turn on profiling data generation.");
190
191 try
192 {
193 arser.parse(argc, argv);
194 }
195 catch (const std::runtime_error &err)
196 {
197 std::cerr << err.what() << std::endl;
198 std::cout << arser;
199 return 255;
200 }
201
202 {
203 // only one of qdqw, qwmm, rq, fq, cq, fake_quant option can be used
204 int32_t opt_used = arser[qdqw] ? 1 : 0;
205 opt_used += arser[qwmm] ? 1 : 0;
206 opt_used += arser[rq] ? 1 : 0;
207 opt_used += arser[fq] ? 1 : 0;
208 opt_used += arser[cq] ? 1 : 0;
209 opt_used += arser[fake_quant] ? 1 : 0;
210 if (opt_used != 1)
211 {
212 print_exclusive_options();
213 return 255;
214 }
215 }
216
217 if (arser.get<bool>("--verbose"))
218 {
219 // The third parameter of setenv means REPLACE.
220 // If REPLACE is zero, it does not overwrite an existing value.
221 setenv("LUCI_LOG", "100", 0);
222 }
223
224 if (arser[qdqw])
225 {
226 auto values = arser.get<std::vector<std::string>>(qdqw);
227 if (values.size() != 3)
228 {
229 std::cerr << arser;
230 return 255;
231 }
232 options->enable(Algorithms::QuantizeDequantizeWeights);
233
234 options->param(AlgorithmParameters::Quantize_input_model_dtype, values.at(0));
235 options->param(AlgorithmParameters::Quantize_output_model_dtype, values.at(1));
236 options->param(AlgorithmParameters::Quantize_granularity, values.at(2));
237
238 if (arser[cfg])
239 {
240 auto filename = arser.get<std::string>(cfg);
241 try
242 {
243 auto layer_params = read_layer_params(filename);
244
245 options->layer_params(AlgorithmParameters::Quantize_layer_params, layer_params);
246 }
247 catch (const std::runtime_error &e)
248 {
249 std::cerr << e.what() << '\n';
250 return 255;
251 }
252 }
253 }
254
255 if (arser[qwmm])
256 {
257 auto values = arser.get<std::vector<std::string>>(qwmm);
258 if (values.size() != 3)
259 {
260 std::cerr << arser;
261 return 255;
262 }
263 options->enable(Algorithms::QuantizeWithMinMax);
264
265 options->param(AlgorithmParameters::Quantize_input_model_dtype, values.at(0));
266 options->param(AlgorithmParameters::Quantize_output_model_dtype, values.at(1));
267 options->param(AlgorithmParameters::Quantize_granularity, values.at(2));
268
269 if (arser["--input_type"])
270 options->param(AlgorithmParameters::Quantize_input_type,
271 arser.get<std::string>("--input_type"));
272
273 if (arser["--output_type"])
274 options->param(AlgorithmParameters::Quantize_output_type,
275 arser.get<std::string>("--output_type"));
276
277 if (arser[tf_maxpool] and arser.get<bool>(tf_maxpool))
278 options->param(AlgorithmParameters::Quantize_TF_style_maxpool, "True");
279
280 if (arser[cfg])
281 {
282 auto filename = arser.get<std::string>(cfg);
283 try
284 {
285 auto layer_params = read_layer_params(filename);
286
287 options->layer_params(AlgorithmParameters::Quantize_layer_params, layer_params);
288 }
289 catch (const std::runtime_error &e)
290 {
291 std::cerr << e.what() << '\n';
292 return 255;
293 }
294 }
295 }
296
297 if (arser[rq])
298 {
299 auto values = arser.get<std::vector<std::string>>(rq);
300 if (values.size() != 2)
301 {
302 std::cerr << arser;
303 return 255;
304 }
305 options->enable(Algorithms::Requantize);
306
307 options->param(AlgorithmParameters::Quantize_input_model_dtype, values.at(0));
308 options->param(AlgorithmParameters::Quantize_output_model_dtype, values.at(1));
309 }
310
311 if (arser[fq])
312 {
313 auto values = arser.get<std::vector<std::vector<std::string>>>(fq);
314
315 std::vector<std::string> tensors;
316 std::vector<std::string> scales;
317 std::vector<std::string> zero_points;
318
319 for (auto const value : values)
320 {
321 if (value.size() != 3)
322 {
323 std::cerr << arser;
324 return 255;
325 }
326
327 tensors.push_back(value[0]);
328 scales.push_back(value[1]);
329 zero_points.push_back(value[2]);
330 }
331
332 options->enable(Algorithms::ForceQuantParam);
333
334 options->params(AlgorithmParameters::Quantize_tensor_names, tensors);
335 options->params(AlgorithmParameters::Quantize_scales, scales);
336 options->params(AlgorithmParameters::Quantize_zero_points, zero_points);
337 }
338
339 if (arser[cq])
340 {
341 auto values = arser.get<std::vector<std::vector<std::string>>>(cq);
342
343 std::vector<std::string> src;
344 std::vector<std::string> dst;
345
346 for (auto const value : values)
347 {
348 if (value.size() != 2)
349 {
350 std::cerr << arser;
351 return 255;
352 }
353
354 src.push_back(value[0]);
355 dst.push_back(value[1]);
356 }
357
358 options->enable(Algorithms::CopyQuantParam);
359
360 options->params(AlgorithmParameters::Quantize_src_tensor_names, src);
361 options->params(AlgorithmParameters::Quantize_dst_tensor_names, dst);
362 }
363
364 if (arser[fake_quant])
365 options->enable(Algorithms::ConvertToFakeQuantizedModel);
366
367 std::string input_path = arser.get<std::string>("input");
368 std::string output_path = arser.get<std::string>("output");
369
370 if (arser[gpd])
371 settings->set(luci::UserSettings::Key::ProfilingDataGen, true);
372
373 // Load model from the file
374 luci::ImporterEx importerex;
375 auto module = importerex.importVerifyModule(input_path);
376 if (module.get() == nullptr)
377 return EXIT_FAILURE;
378
379 for (size_t idx = 0; idx < module->size(); ++idx)
380 {
381 auto graph = module->graph(idx);
382
383 // quantize the graph
384 quantizer.quantize(graph);
385
386 if (!luci::validate(graph))
387 {
388 std::cerr << "ERROR: Quantized graph is invalid" << std::endl;
389 return 255;
390 }
391 }
392
393 // Export to output Circle file
394 luci::CircleExporter exporter;
395
396 luci::CircleFileExpContract contract(module.get(), output_path);
397
398 if (!exporter.invoke(&contract))
399 {
400 std::cerr << "ERROR: Failed to export '" << output_path << "'" << std::endl;
401 return 255;
402 }
403
404 return 0;
405}
