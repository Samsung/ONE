28TEST(TRANSFORMATIONS, Switcher_Conv2D_NCHW2NHWC)
29{
30 mir::Graph g;
31 mir::TensorType input_type{mir::DataType::FLOAT32, {1, 3, 299, 299}};
32 auto *input = g.create<mir::ops::InputOp>(input_type);
33
34 mir::TensorType kernel_type{mir::DataType::FLOAT32, {3, 32, 3, 3}};
35 auto *kernel = g.create<mir::ops::InputOp>(kernel_type);
36 // Conv2DOp
37 mir::Conv2DOpAttributes attributes;
38 attributes.strides = {2, 5};
39 attributes.padding_before = {8, 1};
40 attributes.padding_after = {7, 9};
41 attributes.data_format = mir::DataFormat::NCHW;
42 auto *conv = g.create<mir::ops::Conv2DOp>(input->getOutput(0), kernel->getOutput(0), attributes);
43
44 auto *output = g.create<mir::ops::OutputOp>(conv->getOutput(0));
45
46 nnc::DataFormatSwitcher switcher(mir::DataFormat::NHWC);
47
48 switcher.run(&g);
49
50 auto *trans_out = output->getInput(0)->getNode();
51 auto *conv_ = trans_out->getInput(0)->getNode();
52 auto *trans_in = conv_->getInput(0)->getNode();
53 auto *input_ = trans_in->getInput(0)->getNode();
54
55 ASSERT_EQ(trans_out->getType(), mir::Operation::Type::transpose);
56 ASSERT_NE(conv_, conv);
57 ASSERT_EQ(trans_in->getType(), mir::Operation::Type::transpose);
58 ASSERT_EQ(input_, input);
59
60 auto &in_axis_order = dynamic_cast<mir::ops::TransposeOp *>(trans_in)->getAxisOrder();
61 auto &out_axis_order = dynamic_cast<mir::ops::TransposeOp *>(trans_out)->getAxisOrder();
62
63 ASSERT_EQ(in_axis_order.size(), 4);
64 ASSERT_EQ(in_axis_order, std::vector<size_t>({0, 2, 3, 1}));
65
66 ASSERT_EQ(out_axis_order.size(), 4);
67 ASSERT_EQ(out_axis_order, std::vector<size_t>({0, 3, 1, 2}));
68 // Check Conv2D params
69 auto *nhwc_conv = dynamic_cast<mir::ops::Conv2DOp *>(conv_);
70 ASSERT_EQ(nhwc_conv->getDataFormat(), mir::DataFormat::NHWC);
71 ASSERT_EQ(nhwc_conv->getStrides(), std::vector<int32_t>({2, 5}));
72 ASSERT_EQ(nhwc_conv->getPaddingBefore(), std::vector<int32_t>({8, 1}));
73 ASSERT_EQ(nhwc_conv->getPaddingAfter(), std::vector<int32_t>({7, 9}));
74}
