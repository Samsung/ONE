31template <class CIRCLENODE>
32loco::TensorShape use_paddings(const CIRCLENODE *node, const luci::CircleConst *paddings)
33{
34 const loco::DataType S32 = loco::DataType::S32;
35 const loco::DataType S64 = loco::DataType::S64;
36
37 auto input_shape = luci::shape_get(node->input()).template as<loco::TensorShape>();
38 // TODO support other data type
39 LUCI_ASSERT(paddings->dtype() == S32 || paddings->dtype() == S64, "Support int 32/64 for now");
40 LUCI_ASSERT(paddings->rank() == 2, "paddings should be rank 2");
41
42 int32_t n = paddings->dim(0).value();
43 int32_t v = paddings->dim(1).value();
44
45 LUCI_ASSERT(v == 2, "paddings should be [n, 2]");
46 LUCI_ASSERT(n == int32_t(input_shape.rank()),
47 "paddings [n, 2] should have same value of input rank");
48
49 loco::TensorShape output_shape;
50
51 output_shape.rank(input_shape.rank());
52 for (int32_t ni = 0; ni < n; ++ni)
53 {
54 if (not input_shape.dim(ni).known())
55 {
56 output_shape.dim(ni).unset();
57 continue;
58 }
59 int32_t idx = ni * 2;
60 int value = input_shape.dim(ni).value();
61 if (paddings->dtype() == S32)
62 {
63 value += paddings->at<S32>(idx + 0); // left
64 value += paddings->at<S32>(idx + 1); // right
65 }
66 else
67 {
68 auto pl = paddings->at<S64>(idx + 0);
69 auto pr = paddings->at<S64>(idx + 1);
70 auto max = static_cast<int64_t>(std::numeric_limits<int32_t>::max());
71 auto low = static_cast<int64_t>(std::numeric_limits<int32_t>::lowest());
72 LUCI_ASSERT(pl <= max, "paddings is over 32 bit limit");
73 LUCI_ASSERT(pl >= low, "paddings is over 32 bit limit");
74 LUCI_ASSERT(pr <= max, "paddings is over 32 bit limit");
75 LUCI_ASSERT(pr >= low, "paddings is over 32 bit limit");
76 value += static_cast<int32_t>(pl); // left
77 value += static_cast<int32_t>(pr); // right
78 }
79 output_shape.dim(ni) = value;
80 }
81
82 return output_shape;
83}
