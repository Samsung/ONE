59void KernelGenerator::visit(const ir::operation::Bulk &node)
60{
61 using ir::operation::Bulk;
62
63 std::vector<IPortableTensor *> output_tensors;
64 for (auto &ofm_idx : node.getOutputs())
65 output_tensors.emplace_back(_tensor_reg->getPortableTensor(ofm_idx));
66
67 std::vector<const IPortableTensor *> input_tensors;
68 for (auto &ifm_idx : node.getInputs())
69 input_tensors.emplace_back(_tensor_reg->getPortableTensor(ifm_idx));
70
71 // parameters
72 const auto binary_path = node.param().binary_path;
73
74 auto fn = std::make_unique<ops::BulkLayer>();
75
76 fn->configure(input_tensors, output_tensors, binary_path, _dev_context);
77
78 _return_fn = std::move(fn);
79}
