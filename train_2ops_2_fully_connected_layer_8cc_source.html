31std::unique_ptr<backend::train::Tensor>
32createTransposedTensor(const backend::IPortableTensor *origin_tensor)
33{
34 const auto &origin_shape = origin_tensor->getShape();
35 assert(origin_shape.rank() == 2);
36
37 auto transposed_info = origin_tensor->get_info();
38 auto transposed_shape = ir::Shape{origin_shape.dim(1), origin_shape.dim(0)};
39 transposed_info.shape(transposed_shape);
40
41 return std::make_unique<backend::train::Tensor>(transposed_info, origin_tensor->layout());
42}

66void FullyConnectedLayer::configure(const IPortableTensor *input, const IPortableTensor *weights,
67 const IPortableTensor *bias, IPortableTensor *output,
68 IPortableTensor *back_prop_input, IPortableTensor *grad_weights,
69 IPortableTensor *grad_bias,
70 const IPortableTensor *back_prop_output,
71 ir::Activation activation,
72 ir::FullyConnectedWeightsFormat weights_format,
73 const std::shared_ptr<train::ExternalContext> &external_context)
74{
75 cpu::ops::FullyConnectedLayer::configure(input, weights, bias, activation, weights_format, output,
76 external_context);
77
78 _back_prop_input = back_prop_input;
79 _grad_weights = grad_weights;
80 _grad_bias = grad_bias;
81 _back_prop_output = back_prop_output;
82
83 if (weights_format != ir::FullyConnectedWeightsFormat::Default)
84 throw std::runtime_error{
85 "train FullyConnectedLayer: Weight formats other than default are not supported."};
86
87 if (input->get_info().shape().rank() != 2 || weights->get_info().shape().rank() != 2 ||
88 output->get_info().shape().rank() != 2 || back_prop_input->get_info().shape().rank() != 2 ||
89 grad_weights->get_info().shape().rank() != 2 ||
90 back_prop_output->get_info().shape().rank() != 2)
91 throw std::runtime_error{
92 "train FullyConnectedLayer: Input other ranks than 2 are not supported."};
93
94 _transposed_weights = createTransposedTensor(weights);
95 _transposed_weights->setBuffer(std::make_shared<basic::Allocator>(weights->total_size()));
96
97 _transposed_input = createTransposedTensor(input);
98 _transposed_input->setBuffer(std::make_shared<basic::Allocator>(input->total_size()));
99
100 _transposed_back_prop_output = createTransposedTensor(back_prop_output);
101 _transposed_back_prop_output->setBuffer(
102 std::make_shared<basic::Allocator>(back_prop_output->total_size()));
103
104 if (activation != ir::Activation::NONE)
105 {
106 _act_back_prop_output =
107 std::make_unique<Tensor>(_back_prop_output->get_info(), _back_prop_output->layout());
108 _act_back_prop_output->setBuffer(
109 std::make_shared<basic::Allocator>(_back_prop_output->total_size()));
110 }
111}

115void FullyConnectedLayer::backward()
116{
117 const auto data_type = _back_prop_output->data_type();
118 assert(data_type == _input->data_type());
119 switch (data_type)
120 {
121 case OperandType::FLOAT32:
122 {
123 assert(data_type == _grad_weights->data_type());
124 assert(_grad_bias == nullptr || data_type == _grad_bias->data_type());
125 backwardFloat32();
126 break;
127 }
128 default:
129 throw std::runtime_error{"train FullyConnectedLayer: unsupported data type"};
130 }
131}

133void FullyConnectedLayer::backwardFloat32()
134{
135 // Calculate gradient for activation
136 const IPortableTensor *backprop_act;
137 switch (_activation)
138 {
139 case ir::Activation::NONE:
140 backprop_act = _back_prop_output;
141 break;
142 case ir::Activation::RELU:
143 nnfw::cker::train::ReLUGrad(getShape(_output), getBuffer<float>(_output),
144 getShape(_back_prop_output), getBuffer<float>(_back_prop_output),
145 getShape(_act_back_prop_output.get()),
146 getBuffer<float>(_act_back_prop_output.get()));
147 backprop_act = _act_back_prop_output.get();
148 break;
149 default:
150 throw std::runtime_error("train FullyConnectedLayer: Unsupported activation type yet");
151 }
152
153 // Initialize TransposeParams
154 nnfw::cker::TransposeParams transpose_param;
155 transpose_param.perm_count = 2;
156 transpose_param.perm[0] = 1;
157 transpose_param.perm[1] = 0;
158
159 // Initialize FullyConnectedParams
160 nnfw::cker::FullyConnectedParams op_params;
161 float output_activation_min = 0;
162 float output_activation_max = 0;
163 CalculateActivationRange(ir::Activation::NONE, &output_activation_min, &output_activation_max);
164 op_params.activation = nnfw::cker::FusedActivationFunctionType::kNone;
165 op_params.float_activation_min = output_activation_min;
166 op_params.float_activation_max = output_activation_max;
167 op_params.lhs_cacheable = false;
168 op_params.rhs_cacheable = false;
169
170 // Transpose and compute gradient for input
171 // ∂L/∂X = fc(Incoming gradient, transposed W)
172 auto transposed_weights = _transposed_weights.get();
173 assert(transposed_weights->getShape().rank() == 2);
174 nnfw::cker::Transpose(transpose_param, getShape(_weights), getBuffer<float>(_weights),
175 getShape(transposed_weights), getBuffer<float>(transposed_weights));
176
177 nnfw::cker::FullyConnected(op_params, getShape(backprop_act), getBuffer<float>(backprop_act),
178 getShape(transposed_weights), getBuffer<float>(transposed_weights),
179 getShape(nullptr), nullptr, getShape(_back_prop_input),
180 getBuffer<float>(_back_prop_input));
181
182 // Transpose and compute gradient for weights
183 // ∂L/∂W = fc(transposed incomming gradient, transposed X)
184 auto transposed_input = _transposed_input.get();
185 assert(transposed_input->getShape().rank() == 2);
186 nnfw::cker::Transpose(transpose_param, getShape(_input), getBuffer<float>(_input),
187 getShape(transposed_input), getBuffer<float>(transposed_input));
188
189 auto transposed_back_prop_output = _transposed_back_prop_output.get();
190 assert(transposed_back_prop_output->getShape().rank() == 2);
191 nnfw::cker::Transpose(transpose_param, getShape(backprop_act), getBuffer<float>(backprop_act),
192 getShape(transposed_back_prop_output),
193 getBuffer<float>(transposed_back_prop_output));
194
195 nnfw::cker::FullyConnected(
196 op_params, getShape(transposed_back_prop_output), getBuffer<float>(transposed_back_prop_output),
197 getShape(transposed_input), getBuffer<float>(transposed_input), getShape(nullptr), nullptr,
198 getShape(_grad_weights), getBuffer<float>(_grad_weights));
199
200 // Compute gradient for bias
201 if (_bias)
202 {
203 assert(_grad_bias);
204 nnfw::cker::train::FullyConnectedBiasGrad(getShape(backprop_act),
205 getBuffer<float>(backprop_act), getShape(_grad_bias),
206 getBuffer<float>(_grad_bias));
207 }
208}
