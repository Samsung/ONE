54void ConstantInitializer::visit(const ir::operation::SpaceToBatchND &node)
55{
56 const auto &block_size_index = node.getInputs().at(ir::operation::SpaceToBatchND::BLOCK_SIZE);
57 const auto &block_size_obj = _operands.at(block_size_index);
58
59 if (block_size_obj.isConstant())
60 {
61 _init_map[block_size_index] = acl_common::initReverseOrder<int32_t>;
62 }
63
64 const auto &paddings_index = node.getInputs().at(ir::operation::SpaceToBatchND::PADDINGS);
65 const auto &paddings_obj = _operands.at(paddings_index);
66 if (paddings_obj.isConstant())
67 {
68 _init_map[paddings_index] = [](const ir::Operand &model_obj, backend::ITensor &obj) {
69 assert(model_obj.data());
70 const auto &shape = model_obj.shape();
71 const auto base = reinterpret_cast<const int32_t *>(model_obj.data()->base());
72 assert(model_obj.shape().rank() == 2);
73 assert(obj.getShape().dim(0) == 2);
74 obj.access([&](ITensor &tensor) {
75 for (auto i = 0; i < shape.dim(0); ++i)
76 {
77 for (auto j = 0; j < shape.dim(1); ++j)
78 {
79 const int32_t value = base[i * 2 + j];
80 int32_t *into = reinterpret_cast<int32_t *>(
81 tensor.buffer() + tensor.calcOffset({shape.dim(0) - i - 1, j}));
82 *into = value;
83 }
84 }
85 });
86 };
87 }
88}

90void ConstantInitializer::visit(const ir::operation::Reverse &node)
91{
92 const auto &output_index = node.getOutputs().at(0);
93
94 const auto &input_index = node.getInputs().at(ir::operation::Reverse::Input::INPUT);
95 const auto &input_obj = _operands.at(input_index);
96
97 const auto &axis_index = node.getInputs().at(ir::operation::Reverse::Input::AXIS);
98 const auto &axis_obj = _operands.at(axis_index);
99
100 const auto ifm_rank = input_obj.shape().rank();
101 const auto frontend_layout = this->_current_layout;
102
103 auto output_tensor = this->_tensor_reg->getITensor(output_index);
104 const auto backend_layout = output_tensor->layout();
105
106 if (axis_obj.isConstant())
107 {
108 _init_map[axis_index] = [ifm_rank, frontend_layout, backend_layout](const ir::Operand &operand,
109 backend::ITensor &obj) {
110 assert(operand.data());
111
112 const auto axis_value = *(reinterpret_cast<const int32_t *>(operand.data()->base()));
113 int32_t axis_tmp = axis_value;
114 if (axis_tmp < 0)
115 {
116 axis_tmp = axis_tmp + ifm_rank;
117 }
118
119 auto axis =
120 acl_common::ToARMComputeAxis(ifm_rank, axis_tmp, frontend_layout, backend_layout).value();
121
122 obj.access([&](ITensor &tensor) {
123 int32_t *into = reinterpret_cast<int32_t *>(tensor.buffer());
124 *into = (int32_t)axis;
125 });
126 };
127 }
