41public:
42 // Only one of two method 'setBuffer' must be called once
43
47 void setBuffer(uint8_t *buffer) { _buffer = buffer; }

52 void setBuffer(const std::shared_ptr<Allocator> &alloc)

54 _allocator = alloc;

57
61 void deallocBuffer() override;
62
63public:
64 uint8_t *buffer() const override { return _buffer; }
65 void set_dynamic() override { _info.setDynamic(); }
66 bool applyShape(const ir::Shape &new_shape) override;
67
68 virtual void increase_ref()
69 {
70 assert(is_dynamic() ||
71 // when not dynamic
72 (_buffer != nullptr));
73
74 ++_num_references;
75 }
76
77 virtual void decrease_ref()
78 {
79 assert(_buffer != nullptr || _allocator != nullptr);
80 assert(_num_references > 0);
81 --_num_references;
82 // constant tensor and dynamic tensor has _allocator
83 if (_num_references == 0)
84 {
85 if (_buffer != nullptr)
86 _buffer = nullptr;
87 if (_allocator != nullptr)
88 {
89 _allocator->release();
90 _allocator = nullptr;
91 }
92 }
93 }
94
98 virtual void reset_ref()
99 {
100 assert(_buffer != nullptr || _allocator != nullptr);
101 assert(_num_references > 0);
102 _num_references = 0;
103
104 // Only constant tensor has allocator pointer
105 if (_buffer != nullptr)
106 _buffer = nullptr;
107 else
108 {
109 _allocator->release();
110 _allocator = nullptr;
111 }
112 }
113
114 virtual int32_t num_references() { return _num_references; }
115
116 void setShape(const ir::Shape &new_shape) override;
117
118protected:
119 uint8_t *_buffer;
120 size_t _size;
121 int32_t _num_references;
122 DynamicMemoryManager *_dynamic_mem_mgr;
123
124private:
131 std::shared_ptr<Allocator> _allocator;
132};

142class ExternalTensor : public Tensor
143{
144public:
145 ExternalTensor() = delete;
146 virtual ~ExternalTensor();
147
148public:
149 ExternalTensor(const ir::OperandInfo &info) : Tensor(info, nullptr)
150 {
151 assert(_info.isConstant());
152 assert(_info.isDynamic() == false);
153 }
154
155public:
161 void setData(const std::shared_ptr<ir::Data> data)
162 {
163 assert(data != nullptr);
164 _data = data;
165 // Note. Some op such as cker::Conv could take buffer as nullptr.
166 // That's why _buffer also would be used
167 _buffer = const_cast<uint8_t *>(_data->base());
168 }
169
170public:
171 uint8_t *buffer() const override { return _buffer; }
172
173 void set_dynamic() override
174 {
175 throw std::runtime_error("This tensor does not support changing dynamic");
176 }
177
178 void setShape(const ir::Shape &) override
179 {
180 throw std::runtime_error("This tensor does not support changing shape");
181 }
182
183 void increase_ref() override { ++_num_references; }
184
185 void decrease_ref() override
186 {
187 assert(_data != nullptr);
188 assert(_num_references > 0);
189 --_num_references;
190 if (_num_references == 0)
191 {
192 _data.reset();
193 _buffer = nullptr;
194 }
195 }
196
200 void reset_ref() override
201 {
202 assert(_data != nullptr);
203 assert(_num_references > 0);
204 _num_references = 0;
205
206 _data.reset();
207 _buffer = nullptr;
208 }
209
210 int32_t num_references() override { return _num_references; }
211
212private:
213 std::shared_ptr<const ir::Data> _data;
214};
