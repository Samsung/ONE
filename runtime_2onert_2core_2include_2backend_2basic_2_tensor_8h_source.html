169class ExternalTensor : public Tensor

175public:
176 ExternalTensor(const ir::OperandInfo &info, const ir::Layout layout)
177 : Tensor(info, layout, nullptr)

179 assert(_layout == ir::Layout::NHWC);
180 assert(_info.isConstant());
181 assert(_info.isDynamic() == false);
182 }
183
184public:
190 void setData(const std::shared_ptr<ir::Data> data)
191 {
192 assert(data != nullptr);
193 _data = data;
194 // Note. Some op such as cker::Conv could take buffer as nullptr.
195 // That's why _buffer also would be used
196 _buffer = const_cast<uint8_t *>(_data->base());
197 }
198
199public:
200 uint8_t *buffer() const override { return _buffer; }
201
202 bool is_constant() const override { return true; }
203 bool is_dynamic() const override { return false; }
204 void set_dynamic() override
205 {
206 throw std::runtime_error("This tensor does not support changing dynamic");
207 }
208
209 void setShape(const ir::Shape &) override
210 {
211 throw std::runtime_error("This tensor does not support changing shape");
212 }
213
214 void increase_ref() override { ++_num_references; }
215
216 void decrease_ref() override
217 {
218 assert(_data != nullptr);
219 assert(_num_references > 0);
220 --_num_references;
221 if (_num_references == 0)
222 {
223 _data.reset();
224 _buffer = nullptr;
225 }
226 }
227
231 void reset_ref() override
232 {
233 assert(_data != nullptr);
234 assert(_num_references > 0);
235 _num_references = 0;
236
237 _data.reset();
238 _buffer = nullptr;
239 }
240
241 int32_t num_references() override { return _num_references; }
242
243private:
244 std::shared_ptr<const ir::Data> _data;
245};
