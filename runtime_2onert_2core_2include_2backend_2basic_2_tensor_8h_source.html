37public:

40public:
41 // Only one of two method 'setBuffer' must be called once
42
46 void setBuffer(uint8_t *buffer) { _buffer = buffer; }
47
51 void setBuffer(const std::shared_ptr<Allocator> &alloc)
52 {
53 _allocator = alloc;
54 _buffer = alloc->base();
55 }
56
60 void deallocBuffer() override;
61
62public:
63 uint8_t *buffer() const override { return _buffer; }
64 void set_dynamic() override { _info.setDynamic(); }
65 bool applyShape(const ir::Shape &new_shape) override;
66
67 virtual void increase_ref()
68 {
69 assert(is_dynamic() ||
70 // when not dynamic
71 (_buffer != nullptr));
72
73 ++_num_references;
74 }
75
76 virtual void decrease_ref()
77 {
78 assert(_buffer != nullptr || _allocator != nullptr);
79 assert(_num_references > 0);
80 --_num_references;
81 // constant tensor and dynamic tensor has _allocator
82 if (_num_references == 0)
83 {
84 if (_buffer != nullptr)
85 _buffer = nullptr;
86 if (_allocator != nullptr)
87 {
88 _allocator->release();
89 _allocator = nullptr;
90 }
91 }
92 }
93
97 virtual void reset_ref()
98 {
99 assert(_buffer != nullptr || _allocator != nullptr);
100 assert(_num_references > 0);
101 _num_references = 0;
102
103 // Only constant tensor has allocator pointer
104 if (_buffer != nullptr)
105 _buffer = nullptr;
106 else
107 {
108 _allocator->release();
109 _allocator = nullptr;
110 }
111 }
112
113 virtual int32_t num_references() { return _num_references; }
114
115 void setShape(const ir::Shape &new_shape) override;
116
117protected:
118 uint8_t *_buffer;
119 size_t _size;
120 int32_t _num_references;
121 DynamicMemoryManager *_dynamic_mem_mgr;
122
123private:
130 std::shared_ptr<Allocator> _allocator;
131};

141class ExternalTensor : public Tensor
142{
143public:
144 ExternalTensor() = delete;
145 virtual ~ExternalTensor();
146
147public:
148 ExternalTensor(const ir::OperandInfo &info) : Tensor(info, nullptr)
149 {
150 assert(_info.isConstant());
151 assert(_info.isDynamic() == false);
152 }
153
154public:
160 void setData(const std::shared_ptr<ir::Data> data)
161 {
162 assert(data != nullptr);
163 _data = data;
164 // Note. Some op such as cker::Conv could take buffer as nullptr.
165 // That's why _buffer also would be used
166 _buffer = const_cast<uint8_t *>(_data->base());
167 }
168
169public:
170 uint8_t *buffer() const override { return _buffer; }
171
172 void set_dynamic() override
173 {
174 throw std::runtime_error("This tensor does not support changing dynamic");
175 }
176
177 void setShape(const ir::Shape &) override
178 {
179 throw std::runtime_error("This tensor does not support changing shape");
180 }
181
182 void increase_ref() override { ++_num_references; }
183
184 void decrease_ref() override
185 {
186 assert(_data != nullptr);
187 assert(_num_references > 0);
188 --_num_references;
189 if (_num_references == 0)
190 {
191 _data.reset();
192 _buffer = nullptr;
193 }
194 }
195
199 void reset_ref() override
200 {
201 assert(_data != nullptr);
202 assert(_num_references > 0);
203 _num_references = 0;
204
205 _data.reset();
206 _buffer = nullptr;
207 }
208
209 int32_t num_references() override { return _num_references; }
210
211private:
212 std::shared_ptr<const ir::Data> _data;
213};
