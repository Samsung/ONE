63void SingleModelExecutors::execute(const ExecutionContext &ctx)
64{
65 // UserTensor for Input/Output
66 std::vector<std::unique_ptr<backend::builtin::UserTensor>> tensorpool;
67
68 // EdgeTensor for Input Quantization / Output Dequantization
69 std::vector<std::unique_ptr<EdgeTensor>> qtensorpool;
70
71 // Input/Output Tensor vector for executor
72 std::vector<backend::IPortableTensor *> inputs(ctx.desc.inputs.size());
73 std::vector<backend::IPortableTensor *> outputs(ctx.desc.outputs.size());
74
75 // Vector for input quantization I/O
76 std::vector<backend::ITensor *> input_tensors;
77 std::vector<backend::ITensor *> input_qtensors;
78 std::vector<ir::PermuteType> input_permute_types;
79
80 // Vector for output dequantization I/O
81 std::vector<backend::ITensor *> output_qtensors;
82 std::vector<backend::ITensor *> output_tensors;
83 std::vector<ir::PermuteType> output_permute_types;
84
85 // Prepare UserTensor and EdgeTensor for input quantization
86 for (uint32_t i = 0; i < inputs.size(); i++)
87 {
88 auto &desc = ctx.desc.inputs[i];
89
90 // Input is optional if buffer is nullptr, and optional input's size is 0
91 if (desc->buffer == nullptr && (desc->size != 0 || desc->info.total_size() != 0))
92 throw std::runtime_error{"Input " + std::to_string(i) + "'s buffer is not set."};
93
94 // TODO: Create UserTensor only that will be set into IOTensor
95 tensorpool.emplace_back(std::make_unique<backend::builtin::UserTensor>(
96 desc->info, desc->layout, const_cast<uint8_t *>(static_cast<const uint8_t *>(desc->buffer)),
97 desc->size));
98
99 auto user_type = desc->info.typeInfo().type();
100 auto &model_info = entryExecutor()->inputInfo(i).typeInfo();
101 auto model_type = model_info.type();
102 if ((user_type != model_type && user_type == ir::DataType::FLOAT32) ||
103 (desc->layout == ir::Layout::NCHW))
104 {
105 auto quantized_info = desc->info;
106 quantized_info.typeInfo(model_info);
107 qtensorpool.emplace_back(
108 std::make_unique<EdgeTensor>(quantized_info, entryExecutor()->inputLayout(i)));
109 qtensorpool.back()->allocate_buffer();
110
111 input_tensors.push_back(tensorpool.back().get());
112 input_qtensors.push_back(qtensorpool.back().get());
113 inputs[i] = qtensorpool.back().get();
114 if (desc->layout == ir::Layout::NCHW)
115 input_permute_types.push_back(ir::PermuteType::NCHW_TO_NHWC);
116 else
117 input_permute_types.push_back(ir::PermuteType::COPY);
118 }
119 else
120 inputs[i] = tensorpool.back().get();
121 }
122
123 // Prepare UserTensor and EdgeTensor for output dequantization
124 for (uint32_t i = 0; i < outputs.size(); i++)
125 {
126 auto &desc = ctx.desc.outputs[i];
127 const auto output_io_tensor =
128 dynamic_cast<const backend::builtin::IOTensor *>(outputTensor(ir::IOIndex{i}));
129 if (!output_io_tensor)
130 throw std::runtime_error{"Output tensor must be IOTensor"};
131 bool skip_set_output = output_io_tensor->hasBackendTensor();
132
133 // Output is optional if buffer is nullptr, and optional output's size is 0
134 if (desc->buffer == nullptr && (desc->size != 0 || desc->info.total_size() != 0) &&
135 !skip_set_output)
136 throw std::runtime_error{"Output " + std::to_string(i) + "'s buffer is not set."};
137
138 tensorpool.emplace_back(std::make_unique<backend::builtin::UserTensor>(
139 desc->info, desc->layout, static_cast<uint8_t *>(desc->buffer), desc->size));
140
141 auto user_type = desc->info.typeInfo().type();
142 auto &model_info = entryExecutor()->outputInfo(i).typeInfo();
143 auto model_type = model_info.type();
144 if ((user_type != model_type && user_type == ir::DataType::FLOAT32) ||
145 (desc->layout == ir::Layout::NCHW))
146 {
147 if (skip_set_output)
148 std::runtime_error("When outputs are allocated internally, backend-aware quantization is "
149 "not yet supported.");
150
151 auto quantized_info = desc->info;
152 quantized_info.typeInfo(model_info);
153 qtensorpool.emplace_back(
154 std::make_unique<EdgeTensor>(quantized_info, entryExecutor()->outputLayout(i)));
155 qtensorpool.back()->allocate_buffer();
156
157 output_qtensors.push_back(qtensorpool.back().get());
158 output_tensors.push_back(tensorpool.back().get());
159 outputs[i] = qtensorpool.back().get();
160 if (desc->layout == ir::Layout::NCHW)
161 output_permute_types.push_back(ir::PermuteType::NHWC_TO_NCHW);
162 else
163 output_permute_types.push_back(ir::PermuteType::COPY);
164 }
165 else
166 outputs[i] = tensorpool.back().get();
167 }
168
169 // Run quantization
170 if (input_tensors.size() > 0)
171 {
172 auto input_quantize_layer = PermuteLayer(input_tensors, input_qtensors, input_permute_types);
173 input_quantize_layer.prepare();
174 input_quantize_layer.run();
175 }
176
177 // Executor
178 entryExecutor()->execute(inputs, outputs, ctx.options);
179
180 // Run dequantization
181 if (output_tensors.size() != 0)
182 {
183 auto output_dequantize_layer =
184 PermuteLayer(output_qtensors, output_tensors, output_permute_types);
185 output_dequantize_layer.prepare();
186 output_dequantize_layer.run();
187 }
188
189 // Get dynamic shape inference result
190 for (uint32_t i = 0; i < outputs.size(); i++)
191 {
192 if (ctx.desc.outputs[i]->buffer == nullptr)
193 {
194 // Output is optional if buffer is nullptr
195 continue;
196 }
197
198 ctx.desc.outputs[i]->info.shape(outputs[i]->getShape());
199 }
200}
