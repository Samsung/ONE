26class EdgeTensor : public onert::backend::IPortableTensor
27{
28public:
29 EdgeTensor(const onert::ir::OperandInfo &info, onert::ir::Layout layout)
30 : onert::backend::IPortableTensor(info), _layout{layout}, _buffer{nullptr}, _ref_count{0}
31 {
32 }
33 ~EdgeTensor() = default;
34
35 uint8_t *buffer() const override { return _buffer.get(); }
36 onert::ir::Layout layout() const override { return _layout; }
37 void set_dynamic() override { _info.setDynamic(); }
38 bool applyShape(const onert::ir::Shape &new_shape) override
39 {
40 bool previously_dynamic = is_dynamic();
41 if (!previously_dynamic || _buffer == nullptr)
42 {
43 // Always set shape - when buffer with same size was already allocated, shape could differ
44 setShape(new_shape);
45 set_dynamic();
46 const auto total_size = get_info().total_size();
47 _buffer = std::make_unique<uint8_t[]>(total_size);
48 }
49 else
50 {
51 auto previous_size = total_size();
52 auto new_size = new_shape.num_elements() * onert::ir::sizeOfDataType(data_type());
53 if (previous_size != new_size)
54 {
55 setShape(new_shape);
56 set_dynamic();
57 const auto total_size = get_info().total_size();
58 _buffer = std::make_unique<uint8_t[]>(total_size);
59 }
60 else
61 { // when buffer with same size was already allocated, shape could differ
62 setShape(new_shape);
63 }
64 }
65 return true;
66 }
67
68 void setShape(const onert::ir::Shape &new_shape) override { _info.shape(new_shape); }
69
70 void allocate_buffer()
71 {
72 const auto total_size = _info.total_size();
73 _buffer = std::make_unique<uint8_t[]>(total_size);
74 _ref_count = 1;
75 }
76
77 void increase_ref() { _ref_count++; }
78
79 void decrease_ref()
80 {
81 assert(_ref_count > 0);
82 _ref_count--;
83 if (_ref_count == 0)
84 {
85 _buffer.reset();
86 }
87 }
88
89private:
90 onert::ir::Layout _layout;
91 std::unique_ptr<uint8_t[]> _buffer;
92 int32_t _ref_count;
93};

142void SingleModelExecutors::execute(const ExecutionContext &ctx)
143{
144 // UserTensor for Input/Output
145 std::vector<std::unique_ptr<backend::builtin::UserTensor>> tensorpool;
146
147 // EdgeTensor for Input Quantization / Output Dequantization
148 std::vector<std::unique_ptr<EdgeTensor>> qtensorpool;
149
150 // Input/Output Tensor vector for executor
151 std::vector<backend::IPortableTensor *> inputs(ctx.desc.inputs.size());
152 std::vector<backend::IPortableTensor *> outputs(ctx.desc.outputs.size());
153
154 // Vector for input quantization I/O
155 std::vector<backend::ITensor *> input_tensors;
156 std::vector<backend::ITensor *> input_qtensors;
157
158 // Vector for output dequantization I/O
159 std::vector<backend::ITensor *> output_qtensors;
160 std::vector<backend::ITensor *> output_tensors;
161
162 // Prepare UserTensor and EdgeTensor for input quantization
163 for (uint32_t i = 0; i < inputs.size(); i++)
164 {
165 auto &desc = ctx.desc.inputs[i];
166
167 // Input is optional if buffer is nullptr, and optional input's size is 0
168 if (desc->buffer == nullptr && (desc->size != 0 || desc->info.total_size() != 0))
169 throw std::runtime_error{"Input " + std::to_string(i) + "'s buffer is not set."};
170
171 tensorpool.emplace_back(std::make_unique<backend::builtin::UserTensor>(
172 desc->info, desc->layout, const_cast<uint8_t *>(static_cast<const uint8_t *>(desc->buffer)),
173 desc->size));
174
175 auto user_type = desc->info.typeInfo().type();
176 auto &model_info = entryExecutor()->inputInfo(i).typeInfo();
177 auto model_type = model_info.type();
178 if (user_type != model_type && user_type == ir::DataType::FLOAT32)
179 {
180 auto quantized_info = desc->info;
181 quantized_info.typeInfo(model_info);
182 qtensorpool.emplace_back(
183 std::make_unique<EdgeTensor>(quantized_info, entryExecutor()->inputLayout(i)));
184 qtensorpool.back()->allocate_buffer();
185
186 input_tensors.push_back(tensorpool.back().get());
187 input_qtensors.push_back(qtensorpool.back().get());
188 inputs[i] = qtensorpool.back().get();
189 }
190 else
191 inputs[i] = tensorpool.back().get();
192 }
193
194 // Prepare UserTensor and EdgeTensor for output dequantization
195 for (uint32_t i = 0; i < outputs.size(); i++)
196 {
197 auto &desc = ctx.desc.outputs[i];
198
199 // Output is optional if buffer is nullptr, and optional output's size is 0
200 if (desc->buffer == nullptr && (desc->size != 0 || desc->info.total_size() != 0))
201 throw std::runtime_error{"Output " + std::to_string(i) + "'s buffer is not set."};
202
203 tensorpool.emplace_back(std::make_unique<backend::builtin::UserTensor>(
204 desc->info, desc->layout, static_cast<uint8_t *>(desc->buffer), desc->size));
205
206 auto user_type = desc->info.typeInfo().type();
207 auto &model_info = entryExecutor()->outputInfo(i).typeInfo();
208 auto model_type = model_info.type();
209 if (user_type != model_type && user_type == ir::DataType::FLOAT32)
210 {
211 auto quantized_info = desc->info;
212 quantized_info.typeInfo(model_info);
213 qtensorpool.emplace_back(
214 std::make_unique<EdgeTensor>(quantized_info, entryExecutor()->outputLayout(i)));
215 qtensorpool.back()->allocate_buffer();
216
217 output_qtensors.push_back(qtensorpool.back().get());
218 output_tensors.push_back(tensorpool.back().get());
219 outputs[i] = qtensorpool.back().get();
220 }
221 else
222 outputs[i] = tensorpool.back().get();
223 }
224
225 // Run quantization
226 if (input_tensors.size() > 0)
227 {
228 auto input_quantize_layer = PermuteLayer(input_tensors, input_qtensors);
229 input_quantize_layer.prepare();
230 input_quantize_layer.run();
231 }
232
233 // Executor
234 entryExecutor()->execute(inputs, outputs, ctx.options);
235
236 // Run dequantization
237 if (output_tensors.size() != 0)
238 {
239 auto output_dequantize_layer = PermuteLayer(output_qtensors, output_tensors);
240 output_dequantize_layer.prepare();
241 output_dequantize_layer.run();
242 }
243
244 // Get dynamic shape inference result
245 for (uint32_t i = 0; i < outputs.size(); i++)
246 {
247 if (ctx.desc.outputs[i]->buffer == nullptr)
248 {
249 // Output is optional if buffer is nullptr
250 continue;
251 }
252
253 ctx.desc.outputs[i]->info.shape(outputs[i]->getShape());
254 }
255}
