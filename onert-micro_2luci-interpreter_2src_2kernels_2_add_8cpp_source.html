36void Add::configure()
37{
38 LUCI_INTERPRETER_CHECK(input1()->element_type() == input2()->element_type());
39 LUCI_INTERPRETER_CHECK(input1()->element_type() == output()->element_type());
40 if (input1()->element_type() == DataType::S16)
41 {
42 LUCI_INTERPRETER_CHECK(input1()->zero_points().size() == 1 &&
43 input2()->zero_points().size() == 1);
44 LUCI_INTERPRETER_CHECK(input1()->zero_point() == 0 && input2()->zero_point() == 0 &&
45 output()->zero_point() == 0);
46 }
47
48 // TODO: enable it only if kernel with dynamic shapes
49 output()->resize(calculateShapeForBroadcast(input1()->shape(), input2()->shape()));
50}

52void Add::execute() const
53{
54 switch (input1()->element_type())
55 {
56 case DataType::FLOAT32:
57 evalFloat();
58 break;
59 case DataType::S64:
60 evalInteger<int64_t>();
61 break;
62 case DataType::S32:
63 evalInteger<int32_t>();
64 break;
65 case DataType::U8:
66 evalQuantized();
67 break;
68 case DataType::S16:
69 evalQuantizedS16();
70 break;
71 default:
72 assert(false && "Unsupported type.");
73 }
74}

120void Add::evalQuantized() const
121{
122 const auto input1_scale = static_cast<double>(input1()->scale());
123 const auto input2_scale = static_cast<double>(input2()->scale());
124 const auto output_scale = static_cast<double>(output()->scale());
125
126 const int left_shift = 20;
127 const double twice_max_input_scale = 2 * std::max(input1_scale, input2_scale);
128 const double real_input1_multiplier = input1_scale / twice_max_input_scale;
129 const double real_input2_multiplier = input2_scale / twice_max_input_scale;
130 const double real_output_multiplier = twice_max_input_scale / ((1 << left_shift) * output_scale);
131
132 int32_t input1_multiplier{}, input2_multiplier{}, output_multiplier{};
133 int input1_shift{}, input2_shift{}, output_shift{};
134 quantizeMultiplierSmallerThanOneExp(real_input1_multiplier, &input1_multiplier, &input1_shift);
135 quantizeMultiplierSmallerThanOneExp(real_input2_multiplier, &input2_multiplier, &input2_shift);
136 quantizeMultiplierSmallerThanOneExp(real_output_multiplier, &output_multiplier, &output_shift);
137
138 int32_t activation_min{};
139 int32_t activation_max{};
140 calculateActivationRangeQuantized(_params.activation, output(), &activation_min, &activation_max);
141
142 tflite::ArithmeticParams params{};
143 params.left_shift = left_shift;
144 // The kernel expects inputs' zero points to be negated.
145 params.input1_offset = -input1()->zero_point(); // Note the '-'.
146 params.input1_multiplier = input1_multiplier;
147 params.input1_shift = input1_shift;
148 params.input2_offset = -input2()->zero_point(); // Note the '-'.
149 params.input2_multiplier = input2_multiplier;
150 params.input2_shift = input2_shift;
151 params.output_offset = output()->zero_point();
152 params.output_multiplier = output_multiplier;
153 params.output_shift = output_shift;
154 params.quantized_activation_min = activation_min;
155 params.quantized_activation_max = activation_max;
156
157 const bool need_broadcast = tflite::reference_ops::ProcessBroadcastShapes(
158 getTensorShape(input1()), getTensorShape(input2()), &params);
159
160 if (need_broadcast)
161 {
162 tflite::reference_ops::BroadcastAdd4DSlow(
163 params, getTensorShape(input1()), getTensorData<uint8_t>(input1()), getTensorShape(input2()),
164 getTensorData<uint8_t>(input2()), getTensorShape(output()), getTensorData<uint8_t>(output()));
165 }
166 else
167 {
168 tflite::reference_ops::Add(params, getTensorShape(input1()), getTensorData<uint8_t>(input1()),
169 getTensorShape(input2()), getTensorData<uint8_t>(input2()),
170 getTensorShape(output()), getTensorData<uint8_t>(output()));
171 }
172}

174void Add::evalQuantizedS16() const
175{
176 const auto input1_scale = static_cast<double>(input1()->scale());
177 const auto input2_scale = static_cast<double>(input2()->scale());
178 const auto output_scale = static_cast<double>(output()->scale());
179
180 constexpr int left_shift = 12;
181 const double twice_max_input_scale = 2 * std::max(input1_scale, input2_scale);
182 const double real_input1_multiplier = input1_scale / twice_max_input_scale;
183 const double real_input2_multiplier = input2_scale / twice_max_input_scale;
184 const double real_output_multiplier = twice_max_input_scale / ((1 << left_shift) * output_scale);
185
186 int32_t input1_multiplier{}, input2_multiplier{}, output_multiplier{};
187 int input1_shift{}, input2_shift{}, output_shift{};
188 quantizeMultiplierSmallerThanOneExp(real_input1_multiplier, &input1_multiplier, &input1_shift);
189 quantizeMultiplierSmallerThanOneExp(real_input2_multiplier, &input2_multiplier, &input2_shift);
190 quantizeMultiplierSmallerThanOneExp(real_output_multiplier, &output_multiplier, &output_shift);
191
192 int32_t activation_min{};
193 int32_t activation_max{};
194 calculateActivationRangeQuantized(_params.activation, output(), &activation_min, &activation_max);
195
196 auto fn = [input1_multiplier, input1_shift, //
197 input2_multiplier, input2_shift, //
198 output_multiplier, output_shift, //
199 activation_min, activation_max](int16_t input1_val, int16_t input2_val) {
200 const int32_t shifted_input1_val = static_cast<int32_t>(input1_val) << left_shift;
201 const int32_t shifted_input2_val = static_cast<int32_t>(input2_val) << left_shift;
202 const int32_t scaled_input1_val = tflite::MultiplyByQuantizedMultiplierSmallerThanOneExp(
203 shifted_input1_val, input1_multiplier, input1_shift);
204 const int32_t scaled_input2_val = tflite::MultiplyByQuantizedMultiplierSmallerThanOneExp(
205 shifted_input2_val, input2_multiplier, input2_shift);
206 const int32_t raw_sum = scaled_input1_val + scaled_input2_val;
207 const int32_t raw_output = tflite::MultiplyByQuantizedMultiplierSmallerThanOneExp(
208 raw_sum, output_multiplier, output_shift);
209 const int32_t clamped_output = std::min(activation_max, std::max(activation_min, raw_output));
210 return static_cast<int16_t>(clamped_output);
211 };

214 getTensorShape(input2()), getTensorData<int16_t>(input2()),
215 getTensorShape(output()), getTensorData<int16_t>(output()), fn);
216}
