59inline std::string decorated_name(const char *input)
60{
61 const int min_prefix = 16;
62 std::string prefix(input);
63 auto len_prefix = prefix.size();
64 if (len_prefix > min_prefix)
65 return "[" + prefix + "] ";
66 std::string spaces((min_prefix - len_prefix) / 2, ' ');
67 return (len_prefix % 2 ? "[ " : "[") + spaces + prefix + spaces + "] ";
68}
