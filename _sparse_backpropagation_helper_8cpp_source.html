45void generateRankCombinations(
46 const std::unordered_map<uint16_t, std::unordered_set<uint16_t>> &index_to_possible_ranks,
47 const std::vector<uint16_t> &indices, size_t currentIndex,
48 std::vector<uint16_t> &currentCombination, std::vector<std::vector<uint16_t>> &result)
49{
50 if (currentIndex == indices.size())
51 {
52 result.push_back(currentCombination);
53 return;
54 }
55 uint16_t index = indices[currentIndex];
56 for (uint16_t rank : index_to_possible_ranks.at(index))
57 {
58 currentCombination.push_back(rank);
59 generateRankCombinations(index_to_possible_ranks, indices, currentIndex + 1, currentCombination,
60 result);
61 currentCombination.pop_back();
62 }
63}

65// Find pairs: selected ops indexes - divided dim max rank value
66std::unordered_map<uint16_t, uint32_t> findTrainableTensorsMaxDivideRankAccordingToOperatorIndex(
67 const std::unordered_set<uint16_t> &selected_op_indexes,
68 const onert_micro::core::reader::OMCircleReader &reader)
69{
70 std::unordered_map<uint16_t, uint32_t> operator_index_to_tensor_max_divide_rank;
71
72 // Read ops
73 auto operators = reader.operators();
74 assert(operators != nullptr);
75
76 auto op_size = operators->size();
77
78 // Obtain operation codes
79 auto op_codes = reader.opcodes();
80
81 const auto tensors = reader.tensors();
82
83 // Go over selected best op indexes
84 for (auto op_index : selected_op_indexes)
85 {
86 auto cur_op = operators->operator[](op_index);
87
88 // Get opcode index
89 uint32_t cur_opcode_index = cur_op->opcode_index();
90 assert(cur_opcode_index < op_codes->size());
91
92 const auto opcode = op_codes->operator[](cur_opcode_index);
93
94 const auto inputs_tensors = cur_op->inputs();
95
96 uint32_t tensor_divided_dim_value = 0;
97
98 switch (opcode->builtin_code())
99 {
100 case circle::BuiltinOperator_FULLY_CONNECTED:
101 case circle::BuiltinOperator_CONV_2D:
102 {
103 assert(inputs_tensors->size() >= 2);
104 auto tensor_index = inputs_tensors->operator[](1);
105 assert(tensor_index != -1);
106 assert(tensor_index < tensors->size());
107 auto tensor = tensors->operator[](tensor_index);
108
109 // For FC and Conv2D op tool provide rank over 0 dimensional
110 tensor_divided_dim_value = tensor->shape()->operator[](0);
111
112 break;
113 }
114 default:
115 assert(false && "Unsupported type");
116 tensor_divided_dim_value = 0; // Not supported
117 }
118
119 assert(tensor_divided_dim_value != 0);
120 operator_index_to_tensor_max_divide_rank[op_index] = tensor_divided_dim_value;
121 }
122
123 return operator_index_to_tensor_max_divide_rank;
124}

126void recursiveGenerateAllPossibleOpIndexesSetsHelper(
127 std::vector<std::unordered_set<uint16_t>> &result, std::unordered_set<uint16_t> &cur_set,
128 std::unordered_set<uint16_t>::const_iterator cur_it_set_value,
129 std::unordered_set<uint16_t>::const_iterator &end_it_set)
130{
131 // If we reach end of the initial set then finish
132 if (cur_it_set_value == end_it_set)
133 {
134 // If set is not empty add to final result
135 if (cur_set.empty() == false)
136 result.push_back(cur_set);
137 return;
138 }
139
140 // Add value to current set
141 uint16_t cur_index = *cur_it_set_value;
142 cur_set.insert(cur_index);
143 // Run further and move iterator to next position
144 cur_it_set_value++;
145 recursiveGenerateAllPossibleOpIndexesSetsHelper(result, cur_set, cur_it_set_value, end_it_set);
146 // Remove current index from set
147 cur_set.erase(cur_index);
148 // Run again recursive functions but now without current index
149 recursiveGenerateAllPossibleOpIndexesSetsHelper(result, cur_set, cur_it_set_value, end_it_set);
150}

154// Find is left train result is better then right
155bool training_configure_tool::cmpTrainResults(const training_configure_tool::TrainResult &left,
156 const training_configure_tool::TrainResult &right,
157 const float acceptable_diff)
158{
159 // Metrics should be the same
160 assert(left.best_metrics_results.first == right.best_metrics_results.first);
161 OMMetrics metric = left.best_metrics_results.first;
162 float left_metric_res = left.best_metrics_results.second;
163 float right_metric_res = right.best_metrics_results.second;
164
165 bool is_in_acceptable_diff = std::abs(left_metric_res - right_metric_res) <= acceptable_diff;
166 if (is_in_acceptable_diff)
167 {
168 return left.peak_memory_footprint < right.peak_memory_footprint;
169 }
170
171 switch (metric)
172 {
173 case onert_micro::ACCURACY:
174 {
175 return left.best_metrics_results.second > right.best_metrics_results.second;
176 }
177 break;
178 case onert_micro::CROSS_ENTROPY_METRICS:
179 case onert_micro::MSE_METRICS:
180 case onert_micro::MAE_METRICS:
181 {
182
183 return left.best_metrics_results.second < right.best_metrics_results.second;
184 }
185 break;
186 default:
187 assert(false && "Unsupported type");
188 break;
189 }
190 return true;
191}

193// Remove operations indexes sets with peak memory footprint greater then given restriction:
194// 1 - Run train interpreter with all this sets with single train sample and single test sample
195// to obtain approximately peak memory footprint for each set 2 - Cut according to max peak
196// memory
197std::vector<std::unordered_set<uint16_t>>
198training_configure_tool::selectOpIndexesSetsAccordingToMemoryRestriction(
199 const std::vector<std::unordered_set<uint16_t>> &op_indexes_sets, onert_micro::OMConfig config,
200 training_configure_tool::TrainData train_data)
201{
202 // It 0 - then is not set
203 if (train_data.memory_above_restriction == 0)
204 {
205 return op_indexes_sets;
206 }
207
208 std::vector<std::unordered_set<uint16_t>> result;
209
210 // To obtain real estimation we need minimum batch_size = 2 and num_train_data_samples = 4
211 // Change config train and test sample values
212 train_data.num_test_data_samples = 0;
213 train_data.num_train_data_samples = std::min(4, train_data.num_train_data_samples);
214 // To disable tests
215 train_data.metrics_to_check_best_config = NONE;
216 // Set number of the epochs and batch size to one
217 config.training_context.epochs = 1;
218 config.training_context.batch_size = std::min(2u, config.training_context.batch_size);
219
220 for (const auto &op_indexes_set : op_indexes_sets)
221 {
222#if PRINT
223 printf("Start checking: { ");
224 for (auto i : op_indexes_set)
225 {
226 printf("%d ", i);
227 }
228 printf("}\n");
229#endif
230 // Construct mapping with current indexes - use default train ALL parts
231 std::unordered_map<uint16_t, OpTrainableRank> train_op_ranks;
232 for (auto index : op_indexes_set)
233 {
234 train_op_ranks[index] = ALL;
235 }
236
237 std::vector<char> tmp_buffer;
238 // Create data with current buffer information
239 createResultData({train_op_ranks}, tmp_buffer);
240 config.training_context.training_config_info_data = tmp_buffer.data();
241
242 TrainResult train_result;
243 // Run train with this information
244 runTrainProcessWithCurConfig(config, train_data, train_result);
245#if PRINT
246 printf("CURRENT MEMORY PEAK = %zu\n", train_result.peak_memory_footprint);
247#endif
248 if (train_result.peak_memory_footprint < train_data.memory_above_restriction)
249 {
250#if PRINT
251 printf("Added to the result\n");
252#endif
253 result.push_back(op_indexes_set);
254 }
255 }
256
257 return result;
258}

274// To find all trainable ops indexes in the model - initial_train_op_indexes
275std::unordered_set<uint16_t>
276training_configure_tool::findAllTrainableOps(const char *circle_model_path)
277{
278 std::unordered_set<uint16_t> result;
279
280 training_configure_tool::DataBuffer model_ptr =
281 training_configure_tool::readFile(circle_model_path);
282
283 // Init reader
284 OMStatus status = Ok;
285 core::reader::OMCircleReader reader;
286 assert(model_ptr.data() != nullptr);
287 status = reader.parse(model_ptr.data());
288 assert(status == Ok);
289 // return empty set
290 if (status != Ok)
291 return result;
292 // TODO: support multi subgraph models
293 status = reader.select_subgraph(0);
294 // return empty set
295 if (status != Ok)
296 return result;
297
298 // Read ops
299 auto operators = reader.operators();
300 assert(operators != nullptr);
301
302 auto op_size = operators->size();
303
304 // Obtain operation codes
305 auto op_codes = reader.opcodes();
306
307 // Run through all ops
308 for (uint32_t i = 0; i < op_size; ++i)
309 {
310 auto cur_op = operators->operator[](i);
311
312 // Get opcode index
313 uint32_t cur_opcode_index = cur_op->opcode_index();
314 assert(cur_opcode_index < op_codes->size());
315
316 const auto opcode = op_codes->operator[](cur_opcode_index);
317
318 // If op is trainable - insert it
319 if (isTrainableWeights(opcode))
320 result.insert(static_cast<uint16_t>(i));
321 }
322
323 return result;
324}

326std::vector<std::unordered_map<uint16_t, training_configure_tool::OpTrainableRank>>
327training_configure_tool::findAllTensorsRanksCombinations(
328 const std::unordered_set<uint16_t> &selected_op_indexes, onert_micro::OMConfig config,
329 training_configure_tool::TrainData train_data)
330{
331 // 1 - Find pairs: selected ops indexes - divided dim max rank value
332 // 2 - Find for every tensor index every possible rank according to its opcode and size
333 // 3 - Get result
334 std::vector<std::unordered_map<uint16_t, training_configure_tool::OpTrainableRank>> result;
335
336 training_configure_tool::DataBuffer model_ptr =
337 training_configure_tool::readFile(train_data.circle_model_path);
338
339 // Init reader
340 OMStatus status = Ok;
341 core::reader::OMCircleReader reader;
342 assert(model_ptr.data() != nullptr);
343 status = reader.parse(model_ptr.data());
344 assert(status == Ok);
345 // return empty set
346 if (status != Ok)
347 return result;
348 // TODO: support multi subgraph models
349 status = reader.select_subgraph(0);
350 // return empty set
351 if (status != Ok)
352 return result;
353
354 // 1 - Find pairs: selected ops indexes - divided dim max rank value
355 std::unordered_map<uint16_t, uint32_t> operator_index_to_tensor_index =
356 findTrainableTensorsMaxDivideRankAccordingToOperatorIndex(selected_op_indexes, reader);
357 assert(operator_index_to_tensor_index.size() == selected_op_indexes.size());
358 // 2 - Find for every tensor index every possible rank according to its opcode and size
359 std::unordered_map<uint16_t, std::unordered_set<uint16_t>> op_index_to_all_possible_ranks;
360 for (auto &p : operator_index_to_tensor_index)
361 {
362 const auto op_index = p.first;
363 const auto max_value = p.second;
364
365 uint16_t cur_value = 2;
366 op_index_to_all_possible_ranks[op_index] = {ALL, ONLY_BIAS};
367 while (cur_value < uint32_t(OpTrainableRank::MAX_VALUE) and cur_value <= max_value)
368 {
369 auto new_value = cur_value * 2;
370 while (cur_value < uint16_t(OpTrainableRank::MAX_VALUE) and cur_value < new_value)
371 {
372 op_index_to_all_possible_ranks[op_index].insert(cur_value);
373 cur_value++;
374 }
375 }
376 }
377 // Get all op indices
378 std::vector<uint16_t> indices(selected_op_indexes.begin(), selected_op_indexes.end());
379 std::vector<std::vector<uint16_t>> rank_combinations;
380 std::vector<uint16_t> cur_v;
381 generateRankCombinations(op_index_to_all_possible_ranks, indices, 0, cur_v, rank_combinations);
382
383 for (const auto &ranks : rank_combinations)
384 {
385 std::unordered_map<uint16_t, OpTrainableRank> combination;
386 for (size_t i = 0; i < indices.size(); ++i)
387 {
388 combination[indices[i]] = OpTrainableRank(ranks[i]);
389 }
390 result.push_back(std::move(combination));
391 }
392
393 return result;
394}
