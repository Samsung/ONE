34template <typename T, std::enable_if_t<std::is_floating_point<T>::value, bool> = true>
35struct FloorDivFn
36{
37 T operator()(T lhs, T rhs)
38 {
39 return std::floor(static_cast<double>(lhs) / static_cast<double>(rhs));
40 }
41};
42template <typename T, std::enable_if_t<std::is_floating_point<T>::value, bool> = true>
43struct FloorModFn
44{
45 T operator()(T lhs, T rhs)
46 {
47 T trunc_mod = std::fmod(lhs, rhs);
48 return (trunc_mod != 0) && ((rhs < 0) != (trunc_mod < 0)) ? (trunc_mod + rhs) : trunc_mod;
49 }
50};
51template <typename T> struct MaximumFn
52{
53 T operator()(T lhs, T rhs) { return std::max(lhs, rhs); }
54};
55template <typename T> struct MinimumFn
56{
57 T operator()(T lhs, T rhs) { return std::min(lhs, rhs); }
58};

61template <typename T, typename Fn>
62inline void BinaryOp(const int flat_size, const T *input1_data, const T *input2_data,
63 T *output_data)
64{
65 Fn func;
66 for (int i = 0; i < flat_size; ++i)
67 {
68 output_data[i] = func(input1_data[i], input2_data[i]);
69 }

72template <typename T, typename Fn>
73inline void
74BroadcastBinaryOp4DSlow(const core::OMRuntimeShape &input1_shape, const float *input1_data,
75 const core::OMRuntimeShape &input2_shape, const float *input2_data,
76 const core::OMRuntimeShape &output_shape, float *output_data)
77{
78 NdArrayDesc<4> desc1;
79 NdArrayDesc<4> desc2;
80 NdArrayDescsForElementwiseBroadcast(input1_shape, input2_shape, &desc1, &desc2);
81
82 const core::OMRuntimeShape extended_output_shape =
83 core::OMRuntimeShape::extendedShape(4, output_shape);
84
85 // In Tensorflow, the dimensions are canonically named (batch_number, row,
86 // col, channel), with extents (batches, height, width, depth), with the
87 // trailing dimension changing most rapidly (channels has the smallest stride,
88 // typically 1 element).
89 //
90 // In generated C code, we store arrays with the dimensions reversed. The
91 // first dimension has smallest stride.
92 //
93 // We name our variables by their Tensorflow convention, but generate C code
94 // nesting loops such that the innermost loop has the smallest stride for the
95 // best cache behavior.
96
97 Fn func;
98 for (int b = 0; b < extended_output_shape.dims(0); ++b)
99 {
100 for (int y = 0; y < extended_output_shape.dims(1); ++y)
101 {
102 for (int x = 0; x < extended_output_shape.dims(2); ++x)
103 {
104 for (int c = 0; c < extended_output_shape.dims(3); ++c)
105 {
106 const int output_data_offset =
107 ((b * extended_output_shape.dims(1) + y) * extended_output_shape.dims(2) + x) *
108 extended_output_shape.dims(3) +
109 c;
110
111 output_data[output_data_offset] = func(input1_data[subscriptToIndex(desc1, b, y, x, c)],
112 input2_data[subscriptToIndex(desc2, b, y, x, c)]);
113 }
114 }
115 }
116 }
117}
