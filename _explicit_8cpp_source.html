35std::vector<uint8_t> ExplicitDataChef<std::string>::generate(int32_t count) const
36{
37 std::vector<uint8_t> res;
38
39 // write count
40 write_value(res, count);
41
42 // write first item offset
43 int32_t start = sizeof(int32_t) * (count + 2);
44 write_value(res, start);
45
46 // write succeeding items offset (or the end)
47 int32_t offset = start;
48 for (uint32_t n = 0; n < count; ++n)
49 {
50 std::string const value = (n < _values.size()) ? _values.at(n) : std::string{};
51 offset += value.length();
52 write_value(res, offset);
53 }
54
55 for (uint32_t n = 0; n < count; ++n)
56 {
57 std::string const value = (n < _values.size()) ? _values.at(n) : std::string{};
58 const uint8_t *arr = reinterpret_cast<const uint8_t *>(value.c_str());
59
60 for (uint32_t b = 0; b < value.length(); ++b)
61 {
62 res.emplace_back(arr[b]);
63 }
64 }
65
66 return res;
67}
