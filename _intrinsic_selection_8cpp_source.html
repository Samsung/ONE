29coco::Instr *rewrite(coco::Instr *curr)
30{
31 auto m = curr->module();
32 assert(m != nullptr);
33
34 if (auto eval = coco::safe_cast<coco::Eval>(curr))
35 {
36 if (auto concat_f = eval->op()->asConcatF())
37 {
38 auto fst_load = concat_f->left()->asLoad();
39 auto snd_load = concat_f->right()->asLoad();
40
41 if (fst_load && snd_load && (concat_f->axis() == coco::ConcatF::Axis::Depth))
42 {
43 // Here is the pattern of interest
44 //
45 // %ofm = eval(ConcatF(Depth, Load(%left), Load(%right)))
46 //
47 auto fst_feature = fst_load->object()->asFeature();
48 auto snd_feature = snd_load->object()->asFeature();
49 assert((fst_feature != nullptr) && (snd_feature != nullptr));
50
51 auto out_feature = eval->out()->asFeature();
52 assert(out_feature != nullptr);
53
54 eval->out(nullptr);
55
56 auto depth_concat = m->entity()->instr()->create<ANNDepthConcatF>();
57
58 depth_concat->out(out_feature);
59 depth_concat->fst(fst_feature);
60 depth_concat->snd(snd_feature);
61
62 return depth_concat;
63 }
64
65 return nullptr;
66 }
67 }
68
69 return nullptr;
70}

77void select_intrinsic(enco::Code *code)
78{
79 auto m = code->module();
80
81 for (auto blk = m->block()->head(); blk; blk = blk->next())
82 {
83 auto ins = blk->instr()->head();
84
85 while (ins)
86 {
87 if (auto rewritten_ins = rewrite(ins))
88 {
89 rewritten_ins->insertBefore(ins);
90 ins->detach();
91
92 ins = rewritten_ins;
93 }
94
95 ins = ins->next();
96 }
97 }
