43// Layers with trainable weights
44// Note: needed not to store some layers with const intputs but it is not trainable (for example
45// Reshape)
46bool isTrainableWeights(const circle::OperatorCode *opcode)
47{
48 switch (opcode->builtin_code())
49 {
50 case circle::BuiltinOperator_FULLY_CONNECTED:
51 case circle::BuiltinOperator_CONV_2D:
52 return true;
53 default:
54 return false;
55 }
56}

60OMStatus OMCheckpointLoader::validateCheckpointData(core::OMRuntimeContext &context,
61 const char *data, OMConfig &config)
62{
63 OMStatus status = Ok;
64
65 // Validate magic number
66 uint16_t mag_num = 0;
67 std::memcpy(&mag_num, &data[MAGIC_NUMBER_OFFSET], sizeof(mag_num));
68 assert(mag_num == MAGIC_NUMBER && "False MAGIC NUMBER, check correctness of checkpoint file");
69 if (mag_num != MAGIC_NUMBER)
70 return FailReadCheckpointFile;
71
72 // Validate schema version
73 uint8_t version = 0;
74 std::memcpy(&version, &data[SCHEMA_VERSION_OFFSET], sizeof(version));
75 assert(version == SCHEMA_VERSION &&
76 "False SCHEMA_VERSION NUMBER, check correctness of checkpoint file");
77 if (version != SCHEMA_VERSION)
78 return FailReadCheckpointFile;
79
80 // Validate count of tensors
81 uint32_t num_tensors = context.getCircleTensors()->size();
82 uint32_t num_tensors_in_file = 0;
83 std::memcpy(&num_tensors_in_file, &data[NUM_BUFFERS_OFFSET], sizeof(num_tensors_in_file));
84 assert(num_tensors == num_tensors_in_file &&
85 "Number of tensors in circle and in checkpoint file should be the same");
86 if (num_tensors != num_tensors_in_file)
87 return FailReadCheckpointFile;
88
89 // Validate m, v and other parameters offset
90 uint32_t m_offset;
91 uint32_t v_offset;
92 uint32_t other_params_offset;
93 std::memcpy(&m_offset, &data[M_OFFSET], sizeof(m_offset));
94 std::memcpy(&v_offset, &data[V_OFFSET], sizeof(v_offset));
95 std::memcpy(&other_params_offset, &data[OTHER_PARAMS_OFFSET], sizeof(other_params_offset));
96
97 assert(other_params_offset > 0);
98 if (other_params_offset == 0)
99 return FailReadCheckpointFile;
100
101 if (config.training_context.optimizer == SGD)
102 {
103 assert(m_offset == 0 and v_offset == 0);
104 if (m_offset != 0 or v_offset != 0)
105 return FailReadCheckpointFile;
106 }
107
108 return Ok;
109}

111OMStatus OMCheckpointLoader::loadBuffers(core::OMRuntimeContext &context,
112 OMTrainingStorage &train_storage, const char *data,
113 OMConfig &config)
114{
115 OMStatus status = Ok;
116
117 // Read v, m and other params offsets
118 uint32_t m_offset;
119 uint32_t v_offset;
120 uint32_t other_params_offset;
121 std::memcpy(&m_offset, &data[M_OFFSET], sizeof(m_offset));
122 std::memcpy(&v_offset, &data[V_OFFSET], sizeof(v_offset));
123 std::memcpy(&other_params_offset, &data[OTHER_PARAMS_OFFSET], sizeof(other_params_offset));
124
125 uint32_t weight_offset_pos = WEIGHT_OFFSET;
126
127 // If optimizer is Adam then reset its state
128 optimizers::Adam *adam_opt = nullptr;
129 if (config.training_context.optimizer == ADAM)
130 {
131 adam_opt = train_storage.getAdam();
132 assert(adam_opt != nullptr);
133
134 adam_opt->fullReset();
135 }
136
137 auto tensors = context.getCircleTensors();
138 auto tensors_size = tensors->size();
139 const auto *operators = context.getCircleOperators();
140 const auto num_kernels = operators->size();
141 uint32_t num_train_layers = config.training_context.num_of_train_layers == 0
142 ? num_kernels
143 : config.training_context.num_of_train_layers;
144 uint32_t last_node_pos = std::min(num_kernels, num_train_layers);
145 // Goes among trainable ops
146 const auto *op_codes = context.getCircleOpcodes();
147 for (uint32_t index = 0; index < last_node_pos; ++index)
148 {
149 uint32_t cur_op_index = num_kernels - index - 1;
150 auto *cur_op = operators->operator[](cur_op_index);
151
152 auto input_tensors = cur_op->inputs();
153
154 for (uint32_t i = 0; i < input_tensors->size(); ++i)
155 {
156 const auto input_tensor_index = input_tensors->operator[](i);
157 // Check is const
158 if (not context.isConstTensor(input_tensor_index))
159 continue;
160
161 uint32_t cur_opcode_index = cur_op->opcode_index();
162
163 assert(cur_opcode_index < op_codes->size());
164
165 const auto opcode = op_codes->operator[](cur_opcode_index);
166
167 // Check it is trainable const
168 if (not isTrainableWeights(opcode))
169 continue;
170
171 // Get current weight file pos
172 uint32_t cur_weight_offset_pos = weight_offset_pos + input_tensor_index * 4;
173
174 // Read current tensor offset
175 uint32_t cur_tensor_offset;
176 std::memcpy(&cur_tensor_offset, &data[cur_weight_offset_pos], sizeof(cur_tensor_offset));
177
178 // Check is it save data or not
179 // Note: zero means there are no data - it is error
180 if (cur_tensor_offset == 0)
181 return FailReadCheckpointFile;
182
183 // Read weight data and save it
184 const auto tensor = tensors->operator[](input_tensor_index);
185 assert(tensor != nullptr);
186
187 OMRuntimeShape shape(tensor);
188 auto type_size = sizeof(OMDataType(tensor->type()));
189
190 size_t buffer_size = shape.flatSize() * type_size;
191 // Get pointer to the data in model
192 uint8_t *weight_data_in_model_ptr;
193
194 status = context.getConstDataByTensorIndex(&weight_data_in_model_ptr, input_tensor_index);
195 assert(status == Ok);
196 assert(weight_data_in_model_ptr != nullptr);
197 if (status != Ok or weight_data_in_model_ptr == nullptr)
198 return status;
199
200 std::memcpy(weight_data_in_model_ptr, &data[cur_tensor_offset], buffer_size);
201
202 if (config.training_context.optimizer == SGD)
203 continue;
204
205 // For Adam read m and v buffer
206 assert(config.training_context.optimizer == ADAM);
207
208 // If no saved Adam state then continue
209 if (m_offset == 0 or v_offset == 0)
210 {
211 assert(v_offset == 0);
212 assert(m_offset == 0);
213 continue;
214 }
215
216 // Get current v and m file pos
217 uint32_t cur_m_offset_pos = m_offset + input_tensor_index * 4;
218 uint32_t cur_v_offset_pos = v_offset + input_tensor_index * 4;
219
220 // Read current v and m offset
221 uint32_t cur_m_offset;
222 uint32_t cur_v_offset;
223 std::memcpy(&cur_m_offset, &data[cur_m_offset_pos], sizeof(cur_m_offset));
224 std::memcpy(&cur_v_offset, &data[cur_v_offset_pos], sizeof(cur_v_offset));
225
226 // Cannot be zero due to weight already not zero
227 assert(cur_m_offset != 0 and cur_v_offset != 0);
228 if (cur_m_offset == 0 or cur_v_offset == 0)
229 return FailReadCheckpointFile;
230
231 assert(adam_opt != nullptr);
232
233 // Allocate memory for m and v current buffer
234 uint8_t *m_buffer = nullptr;
235 uint8_t *v_buffer = nullptr;
236 status = memory::OMMemoryManager::allocateMemory(buffer_size, &m_buffer);
237 assert(status == Ok);
238 assert(m_buffer != nullptr);
239 status = memory::OMMemoryManager::allocateMemory(buffer_size, &v_buffer);
240 assert(status == Ok);
241 assert(v_buffer != nullptr);
242
243 // Read m and v buffer
244 std::memcpy(m_buffer, &data[cur_m_offset], buffer_size);
245 std::memcpy(v_buffer, &data[cur_v_offset], buffer_size);
246
247 // Save m and v buffer in adam optimizer state
248 adam_opt->setExponentAvgDataByTensorIndex(input_tensor_index, m_buffer);
249 adam_opt->setExponentAvgSquaresDataByTensorIndex(input_tensor_index, v_buffer);
250 }
251 }
252
253 // Read other parameters: cur step num and cur epoch num
254 uint32_t cur_step_offset = other_params_offset;
255 uint32_t cur_epoch_offset = other_params_offset + 4;
256
257 uint32_t cur_step;
258 std::memcpy(&cur_step, &data[cur_step_offset], sizeof(cur_step));
259
260 uint32_t cur_epoch;
261 std::memcpy(&cur_epoch, &data[cur_epoch_offset], sizeof(cur_epoch));
262
263 // Save it in config
264 config.training_context.num_step = cur_step;
265 config.training_context.num_epoch = cur_epoch;
266
267 return status;
268}

270// To check checkpoint file format please see https://github.com/Samsung/ONE/discussions/13037
271OMStatus OMCheckpointLoader::loadCheckpointData(core::OMRuntimeContext &context,
272 OMTrainingStorage &train_storage, const char *data,
273 OMConfig &config)
274{
275
276 // Validate current checkpoint file data
277 OMStatus status = validateCheckpointData(context, data, config);
278 assert(status == Ok);
279 if (status != Ok)
280 return status;
281
282 // Read and save buffer
283 status = loadBuffers(context, train_storage, data, config);
284 assert(status == Ok);
285
286 return status;
287}
