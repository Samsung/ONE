51void StaticTensorManager::allocateNonconsts(void)
52{
53 _nonconst_mgr->allocate();
54
55 for (auto &&[ind, tensor] : _tensors->native_tensors())
56 {
57 if (!_as_constants[ind] && !tensor->is_dynamic())
58 {
59 auto *buffer = _nonconst_mgr->getBuffer(ind);
60 tensor->setBuffer(buffer);
61
62 VERBOSE(CPU_StaticTensorManager)
63 << "TENSOR " << ind << " : " << static_cast<void *>(buffer) << std::endl;
64 }
65 }
66}

70void StaticTensorManager::buildTensor(const ir::OperandIndex &ind,
71 const ir::OperandInfo &tensor_info, bool as_const)
72{
73 assert(!_tensors->getNativeTensor(ind));
74 if (as_const)
75 {
76 auto tensor = std::make_unique<ExternalTensor>(tensor_info);
77 _tensors->setNativeTensor(ind, std::move(tensor));
78 }
79 else
80 {
81 auto tensor =
82 std::make_unique<Tensor>(tensor_info, _dynamic_tensor_manager->dynamic_mem_mgr().get());
83 _tensors->setNativeTensor(ind, std::move(tensor));
84 }
85 _as_constants[ind] = as_const;
86}

88void StaticTensorManager::claimPlan(const ir::OperandIndex &ind, uint32_t size)
89{
90 assert(_tensors->getNativeTensor(ind));
91
92 // This method is called only when a tensor has proper shape
93 assert(!_tensors->getNativeTensor(ind)->is_dynamic());
94
95 if (!_as_constants[ind])

97}
