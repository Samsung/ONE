30void check_const_opcode(luci::CircleNode *node)
31{
32 if (node == nullptr)
33 return;
34
35 if (node->opcode() != luci::CircleOpcode::CIRCLECONST and
36 node->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
37 {
38 throw std::runtime_error("Unsupported non const input " + node->name());
39 }
40}
41
42struct ConstInputChecker final : public luci::CircleNodeMutableVisitor<void>
43{
44// INPUT_NAME is name for input const for current NODE
45#define CHECK_NODE_WITH_ONE_INPUT_CONST(NODE, INPUT_NAME) \
46 void visit(NODE *node) \
47 { \
48 const auto input = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME()); \
49 check_const_opcode(input); \
50 }
51
52// INPUT_NAME_1 and INPUT_NAME_2 are names for input const for current NODE
53#define CHECK_NODE_WITH_TWO_INPUT_CONST(NODE, INPUT_NAME_1, INPUT_NAME_2) \
54 void visit(NODE *node) \
55 { \
56 const auto input_1 = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME_1()); \
57 const auto input_2 = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME_2()); \
58 \
59 check_const_opcode(input_1); \
60 check_const_opcode(input_2); \
61 }
62
63// INPUT_NAME_1, INPUT_NAME_2 and INPUT_NAME_3 are names for input const for current NODE
64#define CHECK_NODE_WITH_THREE_INPUT_CONST(NODE, INPUT_NAME_1, INPUT_NAME_2, INPUT_NAME_3) \
65 void visit(NODE *node) \
66 { \
67 const auto input_1 = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME_1()); \
68 const auto input_2 = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME_2()); \
69 const auto input_3 = dynamic_cast<luci::CircleNode *>(node->INPUT_NAME_3()); \
70 \
71 check_const_opcode(input_1); \
72 check_const_opcode(input_2); \
73 check_const_opcode(input_3); \
74 }
75
76 // Skip other circle node
77 void visit(luci::CircleNode *) {}
78
79 // Ops that receive one const nodes as inputs
80 CHECK_NODE_WITH_ONE_INPUT_CONST(luci::CirclePRelu, alpha)
81
82 // Ops that receive two const node as an inputs
83 CHECK_NODE_WITH_TWO_INPUT_CONST(luci::CircleConv2D, filter, bias)
84 CHECK_NODE_WITH_TWO_INPUT_CONST(luci::CircleDepthwiseConv2D, filter, bias)
85 CHECK_NODE_WITH_TWO_INPUT_CONST(luci::CircleFullyConnected, weights, bias)
86 CHECK_NODE_WITH_TWO_INPUT_CONST(luci::CircleInstanceNorm, gamma, beta)
87
88 // Ops that receive three const nodes as an inputs
89 CHECK_NODE_WITH_THREE_INPUT_CONST(luci::CircleTransposeConv, inputSizes, filter, bias)
90
91#undef CHECK_NODE_WITH_ONE_INPUT_CONST
92#undef CHECK_NODE_WITH_TWO_INPUT_CONST
93#undef CHECK_NODE_WITH_THREE_INPUT_CONST
94};

101bool QuantizePreCheckerPass::run(loco::Graph *g)
102{
103 LOGGER(l);
104 INFO(l) << "QuantizePreCheckerPass Start" << std::endl;
105
106 for (auto node : loco::active_nodes(loco::output_nodes(g)))
107 {
108 // Check const inputs
109 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
110 ConstInputChecker checker{};
111 circle_node->accept(&checker);
112 }
113
114 INFO(l) << "QuantizePreCheckerPass End" << std::endl;
115
116 return false; // one time run
117}
