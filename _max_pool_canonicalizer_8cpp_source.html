29bool canonicalize_maxpool2d(loco::Graph *graph, moco::TFMaxPool *node)
30{
31 LOGGER(l);
32
51 auto data_layout = plier::tf::as_data_layout(node->data_layout());
52
53 auto feature_enc = graph->nodes()->create<loco::FeatureEncode>();
54 auto maxPool2d_node = graph->nodes()->create<loco::MaxPool2D>();
55 auto feature_dec = graph->nodes()->create<loco::FeatureDecode>();
56
57 set_feature_enc(feature_enc, data_layout);
58 set_feature_dec(feature_dec, data_layout);
59
60 // paddata to pad
61 auto input_shape = moco::node_shape(node->input());
62 assert(input_shape.domain() != loco::Domain::Unknown);
63
64 auto node_stride = moco::stride_of(node->strides(), node->data_layout());
65 auto node_window = moco::window_of(node->ksize(), node->data_layout());
66
67 moco::Padding2DInference infer_padding2d;
68
69 infer_padding2d.padding(node->padding());
70 infer_padding2d.stride(node_stride);
71 infer_padding2d.window(node_window);
72
73 auto input_feature_shape = moco::as_feature_shape(input_shape, node->data_layout());
74 auto input_plane_shape = moco::make_plane_shape(input_feature_shape);
75
76 *maxPool2d_node->pad() = infer_padding2d(input_plane_shape);
77 *maxPool2d_node->stride() = node_stride;
78 *maxPool2d_node->window() = node_window;
79
80 INFO(l) << "Canonicalize TFMaxPool pad = T " << maxPool2d_node->pad()->top() << ", L "
81 << maxPool2d_node->pad()->left() << ", B " << maxPool2d_node->pad()->bottom() << ", R "
82 << maxPool2d_node->pad()->right() << std::endl;
83
84 // update graph
85 auto node_A = node->input();
86
87 // update connections
88 feature_enc->input(node_A);
89 maxPool2d_node->ifm(feature_enc);
90 feature_dec->input(maxPool2d_node);
91
92 // replace node
