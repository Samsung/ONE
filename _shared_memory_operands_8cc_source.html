24// To handle cases like Reshape->Reshape->Reshape... chain where the memory is shared.
25// In such a case we should re-assign indexes to the first Reshape input.
26void reassign_indexes_to_single_sources(
27 ir::OperandIndexMap<ir::OperandIndex> &shared_memory_operand_map)
28{
29 for (auto [shared_ind, source_ind] : shared_memory_operand_map)
30 {
31 bool other_source_found = false;
32 auto it = std::end(shared_memory_operand_map);
33 while ((it = shared_memory_operand_map.find(source_ind)) != std::end(shared_memory_operand_map))
34 {
35 source_ind = shared_memory_operand_map[source_ind];
36 other_source_found = true;
37 }
38 if (other_source_found)
39 {
40 shared_memory_operand_map[shared_ind] = source_ind;
41 }
42 }
43}

45bool is_memory_sharing_allowed(const ir::IGraph &graph, const ir::IOperation &op)
46{
47 const std::unordered_set<ir::OpCode> ops_with_possible_memory_sharing = {
48 ir::OpCode::Reshape, ir::OpCode::ExpandDims, ir::OpCode::Squeeze};
49
50 if (ops_with_possible_memory_sharing.find(op.opcode()) ==
51 std::end(ops_with_possible_memory_sharing))
52 {
53 return false;
54 }
55 if (graph.operands().at(op.getInputs().at(0)).info().isDynamic())
56 {
57 return false;
58 }
59 if (graph.operands().at(op.getOutputs().at(0)).info().isDynamic())
60 {
61 return false;
62 }
63 const auto op_input_output = {op.getInputs().at(0), op.getOutputs().at(0)};
64 const bool is_model_input_output = std::any_of(
65 std::begin(op_input_output), std::end(op_input_output), [&graph](const ir::OperandIndex &ind) {
66 return graph.getInputs().contains(ind) || graph.getOutputs().contains(ind);
67 });
68 return !is_model_input_output;
69};

73ir::OperandIndexMap<ir::OperandIndex> findSharedMemoryOperandIndexes(const ir::IGraph &graph)
74{
75 ir::OperandIndexMap<ir::OperandIndex> shared_memory_operand_map;
76 graph.operations().iterate([&](const ir::OperationIndex &, const ir::IOperation &op) {
77 if (is_memory_sharing_allowed(graph, op))
78 {
79 assert(op.getInputs().size() == 1 || op.getInputs().size() == 2);
80 assert(op.getOutputs().size() == 1);
81 shared_memory_operand_map[op.getOutputs().at(0)] = op.getInputs().at(0);
82 }
83 });
84 reassign_indexes_to_single_sources(shared_memory_operand_map);
85 return shared_memory_operand_map;
86}
