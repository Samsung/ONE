71template <typename Tensor, typename TrainableTensor, typename BackPropTensor,
72 typename GradientTensor>
73class PortableTensorRegistryTemplate : public backend::train::ITensorRegistry
74{
75public:
76 using TrainingTensors = std::tuple<TrainableTensor *, GradientTensor *>;
77
78public:
79 ITensor *getITensor(const ir::OperandIndex &index) override
80 {
81 auto _migrant_tensor = _migrant.find(index);
82 if (_migrant_tensor != _migrant.end())
83 return _migrant_tensor->second;
84 return getNativeITensor(index);
85 }
86
87 ITensor *getNativeITensor(const ir::OperandIndex &index) override
88 {
89 ITensor *tensor = getTrainableTensor(index);
90 if (tensor == nullptr)
91 tensor = getNonConstTensor(index);
92 return tensor;
93 }
94
95 ITensor *getBackPropITensor(const ir::OperandIndex &index) override
96 {
97 return getBackPropTensor(index);
98 }
99
100 ITensor *getGradientITensor(const ir::OperandIndex &index) override
101 {
102 return getGradientTensor(index);
103 }
104
105 IPortableTensor *getPortableTensor(const ir::OperandIndex &index)
106 {
107 auto tensor = _trainable.find(index);
108 if (tensor != _trainable.end())
109 {
110 if (tensor->second)
111 return tensor->second.get();
112 }
113 return getNonConstTensor(index);
114 }
115
116 Tensor *getNonConstTensor(const ir::OperandIndex &index)
117 {
118 auto tensor = _non_const.find(index);
119 if (tensor != _non_const.end())
120 return tensor->second.get();
121 return nullptr;
122 }
123
124 TrainableTensor *getTrainableTensor(const ir::OperandIndex &index)
125 {
126 auto tensor = _trainable.find(index);
127 if (tensor != _trainable.end())
128 return tensor->second.get();
129
130 return nullptr;
131 }
132
133 BackPropTensor *getBackPropTensor(const ir::OperandIndex &index)
134 {
135 auto tensor = _back_prop.find(index);
136 if (tensor != _back_prop.end())
137 return tensor->second.get();
138 return nullptr;
139 }
140
141 GradientTensor *getGradientTensor(const ir::OperandIndex &index)
142 {
143 auto tensor = _gradient.find(index);
144 if (tensor != _gradient.end())
145 return tensor->second.get();
146 return nullptr;
147 }
148
149 TrainingTensors getTrainingTensors(const ir::OperandIndex &index)
150 {
151 auto trainable = getTrainableTensor(index);
152 if (trainable == nullptr)
153 throw std::runtime_error{
154 "Tried to get a trainable tensor but the corresponding tensor does not exist."};
155
156 auto gradient = getGradientTensor(index);
157 if (gradient == nullptr)
158 throw std::runtime_error{
159 "Tried to get a gradient tensor but the corresponding tensor does not exist."};
160
161 return TrainingTensors{std::make_pair(trainable, gradient)};
162 }
163
164 bool setMigrantTensor(const ir::OperandIndex &index, IPortableTensor *tensor) override
165 {
166 assert(tensor != nullptr);
167 if (getITensor(index) != nullptr)
168 throw std::runtime_error{
169 "Tried to set a trainable tensor but another tensor already exists."};
170
171 _migrant[index] = tensor;
172 return true;
173 }
174
175 void setNonConstTensor(const ir::OperandIndex &index, std::unique_ptr<Tensor> tensor)
176 {
177 assert(tensor != nullptr);
178 if (getITensor(index) != nullptr)
179 throw std::runtime_error{
180 "Tried to set a trainable tensor but another tensor already exists."};
181
182 _non_const[index] = std::move(tensor);
183 }
184
185 void setTrainableTensor(const ir::OperandIndex &index, std::unique_ptr<TrainableTensor> tensor)
186 {
187 assert(tensor != nullptr);
188 if (getITensor(index) != nullptr)
189 throw std::runtime_error{
190 "Tried to set a trainable tensor but another tensor already exists."};
191
192 _trainable[index] = std::move(tensor);
193 }
194
195 void setBackPropTensor(const ir::OperandIndex &index, std::unique_ptr<BackPropTensor> tensor)
196 {
197 assert(tensor != nullptr);
198 auto itr = _back_prop.find(index);
199 if (itr != _back_prop.end())
200 throw std::runtime_error{"Tried to set a back propagation tensor but another back "
201 "propagation tensor already exists."};
202
203 _back_prop[index] = std::move(tensor);
204 }
205
206 void setGradientTensor(const ir::OperandIndex &index, std::unique_ptr<GradientTensor> tensor)
207 {
208 assert(tensor != nullptr);
209 auto itr = _gradient.find(index);
210 if (itr != _gradient.end())
211 throw std::runtime_error{
212 "Tried to set a gradient tensor but another gradient tensor already exists."};
213
214 _gradient[index] = std::move(tensor);
215 }
216
217 const ir::OperandIndexMap<std::unique_ptr<TrainableTensor>> &trainable_tensors()
218 {
219 return _trainable;
220 }
221 const ir::OperandIndexMap<std::unique_ptr<Tensor>> &nonconst_tensors() { return _non_const; }
222 const ir::OperandIndexMap<std::unique_ptr<Tensor>> &back_prop_tensors() { return _back_prop; }
223 const ir::OperandIndexMap<std::unique_ptr<GradientTensor>> &gradient_tensors()
224 {
225 return _gradient;
226 }
227
228private:
229 // Native tensors
230 ir::OperandIndexMap<std::unique_ptr<Tensor>> _non_const;
231 ir::OperandIndexMap<std::unique_ptr<TrainableTensor>> _trainable;
232
233 // Migrant tensors
234 ir::OperandIndexMap<IPortableTensor *> _migrant;
235
236 // Tensors for backpropagation
237 ir::OperandIndexMap<std::unique_ptr<BackPropTensor>> _back_prop;
238
239 // Tensors for updating trainable tensors
240 ir::OperandIndexMap<std::unique_ptr<GradientTensor>> _gradient;
241};
