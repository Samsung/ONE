104 IPortableTensor *getPortableTensor(const ir::OperandIndex &index)
105 {
106 auto tensor = _trainable.find(index);
107 if (tensor != _trainable.end())
108 {
109 if (tensor->second)

112 return getNonConstTensor(index);

132 DerivativeTensor *getDerivativeTensor(const ir::OperandIndex &index)

134 auto tensor = _derivative.find(index);

136 return tensor->second.get();
137 return nullptr;
138 }
139
140 GradientTensor *getGradientTensor(const ir::OperandIndex &index)

142 auto tensor = _gradient.find(index);

145 return nullptr;

147
148 TrainingTensors getTrainingTensors(const ir::OperandIndex &index)
149 {
150 auto trainable = getTrainableTensor(index);
151 if (trainable == nullptr)
152 throw std::runtime_error{
153 "Tried to get a trainable tensor but the corresponding tensor does not exist."};
154
155 auto gradient = getGradientTensor(index);
156 if (gradient == nullptr)
157 throw std::runtime_error{
158 "Tried to get a gradient tensor but the corresponding tensor does not exist."};
159
160 return TrainingTensors{std::make_pair(trainable, gradient)};
161 }
162
163 bool setMigrantTensor(const ir::OperandIndex &index, IPortableTensor *tensor) override
164 {
165 assert(tensor != nullptr);
166 if (getITensor(index) != nullptr)
167 throw std::runtime_error{
168 "Tried to set a trainable tensor but another tensor already exists."};
169
170 _migrant[index] = tensor;
171 return true;
172 }
173
174 void setNonConstTensor(const ir::OperandIndex &index, std::unique_ptr<Tensor> tensor)
175 {
176 assert(tensor != nullptr);
177 if (getITensor(index) != nullptr)
178 throw std::runtime_error{
179 "Tried to set a trainable tensor but another tensor already exists."};
180
181 _non_const[index] = std::move(tensor);
182 }
183
184 void setTrainableTensor(const ir::OperandIndex &index, std::unique_ptr<TrainableTensor> tensor)
185 {
186 assert(tensor != nullptr);
187 if (getITensor(index) != nullptr)
188 throw std::runtime_error{
189 "Tried to set a trainable tensor but another tensor already exists."};
190
191 _trainable[index] = std::move(tensor);
192 }
193
194 void setDerivativeTensor(const ir::OperandIndex &index, std::unique_ptr<DerivativeTensor> tensor)
195 {
196 assert(tensor != nullptr);
197 auto itr = _derivative.find(index);
198 if (itr != _derivative.end())
199 throw std::runtime_error{
200 "Tried to set a derivative tensor but another derivative tensor already exists."};
201
202 _derivative[index] = std::move(tensor);
203 }
204
205 void setGradientTensor(const ir::OperandIndex &index, std::unique_ptr<GradientTensor> tensor)
206 {
207 assert(tensor != nullptr);
208 auto itr = _gradient.find(index);
209 if (itr != _gradient.end())
210 throw std::runtime_error{
211 "Tried to set a gradient tensor but another gradient tensor already exists."};
212
213 _gradient[index] = std::move(tensor);
214 }
215
216 const ir::OperandIndexMap<std::unique_ptr<TrainableTensor>> &trainable_tensors()
217 {
218 return _trainable;
219 }
220 const ir::OperandIndexMap<std::unique_ptr<Tensor>> &nonconst_tensors() { return _non_const; }
221 const ir::OperandIndexMap<std::unique_ptr<Tensor>> &derivative_tensors() { return _derivative; }
222 const ir::OperandIndexMap<std::unique_ptr<GradientTensor>> &gradient_tensors()
223 {
224 return _gradient;
225 }
226
227private:
228 // Native tensors
229 ir::OperandIndexMap<std::unique_ptr<Tensor>> _non_const;
230 ir::OperandIndexMap<std::unique_ptr<TrainableTensor>> _trainable;
231
232 // Migrant tensors
233 ir::OperandIndexMap<IPortableTensor *> _migrant;
234
235 // Tensors for backpropagation
236 ir::OperandIndexMap<std::unique_ptr<DerivativeTensor>> _derivative;
237
238 // Tensors for updating trainable tensors
239 ir::OperandIndexMap<std::unique_ptr<GradientTensor>> _gradient;
240};
