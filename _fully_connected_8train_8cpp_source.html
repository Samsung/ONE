30Status computeGradients(const circle::Operator *op, CircleReader *reader,
31 GradientCalculationStorage *gradient_calculation_storage,
32 const circle::Tensor *weight, TrainableWeightStorage *weight_storage)
33{
34
35 const auto input_index = op->inputs()->operator[](0);
36 const auto output_index = op->outputs()->operator[](0);
37
38 assert(input_index != -1);
39 assert(output_index != -1);
40
41 const auto input = reader->tensors()[input_index];
42 const auto output = reader->tensors()[output_index];
43
44 assert(input != nullptr);
45 assert(output != nullptr);
46
47 uint8_t *input_data = nullptr;
48 Status status = gradient_calculation_storage->getDataByTensor(input, &input_data);
49
50 assert(input_data != nullptr);
51
52 float *input_data_float = reinterpret_cast<float *>(input_data);
53
54 if (status != Ok)
55 return status;
56
57 // output_data - save derivative of the error function by output activation for the current layer
58 uint8_t *output_data = nullptr;
59 status = gradient_calculation_storage->getDataByTensor(output, &output_data);
60
61 assert(input_data != nullptr);
62
63 float *output_data_float = reinterpret_cast<float *>(output_data);
64
65 if (status != Ok)
66 return status;
67
68 uint8_t *gradients_values = nullptr;
69
70 const auto rows = Tensor::dim(weight, 0);
71 const auto cols = Tensor::dim(weight, 1);
72
73 status = gradient_calculation_storage->getGradients(weight, &gradients_values);
74 float *gradient_values_float = reinterpret_cast<float *>(gradients_values);
75 assert(gradient_values_float != nullptr);
76
77 if (status != Ok or gradient_values_float == nullptr)
78 return status;
79
80 // Update output activation gradient with activations:
81 const auto options = op->builtin_options_as_FullyConnectedOptions();
82 const auto activation_type = luci_actfunc(options->fused_activation_function());
83 switch (activation_type)
84 {
85 case FusedActFunc::RELU:
86 {
87 for (int row = 0; row < rows; ++row)
88 output_data_float[row] = std::max(output_data_float[row], 0.f);
89 }
90 break;
91 case FusedActFunc::NONE:
92 break;
93 default:
94 assert(false && "Not supported type now");
95 return Error;
96 }
97
98 // Update weight gradients
99 for (int row = 0; row < rows; ++row)
100 {
101 for (int col = 0; col < cols; ++col)
102 {
103 gradient_values_float[col + row * cols] += output_data_float[row] * input_data_float[col];
104 }
105 }
106
107 uint8_t *weight_data = nullptr;
108
109 status = weight_storage->getTrainWeightDataByTensor(weight, &weight_data);
110 if (status != Ok)
111 return status;
112
113 assert(weight_data != nullptr);
114
115 float *weight_data_float = reinterpret_cast<float *>(weight_data);
116
117 // calculate derivative of the error function by input activation for the current layer
118 for (int col = 0; col < cols; ++col)
119 {
120 input_data_float[col] = 0;
121 }
122
123 for (int row = 0; row < rows; ++row)
124 {
125 for (int col = 0; col < cols; ++col)
126 {
127 input_data_float[row] += weight_data_float[row + col * rows] * output_data_float[col];
128 }
129 }
130
131 return Ok;
132}

134Status updateWeights(const circle::Operator *op, CircleReader *reader,
135 GradientCalculationStorage *gradient_calculation_storage,
136 const TrainingSettings &settings, TrainableWeightStorage *weight_storage,
137 const circle::Tensor *weight)
138{
139 uint8_t *weight_data = nullptr;
140 Status status = weight_storage->getTrainWeightDataByTensor(weight, &weight_data);
141
142 assert(weight_data != nullptr);
143
144 float *weight_data_float = reinterpret_cast<float *>(weight_data);
145
146 if (status != Ok)
147 return status;
148
149 uint8_t *gradients_values = nullptr;
150
151 const auto rows = Tensor::dim(weight, 0);
152 const auto cols = Tensor::dim(weight, 1);
153
154 status = gradient_calculation_storage->getGradients(weight, &gradients_values);
155 float *gradient_values_float = nullptr;
156
157 gradient_values_float = reinterpret_cast<float *>(gradients_values);
158
159 assert(gradient_values_float != nullptr);
160
161 if (status != Ok or gradient_values_float == nullptr)
162 return status;
163
164 for (int row = 0; row < rows; ++row)
165 {
166 for (int col = 0; col < cols; ++col)
167 {
168 weight_data_float[col + row * cols] -=
169 settings.learning_rate * gradient_values_float[col + row * cols] / settings.batch_size;
170 }
171 }
172
173 return Ok;
174}
