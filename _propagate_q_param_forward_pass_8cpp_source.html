30bool copy_qparam(luci::CircleQuantParam *src, luci::CircleQuantParam *dst)
31{
32 assert(src->scale.size() == dst->scale.size());
33 assert(src->zerop.size() == dst->zerop.size());
34
35 // src and dst have the same qparam
36 if (std::equal(src->scale.begin(), src->scale.end(), dst->scale.begin()) &&
37 std::equal(src->zerop.begin(), src->zerop.end(), dst->zerop.begin()) &&
38 src->quantized_dimension == dst->quantized_dimension)
39 return false;
40
41 dst->scale.assign(src->scale.begin(), src->scale.end());
42 dst->zerop.assign(src->zerop.begin(), src->zerop.end());
43 dst->quantized_dimension = src->quantized_dimension;
44 return true;
45}

47bool copy_qparam(luci::CircleNode *src, luci::CircleNode *dst)
48{
49 // Skip nodes that do not have quantparams
50 auto src_qparam = src->quantparam();
51 if (not src_qparam)
52 return false;
53
54 auto dst_qparam = dst->quantparam();
55 if (not dst_qparam)
56 return false;
57
58 return copy_qparam(src_qparam, dst_qparam);
59}

61// Visitor to propagate quantization parameters
62struct PropagateQParamForward final : public luci::CircleNodeMutableVisitor<bool>
63{
64 PropagateQParamForward() = default;
65
66 bool visit(luci::CircleNode *) { return false; }
67
68 bool visit(luci::CircleGather *node)
69 {
70 auto input_node = loco::must_cast<luci::CircleNode *>(node->params());
71 return copy_qparam(input_node, node);
72 }
73
74 bool visit(luci::CircleReshape *node)
75 {
76 auto input_node = loco::must_cast<luci::CircleNode *>(node->tensor());
77 return copy_qparam(input_node, node);
78 }
79
80 bool visit(luci::CircleTranspose *node)
81 {
82 auto input_node = loco::must_cast<luci::CircleNode *>(node->a());
83 return copy_qparam(input_node, node);
84 }
85
86 bool visit(luci::CircleStridedSlice *node)
87 {
88 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
89 return copy_qparam(input_node, node);
90 }
91
92 bool visit(luci::CircleSplitOut *node)
93 {
94 auto split = loco::must_cast<luci::CircleSplit *>(node->input());
95 auto input_node = loco::must_cast<luci::CircleNode *>(split->input());
96 return copy_qparam(input_node, node);
97 }
98
99 bool visit(luci::CircleSplitVOut *node)
100 {
101 auto splitv = loco::must_cast<luci::CircleSplitV *>(node->input());
102 auto input_node = loco::must_cast<luci::CircleNode *>(splitv->input());
103 return copy_qparam(input_node, node);
104 }
105
106 bool visit(luci::CircleUnpackOut *node)
107 {
108 auto unpack = loco::must_cast<luci::CircleUnpack *>(node->input());
109 auto input_node = loco::must_cast<luci::CircleNode *>(unpack->value());
110 return copy_qparam(input_node, node);
111 }
112
113 // Propagate qparam across Quantize op to ensure
114 // special qparams (pre-defined values, integer scale)
115 bool visit(luci::CircleQuantize *node)
116 {
117 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
118
119 // Skip if input_node is not quantized activation
120 if (input_node->dtype() != loco::DataType::U8 and input_node->dtype() != loco::DataType::S16)
121 return false;
122
123 // If input_node and node have the same dtype, Quantize op
124 // will do rescale, not requantize for mixed-precision
125 if (input_node->dtype() == node->dtype())
126 return false;
127
128 assert(node->dtype() == loco::DataType::U8 or node->dtype() == loco::DataType::S16);
129
130 auto prev_qparam = node->quantparam();
131 assert(prev_qparam);
132 assert(prev_qparam->scale.size() == 1);
133 assert(prev_qparam->zerop.size() == 1);
134
135 const auto prev_scale = prev_qparam->scale[0];
136 const auto prev_zerop = prev_qparam->zerop[0];
137
138 auto qtype = luci::activation_qtype(input_node);
139 switch (qtype)
140 {
141 case luci::ActivationQType::PreDefinedLogistic:
142 case luci::ActivationQType::PreDefinedTanh:
143 case luci::ActivationQType::PreDefinedSoftmax:
144 node->quantparam(luci::make_predefined_qparam(qtype, node->dtype()));
145 break;
146 case luci::ActivationQType::IntScale:
147 luci::set_int_scale(node);
148 break;
149 default:
150 // This assert ensures this switch-satement handles all ActivationQTypes
151 // TODO Find a better design to remove coupling with ActivationQType
152 assert(qtype == luci::ActivationQType::MinMax);
153 break;
154 }
155
156 assert(node->quantparam());
157 assert(node->quantparam()->scale.size() == 1);
158 assert(node->quantparam()->zerop.size() == 1);
159
160 const auto scale = node->quantparam()->scale[0];
161 const auto zerop = node->quantparam()->zerop[0];
162
163 // Compare qparam with saved values to detect update
164 return scale != prev_scale or zerop != prev_zerop;
165 }
166};

173bool PropagateQParamForwardPass::run(loco::Graph *g)
174{
175 bool changed = false;
176 LOGGER(l);
177 for (auto node : loco::active_nodes(loco::output_nodes(g)))
178 {
179 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
180 INFO(l) << "PropagateQParamForwardPass visit node: " << circle_node->name() << std::endl;
181
182 PropagateQParamForward pqp;
183 if (circle_node->accept(&pqp))
184 changed = true;
185
186 if (_TF_style_maxpool)
187 {
188 if (auto maxpool = dynamic_cast<luci::CircleMaxPool2D *>(node))
189 {
190 auto input = loco::must_cast<luci::CircleNode *>(maxpool->value());
191 copy_qparam(input, maxpool);
192 }
193 }
194 }
195
196 return changed;
197}
