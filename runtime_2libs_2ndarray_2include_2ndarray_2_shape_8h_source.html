39 Shape(std::initializer_list<size_t> list) noexcept : _dims{}, _rank(list.size())

43
44 size_t dim(int i) const noexcept { return _dims.at(i); }

48 size_t element_count() const noexcept

50 uint32_t res = 1;

55 }
