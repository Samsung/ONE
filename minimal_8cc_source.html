21uint64_t num_elems(const nnfw_tensorinfo *ti)
22{
23 uint64_t n = 1;
24 for (uint32_t i = 0; i < ti->rank; ++i)
25 {
26 n *= ti->dims[i];
27 }
28 return n;
29}
