32void prepareFC(ExecEnv *env, const ir::Operation &node)
33{
34 const auto in_index = node.getInputs().at(ir::operation::FullyConnected::INPUT);
35 const auto kernel_index = node.getInputs().at(ir::operation::FullyConnected::WEIGHT);
36 const auto bias_index = node.getInputs().at(ir::operation::FullyConnected::BIAS);
37 const auto out_index = node.getOutputs().at(0);
38
39 const auto in_tensor = env->tensorAt(in_index);
40 const auto kernel_tensor = env->tensorAt(kernel_index);
41 const auto bias_tensor = env->tensorAt(bias_index);
42
43 UNUSED_RELEASE(in_tensor);
44 UNUSED_RELEASE(kernel_tensor);
45 UNUSED_RELEASE(bias_tensor);
46
47 assert(in_tensor->getShape().rank() >= 2);
48 assert(kernel_tensor->getShape().rank() == 2);
49 assert(bias_tensor->getShape().rank() == 1);
50
51 const auto input_size_with_batch = in_tensor->num_elements();
52 const auto num_units = kernel_tensor->getShape().dim(0);
53 const auto input_size = kernel_tensor->getShape().dim(1);
54 const int32_t batch_size = input_size_with_batch / input_size;
55 assert(input_size_with_batch % input_size == 0);
56 assert(num_units == bias_tensor->getShape().dim(0));
57
58 // Make output tensor info
59 ir::Shape output_shape(2);
60 output_shape.dim(0) = batch_size;
61 output_shape.dim(1) = num_units;
62 const auto out_info =
63 ir::OperandInfo::createStaticInfo(output_shape, in_tensor->tensorInfo().typeInfo());
64 env->allocateIfNeeded(out_index, out_info);
65
66 auto out_tensor = env->tensorAt(out_index);
67 UNUSED_RELEASE(out_tensor);
68
69 // Handle same ifm & ofm data type only
70 assert(in_tensor->data_type() == out_tensor->data_type());
71 assert(out_tensor->getShape().rank() == 2);
72 assert(out_tensor->getShape().dim(0) == batch_size);
73 assert(out_tensor->getShape().dim(1) == num_units);
74}

76void invoke(const ITensor *ifm_tensor, const ITensor *ker_tensor, const ITensor *bias_tensor,
77 const ITensor *ofm_tensor, const ir::operation::FullyConnected::Param &param)
78{
79 const auto ifm_buffer = ifm_tensor->bufferRO();
80 const auto ker_buffer = ker_tensor->bufferRO();
81 const auto bias_buffer = bias_tensor->bufferRO();
82 auto ofm_buffer = ofm_tensor->buffer();
83
84 // Calculate
85 nnfw::cker::FullyConnectedParams cker_param;
86 cker_param.activation = convertActivationType(param.activation);
87 const auto cker_ifm_shape = convertShape(ifm_tensor->tensorInfo().shape());
88 const auto cker_ker_shape = convertShape(ker_tensor->tensorInfo().shape());
89 const auto cker_bias_shape = convertShape(bias_tensor->tensorInfo().shape());
90 const auto cker_ofm_shape = convertShape(ofm_tensor->tensorInfo().shape());
91 const float *ifm_ptr = reinterpret_cast<const float *>(ifm_buffer);
92 const float *ker_ptr = reinterpret_cast<const float *>(ker_buffer);
93 const float *bias_ptr = reinterpret_cast<const float *>(bias_buffer);
94 float *ofm_ptr = reinterpret_cast<float *>(ofm_buffer);
95
96 nnfw::cker::FullyConnected(cker_param, cker_ifm_shape, ifm_ptr, cker_ker_shape, ker_ptr,
97 cker_bias_shape, bias_ptr, cker_ofm_shape, ofm_ptr);
98}

100void invokeFC(const ExecEnv *env, const ir::Operation &node)
101{
102 const auto &conv_node =
103 nnfw::misc::polymorphic_downcast<const ir::operation::FullyConnected &>(node);
104
105 const auto ifm_index = node.getInputs().at(ir::operation::FullyConnected::INPUT);
106 const auto ker_index = node.getInputs().at(ir::operation::FullyConnected::WEIGHT);
107 const auto bias_index = node.getInputs().at(ir::operation::FullyConnected::BIAS);
108 const auto ofm_index = node.getOutputs().at(0);
109
110 const auto ifm_tensor = env->tensorAt(ifm_index);
111 const auto ker_tensor = env->tensorAt(ker_index);
112 const auto bias_tensor = env->tensorAt(bias_index);
113 const auto ofm_tensor = env->tensorAt(ofm_index);
114
115 const auto data_type = ifm_tensor->data_type();
116 if (data_type == ir::DataType::FLOAT32)
117 {
118 invoke(ifm_tensor, ker_tensor, bias_tensor, ofm_tensor, conv_node.param());
119 }
120 else
121 {
122 throw std::runtime_error{"NYI: Support float only"};
123 }
124}
