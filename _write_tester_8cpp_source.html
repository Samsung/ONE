84/*
85 * @brief WriteTester main
86 *
87 * Give two Circle file as an argument
88 *
89 * This will use luci_import to read the first file and get loco graph
90 * With the graph, this will use luci_export to write to the second file
91 * Like ReadTester, LUCI_LOG=1 environment variable is available to dump the graph
92 */
93int entry(int argc, char **argv)
94{
95 if (argc != 3)
96 {
97 show_error_message(argv[0], std::cerr, "In/Out Circle file path is not specified");
98 return 255;
99 }
100
101 std::string input_path = argv[1];
102 std::string output_path = argv[2];
103
104 std::cout << "[INFO] Circle from '" << input_path << "' to '" << output_path << "'" << std::endl;
105
106 auto module = ReadModule(input_path);
107 if (module == nullptr)
108 return EXIT_FAILURE;
109
110 // Export to output Circle file
111 luci::CircleExporter exporter;
112
113 CircleExpContract contract(module.get(), output_path);
114
115 return exporter.invoke(&contract) ? 0 : 255;
116}
