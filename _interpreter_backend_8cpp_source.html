57static void writeTensorToHDF5File(const TensorVariant &tensor, std::string tensor_name,
58 const std::string &destination)
59{
60
61 // Prepare shape, rank, dims, numElems
62 auto &shape = tensor.getShape();
63 const int32_t rank = shape.rank();
64 hsize_t dims[rank];
65 for (int32_t axis = 0; axis < rank; ++axis)
66 {
67 dims[axis] = static_cast<hsize_t>(shape.dim(axis));
68 }
69
70 // Create array from tensor
71 std::vector<char> values;
72 const auto elem_size = tensor.getElementSize();
73 values.resize(elem_size * shape.numElements());
74 char *values_ptr = values.data();
75 ShapeRange out_range(shape);
76 for (auto &out_idx : out_range)
77 {
78 std::memcpy(values_ptr, tensor.at(out_idx), elem_size);
79 values_ptr += elem_size;
80 }
81
82 // Backslashes are not allowed in tensor names
83 std::replace(tensor_name.begin(), tensor_name.end(), '/', '_');
84 std::string filename = destination + "/" + tensor_name + ".hdf5";
85
86 // Write to .hdf5 file
87 H5::H5File h5File(filename, H5F_ACC_TRUNC);
88 H5::DataSpace dataspace(rank, dims);
89 H5::DataType h5_data_type;
90
91 if (tensor.getDataType() == DataType::FLOAT32)
92 h5_data_type = H5::PredType::NATIVE_FLOAT;
93 else if (tensor.getDataType() == DataType::UINT8)
94 h5_data_type = H5::PredType::NATIVE_UINT8;
95 else
96 throw std::runtime_error("NYI writing that DataType!");
97
98 auto dataset = h5File.createDataSet(tensor_name, h5_data_type, dataspace);
99 dataset.write(values.data(), h5_data_type);
100}

104static TensorVariant readTensorFromFile(const std::string &filename, const TensorType &type)
105{
106 const std::size_t input_data_size =
107 type.getShape().numElements() * getDataTypeSize(type.getElementType());
108
109 std::ifstream stream(filename, std::ios::in | std::ios::binary);
110 if (stream.fail())
111 throw std::runtime_error("Couldn't open file \"" + filename + "\".");
112
113 stream.seekg(0, std::ios::end);
114 std::streampos end = stream.tellg();
115 stream.seekg(0, std::ios::beg);
116 std::streampos begin = stream.tellg();
117 int64_t file_size = end - begin;
118
119 if (static_cast<std::size_t>(file_size) != input_data_size)
120 throw std::runtime_error("File \"" + filename +
121 "\" has incorrect size: " + std::to_string(file_size) +
122 "(expected: " + std::to_string(input_data_size) + ").");
123
124 std::unique_ptr<char[]> data(new char[input_data_size]);
125 stream.read(data.get(), input_data_size);
126 if (stream.fail())
127 throw std::runtime_error("Couldn't read file \"" + filename + "\".");
128
129 return TensorVariant(type, data.get());
130}

137void InterpreterBackend::run(mir::Graph *graph)
138{
139 assert(graph);
140
141 mir_interpreter::MIRInterpreter interpreter;
142
143 for (const auto *input_op : graph->getInputs())
144 {
145 const Operation::Output *input = input_op->getOutput(0);
146
147 std::string tensor_name = input->getName();
148 assert(!tensor_name.empty());
149 std::replace(tensor_name.begin(), tensor_name.end(), '/', '_');
150 std::string filename = _input_dir + "/" + tensor_name + ".dat";
151
152 TensorVariant tensor = readTensorFromFile(filename, input->getType());
153 interpreter.setTensor(input, std::move(tensor));
154 }
155
156 graph->accept(&interpreter);
157
158 for (const auto *output_op : graph->getOutputs())
159 {
160 const auto &output_name = output_op->getInput(0)->getName();
161
162#ifdef NNC_HDF5_SUPPORTED
163 const auto &tensor = interpreter.getTensor(output_op->getInput(0));
164 writeTensorToHDF5File(tensor, output_name, _output_dir);
165#else
166 std::cout << "Result <" << output_name << "> wasn't saved, due to lack of HDF5" << std::endl;
167#endif // NNC_HDF5_SUPPORTED
168 }
169}
