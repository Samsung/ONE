335void InsertScaleShift::run(loco::Graph *g)
336{
337 // Create a map for pattern matching
338 // { back(string) -> EqualizationPattern*}
339 // This assumes that each EqualizePattern has a unique 'back'
340 std::map<std::string, EqualizePattern *> pattern_by_back;
341 {
342 for (auto &pattern : _patterns)
343 {
344 auto back = pattern.back;
345 pattern_by_back[back] = &pattern;
346 }
347 }
348
349 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
350 {
351 auto cnode = loco::must_cast<luci::CircleNode *>(node);
352
353 if (pattern_by_back.find(cnode->name()) == pattern_by_back.end())
354 continue;
355
356 auto pattern = pattern_by_back.at(cnode->name());
357 InsertScaleShiftVisitor issv(pattern);
358 cnode->accept(&issv);
359 }
360}
