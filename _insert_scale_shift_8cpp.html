337void InsertScaleShift::run(loco::Graph *g)
338{
339 // Create a map for pattern matching
340 // { back(string) -> EqualizationPattern*}
341 // This assumes that each EqualizePattern has a unique 'back'
342 std::map<std::string, EqualizePattern *> pattern_by_back;
343 {
344 for (auto &pattern : _patterns)
345 {
346 auto back = pattern.back;
347 pattern_by_back[back] = &pattern;
348 }
349 }
350
351 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
352 {
353 auto cnode = loco::must_cast<luci::CircleNode *>(node);
354
355 if (pattern_by_back.find(cnode->name()) == pattern_by_back.end())
356 continue;
357
358 auto pattern = pattern_by_back.at(cnode->name());
359 InsertScaleShiftVisitor issv(pattern);
360 cnode->accept(&issv);
361 }
362}
