30void ModuleLoader::load(bool use_static_memory_manager)
31{
32 const circle::Model *model = circle::GetModel(_model_data_raw);
33
34 CircleReader reader;
35 if (!reader.parse(model))
36 assert(false && "Error during parse");
37
38 for (size_t i = 0; i < reader.num_subgraph(); ++i)
39 {
40 _runtime_graphs.emplace_back(
41 _runtime_module->addGraph(_memory_manager, use_static_memory_manager));
42 }
43
44 for (size_t i = 0; i < reader.num_subgraph(); ++i)
45 {
46 if (!reader.select_subgraph(i))
47 assert(false && "Error during select subgraph");
48 IBaseRuntimeGraph *runtime_graph = _runtime_graphs.at(i);
49 GraphLoader loader(&reader, runtime_graph, _memory_manager, &_index_to_tensor);
50
51 loader.initInputTensors(use_static_memory_manager);
52 loader.loadTensors(use_static_memory_manager);
53 loader.loadOperators(use_static_memory_manager);
54 }
55
56 // For Dynamic Memory manager we build memory allocate/deallocate plan and then configure kernels.
57 // For Static Memory manager we only configure kernels.
58 if (not use_static_memory_manager)
59 {
60 // Dynamic memory manager case
61 for (size_t i = 0; i < reader.num_subgraph(); ++i)
62 {
63 IBaseRuntimeGraph *runtime_graph = _runtime_graphs.at(i);
64 runtime_graph->configure();
65 }
66 }
67 else
68 {
69 // Static memory manager case
70 for (size_t i = 0; i < reader.num_subgraph(); ++i)
71 {
72 _runtime_graphs.at(i)->configure_kernels();
73 }
74 }
75}
