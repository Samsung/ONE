52template <typename T_MemoryManager, typename T_PoolManager, typename T_LifetimeManager,
53 typename T_Allocator>
54class AclInternalBufferManager : public IInternalBufferManager
55{
56public:
57 AclInternalBufferManager() : _allocator{nullptr}
58 {
59 std::shared_ptr<T_LifetimeManager> lifetime_mgr = std::make_shared<T_LifetimeManager>();
60 std::shared_ptr<T_PoolManager> pool_mgr = std::make_shared<T_PoolManager>();
61
62 _internal_manager = std::make_shared<T_MemoryManager>(lifetime_mgr, pool_mgr);
63 assert(_internal_manager);
64 }
65
66 virtual ~AclInternalBufferManager() = default;
67
71 void allocate(void) override
72 {
73 _allocator = std::make_shared<T_Allocator>();
74 _internal_manager->populate(*_allocator, 1);
75 }
76
80 void deallocate(void) override { _internal_manager->clear(); }
81
85 std::shared_ptr<::arm_compute::IMemoryManager> internal_buffer_manager(void) override
86 {
87 return _internal_manager;
88 }
89
90private:
91 std::shared_ptr<T_Allocator> _allocator;
92 std::shared_ptr<T_MemoryManager> _internal_manager;
93};
