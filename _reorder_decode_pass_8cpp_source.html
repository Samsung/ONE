64void gather_candidates(loco::Graph *g, std::queue<loco::FeatureDecode *> &q)
65{
66 Collector collector{&q};
67
68 for (auto node : loco::all_nodes(g))
69 {
70 if (node->dialect() == loco::CanonicalDialect::get())
71 {
72 auto canonical_node = loco::must_cast<loco::CanonicalNode *>(node);
73 canonical_node->accept(&collector);
74 }
75 }
76}

83bool ReorderDecodePass<loco::TensorBiasAdd>::run(loco::Graph *g)
84{
85 std::queue<loco::FeatureDecode *> q;
86
87 gather_candidates(g, q);
88
89 bool changed = false;
90
91 while (!q.empty())
92 {
93 auto cur_decode = q.front();
94 q.pop();
95
96 // Collector IS EXPECTED TO guarantee this property
97 assert(cur_decode->input() != nullptr);
98
99 for (auto u : loco::succs(cur_decode))
100 {
125 if (loco::succs(u).empty())
126 {
127 continue;
128 }
129
130 // Q. Is it better to create an independent transform for this rewriting rule?
131 if (isTensorBiasAdd(u))
132 {
133 auto old_badd = loco::must_cast<loco::TensorBiasAdd *>(u);
134
162 auto new_badd = g->nodes()->create<loco::FeatureBiasAdd>();
163 auto new_decode = g->nodes()->create<loco::FeatureDecode>();
164
165 new_badd->value(cur_decode->input());
166 new_badd->bias(old_badd->bias());
167
168 new_decode->input(new_badd);
169 new_decode->decoder(cur_decode->decoder()->clone());
170
171 loco::replace(u).with(new_decode);
172
173 // Enque FeatureDeocde(1') for the further optimization.
174 q.push(new_decode);
175
176 changed = true;
177 }
178 }
179 }
180
181 return changed;
182}
