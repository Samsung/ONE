30// struct for additional information for the node. it helps build allocations plan for nodes.
31struct AllocationNodeInformation
32{
33
34 AllocationNodeInformation()
35 {
36 offset = 0;
37 size = 0;
38 node_num = -1;
39 first_node = -1;
40 last_node = -1;
41 is_temp = false;
42 breadth = 0;
43 }
44 // memory offset from the beginning of the buffer
45 uint32_t offset;
46 // node required size
47 uint32_t size;
48 // the value assigned to the node
49 uint32_t node_num;
50 // the value of the node_num of the node when current node first use.
51 // Used to build the usage interval of the current node
52 uint32_t first_node;
53 // the value of the node_num of the node when current node last use.
54 // Used to build the usage interval of the current node
55 uint32_t last_node;
56 // is the current node temporary or not
57 bool is_temp;
58 // Breadth is a sum of live tensors sizes at the moment of execution of given node
59 uint32_t breadth;
60
61 bool operator<(const AllocationNodeInformation &other) const { return offset < other.offset; }
62};

75 : _graph(graph), _runtime_type(runtime_type), _allocating_mode(allocating_mode)
