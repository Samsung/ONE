31void KernelGenerator::visit(const ir::operation::RmsNorm &node)
32{
33 const auto ofm_index{node.getOutputs().at(0)};
34 const auto ifm_index{node.getInputs().at(ir::operation::RmsNorm::Input::INPUT)};
35 const auto gamma_index{node.getInputs().at(ir::operation::RmsNorm::Input::GAMMA)};
36
37 auto ofm_tensor = _tensor_reg->getPortableTensor(ofm_index);
38 auto ifm_tensor = _tensor_reg->getPortableTensor(ifm_index);
39 auto gamma_tensor = _tensor_reg->getPortableTensor(gamma_index);
40 auto epsilon = node.param().epsilon;
41
42 auto fn = std::make_unique<ops::RmsNormLayer>();
43
44 fn->configure(ifm_tensor, gamma_tensor, epsilon, ofm_tensor);
45
46 _return_fn = std::move(fn);
47}

66void RmsNormLayer::run()
67{
68 switch (_input->data_type())
69 {
70 case OperandType::FLOAT32:
71 nnfw::cker::RmsNormParams param;
72 param.epsilon = _epsilon;
73 nnfw::cker::RmsNorm(param, getShape(_input), getBuffer<float>(_input), getShape(_gamma),
74 getBuffer<float>(_gamma), getShape(_output), getBuffer<float>(_output));
75 break;
76
77 default:
78 throw std::runtime_error{"RmsNorm: Unsupported data type"};
79 }
