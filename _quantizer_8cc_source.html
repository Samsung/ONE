43void fillQuantizeOptionParam(QuantizerOptions *options, QuantizeType qtype)
44{
45 std::string output_type = "";
46 switch (qtype)
47 {
48 case QuantizeType::ODC_QTYPE_U8_ASYM:
49 output_type = "uint8";
50 break;
51 case QuantizeType::ODC_QTYPE_I16_SYM:
52 output_type = "int16";
53 break;
54 case QuantizeType::ODC_QTYPE_WO_I8_SYM:
55 output_type = "wo_int8";
56 break;
57 case QuantizeType::ODC_QTYPE_WO_I16_SYM:
58 output_type = "wo_int16";
59 break;
60 default:
61 throw std::runtime_error("Invalid quantization type");
62 }
63 options->param(QuantizerOptions::AlgorithmParameters::Quantize_output_model_dtype, output_type);
64 options->param(QuantizerOptions::AlgorithmParameters::Quantize_input_model_dtype, "float32");
65 options->param(QuantizerOptions::AlgorithmParameters::Quantize_granularity, "channel");
66
67 if (qtype == QuantizeType::ODC_QTYPE_U8_ASYM)
68 {
69 options->param(QuantizerOptions::AlgorithmParameters::Quantize_input_type, "uint8");
70 options->param(QuantizerOptions::AlgorithmParameters::Quantize_output_type, "uint8");
71 }
72 else if (qtype == QuantizeType::ODC_QTYPE_I16_SYM)
73 {
74 options->param(QuantizerOptions::AlgorithmParameters::Quantize_input_type, "int16");
75 options->param(QuantizerOptions::AlgorithmParameters::Quantize_output_type, "int16");
76 }
77}

81int Quantizer::quantize(const char *in, const char *out, QuantizeType qtype)
82{
83 if (not in || not out)
84 return 1;
85
86 bool full_quantize = false;
87 if (qtype == QuantizeType::ODC_QTYPE_U8_ASYM || qtype == QuantizeType::ODC_QTYPE_I16_SYM)
88 full_quantize = true;
89
90 // Load model from the file
91 luci::ImporterEx importerex;
92 auto module = importerex.importVerifyModule(std::string(in));
93 if (module.get() == nullptr)
94 return 1;
95
96 luci::CircleQuantizer quantizer;
97 auto options = quantizer.options();
98 if (options == nullptr)
99 return 1;
100
101 // Fill quantization type param
102 fillQuantizeOptionParam(options, qtype);
103
104 // Additional phase for full quantization
105 if (full_quantize)
106 {
107 // Fake quantization
108 options->enable(QuantizerOptions::Algorithm::QuantizeDequantizeWeights);
109 for (size_t idx = 0; idx < module->size(); ++idx)
110 {
111 auto graph = module->graph(idx);
112
113 // quantize the graph
114 quantizer.quantize(graph);
115 }
116
117 // Record minmax by minmax-embedder
118 // TODO use workspace to find minmax file
119 auto minmax_path = util::getConfigString(util::config::MINMAX_FILEPATH);
120 Embedder().embed(module.get(), minmax_path, {1.f, 99.f});
121 }
122
123 if (full_quantize)
124 options->enable(QuantizerOptions::Algorithm::QuantizeWithMinMax);
125 else
126 options->enable(QuantizerOptions::Algorithm::QuantizeWeights);
127
128 for (size_t idx = 0; idx < module->size(); ++idx)
129 {
130 auto graph = module->graph(idx);
131
132 // quantize the graph
133 quantizer.quantize(graph);
134
135 // Skip validate
136 // TODO Validate if needed
137#if 0
138 if (!luci::validate(graph))
139 {
140 std::cerr << "ERROR: Quantized graph is invalid" << std::endl;
141 return 1;
142 }
143#endif
144 }
145
146 // Export to output Circle file
147 luci::CircleExporter exporter;
148 luci::CircleFileExpContract contract(module.get(), std::string(out));
149
150 if (!exporter.invoke(&contract))
151 return 1;
152
153 // Return 0 when luci::CircleQuantizer::Options::Algorithm::QuantizeWeights is ready
154 return 0;
155}
