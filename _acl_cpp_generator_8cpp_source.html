37void AclCppCodeGenerator::run(mir::Graph *data)
38{
39 mir::Graph *g = data;
40 assert(g);
41
42 // Create a directory for generated artifact files.
43 fs::create_directory(_output_dir);
44
45 string base_path = _output_dir + "/" + _artifact_name;
46 string code_path = base_path + ".cpp";
47 string decl_path = base_path + ".h";
48 string par_path = base_path + ".par";
49
50 // Create the source and header files output streams.
51 ofstream code_out(code_path);
52
53 if (code_out.fail())
54 throw AclCppException("Can not open code output file: " + code_path);
55
56 ofstream decl_out(decl_path);
57
58 if (decl_out.fail())
59 throw AclCppException("Can not open declaration output file: " + decl_path);
60
61 ofstream par_out(par_path, ios_base::out | ios_base::binary);
62
63 if (par_out.fail())
64 throw AclCppException("Can not open parameter output file: " + par_path);
65
66 ArtifactGeneratorCppCode code_gen(code_out);
67 ArtifactGeneratorCppDecl decl_gen(decl_out);
68
69 // Generate the artifact.
70 AclCppOpGenerator op_generator(_artifact_name, par_out);
71 auto dom = op_generator.generate(g);
72 dom.accept(&code_gen);
73 dom.accept(&decl_gen);
74}
