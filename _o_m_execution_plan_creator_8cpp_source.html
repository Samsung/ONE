25OMStatus OMExecutionPlanCreator::createExecutionPlan(core::OMRuntimeStorage &runtime_storage,
26 core::OMRuntimeContext &runtime_context,
27 core::memory::OMRuntimeAllocator &allocator,
28 const OMConfig &configs)
29{
30 bool keep_input = configs.keep_input;
31
32 std::vector<std::vector<uint16_t>> &alloc_plan = allocator.getAllocPlan();
33 std::vector<std::vector<uint16_t>> &dealloc_plan = allocator.getDeallocPlan();
34
35 using Lifetime = std::pair<int32_t, int32_t>;
36
37 std::map<uint16_t, Lifetime> lifetimes;
38
39 const reader::CircleOperators *operators = runtime_context.getCircleOperators();
40
41 const size_t num_kernels = operators->size();
42
43 if (not keep_input)
44 {
45 auto graph_inputs = runtime_context.getCircleInputs();
46 for (const auto input_ind : *graph_inputs)
47 {
48 assert(lifetimes.count(input_ind) == 0);
49 lifetimes[input_ind] = Lifetime(-1, 0);
50 }
51 }
52
53 for (int32_t index = 0; index < num_kernels; ++index)
54 {
55 auto *cur_op = operators->operator[](index);
56
57 const auto *op_inputs = cur_op->inputs();
58 const auto *op_outputs = cur_op->outputs();
59 auto kernel_type = runtime_storage.getKernelType(index);
60 for (int32_t j = 0; j < op_inputs->size(); ++j)
61 {
62 const auto input_index = op_inputs->operator[](j);
63
64 if (input_index == -1)
65 continue;
66
67 // Pass constant tensors
68 if (runtime_context.isConstTensor(input_index))
69 continue;
70
71 if (lifetimes.count(input_index) > 0)
72 {
73 if (kernel_type == Inplace)
74 lifetimes.at(input_index).second = -1;
75 else
76 lifetimes.at(input_index).second = index;
77 }
78 }
79
80 for (int32_t j = 0; j < op_outputs->size(); ++j)
81 {
82 const auto output_index = op_outputs->operator[](j);
83
84 if (kernel_type == Inplace)
85 lifetimes[output_index] = Lifetime(-1, index);
86 else
87 lifetimes[output_index] = Lifetime(index, index);
88 }
89 }
90 auto graph_outputs = runtime_context.getCircleOutputs();
91 for (const auto output_ind : *graph_outputs)
92 {
93 if (lifetimes.count(output_ind) > 0)
94 lifetimes.at(output_ind).second = static_cast<int32_t>(num_kernels);
95 }
96
97 alloc_plan.assign(num_kernels, std::vector<uint16_t>());
98 dealloc_plan.assign(num_kernels + 1, std::vector<uint16_t>());
99
100 for (const auto &item : lifetimes)
101 {
102 if (item.second.first != -1)
103 alloc_plan[item.second.first].push_back(item.first);
104 if (item.second.second != -1)
105 dealloc_plan[item.second.second].push_back(item.first);
106 }
107
108 return Ok;
109}
