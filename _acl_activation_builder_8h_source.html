47template <typename T_Tensor, typename T_ActivationLayer, typename T_ExecFunction>
48std::unique_ptr<exec::IFunction>
49AclActivationBuilder<T_Tensor, T_ActivationLayer, T_ExecFunction>::generateReLU(T_Tensor *ifm_alloc)
50{
51 const ::arm_compute::ActivationLayerInfo act_info{
52 ::arm_compute::ActivationLayerInfo::ActivationFunction::RELU};
53
54 auto fn = std::make_unique<T_ActivationLayer>();
55
56 fn->configure(ifm_alloc, nullptr, act_info);
57
58 return asFunction<T_ExecFunction>(std::move(fn));
59}

91template <typename T_Tensor, typename T_ActivationLayer, typename T_ExecFunction>
92std::unique_ptr<exec::IFunction>
93AclActivationBuilder<T_Tensor, T_ActivationLayer, T_ExecFunction>::generate(ir::Activation code,
94 T_Tensor *ifm_alloc)
95{
96 switch (code)
97 {
98 case ir::Activation::NONE:
99 {
100 return std::make_unique<exec::NopFunction>();
101 }
102 case ir::Activation::RELU:
103 {
104 return generateReLU(ifm_alloc);
105 }
106 case ir::Activation::RELU1:
107 {
108 return generateReLU1(ifm_alloc);
109 }
110 case ir::Activation::RELU6:
111 {
112 return generateReLU6(ifm_alloc);
113 }
114 default:
115 {
116 throw std::runtime_error("Not supported, yet");
117 }
118 }
119}
