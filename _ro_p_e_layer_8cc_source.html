29void KernelGenerator::visit(const ir::operation::RoPE &node)
30{
31 const auto input_index{node.getInputs().at(ir::operation::RoPE::Input::INPUT)};
32 const auto sin_table{node.getInputs().at(ir::operation::RoPE::Input::SIN_TABLE)};
33 const auto cos_table{node.getInputs().at(ir::operation::RoPE::Input::COS_TABLE)};
34 const auto output_index{node.getOutputs().at(ir::operation::RoPE::Output::OUTPUT)};
35
36 auto mode = ops::getRoPEMode(node.param().mode);
37
38 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
39 auto sin_tensor = _tensor_reg->getPortableTensor(sin_table);
40 auto cos_tensor = _tensor_reg->getPortableTensor(cos_table);
41 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
42
43 auto fn = std::make_unique<ops::RoPELayer>();
44
45 fn->configure(input_tensor, sin_tensor, cos_tensor, mode, output_tensor);
46 _return_fn = std::move(fn);
47}

89void RoPELayer::run()
90{
91 switch (_input->data_type())
92 {
93 case OperandType::FLOAT32:
94 rope<float>();
95 break;
96 case OperandType::INT32:
97 rope<int32_t>();
98 break;
99 case OperandType::INT64:
100 rope<int64_t>();
101 break;
102 case OperandType::QUANT_UINT8_ASYMM:
103 rope<uint8_t>();
104 break;
105 case OperandType::QUANT_INT8_ASYMM:
106 rope<int8_t>();
107 break;
108 default:
109 throw std::runtime_error("RoPE: unsupported data type");
110 }
111}
