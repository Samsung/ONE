32template <typename T>
33OMStatus calculateActivationRange(circle::ActivationFunctionType activation, T *activation_min,
34 T *activation_max)
35{
36 switch (activation)
37 {
38 case circle::ActivationFunctionType::ActivationFunctionType_NONE:
39 *activation_min = std::numeric_limits<T>::lowest();
40 *activation_max = std::numeric_limits<T>::max();
41 break;
42 case circle::ActivationFunctionType::ActivationFunctionType_RELU:
43 *activation_min = 0;
44 *activation_max = std::numeric_limits<T>::max();
45 break;
46 case circle::ActivationFunctionType::ActivationFunctionType_RELU_N1_TO_1:
47 *activation_min = -1;
48 *activation_max = 1;
49 break;
50 case circle::ActivationFunctionType::ActivationFunctionType_RELU6:
51 *activation_min = 0;
52 *activation_max = 6;
53 break;
54 default:
55 assert(false && "Unsupported activation.");
56 return UnsupportedActivation;
57 }
58
59 return Ok;
60}

62inline double getQuantizedConvolutionMultipler(float input_scale, float filter_scale,
63 float output_scale)
64{
65 const double input_product_scale = static_cast<double>(input_scale * filter_scale);
66
67 assert(input_product_scale >= 0);
68
69 assert(output_scale != 0.f);
70
71 return input_product_scale / static_cast<double>(output_scale);
72}

75// significand, and shift representation of its exponent.
76//
77// Handles an arbitrary positive multiplier. The 'shift' output-value is

91inline std::vector<double>
92getQuantizedConvolutionMultiplers(float input_scale, const flatbuffers::Vector<float> *filter_scale,
93 float output_scale)
94{
95 std::vector<double> effective_output_scales;
96 size_t n = filter_scale->size();
97 effective_output_scales.reserve(n);
98 for (size_t i = 0; i < n; ++i)
99 {
100 effective_output_scales.push_back(
101 getQuantizedConvolutionMultipler(input_scale, filter_scale->operator[](i), output_scale));
102 }
103 return effective_output_scales;
104}

111inline int computeOutSize(circle::Padding padding, int image_size, int filter_size, int stride,
112 int dilation_rate = 1)
113{
114 int effective_filter_size = (filter_size - 1) * dilation_rate + 1;
115
116 if (stride == 0)
117 return 0;
118
119 switch (padding)
120 {
121 case circle::Padding_SAME:
122 return (image_size + stride - 1) / stride;
123 case circle::Padding_VALID:
124 return (image_size + stride - effective_filter_size) / stride;
125 default:
126 return 0;
127 }
128}

130inline int computePadding(int32_t stride, int32_t dilation_rate, int32_t in_size,
131 int32_t filter_size, int32_t out_size)
132{
133 int32_t effective_filter_size = (filter_size - 1) * dilation_rate + 1;
134 int32_t padding = ((out_size - 1) * stride + effective_filter_size - in_size) / 2;
135 return padding > 0 ? padding : 0;
136}

138inline void computePaddingHeightWidth(int32_t stride_height, int32_t stride_width,
139 int32_t dilation_rate_height, int32_t dilation_rate_width,
140 int32_t in_height, int32_t in_width, int32_t filter_height,
141 int32_t filter_width, circle::Padding padding,
142 int32_t *padding_h, int32_t *padding_w)
143{
144
145 int out_width =
146 computeOutSize(padding, in_width, filter_width, stride_width, dilation_rate_width);
147 int out_height =
148 computeOutSize(padding, in_height, filter_height, stride_height, dilation_rate_height);
149
150 *padding_h =
151 computePadding(stride_height, dilation_rate_height, in_height, filter_height, out_height);
152
153 *padding_w = computePadding(stride_width, dilation_rate_width, in_width, filter_width, out_width);
154}
