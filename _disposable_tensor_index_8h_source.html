43 DisposableTensorIndex(const ir::OperationIndex &op_index, const ir::OperandIndex &operand_index)
44 : _op_index{op_index}, _operand_index{operand_index}
45 {
46 assert(op_index.valid());
47 assert(operand_index.valid());
48 // NOTE This constraint is necessary to check overflow of hash value
49 assert((op_index.value() < (1 << 16) && operand_index.value() < (1 << 16)) &&
50 "DisposableTensorIndex does not support index value of operation or "
51 "operand greater than 65535");
52 }

107template <> struct hash<onert::backend::train::DisposableTensorIndex>
108{
109 size_t operator()(const onert::backend::train::DisposableTensorIndex &index) const noexcept
110 {
111 const auto op_index = index.op_index();
112 const auto operand_index = index.operand_index();
113
114 assert(sizeof(op_index) <= 4);
115 assert(sizeof(operand_index) <= 4);
116 static_assert(sizeof(size_t) >= sizeof(uint32_t),
117 "DisposableTensorIndex's hash creation error, size_t size is less than uint32_t");
118
119 return (static_cast<size_t>(op_index.value())) << 16 |
120 static_cast<size_t>(operand_index.value());
121 }
