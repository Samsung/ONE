31template <typename T>
32void evalLogic(const IPortableTensor *input, IPortableTensor *output, const std::vector<int> &axes,
33 bool keep_dims, T init_value, nnfw::cker::Reduce &reduce_kernel,
34 T reducer(const T current, const T in))
35{
36 reduce_kernel.prepare(input->getShape().rank(), axes.size());
37 bool result =
38 reduce_kernel.ReduceGeneric<T>(getShape(input), getBuffer<T>(input), getShape(output),
39 getBuffer<T>(output), axes, keep_dims, init_value, reducer);
40
41 if (!result)
42 {
43 throw std::runtime_error{"Reduce: Fail to run"};
44 }
45}

47template <typename T>
48std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
49evalType(bool keep_dims, nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
50{
51 switch (reduce_type)
52 {
53 case ReduceType::kSum:
54 return std::bind(&evalLogic<T>, std::placeholders::_1, std::placeholders::_2,
55 std::placeholders::_3, keep_dims, static_cast<T>(0), reduce_kernel,
56 [](const T current, const T in) -> T { return in + current; });
57 break;
58 case ReduceType::kProd:
59 return std::bind(&evalLogic<T>, std::placeholders::_1, std::placeholders::_2,
60 std::placeholders::_3, keep_dims, static_cast<T>(1), reduce_kernel,
61 [](const T current, const T in) -> T { return in * current; });
62 break;
63 case ReduceType::kMax:
64 return std::bind(
65 &evalLogic<T>, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3,
66 keep_dims, std::numeric_limits<T>::lowest(), reduce_kernel,
67 [](const T current, const T in) -> T { return (in > current) ? in : current; });
68 break;
69 case ReduceType::kMin:
70 return std::bind(
71 &evalLogic<T>, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3,
72 keep_dims, std::numeric_limits<T>::max(), reduce_kernel,
73 [](const T current, const T in) -> T { return (in < current) ? in : current; });
74 break;
75 default:
76 throw std::runtime_error{"Reduce: Unsupported reduce type"};
77 }
78}

80// Template specialization for bool type
81template <>
82std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
83evalType<bool>(bool keep_dims, nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
84{
85 static_assert(sizeof(bool) == 1, "cpu backend supports bool type which is 1 byte");
86 switch (reduce_type)
87 {
88 case ReduceType::kAny:
89 return std::bind(&evalLogic<bool>, std::placeholders::_1, std::placeholders::_2,
90 std::placeholders::_3, keep_dims, false, reduce_kernel,
91 [](const bool current, const bool in) -> bool { return in || current; });
92 break;
93 case ReduceType::kAll:
94 return std::bind(&evalLogic<bool>, std::placeholders::_1, std::placeholders::_2,
95 std::placeholders::_3, keep_dims, true, reduce_kernel,
96 [](const bool current, const bool in) -> bool { return in && current; });
97 break;
98 default:
99 throw std::runtime_error{"Reduce: Unsupported reduce type"};
100 }
101}

103std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
104generateKernelGeneric(const IPortableTensor *input, bool keep_dims,
105 nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
106{
107 switch (input->data_type())
108 {
109 case OperandType::FLOAT32:
110 return evalType<float>(keep_dims, reduce_kernel, reduce_type);
111 case OperandType::INT32:
112 return evalType<int32_t>(keep_dims, reduce_kernel, reduce_type);
113 case OperandType::BOOL8:
114 return evalType<bool>(keep_dims, reduce_kernel, reduce_type);
115 default:
116 throw std::runtime_error{"Reduce(generic): unsupported data type"};
117 }
118}

120// TODO Refine this function
121void evalSumQuantized(const IPortableTensor *input, IPortableTensor *output,
122 const std::vector<int> &axes, bool keep_dims,
123 nnfw::cker::Reduce &reduce_kernel)
124{
125 const bool same_scale = (input->data_scale() == output->data_scale() &&
126 input->data_zero_point() == output->data_zero_point());
127
128 reduce_kernel.prepare(input->getShape().rank(), axes.size());
129
130 if (!same_scale)
131 {
132 std::vector<int32_t> temp_sum(output->getShape().num_elements());
133 bool result = reduce_kernel.QuantizedMeanOrSum<uint8_t, int32_t>(
134 getBuffer<uint8_t>(input), input->data_zero_point(), input->data_scale(), getShape(input),
135 getBuffer<uint8_t>(output), output->data_zero_point(), output->data_scale(), getShape(output),
136 axes, keep_dims, temp_sum.data(), true,
137 [](const int32_t current, const uint8_t in) -> int32_t {
138 const int32_t actual_in = static_cast<int32_t>(in);
139 return current + actual_in;
140 });
141
142 if (!result)
143 {
144 throw std::runtime_error{"Reduce: Fail to run"};
145 }
146
147 return;
148 }
149
150 const auto kernel = generateKernelGeneric(input, keep_dims, reduce_kernel, ReduceType::kSum);
151 kernel(input, output, axes);
152}

165void ReduceLayer::configure(const IPortableTensor *input, const IPortableTensor *axes,
166 IPortableTensor *output, ReduceType reduceType, bool keep_dims)
167{
168 _input = input;
169 _axes = axes;
170 _output = output;
171 _reduceType = reduceType;
172
173 switch (_reduceType)
174 {
175 case ReduceType::kSum:
176 if (_input->data_type() == OperandType::QUANT_UINT8_ASYMM)
177 {
178 _kernel = std::bind(&evalSumQuantized, std::placeholders::_1, std::placeholders::_2,
179 std::placeholders::_3, keep_dims, *_reduce_kernel);
180 return;
181 }
182 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kSum);
183 break;
184 case ReduceType::kProd:
185 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kProd);
186 break;
187 case ReduceType::kMax:
188 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kMax);
189 break;
190 case ReduceType::kMin:
191 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kMin);
192 break;
193 case ReduceType::kAny:
194 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kAny);
195 break;
196 case ReduceType::kAll:
197 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kAll);
198 break;
199 default:
200 throw std::runtime_error{"Reduce: Unsupported reduce type"};
201 }
202}

204void ReduceLayer::run()
205{
206 const auto axes = getReducerAxes(_axes);
207#ifdef USE_NEON
208 int32_t rank = _input->getShape().rank();
209 if (_input->data_type() == ir::DataType::FLOAT32 && _reduceType == ReduceType::kSum &&
210 axes.size() == 1 && (axes[0] == -1 || axes[0] == rank - 1))
211 {
212 OptimizedReduceSum(getBuffer<float>(_input), getShape(_input), getBuffer<float>(_output));
213 return;
214 }
215#endif // NEON
216 _kernel(_input, _output, axes);
217}

224void MeanLayer::MeanFloat32()
225{
226 const auto inputShape = getShape(_input);
227 const auto axisVec = getReducerAxes(_axes);
228 bool axis_is_1_and_2 =
229 _keep_dims && inputShape.DimensionsCount() == 4 && axisVec.size() == 2 &&
230 ((axisVec[0] == 1 && axisVec[1] == 2) || (axisVec[0] == 2 && axisVec[1] == 1));
231
232 if (axis_is_1_and_2)
233 {
234 nnfw::cker::MeanAxis1And2(inputShape, getBuffer<float>(_input), getShape(_output),
235 getBuffer<float>(_output));
236 }
237 else
238 {
239 nnfw::cker::Mean(inputShape, getBuffer<float>(_input), getShape(_output),
240 getBuffer<float>(_output), axisVec);
241 }
242}

264void MeanLayer::run()
265{
266 if (_input->data_type() == OperandType::FLOAT32)
267 {
268 MeanFloat32();
269 }
270 else if (_input->data_type() == OperandType::QUANT_UINT8_ASYMM)
271 {
272 MeanQuant8();
273 }
274 else
275 {
276 throw std::runtime_error{"Mean: unsupported data type"};
277 }
