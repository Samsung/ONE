53void KernelGenerator::visit(const ir::operation::Reduce &node)
54{
55 const auto output_index{node.getOutputs().at(0)};
56 const auto input_index{node.getInputs().at(ir::operation::Reduce::Input::INPUT)};
57 const auto axes_index{node.getInputs().at(ir::operation::Reduce::Input::AXES)};
58
59 const auto keep_dims = node.param().keep_dims;
60 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
61 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
62 auto axes_tensor = _tensor_reg->getPortableTensor(axes_index);
63
64 if (node.param().reduce_type == ir::operation::Reduce::ReduceType::MEAN)
65 {
66 auto fn = std::make_unique<ops::MeanLayer>();
67
68 fn->configure(input_tensor, axes_tensor, output_tensor, keep_dims);
69
70 _return_fn = std::move(fn);
71 }
72 else
73 {
74 auto fn = std::make_unique<ops::ReduceLayer>();
75
76 const auto reduce_type = convertReduceType(node.param().reduce_type);
77 fn->configure(input_tensor, axes_tensor, output_tensor, reduce_type, keep_dims);
78
79 _return_fn = std::move(fn);
80 }
81}

91template <typename T>
92void evalLogic(const IPortableTensor *input, IPortableTensor *output, const std::vector<int> &axes,
93 bool keep_dims, T init_value, nnfw::cker::Reduce &reduce_kernel,
94 T reducer(const T current, const T in))
95{
96 reduce_kernel.prepare(input->getShape().rank(), axes.size());
97 bool result =
98 reduce_kernel.ReduceGeneric<T>(getShape(input), getBuffer<T>(input), getShape(output),
99 getBuffer<T>(output), axes, keep_dims, init_value, reducer);
100
101 if (!result)
102 {
103 throw std::runtime_error{"Reduce: Fail to run"};
104 }
105}

107template <typename T>
108std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
109evalType(bool keep_dims, nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
110{
111 switch (reduce_type)
112 {
113 case ReduceType::kSum:
114 return std::bind(&evalLogic<T>, std::placeholders::_1, std::placeholders::_2,
115 std::placeholders::_3, keep_dims, static_cast<T>(0), reduce_kernel,
116 [](const T current, const T in) -> T { return in + current; });
117 break;
118 case ReduceType::kProd:
119 return std::bind(&evalLogic<T>, std::placeholders::_1, std::placeholders::_2,
120 std::placeholders::_3, keep_dims, static_cast<T>(1), reduce_kernel,
121 [](const T current, const T in) -> T { return in * current; });
122 break;
123 case ReduceType::kMax:
124 return std::bind(
125 &evalLogic<T>, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3,
126 keep_dims, std::numeric_limits<T>::lowest(), reduce_kernel,
127 [](const T current, const T in) -> T { return (in > current) ? in : current; });
128 break;
129 case ReduceType::kMin:
130 return std::bind(
131 &evalLogic<T>, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3,
132 keep_dims, std::numeric_limits<T>::max(), reduce_kernel,
133 [](const T current, const T in) -> T { return (in < current) ? in : current; });
134 break;
135 default:
136 throw std::runtime_error{"Reduce: Unsupported reduce type"};
137 }
138}

140// Template specialization for bool type
141template <>
142std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
143evalType<bool>(bool keep_dims, nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
144{
145 static_assert(sizeof(bool) == 1, "cpu backend supports bool type which is 1 byte");
146 switch (reduce_type)
147 {
148 case ReduceType::kAny:
149 return std::bind(&evalLogic<bool>, std::placeholders::_1, std::placeholders::_2,
150 std::placeholders::_3, keep_dims, false, reduce_kernel,
151 [](const bool current, const bool in) -> bool { return in || current; });
152 break;
153 case ReduceType::kAll:
154 return std::bind(&evalLogic<bool>, std::placeholders::_1, std::placeholders::_2,
155 std::placeholders::_3, keep_dims, true, reduce_kernel,
156 [](const bool current, const bool in) -> bool { return in && current; });
157 break;
158 default:
159 throw std::runtime_error{"Reduce: Unsupported reduce type"};
160 }
161}

163std::function<void(const IPortableTensor *, IPortableTensor *, const std::vector<int> &)>
164generateKernelGeneric(const IPortableTensor *input, bool keep_dims,
165 nnfw::cker::Reduce &reduce_kernel, ReduceType reduce_type)
166{
167 switch (input->data_type())
168 {
169 case OperandType::FLOAT32:
170 return evalType<float>(keep_dims, reduce_kernel, reduce_type);
171 case OperandType::INT32:
172 return evalType<int32_t>(keep_dims, reduce_kernel, reduce_type);
173 case OperandType::BOOL8:
174 return evalType<bool>(keep_dims, reduce_kernel, reduce_type);
175 default:
176 throw std::runtime_error{"Reduce(generic): unsupported data type"};
177 }
178}

181void evalSumQuantized(const IPortableTensor *input, IPortableTensor *output,
182 const std::vector<int> &axes, bool keep_dims,
183 nnfw::cker::Reduce &reduce_kernel)
184{
185 const bool same_scale = (input->data_scale() == output->data_scale() &&
186 input->data_zero_point() == output->data_zero_point());
187
188 reduce_kernel.prepare(input->getShape().rank(), axes.size());
189
190 if (!same_scale)
191 {
192 std::vector<int32_t> temp_sum(output->getShape().num_elements());
193 bool result = reduce_kernel.QuantizedMeanOrSum<uint8_t, int32_t>(
194 getBuffer<uint8_t>(input), input->data_zero_point(), input->data_scale(), getShape(input),
195 getBuffer<uint8_t>(output), output->data_zero_point(), output->data_scale(), getShape(output),
196 axes, keep_dims, temp_sum.data(), true,
197 [](const int32_t current, const uint8_t in) -> int32_t {
198 const int32_t actual_in = static_cast<int32_t>(in);
199 return current + actual_in;
200 });
201
202 if (!result)
203 {
204 throw std::runtime_error{"Reduce: Fail to run"};
205 }
206
207 return;
208 }
209
210 const auto kernel = generateKernelGeneric(input, keep_dims, reduce_kernel, ReduceType::kSum);
211 kernel(input, output, axes);
212}

225void ReduceLayer::configure(const IPortableTensor *input, const IPortableTensor *axes,
226 IPortableTensor *output, ReduceType reduceType, bool keep_dims)
227{
228 _input = input;
229 _axes = axes;
230 _output = output;
231 _reduceType = reduceType;
232
233 switch (_reduceType)
234 {
235 case ReduceType::kSum:
236 if (_input->data_type() == OperandType::QUANT_UINT8_ASYMM)
237 {
238 _kernel = std::bind(&evalSumQuantized, std::placeholders::_1, std::placeholders::_2,
239 std::placeholders::_3, keep_dims, *_reduce_kernel);
240 return;
241 }
242 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kSum);
243 break;
244 case ReduceType::kProd:
245 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kProd);
246 break;
247 case ReduceType::kMax:
248 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kMax);
249 break;
250 case ReduceType::kMin:
251 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kMin);
252 break;
253 case ReduceType::kAny:
254 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kAny);
255 break;
256 case ReduceType::kAll:
257 _kernel = generateKernelGeneric(_input, keep_dims, *_reduce_kernel, ReduceType::kAll);
258 break;
259 default:
260 throw std::runtime_error{"Reduce: Unsupported reduce type"};
261 }
262}

264void ReduceLayer::run()
265{
266 const auto axes = getReducerAxes(_axes);
267#ifdef USE_NEON
268 int32_t rank = _input->getShape().rank();
269 if (_input->data_type() == ir::DataType::FLOAT32 && _reduceType == ReduceType::kSum &&
270 axes.size() == 1 && (axes[0] == -1 || axes[0] == rank - 1))
271 {
272 OptimizedReduceSum(getBuffer<float>(_input), getShape(_input), getBuffer<float>(_output));
273 return;
274 }
275#endif // NEON
276 _kernel(_input, _output, axes);
277}

284void MeanLayer::MeanFloat32()
285{
286 const auto inputShape = getShape(_input);
287 const auto axisVec = getReducerAxes(_axes);
288 bool axis_is_1_and_2 =
289 _keep_dims && inputShape.DimensionsCount() == 4 && axisVec.size() == 2 &&
290 ((axisVec[0] == 1 && axisVec[1] == 2) || (axisVec[0] == 2 && axisVec[1] == 1));
291
292 if (axis_is_1_and_2)
293 {
294 nnfw::cker::MeanAxis1And2(inputShape, getBuffer<float>(_input), getShape(_output),
295 getBuffer<float>(_output));
296 }
297 else
298 {
299 nnfw::cker::Mean(inputShape, getBuffer<float>(_input), getShape(_output),
300 getBuffer<float>(_output), axisVec);
301 }
302}
