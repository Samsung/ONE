62bool decompose_softmax(luci::CircleSoftmax *softmax)
63{
64 if (!softmax)
65 return false;
66
67 if (softmax->dtype() != loco::DataType::FLOAT32)
68 return false;
69
70 auto const input = loco::must_cast<luci::CircleNode *>(softmax->logits());
71 auto g = softmax->graph();
72
73 auto const beta = softmax->beta();
74 auto const name = softmax->name();
75 assert(name.length() > 0);
76
77 // fill reduction index (-1) for CircleReduceMax and CircleSum
78 auto index_const = g->nodes()->create<luci::CircleConst>();
79 index_const->shape({}); // scalar
80 index_const->dtype(loco::DataType::S32);
81 index_const->rank(0);
82 index_const->size<loco::DataType::S32>(1);
83 index_const->at<loco::DataType::S32>(0) = -1;
84 index_const->name(name + "/Softmax/reduction_index");
85 luci::add_origin(index_const, luci::get_origin(softmax));
86
87 // Create CircleReduceMax operation
88 auto max = g->nodes()->create<luci::CircleReduceMax>();
89 max->input(input);
90 max->reduction_indices(index_const);
91 max->keep_dims(true);
92 max->name(name + "/Softmax/max");
93 luci::add_origin(max, luci::get_origin(softmax));
94
95 // Create CircleSub operation
96 auto sub = g->nodes()->create<luci::CircleSub>();
97 sub->x(input);
98 sub->y(max);
99 sub->fusedActivationFunction(luci::FusedActFunc::NONE);
100 sub->name(name + "/Softmax/sub");
101 luci::add_origin(sub, luci::get_origin(softmax));
102
103 // input for exp can be either sub or mul (in case beta != 1)
104 loco::Node *exp_input = sub;
105
106 // multiply sub by beta in case it is nonunit
107 if (std::abs(beta - 1.f) > 1.e-05f)
108 {
109 // Create constant for beta
110 auto beta_const = g->nodes()->create<luci::CircleConst>();
111 beta_const->shape({}); // scalar
112 beta_const->dtype(loco::DataType::FLOAT32);
113 beta_const->rank(0);
114 beta_const->size<loco::DataType::FLOAT32>(1);
115 beta_const->at<loco::DataType::FLOAT32>(0) = beta;
116 beta_const->name(name + "/Softmax/beta_const");
117 luci::add_origin(beta_const, luci::get_origin(softmax));
118
119 // Create CircleMul
120 auto mul = g->nodes()->create<luci::CircleMul>();
121 mul->x(sub);
122 mul->y(beta_const);
123 mul->fusedActivationFunction(luci::FusedActFunc::NONE);
124 mul->name(name + "/Softmax/beta_mul");
125 luci::add_origin(mul, luci::get_origin(softmax));
126
127 exp_input = mul;
128 }
129
130 // Create CircleExp operation
131 auto exp = g->nodes()->create<luci::CircleExp>();
132 exp->x(exp_input);
133 exp->name(name + "/Softmax/exp");
134 luci::add_origin(exp, luci::get_origin(softmax));
135
136 // Create CircleSum operation
137 auto sum = g->nodes()->create<luci::CircleSum>();
138 sum->input(exp);
139 sum->reduction_indices(index_const);
140 sum->keep_dims(true);
141 sum->name(name + "/Softmax/sum");
142 luci::add_origin(sum, luci::get_origin(softmax));
143
144 // Create CircleDiv operation
145 auto div = g->nodes()->create<luci::CircleDiv>();
146 div->x(exp);
147 div->y(sum);
148 div->fusedActivationFunction(luci::FusedActFunc::NONE);
149 div->name(name + "/Softmax/div");
150 luci::add_origin(div, luci::get_origin(softmax));
151
152 replace(softmax).with(div);
153
154 return true;
155}

162bool DecomposeSoftmaxPass::run(loco::Graph *g)
163{
164 bool changed = false;
165
166 for (auto node : loco::active_nodes(loco::output_nodes(g)))
167 {
168 if (auto softmax = dynamic_cast<luci::CircleSoftmax *>(node))
169 {
170 if (decompose_softmax(softmax))
171 changed = true;
172 }
173 }
174
175 return changed;
176}
