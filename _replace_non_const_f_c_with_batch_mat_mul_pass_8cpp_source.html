28luci::CircleNode *fromActivation(luci::CircleNode *inp, luci::FusedActFunc act)
29{
30 switch (act)
31 {
32 case luci::FusedActFunc::NONE:
33 return inp;
34 case luci::FusedActFunc::RELU:
35 {
36 auto n = inp->graph()->nodes()->create<luci::CircleRelu>();
37 n->features(inp);
38 return n;
39 }
40 case luci::FusedActFunc::RELU6:
41 {
42 auto n = inp->graph()->nodes()->create<luci::CircleRelu6>();
43 n->features(inp);
44 return n;
45 }
46 case luci::FusedActFunc::RELU_N1_TO_1:
47 {
48 auto n = inp->graph()->nodes()->create<luci::CircleReluN1To1>();
49 n->features(inp);
50 return n;
51 }
52 case luci::FusedActFunc::TANH:
53 {
54 auto n = inp->graph()->nodes()->create<luci::CircleTanh>();
55 n->x(inp);
56 return n;
57 }
58 case luci::FusedActFunc::SIGN_BIT:
59 {
60 throw std::invalid_argument("no matching node to create from fused activation");
61 }
62 default:
63 throw std::invalid_argument("invalid fused activation");
64 }
65}

67// Create CircleReshape where
68// - dtype is same with node
69// - shape is same with node
70// NOTE: User should set input(tensor) of the returned Op.
71luci::CircleReshape *create_reshape(luci::CircleFullyConnected *node)
72{
73 assert(node); // FIX_CALLER_UNLESS
74
75 auto g = node->graph();
76
77 auto reshape = g->nodes()->create<luci::CircleReshape>();
78 reshape->name(node->name() + "/reshape");
79 reshape->dtype(node->dtype());
80 luci::add_origin(reshape, luci::get_origin(node));
81
82 auto shape_const = g->nodes()->create<luci::CircleConst>();
83 shape_const->dtype(loco::DataType::S32);
84 shape_const->rank(1);
85 shape_const->dim(0).set(node->rank());
86 shape_const->size<loco::DataType::S32>(node->rank());
87 for (uint32_t i = 0; i < node->rank(); i++)
88 {
89 assert(node->dim(i).known()); // FIX_CALLER_UNLESS
90 shape_const->at<loco::DataType::S32>(i) = node->dim(i).value();
91 }
92 shape_const->shape_status(luci::ShapeStatus::VALID);
93 shape_const->name(node->name() + "/shape");
94 luci::add_origin(shape_const, luci::get_origin(node));
95
96 reshape->shape(shape_const);
97
98 return reshape;
99}

129bool replace_fc_with_matmul(luci::CircleFullyConnected *fc)
130{
131 luci::CircleNode *x = nullptr;
132 luci::CircleNode *y = nullptr;
133 luci::CircleTranspose *ty = nullptr;
134 luci::CircleTranspose *tx = nullptr;
135 bool adj_x = false;
136 bool adj_y = true;
137
138 if (dynamic_cast<luci::CircleConst *>(fc->weights()))
139 return false; // NonConst
140
141 if ((ty = dynamic_cast<luci::CircleTranspose *>(fc->weights()))) // is y a transpose?
142 {
143 adj_y = false;
144 if (dynamic_cast<luci::CircleConst *>(ty->a()))
145 return false;
146 else
147 y = loco::must_cast<luci::CircleNode *>(ty->a());
148 }
149 else
150 { // y is not transpose and not const
151 y = loco::must_cast<luci::CircleNode *>(fc->weights());
152 }
153 if ((tx = dynamic_cast<luci::CircleTranspose *>(fc->input())))
154 {
155 adj_x = true;
156 x = loco::must_cast<luci::CircleNode *>(tx->a());
157 }
158 else
159 {
160 x = loco::must_cast<luci::CircleNode *>(fc->input());
161 }
162
163 if (x->dtype() != loco::DataType::FLOAT32 || y->dtype() != loco::DataType::FLOAT32)
164 return false;
165
166 auto bc = dynamic_cast<luci::CircleConst *>(fc->bias());
167 // NOTE bias can be empty as CircleOutputExclude type
168 // NOTE we can only handle bias as FLOAT32 type as of now
169 if (nullptr != bc && bc->dtype() != loco::DataType::FLOAT32)
170 return false;
171
172 auto name = fc->name();
173 assert(name.length() > 0);
174
175 auto matmul = fc->graph()->nodes()->create<luci::CircleBatchMatMul>();
176 matmul->x(x);
177 matmul->y(y);
178 matmul->adj_x(adj_x);
179 matmul->adj_y(adj_y);
180 matmul->name(name);
181 matmul->dtype(fc->dtype());
182
183 luci::add_origin(matmul, luci::get_origin(fc));
184
185 auto reshape = create_reshape(fc);
186 reshape->tensor(matmul);
187
188 auto all_zero = [](const luci::CircleConst *c) {
189 bool ac = true;
190 for (uint32_t i = 0; i < c->size<loco::DataType::FLOAT32>() && ac; i++)
191 {
192 ac &= c->at<loco::DataType::FLOAT32>(i) == 0.0f;
193 }
194 return ac;
195 };
196
197 if (nullptr != bc && !all_zero(bc))
198 {
199 auto bias_add = fc->graph()->nodes()->create<luci::CircleAdd>();
200 bias_add->x(reshape);
201 bias_add->y(bc);
202 bias_add->name(fc->name() + "/bias_add");
203 bias_add->dtype(fc->dtype());
204 add_origin(bias_add, get_origin(fc));
205 bias_add->fusedActivationFunction(fc->fusedActivationFunction());
206 loco::replace(fc).with(bias_add);
207 }
208 else
209 {
210 // NOTE bias doesn't exist or bias is all zero
211 auto n = fromActivation(reshape, fc->fusedActivationFunction());
212 add_origin(n, luci::get_origin(fc));
213 n->name(fc->name() + "fusedActivation");
214 n->dtype(fc->dtype());
215 loco::replace(fc).with(n);
216 }
217
218 return true;
219}

225bool ReplaceNonConstFCWithBatchMatMulPass::run(loco::Graph *g)
226{
227 bool changed = false;
228 for (auto node : loco::active_nodes(loco::output_nodes(g)))
229 {
230 if (auto fc = dynamic_cast<luci::CircleFullyConnected *>(node))
231 {
232 if (replace_fc_with_matmul(fc))
233 changed = true;
234 }
235 }
236
237 return changed;
238}
