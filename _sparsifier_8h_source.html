28template <typename T> class Sparsifier
29{
30public:
31 /*
32 * Creates a dense to sparse converter.
33 * @param shape Shape of the dense tensor.
34 * @param traversal_order In what order to traverse all dimensions,
35 * including block dimensions.
36 * @param format Whether each dimension in converted tensor is
37 * dense or sparse (not in the traversal order).
38 * @param block_size Size of each block dimension.
39 * @param block_map Map from block dimension to original tensor
40 * dimension.
41 */
42 Sparsifier(const std::vector<int> &shape, const std::vector<int> &traversal_order,
43 const std::vector<DimensionType> &format, const std::vector<int> &block_size = {},
44 const std::vector<int> &block_map = {});
45
46 std::vector<T> GetData() { return _data; }
47 std::vector<std::vector<int>> GetDimMetadata() { return _dim_metadata; }
48
49 void DenseToSparse(const T *src_data);
50
51private:
52 // Check if val is equal to zero.
53 bool IsZero(const T val);
54
55 // Shape of the conceptual dense tensor.
56 std::vector<int> _dense_shape;
57 // Shape of the dense tensor with inner blocks reduced. For example, a (4, 4)
58 // tensor with (2, 2) block has blocked_shape (2, 2).
59 std::vector<int> _blocked_shape;
60 // Total number of elements in the dense tensor.
61 uint64_t _dense_size;
62 // Has n(original dimension)+k(block_dimension) elements.
63 std::vector<int> _traversal_order;
64 // Format of each dimension in the traversal order.
65 std::vector<DimensionType> _format;
66 // Size of each block dimension, in the same order as block map.
67 std::vector<int> _block_size;
68 // Map from block dimension to the original tensor dimension.
69 std::vector<int> _block_map;
70 // Metadata of each dimension in the traversal order.
71 // Each dimension needs two vectors. For dense dimensions, the first vector
72 // stores the size of that dimension, and the second vector is empty. For
73 // sparse dimensions, the first vector stores the segments and the second one
74 // stores the indices.
75 std::vector<std::vector<int>> _dim_metadata;
76 // Actual buffer holding data after conversion. Could be sparse buffer or
77 // dense buffer.
78 std::vector<T> _data;
79};
