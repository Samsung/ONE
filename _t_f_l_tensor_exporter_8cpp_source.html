73struct NoOpDetector final : public loco::CanonicalNodeMutableVisitor<bool>
74{
75 bool visit(loco::BiasEncode *) final
76 {
77 // BiasEncode is always noop
78 return true;
79 }
80
81 bool visit(loco::FilterEncode *node) final
82 {
83 auto encoder = loco::must_cast<loco::PermutingEncoder<loco::Domain::Filter> *>(node->encoder());
84 auto perm = encoder->perm();
85
86 return isNHWC(perm);
87 }
88
89 bool visit(loco::FeatureEncode *node) final
90 {
91 auto encoder =
92 loco::must_cast<loco::PermutingEncoder<loco::Domain::Feature> *>(node->encoder());
93 auto perm = encoder->perm();
94 return isNHWC(perm);
95 }
96
97 bool visit(loco::FeatureDecode *node) final
98 {
99 auto decoder =
100 loco::must_cast<loco::PermutingDecoder<loco::Domain::Feature> *>(node->decoder());
101 auto perm = decoder->perm();
102 return isNHWC(perm);
103 }
104
105 // Return false by default
106 bool visit(loco::Node *) final { return false; }
107};

109bool isNoOp(loco::Node *node)
110{
111 if (auto canonical_node = dynamic_cast<loco::CanonicalNode *>(node))
112 {
113 NoOpDetector d;
114 return canonical_node->accept(&d);
115 }
116 return false;
117}

119void allocateTFLiteTensor(loco::Node *node, TFLTensorContext &ctx)
120{
121 if (isNoOp(node))
122 {
123 assert(node->arity() == 1 && node->arg(0) != nullptr);
124 set_tensor_index(node, get_tensor_index(node->arg(0)));
125 return;
126 }
127
128 auto tensor_index = static_cast<TFLTensorIndex>(ctx.size());
129 // TODO Use Graph-level metadata for Input & Output
130 auto tensor_name = "t_" + std::to_string(tensor_index);
131
132 TFLTensorInfo tensor_info;
133
134 tensor_info.name(tensor_name);
135 tensor_info.dtype(TypeInference::get(node));
136 tensor_info.shape(ShapeInference::get(node));
137
138 if (auto const_node = dynamic_cast<locoex::TFLConst *>(node))
139 tensor_info.tfl_content(const_node);
140
141 set_tensor_index(node, tensor_index);
142
143 ctx.emplace_back(tensor_info);
144}

169template <loco::DataType DT>
170flatbuffers::Offset<tflite::Buffer> encodeOpBufferByDType(FlatBufferBuilder &builder,
171 locoex::TFLConst *c)
172{
173 using NativeType = typename loco::DataTypeImpl<DT>::Type;
174
175 std::vector<NativeType> raw_data;
176 const uint32_t size = c->size<DT>();
177 raw_data.reserve(size);
178 for (uint32_t i = 0; i < size; ++i)
179 {
180 raw_data.push_back(c->at<DT>(i));
181 }
182 const size_t raw_size = size * sizeof(NativeType);
183 auto array_offset = builder.CreateVector(reinterpret_cast<uint8_t *>(raw_data.data()), raw_size);
184 return CreateBuffer(builder, array_offset);
185}
186
187template <>
188flatbuffers::Offset<tflite::Buffer> encodeOpBuffer(FlatBufferBuilder &builder, locoex::TFLConst *c)
189{
190 if (c->dtype() == loco::DataType::FLOAT32)
191 {
192 return encodeOpBufferByDType<loco::DataType::FLOAT32>(builder, c);
193 }
194 else if (c->dtype() == loco::DataType::S32)
195 {
196 return encodeOpBufferByDType<loco::DataType::S32>(builder, c);
197 }
198
199 INTERNAL_EXN_V("Unsupported datatype", oops::to_uint32(c->dtype()));
200}

209void exportOpDefinedTensor(const TFLTensorInfo &info, FlatBufferBuilder &builder,
210 SerializedModelData &gd)
211{
212 // Create and register output tensor shape
213 auto shape_offset = encodeShape(builder, info.shape());
214
215 // encode and register output tensor buffer
216 auto buffer = info.tfl_content() == nullptr ? encodeOpBuffer(builder)
217 : encodeOpBuffer(builder, info.tfl_content());
218
219 auto buffer_id = static_cast<uint32_t>(gd._buffers.size());
220 gd._buffers.push_back(buffer);
221
222 auto name_offset = builder.CreateString(info.name());
223 auto tensor_offset = CreateTensor(builder, shape_offset, info.dtype(), buffer_id, name_offset,
224 /*quantization*/ 0, /*is_variable*/ false);
225 gd._tensors.push_back(tensor_offset);
226}
227
228void exportOpDefinedTensors(loco::Graph *g, FlatBufferBuilder &builder, SerializedModelData &gd)
229{
230 TFLTensorContext tensor_ctx;
231
232 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
233 {
234 allocateTFLiteTensor(node, tensor_ctx);
235 }
236
237 // add one empty buffer
238 // note: there's a comment in tflite fbs file
239 // - Note the 0th entry of this array must be an empty buffer (sentinel).
240 // - This is a convention so that tensors without a buffer can provide 0 as
241 // - their buffer.
242 auto buffer = encodeOpBuffer(builder);
243 gd._buffers.push_back(buffer);
244
245 for (const auto &tensor_info : tensor_ctx)
246 {
247 exportOpDefinedTensor(tensor_info, builder, gd);
248 }
249}
