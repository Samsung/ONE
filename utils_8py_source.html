100def parse_cfg(config_path: Union[str, None],
101 section_to_parse: str,
102 args,
103 quiet: bool = False):
104 """
105 parse configuration file and store the information to args
106
107 :param config_path: path to configuration file
108 :param section_to_parse: section name to parse
109 :param args: object to store the parsed information
110 :param quiet: raise no error when given section doesn't exist
111 """
112 if config_path is None:
113 return
114
115 parser = get_config_parser()
116 parser.read(config_path)
117
118 if not parser.has_section(section_to_parse) and quiet:
119 return
120
121 if not parser.has_section(section_to_parse):
122 raise AssertionError('configuration file must have \'' + section_to_parse +
123 '\' section')
124
125 for key in parser[section_to_parse]:
126 if is_accumulated_arg(key, section_to_parse):
127 if not is_valid_attr(args, key):
128 setattr(args, key, [parser[section_to_parse][key]])
129 else:
130 getattr(args, key).append(parser[section_to_parse][key])
131 continue
132 if hasattr(args, key) and getattr(args, key):
133 continue
134 setattr(args, key, parser[section_to_parse][key])

157def run(cmd, err_prefix=None, logfile=None):
158 """Execute command in subprocess
159
160 Args:
161 cmd: command to be executed in subprocess
162 err_prefix: prefix to be put before every stderr lines
163 logfile: file stream to which both of stdout and stderr lines will be written
164 """
165 with subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE) as p:
166 import select
167 inputs = set([p.stdout, p.stderr])
168 while inputs:
169 readable, _, _ = select.select(inputs, [], [])
170 for x in readable:
171 line = x.readline()
172 if len(line) == 0:
173 inputs.discard(x)
174 continue
175 if x == p.stdout:
176 out = sys.stdout
177 if x == p.stderr:
178 out = sys.stderr
179 if err_prefix:
180 line = f"{err_prefix}: ".encode() + line
181 out.buffer.write(line)
182 out.buffer.flush()
183 if logfile != None:
184 logfile.write(line)
185 if p.returncode != 0:
186 sys.exit(p.returncode)

201def get_optimization_list(get_name=False):
202 """
203 returns a list of optimization. If `get_name` is True,
204 only basename without extension is returned rather than full file path.
205
206 [one hierarchy]
207 one
208 ├── backends
209 ├── bin
210 ├── doc
211 ├── include
212 ├── lib
213 ├── optimization
214 └── test
215
216 Optimization options must be placed in `optimization` folder
217 """
218 dir_path = os.path.dirname(os.path.realpath(__file__))
219
220 # optimization folder
221 files = [
222 f for f in glob.glob(dir_path + '/../../optimization/O*.cfg', recursive=True)
223 ]
224 # exclude if the name has space
225 files = [s for s in files if not ' ' in s]
226
227 opt_list = []
228 for cand in files:
229 base = ntpath.basename(cand)
230 if os.path.isfile(cand) and os.access(cand, os.R_OK):
231 opt_list.append(cand)
232
233 if get_name == True:
234 # NOTE the name includes prefix 'O'
235 # e.g. O1, O2, ONCHW not just 1, 2, NCHW
236 opt_list = [ntpath.basename(f) for f in opt_list]
237 opt_list = [remove_suffix(s, '.cfg') for s in opt_list]
238
239 return opt_list

242def get_arg_parser(target: Optional[str], cmd: str) -> Optional[ArgumentParser]:
243 if not target:
244 return None
245
246 dir_path = os.path.dirname(os.path.realpath(__file__))
247 # for python module naming convention
248 target_name = target.replace('-', '_')
249 command_schema_path = dir_path + f'/../../backends/command/{cmd}/{target_name}.py'
250 if not os.path.isfile(command_schema_path):
251 return None
252
253 # https://docs.python.org/3/library/importlib.html#importing-a-source-file-directly
254 spec = importlib.util.spec_from_file_location(target_name, command_schema_path)
255 module = importlib.util.module_from_spec(spec)
256 sys.modules[target_name] = module
257 spec.loader.exec_module(module)
258
259 if not hasattr(module, "command_schema"):
260 raise RuntimeError('You must implement "command_schema" function')
261
262 parser: ArgumentParser = module.command_schema()
263 parser.target = target
264 return parser

267def detect_one_import_drivers(search_path):
268 """Looks for import drivers in given directory
269
270 Args:
271 search_path: path to the directory where to search import drivers
272
273 Returns:
274 dict: each entry is related to single detected driver,
275 key is a config section name, value is a driver name
276
277 """
278 import_drivers_dict = {}
279 for module_name in os.listdir(search_path):
280 full_path = os.path.join(search_path, module_name)
281 if not os.path.isfile(full_path):
282 continue
283 if module_name.find("one-import-") != 0:
284 continue
285 module_loader = importlib.machinery.SourceFileLoader(module_name, full_path)
286 module_spec = importlib.util.spec_from_loader(module_name, module_loader)
287 module = importlib.util.module_from_spec(module_spec)
288 try:
289 module_loader.exec_module(module)
290 if hasattr(module, "get_driver_cfg_section"):
291 section = module.get_driver_cfg_section()
292 import_drivers_dict[section] = module_name
293 except:
294 pass
295 return import_drivers_dict
