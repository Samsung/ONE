28template <loco::DataType DT>
29bool is_scalar_with_value(luci::CircleConst *node, typename loco::DataTypeImpl<DT>::Type val)
30{
31 if (node->dtype() != DT)
32 return false;
33 if (node->rank() != 0)
34 return false;
35 if (node->size<DT>() != 1)
36 return false;
37 if (node->at<DT>(0) != static_cast<typename loco::DataTypeImpl<DT>::Type>(val))
38 return false;
39
40 return true;
41}

63template <loco::DataType DT> bool transform_min_max_pattern(luci::CircleMaximum *maxi)
64{
65 if (not maxi)
66 return false;
67
68 if (maxi->dtype() != DT)
69 return false;
70
71 luci::CircleConst *maxi_const = nullptr;
72 luci::CircleMinimum *mini = nullptr;
73
74 // There are two ways Maximum takes inputs.
75 // 1. Maximum(x = CircleConst, y = CircleMinimum)
76 // 2. Maximum(x = CircleMinimum, y = CircleConst)
77 if (not luci::fill(&maxi_const, &mini).with_commutative_args_of(maxi))
78 return false;
79
80 // Maximum constant should be scalar whose value is 0.
81 if (not is_scalar_with_value<DT>(maxi_const,
82 static_cast<typename loco::DataTypeImpl<DT>::Type>(0)))
83 return false;
84
85 luci::CircleConst *mini_const = nullptr;
86 loco::Node *mini_input = nullptr;
87
88 // There are two ways Miminum takes inputs.
89 // 1. Miminum(x = CircleNode, y = CircleConst)
90 // 2. Miminum(x = CircleConst, y = CircleNode)
91 if (not luci::fill(&mini_const, &mini_input).with_commutative_args_of(mini))
92 return false;
93
94 // Miminum constant should be scalar whose value is 6.
95 if (not is_scalar_with_value<DT>(mini_const,
96 static_cast<typename loco::DataTypeImpl<DT>::Type>(6)))
97 return false;
98
99 auto name = maxi->name();
100 assert(name.length() > 0);
101
102 // Create Relu6 op
103 auto relu6 = mini->graph()->nodes()->create<luci::CircleRelu6>();
104 relu6->features(mini_input);
105 relu6->name(name + "/Relu6");
106 luci::add_origin(relu6, luci::composite_origin({luci::get_origin(maxi), luci::get_origin(mini)}));
107
108 replace(maxi).with(relu6);
109
110 return true;
111}

118bool TransformMinMaxToRelu6Pass::run(loco::Graph *g)
119{
120 bool changed = false;
121
122 for (auto node : loco::active_nodes(loco::output_nodes(g)))
123 {
124 if (auto maxi = dynamic_cast<luci::CircleMaximum *>(node))
125 {
126 if (transform_min_max_pattern<loco::DataType::FLOAT32>(maxi))
127 changed = true;
128 }
129 }
130
131 return changed;
132}
