25void PercentileComputer::update_qparam(MinMaxObserver *observer)
26{
27 auto minmax_map = observer->minMaxData()->getMap();
28 for (auto iter = minmax_map->begin(); iter != minmax_map->end(); ++iter)
29 {
30 auto node = iter->first;
31 auto minmax = iter->second;
32
33 auto min = getNthPercentile(minmax.min_vector, _min_percentile);
34 auto max = getNthPercentile(minmax.max_vector, _max_percentile);
35
36 auto quantparam = std::make_unique<luci::CircleQuantParam>();
37 quantparam->min.push_back(min);
38 quantparam->max.push_back(max);
39
40 assert(node->quantparam() == nullptr);
41
42 auto mutable_node = const_cast<luci::CircleNode *>(node);
43 mutable_node->quantparam(std::move(quantparam));
44 }
45}
