29std::set<coco::Copy *> linked_copy_instrs(coco::Module *m)
30{
31 std::set<coco::Copy *> res;
32
33 for (uint32_t n = 0; n < m->entity()->instr()->size(); ++n)
34 {
35 auto ins = m->entity()->instr()->at(n);
36 assert(ins != nullptr);
37
38 if (ins->parent() && ins->parent()->parent())
39 {
40 if (auto copy = ins->asCopy())
41 {
42 res.insert(copy);
43 }
44 }
45 }
46
47 return res;
48}

55void eliminate_indirect_copy(enco::Code *code)
56{
57 auto m = code->module();
58
