30std::string export_file_path(const std::string &output_base, const std::string &input,
31 const std::string &ext)

34 auto pos = filename_ext.find_last_of(".");
35 assert(pos > 0);

46void graph_io_to_config_part(loco::Graph *graph, crew::Part &part)
47{
48 assert(graph != nullptr);
49
50 auto *gis = graph->inputs();
51 auto *gos = graph->outputs();
52 for (uint32_t i = 0; i < gis->size(); ++i)
53 {
54 auto *gi = gis->at(i);
55 assert(gi != nullptr);
56 part.inputs.push_back(gi->name());
57 }
58 for (uint32_t i = 0; i < gos->size(); ++i)
59 {
60 auto *go = gos->at(i);
61 assert(go != nullptr);
62 part.outputs.push_back(go->name());
63 }
64}

66void pms2config(const luci::PartedModules &pms, crew::PConfig &pconfig)
67{
68 for (auto &pmodule : pms.pmodules)
69 {
70 auto *graph = pmodule.module->graph();
71
72 crew::Part part;
73 part.model_file = pmodule.name;
74 graph_io_to_config_part(graph, part);
75
76 pconfig.parts.push_back(part);
77 }
78}

85bool export_part_conn_json(const std::string &output_base, const std::string &input,
86 const luci::Module *source, luci::PartedModules &pms)
87{
88 crew::PConfig pconfig;
89
90 // TODO is graph I/O using main graph is enough?
91 auto *graph = source->graph();
92
93 pconfig.source.model_file = input;
94 graph_io_to_config_part(graph, pconfig.source);
95
96 pms2config(pms, pconfig);
97
98 auto filepath_json = export_file_path(output_base, input, ".json");
99 std::ofstream fs(filepath_json.c_str(), std::ofstream::binary | std::ofstream::trunc);
100 if (not fs.good())
101 {
102 std::cerr << "ERROR: Failed to create file: " << filepath_json;
103 return false;
104 }
105 if (not write_json(fs, pconfig))
106 {
107 std::cerr << "ERROR: Failed to write json file: " << filepath_json;
108 return false;
109 }
110 fs.close();
111
112 return true;
113}
