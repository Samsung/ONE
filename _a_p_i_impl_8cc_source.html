204NNFW_STATUS nnfw_query_info_u32(nnfw_session *session, NNFW_INFO_ID id, uint32_t *val)
205{
206 (void)session;
207 switch (id)
208 {
209 case NNFW_INFO_ID_VERSION:
210 if (val)
211 {
212 *val = NNFW_VERSION;
213 return NNFW_STATUS_NO_ERROR;
214 }
215 break;
216 default:
217 return NNFW_STATUS_ERROR;
218 }
219 // It should not be reached.
220 return NNFW_STATUS_ERROR;
221}
