30uint64_t nowMicros()
31{
32 struct timespec ts;
33 clock_gettime(CLOCK_MONOTONIC, &ts);
34 return static_cast<uint64_t>(ts.tv_nsec) / 1e3 + static_cast<uint64_t>(ts.tv_sec) * 1e6;
35}

37void SleepForMicros(uint64_t micros)
38{
39 timespec sleep_time;
40 sleep_time.tv_sec = micros / 1e6;
41 micros -= sleep_time.tv_sec * 1e6;
42 sleep_time.tv_nsec = micros * 1e3;
43 nanosleep(&sleep_time, nullptr);
44}

62void Phases::run(const std::string &tag, const PhaseFunc &exec, const PhaseFunc *post,
63 uint32_t loop_num, bool option_disable)
64{
65 Phase phase{tag, loop_num};
66 PhaseEnum p = getPhaseEnum(tag);
67 for (uint32_t i = 0; i < loop_num; ++i)
68 {
69 if (!option_disable && _option.memory)
70 _mem_poll->start(p);
71
72 uint64_t t = 0u;
73 t = nowMicros();
74
75 exec(phase, i);
76
77 t = nowMicros() - t;
78
79 if (!option_disable && _option.memory)
80 _mem_poll->end(p);
81
82 phase.time.emplace_back(t);
83
84 if (!option_disable && _option.memory)
85 {
86 phase.memory[MemoryType::RSS].emplace_back(_mem_poll->getRssMap().at(p));
87 phase.memory[MemoryType::HWM].emplace_back(_mem_poll->getHwmMap().at(p));
88 phase.memory[MemoryType::PSS].emplace_back(_mem_poll->getPssMap().at(p));
89 }
90
91 if (post)
92 (*post)(phase, i);
93
94 if (_option.run_delay > 0 && p == PhaseEnum::EXECUTE && i != loop_num - 1)
95 {
96 SleepForMicros(_option.run_delay);
97 }
98 }
99
100 _mem_after_run = getVmHWM();
101
102 if (p == PhaseEnum::END_OF_PHASE)
103 {
104 return;
105 }
106
107 assert(_phases.find(tag) == _phases.end());
108 _phases.emplace(tag, phase);
109}
