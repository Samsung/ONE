24bool remove_quant_dequant(luci::CircleDequantize *dequant)
25{
26 assert(dequant != nullptr);
27
28 auto quantize = dynamic_cast<luci::CircleQuantize *>(dequant->input());
29 if (quantize == nullptr)
30 return false;
31
32 auto input_node = loco::must_cast<luci::CircleNode *>(quantize->input());
33
34 replace(dequant).with(input_node);
35
36 return true;
37}

62bool RemoveQuantDequantSeqPass::run(loco::Graph *g)
