96enum class Direction
97{
98 Forward,
99 Backward,
100};
101
102template <Direction> class PlaneInference;
103
104template <> class PlaneInference<Direction::Forward> final
105{
106public:
107 PlaneShape operator()(const PlaneShape &in) const
108 {
109 assert(_pad != nullptr);
110 assert(_window != nullptr);
111 assert(_stride != nullptr);
112
113 uint32_t const raw_input_height = in.height.value();
114 uint32_t const raw_input_width = in.width.value();
115
116 uint32_t const raw_window_height = _window->vertical();
117 uint32_t const raw_window_width = _window->horizontal();
118
119 uint32_t const vertical_padding = _pad->top() + _pad->bottom();
120 uint32_t const horizontal_padding = _pad->left() + _pad->right();
121
122 uint32_t const effective_input_height = raw_input_height + vertical_padding;
123 uint32_t const effective_input_width = raw_input_width + horizontal_padding;
124
125 // NOTE To support "dilation" later
126 uint32_t const effective_window_height = raw_window_height;
127 uint32_t const effective_window_width = raw_window_width;
128
129 uint32_t const vertical_stride = _stride->vertical();
130 uint32_t const horizontal_stride = _stride->horizontal();
131
132 assert((effective_input_height - effective_window_height) % vertical_stride == 0);
133 assert((effective_input_width - effective_window_width) % horizontal_stride == 0);
134
135 PlaneShape res;
136
137 res.height = (effective_input_height - effective_window_height) / vertical_stride + 1;
138 res.width = (effective_input_width - effective_window_width) / horizontal_stride + 1;
139
140 return res;
141 }
142
143public:
144 void pad(const loco::Padding2D *value) { _pad = value; }
145 void window(const loco::Window<2> *value) { _window = value; }
146 void stride(const loco::Stride<2> *value) { _stride = value; }
147
148private:
149 const loco::Padding2D *_pad = nullptr;
150 const loco::Window<2> *_window = nullptr;
151 const loco::Stride<2> *_stride = nullptr;
152};

154template <> class PlaneInference<Direction::Backward> final
155{
156public:
157 PlaneShape operator()(const PlaneShape &in) const
158 {
159 assert(_pad != nullptr);
160 assert(_window != nullptr);
161 assert(_stride != nullptr);
162
163 uint32_t const input_height = in.height.value();
164 uint32_t const input_width = in.width.value();
165
166 uint32_t const vertical_padding = _pad->top() + _pad->bottom();
167 uint32_t const horizontal_padding = _pad->left() + _pad->right();
168
169 uint32_t const raw_window_height = _window->vertical();
170 uint32_t const raw_window_width = _window->horizontal();
171
172 // TODO Support "dilation"
173 uint32_t const effective_window_height = raw_window_height;
174 uint32_t const effective_window_width = raw_window_width;
175
176 uint32_t const vertical_stride = _stride->vertical();
177 uint32_t const horizontal_stride = _stride->horizontal();
178
179 PlaneShape res;
180
181 res.height = vertical_stride * (input_height - 1) + effective_window_height - vertical_padding;
182 res.width = horizontal_stride * (input_width - 1) + effective_window_width - horizontal_padding;
183
184 return res;
185 }
186
187public:
188 void pad(const loco::Padding2D *value) { _pad = value; }
189 void window(const loco::Window<2> *value) { _window = value; }
190 void stride(const loco::Stride<2> *value) { _stride = value; }
191
192private:
193 const loco::Padding2D *_pad = nullptr;
194 const loco::Window<2> *_window = nullptr;
195 const loco::Stride<2> *_stride = nullptr;
196};

209class ForwardShapeInferenceAlgorithm final : public loco::CanonicalNodeVisitor<loco::NodeShape>
210{
211public:
212 ForwardShapeInferenceAlgorithm(const loco::ShapeInferenceRule::Context *ctx) : _ctx{ctx}
213 {
214 // DO NOTHING
215 }
216
217private:
218 const loco::ShapeInferenceRule::Context *_ctx;
219
220private:
221 bool shape_known(const loco::Node *node) const { return _ctx->known(node); }
222 loco::NodeShape node_shape(const loco::Node *node) const { return _ctx->get(node); }
223
224private:
225 loco::NodeShape eltwise_binary_node_shape(const loco::Node *node)
226 {
227 // This helper works only for binary node.
228 assert(node->arity() == 2);
229
230 auto lhs_shape = node_shape(node->arg(0));
231 auto rhs_shape = node_shape(node->arg(1));
232
233 // ASSERT: lhs_shape == rhs_shape
234
235 return lhs_shape;
236 }
237
238public:
239 // CASE: AvgPool2D
240 loco::NodeShape visit(const loco::AvgPool2D *node) final
241 {
242 PlaneInference<Direction::Forward> infer_plane_shape;
243
244 infer_plane_shape.pad(node->pad());
245 infer_plane_shape.window(node->window());
246 infer_plane_shape.stride(node->stride());
247
248 auto input_feature_shape = node_shape(node->ifm()).as<loco::FeatureShape>();
249 auto input_plane_shape = make_plane_shape(input_feature_shape);
250 auto output_plane_shape = infer_plane_shape(input_plane_shape);
251 auto output_feature_shape = input_feature_shape; // AvgPool2D does not change count/depth
252
253 // Update the height/width of output_feature_shape with that of output_plane_shape
254 update(output_feature_shape).with(output_plane_shape);
255
256 return loco::NodeShape{output_feature_shape};
257 }
258
259 // CASE: BiasDecode
260 loco::NodeShape visit(const loco::BiasDecode *node) final
261 {
262 // The input of BiasDecode SHOULD BE a bias!
263 assert(node_shape(node->input()).domain() == loco::Domain::Bias);
264 auto input_bias_shape = node_shape(node->input()).as<loco::BiasShape>();
265
266 loco::TensorShape output_tensor_shape;
267
268 output_tensor_shape.rank(1);
269 output_tensor_shape.dim(0) = input_bias_shape.length();
270
271 return loco::NodeShape{output_tensor_shape};
272 }
273
274 // CASE: BiasEncode
275 loco::NodeShape visit(const loco::BiasEncode *node) final
276 {
277 // The input of BiasEncode SHOULD BE a tensor!
278 assert(node_shape(node->input()).domain() == loco::Domain::Tensor);
279 auto input_tensor_shape = node_shape(node->input()).as<loco::TensorShape>();
280
281 loco::BiasShape output_bias_shape;
282
283 output_bias_shape.length() = input_tensor_shape.dim(0);
284
285 return loco::NodeShape{output_bias_shape};
286 }

288 // CASE: ConstGen
289 loco::NodeShape visit(const loco::ConstGen *node) final
290 {
291 loco::TensorShape tensor_shape;
292
293 tensor_shape.rank(node->rank());
294 for (uint32_t axis = 0; axis < node->rank(); ++axis)
295 {
296 tensor_shape.dim(axis) = node->dim(axis);
297 }
298
299 return loco::NodeShape{tensor_shape};
300 }

302 // CASE: Conv2D
303 loco::NodeShape visit(const loco::Conv2D *node) final
304 {
305 auto filter_shape = node_shape(node->ker()).as<loco::FilterShape>();
306 auto filter_window = window_of(filter_shape);
307
308 PlaneInference<Direction::Forward> infer_plane_shape;
309
310 infer_plane_shape.pad(node->pad());
311 infer_plane_shape.window(&filter_window);
312 infer_plane_shape.stride(node->stride());
313
314 auto input_feature_shape = node_shape(node->ifm()).as<loco::FeatureShape>();
315 auto input_plane_shape = make_plane_shape(input_feature_shape);
316 auto output_plane_shape = infer_plane_shape(input_plane_shape);
317
318 loco::FeatureShape output_feature_shape;
319
320 // "COUNT" does not change
321 output_feature_shape.count() = input_feature_shape.count();
322 // "DEPTH" depends on # of filters
323 output_feature_shape.depth() = filter_shape.count();
324 // Update the height/width of output_feature_shape with that of output_plane_shape
325 update(output_feature_shape).with(output_plane_shape);
326
327 return loco::NodeShape{output_feature_shape};
328 }

330 // CASE: DepthwiseConv2D
331 loco::NodeShape visit(const loco::DepthwiseConv2D *node) final
332 {
333 auto depthwise_filter_shape = node_shape(node->ker()).as<loco::DepthwiseFilterShape>();
334 auto dpethwise_filter_window = window_of(depthwise_filter_shape);
335
336 PlaneInference<Direction::Forward> infer_plane_shape;
337
338 infer_plane_shape.pad(node->pad());
339 infer_plane_shape.window(&dpethwise_filter_window);
340 infer_plane_shape.stride(node->stride());
341
342 auto input_feature_shape = node_shape(node->ifm()).as<loco::FeatureShape>();
343 auto input_plane_shape = make_plane_shape(input_feature_shape);
344 auto output_plane_shape = infer_plane_shape(input_plane_shape);
345
346 loco::FeatureShape output_feature_shape;
347
348 // "COUNT" does not change
349 output_feature_shape.count() = input_feature_shape.count();
350 // "DEPTH" depends on [in_channels * channel_multiplier] of filters
351 output_feature_shape.depth() = loco::Dimension(depthwise_filter_shape.depth().value() *
352 depthwise_filter_shape.multiplier().value());
353 // Update the height/width of output_feature_shape with that of output_plane_shape
354 update(output_feature_shape).with(output_plane_shape);
355
356 return loco::NodeShape{output_feature_shape};
357 }

557 // CASE: TensorConcat
558 loco::NodeShape visit(const loco::TensorConcat *node)
559 {
560 auto const lhs_shape = node_shape(node->lhs()).as<loco::TensorShape>();
561 auto const rhs_shape = node_shape(node->rhs()).as<loco::TensorShape>();
562
563 assert(lhs_shape.rank() == rhs_shape.rank());
564 uint32_t const out_rank = lhs_shape.rank();
565
566 loco::TensorShape out_shape;
567
568 out_shape.rank(out_rank);
569
570 for (uint32_t axis = 0; axis < out_rank; ++axis)
571 {
572 if (axis == node->axis())
573 {
574 out_shape.dim(axis) = lhs_shape.dim(axis).value() + rhs_shape.dim(axis).value();
575 }
576 else
577 {
578 assert(lhs_shape.dim(axis) == rhs_shape.dim(axis));
579 out_shape.dim(axis) = lhs_shape.dim(axis);
580 }
581 }
582
583 return loco::NodeShape{out_shape};
584 }

619 // CASE: TensorTranspose
620 loco::NodeShape visit(const loco::TensorTranspose *node) final
621 {
622 loco::TensorShape output_shape;
623
624 auto input_shape = node_shape(node->input()).as<loco::TensorShape>();
625 assert(input_shape.rank() == node->perm()->size());
626
627 output_shape.rank(input_shape.rank());
628
629 for (uint32_t output_axis = 0; output_axis < output_shape.rank(); output_axis++)
630 {
631 auto new_dim = input_shape.dim(node->perm()->axis(output_axis));
632 output_shape.dim(output_axis) = new_dim;
633 }
634
635 return loco::NodeShape(output_shape);
636 }

643
644 PlaneInference<Direction::Backward> infer_plane_shape;
645
646 infer_plane_shape.pad(node->pad());
647 infer_plane_shape.window(&filter_window);
648 infer_plane_shape.stride(node->stride());
649
650 auto input_feature_shape = node_shape(node->ifm()).as<loco::FeatureShape>();
651 auto input_plane_shape = make_plane_shape(input_feature_shape);
652 auto output_plane_shape = infer_plane_shape(input_plane_shape);
653
654 loco::FeatureShape output_feature_shape;
655
656 // "COUNT" does not change
657 output_feature_shape.count() = input_feature_shape.count();
658 // Output "DEPTH" depends on count of filters
659 output_feature_shape.depth() = filter_shape.count();
660 // Update the height/width of output_feature_shape with that of output_plane_shape
661 update(output_feature_shape).with(output_plane_shape);
662
663 return loco::NodeShape{output_feature_shape};
664 }

666 // CASE: TensorConstantPad
667 loco::NodeShape visit(const loco::TensorConstantPad *node) final
668 {
669 auto const tensor_shape = loco::shape_get(node->input()).as<loco::TensorShape>();
670 auto padding = node->padding();
671
672 loco::TensorShape out_shape;
673 out_shape.rank(tensor_shape.rank());
674 for (uint32_t axis = 0; axis < out_shape.rank(); ++axis)
675 {
676 out_shape.dim(axis) =
677 tensor_shape.dim(axis).value() + padding->front(axis) + padding->back(axis);
678 }
679
680 return loco::NodeShape{out_shape};
681 }
