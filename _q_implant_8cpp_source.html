47loco::DataType str_to_dtype(const std::string &str)
48{
49 auto lower_case_str = str;
50 std::transform(lower_case_str.begin(), lower_case_str.end(), lower_case_str.begin(),
51 [](unsigned char c) { return std::tolower(c); });
52
53 if (lower_case_str.compare("int8") == 0)
54 return loco::DataType::S8;
55 if (lower_case_str.compare("uint8") == 0)
56 return loco::DataType::U8;
57 if (lower_case_str.compare("int16") == 0)
58 return loco::DataType::S16;
59 if (lower_case_str.compare("int32") == 0)
60 return loco::DataType::S32;
61 if (lower_case_str.compare("int64") == 0)
62 return loco::DataType::S64;
63
64 throw std::runtime_error("Invalid dtype detected. " + str);
65}

85Json::Value load_json(const std::string &path)
86{
87 Json::Value root;
88 std::ifstream ifs(path);
89
90 // Failed to open cfg file
91 if (not ifs.is_open())
92 throw std::runtime_error("Cannot open config file. " + path);
93
94 Json::CharReaderBuilder builder;
95 JSONCPP_STRING errs;
96
97 // Failed to parse
98 if (not parseFromStream(builder, ifs, &root, &errs))
99 throw std::runtime_error("Cannot parse config file (json format). " + errs);
100
101 return root;
102}

148template <loco::DataType DT> void set_value(luci::CircleConst *node, const std::string &value_path)
149{
150 assert(node); // FIX CALLER UNLESS
151 assert(node->dtype() == DT); // FIX CALLER UNLESS
152
153 std::vector<unsigned long> shape;
154 bool fortran_order;
155 std::vector<typename loco::DataTypeImpl<DT>::Type> values;
156 npy::LoadArrayFromNumpy(value_path, shape, fortran_order, values);
157
158 THROW_UNLESS(shape.size() == node->rank());
159 THROW_UNLESS(fortran_order == false);
160
161 uint32_t value_size = 1;
162 for (uint32_t i = 0; i < node->rank(); i++)
163 {
164 THROW_UNLESS(node->dim(i).known());
165 THROW_UNLESS(node->dim(i).value() == shape[i]);
166
167 value_size *= node->dim(i).value();
168 }
169
170 node->size<DT>(value_size);
171 for (uint32_t i = 0; i < value_size; i++)
172 {
173 node->at<DT>(i) = values.at(i);
174 }
175}

205void QImplant::write(loco::Graph *g)
206{
207 const auto root = load_json(_path);
208 const auto dir_path = directory_path(_path);
209
210 std::unordered_map<std::string, luci::CircleNode *> name_to_node;
211 for (auto node : loco::all_nodes(g))
212 {
213 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
214 if (circle_node->opcode() == luci::CircleOpcode::CIRCLEOUTPUT)
215 {
216 continue;
217 }
218
219 name_to_node[circle_node->name()] = circle_node;
220 }
221
222 THROW_UNLESS(root.isObject());
223
224 for (const auto tensor_name : root.getMemberNames())
225 {
226 const auto tensor = root[tensor_name];
227
228 verify_tensor(tensor);
229
230 const auto scale_path = dir_path + '/' + tensor["scale"].asString();
231 const auto zerop_path = dir_path + '/' + tensor["zerop"].asString();
232 const auto quantized_dimension = tensor["quantized_dimension"].asUInt();
233 const auto dtype = str_to_dtype(tensor["dtype"].asString());
234
235 auto node = name_to_node.at(tensor_name);
236
237 // Node must be fp32
238 THROW_UNLESS(node->dtype() == loco::DataType::FLOAT32);
239
240 node->quantparam(std::make_unique<luci::CircleQuantParam>());
241
242 set_dtype(node, dtype);
243 set_scale(node, scale_path);
244 set_zerop(node, zerop_path);
245 set_quantized_dimension(node, quantized_dimension);
246
247 if (tensor.isMember("value"))
248 {
249 auto const_node = loco::must_cast<luci::CircleConst *>(node);
250 const auto value_path = dir_path + '/' + tensor["value"].asString();
251
252 set_value(const_node, value_path, dtype);
253 }
254 }
255
256 forward_qparam(g);
257
258 // Update output nodes
259 auto graph_outputs = g->outputs();
260 assert(graph_outputs); // FIX_CALLER_UNLESS
261 for (auto node : loco::output_nodes(g))
262 {
263 auto out_node = loco::must_cast<luci::CircleOutput *>(node);
264 auto from_node = loco::must_cast<luci::CircleNode *>(out_node->from());
265
266 THROW_UNLESS(from_node->quantparam());
267
268 out_node->quantparam(std::make_unique<luci::CircleQuantParam>());
269 out_node->quantparam()->scale = from_node->quantparam()->scale;
270 out_node->quantparam()->zerop = from_node->quantparam()->zerop;
271 out_node->quantparam()->quantized_dimension = from_node->quantparam()->quantized_dimension;
272 out_node->dtype(from_node->dtype());
273
274 auto graph_output = graph_outputs->at(out_node->index());
275 graph_output->dtype(out_node->dtype());
276 }
277
278 // Verify quantized model
279 for (auto node : loco::active_nodes(loco::output_nodes(g)))
280 {
281 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
282
283 // Throw an exception if dtype is not float32
284 // TODO Operator-level verification (ex: using QuantizedModelVerifier)
285 THROW_UNLESS(circle_node->dtype() != loco::DataType::FLOAT32);
286 }
287}

289void QImplant::forward_qparam(loco::Graph *g)
290{
291 /*
292 * TODO: add comment about how to add to the set
293 *
294 * If the operator doesn't change input tensor value,
295 * (Operator don't change input quantization parameter to output tensor)
296 * the operator's quantization parameter can be forwarded
297 */
298 std::set<luci::CircleOpcode> forwardable_opcode;
299 forwardable_opcode.emplace(luci::CircleOpcode::RESHAPE);
300 forwardable_opcode.emplace(luci::CircleOpcode::SPLIT);
301 forwardable_opcode.emplace(luci::CircleOpcode::CIRCLESPLITOUT);
302 forwardable_opcode.emplace(luci::CircleOpcode::TRANSPOSE);
303 // TODO add more Ops
304
305 auto forwardable = [&forwardable_opcode](luci::CircleOpcode opcode) {
306 return forwardable_opcode.find(opcode) != forwardable_opcode.end();
307 };
308
309 for (auto node : loco::active_nodes(loco::output_nodes(g)))
310 {
311 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
312
313 auto quantparam = circle_node->quantparam();
314 if (quantparam == nullptr)
315 continue;
316
317 for (auto successor : loco::succs(node))
318 {
319 auto successor_node = loco::must_cast<luci::CircleNode *>(successor);
320 if (successor_node->quantparam() == nullptr)
321 {
322 if (!forwardable(successor_node->opcode()))
323 continue;
324 copy_quantparam(circle_node, successor_node);
325 copy_dtype(circle_node, successor_node);
326 }
327 }
328 }
