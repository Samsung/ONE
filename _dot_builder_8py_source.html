28# Class to build dot graph from qerror_map
29class DotBuilder:
30 def __init__(self, circle_path: str, dot_path: str, metric: str, colors: str):
31 '''
32 circle_path: Path to the fp32 circle model (required to build graph)
33 dot_path: Path to the saved dot file
34 metric: Metric name (ex: MPEIR, MSE)
35 colors: List of color slots [{'b': begin, 'e': end, 'c':color}, ..]
36 '''
37 with open(circle_path, 'rb') as f:
38 self._model = Model.Model.GetRootAsModel(f.read())
39
40 if self._model.SubgraphsLength() != 1:
41 raise RuntimeError("Only one subgraph is supported")
42
43 self._name = Path(circle_path).name
44 self._dot_path = dot_path
45 self._metric = metric
46 self._colors = colors
47
48 # Return color (RGB) for the given qerror
49 def _get_color(self, qerror: float):
50 # Find a slot where qerror is in the range of [begin, end]
51 for slot in self._colors:
52 begin = slot['b']
53 end = slot['e']
54 if (qerror > begin or math.isclose(
55 qerror, begin)) and (qerror < end or math.isclose(qerror, end)):
56 return slot['c']
57
58 # Use the first color if qerror is smaller than the first begin
59 if qerror < self._colors[0]['b']:
60 return self._colors[0]['c']
61
62 # Use the last color if qerror is larger than the last end
63 if qerror > self._colors[-1]['e']:
64 return self._colors[-1]['c']
65
66 raise RuntimeError("Color ID not found. QError: " + str(qerror))
67
68 # Generate a pydot.Node object which represents the color table
69 def _gen_color_table(self):
70 color_table = "< <table>"
71 for slot in self._colors:
72 begin = slot['b']
73 end = slot['e']
74 color = slot['c']
75 color_table += "<tr> <td bgcolor=\""
76 color_table += color
77 color_table += "\">"
78 color_table += self._metric + ": {:.4f}".format(
79 begin) + " ~ " + "{:.4f}".format(end)
80 color_table += "</td> </tr>"
81 color_table += "</table> >"
82 return pydot.Node("color_table", shape='none', label=color_table)
83
84 # Save dot graph to self._dot_path
85 def save(self, qerror_map: dict):
86 '''
87 qerror_map: Dictionary of {op_name (str) -> qerror (float)}
88 '''
89 # Build graph
90 DOT = pydot.Dot(self._name, graph_type="digraph")
91
92 # Add color table
93 DOT.add_node(self._gen_color_table())
94
95 # Dictionary from output tensor name to Op name {str -> str}
96 # This dict is for handling Ops with multiple output tensors.
97 # We use the first output tensor's name as the Op name, following
98 # the implementation of luci IR
99 output_to_op = dict()
100
101 graph = self._model.Subgraphs(0)
102
103 # Add Input nodes
104 for i in range(graph.InputsLength()):
105 name = _tensor_name(graph, graph.Inputs(i))
106 output_to_op[name] = name
107 DOT.add_node(pydot.Node(name))
108
109 # Add Output nodes
110 for i in range(graph.OutputsLength()):
111 name = _tensor_name(graph, graph.Outputs(i))
112 output_to_op[name] = name
113 DOT.add_node(pydot.Node(name))
114
115 # Add Edges
116 for i in range(graph.OperatorsLength()):
117 op = graph.Operators(i)
118 # Name of the first output tensor
119 op_name = _tensor_name(graph, op.Outputs(0))
120 if op.OutputsLength() == 0:
121 print(op_name)
122 continue
123
124 if op_name in qerror_map:
125 qerror = qerror_map[op_name]
126 node = pydot.Node(
127 op_name,
128 style="filled",
129 fillcolor=self._get_color(qerror),
130 xlabel=self._metric + ": {:.4f}".format(qerror))
131 else:
132 # qerror_map does not have qerror info for the op. Color gray.
133 # When this happen? visq does not collect qerror info of some Ops
134 # For example, Reshape Op does not change values, so its qerror
135 # info is not collected.
136 node = pydot.Node(op_name, style="filled", fillcolor='gray')
137
138 DOT.add_node(node)
139
140 for output_idx in range(op.OutputsLength()):
141 output_name = _tensor_name(graph, op.Outputs(output_idx))
142 # Set Op name as the first output tensor name (op_name)
143 output_to_op[output_name] = op_name
144
145 for j in range(op.InputsLength()):
146 op_input = op.Inputs(j)
147
148 # Optional input case (ex: For TConv with no bias, bias is -1)
149 if op_input == -1:
150 continue
151
152 op_input_name = _tensor_name(graph, op_input)
153 if op_input_name not in output_to_op:
154 continue
155
156 # Use the saved name to handle multiple outputs
157 op_input_name = output_to_op[op_input_name]
158 DOT.add_edge(pydot.Edge(op_input_name, op_name))
159
160 DOT.write(self._dot_path)
