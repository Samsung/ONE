39void execute_node(loco::TensorConstantPad *pad)
40{
41 validate(pad, "TensorConstantPad is nullptr");
42
43 auto input_data = annot_data(pad->input());
44 auto input_domain = annot_domain(pad->input());
45 validate(input_data, "Input not ready");
46 validate(input_domain == loco::Domain::Tensor, "Input domain of TensorConstantPad is not Tensor");
47
48 auto input_shape = input_data->shape();
49 const uint32_t input_rank = input_shape->rank();
50
51 auto padding = pad->padding();
52 validate(input_rank == padding->rank(), "input and padding should have same rank");
53
54 auto constant_node = pad->constant();
55 auto constant_data = annot_data(constant_node);
56 validate(constant_data != nullptr, "constant is not found");
57 validate(constant_data->dtype() == input_data->dtype(), "constant and input have same data type");
58 validate(constant_data->shape()->rank() == 1 && constant_data->shape()->dim(0) == 1,
59 "constant should have one rank with one dimension at zero axis");
60
61 std::unique_ptr<NodeData> pad_data = nullptr;
62 Index base_index;
63 base_index.resize(input_rank);
64
65 // Tensor is padded by relocating its base.
66 // padded output index = input index + base index
67 for (uint32_t axis = 0; axis < padding->rank(); axis++)
68 {
69 base_index.at(axis) = padding->front(axis);
70 }
71
72 // calculate output shape
73 Shape output_shape;
74 output_shape.resize(input_rank);
75 for (uint32_t i = 0; i < input_rank; i++)
76 {
77 output_shape.dim(i) = input_shape->dim(i) + padding->front(i) + padding->back(i);
78 }
79
80 switch (input_data->dtype())
81 {
82 case loco::DataType::FLOAT32:
83 {
84 auto input_buf = input_data->as_f32_bufptr();
85 auto constant_data_buf = constant_data->as_f32_bufptr();
86 const auto constant_value = constant_data_buf->at(Index{0});
87
88 auto output_buf = make_buffer<float, LexicalLayout>(output_shape);
89
90 for (IndexEnumerator ie{*input_shape}, oe{output_shape}; oe.valid(); oe.advance())
91 {
92 auto input_index = ie.current();
93 auto output_index = oe.current();
94
95 if ((input_index + base_index) == output_index)
96 {
97 output_buf.at(output_index) = input_buf->at(input_index);
98 ie.advance();
99 }
100 else
101 {
102 output_buf.at(output_index) = constant_value;
103 }
104 }
105
106 pad_data = make_data(output_buf);
107 break;
108 }
109 default:
110 throw std::runtime_error("NYI for this DataType");
111 }
112
113 assert(pad_data != nullptr);
114 annot_data(pad, std::move(pad_data));
115 annot_domain(pad, annot_domain(pad->input()));
116}
