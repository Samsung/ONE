35 std::vector<uint32_t>
36 ComputeScratchpadSizeBatchMatMul(const luci::CircleBatchMatMul *batch_mat_mul) final
37 {
38 const auto lhs = loco::must_cast<luci::CircleNode *>(batch_mat_mul->x());
39 const auto rhs = loco::must_cast<luci::CircleNode *>(batch_mat_mul->y());
40
41 std::vector<uint32_t> scratchpad_sizes;
42
43 // Scratchpad for lhs
44 uint32_t scratchpad_size = 1;
45 for (int32_t i = 0; i < lhs->rank(); ++i)
46 scratchpad_size *= lhs->dim(i).value();
47
48 scratchpad_sizes.push_back(scratchpad_size * loco::size(lhs->dtype()));
49
50 // Scratchpad for rhs
51 scratchpad_size = 1;
52 for (int32_t i = 0; i < rhs->rank(); ++i)
53 scratchpad_size *= rhs->dim(i).value();
54
55 scratchpad_sizes.push_back(scratchpad_size * loco::size(rhs->dtype()));
56
57 return scratchpad_sizes;
58 }

60 uint32_t ComputeScratchpadSizeConv2d(const luci::CircleConv2D *conv) final
61 {
62 const auto conv_input = loco::must_cast<luci::CircleNode *>(conv->input());
63 const auto filter = loco::must_cast<luci::CircleNode *>(conv->filter());
64
65 const uint32_t stride_height = conv->stride()->h();
66 const uint32_t stride_width = conv->stride()->w();
67
68 const uint32_t dilation_height_factor = conv->dilation()->h();
69 const uint32_t dilation_width_factor = conv->dilation()->w();
70
71 const uint32_t filter_height = filter->dim(1).value();
72 const uint32_t filter_width = filter->dim(2).value();
73
74 const bool need_dilated_im2col = dilation_height_factor != 1 || dilation_width_factor != 1;
75 const bool need_non_dilated_im2col =
76 stride_height != 1 || stride_width != 1 || filter_height != 1 || filter_width != 1;
77 const bool need_im2col = conv_input->dtype() != loco::DataType::S16 &&
78 (need_dilated_im2col || need_non_dilated_im2col);
79
80 if (!need_im2col)
81 {
82 return 0;
83 }
84
85 const uint32_t input_depth = conv_input->dim(3).value();
86 const uint32_t batches = conv_input->dim(0).value();
87
88 const uint32_t output_height = conv->dim(1).value();
89 const uint32_t output_width = conv->dim(2).value();
90
91 return batches * output_height * output_width * input_depth * filter_height * filter_width *
92 size(conv_input->dtype());
93 }

102 std::vector<uint32_t> ComputeScratchpadSizeSVDF(const luci::CircleSVDF *svdf) final
103 {
104 const auto svdf_input = loco::must_cast<luci::CircleNode *>(svdf->input());
105 const auto weight_feature_input = loco::must_cast<luci::CircleNode *>(svdf->weight_feature());
106
107 if (svdf_input->dtype() == loco::DataType::FLOAT32 and
108 (weight_feature_input->dtype() == loco::DataType::S8 or
109 weight_feature_input->dtype() == loco::DataType::U8))
110 {
111 throw std::runtime_error("Hybrid type is not currently supported for linux platform");
112 }
113
114 std::vector<uint32_t> scratchpad_sizes;
115
116 const auto batch_size = svdf_input->dim(0).value();
117 const auto num_filters = weight_feature_input->dim(0).value();
118 const auto rank = svdf->svdf_rank();
119 const auto num_units = num_filters / rank;
120
121 if (svdf_input->dtype() == loco::DataType::S8)
122 {
123 scratchpad_sizes.push_back(batch_size * num_filters * sizeof(int32_t));
124 scratchpad_sizes.push_back(batch_size * num_units * sizeof(int32_t));
125 }
126 else
127 {
128 scratchpad_sizes.push_back(batch_size * num_filters * sizeof(float));
129 }
130
131 return scratchpad_sizes;
132 }
