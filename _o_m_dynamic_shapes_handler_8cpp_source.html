27// Function to obtain index for current operation which has weight tensor, which will be sparse
28// updated If operation don't support such behaviour - return -1
29int32_t getWeightTensorIndexForOperatorWithOpcode(const circle::OperatorCode *opcode)
30{
31 switch (opcode->builtin_code())
32 {
33 // For Conv2D and for FullyConnected return index for weight tensor - is 1
34 case circle::BuiltinOperator_CONV_2D:
35 case circle::BuiltinOperator_FULLY_CONNECTED:
36 return 1;
37 default:
38 break;
39 }
40 return -1;
41}

43OMRuntimeShape createDynamicRuntimeShapeForOperator(OMRuntimeShape shape,
44 const circle::OperatorCode *opcode,
45 const float partition_size)
46{
47 assert(partition_size > 1.0);
48 switch (opcode->builtin_code())
49 {
50 // For Conv2D and for FullyConnected return index for weight tensor - is 1
51 case circle::BuiltinOperator_CONV_2D:
52 case circle::BuiltinOperator_FULLY_CONNECTED:
53 {
54 auto first_dim_val = shape.dims(0);
55 assert(partition_size <= static_cast<float>(first_dim_val));
56 assert(partition_size > 0);
57 if (partition_size == 0)
58 return shape;
59 first_dim_val = static_cast<int32_t>(static_cast<float>(first_dim_val) / partition_size);
60 assert(first_dim_val > 0);
61 shape.setDim(0, first_dim_val);
62 }
63 break;
64 default:
65 break;
66 }
67
68 return shape;
69}

73/*
74 * Import dynamic shapes from train config file data:
75 * Some tensors can have sparse tensor backpropagation scheme (train rank)
76 */
77OMStatus OMDynamicShapesHandler::importDynamicShapesFromTrainConfig(
78 core::OMRuntimeStorage &storage, core::OMRuntimeContext &context,
79 core::train::OMTrainingStorage &train_storage)
80{
81 std::unordered_map<uint16_t, uint8_t> train_op_indexes_to_train_rank =
82 context.getTrainableOpsIndexes();
83 const auto opcodes = context.getCircleOpcodes();
84
85 // Goes over pairs of op index and train rank value
86 for (auto &p : train_op_indexes_to_train_rank)
87 {
88 const uint16_t op_index = p.first;
89 const auto train_rank = static_cast<OpTrainableRankType>(p.second);
90
91 switch (train_rank)
92 {
93 case core::LOWER_1_2_PART:
94 case core::UP_1_2_PART:
95 {
96 const auto cur_op = context.getCircleOperatorAt(op_index);
97 const auto opcode = opcodes->operator[](cur_op->opcode_index());
98
99 int32_t res_index = getWeightTensorIndexForOperatorWithOpcode(opcode);
100 // The operation doesn't support such behaviour
101 if (res_index == -1)
102 continue;
103
104 auto tensor_local_index = static_cast<uint16_t>(res_index);
105 auto tensor_index = cur_op->inputs()->operator[](tensor_local_index);
106 auto tensor = context.getTensorByIndex(tensor_index);
107 OMRuntimeShape old_shape(tensor);
108 const float partition_size = 2.f;
109 OMRuntimeShape new_shape =
110 createDynamicRuntimeShapeForOperator(old_shape, opcode, partition_size);
111 storage.setDynamicRuntimeShape(tensor_index, new_shape);
112
113 train_storage.addTrainRank(tensor_index, train_rank);
114
115 break;
116 }
117 default:
118 continue;
119 }
120 }
121
122 return Ok;
123}
