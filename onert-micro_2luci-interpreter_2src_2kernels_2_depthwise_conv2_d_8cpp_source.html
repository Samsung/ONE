30int32_t compute_padding_h(const circle::Tensor *input, const circle::Tensor *filter,
31 const circle::DepthwiseConv2DOptions *options)
32{
33 const int32_t input_height = Tensor::dim(input, 1);
34 const int32_t filter_height = Tensor::dim(filter, 1);
35 const int32_t output_height =
36 kernels::computeOutputSize(luci_padding(options->padding()), input_height, filter_height,
37 options->stride_h(), options->dilation_h_factor());
38
39 const auto padding_height = kernels::computePadding(
40 options->stride_h(), options->dilation_h_factor(), input_height, filter_height, output_height);

42}

61void evalFloat(const circle::Tensor *input, const circle::Tensor *filter,
62 const circle::Tensor *bias, const circle::Tensor *output,
63 const circle::DepthwiseConv2DOptions *options, BaseRuntimeGraph *runtime_graph)
64{
65 float activation_min{};
66 float activation_max{};
67 kernels::calculateActivationRange(luci_actfunc(options->fused_activation_function()),
68 &activation_min, &activation_max);
69
70 luci_interpreter_pal::ConvParams params{};
71 params.padding_values.height = compute_padding_h(input, filter, options);
72 params.padding_values.width = compute_padding_w(input, filter, options);
73 params.stride_height = options->stride_h();
74 params.stride_width = options->stride_w();
75 params.dilation_height_factor = options->dilation_h_factor();
76 params.dilation_width_factor = options->dilation_w_factor();
77 params.float_activation_min = activation_min;
78 params.float_activation_max = activation_max;
79 params.depth_multiplier = options->depth_multiplier();
80
81 auto *input_data = runtime_graph->getDataByTensor(input);
82 auto *output_data = runtime_graph->getDataByTensor(output);
83
84 auto *filter_data = runtime_graph->getConstDataByTensor(filter);
85 auto *bias_data = runtime_graph->getConstDataByTensor(bias);
86
87 int32_t input_shape[kMaxSmallSize];
88 kernels::getTensorDims(input, runtime_graph, input_shape);
89
90 int32_t filter_shape[kMaxSmallSize];
91 kernels::getTensorDims(filter, runtime_graph, filter_shape);
92
93 int32_t output_shape[kMaxSmallSize];
94 kernels::getTensorDims(output, runtime_graph, output_shape);
95
96 luci_interpreter_pal::DepthwiseConv2D(
97 params, input_shape, kernels::getTensorData<float>(input_data), filter_shape,
98 kernels::getTensorData<float>(filter_data), kernels::getTensorData<float>(bias_data),
99 output_shape, kernels::getTensorData<float>(output_data));
100}

106void configure_kernel_CircleDepthwiseConv2D(const circle::Operator *cur_op,
107 BaseRuntimeGraph *runtime_graph)
108{
109 const auto input_index = cur_op->inputs()->operator[](0);
110 const auto filter_index = cur_op->inputs()->operator[](1);
111 const auto bias_index = cur_op->inputs()->operator[](2);
112 const auto output_index = cur_op->outputs()->operator[](0);
113
114 assert(input_index != -1);
115 assert(filter_index != -1);
116 assert(output_index != -1);
117
118 const auto input = runtime_graph->getCircleTensorByIndex(input_index);
119 const auto filter = runtime_graph->getCircleTensorByIndex(filter_index);
120 const auto bias = runtime_graph->getCircleTensorByIndex(bias_index);
121 const auto output = runtime_graph->getCircleTensorByIndex(output_index);
122
123 assert(input != nullptr);
124 assert(filter != nullptr);
125
126 auto filter_data = runtime_graph->getConstDataByTensor(filter);
127
128 assert(filter_data != nullptr);
129
130 const auto *options = cur_op->builtin_options_as_DepthwiseConv2DOptions();
131
132 if (Tensor::element_type(input) == DataType::FLOAT32 &&
133 Tensor::element_type(filter) == DataType::FLOAT32)
134 {
135 LUCI_INTERPRETER_CHECK(bias == nullptr || Tensor::element_type(bias) == DataType::FLOAT32);
136 }
137 else
138 {
139 assert(false && "Unsupported type.");
140 }
141 LUCI_INTERPRETER_CHECK(Tensor::element_type(output) == Tensor::element_type(input));
142 LUCI_INTERPRETER_CHECK(Tensor::num_dims(input) == 4 && Tensor::num_dims(filter) == 4);
143
144 const int32_t output_depth = Tensor::dim(output, 3);
145 LUCI_INTERPRETER_CHECK(bias == nullptr ||
146 (Tensor::num_dims(bias) == 1 && Tensor::dim(bias, 0) == output_depth));
147
148 switch (options->fused_activation_function())
149 {
150 case circle::ActivationFunctionType_NONE:
151 case circle::ActivationFunctionType_RELU:
152 case circle::ActivationFunctionType_RELU6:
153 case circle::ActivationFunctionType_RELU_N1_TO_1:
154 break;
155 default:
156 assert(false && "Unsupported fused activation");
157 }
158}

160void execute_kernel_CircleDepthwiseConv2D(const circle::Operator *cur_op,
161 BaseRuntimeGraph *runtime_graph)
162{
163 const auto input_index = cur_op->inputs()->operator[](0);
164 const auto weight_index = cur_op->inputs()->operator[](1);
165 const auto bias_index = cur_op->inputs()->operator[](2);
166 const auto output_index = cur_op->outputs()->operator[](0);
167
168 assert(input_index != -1);
169 assert(weight_index != -1);
170 assert(output_index != -1);
171
172 const auto input = runtime_graph->getCircleTensorByIndex(input_index);
173 const auto weights = runtime_graph->getCircleTensorByIndex(weight_index);
174 const auto bias = runtime_graph->getCircleTensorByIndex(bias_index);
175 const auto output = runtime_graph->getCircleTensorByIndex(output_index);
176
177 assert(input != nullptr);
178 assert(weights != nullptr);
179 assert(output != nullptr);
180
181 const auto *options = cur_op->builtin_options_as_DepthwiseConv2DOptions();
182
183 switch (Tensor::element_type(input))
184 {
185#ifndef DIS_FLOAT
186 case DataType::FLOAT32:
187 if (Tensor::element_type(weights) == DataType::FLOAT32)
188 {
189 evalFloat(input, weights, bias, output, options, runtime_graph);
190 break;
191 }
192#endif // DIS_FLOAT
193 default:
194 assert(false && "Unsupported type.");
195 }
196}
