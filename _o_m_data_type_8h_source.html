144inline uint32_t size(OMDataType data_type)
145{
146 switch (data_type)
147 {
148 case OMDataType::S8:
149 return sizeof(OMDataTypeImpl<OMDataType::S8>::Type);
150 case OMDataType::U8:
151 return sizeof(OMDataTypeImpl<OMDataType::U8>::Type);
152 case OMDataType::S16:
153 return sizeof(OMDataTypeImpl<OMDataType::S16>::Type);
154 case OMDataType::U16:
155 return sizeof(OMDataTypeImpl<OMDataType::U16>::Type);
156 case OMDataType::S32:
157 return sizeof(OMDataTypeImpl<OMDataType::S32>::Type);
158 case OMDataType::U32:
159 return sizeof(OMDataTypeImpl<OMDataType::U32>::Type);
160 case OMDataType::S64:
161 return sizeof(OMDataTypeImpl<OMDataType::S64>::Type);
162 case OMDataType::U64:
163 return sizeof(OMDataTypeImpl<OMDataType::U64>::Type);
164 case OMDataType::FLOAT16:
165 return sizeof(OMDataTypeImpl<OMDataType::FLOAT16>::Type);
166 case OMDataType::FLOAT32:
167 return sizeof(OMDataTypeImpl<OMDataType::FLOAT32>::Type);
168 case OMDataType::FLOAT64:
169 return sizeof(OMDataTypeImpl<OMDataType::FLOAT64>::Type);
170 case OMDataType::BOOL:
171 return sizeof(OMDataTypeImpl<OMDataType::BOOL>::Type);
172 default:
173 // TODO Support remaining data types.
174 assert(false);
175 return 1; // Avoid compiler warning.
176 }
177}
