30uint64_t num_elements(const loco::NodeMixin<loco::NodeTrait::TensorShape> &shape)
31{
32 if (shape.rank() == 0)
33 {
34 return 0;
35 }
36
37 uint64_t res = 1;
38
39 for (uint32_t axis = 0; axis < shape.rank(); ++axis)
40 {
41 assert(shape.dim(axis).known());
42 res *= shape.dim(axis).value();
43 }
44
45 return res;
46}

50bool skip(const loco::Node *node)
51{
52 static std::set<uint32_t> skip_op = {
53 // TODO Current implementation works for 'Tensor' domain only. Support other domains such as
54 // `Feature`, `Filter`, `Bias`, etc.
55 static_cast<uint32_t>(loco::CanonicalOpcode::FilterEncode),
56 static_cast<uint32_t>(loco::CanonicalOpcode::FeatureEncode),
57 static_cast<uint32_t>(loco::CanonicalOpcode::BiasEncode),
58 static_cast<uint32_t>(loco::CanonicalOpcode::DepthwiseFilterEncode),
59
60 // We don't perform constant folding for Push
61 static_cast<uint32_t>(loco::CanonicalOpcode::Push),
62
63 // TensorBroadcast is a good hint for optimization
64 // TODO Let this option be controlled by driver using logo
65 static_cast<uint32_t>(loco::CanonicalOpcode::TensorBroadcast),
66 };
67
68 if (node->dialect() == loco::CanonicalDialect::get())
69 {
70 if (skip_op.find(node->opnum()) != skip_op.end())
71 return true;
72 }
73
74 return false;
75}

78bool foldable(const loco::Node *node)
79{
80 if (node->dialect() == loco::CanonicalDialect::get())
81 {
82 if (skip(node))
83 return false;
84
85 if (node->arity() == 0) // e.g., when a node is e.g, ConstGen or Pull
86 return false;
87
88 // When all args are ConstGen, let's do Constant Folding Transforms
89 for (int i = 0; i < node->arity(); i++)
90 {
91 if (node->arg(i)->opnum() != static_cast<uint32_t>(loco::CanonicalOpcode::ConstGen))
92 return false;
93 }
94
95 return true;
96 }
97 else
98 {
99 return false;
100 }
101}

103void fold(loco::Graph *graph, loco::Node *node)
104{
105 assert(foldable(node)); // sanity check to find a mistake when this function is reused later
106
107 // calcluate foldable node
108 locomotiv::Session sess(graph, std::vector<loco::Node *>{node});
109 sess.infer();
110 auto data = sess.get_output(0);
111
112 assert(data != nullptr);
113
114 auto shape = data->shape();
115 auto dtype = data->dtype();
116
117 // build ConstGen
118 auto new_const = graph->nodes()->create<loco::ConstGen>();
119 {
120 new_const->dtype(dtype);
121
122 new_const->rank(shape->rank());
123 for (int d = 0; d < shape->rank(); d++)
124 new_const->dim(d) = shape->dim(d);
125
126 auto count = num_elements(*new_const);
127
128 if (dtype == loco::DataType::FLOAT32)
129 {
130 new_const->size<loco::DataType::FLOAT32>(count);
131
132 auto const_buf = data->as_f32_bufptr()->base();
133 for (int x = 0; x < count; x++)
134 new_const->at<loco::DataType::FLOAT32>(x) = const_buf[x];
135 }
136 else if (dtype == loco::DataType::S32)
137 {
138 new_const->size<loco::DataType::S32>(count);
139
140 auto const_buf = data->as_s32_bufptr()->base();
141 for (int x = 0; x < count; x++)
142 new_const->at<loco::DataType::S32>(x) = const_buf[x];
143 }
144 }
145
146 // replace node with new_const
147 loco::replace(node).with(new_const);
148}

155bool ConstantFoldingPass::run(loco::Graph *graph)
156{
157 auto outputs = loco::output_nodes(graph);
158
159 bool changed = false;
160 for (auto node : loco::postorder_traversal(outputs))
161 {
162 if (foldable(node))
163 {
164 fold(graph, node);
165 changed = true;
166 }
167 }
168
169 return changed;
170}
