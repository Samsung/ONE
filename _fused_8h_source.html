67template <FusedActivationFunctionType Ac>
68void AddBiasAndEvalActivationFunction(const float *bias_data, const Dims<4> &bias_dims,
69 float *array_data, const Dims<4> &array_dims)
70{
71 const int bias_size = bias_dims.sizes[3] * bias_dims.strides[3];
72 const int array_size = array_dims.sizes[3] * array_dims.strides[3];
73 DCHECK_EQ((array_size % bias_size), 0);
74 for (int array_offset = 0; array_offset < array_size; array_offset += bias_size)
75 {
76 for (int i = 0; i < bias_size; i++)
77 {
78 array_data[array_offset + i] =
79 ActivationFunction<Ac>(array_data[array_offset + i] + bias_data[i]);
80 }
81 }
82}
