61void TrixMetaReader::init(const char *path)
62{
63 assert(path);
64 _meta = getNPUmodel_metadata(path, false);
65 if (_meta == nullptr)
66 {
67 throw std::runtime_error("Failed to get TRIX model metadata");
68 }
69 if (NPUBIN_VERSION(_meta->magiccode) != 3)
70 {
71 throw std::runtime_error("TRIX model metadata version mismatched.");
72 }

143void TrixLoader::loadInputOperands(TrixMetaReader &meta, ir::Graph &subg)
144{
145 for (uint32_t i = 0; i < meta.input_seg_num(); ++i)
146 {
147 // Shape
148 ir::Shape shape;
149 for (uint32_t d = 0; d < MAX_RANK; ++d)
150 shape.append(meta.input_seg_dims(i, d));
151
152 // TypeInfo
153 ir::TypeInfo type_info(toDataType(meta.input_seg_quant_type(i)), meta.input_seg_quant_scale(i),
154 meta.input_seg_quant_zp(i));
155
156 _origin_input_shapes.push_back(shape);
157 // Create operand
158 subg.addOperand(shape, type_info);
159 subg.addInput(ir::OperandIndex(i), "tvn_input" + std::to_string(i));
160 }
161}

163void TrixLoader::loadOutputOperands(TrixMetaReader &meta, ir::Graph &subg)
164{
165 for (uint32_t i = 0; i < meta.output_seg_num(); ++i)
166 {
167 // Shape
168 ir::Shape shape;
169 for (uint32_t d = 0; d < MAX_RANK; ++d)
170 shape.append(meta.output_seg_dims(i, d));
171
172 // TypeInfo
173 ir::TypeInfo type_info(toDataType(meta.output_seg_quant_type(i)),
174 meta.output_seg_quant_scale(i), meta.output_seg_quant_zp(i));
175
176 _origin_output_shapes.push_back(shape);
177 // Create operand
178 subg.addOperand(shape, type_info);
179 subg.addOutput(ir::OperandIndex(subg.getInputs().size() + i), "tvn_out" + std::to_string(i));
180 }
181}

183std::unique_ptr<ir::Graph> TrixLoader::loadSubgraph()
184{
185 auto subg = std::make_unique<ir::Graph>();
186
187 if (_model_path.size() == 1)
188 {
189 // Single model
190 TrixMetaReader meta;
191 meta.init(_model_path.front().c_str());
192 loadInputOperands(meta, *subg);
193 loadOutputOperands(meta, *subg);
194 }
195 else
196 {
197 // Multiple models
198 TrixMetaReader head_model_meta, tail_model_meta;
199 head_model_meta.init(_model_path.front().c_str());
200 tail_model_meta.init(_model_path.back().c_str());
201 loadInputOperands(head_model_meta, *subg);
202 loadOutputOperands(tail_model_meta, *subg);
203 }
204
205 // Create operations
206 loadBulk(*subg);
207
208 subg->verify();
209 return subg;
210}

212void TrixLoader::loadModel(std::unique_ptr<ir::Model> &model)
213{
214 // one subgraph only
215 auto subg = loadSubgraph();
216 model->push(ir::SubgraphIndex(0), std::move(subg));
217}

219std::unique_ptr<ir::Model> TrixLoader::loadFromFile(const std::string &file_path)
220{
221 auto model = std::make_unique<ir::Model>();
222
223 std::stringstream ss{file_path};
224 std::string path;
225 while (std::getline(ss, path, ';'))
226 {
227 // model path will be used to set Bulk param
228 // , and it can be more than one for multiple models
229 _model_path.push_back(path);
230 }
231
232 loadModel(model);
233
234 return model;
235}
