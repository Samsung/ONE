31TfLiteFusedActivation get_tflite_activation(Activation activation)
32{
33 switch (activation)
34 {
35 case FusedActFunc::RELU:
36 return kTfLiteActRelu;
37 case FusedActFunc::RELU6:
38 return kTfLiteActRelu6;
39 case FusedActFunc::RELU_N1_TO_1:
40 return kTfLiteActReluN1To1;
41 case FusedActFunc::TANH:
42 return kTfLiteActTanh;
43 case FusedActFunc::SIGN_BIT:
44 return kTfLiteActSignBit;
45 case FusedActFunc::NONE:
46 return kTfLiteActNone;
47 default:
48 assert(false && "Unsupported activation type");
49 }
50}

66void SVDF::configure()
67{
68 const Shape &input_shape = input()->shape();
69 const Shape &weight_features_shape = weight_feature()->shape();
70 const Shape &weight_time_shape = weight_time()->shape();
71
72 // Validate Input Tensor:
73 LUCI_INTERPRETER_CHECK(input()->element_type() == DataType::FLOAT32 ||
74 input()->element_type() == DataType::S8);
75 LUCI_INTERPRETER_CHECK(input_shape.num_dims() == 2);
76
77 // Validate inputs and output types
78 if (input()->element_type() == DataType::S8)
79 {
80 LUCI_INTERPRETER_CHECK(weight_feature()->element_type() == DataType::S8);
81 LUCI_INTERPRETER_CHECK(weight_time()->element_type() == DataType::S16 ||
82 weight_time()->element_type() == DataType::S8);
83 if (bias())
84 LUCI_INTERPRETER_CHECK(bias()->element_type() == DataType::S32);
85
86 LUCI_INTERPRETER_CHECK(input_activation_state()->element_type() == DataType::S16 ||
87 input_activation_state()->element_type() == DataType::S8);
88 LUCI_INTERPRETER_CHECK(output()->element_type() == DataType::S8);
89
90 // Note: now tflite support only ReLU activation for integer SVDF
91 LUCI_INTERPRETER_CHECK(params().activation == FusedActFunc::RELU);
92 }
93 else if (weight_feature()->element_type() == DataType::FLOAT32)
94 {
95 LUCI_INTERPRETER_CHECK(weight_feature()->element_type() == DataType::FLOAT32);
96 LUCI_INTERPRETER_CHECK(weight_time()->element_type() == DataType::FLOAT32);
97 LUCI_INTERPRETER_CHECK(input_activation_state()->element_type() == DataType::FLOAT32);
98 if (bias())
99 LUCI_INTERPRETER_CHECK(bias()->element_type() == DataType::FLOAT32);
100 LUCI_INTERPRETER_CHECK(output()->element_type() == DataType::FLOAT32);
101 }
102 else if ((weight_feature()->element_type() == DataType::U8 ||
103 weight_feature()->element_type() == DataType::S8) &&
104 input()->element_type() == DataType::FLOAT32)
105 {
106 // TODO:: support hybrid SVDF op
107 assert(false && "Hybrid type is not currently supported");
108 }
109 else
110 {
111 assert(false && "Unsupported type.");
112 }
113
114 // Check all the parameters of tensor match within themselves and match the
115 // input configuration.
116 const int rank = params().svdf_rank;
117 const int batch_size = input_shape.dim(0);
118 const int num_filters = weight_features_shape.dim(0);
119 LUCI_INTERPRETER_CHECK(rank != 0);
120 LUCI_INTERPRETER_CHECK(num_filters % rank == 0);
121
122 const int num_units = num_filters / rank;
123 const int memory_size = weight_time_shape.dim(1);
124
125 // Validate Weight_Feature Input Tensor:
126 LUCI_INTERPRETER_CHECK(weight_features_shape.num_dims() == 2);
127 LUCI_INTERPRETER_CHECK(weight_features_shape.dim(1) == input_shape.dim(1));
128
129 // Validate Weight_Time Input Tensor:
130 LUCI_INTERPRETER_CHECK(weight_time_shape.num_dims() == 2);
131 LUCI_INTERPRETER_CHECK(weight_time_shape.dim(0) == num_filters);
132
133 // Validate Bias
134 if (bias())
135 LUCI_INTERPRETER_CHECK(bias()->shape().dim(0) == num_units);
136
137 // Validate Input Activation State
138 LUCI_INTERPRETER_CHECK(input_activation_state()->shape().num_dims() == 2);
139 LUCI_INTERPRETER_CHECK(input_activation_state()->shape().dim(0) == batch_size);
140 LUCI_INTERPRETER_CHECK(input_activation_state()->shape().dim(1) == memory_size * num_filters);
141
142 // Resize scratchpad_state to input_activation_state
143 auto scratchpad_activation_state = getOutputTensors()[1];
144 scratchpad_activation_state->resize({batch_size, memory_size * num_filters});
145
146 // TODO: enable it only if kernel with dynamic shapes
147 // Resize output tensor
148 output()->resize({batch_size, num_units});
149
150 luci_interpreter_pal::SetupScratchpadTensor(
151 input()->element_type(), weight_feature()->element_type(), getOutputTensors()[2],
152 getOutputTensors()[3], getOutputTensors()[4], getOutputTensors()[5], getOutputTensors()[6],
153 getOutputTensors()[7], input_shape, weight_time_shape, batch_size, num_filters, num_units);
154}

156void SVDF::execute() const
157{
158 switch (weight_feature()->element_type())
159 {
160 case DataType::FLOAT32:
161 evalFloat();
162 break;
163 case DataType::S8:
164 {
165 if (input()->element_type() == DataType::S8)
166 evalInteger();
167 else
168 // TODO:: support hybrid SVDF op
169 assert(false && "Hybrid type is not currently supported");
170 break;
171 }
172 default:
173 assert(false && "Unsupported type");
174 }
175}

177void SVDF::evalInteger() const
178{
179 const auto effective_scale_1 = static_cast<double>(input()->scale() * weight_feature()->scale() /
180 input_activation_state()->scale());
181 const auto effective_scale_2 = static_cast<double>(input_activation_state()->scale() *
182 weight_time()->scale() / output()->scale());
183
184 int32_t effective_scale_1_a;
185 int effective_scale_1_b;
186 int32_t effective_scale_2_a;
187 int effective_scale_2_b;
188
189 tflite::QuantizeMultiplier(effective_scale_1, &effective_scale_1_a, &effective_scale_1_b);
190 tflite::QuantizeMultiplier(effective_scale_2, &effective_scale_2_a, &effective_scale_2_b);
191
192 TfLiteSVDFParams params_svdf{};
193 params_svdf.asymmetric_quantize_inputs = params().asymmetric_quantize_inputs;
194 params_svdf.rank = params().svdf_rank;
195 params_svdf.activation = get_tflite_activation(params().activation);
196
197 auto scratchpad_activation_state = getOutputTensors()[1];
198 // Note: it is expected that activation_state input variable tensor reset to zero,
199 // also expected that this variable tensor doesn't have buffer
200 auto scratchpad_data = getTensorData<int16_t>(scratchpad_activation_state);
201 std::fill_n(scratchpad_data, scratchpad_activation_state->shape().num_elements(), 0);
202
203 auto scratchpad = getOutputTensors()[2];
204 auto output_temp = getOutputTensors()[3];
205
206 int32_t input_zp = input()->zero_point();
207 int32_t output_zp = output()->zero_point();
208 luci_interpreter_pal::IntegerSVDF(
209 params_svdf, getTensorShape(input()), getTensorData<int8_t>(input()),
210 getTensorShape(weight_feature()), getTensorData<int8_t>(weight_feature()),
211 getTensorShape(weight_time()), getTensorData<int16_t>(weight_time()), getTensorShape(bias()),
212 getTensorData<int32_t>(bias()), scratchpad_data, getTensorShape(output()),
213 getTensorData<int8_t>(output()), getTensorData<int32_t>(scratchpad),
214 getTensorData<int32_t>(output_temp), effective_scale_1_a, effective_scale_1_b,
215 effective_scale_2_a, effective_scale_2_b, input_zp, output_zp);
216}

222 params_svdf.rank = params().svdf_rank;

235 getTensorShape(weight_feature()), getTensorData<float>(weight_feature()),
236 getTensorShape(weight_time()), getTensorData<float>(weight_time()), getTensorShape(bias()),
237 getTensorData<float>(bias()), getTensorData<float>(scratchpad_1), scratchpad_data,
238 getTensorShape(output()), getTensorData<float>(output()));
239}
