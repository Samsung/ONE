30void KernelGenerator::visit(const ir::operation::OneHot &node)
31{
32 const auto output_index{node.getOutputs().at(0)};
33 const auto indices_index{node.getInputs().at(ir::operation::OneHot::INDICES)};
34 const auto depth_index{node.getInputs().at(ir::operation::OneHot::Input::DEPTH)};
35 const auto onvalue_index{node.getInputs().at(ir::operation::OneHot::Input::ON_VALUE)};
36 const auto offvalue_index{node.getInputs().at(ir::operation::OneHot::Input::OFF_VALUE)};
37
38 const auto axis = node.param().axis;
39
40 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
41 auto indices_tensor = _tensor_reg->getPortableTensor(indices_index);
42 auto depth_tensor = _tensor_reg->getPortableTensor(depth_index);
43 auto onvalue_tensor = _tensor_reg->getPortableTensor(onvalue_index);
44 auto offvalue_tensor = _tensor_reg->getPortableTensor(offvalue_index);
45
46 assert(indices_tensor->data_type() == OperandType::INT32);
47 assert(axis <= static_cast<int>(indices_tensor->getShape().rank()));
48
49 auto fn = std::make_unique<ops::OneHotLayer>();
50
51 fn->configure(indices_tensor, depth_tensor, onvalue_tensor, offvalue_tensor, output_tensor, axis);
52
53 _return_fn = std::move(fn);
54}
