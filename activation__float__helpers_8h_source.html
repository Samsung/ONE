49// Hard-Swish
50#define hard_swish_op(DATA_TYPE, x, A_VAL, B_VAL) \
51 (x * ((min(max((x + (DATA_TYPE)3.0), (DATA_TYPE)0.0), (DATA_TYPE)6.0)) * (DATA_TYPE)0.166666667))
52
53// Logistic Activation
54#define logistic_op(DATA_TYPE, x, A_VAL, B_VAL) ((DATA_TYPE)1.0 / ((DATA_TYPE)1.0 + exp(-x)))
55
56// Hyperbolic Tangent Activation
57#define tanh_op(DATA_TYPE, x, A_VAL, B_VAL) ((DATA_TYPE)A_VAL * tanh((DATA_TYPE)B_VAL * x))
58
59// RELU Tangent Activation
60#define relu_op(DATA_TYPE, x, A_VAL, B_VAL) (max((DATA_TYPE)0.0, x))
61
62// Bounded RELU Activation
63#define brelu_op(DATA_TYPE, x, A_VAL, B_VAL) (min((DATA_TYPE)A_VAL, max((DATA_TYPE)0.0, x)))
64
65// Lower Upper Bounded RELU Activation
66#define lu_brelu_op(DATA_TYPE, x, A_VAL, B_VAL) (min(max(x, (DATA_TYPE)B_VAL), (DATA_TYPE)A_VAL))
67
68// Leaky RELU Activation
69#define lrelu_op(DATA_TYPE, x, A_VAL, B_VAL) \
70 ((min(x, (DATA_TYPE)0.0) * (DATA_TYPE)A_VAL) + max(x, (DATA_TYPE)0.0))
71
72// Soft RELU Activation
73#define srelu_op(DATA_TYPE, x, A_VAL, B_VAL) (log((DATA_TYPE)1.0 + exp(x)))
74
75// ELU Activation
76#define elu_op(DATA_TYPE, x, A_VAL, B_VAL) \
77 (select(((DATA_TYPE)A_VAL * (exp(x) - (DATA_TYPE)1.0)), x, isgreaterequal(x, (DATA_TYPE)0.0)))
78
79// Absolute Activation
80#define abs_op(DATA_TYPE, x, A_VAL, B_VAL) (fabs(x))
81
82// Square Activation
83#define square_op(DATA_TYPE, x, A_VAL, B_VAL) (x * x)
84
85// Square-root Activation
86#define sqrt_op(DATA_TYPE, x, A_VAL, B_VAL) (sqrt(x))
87
88// Linear Activation
89#define linear_op(DATA_TYPE, x, A_VAL, B_VAL) (MLA((DATA_TYPE)B_VAL, (DATA_TYPE)A_VAL, x))
90
91// Identity Activation
92#define identity_op(DATA_TYPE, x, A_VAL, B_VAL) (x)
93
94#define ACT_OP(op, DATA_TYPE, x, A_VAL, B_VAL) op##_op(DATA_TYPE, x, A_VAL, B_VAL)
95
96#define ACTIVATION(op, DATA_TYPE, x, A_VAL, B_VAL) ACT_OP(op, DATA_TYPE, x, A_VAL, B_VAL)
