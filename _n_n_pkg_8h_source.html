46struct ModelEdgeHash
47{
48 size_t operator()(const ::onert::ir::ModelEdge &edge) const noexcept
49 {
50 const auto &[from_model_index, from_sug_index, from_io_index] = edge.from;
51 unsigned long long h1 =
52 (from_model_index.value() << 24) | (from_sug_index.value() << 16) | from_io_index.value();
53
54 const auto &[to_model_index, to_sug_index, to_io_index] = edge.to;
55 unsigned long long h2 =
56 (to_model_index.value() << 24) | (to_sug_index.value() << 16) | to_io_index.value();
57 return h1 + h2;
58 }
59};

194 void verify(void)
195 {
196 // Verify edges information
197 //
198 // Only duplicates of nnpkg output and Edge `from` are possible.
199 // | Whether duplicates are possible | Edge `to` | Edge `from` |
200 // | nnpkg input (input of subgraph) | X (*1) | X (*2) |
201 // | nnpkg output (output of subgraph) | X (*2) | O |
202 // *1. The subjects who determine values of each buffer are different.
203 // - nnpkg input : user input
204 // - Edge `to` : output of another subgraph
205 // *2. `IOIndex` of inputs and outputs of subgraph is distinct.
206 //
207 for (const auto &edge : _edges.edges)
208 {
209 if (std::find(_edges.pkg_inputs.begin(), _edges.pkg_inputs.end(), edge.to) !=
210 _edges.pkg_inputs.end())
211 {
212 throw std::runtime_error{
213 "Invalid edge information. NNPkg inputs and Edge `to` cannot be duplicated"};
214 }
215 }
216 }
