46struct ModelEdgeHash
47{
48 size_t operator()(const ::onert::ir::ModelEdge &edge) const noexcept
49 {
50 unsigned long long h1 = (std::get<0>(edge.from).value() << 24) |
51 (std::get<1>(edge.from).value() << 16) | std::get<2>(edge.from).value();
52 unsigned long long h2 = (std::get<0>(edge.to).value() << 24) |
53 (std::get<1>(edge.to).value() << 16) | std::get<2>(edge.to).value();
54 return h1 + h2;
55 }
56};

185 void verify(void)
186 {
187 // Verify edges information
188 //
189 // Only duplicates of nnpkg output and Edge `from` are possible.
190 // | Whether duplicates are possible | Edge `to` | Edge `from` |
191 // | nnpkg input (input of subgraph) | X (*1) | X (*2) |
192 // | nnpkg output (output of subgraph) | X (*2) | O |
193 // *1. The subjects who determine values of each buffer are different.
194 // - nnpkg input : user input
195 // - Edge `to` : output of another subgraph
196 // *2. `IOIndex` of inputs and outputs of subgraph is distinct.
197 //
198 for (const auto &edge : _edges.edges)
199 {
200 if (std::find(_edges.pkg_inputs.begin(), _edges.pkg_inputs.end(), edge.to) !=
201 _edges.pkg_inputs.end())
202 {
203 throw std::runtime_error{
204 "Invalid edge information. NNPkg inputs and Edge `to` cannot be duplicated"};
205 }
206 }
207 }
