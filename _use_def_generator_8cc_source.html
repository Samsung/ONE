35UseDefGenerator::UseDefGenerator(const TrainableGraph &tgraph)
36 : _tgraph{tgraph}, _node_to_idx{}, _training_usedefs{}
37{
38 const auto order = _tgraph.topolSortOperations();
39 for (const auto &index : order)
40 {
41 const auto &node = _tgraph.operation(index);
42 assert(_node_to_idx.find(&node) == _node_to_idx.end());
43 _node_to_idx[&node] = index;
44 }
45
46 // Check whether loss exists
47 assert(std::any_of(order.begin(), order.end(),
48 [&](const auto &index) {
49 return _tgraph.operation(index).opcode() == ir::OpCode::Loss;
50 }) &&
51 "Loss does not exist");
52}

54UseDefChains UseDefGenerator::operator()()
55{
56 const auto &graph = _tgraph.graph();
57 assert(ir::verifier::EdgeChecker().verify(graph));
58
59 _training_usedefs.clear();
60 graph.operands().iterate([&](const ir::OperandIndex &idx, const ir::Operand &operand) {
61 // Initialize as emtpy UseDefChain
62 const auto empty_usedef_chain = UseDefChain{operand};
63 _training_usedefs.emplace(TrainingOperandIndex{idx, true}, empty_usedef_chain);
64 _training_usedefs.emplace(TrainingOperandIndex{idx, false}, empty_usedef_chain);
65 });
66
67 initForForwardingNodes();
68
69 initForBackwardingNodes();
70
71 return _training_usedefs;
72}

74void UseDefGenerator::visit(const train::operation::Conv2D &node)
75{
76 assert(_node_to_idx.find(&node) != _node_to_idx.end());
77 const auto &op_index = _node_to_idx.at(&node);
78 const auto backwarding_op_index = TrainingOperationIndex{op_index, false};
79
80 // Insert use of forwarding inputs
81 const auto &in_index = node.getInputs().at(train::operation::Conv2D::Input::INPUT);
82 const auto in_forwarding_index = TrainingOperandIndex{in_index, true};
83 insertUse(in_forwarding_index, backwarding_op_index);
84
85 const auto &weights_index = node.getInputs().at(train::operation::Conv2D::Input::KERNEL);
86 const auto weights_forwarding_index = TrainingOperandIndex{weights_index, true};
87 insertUse(weights_forwarding_index, backwarding_op_index);
88
89 // Insert use of forwarding output
90 if (node.param().activation != ir::Activation::NONE)
91 {
92 const auto &out_index = node.getOutputs().at(0);
93 const auto out_forwarding_index = TrainingOperandIndex{out_index, true};
94 insertUse(out_forwarding_index, backwarding_op_index);
95 }
96
97 // Set def of backwarding inputs
98 const auto outgoing_index = TrainingOperandIndex{in_index, false};
99 insertBackPropDef(outgoing_index, backwarding_op_index);
100
101 const auto weights_gradient_index = TrainingOperandIndex{weights_index, false};
102 insertDef(weights_gradient_index, backwarding_op_index);
103
104 const auto &bias_index = node.getInputs().at(ir::operation::Conv2D::Input::BIAS);
105 if (bias_index.valid())
106 {
107 const auto bias_gradient_index = TrainingOperandIndex{bias_index, false};
108 insertDef(bias_gradient_index, backwarding_op_index);
109 }
110}

112void UseDefGenerator::visit(const train::operation::Loss &node)
113{
114 assert(_node_to_idx.find(&node) != _node_to_idx.end());
115 const auto &op_index = _node_to_idx.at(&node);
116 const auto backwarding_op_index = TrainingOperationIndex{op_index, false};
117
118 for (const auto &in_index : node.getInputs() | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED)
119 {
120 // Insert use of forwarding inputs
121 const auto in_forwarding_index = TrainingOperandIndex{in_index, true};
122 insertUse(in_forwarding_index, backwarding_op_index);
123 }
124
125 // Set def of backwarding(backprop) y_pred
126 const auto &y_pred_index = node.getInputs().at(train::operation::Loss::Input::Y_PRED);
127 assert(!_tgraph.operands().at(y_pred_index).isConstant());
128 const auto y_pred_outgoing_index = TrainingOperandIndex{y_pred_index, false};
129 insertBackPropDef(y_pred_outgoing_index, backwarding_op_index);
130
131 // Set def of backwarding(backprop) y_true
132 const auto &y_true_index = node.getInputs().at(train::operation::Loss::Input::Y_TRUE);
133 assert(!_tgraph.operands().at(y_true_index).isConstant());
134 const auto y_true_outgoing_index = TrainingOperandIndex{y_true_index, false};
135 insertBackPropDef(y_true_outgoing_index, backwarding_op_index);
136
137 // Remove use of backwarding output
138 const auto &out_index = node.getOutputs().at(0);
139 const auto incoming_index = TrainingOperandIndex{out_index, false};
140 auto &usedef_chain = _training_usedefs.at(incoming_index);
141 usedef_chain.removeTrainingUse(backwarding_op_index);
142}

144void UseDefGenerator::visit(const train::operation::Reshape &node)
145{
146 assert(_node_to_idx.find(&node) != _node_to_idx.end());
147 const auto &op_index = _node_to_idx.at(&node);
148 const auto backwarding_op_index = TrainingOperationIndex{op_index, false};
149
150 // Insert use of backwarding(backprop) output
151 const auto &out_index = node.getOutputs().at(0);
152 const auto incoming_index = TrainingOperandIndex{out_index, false};
153 insertUse(incoming_index, backwarding_op_index);
154
155 // Set def of backwarding(backprop) input
156 const auto &in_index = node.getInputs().at(train::operation::Reduce::Input::INPUT);
157 const auto outgoing_index = TrainingOperandIndex{in_index, false};
158 insertBackPropDef(outgoing_index, backwarding_op_index);
159}

179void UseDefGenerator::insertBackPropDef(const TrainingOperandIndex &operand_index,
180 const TrainingOperationIndex &op_index)
181{
182 // NOTE There is no need to set def of constant backwarding(backprop) inputs
183 // because it won't be back-propagated.
184 if (!_tgraph.operands().at(operand_index.index()).isConstant())
185 {
186 insertDef(operand_index, op_index);

190void UseDefGenerator::initForForwardingNodes()
191{
192 // Initialize training def-uses of forwarding operands for only forwarding nodes
193 // (i.e. forwarding nodes that do not have any backwarding node)
194 _tgraph.operands().iterate([&](const ir::OperandIndex &idx, const ir::Operand &operand) {
195 // Append forwarding def-uses as it is
196 const bool is_forward = true;
197 const auto forwarding_operand_index = TrainingOperandIndex{idx, is_forward};
198
199 const auto def = operand.getDef();
200 if (def.valid())
201 {
202 insertDef(forwarding_operand_index, TrainingOperationIndex{def, is_forward});
203 auto &usedef_chain = _training_usedefs.at(forwarding_operand_index);
204 usedef_chain.insertTrainingDef(TrainingOperationIndex{def, is_forward});
205 }
206
207 assert(_training_usedefs.at(forwarding_operand_index).getTrainingUses().size() == 0);
208 const auto uses = operand.getUses();
209 for (const auto &use : uses)
210 insertUse(forwarding_operand_index, TrainingOperationIndex{use, is_forward});
211 });
212}

214void UseDefGenerator::initForBackwardingNodes()
215{
216 const auto backward_order = _tgraph.essentialBackwardOrder();
217 // Initialize training uses of forwarding operands and def-uses of backwarding operands for
218 // backwarding nodes (i.e. backwarding nodes that do not have any forwarding node)
219 for (const auto &op_index : backward_order)
220 {
221 const auto &node = _tgraph.operation(op_index);
222
223 // Insert use of backwarding operands(only output)
224 {
225 if (node.getOutputs().size() > 1)
226 throw std::runtime_error(
227 "UseDefGenerator does not support multiple outputs of training operation");
228
229 const auto &output = node.getOutputs().at(0);
230 const auto backwarding_op_index = TrainingOperationIndex{op_index, false};
231 const auto incoming_index = TrainingOperandIndex{output, false};
232 insertUse(incoming_index, backwarding_op_index);
233 }
234
235 // Insert uses of forwarding operands and insert defs of backwarding operands
236 node.accept(*this);
237 }
