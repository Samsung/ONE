41void PRelu::configure()
42{
43 LUCI_INTERPRETER_CHECK(input()->element_type() == output()->element_type());
44 LUCI_INTERPRETER_CHECK(alpha()->element_type() == output()->element_type());
45 LUCI_INTERPRETER_CHECK(input()->scales().size() <= 1);
46 LUCI_INTERPRETER_CHECK(output()->scales().size() <= 1);
47
48 if (input()->element_type() == DataType::U8)
49 {
50 LUCI_INTERPRETER_CHECK(alpha()->scales().size() <= 1); // remove when CWQ kernel arrives
51 _alpha_multipliers.resize(1);
52 double alpha_multiplier = input()->scale() * alpha()->scale() / output()->scale();
53 quantizeMultiplier(alpha_multiplier, &_alpha_multipliers[0].multiplier,
54 &_alpha_multipliers[0].shift);
55 double identity_multiplier = input()->scale() / output()->scale();
56 quantizeMultiplier(identity_multiplier, &_output_multiplier_identity, &_output_shift_identity);
57 }
58 else if (input()->element_type() == DataType::S16)
59 {
60 // Common check for correctness of quant params
61 LUCI_INTERPRETER_CHECK(input()->zero_point() == 0 && output()->zero_point() == 0);
62 for (size_t channel = 0; channel < alpha()->zero_points().size(); ++channel)
63 {
64 LUCI_INTERPRETER_CHECK(alpha()->zero_points()[channel] == 0);
65 }
66 // PRelu specific checks for CWQ
67 LUCI_INTERPRETER_CHECK(alpha()->quantized_dimension() == alpha()->shape().num_dims() - 1);
68 LUCI_INTERPRETER_CHECK(static_cast<int32_t>(alpha()->scales().size()) ==
69 alpha()->shape().dim(alpha()->quantized_dimension()));
70 LUCI_INTERPRETER_CHECK(alpha()->shape().num_elements() ==
71 input()->shape().dim(input()->shape().num_dims() - 1));
72
73 // all dimension of alpha except last one should be size 1
74 for (int dim = 0; dim < alpha()->shape().num_dims() - 1; ++dim)
75 {
76 LUCI_INTERPRETER_CHECK(alpha()->shape().dim(dim) == 1);
77 }
78
79 std::vector<double> real_multipliers =
80 getQuantizedConvolutionMultiplers(input()->scale(), alpha()->scales(), output()->scale());
81
82 _alpha_multipliers = quantizeMultipliers(real_multipliers);
83
84 double identity_multiplier = input()->scale() / output()->scale();
85 quantizeMultiplier(identity_multiplier, &_output_multiplier_identity, &_output_shift_identity);
86 }
87 // TODO: enable it only if kernel with dynamic shapes
88 output()->resize(calculateShapeForBroadcast(input()->shape(), alpha()->shape()));
89}

91void PRelu::execute() const
92{
93 switch (input()->element_type())
94 {
95 case DataType::FLOAT32:
96 evalFloat();
97 break;
98 case DataType::U8:
99 evalQuantized();
100 break;
101 case DataType::S16:
102 evalQuantizedS16();
103 break;
104 default:
105 assert(false && "Unsupported type.");
106 }
107}

109void PRelu::evalFloat() const
110{
111 const auto input_data = getTensorData<float>(input());
112 const auto alpha_data = getTensorData<float>(alpha());
113 const auto size = getTensorShape(input()).FlatSize();
114 auto output_data = getTensorData<float>(output());
115
116 auto PReluFunc = [](float input, float alpha) { return input >= 0.0 ? input : input * alpha; };
117
118 if (input()->shape() != alpha()->shape())
119 {
120 tflite::reference_ops::BroadcastBinaryFunction4DSlow<float, float, float>(
121 getTensorShape(input()), getTensorData<float>(input()), getTensorShape(alpha()),
122 getTensorData<float>(alpha()), getTensorShape(output()), getTensorData<float>(output()),
123 PReluFunc);
124 }
125 else
126 {
127 for (auto i = decltype(size){0}; i < size; ++i)
128 {
129 if (input_data[i] >= 0)
130 output_data[i] = input_data[i];
131 else
132 output_data[i] = input_data[i] * alpha_data[i];
133 }
134 }
135}

137void PRelu::evalQuantized() const
138{
139 tflite::PreluParams op_params{};
140
141 op_params.input_offset = -input()->zero_point(); // Note the '-'.
142 op_params.alpha_offset = -alpha()->zero_point(); // Note the '-'.
143 op_params.output_offset = output()->zero_point();
144 op_params.output_shift_1 = _output_shift_identity;
145 op_params.output_multiplier_1 = _output_multiplier_identity;
146 op_params.output_shift_2 = _alpha_multipliers[0].shift;
147 op_params.output_multiplier_2 = _alpha_multipliers[0].multiplier;
148
149 if (input()->shape() != alpha()->shape())
150 {
151 tflite::reference_ops::BroadcastPrelu4DSlow(
152 op_params, getTensorShape(input()), getTensorData<uint8_t>(input()), getTensorShape(alpha()),
153 getTensorData<uint8_t>(alpha()), getTensorShape(output()), getTensorData<uint8_t>(output()));
154 }
155 else
156 {
157 tflite::reference_ops::Prelu<uint8_t>(
158 op_params, getTensorShape(input()), getTensorData<uint8_t>(input()), getTensorShape(alpha()),
159 getTensorData<uint8_t>(alpha()), getTensorShape(output()), getTensorData<uint8_t>(output()));
160 }
161}

163static inline int16_t evalElemS16PRelu(int16_t input_val, int16_t alpha_val,
164 const ChannelQuantMultipliers &identity_mult,
165 const ChannelQuantMultipliers &alpha_mult)
166{
167 constexpr int32_t quantized_min = std::numeric_limits<int16_t>::min();
168 constexpr int32_t quantized_max = std::numeric_limits<int16_t>::max();
169
170 const int32_t output_val =
171 input_val >= 0
172 ? tflite::MultiplyByQuantizedMultiplier(static_cast<int32_t>(input_val),
173 identity_mult.multiplier, identity_mult.shift)
174 : tflite::MultiplyByQuantizedMultiplier(static_cast<int32_t>(input_val * alpha_val),
175 alpha_mult.multiplier, alpha_mult.shift);
176 const int32_t clamped_output = std::min(quantized_max, std::max(quantized_min, output_val));
177 return clamped_output;
178}

180void PRelu::evalQuantizedS16() const
181{
182 // Note that this kernel assumes alpha is CWQ
183 tflite::RuntimeShape input_shape = getTensorShape(input());
184 const int16_t *input_data = input()->data<int16_t>();
185 const int16_t *alpha_data = alpha()->data<int16_t>();
186 int16_t *output_data = output()->data<int16_t>();
187
188 const ChannelQuantMultipliers pos_mult{_output_shift_identity, _output_multiplier_identity};
189
190 const int last_dim = input()->shape().num_dims() - 1;
191
192 int32_t outer_dims_size = 1;
193 for (int i = 0; i < last_dim; ++i)
194 outer_dims_size *= input_shape.Dims(i);
195 int32_t quant_dim_size = input_shape.Dims(last_dim);
196
197 for (int32_t outer_dims = 0; outer_dims < outer_dims_size; ++outer_dims)
198 for (int32_t quant_channel = 0; quant_channel < quant_dim_size; ++quant_channel)
199 {
200 const ChannelQuantMultipliers &neg_mult = _alpha_multipliers[quant_channel];
201 size_t offset = static_cast<size_t>(outer_dims) * static_cast<size_t>(quant_dim_size);
202 offset += quant_channel;
203
204 output_data[offset] =
205 evalElemS16PRelu(input_data[offset], alpha_data[quant_channel], pos_mult, neg_mult);
206 }
207}
