167std::ostream &operator<<(std::ostream &os, const NodeDesc &d)
168{
169 assert(d.state() != NodeDesc::State::Invalid);
170
171 std::vector<std::string> values;
172
173 for (uint32_t n = 0; n < d.args().count(); ++n)
174 {
175 values.emplace_back(d.args().at(n).first + ": " + d.args().at(n).second);
176 }
177
178 if (d.state() == NodeDesc::State::PartiallyKnown)
179 {
180 values.emplace_back("...");
181 }
182
183 os << d.opname();
184 os << "(";
185 if (values.size() > 0)
186 {
187 os << values.at(0);
188 for (uint32_t n = 1; n < values.size(); ++n)
189 {
190 os << ", " << values.at(n);
191 }
192 }
193 os << ")";
194
195 return os;
196}

214void FormattedGraphImpl<Formatter::LinearV1>::dump(std::ostream &os) const
215{
216 struct SymbolTableImpl final : public SymbolTable
217 {
218 std::string lookup(const loco::Node *node) const final
219 {
220 if (node == nullptr)
221 {
222 return "(null)";
223 }
224
225 return _content.at(node);
226 }
227
228 std::map<const loco::Node *, std::string> _content;
229 };
230
231 SymbolTableImpl symbols;
232
233 auto symbol = [&symbols](const loco::Node *node) { return symbols.lookup(node); };
234
235 for (uint32_t n = 0; n < _graph->nodes()->size(); ++n)
236 {
237 symbols._content[_graph->nodes()->at(n)] = pp::fmt("%", n);
238 }
239
240 // Find the disjoint node clusters
241 //
242 // TODO Move this implementation into loco Algorithms.h
243 std::map<loco::Node *, loco::Node *> parents;
244
245 for (auto node : loco::all_nodes(_graph))
246 {
247 parents[node] = nullptr;
248 }
249
250 for (auto node : loco::all_nodes(_graph))
251 {
252 for (uint32_t n = 0; n < node->arity(); ++n)
253 {
254 if (auto arg = node->arg(n))
255 {
256 parents[arg] = node;
257 }
258 }
259 }
260
261 auto find = [&parents](loco::Node *node) {
262 loco::Node *cur = node;
263
264 while (parents.at(cur) != nullptr)
265 {
266 cur = parents.at(cur);
267 }
268
269 return cur;
270 };
271
272 std::set<loco::Node *> roots;
273
274 for (auto node : loco::all_nodes(_graph))
275 {
276 roots.insert(find(node));
277 }
278
279 std::map<loco::Node *, std::set<loco::Node *>> clusters;
280
281 // Create clusters
282 for (auto root : roots)
283 {
284 clusters[root] = std::set<loco::Node *>{};
285 }
286
287 for (auto node : loco::all_nodes(_graph))
288 {
289 clusters.at(find(node)).insert(node);
290 }
291
292 std::unique_ptr<locop::NodeSummaryBuilder> node_summary_builder;
293
294 if (_factory)
295 {
296 // Use User-defined NodeSummaryBuilder if NodeSummaryBuilderFactory is present
297 node_summary_builder = _factory->create(&symbols);
298 }
299 else
300 {
301 // Use Built-in NodeSummaryBuilder otherwise
302 node_summary_builder = std::make_unique<GenericNodeSummaryBuilder>(&symbols);
303 }
304
305 // Print Graph Input(s)
306 for (uint32_t n = 0; n < _graph->inputs()->size(); ++n)
307 {
308 auto input = _graph->inputs()->at(n);
309
310 std::string name = input->name();
311
312 std::string shape = "?";
313 if (input->shape() != nullptr)
314 {
315 shape = pp::fmt(formatted_tensor_shape(input->shape()));
316 }
317
318 // TODO Print dtype
319 os << pp::fmt("In #", n, " { name: ", name, ", shape: ", shape, " }") << std::endl;
320 }
321
322 // Print Graph Output(s)
323 for (uint32_t n = 0; n < _graph->outputs()->size(); ++n)
324 {
325 auto output = _graph->outputs()->at(n);
326
327 std::string name = output->name();
328
329 std::string shape = "?";
330 if (output->shape() != nullptr)
331 {
332 shape = pp::fmt(formatted_tensor_shape(output->shape()));
333 }
334
335 // TODO Print dtype
336 os << pp::fmt("Out #", n, " { name: ", name, ", shape: ", shape, " }") << std::endl;
337 }
338
339 if (_graph->inputs()->size() + _graph->outputs()->size() != 0)
340 {
341 os << std::endl;
342 }
343
344 for (auto it = clusters.begin(); it != clusters.end(); ++it)
345 {
346 std::vector<loco::Node *> cluster_outputs;
347
348 for (auto node : it->second)
349 {
350 // NOTE This is inefficient but anyway working :)
351 if (loco::succs(node).empty())
352 {
353 cluster_outputs.emplace_back(node);
354 }
355 }
356
357 for (auto node : loco::postorder_traversal(cluster_outputs))
358 {
359 locop::NodeSummary node_summary;
360
361 // Build a node summary
362 if (!node_summary_builder->build(node, node_summary))
363 {
364 throw std::runtime_error{"Fail to build a node summary"};
365 }
366
367 for (uint32_t n = 0; n < node_summary.comments().count(); ++n)
368 {
369 os << "; " << node_summary.comments().at(n) << std::endl;
370 }
371
372 os << symbol(node);
373
374 if (loco::shape_known(node))
375 {
376 auto node_shape = loco::shape_get(node);
377 os << " : " << str(node_shape.domain());
378 os << "<";
379 os << str(node_shape);
380 os << ", ";
381 // Show DataType
382 os << (loco::dtype_known(node) ? str(loco::dtype_get(node)) : std::string{"?"});
383 os << ">";
384 }
385
386 os << " = " << node_summary << std::endl;
387 }
388 os << std::endl;
389 }
390}
