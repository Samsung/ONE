32// struct to carry Input/Weights/Bias
33struct IWB
34{
35 CircleNode *input = nullptr;
36 CircleNode *weights = nullptr;
37 CircleConst *bias = nullptr;
38
39 IWB(loco::Node *i, loco::Node *w, loco::Node *b)
40 {
41 input = dynamic_cast<luci::CircleNode *>(i);
42 weights = dynamic_cast<luci::CircleNode *>(w);
43 bias = dynamic_cast<luci::CircleConst *>(b);
44 }
45
46 // Return true if bias can be quantized with valid input an weights
47 operator bool()
48 {
49 if (bias == nullptr || is_quantized(bias))
50 return false;
51 if (input == nullptr || weights == nullptr)
52 return false;
53 return true;
54 }
55};

57// Create a new const node from an existing node.
58// The new node has the following characteristics
59// type: T
60// shape: same with 'node' (given as an argument)
61// buffer size: 'size' (given as an argument)
62// Note that contents are not filled in this function.
63template <loco::DataType T>
64luci::CircleConst *create_empty_const_from(luci::CircleConst *node, uint32_t size)
65{
66 auto new_node = node->graph()->nodes()->create<CircleConst>();
67 // TODO: We don't have any naming convention for quantized nodes yet.
68 // Fix this when we have one.
69 new_node->name(node->name());
70 new_node->dtype(T);
71 new_node->rank(node->rank());
72 for (uint32_t i = 0; i < node->rank(); i++)
73 new_node->dim(i).set(node->dim(i).value());
74
75 new_node->size<T>(size);
76 new_node->shape_status(luci::ShapeStatus::VALID);
77
78 return new_node;
79}

81CircleConst *asym_quant_bias_per_layer(CircleConst *node, float input_scale, float weight_scale,
82 float *scaling_factor, int64_t *zp)
83{
84 float scale = input_scale * weight_scale;
85 const float scaling_factor_inv = (scale == 0) ? 0 : 1.0 / scale;
86
87 uint32_t size = node->size<loco::DataType::FLOAT32>();
88 std::vector<int32_t> quantized_values(size);
89 for (uint32_t i = 0; i < size; ++i)
90 {
91 quantized_values[i] =
92 static_cast<int32_t>(std::round(node->at<loco::DataType::FLOAT32>(i) * scaling_factor_inv));
93 }
94
95 auto new_bias = create_empty_const_from<loco::DataType::S32>(node, size);
96
97 const int32_t kMinScale = std::numeric_limits<int32_t>::lowest();
98 const int32_t kMaxScale = std::numeric_limits<int32_t>::max();
99 for (uint32_t i = 0; i < size; ++i)
100 {
101 new_bias->at<loco::DataType::S32>(i) =
102 std::min(kMaxScale, std::max(kMinScale, quantized_values[i]));
103 }
104 *scaling_factor = scale;
105 *zp = 0;
106
107 return new_bias;
108}

110CircleConst *quant_bias_per_channel(CircleConst *node, float input_scale,
111 std::vector<float> &weight_scale,
112 std::vector<float> &scaling_factor, std::vector<int64_t> &zp)
113{
114 float scaling_factor_inv{0};
115
116 uint32_t size = node->size<loco::DataType::FLOAT32>();
117 std::vector<int32_t> quantized_values(size);
118
119 for (uint32_t i = 0; i < size; ++i)
120 {
121 scaling_factor[i] = input_scale * weight_scale[i];
122 scaling_factor_inv = (scaling_factor[i] == 0) ? 0 : 1.0 / scaling_factor[i];
123 quantized_values[i] =
124 static_cast<int32_t>(std::round(node->at<loco::DataType::FLOAT32>(i) * scaling_factor_inv));
125 zp[i] = 0;
126 }
127
128 auto new_bias = create_empty_const_from<loco::DataType::S32>(node, size);
129
130 const int32_t kMinScale = std::numeric_limits<int32_t>::lowest();
131 const int32_t kMaxScale = std::numeric_limits<int32_t>::max();
132 for (uint32_t i = 0; i < size; ++i)
133 {
134 new_bias->at<loco::DataType::S32>(i) =
135 std::min(kMaxScale, std::max(kMinScale, quantized_values[i]));
136 }
137
138 return new_bias;
139}

143 std::vector<float> &scaling_factor,

175// Return a quantized bias node
176CircleConst *QuantizeBias::quantized_bias(CircleNode *input, const CircleNode *weight,
177 CircleNode *bias)
178{
179 auto const_bias = loco::must_cast<luci::CircleConst *>(bias);
180 assert(const_bias->dtype() == loco::DataType::FLOAT32);
181
182 // If input is const, it is quantized here, not in QuantizeActivation
183 if (auto const_input = dynamic_cast<luci::CircleConst *>(input))
184 {
185 quant_const(const_input, output_type);
186 }
187
188 CircleConst *new_bias = nullptr;
189
190 if (granularity == QuantizationGranularity::ChannelWise)
191 {
192 auto input_q = input->quantparam();
193 assert(input_q);
194 assert(input_q->scale.size() == 1); // input scale's layer-wise
195 auto input_scale = input_q->scale[0];
196
197 assert(weight->quantparam() != nullptr); // weight scale's channel-wise
198 auto weight_scale = weight->quantparam()->scale;
199
200 uint32_t size = const_bias->size<loco::DataType::FLOAT32>();
201 assert(size == weight_scale.size());
202 std::vector<float> scaling_factor(size);
203 std::vector<int64_t> zp(size);
204
205 if (const_bias->rank() == 0)
206 {
207 // TODO Support quantization of scalar bias
208 throw std::runtime_error("Quantization of scalar bias is not yet supported (" +
209 const_bias->name() + ")");
210 }
211 if (size != const_bias->dim(const_bias->rank() - 1).value())
212 {
213 throw std::runtime_error(const_bias->name() +
214 " (bias) should have the shape of [1, 1, .. 1, channel]");
215 }
216
217 if (output_type == loco::DataType::U8)
218 {
219 new_bias = quant_bias_per_channel(const_bias, input_scale, weight_scale, scaling_factor, zp);
220 }
221 else if (output_type == loco::DataType::S16)
222 {
223 new_bias =
224 int16_quant_bias_per_channel(const_bias, input_scale, weight_scale, scaling_factor, zp);
225 }
226 else
227 {
228 throw std::runtime_error("Unsupported quantization type.");
229 }
230
231 auto quantparam = std::make_unique<CircleQuantParam>();
232 quantparam->scale = scaling_factor;
233 quantparam->zerop = zp;
234 quantparam->quantized_dimension = const_bias->rank() - 1;
235 assert(new_bias->quantparam() == nullptr); // bias should not be quantized before
236 new_bias->quantparam(std::move(quantparam));
237
238 return new_bias;
239 }
240 else
241 {
242 auto input_q = input->quantparam();
243 assert(input_q);
244 assert(input_q->scale.size() == 1); // Only support per-layer quant
245 auto input_scale = input_q->scale[0];
246
247 auto weight_q = weight->quantparam();
248 assert(weight_q);
249 assert(weight_q->scale.size() == 1); // Only support per-layer quant
250 auto weight_scale = weight_q->scale[0];
251
252 float scaling_factor{0};
253 int64_t zp{0};
254 new_bias =
255 asym_quant_bias_per_layer(const_bias, input_scale, weight_scale, &scaling_factor, &zp);
256 auto quantparam = std::make_unique<CircleQuantParam>();
257 quantparam->scale.push_back(scaling_factor);
258 quantparam->zerop.push_back(zp);
259 assert(new_bias->quantparam() == nullptr); // bias should not be quantized before
260 new_bias->quantparam(std::move(quantparam));
261
262 return new_bias;
263 }
