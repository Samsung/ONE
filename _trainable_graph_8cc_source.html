37void disableUnusedBackwardNodes(const UseDefChains &training_usedefs, TrainableGraph &tgraph)
38{
39 // Disable backward nodes that will be unused
40 const auto border = tgraph.btopolSortOperations();
41 for (const auto &op_index : border)
42 {
43 const auto &node = tgraph.operations().at(op_index);
44 const auto &candidates =
45 (node.getInputs() + node.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
46 const bool is_backward_op_used =
47 std::any_of(candidates.begin(), candidates.end(), [&](const OperandIndex &operand) {
48 const auto training_op_index = TrainingOperationIndex{op_index, false};
49 const auto forwarding_index = TrainingOperandIndex{operand, true};
50 const auto &forwarding_uses = training_usedefs.at(forwarding_index).getTrainingUses();
51 const auto backwarding_index = TrainingOperandIndex{operand, false};
52 const auto &backwarding_uses = training_usedefs.at(backwarding_index).getTrainingUses();
53 return forwarding_uses.find(training_op_index) != forwarding_uses.end() ||
54 backwarding_uses.find(training_op_index) != backwarding_uses.end();
55 });
56
57 // NOTE Backward op does not define any incoming operand in backwarding
58 const auto &inputs = node.getInputs() | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
59 const bool is_backward_op_def =
60 std::any_of(inputs.begin(), inputs.end(), [&](const OperandIndex &input) {
61 const auto training_op_index = TrainingOperationIndex{op_index, false};
62 const auto outcoming_index = TrainingOperandIndex{input, false};
63 const auto &backwarding_defs = training_usedefs.at(outcoming_index).getTrainingUses();
64 return backwarding_defs.find(training_op_index) != backwarding_defs.end();
65 });
66
67 if (is_backward_op_used || is_backward_op_def)
68 tgraph.enableBackward(op_index);
69 else
70 tgraph.disableBackward(op_index);
71 }
72}

206void TrainableGraph::validateTopologicalOrder(std::vector<ir::OperationIndex> order,
207 bool is_forward) const
208{
209 if (!is_forward)
210 std::reverse(order.begin(), order.end());
211
212 const std::string order_type = is_forward ? "forward" : "backward";
213
214 std::map<ir::OperationIndex, uint32_t> position;
215 for (uint32_t p = 0; p < order.size(); ++p)
216 {
217 auto index = order[p];
218 // TODO: replace this with `std::map::contains` after C++20
219 if (position.find(index) != position.end())
220 throw std::runtime_error{"Invalid " + order_type + " topological order: duplicate node @" +
221 std::to_string(index.value())};
222
223 position[index] = p;
224 }
225
226 operations().iterate([&](const ir::OperationIndex &index, const ir::IOperation &op) {
227 if (position.count(index) == 0)
228 return;
229
230 uint32_t p = position[index];
231
232 for (const auto &output : op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED)
233 {
234 const auto &operand = operands().at(output);
235 for (const auto &use : operand.getUses())
236 {
237 if (position.count(use) == 0)
238 continue;
239
240 uint32_t q = position[use];
241 if (p > q)
242 throw std::runtime_error{
243 "Invalid " + order_type + " topological order: inversion between @" +
244 std::to_string(index.value()) + " and @" + std::to_string(use.value())};
245 }
246 }
247 });
248}

277std::vector<ir::OperationIndex> TrainableGraph::btopolSortOperations() const
278{
279 std::vector<ir::OperationIndex> ret;
280 util::Set<ir::OperationIndex> unvisited;
281 ir::OperationIndex loss_idx;
282 operations().iterate([&](const ir::OperationIndex &index, const ir::IOperation &op) {
283 unvisited.add(index);
284 if (op.opcode() == ir::OpCode::Loss)
285 {
286 assert(!loss_idx.valid()); // Should be only one loss
287 loss_idx = index;
288 }
289 });
290
291 std::function<void(const ir::OperationIndex &, const ir::IOperation &)> dfs =
292 [&](const ir::OperationIndex &index, const ir::IOperation &op) -> void {
293 if (!unvisited.contains(index))
294 return;
295 unvisited.remove(index);
296
297 for (const auto &input : op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED)
298 {
299 const auto &operand = operands().at(input);
300 const auto &def = operand.getDef();
301 if (!def.valid())
302 continue;
303 dfs(def, operations().at(def));
304 }
305
306 ret.push_back(index);
307 };
308
309 dfs(loss_idx, operations().at(loss_idx));
310 std::reverse(ret.begin(), ret.end());
311 validateBackwardTopologicalOrder(ret);
312
313 return ret;
314}

316std::vector<ir::OperationIndex> TrainableGraph::essentialBackwardOrder() const
317{
318 auto backward_order = btopolSortOperations();
319 // get rid of all nodes not reachable from a node with trainable parameters
320 backward_order = truncateBackwardOrder(backward_order, [&](const OperationIndex &index) {
321 return operation(index).isRequiredForBackward();
322 });
323
324 return truncateBackwardOrder(backward_order);
325}

327std::vector<ir::OperationIndex> TrainableGraph::truncateBackwardOrder(
328 std::vector<ir::OperationIndex> backward_order,
329 std::function<bool(const ir::OperationIndex &)> alive_cond) const
330{
331 auto forward_order = backward_order;
332 std::reverse(forward_order.begin(), forward_order.end());
333 std::set<ir::OperationIndex> alive;
334
335 for (const auto &index : forward_order)
336 {
337 if (alive_cond(index))
338 alive.insert(index);
339
340 // TODO: replace this with `std::set::contains` after C++20
341 if (alive.find(index) != alive.end())
342 {
343 const auto &op = operations().at(index);
344 for (const auto &output : op.getOutputs())
345 {
346 const auto &operand = operands().at(output);
347 for (const auto &use : operand.getUses())
348 alive.insert(use);
349 }
350 }
351 }
352
353 // TODO: replace this with `std::erase_if(std::vector)` after C++20
354 backward_order.erase(
355 std::remove_if(backward_order.begin(), backward_order.end(),
356 [&](const auto &index) { return alive.find(index) == alive.end(); }),
357 backward_order.end());
358 return backward_order;
359}
