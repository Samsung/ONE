36void disableUnusedBackwardNodes(const UseDefChains &training_usedefs, TrainableGraph &tgraph)
37{
38 // Disable backward nodes that will be unused
39 const auto border = tgraph.btopolSortOperations();
40 for (const auto &op_index : border)
41 {
42 const auto &node = tgraph.operations().at(op_index);
43 const auto &candidates =
44 (node.getInputs() + node.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
45 const bool is_backward_op_used =
46 std::any_of(candidates.begin(), candidates.end(), [&](const OperandIndex &operand) {
47 const auto training_op_index = TrainingOperationIndex{op_index, false};
48 const auto forwarding_index = TrainingOperandIndex{operand, true};
49 const auto &forwarding_uses = training_usedefs.at(forwarding_index).getTrainingUses();
50 const auto backwarding_index = TrainingOperandIndex{operand, false};
51 const auto &backwarding_uses = training_usedefs.at(backwarding_index).getTrainingUses();
52 return forwarding_uses.find(training_op_index) != forwarding_uses.end() ||
53 backwarding_uses.find(training_op_index) != backwarding_uses.end();
54 });
55
56 // NOTE Backward op does not define any incoming operand in backwarding
57 const auto &inputs = node.getUsedInputSet();
58 const bool is_backward_op_def =
59 std::any_of(inputs.begin(), inputs.end(), [&](const OperandIndex &input) {
60 const auto training_op_index = TrainingOperationIndex{op_index, false};
61 const auto outcoming_index = TrainingOperandIndex{input, false};
62 const auto &backwarding_defs = training_usedefs.at(outcoming_index).getTrainingUses();
63 return backwarding_defs.find(training_op_index) != backwarding_defs.end();
64 });
65
66 if (is_backward_op_used || is_backward_op_def)
67 tgraph.enableBackward(op_index);
68 else
69 tgraph.disableBackward(op_index);
70 }
71}

200void TrainableGraph::validateTopologicalOrder(std::vector<ir::OperationIndex> order,
201 bool is_forward) const
202{
203 if (!is_forward)
204 std::reverse(order.begin(), order.end());
205
206 const std::string order_type = is_forward ? "forward" : "backward";
207
208 std::map<ir::OperationIndex, uint32_t> position;
209 for (uint32_t p = 0; p < order.size(); ++p)
210 {
211 auto index = order[p];
212 // TODO: replace this with `std::map::contains` after C++20
213 if (position.find(index) != position.end())
214 throw std::runtime_error{"Invalid " + order_type + " topological order: duplicate node @" +
215 std::to_string(index.value())};
216
217 position[index] = p;
218 }
219
220 operations().iterate([&](const ir::OperationIndex &index, const ir::IOperation &op) {
221 if (position.count(index) == 0)
222 return;
223
224 uint32_t p = position[index];
225
226 for (const auto &output : op.getUsedOutputSet())
227 {
228 const auto &operand = operands().at(output);
229 for (const auto &use : operand.getUses())
230 {
231 if (position.count(use) == 0)
232 continue;
233
234 uint32_t q = position[use];
235 if (p > q)
236 throw std::runtime_error{
237 "Invalid " + order_type + " topological order: inversion between @" +
238 std::to_string(index.value()) + " and @" + std::to_string(use.value())};
239 }
240 }
241 });
242}

271std::vector<ir::OperationIndex> TrainableGraph::btopolSortOperations() const
272{
273 std::vector<ir::OperationIndex> ret;
274 util::Set<ir::OperationIndex> unvisited;
275 ir::OperationIndex loss_idx;
276 operations().iterate([&](const ir::OperationIndex &index, const ir::IOperation &op) {
277 unvisited.add(index);
278 if (op.opcode() == ir::OpCode::Loss)
279 {
280 assert(!loss_idx.valid()); // Should be only one loss
281 loss_idx = index;
282 }
283 });
284
285 std::function<void(const ir::OperationIndex &, const ir::IOperation &)> dfs =
286 [&](const ir::OperationIndex &index, const ir::IOperation &op) -> void {
287 if (!unvisited.contains(index))
288 return;
289 unvisited.remove(index);
290
291 for (const auto &input : op.getUsedInputSet())
292 {
293 const auto &operand = operands().at(input);
294 const auto &def = operand.getDef();
295 if (!def.valid())
296 continue;
297 dfs(def, operations().at(def));
298 }
299
300 ret.push_back(index);
301 };
302
303 dfs(loss_idx, operations().at(loss_idx));
304 std::reverse(ret.begin(), ret.end());
305 validateBackwardTopologicalOrder(ret);
306
307 return ret;
308}

310std::vector<ir::OperationIndex> TrainableGraph::essentialBackwardOrder() const
311{
312 auto backward_order = btopolSortOperations();
313 // get rid of all nodes not reachable from a node with trainable parameters
314 backward_order = truncateBackwardOrder(backward_order, [&](const OperationIndex &index) {
315 return operation(index).isRequiredForBackward();
316 });
317
318 return truncateBackwardOrder(backward_order);
319}

321std::vector<ir::OperationIndex> TrainableGraph::truncateBackwardOrder(
322 std::vector<ir::OperationIndex> backward_order,
323 std::function<bool(const ir::OperationIndex &)> alive_cond) const
324{
325 auto forward_order = backward_order;
326 std::reverse(forward_order.begin(), forward_order.end());
327 std::set<ir::OperationIndex> alive;
328
329 for (const auto &index : forward_order)
330 {
331 if (alive_cond(index))
332 alive.insert(index);
333
334 // TODO: replace this with `std::set::contains` after C++20
335 if (alive.find(index) != alive.end())
336 {
337 const auto &op = operations().at(index);
338 for (const auto &output : op.getOutputs())
339 {
340 const auto &operand = operands().at(output);
341 for (const auto &use : operand.getUses())
342 alive.insert(use);
343 }
344 }
345 }
346
347 // TODO: replace this with `std::erase_if(std::vector)` after C++20
348 backward_order.erase(
349 std::remove_if(backward_order.begin(), backward_order.end(),
350 [&](const auto &index) { return alive.find(index) == alive.end(); }),
351 backward_order.end());
352 return backward_order;
353}
