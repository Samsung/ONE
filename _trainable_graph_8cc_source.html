77void TrainableGraph::verify(void) const
78{
79 _graph.verify();
80
81 operations().iterate([](const onert::ir::OperationIndex &, const onert::ir::IOperation &op) {
82 try
83 {
84 UNUSED_RELEASE(dynamic_cast<const onert::ir::train::ITrainableOperation &>(op));
85 }
86 catch (const std::bad_cast &)
87 {
88 std::runtime_error("TrainableGraph: " + op.name() + " is not a trainable operation");
89 }
90 });
91}
