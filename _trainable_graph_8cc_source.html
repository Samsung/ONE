133std::vector<ir::OperationIndex> TrainableGraph::btopolSortOperations() const
134{
135 std::vector<ir::OperationIndex> ret;
136 util::Set<ir::OperationIndex> unvisited;
137 ir::OperationIndex loss_idx;
138 operations().iterate([&](const ir::OperationIndex &index, const ir::IOperation &op) {
139 unvisited.add(index);
140 if (op.opcode() == ir::OpCode::Loss)
141 {
142 assert(!loss_idx.valid()); // Should be only one loss
143 loss_idx = index;
144 }
145 });
146
147 std::function<void(const ir::OperationIndex &, const ir::IOperation &)> dfs =
148 [&](const ir::OperationIndex &index, const ir::IOperation &op) -> void {
149 if (!unvisited.contains(index))
150 return;
151 unvisited.remove(index);
152 ret.push_back(index);
153
154 for (const auto &input : op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED)
155 {
156 const auto &operand = operands().at(input);
157 const auto &def = operand.getDef();
158 if (!def.valid())
159 return;
160 dfs(def, operations().at(def));
161 }
162 };
163
164 dfs(loss_idx, operations().at(loss_idx));
165
166 return ret;
167}
