122circle::BuiltinOperator CircleReader::builtin_code(const circle::Operator *op) const
123{
124 assert(op != nullptr);
125

127 uint32_t index = op->opcode_index();

145bool CircleReader::select_subgraph(uint32_t sgindex)
146{
147 if (num_subgraph() <= sgindex)
148 {
149 assert(false);
150 return false;
151 }
152
153 // for direct pointer access
154 auto subgraphs = _model->subgraphs();
155 assert(subgraphs != nullptr);
156
157 _current_subgraph = subgraphs->Get(sgindex);
158 assert(_current_subgraph != nullptr);
159
160 return true;
161}
