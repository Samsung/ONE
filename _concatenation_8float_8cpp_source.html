26template <FusedActivationFunctionType Ac, typename Scalar>
27void Concatenation(int concat_dim, const Scalar *const *input_data,
28 const Dims<4> *const *input_dims, int inputs_count, Scalar *output_data,
29 const Dims<4> &output_dims)
30{
31 DCHECK_GT(inputs_count, 1);
32 int concat_size = 0;
33 for (int i = 0; i < inputs_count; i++)
34 {
35 for (int j = 0; j < 4; j++)
36 {
37 if (j != concat_dim)
38 {
39 MatchingArraySize(*input_dims[i], j, output_dims, j);
40 }
41 }
42 concat_size += ArraySize(*input_dims[i], concat_dim);
43 }
44 DCHECK_EQ(concat_size, ArraySize(output_dims, concat_dim));
45 DCHECK(IsPackedWithoutStrides(output_dims));
46 // for now we dont have a model with a Concatenation
47 // with fused activation function.
48 DCHECK(Ac == FusedActivationFunctionType::kNone);
49 int outer_size = 1;
50 for (int i = concat_dim + 1; i < 4; i++)
51 {
52 outer_size *= output_dims.sizes[i];
53 }
54 Scalar *output_ptr = output_data;
55 for (int k = 0; k < outer_size; k++)
56 {
57 for (int i = 0; i < inputs_count; ++i)
58 {
59 const int copy_size = input_dims[i]->sizes[concat_dim] * input_dims[i]->strides[concat_dim];
60 memcpy(output_ptr, input_data[i] + k * copy_size, copy_size * sizeof(Scalar));
61 output_ptr += copy_size;
62 }
63 }
64}

66bool concatenationFloat32(const std::vector<const float *> &inputDataPtrs,
67 const std::vector<Shape> &inputShapes, int32_t axis, float *outputData,
68 const Shape &outputShape)
69{
70 int num_inputs = inputShapes.size();
71 std::vector<Dims<4> *> inputDimsPtr(num_inputs);
72 std::vector<Dims<4>> inputDims(num_inputs);
73 for (int i = 0; i < num_inputs; i++)
74 {
75 inputDims[i] = convertShapeToDims(inputShapes[i]);
76 inputDimsPtr[i] = &inputDims[i];
77 }
78
79 Concatenation<FusedActivationFunctionType::kNone, float>(
80 getNumberOfDimensions(outputShape) - axis - 1, inputDataPtrs.data(), inputDimsPtr.data(),
81 num_inputs, outputData, convertShapeToDims(outputShape));
82
83 return true;
84}
