27bool read_part(const crew::Section &section, crew::Part &part)
28{
29 // construct Source from section_source
30 part.model_file = crew::find(section, "file");
31 if (part.model_file.empty())
32 return false;
33
34 // read inputs for Source
35 for (int32_t i = 1;; ++i)
36 {
37 std::string item = "i" + std::to_string(i);
38 std::string input = crew::find(section, item);
39 if (input.empty())
40 break;
41
42 part.inputs.push_back(input);
43 }
44 // read outputs for Source
45 for (int32_t i = 1;; ++i)
46 {
47 std::string item = "o" + std::to_string(i);
48 std::string output = crew::find(section, item);
49 if (output.empty())
50 break;
51
52 part.outputs.push_back(output);
53 }
54 return true;
55}

62void write_part(crew::JsonExport &je, const crew::Part &part)
63{
64 std::vector<std::string> graph_inputs;
65 std::vector<std::string> graph_outputs;
66
67 for (auto &input : part.inputs)
68 {
69 graph_inputs.push_back(input);
70 }
71 for (auto &output : part.outputs)
72 {
73 graph_outputs.push_back(output);
74 }
75
76 je.key_val("file", part.model_file.c_str(), true);
77 je.key_val("inputs", graph_inputs, true);
78 je.key_val("outputs", graph_outputs, false);
79}

81void write_parts(crew::JsonExport &je, const crew::Parts &parts)
82{
83 uint32_t idx = 1;

85 for (auto &part : parts)

99void part_to_section_io(const crew::Part &part, crew::Section &section)
100{
101 uint32_t idx = 1;
102 for (auto &input : part.inputs)
103 {
104 std::string key = "i" + std::to_string(idx);
105 section.items.emplace(key, input);
106 idx++;
107 }
108 idx = 1;
109 for (auto &output : part.outputs)
110 {
111 std::string key = "o" + std::to_string(idx);
112 section.items.emplace(key, output);
113 idx++;
114 }
115}

122bool read_ini(const std::string &path, PConfig &pconfig)
123{
124 auto sections = crew::read_ini(path);
125
126 auto section_source = crew::find(sections, "source");
127 auto section_models = crew::find(sections, "models");
128 if (section_source.name != "source" || section_models.name != "models")
129 {
130 return false;
131 }
132
133 if (!read_part(section_source, pconfig.source))
134 {
135 return false;
136 }
137
138 // get models list
139 std::vector<std::string> models;
140 for (int32_t i = 1;; ++i)
141 {
142 std::string item = "m" + std::to_string(i);
143 std::string model = crew::find(section_models, item);
144 if (model.empty())
145 break;
146
147 models.push_back(model);
148 }
149
150 for (auto &model : models)
151 {
152 auto section_model = crew::find(sections, model);
153
154 Part part;
155 if (!read_part(section_model, part))
156 {
157 return false;
158 }
159 pconfig.parts.push_back(part);
160 }
161
162 return true;
163}

165bool write_ini(std::ostream &os, const PConfig &pconfig)
166{
167 crew::Sections sections;
168
169 // make [source]
170 crew::Section section_source;
171 section_source.name = "source";
172 section_source.items["file"] = pconfig.source.model_file;
173 part_to_section_io(pconfig.source, section_source);
174 sections.push_back(section_source);
175
176 // make [models]
177 crew::Section section_models;
178 section_models.name = "models";
179 uint32_t idx = 1;
180 for (auto &part : pconfig.parts)
181 {
182 std::string key = "m" + std::to_string(idx);
183 section_models.items[key] = part.model_file;
184 idx++;
185 }
186 sections.push_back(section_models);
187
188 for (auto &part : pconfig.parts)
189 {
190 // make circle model section
191 crew::Section section_model;
192 section_model.name = part.model_file;
193 section_model.items["file"] = part.model_file;
194 part_to_section_io(part, section_model);
195 sections.push_back(section_model);
196 }
197
198 write_ini(os, sections);
199
200 return true;
201}

203bool write_json(std::ostream &os, const PConfig &pconfig)
204{

213 {
214 je.open_bracket("parts");
