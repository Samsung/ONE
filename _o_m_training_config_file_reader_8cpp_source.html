22/*
23 * Validate method for Weight Only Format file to check its correctness
24 */
25OMStatus OMTrainingConfigReader::validate(OMCircleReader *reader)
26{
27 OMStatus status = Ok;
28
29 // Validate magic number
30 uint16_t mag_num = 0;
31 std::memcpy(&mag_num, &_train_config_ptr[MAGIC_NUMBER_FIELD], sizeof(mag_num));
32 assert(mag_num == train_config_file_magic_number &&
33 "False MAGIC NUMBER, check correctness of wof file");
34 if (mag_num != train_config_file_magic_number)
35 return FailReadWOFFile;
36
37 // Validate schema version
38 uint8_t version = 0;
39 std::memcpy(&version, &_train_config_ptr[SCHEMA_VERSION_FIELD], sizeof(version));
40 assert(version == train_config_file_schema_version &&
41 "False MAGIC NUMBER, check correctness of wof file");
42 if (version != train_config_file_schema_version)
43 return FailReadWOFFile;
44
45 // Validate count of ops is not greater than current model has
46 assert(reader != nullptr && "Reader should exist");
47 if (reader == nullptr)
48 return ModelNotImport;
49 uint32_t num_ops = reader->operators()->size();
50 uint32_t num_ops_in_file = 0;
51 std::memcpy(&num_ops_in_file, &_train_config_ptr[NUM_LAYERS_FIELD], sizeof(num_ops_in_file));
52 assert(num_ops_in_file > 0 and num_ops >= num_ops_in_file &&
53 "Number of operators in circle should be greater than train config file has");
54 if (num_ops_in_file > 0 and num_ops < num_ops_in_file)
55 return FailReadWOFFile;
56
57 return status;
58}

61 */
62std::unordered_map<uint16_t, uint8_t> OMTrainingConfigReader::getTrainableOpsIndexes()
63{
64 std::unordered_map<uint16_t, uint8_t> result;
65
66 // If reader is not parsed then return empty vector
67 if (_train_config_ptr == nullptr)
68 return result;
69
70 // Read number of ops
71 uint32_t num_ops_in_file = 0;
72 std::memcpy(&num_ops_in_file, &_train_config_ptr[NUM_LAYERS_FIELD], sizeof(num_ops_in_file));
73
74 assert(num_ops_in_file > 0);
75 // Obtain pointer to the first layer index position in the file
76 char *cur_op_index_ptr = &_train_config_ptr[FIRST_LAYER_INDEX_FIELD];
77 char *cur_op_train_rank_ptr =
78 &_train_config_ptr[FIRST_LAYER_INDEX_FIELD + sizeof(uint16_t) * num_ops_in_file];
79 // Fill result set with indexes and its rank
80 for (uint32_t i = 0; i < num_ops_in_file; ++i)
81 {
82 // Read op index
83 uint16_t cur_op_index;
84 std::memcpy(&cur_op_index, cur_op_index_ptr, sizeof(cur_op_index));
85 cur_op_index_ptr += sizeof(cur_op_index);
86
87 // Read op train rank
88 uint8_t cur_op_train_rank;
89 std::memcpy(&cur_op_train_rank, cur_op_train_rank_ptr, sizeof(cur_op_train_rank));
90 cur_op_train_rank_ptr += sizeof(cur_op_train_rank);
91
92 // Insert op index and op rank
93 result[cur_op_index] = cur_op_train_rank;
94 }
95
96 return result;
97}
