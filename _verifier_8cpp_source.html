58public:
59 void run(void) const
60 {
61 for (auto node : loco::all_nodes(_graph))
62 {
63 // Verify nodes
64 for (uint32_t n = 0; n < node->arity(); ++n)
65 {
66 if (node->arg(n) == nullptr)
67 {
68 notify(ErrorDetail<ErrorCategory::MissingArgument>{node, n});
69 }
70 }
71 }
72 }
73
74private:

76 {
77 for (const auto &listener : _listeners)

79 listener->notify(error);
80 }
81 }

83private:

89 // Owned error listeners
