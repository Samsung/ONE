26static void lowerConv2D(mir::Graph *graph, mir::ops::Conv2DOp *op)
27{
28 mir::Operation::Output *input = op->getInput(0);
29 mir::Operation::Output *kernel = op->getInput(1);
30
31 const std::int32_t in_group_size = kernel->getShape().dim(3);
32 const std::int32_t out_group_size = kernel->getShape().dim(0) / op->getNumGroups();
33
34 if (in_group_size == 1 && out_group_size == 1)
35 {
36 // [O, H, W, I / M] == [M, H, W, 1] -> [H, W, M, 1]
37 std::vector<std::size_t> perm{1, 2, 0, 3};
38 mir::Operation::Output *new_kernel =
39 graph->create<mir::ops::TransposeOp>(kernel, perm)->getOutput(0);
40 mir::Conv2DOpAttributes attributes = op->getAttributes();
41 attributes.num_groups = 1;
42 mir::Operation::Output *new_result =
43 graph->create<mir::ops::DepthwiseConv2DOp>(input, new_kernel, attributes)->getOutput(0);
44 graph->replaceNode(op, new_result->getNode());
45 }
46}

50PassData LowerConv2D::run(PassData data)
51{
52 auto *graph = static_cast<mir::Graph *>(data);
53
54 // Collect candidate ops before actual transformation because the graph will be changed.
55 std::vector<mir::ops::Conv2DOp *> group_conv_ops;
56 for (mir::Operation *op : graph->getNodes())
57 {
58 auto *conv_op = dynamic_cast<mir::ops::Conv2DOp *>(op);
59 if (conv_op != nullptr && conv_op->getNumGroups() != 1)
60 {
61 group_conv_ops.push_back(conv_op);
62 }
63 }
64
65 for (mir::ops::Conv2DOp *op : group_conv_ops)
66 {
67 lowerConv2D(graph, op);
68 }
69
70 return graph;
71}
