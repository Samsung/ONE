26// Check common (non-op-specific) attributes of lhs and rhs
27bool same_common_attributes(const luci::CircleNode *lhs, const luci::CircleNode *rhs)
28{
29 // Opcode
30 if (lhs->opcode() != rhs->opcode())
31 return false;
32
33 // Shape
34 if (lhs->rank() != rhs->rank())
35 return false;
36
37 for (uint32_t i = 0; i < lhs->rank(); i++)
38 {
39 if (lhs->dim(i).known() != rhs->dim(i).known())
40 return false;
41
42 if (lhs->dim(i).value() != rhs->dim(i).value())
43 return false;
44 }
45
46 // Data type
47 if (lhs->dtype() != rhs->dtype())
48 return false;
49
50 // Op version
51 if (lhs->op_version() != rhs->op_version())
52 return false;
53
54 // QuantParam
55 const auto lhs_qparam = lhs->quantparam();
56 const auto rhs_qparam = rhs->quantparam();
57
58 if (lhs_qparam == nullptr and rhs_qparam != nullptr)
59 return false;
60
61 if (lhs_qparam != nullptr and rhs_qparam == nullptr)
62 return false;
63
64 if (lhs_qparam)
65 {
66 assert(rhs_qparam); // FIX_ME_UNLESS
67
68 if (lhs_qparam->scale != rhs_qparam->scale)
69 return false;
70
71 if (lhs_qparam->zerop != rhs_qparam->zerop)
72 return false;
73
74 if (lhs_qparam->min != rhs_qparam->min)
75 return false;
76
77 if (lhs_qparam->max != rhs_qparam->max)
78 return false;
79 }
80
81 return true;
82}

84// Return true if two constants are the same
85bool same_const(const luci::CircleConst *x, const luci::CircleConst *y)
86{
87 assert(x != nullptr); // FIX_CALLER_UNLESS
88 assert(y != nullptr); // FIX_CALLER_UNLESS
89
90 RETURN_FALSE_UNLESS(same_common_attributes(x, y));
91
92 switch (x->dtype())
93 {
94 case loco::DataType::S32:
95 {
96 const auto size_x = x->size<loco::DataType::S32>();
97 const auto size_y = y->size<loco::DataType::S32>();
98 RETURN_FALSE_UNLESS(size_x == size_y);
99
100 for (uint32_t i = 0; i < size_x; i++)
101 {
102 RETURN_FALSE_UNLESS(x->at<loco::DataType::S32>(i) == y->at<loco::DataType::S32>(i));
103 }
104 return true;
105 }
106 // TODO Support more dtypes
107 default:
108 // Simply return false
109 return false;
110 }
111
112 return true;
113}

115// Return true if x and y are semantically equal
116bool same_attributes(const luci::CircleTranspose *x, luci::CircleTranspose *y)
117{
118 assert(x != nullptr); // FIX_CALLER_UNLESS
119 assert(y != nullptr); // FIX_CALLER_UNLESS
120
121 assert(same_common_attributes(x, y)); // FIX_CALLER_UNLESS
122
123 const auto perm_x = dynamic_cast<luci::CircleConst *>(x->perm());
124 const auto perm_y = dynamic_cast<luci::CircleConst *>(y->perm());
125
126 RETURN_FALSE_UNLESS(perm_x);
127 RETURN_FALSE_UNLESS(perm_y);
128
129 // Check perm_x and perm_y are the same
130 RETURN_FALSE_UNLESS(same_const(perm_x, perm_y));
131
132 return true;
133}

135// Use a similar approach as boost's hash_combine
136template <class T> inline void hash_combine(std::size_t &s, const T v)
137{
138 std::hash<T> h;
139 s ^= h(v) + 0x9e3779b9 + (s << 6) + (s >> 2);
140}
141
142template <> inline void hash_combine(std::size_t &s, luci::CircleNode *node)
143{
144 // Shape
145 hash_combine(s, node->rank());
146 for (uint32_t i = 0; i < node->rank(); i++)
147 hash_combine(s, node->dim(i).value());
148
149 // Data type
150 hash_combine(s, static_cast<std::size_t>(node->dtype()));
151
152 // Op version
153 hash_combine(s, node->op_version());
154
155 // Op code
156 hash_combine(s, node->opcode());
157
158 // QuantParam
159 // Let's skip quantparam to reduce burden of hash function
160}

169Expression Expression::build(luci::CircleNode *node)
170{
171 if (node == nullptr)
172 throw std::invalid_argument("node");
173
174 Expression key;
175 {
176 switch (node->opcode())
177 {
178 case luci::CircleOpcode::QUANTIZE:
179 case luci::CircleOpcode::TRANSPOSE:
180 key.inputs.emplace_back(node->arg(0));
181 break;
182 // TODO Add more Ops
183 default:
184 // NYI. Return invalid expression
185 key.op = nullptr;
186 return key;
187 }
188
189 key.op = node;
190 }
191
192 return key;
193}

195bool operator==(const Expression &x, const Expression &y)
196{
197 if (x.inputs != y.inputs)
198 return false;
199
200 // Check general (non-op-specific) attributes
201 if (not same_common_attributes(x.op, y.op))
202 return false;
203
204 assert(x.op->opcode() == y.op->opcode()); // FIX_ME_UNLESS
205
206 // Check op-specific attributes
207 switch (x.op->opcode())
208 {
209 case luci::CircleOpcode::QUANTIZE:
210 {
211 // This Op has no op-specific attribute.
212 // same_common_attributes is enough.
213 return true;
214 }
215 case luci::CircleOpcode::TRANSPOSE:
216 {
217 const auto trans_x = loco::must_cast<luci::CircleTranspose *>(x.op);
218 const auto trans_y = loco::must_cast<luci::CircleTranspose *>(y.op);
219
220 return same_attributes(trans_x, trans_y);
221 }
222 // TODO Implement more operators
223 default:
224 // NYI: Unsupported operators
225 return false;
226 }
227
228 return true;
229}

231std::size_t Expression::Hash::call(const Expression &k) const noexcept
232{
233 std::size_t res = 0;
234 for (const auto input : k.inputs)
235 hash_combine(res, input);
236
237 hash_combine(res, k.op);
238
239 return res;
240}
