58void CLPadLayerEx::configure(const CLCompileContext &compile_context, ICLTensor *input,
59 ICLTensor *output, const PaddingList &padding,
60 PixelValue constant_value, PaddingMode mode)
61{
62 ARM_COMPUTE_ERROR_THROW_ON(
63 validate(input->info(), output->info(), padding, constant_value, mode));
64
65 _perform_pad = std::any_of(padding.begin(), padding.end(),
66 [](PaddingInfo info) { return info.first > 0 || info.second > 0; });
67
68 if (_perform_pad)
69 {
70 _pad_kernel->configure(compile_context, input, output, padding, constant_value, mode);
71 }
72 else
73 {
74 Window copy_window = Window();
75 copy_window.use_tensor_dimensions(output->info()->tensor_shape());
76 // Copy the input to the whole output if no padding is applied
77 _copy_kernel->configure(compile_context, input->info(), output->info(), &copy_window);
78 }
79}
80Status CLPadLayerEx::validate(const ITensorInfo *input, const ITensorInfo *output,
81 const PaddingList &padding, PixelValue constant_value,
82 PaddingMode mode)
83{
84 bool perform_pad = std::any_of(padding.begin(), padding.end(), [](PaddingInfo info) {
85 return info.first > 0 || info.second > 0;
86 });
87
88 if (perform_pad)
89 {
90 ARM_COMPUTE_RETURN_ON_ERROR(
91 CLPadLayerKernelEx::validate(input, output, padding, constant_value, mode));
92 }
93 else
94 {
95 ARM_COMPUTE_RETURN_ON_ERROR(opencl::kernels::ClCopyKernel::validate(input, output));
96 }
97 return Status{};
98}
99void CLPadLayerEx::run()
100{
101 if (_perform_pad)
102 {
103 CLScheduler::get().enqueue(*_pad_kernel);
104 }
105 else
106 {
107 CLScheduler::get().enqueue(*_copy_kernel);
108 }
109}
