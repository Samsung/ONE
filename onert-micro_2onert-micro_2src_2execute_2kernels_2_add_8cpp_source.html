36void calculateQuantParams(core::ArithmeticQuantParams &params, const circle::Tensor *input1,
37 const circle::Tensor *input2, const circle::Tensor *output,
38 circle::ActivationFunctionType act)
39{
40 assert(input1->quantization() != nullptr); // Fix caller
41 assert(input2->quantization() != nullptr); // Fix caller
42 assert(output->quantization() != nullptr); // Fix caller
43
44 assert(input1->quantization()->scale() != nullptr and
45 input1->quantization()->scale()->size() == 1); // Fix caller
46 assert(input2->quantization()->scale() != nullptr and
47 input2->quantization()->scale()->size() == 1); // Fix caller
48 assert(output->quantization()->scale() != nullptr and
49 output->quantization()->scale()->size() == 1); // Fix caller
50
51 assert(input1->quantization()->zero_point() != nullptr and
52 input1->quantization()->zero_point()->size() == 1); // Fix caller
53 assert(input2->quantization()->zero_point() != nullptr and
54 input2->quantization()->zero_point()->size() == 1); // Fix caller
55 assert(output->quantization()->zero_point() != nullptr and
56 output->quantization()->zero_point()->size() == 1); // Fix caller
57
58 // 8bit -> 8bit general quantized path, with general rescalings
59 const auto input1_zp = input1->quantization()->zero_point()->operator[](0);
60 const auto input2_zp = input2->quantization()->zero_point()->operator[](0);
61 const auto output_zp = output->quantization()->zero_point()->operator[](0);
62
63 const auto input1_scale = input1->quantization()->scale()->operator[](0);
64 const auto input2_scale = input2->quantization()->scale()->operator[](0);
65 const auto output_scale = output->quantization()->scale()->operator[](0);
66
67 params.input1_offset = -static_cast<int32_t>(input1_zp);
68 params.input2_offset = -static_cast<int32_t>(input2_zp);
69 params.output_offset = static_cast<int32_t>(output_zp);
70 params.left_shift = (output->type() == circle::TensorType_INT16) ? 15 : 20;
71 const double twice_max_input_scale =
72 2 * static_cast<double>(std::max(input1_scale, input2_scale));
73 const double real_input1_multiplier = static_cast<double>(input1_scale) / twice_max_input_scale;
74 const double real_input2_multiplier = static_cast<double>(input2_scale) / twice_max_input_scale;
75 const double real_output_multiplier =
76 twice_max_input_scale / ((1 << params.left_shift) * static_cast<double>(output_scale));
77
78 quantizeMultiplierSmallerThanOneExp(real_input1_multiplier, &params.input1_multiplier,
79 &params.input1_shift);
80
81 quantizeMultiplierSmallerThanOneExp(real_input2_multiplier, &params.input2_multiplier,
82 &params.input2_shift);
83
84 quantizeMultiplierSmallerThanOneExp(real_output_multiplier, &params.output_multiplier,
85 &params.output_shift);
86
87 calculateActivationRangeQuantized(act, output_zp, output_scale, output->type(),
88 &params.quantized_activation_min,
89 &params.quantized_activation_max);
90}

96OMStatus onert_micro::execute::execute_kernel_CircleAdd(const OMExecuteArgs &execute_args)
97{
98 core::OMRuntimeContext &runtime_context = execute_args.runtime_context;
99 core::OMRuntimeStorage &runtime_storage = execute_args.runtime_storage;
100 uint16_t op_index = execute_args.kernel_index;
101
102 const circle::Tensor *input1;
103 const circle::Tensor *input2;
104 const circle::Tensor *output;
105
106 uint8_t *input1_data;
107 uint8_t *input2_data;
108 uint8_t *output_data;
109
110 uint16_t input1_index = 0;
111 uint16_t input2_index = 0;
112
113 const circle::AddOptions *options;
114 // Read kernel
115 {
116 execute::OMRuntimeKernel runtime_kernel;
117 runtime_kernel.readKernel(op_index, runtime_context);
118
119 input1 = runtime_kernel.inputs[input1TensorIdx];
120 input2 = runtime_kernel.inputs[input2TensorIdx];
121 output = runtime_kernel.outputs[outputTensorIdx];
122 assert(input1 != nullptr);
123 assert(input2 != nullptr);
124 assert(output != nullptr);
125
126 runtime_kernel.getDataFromStorage(op_index, runtime_storage, runtime_context);
127
128 input1_data = runtime_kernel.inputs_data[input1TensorIdx];
129 input2_data = runtime_kernel.inputs_data[input2TensorIdx];
130 output_data = runtime_kernel.outputs_data[outputTensorIdx];
131 assert(input1_data != nullptr);
132 assert(input2_data != nullptr);
133 assert(output_data != nullptr);
134
135 options = runtime_kernel.first_operator->builtin_options_as_AddOptions();
136
137 input1_index = runtime_kernel.inputs_index[input1TensorIdx];
138 input2_index = runtime_kernel.inputs_index[input2TensorIdx];
139 }
140
141 OMStatus status;
142
143 core::OMRuntimeShape input1_shape(input1);
144 core::OMRuntimeShape input2_shape(input2);
145 core::OMRuntimeShape output_shape(output);
146
147#ifndef DIS_DYN_SHAPES
148 // Check dynamic shapes
149 if (runtime_storage.getDynamicTensorSize(input1_index) != -1)
150 input1_shape = output_shape;
151
152 if (runtime_storage.getDynamicTensorSize(input2_index) != -1)
153 input2_shape = output_shape;
154#endif // DIS_DYN_SHAPES
155
156 // Check broadcast property
157 core::BinaryArithmeticBroadcastParams params{};
158 const bool need_broadcast = pal::processBroadcastShapes(input1_shape, input2_shape, &params);
159 switch (input1->type())
160 {
161#ifndef DIS_FLOAT
162 case circle::TensorType_FLOAT32:
163 {
164 execute::calculateActivationRange(options->fused_activation_function(),
165 &params.float_activation_min, &params.float_activation_max);
166 if (need_broadcast)
167 {
168 status = pal::BroadcastAdd4DSlow(
169 params, input1_shape, core::utils::castInputData<float>(input1_data), input2_shape,
170 core::utils::castInputData<float>(input2_data), output_shape,
171 core::utils::castOutputData<float>(output_data));
172 }
173 else
174 {
175 status =
176 pal::Add(params, output_shape.flatSize(), core::utils::castInputData<float>(input1_data),
177 core::utils::castInputData<float>(input2_data),
178 core::utils::castOutputData<float>(output_data));
179 }
180 }
181 break;
182#endif // DIS_FLOAT
183 case circle::TensorType_INT64:
184 {
185 execute::calculateActivationRange(options->fused_activation_function(),
186 &params.int64_activation_min, &params.int64_activation_max);
187
188 if (need_broadcast)
189 {
190 status = pal::BroadcastAdd4DSlow(
191 params, input1_shape, core::utils::castInputData<int64_t>(input1_data), input2_shape,
192 core::utils::castInputData<int64_t>(input2_data), output_shape,
193 core::utils::castOutputData<int64_t>(output_data));
194 }
195 else
196 {
197 status = pal::Add(params, input1_shape.flatSize(),
198 core::utils::castInputData<int64_t>(input1_data),
199 core::utils::castInputData<int64_t>(input2_data),
200 core::utils::castOutputData<int64_t>(output_data));
201 }
202 }
203 break;
204 case circle::TensorType_INT32:
205 {
206 execute::calculateActivationRange(options->fused_activation_function(),
207 &params.int32_activation_min, &params.int32_activation_max);
208
209 if (need_broadcast)
210 {
211 status = pal::BroadcastAdd4DSlow(
212 params, input1_shape, core::utils::castInputData<int32_t>(input1_data), input2_shape,
213 core::utils::castInputData<int32_t>(input2_data), output_shape,
214 core::utils::castOutputData<int32_t>(output_data));
215 }
216 else
217 {
218 status = pal::Add(params, input1_shape.flatSize(),
219 core::utils::castInputData<int32_t>(input1_data),
220 core::utils::castInputData<int32_t>(input2_data),
221 core::utils::castOutputData<int32_t>(output_data));
222 }
223 }
224 break;
225#ifndef DIS_QUANT
226 case circle::TensorType_INT8:
227 {
228 core::ArithmeticQuantParams add_params{};
229
230 calculateQuantParams(add_params, input1, input2, output,
231 options->fused_activation_function());
232
233 if (need_broadcast)
234 {
235 status = pal::BroadcastAdd4DSlow(
236 add_params, input1_shape, core::utils::castInputData<int8_t>(input1_data), input2_shape,
237 core::utils::castInputData<int8_t>(input2_data), output_shape,
238 core::utils::castOutputData<int8_t>(output_data));
239 }
240 else
241 {
242 status = pal::Add(add_params, input1_shape.flatSize(),
243 core::utils::castInputData<int8_t>(input1_data),
244 core::utils::castInputData<int8_t>(input2_data),
245 core::utils::castOutputData<int8_t>(output_data));
246 }
247 }
248 break;
249#endif // DIF_QUANT
250 default:
251 {
252 status = UnsupportedType;
253 assert(false && "Unsupported type.");
254 }
255 }
256
257 return status;
258}
