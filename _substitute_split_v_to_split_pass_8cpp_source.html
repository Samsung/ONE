27// SplitV is substituted to Split if the contents of size_splits are all same
28// For example,
29// size_splits = [32, 32] -> substitute
30// size_splits = [31, 33] -> do not substitute
31bool resolve_splitv(luci::CircleSplitV *sv)
32{
33 auto size_splits = dynamic_cast<luci::CircleConst *>(sv->size_splits());
34 if (not size_splits)
35 return false;
36
37 if (size_splits->dtype() != loco::DataType::S32)
38 return false;
39
40 auto num_split = size_splits->size<loco::DataType::S32>();
41 if (static_cast<int32_t>(num_split) != sv->num_split())
42 return false;
43
44 if (num_split < 1)
45 return false;
46
47 // Check the contents of size_splits are all same
48 auto first_size = size_splits->at<loco::DataType::S32>(0);
49 for (uint32_t i = 1; i < num_split; i++)
50 {
51 if (first_size != size_splits->at<loco::DataType::S32>(i))
52 return false;
53 }
54
55 auto graph = sv->graph();
56 auto split_node = graph->nodes()->create<luci::CircleSplit>();
57 split_node->input(sv->input());
58 split_node->split_dim(sv->split_dim());
59 split_node->num_split(sv->num_split());
60 split_node->name(sv->name());
61 copy_quantparam(sv, split_node);
62 luci::add_origin(split_node, luci::get_origin(sv));
63
64 auto succs = loco::succs(sv);
65 for (auto succ : succs)
66 {
67 auto svo = loco::must_cast<luci::CircleSplitVOut *>(succ);
68 auto so_node = graph->nodes()->create<luci::CircleSplitOut>();
69 so_node->input(split_node);
70 so_node->index(svo->index());
71 so_node->name(svo->name());
72 copy_quantparam(svo, so_node);
73 luci::add_origin(so_node, luci::get_origin(svo));
74
75 replace(svo).with(so_node);
76 }
77
78 return true;
79}

107bool SubstituteSplitVToSplitPass::run(loco::Graph *g)
108{
109 bool changed = false;
110 for (auto node : loco::active_nodes(loco::output_nodes(g)))
111 {
112 if (auto sv = dynamic_cast<luci::CircleSplitV *>(node))
113 {
114 if (resolve_splitv(sv))
115 changed = true;
116 }
117 }
118
119 return changed;
120}
