45public:
46 // It is introduced to reduce virtual method call overhead on inference
47 // So derived class should maintain actual OperandInfo to "_info" field
48 // Ex. CPU backend getShape() in OperationUtils.h is called frequently on inference
49 // and it calls get_info()
50 const ir::OperandInfo &get_info() const { return _info; }
51 const ir::Sparsity *sparsity() const { return _info.typeInfo().sparsity(); }
52
53 // Finalized methods for IPortableTensor by "_info" field read
54 size_t total_size() const override final { return _info.total_size(); }
55 size_t calcOffset(const ir::Coordinates &coords) const override final;
56 ir::DataType data_type() const override final { return _info.typeInfo().type(); }
57 float data_scale() const override final { return _info.typeInfo().scale(); }
58 int32_t data_zero_point() const override final { return _info.typeInfo().zero_point(); }
59 const std::vector<float> &data_scales() const override final { return _info.typeInfo().scales(); }
60 const std::vector<int32_t> &data_zero_points() const override
61 {
62 return _info.typeInfo().zero_points();
63 }
64 bool is_constant() const override final { return _info.isConstant(); }
65 bool is_dynamic() const override final { return _info.isDynamic(); }
66 ir::Shape getShape() const override final { return _info.shape(); }
67
68 // Finalized methods for IPortableTensor by no padding
69 bool has_padding() const final { return false; }
70 void access(const std::function<void(ITensor &tensor)> &fn) final { fn(*this); }
71
72protected:
73 ir::OperandInfo _info;
