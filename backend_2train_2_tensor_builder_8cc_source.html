37void TensorBuilder::registerTensorInfo(const ir::OperandIndex &index, const ir::OperandInfo &info,
38 ir::Layout layout)
39{
40 _tensor_info_map.emplace(index, info);
41 _as_constants[index] = info.isConstant();
42
43 // Train backend supports only one layout as NHWC
44 assert(layout == ir::Layout::NHWC);
45 assert(!info.isDynamic());
46
47 // NOTE For now, whether or not to build operands to trainable tensor depends on whether
48 // the corresponding operand is constant.
49 if (_as_constants[index])
50 {
51 auto tensor = std::make_unique<TrainableTensor>(info, layout);
52 _tensor_reg->setTrainableTensor(index, std::move(tensor));
53 }
54 else
55 {
56 auto tensor = std::make_unique<Tensor>(info, layout);
57 _tensor_reg->setNonConstTensor(index, std::move(tensor));
58 }
59}

61void TensorBuilder::registerBackwardTensorInfo(const ir::OperandIndex &index,
62 const ir::OperandInfo &info, ir::Layout layout)
63{
64 _backward_tensor_info_map.emplace(index, info);
65
66 // Train backend supports only one layout as NHWC
67 assert(layout == ir::Layout::NHWC);
68 assert(!info.isDynamic());
69
70 // NOTE For now, whether or not to build operands to trainable tensor depends on whether
71 // the corresponding operand is constant.
72 assert(_as_constants[index] == info.isConstant());
73 if (_as_constants[index])
74 {
75 auto tensor = std::make_unique<GradientTensor>(info, layout);
76 _tensor_reg->setGradientTensor(index, std::move(tensor));
77
78 // Initialize tensors for gradient variables
79 for (uint32_t i = 0; i < _optimizer->getVarCount(); ++i)
80 {
81 // TODO Optimize memory
82 auto tensor = std::make_unique<Tensor>(info, layout);
83 tensor->setBuffer(std::make_shared<basic::Allocator>(tensor->total_size()));
84 _tensor_reg->getTrainableTensor(index)->appendOptVar(std::move(tensor));
85 }
86 }
87 else
88 {
89 auto tensor = std::make_unique<BackPropTensor>(info, layout);
90 _tensor_reg->setBackPropTensor(index, std::move(tensor));
91 }
92}
