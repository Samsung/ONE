30bool has_same_shape(const luci::CircleNode *node, const std::vector<uint32_t> &shape)
31{
32 const auto rank = node->rank();
33 CHECK_OR_FALSE(rank == shape.size());
34 for (uint32_t idx = 0; idx < rank; idx++)
35 CHECK_OR_FALSE(node->dim(idx).value() == shape.at(idx));
36 return true;
37}

66bool fused_batch_norm_with_conv(luci::CircleAdd *add)
67{
68 luci::CircleMul *mul = nullptr;
69 luci::CircleConst *shift = nullptr;
70 // If CircleMul is not found or constant operand of CircleAdd is not found,
71 // this pass cannot be applied.
72 CHECK_OR_FALSE(luci::fill(&mul, &shift).with_commutative_args_of(add));
73 // If FusedActivationFunction of mul is not none, this pass cannot be applied.
74 CHECK_OR_FALSE(mul->fusedActivationFunction() == luci::FusedActFunc::NONE);
75
76 luci::CircleConv2D *conv = nullptr;
77 luci::CircleConst *scale = nullptr;
78 // If CircleConv2D is not found or constant operand of CircleMul is not found,
79 // this pass cannot be applied.
80 CHECK_OR_FALSE(luci::fill(&conv, &scale).with_commutative_args_of(mul));
81 // If FusedActivationFunction of conv is not none, this pass cannot be applied.
82 CHECK_OR_FALSE(conv->fusedActivationFunction() == luci::FusedActFunc::NONE);
83
84 luci::CircleConst *filter = dynamic_cast<luci::CircleConst *>(conv->filter());
85 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(conv->bias());
86 // If filter or bias of conv is not const, this pass cannot be applied.
87 CHECK_OR_FALSE(filter != nullptr && bias != nullptr);
88 // TODO Support more data type
89 CHECK_OR_FALSE(filter->dtype() == loco::DataType::FLOAT32);
90
91 // If dtype of filter is different with scale and shift, multiplication may be impossible.
92 CHECK_OR_FALSE(filter->dtype() == scale->dtype());
93 CHECK_OR_FALSE(filter->dtype() == shift->dtype());
94
95 CHECK_OR_FALSE(has_same_shape(shift, {1, 1, 1, filter->dim(0).value()}));
96 CHECK_OR_FALSE(has_same_shape(scale, {1, 1, 1, filter->dim(0).value()}));
97
98 auto name = add->name();
99 assert(name.length() > 0);
100
101 luci::CircleConv2D *fused_conv = add->graph()->nodes()->create<luci::CircleConv2D>();
102 luci::CircleConst *fused_filter = add->graph()->nodes()->create<luci::CircleConst>();
103 luci::CircleConst *fused_bias = add->graph()->nodes()->create<luci::CircleConst>();
104
105 uint32_t filter_out_channel = filter->dim(0).value();
106 uint32_t filter_height = filter->dim(1).value();
107 uint32_t filter_width = filter->dim(2).value();
108 uint32_t filter_in_channel = filter->dim(3).value();
109
110 // Copy filter
111 fused_filter->dtype(filter->dtype());
112 fused_filter->size<loco::DataType::FLOAT32>(filter->size<loco::DataType::FLOAT32>());
113 fused_filter->rank(4);
114 fused_filter->dim(0).set(filter_out_channel);
115 fused_filter->dim(1).set(filter_height);
116 fused_filter->dim(2).set(filter_width);
117 fused_filter->dim(3).set(filter_in_channel);
118 fused_filter->shape_status(luci::ShapeStatus::VALID);
119 fused_filter->name(name + "/Conv2D/filter");
120
121 // Fuse scale to new filter
122 for (uint32_t c = 0; c < filter_out_channel; c++)
123 {
124 for (uint32_t h = 0; h < filter_height; h++)
125 {
126 for (uint32_t w = 0; w < filter_width; w++)
127 {
128 for (uint32_t b = 0; b < filter_in_channel; b++)
129 {
130 uint32_t offset = c * filter_height * filter_width * filter_in_channel +
131 h * filter_width * filter_in_channel + w * filter_in_channel + b;
132 fused_filter->at<loco::DataType::FLOAT32>(offset) =
133 filter->at<loco::DataType::FLOAT32>(offset) * scale->at<loco::DataType::FLOAT32>(c);
134 }
135 }
136 }
137 }
138
139 // Copy bias
140 assert(bias->rank() == 1);
141 assert(bias->dim(0).value() == filter_out_channel);
142 fused_bias->dtype(bias->dtype());
143 fused_bias->size<loco::DataType::FLOAT32>(bias->size<loco::DataType::FLOAT32>());
144 fused_bias->rank(1);
145 fused_bias->dim(0).set(filter_out_channel);
146 fused_bias->shape_status(luci::ShapeStatus::VALID);
147 fused_bias->name(name + "/Conv2D/bias");
148
149 // Fuse scale and shift to bias
150 for (uint32_t b = 0; b < filter_out_channel; ++b)
151 {
152 fused_bias->at<loco::DataType::FLOAT32>(b) =
153 bias->at<loco::DataType::FLOAT32>(b) * scale->at<loco::DataType::FLOAT32>(b) +
154 shift->at<loco::DataType::FLOAT32>(b);
155 }
156
157 // Set attributes of fused_conv
158 fused_conv->input(conv->input());
159 fused_conv->filter(fused_filter);
160 fused_conv->bias(fused_bias);
161 fused_conv->fusedActivationFunction(add->fusedActivationFunction());
162 fused_conv->padding(conv->padding());
163 fused_conv->stride()->h(conv->stride()->h());
164 fused_conv->stride()->w(conv->stride()->w());
165 fused_conv->dilation()->h(conv->dilation()->h());
166 fused_conv->dilation()->w(conv->dilation()->w());
167 fused_conv->name(name + "/Conv2D");
168 luci::add_origin(fused_conv, luci::composite_origin({luci::get_origin(add), luci::get_origin(mul),
169 luci::get_origin(conv)}));
170
171 replace(add).with(fused_conv);
172
173 return true;
174}

181bool FuseBatchNormWithConvPass::run(loco::Graph *g)
182{
183 bool changed = false;
184 for (auto node : loco::active_nodes(loco::output_nodes(g)))
185 {
186 if (auto add = dynamic_cast<luci::CircleAdd *>(node))
187 {
188 if (fused_batch_norm_with_conv(add))
189 changed = true;
190 }
191 }
192
193 return changed;
194}
