51bool fused_batch_norm_with_conv(luci::CircleAdd *add)
52{
53 luci::CircleMul *mul = nullptr;
54 luci::CircleConst *shift = nullptr;
55 if (auto add_lhs = dynamic_cast<luci::CircleMul *>(add->x()))
56 {
57 mul = add_lhs;
58 shift = dynamic_cast<luci::CircleConst *>(add->y());
59 }
60 else if (auto add_rhs = dynamic_cast<luci::CircleMul *>(add->y()))
61 {
62 mul = add_rhs;
63 shift = dynamic_cast<luci::CircleConst *>(add->x());
64 }
65
66 // If CircleMul is not found or constant operand of CircleAdd is not found,
67 // this pass cannot be applied.
68 if (mul == nullptr || shift == nullptr)
69 return false;
70
71 // If FusedActivationFunction of mul is not none, this pass cannot be applied.
72 if (mul->fusedActivationFunction() != luci::FusedActFunc::NONE)
73 return false;
74
75 // To apply this pass, shape of shift should be [1, 1, 1, out_channel].
76 if (shift->rank() != 4)
77 return false;
78 for (uint32_t i = 0; i < 3; ++i)
79 if (shift->dim(i).value() != 1)
80 return false;
81
82 luci::CircleConv2D *conv = nullptr;
83 luci::CircleConst *scale = nullptr;
84 if (auto mul_lhs = dynamic_cast<luci::CircleConv2D *>(mul->x()))
85 {
86 conv = mul_lhs;
87 scale = dynamic_cast<luci::CircleConst *>(mul->y());
88 }
89 else if (auto mul_rhs = dynamic_cast<luci::CircleConv2D *>(mul->y()))
90 {
91 conv = mul_rhs;
92 scale = dynamic_cast<luci::CircleConst *>(mul->x());
93 }
94
95 // If CircleConv2D is not found or constant operand of CircleMul is not found,
96 // this pass cannot be applied.
97 if (conv == nullptr || scale == nullptr)
98 return false;
99
100 // To apply this pass, shape of scale should be [1, 1, 1, out_channel].
101 if (scale->rank() != 4)
102 return false;
103 for (uint32_t i = 0; i < 3; ++i)
104 if (scale->dim(i).value() != 1)
105 return false;
106
107 // If FusedActivationFunction of conv is not none, this pass cannot be applied.
108 if (conv->fusedActivationFunction() != luci::FusedActFunc::NONE)
109 return false;
110
111 luci::CircleConst *filter = dynamic_cast<luci::CircleConst *>(conv->filter());
112 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(conv->bias());
113
114 // If filter or bias of conv is not const, this pass cannot be applied.
115 if (filter == nullptr || bias == nullptr)
116 return false;
117
118 // If dtype of filter is different with scale and shift, multiplication may be impossible.
119 if (filter->dtype() != scale->dtype())
120 return false;
121 if (filter->dtype() != shift->dtype())
122 return false;
123
124 // TODO Support more data type
125 if (filter->dtype() != loco::DataType::FLOAT32)
126 return false;
127
128 // Output channel dimension should be same. If not, this pass cannot be applied.
129 if (filter->dim(0).value() != scale->dim(3).value())
130 return false;
131 if (filter->dim(0).value() != shift->dim(3).value())
132 return false;
133
134 auto name = add->name();
135 assert(name.length() > 0);
136
137 luci::CircleConv2D *fused_conv = add->graph()->nodes()->create<luci::CircleConv2D>();
138 luci::CircleConst *fused_filter = add->graph()->nodes()->create<luci::CircleConst>();
139 luci::CircleConst *fused_bias = add->graph()->nodes()->create<luci::CircleConst>();
140
141 uint32_t filter_out_channel = filter->dim(0).value();
142 uint32_t filter_height = filter->dim(1).value();
143 uint32_t filter_width = filter->dim(2).value();
144 uint32_t filter_in_channel = filter->dim(3).value();
145
146 // Copy filter
147 fused_filter->dtype(filter->dtype());
148 fused_filter->size<loco::DataType::FLOAT32>(filter->size<loco::DataType::FLOAT32>());
149 fused_filter->rank(4);
150 fused_filter->dim(0).set(filter_out_channel);
151 fused_filter->dim(1).set(filter_height);
152 fused_filter->dim(2).set(filter_width);
153 fused_filter->dim(3).set(filter_in_channel);
154 fused_filter->shape_status(luci::ShapeStatus::VALID);
155 fused_filter->name(name + "/Conv2D/filter");
156
157 // Fuse scale to new filter
158 for (uint32_t c = 0; c < filter_out_channel; c++)
159 {
160 for (uint32_t h = 0; h < filter_height; h++)
161 {
162 for (uint32_t w = 0; w < filter_width; w++)
163 {
164 for (uint32_t b = 0; b < filter_in_channel; b++)
165 {
166 uint32_t offset = c * filter_height * filter_width * filter_in_channel +
167 h * filter_width * filter_in_channel + w * filter_in_channel + b;
168 fused_filter->at<loco::DataType::FLOAT32>(offset) =
169 filter->at<loco::DataType::FLOAT32>(offset) * scale->at<loco::DataType::FLOAT32>(c);
170 }
171 }
172 }
173 }
174
175 // Copy bias
176 assert(bias->rank() == 1);
177 assert(bias->dim(0).value() == filter_out_channel);
178 fused_bias->dtype(bias->dtype());
179 fused_bias->size<loco::DataType::FLOAT32>(bias->size<loco::DataType::FLOAT32>());
180 fused_bias->rank(1);
181 fused_bias->dim(0).set(filter_out_channel);
182 fused_bias->shape_status(luci::ShapeStatus::VALID);
183 fused_bias->name(name + "/Conv2D/bias");
184
185 // Fuse scale and shift to bias
186 for (uint32_t b = 0; b < filter_out_channel; ++b)
187 {
188 fused_bias->at<loco::DataType::FLOAT32>(b) =
189 bias->at<loco::DataType::FLOAT32>(b) * scale->at<loco::DataType::FLOAT32>(b) +
190 shift->at<loco::DataType::FLOAT32>(b);
191 }
192
193 // Set attributes of fused_conv
194 fused_conv->input(conv->input());
195 fused_conv->filter(fused_filter);
196 fused_conv->bias(fused_bias);
197 fused_conv->fusedActivationFunction(add->fusedActivationFunction());
198 fused_conv->padding(conv->padding());
199 fused_conv->stride()->h(conv->stride()->h());
200 fused_conv->stride()->w(conv->stride()->w());
201 fused_conv->dilation()->h(conv->dilation()->h());
202 fused_conv->dilation()->w(conv->dilation()->w());
203 fused_conv->name(name + "/Conv2D");
204 luci::add_origin(fused_conv, luci::composite_origin({luci::get_origin(add), luci::get_origin(mul),
205 luci::get_origin(conv)}));
206
207 replace(add).with(fused_conv);
208
209 return true;
210}

217bool FuseBatchNormWithConvPass::run(loco::Graph *g)
218{
219 bool changed = false;
220 for (auto node : loco::active_nodes(loco::output_nodes(g)))
221 {
222 if (auto add = dynamic_cast<luci::CircleAdd *>(node))
223 {
224 if (fused_batch_norm_with_conv(add))
225 changed = true;
226 }
227 }
228
229 return changed;
230}
