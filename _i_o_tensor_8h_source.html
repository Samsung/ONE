68 /*
69 * Changes tensor shape and allocate memory since its shape was changed
70 * perhaps by nnfw_set_input_tensorinfo()
71 *
72 * Cases are:
73 * 1) static operand -> nnfw_set_input_tensorinfo() -> execute() -> execute()
74 * (a) (b)
75 *
76 * at (a), operand is static, tensor is static - memory dealloc is not needed
77 * (DynamicTensorManager cannot dealloc memory allocated by StaticTensorManager)
78 * at (b), operand is static, tensor is dynamic - memory dealloc is needed
79 *
80 * 2) dynamic operand -> nnfw_set_input_tensorinfo() -> execute() -> execute()
81 * (a) (b)
82 *
83 * at (a), operand is dynamic, tensor is dynamic - memory dealloc is not needed
84 * since it has not been allocated yet
85 * at (b), operand is dynamic, tensor is dynamic - memory dealloc is needed
86 */
87 bool applyShape(const ir::Shape &shape) override
88 {
89 auto return_val = _tensor->applyShape(shape);
90 if (return_val)
91 {
92 _info.shape(shape);
93 _info.setDynamic();
94 }
95 return return_val;
96 }
97
101 bool hasBackendTensor() const { return _has_backend_tensor; }
102
107 void syncInfoFromBackendTensor()
108 {
109 assert(_tensor != nullptr);
110 assert(_has_backend_tensor);
111 if (_tensor->is_dynamic())
112 {
113 _info = _tensor->get_info();
114 }
115 }
116
117private:
118 IPortableTensor *_tensor{nullptr}; //< The actual tensor that is indirected
119 // "_orig" has UserTensor type original tensor's info with nullptr buffer,
120 // and "_tensor" points to "_user_tensor".
121 // After 1st setTensor(tensor) call, "_tensor" is updated to actual tensor
122 std::unique_ptr<UserTensor> _orig; //< If it is a user tensor, it is managed by this object
123 bool _has_backend_tensor;
124};
