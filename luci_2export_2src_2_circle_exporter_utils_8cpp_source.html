211uint32_t SerializedModelData::registerBuiltinOpcode(circle::BuiltinOperator builtin_code,
212 const std::string &custom_code,
213 const int32_t op_version)
214{
215 assert(op_version > 0);
216
217 auto it = _operator_codes.find(OpCode{builtin_code, custom_code, op_version});
218 if (it != _operator_codes.end())
219 {
220 return it->second;
221 }
222 auto idx = static_cast<uint32_t>(_operator_codes.size());
223 _operator_codes.emplace(OpCode{builtin_code, custom_code, op_version}, idx);
224 return idx;
225}

227circle::Padding getOpPadding(const loco::Padding2D *pad, const loco::Stride<2> *stride,
228 const ShapeDescription &ifm, const ShapeDescription &ofm)
229{
230 // VALID padding
231 if (pad->top() == 0 && pad->bottom() == 0 && pad->left() == 0 && pad->right() == 0)
232 return circle::Padding_VALID;
233
234 // SAME padding
235 //
236 // For same padding, by definition, following equation should hold:
237 // O = floor((I - 1) / S) + 1
238 // where input size I, output size O, stride S
239 //
240 // NOTE input and output 'feature' map are shape of NHWC
241 bool same_padding_criterion_1 =
242 (static_cast<uint32_t>(ofm._dims[1]) == (ifm._dims[1] - 1) / stride->vertical() + 1) &&
243 (static_cast<uint32_t>(ofm._dims[2]) == (ifm._dims[2] - 1) / stride->horizontal() + 1);
244
245 // For same padding, rear padding is same or bigger than front padding by at most 1
246 bool same_padding_criterion_2 =
247 (pad->top() <= pad->bottom()) && (pad->bottom() <= pad->top() + 1) &&
248 (pad->left() <= pad->right()) && (pad->right() <= pad->left() + 1);
249
250 if (same_padding_criterion_1 && same_padding_criterion_2)
251 return circle::Padding_SAME;
252
253 INTERNAL_EXN("Unsupported padding criteria");
254}

256circle::Padding getOpPadding(const luci::Padding pad)
257{
258 if (pad == luci::Padding::VALID)
259 return circle::Padding_VALID;
260 if (pad == luci::Padding::SAME)
261 return circle::Padding_SAME;
262
263 INTERNAL_EXN_V("Unsupported luci::Padding", oops::to_uint32(pad));
264}
