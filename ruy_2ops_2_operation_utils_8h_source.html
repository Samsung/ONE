42inline nnfw::ruy::Shape getTensorShape(const IPortableTensor *tensor)
43{
44 if (tensor == nullptr)
45 return nnfw::ruy::Shape();
46
47 const ir::Shape &shape = tensor->get_info().shape();
48
49 assert(tensor->layout() == ir::Layout::NHWC);
50
51 auto rank = shape.rank();
52 nnfw::ruy::Shape ret(rank);
53 auto data = ret.DimsData();
54 for (int i = 0; i < rank; ++i)
55 {
56 data[i] = shape.dim(i);
57 }
58 return ret;
59}

61inline nnfw::ruy::FusedActivationFunctionType convertActivationType(const ir::Activation activation)
62{
63 switch (activation)
64 {
65 case ir::Activation::NONE:
66 return nnfw::ruy::FusedActivationFunctionType::kNone;
67 case ir::Activation::RELU:
68 return nnfw::ruy::FusedActivationFunctionType::kRelu;
69 case ir::Activation::RELU1:
70 return nnfw::ruy::FusedActivationFunctionType::kRelu1;
71 case ir::Activation::RELU6:
72 return nnfw::ruy::FusedActivationFunctionType::kRelu6;
73 case ir::Activation::TANH:
74 return nnfw::ruy::FusedActivationFunctionType::kTanh;
75 case ir::Activation::SIGMOID:
76 return nnfw::ruy::FusedActivationFunctionType::kSigmoid;
77 default:
78 throw std::runtime_error{"RUY backend: Cannot convert activation type"};
79 }
80}
