25void TrainableConstantInsertionPass::callback(const ir::OperationIndex &node_index,
26 ir::IOperation &node)
27{
28 for (const auto &input : node.getUsedInputSet())
29 {
30 auto &object = _graph.operands().at(input);
31
32 // Skip if the operand is not constant or not shared constant
33 if (!object.isConstant() || object.getUses().size() < 2)
34 continue;
35
36 // Insert new operands for shared constant except for the current node.
37 const auto uses(object.getUses());
38 for (const auto &use_index : uses)
39 {
40 if (use_index == node_index)
41 continue;
42
43 // NOTE The PermuteFactor(backend) of the current node and the use node may be different.
44 // But there is no problem because both nodes' constant operand will have
45 // only one use node.
46 const auto new_index = insertNewOperand(object);
47 updateUseDef(input, new_index, use_index);
48 }
49
50 // The input of the current node will have one use as the current node
51 assert(object.getUses().size() == 1 && object.getUses().contains(node_index));
52 }
53}

55ir::OperandIndex TrainableConstantInsertionPass::insertNewOperand(const ir::Operand &object)
56{
57 ir::Operand new_object(object);
58 new_object.clearDefUse();
59 return _graph.operands().emplace(new_object);
60}

62void TrainableConstantInsertionPass::updateUseDef(const ir::OperandIndex &old_index,
63 const ir::OperandIndex &new_index,
64 const ir::OperationIndex &node_index)
65{
66 const auto backend = _lowered_graph.lower_info().operation.at(node_index);
67
68 // Update the same inputs of a node at once because inputs of an operation have the same
69 // PermuteFactor
70 auto &target_node = _graph.operations().at(node_index);
71 target_node.replaceInputs(old_index, new_index);
72
73 // Update the new operand
74 auto &new_object = _graph.operands().at(new_index);
75 new_object.insertUse(node_index);
76
77 VERBOSE(TrainConstInsertPass) << "New operand " << new_index << " added(copy of " << old_index
78 << ") for " << backend->config()->id() << std::endl;
79
80 // Remove the use node from uses of origin operand
81 // Constant operand has no def.
82 auto &old_object = _graph.operands().at(old_index);
83 assert(!old_object.getDef().valid());
84 old_object.removeUse(node_index);
85}
