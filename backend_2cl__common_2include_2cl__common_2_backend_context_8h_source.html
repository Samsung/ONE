49 FunctionMap genKernels() override
50 {
51 FunctionMap ret;
52
53 // kernel_gen
54 for (auto op_ind : _data.op_order)
55 {
56 auto fn_seq = kernel_gen->generate(op_ind);
57 ret.emplace_back(op_ind, std::move(fn_seq));
58 }
59
60 tensor_builder->allocate();
61 initConsts();
62
63 // NOTE For memory optimization, we want to free some operand data
64 const_cast<ir::Graph &>(*_data.graph)
65 .operands()
66 .iterate([&](const ir::OperandIndex &, ir::Operand &obj) { obj.releaseData(); });
67
68 for (auto &it : ret)
69 {
70 auto &fn_seq = it.second;
71 fn_seq->iterate([&](exec::IFunction &ifunc) {
72 ifunc.prepare();
73 tensor_builder->postFunctionPrepare();
74 });
75 }
76
77 return ret;
78 }
79
80protected:
81 void initConsts()
82 {
83 _data.graph->operations().iterate([&](const ir::OperationIndex &, const ir::Operation &op) {
84 constant_initializer->setLayout(graph()->layout());
85 op.accept(*constant_initializer);
86 });
87
88 _data.graph->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &operand) {
89 if (_data.external_operands.contains(ind) || !operand.isConstant())
90 return;
91 const auto &obj = graph()->operands().at(ind);
92 if (obj.isConstant() && !constant_initializer->exist(ind))
93 {
94 constant_initializer->registerDefaultInitializer(ind, obj);
95 }
96 });
97
98 constant_initializer->run();
99 }

104 void planTensors()
105 {
106 ir::OperandIndexMap<uint32_t> uses_map;
107 ir::OperandIndexMap<uint32_t> def_map;
108 ir::OperandIndexSequence constants;
109
110 // Prepare scanning
111 _data.graph->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
112 if (_data.external_operands.contains(ind))
113 return;
114
115 uses_map[ind] = obj.getUses().size();
116 def_map[ind] = obj.getDef().valid() ? 1 : 0;
117
118 if (obj.isConstant())
119 constants.append(ind);
120
121 if (!tensor_builder->isRegistered(ind))
122 {
123 // These tensors do not exist in any operation (No use and def)
124 const auto info = obj.info();
125 const auto layout = _data.operand_layouts.at(ind);
126 // TODO Change tensor info to have permuted shape
127 registerTensorInfo(ind, info, layout);
128 }
129 });
130
131 // Start scanning to do notify{First|Last}Use for each tensor
132
133 // If a tensor is a constant, increase the use of the tensor and allocate it first.
134 // Increasing use count here makes the tensor never be deallocated, i.e it they will be
135 // deallocated last.
136 VERBOSE(planTensors) << "TENSORS as CONSTANT" << std::endl;
137 for (const auto &ind : constants)
138 {
139 uses_map[ind]++;
140 tensor_builder->notifyFirstUse(ind);
141 }
142
143 // At each operation,
144 // 1. Scan DEF of outputs. If the DEF, allocate it
145 // 2. Scan DEF of inputs. If variable tensor, allocate it
146 // 3. Scan USE of inputs. Decrease the USE and deallocate if the USE is 0
147 for (const auto op_ind : _data.op_order)
148 {
149 const auto &op = graph()->operations().at(op_ind);
150 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
151 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
152
153 // Define outputs
154 for (const auto &ind : op_outputs)
155 {
156 if (!tensor_builder->isRegistered(ind))
157 continue;
158 assert(def_map.find(ind) != def_map.end());
159 if (def_map[ind])
160 {
161 def_map[ind] = 0;
162 tensor_builder->notifyFirstUse(ind);
163 }
164 }
165
166 // Scan variable tensors
167 // This tensor has features like constant. But OperandInfo and LowerInfo treat them as
168 // non-constant because of less memory usage by memory planning in here
169 for (const auto &ind : op_inputs)
170 {
171 if (!tensor_builder->isRegistered(ind))
172 continue;
173 const auto &operand = graph()->operands().at(ind);
174 if (operand.info().isVariable())
175 {
176 // The variable tensor with buffer is not supported yet
177 assert(operand.data() == nullptr);
178 assert(operand.getUses().size() == 1 && !operand.getDef().valid());
179 assert(uses_map[ind] == 1 && def_map[ind] == 0);
180 tensor_builder->notifyFirstUse(ind);
181 }
182 }
183
184 for (const auto &ind : op_inputs)
185 {
186 if (!tensor_builder->isRegistered(ind))
187 continue;
188 assert(uses_map.find(ind) != uses_map.end());
189 assert(uses_map[ind] > 0);
190 uses_map[ind]--;
191 if (uses_map[ind] == 0)
192 {
193 // plan for deallocation of static tensornode
194 tensor_builder->notifyLastUse(ind);
195 }
196 }
197 }
198
199 _data.graph->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &) {
200 if (uses_map[ind] == 0)
201 {
202 tensor_builder->notifyLastUse(ind);
203 }
204 });
205
206 // Dispose and validate
207 for (const auto &ind : constants)
208 {
209 --uses_map[ind];
210 if (uses_map[ind] == 0) // To prevent notifyLastUse from being called twice
211 {
212 tensor_builder->notifyLastUse(ind);
213 }
214 }
215
216 assert(
217 std::all_of(uses_map.begin(), uses_map.end(),
218 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
219
220 assert(
221 std::all_of(def_map.begin(), def_map.end(),
222 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
223 }
224
225public:
226 // TODO Make it protected
227 std::shared_ptr<T_TensorBuilder> tensor_builder;
228 std::shared_ptr<T_ConstantInitializer> constant_initializer;
229 std::shared_ptr<T_KernelGenerator> kernel_gen;
230};
