27bool is_input_of(const luci::PGroup *pgroup_i, const luci::PGroup *pgroup)
28{
29 for (auto *output : pgroup_i->outputs)
30 {
31 for (auto *input : pgroup->inputs)

44bool is_input_same(const luci::PGroup *pgroup, const luci::PGroups *pgroups)
45{
46 assert(pgroups != nullptr);
47 assert(pgroup != nullptr);
48
49 const luci::PGroup *input_pgroup = nullptr;
50 std::string group;
51 for (auto &input : pgroup->inputs)
52 {
53 // We ignore below logic for CircleConst.
54 // CircleConst will be cloned if they are not found in pgroup as an input.
55 // Refer build_graph(), "add CircleConst for inputs"
56 // Reason: CircleConst can be shared as input to multiple nodes
57 // where each node can be placed in different groups. For this case
58 // we need to clone this CircleConst for each graph of the group.
59 if (dynamic_cast<const luci::CircleConst *>(input) != nullptr)
60 continue;
61
62 auto input_group = pgroups->group_of(input);
63 // NOTE: all the nodes should be registered and return should be valid group.
64 // produce_pgroups() should ensure this, except CircleConst, Input, Outputs.
65 // assert here to find if there is any problem with this.
66 assert(not input_group.empty());
67 if (input_group.empty())
68 input_group = pgroups->default_group;
69
70 if (group.empty())
71 group = input_group;
72 else
73 {
74 if (group != input_group)
75 return false;
76 }
77 // if there are multiple inputs, all the inputs should be in same pgroup
78 // https://github.com/Samsung/ONE/issues/6230#issuecomment-801618150
79 // https://github.com/Samsung/ONE/issues/6230#issuecomment-801680531
80 auto pgroup_input = pgroups->pgroup_of(input);
81 if (pgroup_input != nullptr)
82 {
83 if (input_pgroup == nullptr)
84 input_pgroup = pgroup_input;
85 else
86 {
87 if (input_pgroup->group != pgroup_input->group)
88 return false;
89 }
90 }
91 }
92 return true;
93}

105bool is_output_same(const luci::PGroup *pgroup, const luci::PGroups *pgroups)
106{
107 assert(pgroups != nullptr);
108 assert(pgroup != nullptr);
109
110 std::string group;
111 for (auto &output : pgroup->outputs)
112 {
113 // get output_group
114 auto output_group = pgroups->group_of(output);
115 assert(not output_group.empty());
116 if (output_group.empty())
117 output_group = pgroups->default_group;
118
119 // find all PGroup that uses output
120 for (auto &pgroup_user : pgroups->pgroups)
121 {
122 for (auto &user_inputs : pgroup_user->inputs)
123 {
124 if (output == user_inputs)
125 {
126 // OK, these are connected, check group is same
127 if (pgroup_user->group != output_group)
128 return false;
129 }
130 }
131 }
132 }
133
134 return true;
135}

141void merge_into(luci::PGroup *pgroup, luci::PGroup *pgroup_i)
142{
143 for (auto &pnode : pgroup->pnodes)
144 {
145 // update pgroup for this pnode
146 pnode->pgroup = pgroup_i;
147 assert(pnode->group == pgroup_i->group);
148
149 // we don't need to add this in topological order:
150 // all the nodes will be created first then connection will be held
151 pgroup_i->pnodes.push_back(std::move(pnode));
152 // note: pnode is now nullptr as it's moved into pgroup_i->pnodes
153 }
154
155 for (auto &input : pgroup->inputs)
156 {
157 // add inputs of pgroup to pgroup_i if not member of pgroup_i
158 bool found_in_pgroup_i = false;
159 for (auto &pnode : pgroup_i->pnodes)
160 {
161 if (input == pnode->node)
162 {
163 found_in_pgroup_i = true;
164 break;
165 }
166 }
167 // skip if this input is already in the inputs
168 auto fit = std::find(pgroup_i->inputs.begin(), pgroup_i->inputs.end(), input);
169 if (fit != pgroup_i->inputs.end())
170 {
171 found_in_pgroup_i = true;
172 }
173 // note: if we force found_in_pgroup_i to false, for testing there will be
174 // unnecessary inputs
175 if (not found_in_pgroup_i)
176 {
177 // node input maybe in another pgroup
178 pgroup_i->inputs.push_back(input);
179 }
180 }
181 // add outputs of pgroup to pgroup_i outputs if not exist
182 for (auto &output : pgroup->outputs)
183 {
184 auto it = std::find(pgroup_i->outputs.begin(), pgroup_i->outputs.end(), output);
185 if (it == pgroup_i->outputs.end())
186 {
187 pgroup_i->outputs.push_back(output);
188 }
189 }
190}

200std::unique_ptr<luci::PGroups> merge_pgroups(const luci::PGroups *s_pgroups)
201{
202 // Make a copy of pgroups to apply merge action
203 // Q) do we really need a copy?
204 auto d_pgroups = s_pgroups->make_copy();
205
206 // Merge partition graphs
207 // - This is initial implementation that works for limited networks
208 // - if A and B is same group -> if A is input of B -> ... -> merge B into A
209 auto &pgroups = d_pgroups->pgroups;
210 bool changed;
211 do
212 {
213 changed = false;
214 for (auto &pgroup_i : pgroups)
215 {
216 bool merged = false;
217 for (auto it = pgroups.begin(); it != pgroups.end(); ++it)
218 {
219 auto &pgroup = *it;
220
221 // skip if same object
222 if (pgroup->id == pgroup_i->id)
223 continue;
224 // skip if different group
225 if (pgroup->group != pgroup_i->group)
226 continue;
227 // skip if not connected
228 if (!is_input_of(pgroup_i.get(), pgroup.get()))
229 continue;
230 // skip if there are multiple inputs but inputs differ in group
231 if (!is_input_same(pgroup.get(), d_pgroups.get()))
232 continue;
233 // skip if pgroup has different group for other users of pgroup_i
234 if (!is_output_same(pgroup_i.get(), d_pgroups.get()))
235 continue;
236 // TODO add more condition may be needed
237
238 merge_into(pgroup.get(), pgroup_i.get());
239
240 auto eit = d_pgroups->id2pgroup.find(pgroup->id);
241 assert(eit != d_pgroups->id2pgroup.end());
242 d_pgroups->id2pgroup.erase(eit);
243
244 // remove merged pgroup from pgroups
245 pgroups.erase(it);
246
247 merged = true;
248 break;
249 }
250 if (merged)
251 {
252 changed = true;
253 break;
254 }
255 }
256 } while (changed);
257
258 return d_pgroups;
259}
