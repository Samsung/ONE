110TEST(acl_backend_dom_to_text, ArtifactFunctionCall)
111{
112 const char *lit_data = "123";
113 const char *id_data = "some_id";
114 shared_ptr<ArtifactExpr> id = AF::id(id_data);
115 shared_ptr<ArtifactExpr> lit = AF::lit(lit_data);
116 const list<shared_ptr<ArtifactExpr>> args{id, lit};
117
118 shared_ptr<ArtifactId> obj = AF::id("obj");
119
120 using TestCase = tuple<ArtifactCallType, shared_ptr<ArtifactExpr>, const char *>;
121 TestCase test_cases[] = {TestCase{ArtifactCallType::scope, nullptr, "foo(some_id, 123)"},
122 TestCase{ArtifactCallType::obj, obj, "obj.foo(some_id, 123)"},
123 TestCase{ArtifactCallType::ref, obj, "obj->foo(some_id, 123)"},
124 TestCase{ArtifactCallType::scope, obj, "obj::foo(some_id, 123)"}};
125
126 for (const auto &test : test_cases)
127 {
128 ArtifactCallType call_type = get<0>(test);
129 shared_ptr<ArtifactExpr> obj = get<1>(test);
130 const char *ref_output = get<2>(test);
131 checkCall(call_type, "foo", args, obj, ref_output);
132 }
133}

280TEST(acl_backend_dom_to_text, ArtifactForLoop)
281{
282 stringstream code_out;
283 ArtifactGeneratorCppCode code_gen(code_out);
284 const char *var_name = "i";
285 const char *var_type = "int";
286
287 shared_ptr<ArtifactVariable> iter = AF::var(var_type, var_name, {}, {AF::lit("0")});
288 shared_ptr<ArtifactExpr> step =
289 AF::bin(ArtifactBinOp::plusAssign, AF::id(var_name), AF::lit("1"));
290 shared_ptr<ArtifactExpr> cond =
291 AF::bin(ArtifactBinOp::lessOrEq, AF::id(var_name), AF::lit("123"));
292
293 shared_ptr<ArtifactBinaryExpr> expr =
294 AF::bin(ArtifactBinOp::plusAssign, AF::id("hello"), AF::id("world"));
295
296 ArtifactForLoop loop(iter, cond, step);
297
298 loop.getBlock()->addStatement(expr);
299
300 loop.accept(&code_gen);
301 ASSERT_EQ(code_out.str(), "for(int i(0); i <= 123; i += 1) {\n hello += world;\n}\n");
302}
