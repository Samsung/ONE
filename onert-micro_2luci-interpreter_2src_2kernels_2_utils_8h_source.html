41inline int32_t computePadding(int32_t stride, int32_t dilation_rate, int32_t in_size,
42 int32_t filter_size, int32_t out_size)
43{
44 const int32_t effective_filter_size = (filter_size - 1) * dilation_rate + 1;
45 const int32_t padding = ((out_size - 1) * stride + effective_filter_size - in_size) / 2;
46 return padding > 0 ? padding : 0;
47}

49inline int32_t computePaddingWithOffset(int32_t stride, int32_t dilation_rate, int32_t in_size,
50 int32_t filter_size, int32_t out_size, int32_t *offset)
51{
52 int32_t effective_filter_size = (filter_size - 1) * dilation_rate + 1;
53 int32_t total_padding = ((out_size - 1) * stride + effective_filter_size - in_size);
54 total_padding = total_padding > 0 ? total_padding : 0;
55 *offset = total_padding % 2;
56 return total_padding / 2;
57}

59inline int32_t computeOutputSize(Padding padding, int32_t image_size, int32_t filter_size,
60 int32_t stride, int32_t dilation_rate = 1)
61{
62 const int32_t effective_filter_size = (filter_size - 1) * dilation_rate + 1;
63 switch (padding)
64 {
65 case Padding::SAME:
66 return (image_size + stride - 1) / stride;
67 case Padding::VALID:
68 return (image_size + stride - effective_filter_size) / stride;
69 default:
70 assert(false);
71 return 0;
72 }
73}

103template <typename T>
104void fillArithmeticActivationRange(tflite::ArithmeticParams &p, Activation act)
105{
106 static_assert(one_of_types<T, float, int32_t, int64_t>(), "Unsupported dtype");
107
108 if (std::is_same<T, float>::value)
109 calculateActivationRange(act, &p.float_activation_min, &p.float_activation_max);
110 if (std::is_same<T, int32_t>::value)

112 else
113 calculateActivationRange(act, &p.int64_activation_min, &p.int64_activation_max);
114}

117// significand, and shift representation of its exponent.
118//

156inline double getQuantizedConvolutionMultipler(float input_scale, float filter_scale,
157 float output_scale)
158{
159 const double input_product_scale = static_cast<double>(input_scale * filter_scale);
160 LUCI_INTERPRETER_CHECK(input_product_scale >= 0);
161 return input_product_scale / static_cast<double>(output_scale);
162}

164// TODO rename getQuantizedConvolutionMultiplers to something more general

166inline std::vector<double> getQuantizedConvolutionMultiplers(float input_scale,
167 const std::vector<float> &filter_scale,
168 float output_scale)
169{
170 std::vector<double> effective_output_scales;
171 size_t n = filter_scale.size();
172 effective_output_scales.reserve(n);
173 for (size_t i = 0; i < n; ++i)
174 {
175 effective_output_scales.push_back(
176 getQuantizedConvolutionMultipler(input_scale, filter_scale[i], output_scale));
177 }
178 return effective_output_scales;
179}

188inline std::vector<ChannelQuantMultipliers>
189quantizeMultipliers(const std::vector<double> &effective_scale)
190{
191 size_t n = effective_scale.size();
192 std::vector<ChannelQuantMultipliers> params(n);
193 for (size_t i = 0; i < n; ++i)

213inline tflite::RuntimeShape getTensorShape(const Tensor *tensor)
214{
215 if (tensor == nullptr)
216 return tflite::RuntimeShape();
217
218 const Shape &shape = tensor->shape();
219 tflite::RuntimeShape runtime_shape(shape.num_dims());
220 for (int i = 0; i < shape.num_dims(); ++i)
221 {
222 runtime_shape.SetDim(i, shape.dim(i));
223 }
224 return runtime_shape;
225}

237// A list of tensors in a format that can be used by kernels like split and
238// concatenation.
239template <typename T, bool is_const> class VectorOfTensors
240{
241public:
242 using ElementT = typename std::conditional<is_const, const T, T>::type;
243 using TensorT = typename std::conditional<is_const, const Tensor, Tensor>::type;
244
245 // Build with the tensors in 'tensor_list'.
246 explicit VectorOfTensors(const std::vector<TensorT *> &tensor_list)
247 {
248 const int num_tensors = tensor_list.size();
249
250 all_data_.reserve(num_tensors);
251 all_shape_.reserve(num_tensors);
252 all_shape_ptr_.reserve(num_tensors);
253
254 for (TensorT *tensor : tensor_list)
255 {
256 all_data_.push_back(getTensorData<T>(tensor));
257 all_shape_.push_back(getTensorShape(tensor));
258 }
259
260 // Taking the pointer from inside a std::vector is only OK if the vector is
261 // never modified, so we populate all_shape in the previous loop and then we
262 // are free to grab iterators here.
263 for (tflite::RuntimeShape &shape : all_shape_)
264 {
265 all_shape_ptr_.push_back(&shape);
266 }
267 }
268 // Return a pointer to the data pointers of all tensors in the list. For
269 // example:
270 // float* const* f = v.data();
271 // f[0][1] is the second element of the first tensor.
272 ElementT *const *data() const { return all_data_.data(); }
273
274 // Return a pointer the shape pointers of all tensors in the list. For
275 // example:
276 // const RuntimeShape* const* d = v.dims();
277 // dims[1] are the dimensions of the second tensor in the list.
278 const tflite::RuntimeShape *const *shapes() const { return all_shape_ptr_.data(); }
279
280private:
281 std::vector<ElementT *> all_data_;
282 std::vector<tflite::RuntimeShape> all_shape_;
283 std::vector<tflite::RuntimeShape *> all_shape_ptr_;
284};

286// A list of quantized tensors in a format that can be used by kernels like
287// split and concatenation.
288template <bool is_const> class VectorOfQuantizedTensors : public VectorOfTensors<uint8_t, is_const>
289{
290public:

292
293 // Build with the tensors in 'tensor_list'.
294 explicit VectorOfQuantizedTensors(const std::vector<TensorT *> &tensor_list)
295 : VectorOfTensors<uint8_t, is_const>(tensor_list)
296 {
297 for (TensorT *tensor : tensor_list)
298 {
299 zero_point_.push_back(tensor->zero_point());
300 scale_.push_back(tensor->scale());
301 }
302 }
303
304 const float *scale() const { return scale_.data(); }
305 const int32_t *zero_point() const { return zero_point_.data(); }
306
307private:
308 std::vector<int32_t> zero_point_;
309 std::vector<float> scale_;
310};
