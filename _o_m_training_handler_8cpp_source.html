34/*
35 * Calculate backpropagation error between calculated data in forward graph and target data
36 */
37OMStatus OMTrainingHandler::handleError(const OMConfig &config, OMRuntimeStorage &forward_storage,
38 OMRuntimeStorage &backward_storage,
39 OMRuntimeContext &context, uint32_t batch_num)
40{
41 auto forward_outputs = context.getCircleOutputs();
42 // Go over all outputs
43 for (uint32_t i = 0; i < forward_outputs->size(); ++i)
44 {
45 const auto forward_output_index = forward_outputs->operator[](i);
46 const auto forward_output_tensor = context.getTensorByIndex(forward_output_index);
47
48 OMRuntimeShape shape(forward_output_tensor);
49 const auto flat_size = shape.flatSize();
50
51 // Check type
52 assert(forward_output_tensor->type() == circle::TensorType_FLOAT32 && "Unsupported type");
53 if (forward_output_tensor->type() != circle::TensorType_FLOAT32)
54 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
55
56 // Get calculated data
57 uint8_t *calculated_data = nullptr;
58 OMStatus status = forward_storage.getDataByTensorIndex(&calculated_data, forward_output_index);
59 assert(calculated_data != nullptr);
60
61 OMLoss loss_type = config.training_context.loss;
62
63 // Get target data
64 auto data_type_size = sizeof(core::OMDataType(forward_output_tensor->type()));
65 size_t offset = batch_num * data_type_size * flat_size;
66
67 // Need to check loss type to control proper offset.
68 if (loss_type == SPARSE_CROSS_ENTROPY)
69 {
70 offset = batch_num * data_type_size;
71 }
72 uint8_t *target_data = _training_storage.getTargetData(i) + offset;
73
74 // Allocate data for error gradient for current calculated data and target data
75 uint8_t *output_grad_data;
76 core::memory::OMMemoryManager::allocateMemory(flat_size * data_type_size, &output_grad_data);
77 // Save error gradient into backward storage associated with current output tensor index
78 backward_storage.saveDataToTensorIndex(output_grad_data, forward_output_index);
79
80 // Handle different loss types
81 switch (loss_type)
82 {
83 case MSE:
84 {
85 losses_functions::MSE::calculateErrorBackpropagation(
86 flat_size, reinterpret_cast<float *>(calculated_data),
87 reinterpret_cast<float *>(target_data), reinterpret_cast<float *>(output_grad_data));
88 break;
89 }
90 case CROSS_ENTROPY:
91 {
92 losses_functions::CrossEntropy::calculateErrorBackpropagation(
93 flat_size, reinterpret_cast<float *>(calculated_data),
94 reinterpret_cast<float *>(target_data), reinterpret_cast<float *>(output_grad_data));
95 break;
96 }
97 case SPARSE_CROSS_ENTROPY:
98 {
99 losses_functions::SparseCrossEntropy::calculateErrorBackpropagation(
100 flat_size, reinterpret_cast<float *>(calculated_data),
101 reinterpret_cast<float *>(target_data), reinterpret_cast<float *>(output_grad_data));
102 break;
103 }
104 default:
105 {
106 assert(false && "Unsupported loss type");
107 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
108 }
109 }
110 }
111
112 return Ok;
113}

115/*
116 * Update weights with current optimizer logic
117 */
118OMStatus OMTrainingHandler::updateWeights(const OMConfig &config, OMRuntimeContext &context,
119 OMRuntimeStorage &storage)
120{
121 OMStatus status = Ok;
122
123 // Chose optimizer type
124 switch (config.training_context.optimizer)
125 {
126 case SGD:
127 {
128 auto *sgd_optimizer = _training_storage.getSGD();
129 assert(sgd_optimizer != nullptr);
130 if (sgd_optimizer == nullptr)
131 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
132
133 status = sgd_optimizer->updateWeights(config.training_context, context, storage,
134 _training_storage.getTensorIndexToRankTypeTable());
135 assert(status == Ok);
136 // Reset values
137#ifdef OM_MEMORY_ESTIMATE
138 sgd_optimizer->reset(context, storage);
139#else
140 sgd_optimizer->reset();
141#endif // OM_MEMORY_ESTIMATE
142 break;
143 }
144 case ADAM:
145 {
146 auto *adam_optimizer = _training_storage.getAdam();
147 assert(adam_optimizer != nullptr);
148 if (adam_optimizer == nullptr)
149 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
150 status = adam_optimizer->updateWeights(config.training_context, context, storage,
151 _training_storage.getTensorIndexToRankTypeTable());
152 assert(status == Ok);
153 // Reset values
154#ifdef OM_MEMORY_ESTIMATE
155 adam_optimizer->reset(context, storage);
156#else
157 adam_optimizer->reset();
158#endif // OM_MEMORY_ESTIMATE
159 break;
160 }
161 default:
162 {
163 assert(false && "Unsupported type");
164 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
165 }
166 }
167
168 return status;
169}

171/*
172 * Update optimizer state
173 *
174 * WARNING: It is assumed that the backward_storage contains only calculated gradients (this
175 * execution plane creator work).
176 */
177OMStatus OMTrainingHandler::updateOptimizerState(const OMConfig &config,
178 OMRuntimeStorage &backward_storage,
179 OMRuntimeContext &context)
180{
181 OMStatus status = Ok;
182
183 switch (config.training_context.optimizer)
184 {
185 case SGD:
186 {
187 auto *sgd_optimizer = _training_storage.getSGD();
188 assert(sgd_optimizer != nullptr);
189 if (sgd_optimizer == nullptr)
190 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
191
192 sgd_optimizer->handle(backward_storage, context, backward_storage);
193 break;
194 }
195 case ADAM:
196 {
197 auto *adam_optimizer = _training_storage.getAdam();
198 assert(adam_optimizer != nullptr);
199 if (adam_optimizer == nullptr)
200 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
201
202 adam_optimizer->handle(backward_storage, context, backward_storage);
203 break;
204 }
205 default:
206 {
207 assert(false && "Unsupported type");
208 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
209 }
210 }
211
212 return status;
213}

224/*
225 * Evaluate metric according OMMetrics and save it into metric_val
226 *
227 * Warning: 1) assume that all metric_val for all OMMetrics types actually are float values.
228 * 2) metric_val should be initialized with some value before calling this method due to
229 * after calculation for current batch_num (the sequence number of the current sample)
230 * this value is added to metric_val
231 */
232OMStatus OMTrainingHandler::evaluateMetric(OMMetrics metric, void *metric_val,
233 OMRuntimeStorage &storage, OMRuntimeContext &context,
234 uint32_t batch_num)
235{
236 // Go over all outputs and calculate metric
237 auto forward_outputs = context.getCircleOutputs();
238 for (uint32_t i = 0; i < forward_outputs->size(); ++i)
239 {
240 // Get output tensor
241 const auto forward_output_index = forward_outputs->operator[](i);
242 const auto forward_output_tensor = context.getTensorByIndex(forward_output_index);
243
244 OMRuntimeShape shape(forward_output_tensor);
245 const auto flat_size = shape.flatSize();
246
247 // Check type
248 assert(forward_output_tensor->type() == circle::TensorType_FLOAT32 && "Unsupported type");
249 if (forward_output_tensor->type() != circle::TensorType_FLOAT32)
250 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
251
252 // Get calculated data
253 uint8_t *calculated_data = nullptr;
254 OMStatus status = storage.getDataByTensorIndex(&calculated_data, forward_output_index);
255 assert(calculated_data != nullptr);
256
257 // Get target data
264 size_t offset = batch_num * sizeof(core::OMDataType(forward_output_tensor->type())) * flat_size;
265 uint8_t *target_data = _training_storage.getTargetData(i) + offset;
266
267 // Note: always cast it to float
268 float *f_metric_val = reinterpret_cast<float *>(metric_val);
269 switch (metric)
270 {
271 case MSE_METRICS:
272 {
273 // Note: sum up new calculated value for current sample
274 *f_metric_val +=
275 metrics::MSE::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
276 reinterpret_cast<float *>(target_data));
277 break;
278 }
279 case MAE_METRICS:
280 {
281 // Note: sum up new calculated value for current sample
282 *f_metric_val +=
283 metrics::MAE::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
284 reinterpret_cast<float *>(target_data));
285 break;
286 }
287 case CROSS_ENTROPY_METRICS:
288 {
289 // Note: sum up new calculated value for current sample
290 *f_metric_val += metrics::CrossEntropy::calculateValue(
291 flat_size, reinterpret_cast<float *>(calculated_data),
292 reinterpret_cast<float *>(target_data));
293 break;
294 }
295 case ACCURACY:
296 {
297 // Note: sum up new calculated value for current sample
298 *f_metric_val +=
299 metrics::Accuracy::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
300 reinterpret_cast<float *>(target_data));
301 break;
302 }
303 case SPARSE_CROSS_ENTROPY_ACCURACY:
304 {
305 // Note: sum up new calculated value for current sample
306 *f_metric_val += metrics::SparseCrossEntropyAccuracy::calculateValue(
307 flat_size, reinterpret_cast<float *>(calculated_data),
308 reinterpret_cast<float *>(target_data));
309 break;
310 }
311 default:
312 {
313 assert(false && "Unsupported loss type");
314 OM_LOG_AND_RETURN(UnsupportedType, "Unsupported type encountered");
315 }
316 }
317 }
318
319 return Ok;
320}
