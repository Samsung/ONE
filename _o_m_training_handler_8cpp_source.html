32/*
33 * Calculate backpropagation error between calculated data in forward graph and target data
34 */
35OMStatus OMTrainingHandler::handleError(const OMConfig &config, OMRuntimeStorage &forward_storage,
36 OMRuntimeStorage &backward_storage,
37 OMRuntimeContext &context, uint32_t batch_num)
38{
39 auto forward_outputs = context.getCircleOutputs();
40 // Go over all outputs
41 for (uint32_t i = 0; i < forward_outputs->size(); ++i)
42 {
43 const auto forward_output_index = forward_outputs->operator[](i);
44 const auto forward_output_tensor = context.getTensorByIndex(forward_output_index);
45
46 OMRuntimeShape shape(forward_output_tensor);
47 const auto flat_size = shape.flatSize();
48
49 // Check type
50 assert(forward_output_tensor->type() == circle::TensorType_FLOAT32 && "Unsupported type");
51 if (forward_output_tensor->type() != circle::TensorType_FLOAT32)
52 return UnsupportedType;
53
54 // Get calculated data
55 uint8_t *calculated_data = nullptr;
56 OMStatus status = forward_storage.getDataByTensorIndex(&calculated_data, forward_output_index);
57 if (status != Ok)
58 return status;
59 assert(calculated_data != nullptr);
60
61 // Get target data
62 auto data_type_size = sizeof(core::OMDataType(forward_output_tensor->type()));
63 size_t offset = batch_num * data_type_size * flat_size;
64 uint8_t *target_data = _training_storage.getTargetData(i) + offset;
65 OMLoss loss_type = config.training_context.loss;
66
67 // Allocate data for error gradient for current calculated data and target data
68 uint8_t *output_grad_data;
69 core::memory::OMMemoryManager::allocateMemory(flat_size * data_type_size, &output_grad_data);
70 // Save error gradient into backward storage associated with current output tensor index
71 backward_storage.saveDataToTensorIndex(output_grad_data, forward_output_index);
72
73 // Handle different loss types
74 switch (loss_type)
75 {
76 case MSE:
77 {
78 losses_functions::MSE::calculateErrorBackpropagation(
79 flat_size, reinterpret_cast<float *>(calculated_data),
80 reinterpret_cast<float *>(target_data), reinterpret_cast<float *>(output_grad_data));
81 break;
82 }
83 case CROSS_ENTROPY:
84 {
85 losses_functions::CrossEntropy::calculateErrorBackpropagation(
86 flat_size, reinterpret_cast<float *>(calculated_data),
87 reinterpret_cast<float *>(target_data), reinterpret_cast<float *>(output_grad_data));
88 break;
89 }
90 default:
91 {
92 assert(false && "Unsupported loss type");
93 return UnsupportedType;
94 }
95 }
96 }
97
98 return Ok;
99}

101/*
102 * Update weights with current optimizer logic
103 */
104OMStatus OMTrainingHandler::updateWeights(const OMConfig &config, OMRuntimeContext &context)
105{
106 OMStatus status = Ok;
107
108 // Chose optimizer type
109 switch (config.training_context.optimizer)
110 {
111 case SGD:
112 {
113 auto *sgd_optimizer = _training_storage.getSGD();
114 assert(sgd_optimizer != nullptr);
115 if (sgd_optimizer == nullptr)
116 return UnknownError;
117
118 status = sgd_optimizer->updateWeights(config.training_context, context);
119 assert(status == Ok);
120 // Reset values
121 sgd_optimizer->reset();
122 break;
123 }
124 case ADAM:
125 {
126 auto *adam_optimizer = _training_storage.getAdam();
127 assert(adam_optimizer != nullptr);
128 if (adam_optimizer == nullptr)
129 return UnknownError;
130
131 status = adam_optimizer->updateWeights(config.training_context, context);
132 assert(status == Ok);
133 // Reset values
134 adam_optimizer->reset();
135 break;
136 }
137 default:
138 {
139 assert(false && "Unsupported type");
140 return UnsupportedType;
141 }
142 }
143
144 return status;
145}

179 break;
180 }
181 default:
182 {
183 assert(false && "Unsupported type");
184 return UnsupportedType;
185 }
186 }
187
188 return status;

193/*
194 * Evaluate metric according OMMetrics and save it into metric_val
195 *
196 * Warning: 1) assume that all metric_val for all OMMetrics types actually are float values.
197 * 2) metric_val should be initialized with some value before calling this method due to
198 * after calculation for current batch_num (the sequence number of the current sample)
199 * this value is added to metric_val
200 */
201OMStatus OMTrainingHandler::evaluateMetric(OMMetrics metric, void *metric_val,
202 OMRuntimeStorage &storage, OMRuntimeContext &context,
203 uint32_t batch_num)
204{
205 // Go over all outputs and calculate metric
206 auto forward_outputs = context.getCircleOutputs();
207 for (uint32_t i = 0; i < forward_outputs->size(); ++i)
208 {
209 // Get output tensor
210 const auto forward_output_index = forward_outputs->operator[](i);
211 const auto forward_output_tensor = context.getTensorByIndex(forward_output_index);
212
213 OMRuntimeShape shape(forward_output_tensor);
214 const auto flat_size = shape.flatSize();
215
216 // Check type
217 assert(forward_output_tensor->type() == circle::TensorType_FLOAT32 && "Unsupported type");
218 if (forward_output_tensor->type() != circle::TensorType_FLOAT32)
219 return UnsupportedType;
220
221 // Get calculated data
222 uint8_t *calculated_data = nullptr;
223 OMStatus status = storage.getDataByTensorIndex(&calculated_data, forward_output_index);
224 if (status != Ok)
225 return status;
226 assert(calculated_data != nullptr);
227
228 // Get target data
229 size_t offset = batch_num * sizeof(core::OMDataType(forward_output_tensor->type())) * flat_size;
230 uint8_t *target_data = _training_storage.getTargetData(i) + offset;
231
232 // Note: always cast it to float
233 float *f_metric_val = reinterpret_cast<float *>(metric_val);
234 switch (metric)
235 {
236 case MSE_METRICS:
237 {
238 // Note: sum up new calculated value for current sample
239 *f_metric_val +=
240 metrics::MSE::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
241 reinterpret_cast<float *>(target_data));
242 break;
243 }
244 case MAE_METRICS:
245 {
246 // Note: sum up new calculated value for current sample
247 *f_metric_val +=
248 metrics::MAE::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
249 reinterpret_cast<float *>(target_data));
250 break;
251 }
252 case CROSS_ENTROPY_METRICS:
253 {
254 // Note: sum up new calculated value for current sample
255 *f_metric_val += metrics::CrossEntropy::calculateValue(
256 flat_size, reinterpret_cast<float *>(calculated_data),
257 reinterpret_cast<float *>(target_data));
258 break;
259 }
260 case ACCURACY:
261 {
262 // Note: sum up new calculated value for current sample
263 *f_metric_val +=
264 metrics::Accuracy::calculateValue(flat_size, reinterpret_cast<float *>(calculated_data),
265 reinterpret_cast<float *>(target_data));
266 break;
267 }
268 default:
269 {
270 assert(false && "Unsupported loss type");
271 return UnsupportedType;
272 }
273 }
274 }
275
276 return Ok;
277}
