70std::unique_ptr<luci::Module> BisectionSolver::run(const std::string &module_path)
71{
72 LOGGER(l);
73
74 auto module = read_module(module_path);
75
76 float min_depth = 0.f;
77 float max_depth = 0.f;
78 NodeDepthType nodes_depth;
79 if (compute_depth(module.get(), nodes_depth, min_depth, max_depth) !=
80 ParameterizerResult::SUCCESS)
81 {
82 std::cerr << "ERROR: Invalid graph for bisectioning" << std::endl;
83 return nullptr;
84 }
85
86 std::unique_ptr<MAEMetric> metric = std::make_unique<MAEMetric>();
87 DatasetEvaluator evaluator(module.get(), _input_data_path, *metric.get());
88
89 LayerParams layer_params;
90 float int16_qerror =
91 evaluate(evaluator, module_path, "int16" /* default quant_dtype */, layer_params);
92 VERBOSE(l, 0) << "Full int16 model quantization error " << int16_qerror << std::endl;
93
94 float uint8_qerror =
95 evaluate(evaluator, module_path, "uint8" /* default quant_dtype */, layer_params);
96 VERBOSE(l, 0) << "Full uint8 model quantization error " << uint8_qerror << std::endl;
97
98 if (int16_qerror > uint8_qerror)
99 {
100 throw std::runtime_error("Q8 model's qerror is less than Q16 model's qerror.");
101 }
102
103 _qerror = int16_qerror + _qerror_ratio * std::fabs(uint8_qerror - int16_qerror);
104 VERBOSE(l, 0) << "Target quantization error " << _qerror << std::endl;
105
106 if (uint8_qerror <= _qerror)
107 {
108 // no need for bisectioning just return Q8 model
109 if (!_quantizer->quantize(module.get(), "uint8", layer_params))
110 {
111 std::cerr << "ERROR: Failed to quantize model" << std::endl;
112 return nullptr;
113 }
114 }
115
116 int last_depth = -1;
117 float best_depth = -1;
118 LayerParams best_params;
119 if (module->size() != 1)
120 {
121 throw std::runtime_error("Unsupported module");
122 }
123 auto graph = module->graph(0);
124 auto active_nodes = loco::active_nodes(loco::output_nodes(graph));
125 // input and output nodes are not valid for quantization, so let's remove them
126 for (auto node : loco::input_nodes(graph))
127 {
128 active_nodes.erase(node);
129 }
130 for (auto node : loco::output_nodes(graph))
131 {
132 active_nodes.erase(node);
133 }
134
135 // let's decide whether nodes at input are more suspectible to be quantized into Q16, than at
136 // output
137 bool int16_front = true;
138 switch (_algorithm)
139 {
140 case Algorithm::Auto: // TODO
141 case Algorithm::ForceQ16Front:
142 int16_front = true;
143 break;
144 case Algorithm::ForceQ16Back:
145 int16_front = true;
146 break;
147 }
148
149 while (true)
150 {
151 int cut_depth = static_cast<int>(std::floor(0.5f * (min_depth + max_depth)));
152
153 if (last_depth == cut_depth)
154 {
155 break;
156 }
157 last_depth = cut_depth;
158
159 LayerParams layer_params;
160 for (auto &node : active_nodes)
161 {
162 auto cur_node = loco::must_cast<luci::CircleNode *>(node);
163 auto iter = nodes_depth.find(cur_node);
164 if (iter == nodes_depth.end())
165 {
166 continue; // to filter out nodes like weights
167 }
168
169 float depth = iter->second;
170
171 if ((depth <= cut_depth && int16_front) || (depth >= cut_depth && !int16_front))
172 {
173 auto layer_param = std::make_shared<LayerParam>();
174 {
175 layer_param->name = cur_node->name();
176 layer_param->dtype = "int16";
177 layer_param->granularity = "channel";
178 }
179
180 layer_params.emplace_back(layer_param);
181 }
182 }
183
184 float cur_accuracy = evaluate(evaluator, module_path, "uint8", layer_params);
185 VERBOSE(l, 0) << cut_depth << " : " << cur_accuracy << std::endl;
186
187 if (cur_accuracy < _qerror)
188 {
189 int16_front ? (max_depth = cut_depth) : (min_depth = cut_depth);
190 best_params = layer_params;
191 best_depth = cut_depth;
192 }
193 else
194 {
195 int16_front ? (min_depth = cut_depth) : (max_depth = cut_depth);
196 }
197 }
198
199 VERBOSE(l, 0) << "Found the best configuration at " << best_depth << " depth." << std::endl;
200 if (!_quantizer->quantize(module.get(), "uint8", best_params))
201 {
202 std::cerr << "ERROR: Failed to quantize model" << std::endl;
203 return nullptr;
204 }
205
206 return module;
207}
