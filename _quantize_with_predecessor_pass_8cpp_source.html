30// Quantize dst node using src node's qparam
31// Return true if dst node is quantized with src
32// Return false otherwise
33bool quantize_with_same_qparam(luci::CircleNode *src, luci::CircleNode *dst)
34{
35 // src node is not quantized. Skip this case.
36 auto src_qparam = src->quantparam();
37 if (not src_qparam)
38 return false;
39
40 auto dst_qparam = dst->quantparam();
41 // dst node is already quantized. Skip this case.
42 if (dst_qparam)
43 return false;
44
45 luci::copy_quantparam(src, dst);
46
47 dst->dtype(src->dtype());
48
49 return true;
50}

52// Visitor to quantize nodes using predecessors qparams
53struct QuantizeWithPredecessor final : public luci::CircleNodeMutableVisitor<bool>
54{
55 QuantizeWithPredecessor() = default;
56
57 bool visit(luci::CircleNode *) { return false; }
58
59 bool visit(luci::CircleReshape *node)
60 {
61 auto input_node = loco::must_cast<luci::CircleNode *>(node->tensor());
62 return quantize_with_same_qparam(input_node, node);
63 }
64
65 bool visit(luci::CircleTranspose *node)
66 {
67 auto input_node = loco::must_cast<luci::CircleNode *>(node->a());
68 return quantize_with_same_qparam(input_node, node);
69 }
70
71 bool visit(luci::CircleStridedSlice *node)
72 {
73 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
74 return quantize_with_same_qparam(input_node, node);
75 }
76
77 bool visit(luci::CircleSqueeze *node)
78 {
79 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
80 return quantize_with_same_qparam(input_node, node);
81 }
82
83 bool visit(luci::CircleGather *node)
84 {
85 auto input_node = loco::must_cast<luci::CircleNode *>(node->params());
86 return quantize_with_same_qparam(input_node, node);
87 }
88
89 bool visit(luci::CircleNeg *node)
90 {
91 auto input_node = loco::must_cast<luci::CircleNode *>(node->x());
92 // Only support S16 for now
93 if (input_node->dtype() != loco::DataType::S16)
94 return false;
95
96 return quantize_with_same_qparam(input_node, node);
97 }
98
99 bool visit(luci::CircleConcatenation *node)
100 {
101 const auto num_inputs = node->numValues();
102
103 for (uint32_t i = 0; i < num_inputs; i++)
104 {
105 auto input = loco::must_cast<luci::CircleNode *>(node->values(i));
106 // Only support S16 for now
107 if (input->dtype() != loco::DataType::S16)
108 return false;
109
110 if (input->quantparam() == nullptr)
111 return false;
112
113 if (input->quantparam()->scale.size() != 1)
114 return false;
115 }
116
117 luci::CircleNode *max_scale_node = nullptr;
118 float max_scale = 0.0;
119 for (uint32_t i = 0; i < num_inputs; i++)
120 {
121 auto input = loco::must_cast<luci::CircleNode *>(node->values(i));
122 auto qparam = input->quantparam();
123 auto scale = qparam->scale.at(0);
124 if (max_scale < scale)
125 {
126 max_scale = scale;
127 max_scale_node = input;
128 }
129 }
130
131 assert(max_scale_node);
132
133 return quantize_with_same_qparam(max_scale_node, node);
134 }
135};

142bool QuantizeWithPredecessorPass::run(loco::Graph *g)
143{
144 bool changed = false;
145
146 LOGGER(l);
147
148 for (auto node : loco::active_nodes(loco::output_nodes(g)))
149 {
150 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
151 INFO(l) << "QuantizeWithPredecessorPass visit node: " << circle_node->name() << std::endl;
152
153 QuantizeWithPredecessor qwp;
154 if (circle_node->accept(&qwp))
155 {
156 changed = true;
157 }
158 }
159
160 return changed;
161}
