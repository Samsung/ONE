30// Quantize dst node using src node's qparam
31// Return true if dst node is quantized with src
32// Return false otherwise
33bool quantize_with_same_qparam(luci::CircleNode *src, luci::CircleNode *dst)
34{
35 // src node is not quantized. Skip this case.
36 auto src_qparam = src->quantparam();
37 if (not src_qparam)
38 return false;
39
40 auto dst_qparam = dst->quantparam();
41 // dst node is already quantized. Skip this case.
42 if (dst_qparam)
43 return false;
44
45 luci::copy_quantparam(src, dst);
46
47 dst->dtype(src->dtype());
48
49 return true;
50}

52// Visitor to quantize nodes using predecessors qparams
53struct QuantizeWithPredecessor final : public luci::CircleNodeMutableVisitor<bool>
54{
55 QuantizeWithPredecessor() = default;
56
57 bool visit(luci::CircleNode *) { return false; }
58
59 bool visit(luci::CircleReshape *node)
60 {
61 auto input_node = loco::must_cast<luci::CircleNode *>(node->tensor());
62 return quantize_with_same_qparam(input_node, node);
63 }
64
65 bool visit(luci::CircleTranspose *node)
66 {
67 auto input_node = loco::must_cast<luci::CircleNode *>(node->a());
68 return quantize_with_same_qparam(input_node, node);
69 }
70
71 bool visit(luci::CircleStridedSlice *node)
72 {
73 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
74 return quantize_with_same_qparam(input_node, node);
75 }
76
77 bool visit(luci::CircleSqueeze *node)
78 {
79 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
80 return quantize_with_same_qparam(input_node, node);
81 }
82
83 bool visit(luci::CircleGather *node)
84 {
85 auto input_node = loco::must_cast<luci::CircleNode *>(node->params());
86 return quantize_with_same_qparam(input_node, node);
87 }
88
89 bool visit(luci::CircleMul *node)
90 {
91 // Skip if node is already quantized
92 if (luci::is_quantized(node))
93 return false;
94
95 auto x = loco::must_cast<luci::CircleNode *>(node->x());
96 auto y = loco::must_cast<luci::CircleNode *>(node->y());
97
98 // Only support square for now
99 if (x != y)
100 return false;
101
102 // Only support S16 for now
103 if (x->dtype() != loco::DataType::S16)
104 return false;
105
106 const auto input_qparam = x->quantparam();
107 if (not input_qparam)
108 return false;
109
110 if (input_qparam->scale.size() != 1)
111 return false;
112
113 const auto input_scale = input_qparam->scale.at(0);
114
115 const auto s16_max = std::numeric_limits<int16_t>::max();
116 // How to determine a new scale of x^2?
117 // x's scale would have been determined by its max or min
118 //
119 // Max value of x^2 = (s * s16_max)^2
120 // Min value of x^2 = 0
121 // New scale = (s * s16_max)^2 / s16_max = s^2 * s16_max
122 //
123 // NOTE s16_max = -s16_min (symmetric quantization)
124 const auto output_scale = input_scale * input_scale * s16_max;
125
126 auto new_qparam = std::make_unique<luci::CircleQuantParam>();
127 {
128 new_qparam->scale.push_back(output_scale);
129 new_qparam->zerop.push_back(0);
130 }
131
132 node->quantparam(std::move(new_qparam));
133 node->dtype(x->dtype());
134
135 return true;
136 }
137
138 bool visit(luci::CircleNeg *node)
139 {
140 auto input_node = loco::must_cast<luci::CircleNode *>(node->x());
141 // Only support S16 for now
142 if (input_node->dtype() != loco::DataType::S16)
143 return false;
144
145 return quantize_with_same_qparam(input_node, node);
146 }
147
148 bool visit(luci::CircleConcatenation *node)
149 {
150 const auto num_inputs = node->numValues();
151
152 for (uint32_t i = 0; i < num_inputs; i++)
153 {
154 auto input = loco::must_cast<luci::CircleNode *>(node->values(i));
155 // Only support S16 for now
156 if (input->dtype() != loco::DataType::S16)
157 return false;
158
159 if (input->quantparam() == nullptr)
160 return false;
161
162 if (input->quantparam()->scale.size() != 1)
163 return false;
164 }
165
166 luci::CircleNode *max_scale_node = nullptr;
167 float max_scale = 0.0;
168 for (uint32_t i = 0; i < num_inputs; i++)
169 {
170 auto input = loco::must_cast<luci::CircleNode *>(node->values(i));
171 auto qparam = input->quantparam();
172 auto scale = qparam->scale.at(0);
173 if (max_scale < scale)
174 {
175 max_scale = scale;
176 max_scale_node = input;
177 }
178 }
179
180 if (max_scale_node == nullptr)
181 {
182 throw std::runtime_error{"Invalid max_scale_node"};
183 }
184
185 return quantize_with_same_qparam(max_scale_node, node);
186 }
187};

194bool QuantizeWithPredecessorPass::run(loco::Graph *g)
195{
196 bool changed = false;
197
198 LOGGER(l);
199
200 for (auto node : loco::active_nodes(loco::output_nodes(g)))
201 {
202 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
203 INFO(l) << "QuantizeWithPredecessorPass visit node: " << circle_node->name() << std::endl;
204
205 QuantizeWithPredecessor qwp;
206 if (circle_node->accept(&qwp))
207 {
208 changed = true;
209 }
210 }
211
212 return changed;
213}
