24ir::Shape convertShape(const ir::Shape &shape, ir::Layout src_layout, ir::Layout dst_layout)
25{
26 if (shape.rank() != 4)

28
29 if (src_layout == dst_layout)

31

39 dst_NHWC.dim(3) = src_NCHW.dim(1); // C

42 }

50 dst_NCHW.dim(2) = src_NHWC.dim(1); // H
51 dst_NCHW.dim(3) = src_NHWC.dim(2); // W
52
