62void InstanceNorm::evalFloat() const
63{
64 float activation_min, activation_max;
65 calculateActivationRange(params().activation, &activation_min, &activation_max);
66 auto input_shape = getTensorShape(input());
67 auto output_shape = getTensorShape(output());
68 const int32_t batches = tflite::MatchingDim(input_shape, 0, output_shape, 0);
69 const int32_t heights = tflite::MatchingDim(input_shape, 1, output_shape, 1);
70 const int32_t widths = tflite::MatchingDim(input_shape, 2, output_shape, 2);
71 const int32_t channels = tflite::MatchingDim(input_shape, 3, output_shape, 3);
72 const float *input_data = getTensorData<float>(input());
73 const float *gamma_data = getTensorData<float>(gamma());
74 auto gamma_shape = getTensorShape(gamma());
75 bool single_gamma = gamma_shape.DimensionsCount() == 1 && gamma_shape.Dims(0) == 1;
76 const float *beta_data = getTensorData<float>(beta());
77 auto beta_shape = getTensorShape(beta());
78 bool single_beta = beta_shape.DimensionsCount() == 1 && beta_shape.Dims(0) == 1;
79 float *output_data = getTensorData<float>(output());
80 for (int32_t batch = 0; batch < batches; batch++)
81 {
82 for (int32_t channel = 0; channel < channels; channel++)
83 {
84 double sum = 0.0f;
85 double square_sum = 0.0f;
86 int32_t size = heights * widths;
87 for (int32_t height = 0; height < heights; height++)
88 {
89 for (int32_t width = 0; width < widths; width++)
90 {
91 double input_val = input_data[tflite::Offset(input_shape, batch, height, width, channel)];
92 sum += input_val;
93 square_sum += (input_val * input_val);
94 }
95 }
96 double mean = sum / size;
97 double var = square_sum / size - mean * mean;
98
99 double gamma = single_gamma ? gamma_data[0] : gamma_data[channel];
100 double beta = single_beta ? beta_data[0] : beta_data[channel];
101 double a = gamma / (std::sqrt(var + params().epsilon));
102 double b = -mean * a + beta;
103
104 for (int32_t height = 0; height < heights; height++)
105 {
106 for (int32_t width = 0; width < widths; width++)
107 {
108 double input_value =
109 input_data[tflite::Offset(output_shape, batch, height, width, channel)];
110 double output_value = input_value * a + b;
111 output_data[tflite::Offset(output_shape, batch, height, width, channel)] =
112 tflite::ActivationFunctionWithMinMax((float)output_value, activation_min,
113 activation_max);
114 }
115 }
116 }
117 }
118}
