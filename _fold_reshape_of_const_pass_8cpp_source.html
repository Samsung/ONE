35locoex::TFLReshape *as_candidate(loco::Node *node)
36{
37 auto reshape = dynamic_cast<locoex::TFLReshape *>(node);
38 if (not reshape)
39 return nullptr;
40
41 // Only accept Constant input of Reshape
42 if (not dynamic_cast<locoex::TFLConst *>(reshape->tensor()))
43 return nullptr;
44
45 return reshape;
46}

48uint32_t volume(loco::Node *tensor_node)
49{
50 auto shape = loco::shape_get(tensor_node).as<loco::TensorShape>();
51
52 uint32_t vol = 1;
53 for (uint32_t axis = 0; axis < shape.rank(); ++axis)
54 vol *= shape.dim(axis).value();
55
56 return vol;
57}

59void fold_reshape_of_const(locoex::TFLReshape *reshape)
60{
61 const loco::DataType FLOAT32 = loco::DataType::FLOAT32;
62
63 auto const_orig = dynamic_cast<locoex::TFLConst *>(reshape->tensor());
64
65 // Exceptions
66 {
67 EXO_ASSERT(const_orig, "Only support for Reshape-Const pair");
68 // TODO support other data types
69 if (const_orig->dtype() != FLOAT32)
70 INTERNAL_EXN_V("NYI for this data type", oops::to_uint32(const_orig->dtype()));
71
72 if (volume(const_orig) != volume(reshape))
73 INTERNAL_EXN("New shape of Reshape is not matched");
74 }
75
76 auto new_shape = loco::shape_get(reshape).as<loco::TensorShape>();
77
78 // TFLConst to replace
79 auto const_new = reshape->graph()->nodes()->create<locoex::TFLConst>();
80
81 const_new->dtype(FLOAT32);
82 const_new->rank(new_shape.rank());
83 const_new->size<FLOAT32>(const_orig->size<FLOAT32>());
84 for (uint32_t axis = 0; axis < new_shape.rank(); ++axis)
85 const_new->dim(axis) = new_shape.dim(axis);
86
87 for (uint32_t i = 0; i < const_new->size<FLOAT32>(); ++i)
88 {
89 const_new->at<FLOAT32>(i) = const_orig->at<FLOAT32>(i);
90 }
91
92 // replace
93 loco::replace(reshape).with(const_new);
94}

101bool FoldReshapeOfConstPass::run(loco::Graph *g)
102{
103 bool changed = false;
104 for (auto node : loco::active_nodes(loco::output_nodes(g)))
105 {
106 if (auto reshape = as_candidate(node))
107 {
108 fold_reshape_of_const(reshape);
109 changed = true;
110 }
111 }
112
113 return changed;
114}
