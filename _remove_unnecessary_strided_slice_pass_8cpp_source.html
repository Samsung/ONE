28int64_t value_from_circle_const(const luci::CircleConst *node, uint32_t idx)
29{
30 assert(node->rank() == 1 && node->dim(0).value() > idx);

38bool remove_no_effect_strided_slice(luci::CircleStridedSlice *target_node)
39{
40 auto begin_const = dynamic_cast<luci::CircleConst *>(target_node->begin());
41 if (begin_const == nullptr)
42 return false;
43
44 auto strides_const = dynamic_cast<luci::CircleConst *>(target_node->strides());
45 if (strides_const == nullptr)
46 return false;
47
48 auto end_const = dynamic_cast<luci::CircleConst *>(target_node->end());
49 if (end_const == nullptr)
50 return false;
51
52 auto input_node = loco::must_cast<luci::CircleNode *>(target_node->input());
53 for (uint32_t i = 0; i < input_node->rank(); i++)
54 {
55 if (value_from_circle_const(begin_const, i) != 0)
56 return false;
57
58 int64_t strides_value = value_from_circle_const(strides_const, i);
59 if (strides_value != 1)
60 return false;
61
62 int64_t end_value = value_from_circle_const(end_const, i);
63 if (end_value == -1)
64 continue;
65
66 if (end_value != input_node->dim(i).value())
67 return false;
68
69 if (!input_node->dim(i).known())
70 return false;
71 }
72
77 if (target_node->new_axis_mask() != 0 || target_node->shrink_axis_mask() != 0)
78 return false;
79
80 replace(target_node).with(input_node);
81 return true;
82}

111bool RemoveUnnecessaryStridedSlicePass::run(loco::Graph *g)
112{
113 bool changed = false;
114 for (auto node : loco::active_nodes(loco::output_nodes(g)))
115 {
116 auto target_node = dynamic_cast<luci::CircleStridedSlice *>(node);
117 if (target_node != nullptr)
118 if (remove_no_effect_strided_slice(target_node))
119 changed = true;
120 }
121 return changed;
