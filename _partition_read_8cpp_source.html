43luci::PartitionTable parse_table(const crew::Sections &sections)
44{
45 luci::PartitionTable table;
46
47 // default comply as OPCODE
48 table.comply = luci::PartitionTable::COMPLY::OPCODE;
49
50 // read main '[partition]' first
51 for (auto &section : sections)
52 {
53 if (section.name == _section_partition)
54 {
55 auto &items = section.items;
56 if (items.find(_key_backends) == items.end())
57 {
58 throw std::invalid_argument("'backends' is required");
59 }
60 if (items.find(_key_default) == items.end())
61 {
62 throw std::invalid_argument("'default' is required");
63 }
64
65 table.groups = pepper::csv_to_vector<std::string>(items.at(_key_backends));
66 table.default_group = items.at(_key_default);
67
68 auto comply = items.at(_key_comply);
69
70 // check valid comply types
71 if (comply == _comply_opcode)
72 {
73 table.comply = luci::PartitionTable::COMPLY::OPCODE;
74 continue;
75 }
76 if (comply == _comply_opname)
77 {
78 table.comply = luci::PartitionTable::COMPLY::OPNAME;
79 continue;
80 }
81 throw std::runtime_error("Invalid or comply is not set");
82 }
83 }
84
85 // read other sections
86 for (auto &section : sections)
87 {
88 if (section.name == _section_OPCODE)
89 {
90 auto &items = section.items;
91
92 for (auto &item : items)
93 {
94 if (item.first == _key_underscore)
95 {
96 if (table.comply == luci::PartitionTable::COMPLY::OPCODE)
97 table.default_group = item.second;
98 }
99 else
100 {
101 table.byopcodes.emplace(item.first, item.second);
102 }
103 }
104 }
105 else if (section.name == _section_OPNAME)
106 {
107 auto &items = section.items;
108
109 for (auto &item : items)
110 {
111 if (item.first == _key_underscore)
112 {
113 if (table.comply == luci::PartitionTable::COMPLY::OPNAME)
114 table.default_group = item.second;
115 }
116 else
117 {
118 table.byopnames.emplace(item.first, item.second);
119 }
120 }
121 }
122 }
123
124 return table;
125}
