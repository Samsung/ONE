33bool compute_output(uint32_t *output_size, luci::Padding padding, int32_t image_size,
34 int32_t filter_size, int32_t stride, int32_t dilation_rate)
35{
36 auto const effective_filter_size = (filter_size - 1) * dilation_rate + 1;
37 switch (padding)
38 {
39 case luci::Padding::SAME:
40 *output_size = (image_size + stride - 1) / stride;
41 return true;
42
43 case luci::Padding::VALID:
44 *output_size = (image_size + stride - effective_filter_size) / stride;
45 return true;
46
47 default:
48 {
49 LOGGER(l);
50 WARN(l) << "Unsupported padding: " << uint32_t(padding);
51 return false;
52 }
53 }
54}

56uint32_t compute_padding(int32_t stride, int32_t dilation_rate, int32_t in_size,
57 int32_t filter_size, int32_t out_size)
58{
59 auto const effective_filter_size = (filter_size - 1) * dilation_rate + 1;
60 auto const padding = ((out_size - 1) * stride + effective_filter_size - in_size) / 2;
61 return padding > 0 ? padding : 0;
62}

64bool set_kernel_parameters(tflite::DepthwiseParams *params, luci::CircleDepthwiseConv2D *node,
65 uint32_t padding_height, uint32_t padding_width)
66{
67 switch (node->fusedActivationFunction())
68 {
69 case luci::FusedActFunc::NONE:
70 case luci::FusedActFunc::TANH:
71 params->float_activation_min = std::numeric_limits<float>::lowest();
72 params->float_activation_max = std::numeric_limits<float>::max();
73 break;
74 case luci::FusedActFunc::RELU:
75 params->float_activation_min = 0;
76 params->float_activation_max = std::numeric_limits<float>::max();
77 break;
78 case luci::FusedActFunc::RELU_N1_TO_1:
79 params->float_activation_min = -1;
80 params->float_activation_max = 1;
81 break;
82 case luci::FusedActFunc::RELU6:
83 params->float_activation_min = 0;
84 params->float_activation_max = 6;
85 break;
86 default:
87 {
88 LOGGER(l);
89 WARN(l) << "Unsupported activation: " << uint32_t(node->fusedActivationFunction());
90 return false;
91 }
92 }
93
94 params->stride_height = node->stride()->h();
95 params->stride_width = node->stride()->w();
96 params->dilation_height_factor = node->dilation()->h();
97 params->dilation_width_factor = node->dilation()->w();
98 params->depth_multiplier = node->depthMultiplier();
99
100 params->padding_values.height = padding_height;
101 params->padding_values.width = padding_width;
102
103 return true;
104}

119bool fold_depthwise_conv_2d(luci::CircleDepthwiseConv2D *node)
120{
121 LOGGER(l);
122
123 auto const input = dynamic_cast<luci::CircleConst *>(node->input());
124
125 if (input == nullptr)
126 return false; // Constant input is required for folding
127
128 auto const filter = dynamic_cast<luci::CircleConst *>(node->filter());
129
130 if (filter == nullptr)
131 return false; // Constant filter is required for folding
132
133 if (filter->dim(0).value() != 1)
134 return false; // Unsupported batch size
135
136 auto const bias = dynamic_cast<luci::CircleConst *>(node->bias());
137
138 if (bias == nullptr)
139 return false; // Constant bias is required for folding
140
141 auto const input_batches = input->dim(0).value();
142 auto const input_height = input->dim(1).value();
143 auto const input_width = input->dim(2).value();
144 auto const input_depth = input->dim(3).value();
145
146 auto const filter_height = filter->dim(1).value();
147 auto const filter_width = filter->dim(2).value();
148 auto const filter_channels_out = filter->dim(3).value();
149
150 if (filter_channels_out % input_depth != 0)
151 return false; // Wrong input/output depth ratio
152
153 if (node->depthMultiplier() != static_cast<int32_t>(filter_channels_out / input_depth))
154 return false; // Wrong depth multiplier value
155
156 if (bias->rank() != 1 || bias->dim(0).value() != filter_channels_out)
157 return false; // Unsupported bias value
158
159 uint32_t output_height = 0;
160 uint32_t output_width = 0;
161
162 if (!compute_output(&output_height, node->padding(), input_height, filter_height,
163 node->stride()->h(), node->dilation()->h()))
164 return false; // Unsupported output parameters
165
166 if (!compute_output(&output_width, node->padding(), input_width, filter_width,
167 node->stride()->w(), node->dilation()->w()))
168 return false; // Unsupported output parameters
169
170 auto const padding_height = compute_padding(node->stride()->h(), node->dilation()->h(),
171 input_height, filter_height, output_height);
172 auto const padding_width = compute_padding(node->stride()->w(), node->dilation()->w(),
173 input_width, filter_width, output_width);
174
175 tflite::DepthwiseParams params{};
176
177 if (!set_kernel_parameters(&params, node, padding_height, padding_width))
178 return false; // Unsupported kernel parameter values
179
180 auto constant = node->graph()->nodes()->create<luci::CircleConst>();
181 constant->name(node->name());
182 constant->dtype(node->dtype());
183 constant->rank(node->rank());
184 constant->shape_status(luci::ShapeStatus::VALID);
185 for (uint32_t i = 0; i < node->rank(); ++i)
186 constant->dim(i).set(node->dim(i).value());
187
188 constant->size<loco::DataType::FLOAT32>(input_batches * output_height * output_width *
189 filter_channels_out);
190
191 auto const input_data = &input->at<loco::DataType::FLOAT32>(0);
192 auto const filter_data = &filter->at<loco::DataType::FLOAT32>(0);
193 auto const bias_data = &bias->at<loco::DataType::FLOAT32>(0);
194 auto const constant_data = &constant->at<loco::DataType::FLOAT32>(0);
195
196 auto tensor_shape = [](luci::CircleNode *node) {
197 tflite::RuntimeShape runtime_shape(node->rank());
198 for (uint32_t i = 0; i < node->rank(); ++i)
199 runtime_shape.SetDim(i, node->dim(i).value());
200 return runtime_shape;
201 };
202
203 tflite::reference_ops::DepthwiseConv(params, tensor_shape(input), input_data,
204 tensor_shape(filter), filter_data, tensor_shape(bias),
205 bias_data, tensor_shape(constant), constant_data);
206
207 loco::replace(node).with(constant);
208
209 return true;
210}

220bool FoldDepthwiseConv2DPass::run(loco::Graph *g)
221{
222 bool changed = false;
223 for (auto node : loco::active_nodes(loco::output_nodes(g)))
224 {
225 auto depthwise_conv2d = dynamic_cast<CircleDepthwiseConv2D *>(node);
226
227 if (depthwise_conv2d == nullptr)
228 continue;
229
230 switch (depthwise_conv2d->dtype())
231 {
232 case loco::DataType::FLOAT32:
233 changed = fold_depthwise_conv_2d(depthwise_conv2d);
234 break;
235 default:
236 break;
237 }
238 }
239
240 return changed;
241}
