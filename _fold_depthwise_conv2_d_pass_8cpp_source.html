37bool set_params(const luci::CircleDepthwiseConv2D *node, compute::DepthwiseConv2D &cdc)
38{
39 assert(node);
40
41 LOGGER(l);
42
43 auto &params = cdc.params();
44 if (!to_compute(node->padding(), params.padding_type))
45 {
46 WARN(l) << "FoldDepthwiseConv2DPass unsupported padding: " << uint32_t(node->padding());
47 return false;
48 }
49
50 params.stride_height = node->stride()->h();
51 params.stride_width = node->stride()->w();
52 params.dilation_height_factor = node->dilation()->h();
53 params.dilation_width_factor = node->dilation()->w();
54 params.depth_multiplier = node->depthMultiplier();
55
56 compute::FusedActFunc fac;
57 if (!to_compute(node->fusedActivationFunction(), fac))
58 {
59 WARN(l) << "FoldDepthwiseConv2DPass unsupported activation: "
60 << uint32_t(node->fusedActivationFunction());
61 return false;
62 }
63 cdc.fused_act_func(fac);
64
65 return true;
66}

81bool fold_depthwise_conv_2d(luci::CircleDepthwiseConv2D *node)
82{
83 auto const input = dynamic_cast<luci::CircleConst *>(node->input());
84 if (input == nullptr)
85 return false; // Constant input is required for folding
86
87 auto const filter = dynamic_cast<luci::CircleConst *>(node->filter());
88 if (filter == nullptr)
89 return false; // Constant filter is required for folding
90 if (filter->dim(0).value() != 1)
91 return false; // Unsupported batch size
92
93 auto const bias = dynamic_cast<luci::CircleConst *>(node->bias());
94 if (bias == nullptr)
95 return false; // Constant bias is required for folding
96
97 auto static_shape = [](luci::CircleNode *node) {
98 loco::TensorShape shape;
99 shape.rank(node->rank());
100 for (uint32_t i = 0; i < node->rank(); ++i)
101 shape.dim(i) = node->dim(i);
102 return shape;
103 };
104
105 auto const input_data = &input->at<loco::DataType::FLOAT32>(0);
106 auto const filter_data = &filter->at<loco::DataType::FLOAT32>(0);
107 auto const bias_data = &bias->at<loco::DataType::FLOAT32>(0);
108
109 compute::DepthwiseConv2D comp_dwconv2d{};
110 if (!set_params(node, comp_dwconv2d))
111 return false;
112 comp_dwconv2d.input(static_shape(input), input_data);
113 comp_dwconv2d.filter(static_shape(filter), filter_data);
114 comp_dwconv2d.bias(static_shape(bias), bias_data);
115
116 if (!comp_dwconv2d.prepare())
117 return false;
118
119 auto output_shape = comp_dwconv2d.output_shape();
120 assert(is_same_shape(node, output_shape));
121 auto output_size = loco::element_count(&output_shape);
122
123 // result folded constant node
124 auto constant = node->graph()->nodes()->create<luci::CircleConst>();
125 constant->dtype(node->dtype());
126 constant->rank(node->rank());
127 for (uint32_t i = 0; i < output_shape.rank(); ++i)
128 constant->dim(i).set(output_shape.dim(i).value());
129 constant->shape_status(luci::ShapeStatus::VALID);
130 constant->size<loco::DataType::FLOAT32>(output_size);
131 constant->name(node->name());
132
133 auto constant_data = &constant->at<loco::DataType::FLOAT32>(0);
134 comp_dwconv2d.output(constant_data);
135 comp_dwconv2d.compute();
136
137 loco::replace(node).with(constant);
138
139 return true;
140}

147bool FoldDepthwiseConv2DPass::run(loco::Graph *g)
148{
149 bool changed = false;
150 for (auto node : loco::active_nodes(loco::output_nodes(g)))
151 {
152 auto depthwise_conv2d = dynamic_cast<CircleDepthwiseConv2D *>(node);
153
154 if (depthwise_conv2d == nullptr)
155 continue;
156
157 switch (depthwise_conv2d->dtype())
158 {
159 case loco::DataType::FLOAT32:
160 changed = fold_depthwise_conv_2d(depthwise_conv2d);
161 break;
162 default:
163 break;
164 }
165 }
166
167 return changed;
168}
