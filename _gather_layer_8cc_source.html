31void KernelGenerator::visit(const ir::operation::Gather &node)
32{
33 const auto output_index{node.getOutputs().at(0)};
34 const auto input_index{node.getInputs().at(ir::operation::Gather::Input::INPUT)};
35 const auto indices_index{node.getInputs().at(ir::operation::Gather::Input::INDICES)};
36
37 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
38 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
39 auto indices_tensor = _tensor_reg->getPortableTensor(indices_index);
40
41 const auto rank = _ctx.at(input_index).shape().rank();
42 const auto axis = ops::getAxis(rank, node.param().axis);
43
44 auto fn = std::make_unique<ops::GatherLayer>();
45
46 fn->configure(input_tensor, indices_tensor, output_tensor, axis, _external_context.get());
47
48 _return_fn = std::move(fn);
49}

69template <typename InputType> void GatherLayer::runByInputType()
70{
71 using OutputType = InputType;
72 nnfw::cker::GatherParams op_params;
73 op_params.axis = _axis;
74
75 switch (_indices->data_type())
76 {
77 case OperandType::INT32:
78 {
79 using IndicesType = int32_t;
80
81 nnfw::cker::Gather<InputType, IndicesType>(
82 op_params, getShape(_input), getBuffer<InputType>(_input), getShape(_indices),
83 getBuffer<IndicesType>(_indices), getShape(_output), getBuffer<OutputType>(_output));
84 break;
85 }
86 case OperandType::INT64:
87 {
88 using IndicesType = int64_t;
89
90 nnfw::cker::Gather<InputType, IndicesType>(
91 op_params, getShape(_input), getBuffer<InputType>(_input), getShape(_indices),
92 getBuffer<IndicesType>(_indices), getShape(_output), getBuffer<OutputType>(_output));
93 break;
94 }
95 default:
96 throw std::runtime_error("Gather: unsupported indices data type");
97 }

100void GatherLayer::runByGGMLQuantInputType()
101{
102 // Supporting condition
103 // Input: rank 2
104 // Indice: rank < 4 or rank 4 with dim(0) = 1, INT32
105 // Axis: 0
106 if (getShape(_input).DimensionsCount() != 2)
107 throw std::runtime_error("Gather: block quantized input tensor must be rank 2");
108
109 if (getShape(_indices).DimensionsCount() >= 4 &&
110 (getShape(_indices).DimensionsCount() != 4 || getShape(_indices).Dims(0) != 1))
111 throw std::runtime_error("Gather: invalid indices tensor shape");
112
113 if (_indices->data_type() != ir::DataType::INT32)
114 throw std::runtime_error("Gather: indices tensor must be int32 type");
115
116 if (_axis != 0)
117 throw std::runtime_error("Gather: axis must be 0");
118
119 // convert tensor
120 auto input = getGGMLTensor(_input);
121 auto indices = getGGMLTensor(_indices);
122 auto output = getGGMLTensor(_output);
123 {
124 output.op = GGML_OP_GET_ROWS;
125 output.src[0] = &input;
126 output.src[1] = &indices;
127 }
128 auto *nodes = &output;
129
130 // create graph
131 struct ggml_cgraph graph;
132 {
133 memset(&graph, 0, sizeof(graph));
134 graph.n_nodes = 1;
135 graph.nodes = &nodes;
136 }
137
138 // get cplan
139 auto cplan = ggml_graph_plan(&graph, _ctx->maxNumThreads());
140 std::vector<uint8_t> buf(cplan.work_size);
141 cplan.work_data = buf.data();
142
143 // compute
144 ggml_graph_compute(&graph, &cplan);
145}

147void GatherLayer::run()
148{
149 switch (_input->data_type())
150 {
151 case OperandType::FLOAT32:
152 runByInputType<float>();
153 break;
154 case OperandType::QUANT_UINT8_ASYMM:
155 runByInputType<uint8_t>();
156 break;
157 case OperandType::INT32:
158 runByInputType<int32_t>();
159 break;
160 case OperandType::QUANT_GGML_Q4_0:
161 runByGGMLQuantInputType();
162 break;
163 case OperandType::BOOL8:
164 runByInputType<bool>();
165 break;
166 default:
167 throw std::runtime_error("Gather: unsupported input data type");
168 }
169}
