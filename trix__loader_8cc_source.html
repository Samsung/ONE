64void TrixMetaReader::init(const char *path)
65{
66 assert(path);
67 _meta = getNPUmodel_metadata(path, false);
68 if (_meta == nullptr)
69 {
70 throw std::runtime_error("Failed to get TRIX model metadata");
71 }
72 if (NPUBIN_VERSION(_meta->magiccode) != 3)
73 {
74 throw std::runtime_error("TRIX model metadata version mismatched.");
75 }

141void TrixLoader::loadOperationIO(ir::OperandIndexSequence &inputs,

155void TrixLoader::loadBulk(ir::Graph &subg)
156{
157 ir::operation::Bulk::Param param;
158 param.binary_path = _model_path;
159 param.origin_input_shapes = _origin_input_shapes;
160 param.origin_output_shapes = _origin_output_shapes;
161
162 ir::OperandIndexSequence inputs;
163 ir::OperandIndexSequence outputs;
164
165 loadOperationIO(inputs, outputs);
166
167 std::unique_ptr<ir::operation::Bulk> bulk(new ir::operation::Bulk(inputs, outputs, param));
168 subg.addOperation(std::move(bulk));
169}

171ir::OperandIndex TrixLoader::loadOperandFromInput(uint32_t idx, ir::Graph &subg)
172{
173 // Shape
174 ir::Shape shape;
175 for (uint32_t d = 0; d < MAX_RANK; ++d)
176 shape.append(_meta.input_seg_dims(idx, d));
177
178 // TypeInfo
179 ir::TypeInfo type_info(toDataType(_meta.input_seg_quant_type(idx)),
180 _meta.input_seg_quant_scale(idx), _meta.input_seg_quant_zp(idx));
181
182 _origin_input_shapes.push_back(shape);
183 // Create operand
184 const auto operand_index = subg.addOperand(shape, type_info);
185 return operand_index;
186}

205void TrixLoader::loadOperands(ir::Graph &subg)
206{
207 auto in_num = _meta.input_seg_num();
208 for (uint32_t i = 0; i < in_num; ++i)
209 {
210 loadOperandFromInput(i, subg);
211 }
212 auto out_num = _meta.output_seg_num();
213 for (uint32_t i = 0; i < out_num; ++i)
214 {
215 loadOperandFromOutput(i, subg);
216 }
217}

219std::unique_ptr<ir::Graph> TrixLoader::loadSubgraph()
220{
221 auto subg = std::make_unique<ir::Graph>();
222 _meta.init(_model_path.c_str());
223
224 // Load tensors
225 loadOperands(*subg);
226
227 // Set inputs
228 for (uint32_t i = 0; i < _meta.input_seg_num(); ++i)
229 {
230 subg->addInput(inputIdxToOperandIdx(i), "tvn_input" + std::to_string(i));
231 }
232 // Set outputs
233 for (uint32_t i = 0; i < _meta.output_seg_num(); ++i)
234 {
235 subg->addOutput(outputIdxToOperandIdx(i), "tvn_out" + std::to_string(i));
236 }
237 // Create operations
238 loadBulk(*subg);
239
240 // TODO: NHWC only supported at this moment.
241 subg->setLayout(ir::Layout::NHWC);
242 subg->verify();
243 return subg;
244}

246void TrixLoader::loadModel()
247{
248 // one subgraph only
249 auto subg = loadSubgraph();
250 _model->push(ir::SubgraphIndex(0), std::move(subg));
251}

253void TrixLoader::loadFromFile(const std::string &file_path)
254{
255 // model path will be used to set Bulk param
256 _model_path = file_path;
257 // metadata is initialized from model path since it is loadFromFile
258 _meta.init(_model_path.c_str());
259 loadModel();
260}
