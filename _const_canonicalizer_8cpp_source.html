29bool canonicalize_const(loco::Graph *graph, moco::TFConst *node)
30{
31 LOGGER(l);
32
48 INFO(l) << "TFNodeCanonicalize TFConst begin";
49
50 auto const_node = graph->nodes()->create<loco::ConstGen>();
51
52 // copy properties
53 auto dtype = node->dtype();
54 const_node->dtype(dtype);
55
56 auto rank = node->rank();
57
58 if (rank == 0)
59 {
60 // This routine implements a workaround that converts a scalar constant (rank-0 tensor)
61 // into a rank-1 tensor of shape [1].
62 //
63 // TODO Revise this implementation later
64 const_node->rank(1);
65 const_node->dim(0) = 1;
66 }
67 else
68 {
69 const_node->rank(rank);
70
71 for (uint32_t r = 0; r < rank; ++r)
72 {
73 if (node->dim(r).known())
74 const_node->dim(r) = node->dim(r);
75 else
76 const_node->dim(r).unset();
77 }
78 }
79
80 switch (dtype)
81 {
82 case loco::DataType::S32:
83 {
84 uint32_t input_elements = node->size<loco::DataType::S32>();
85 const_node->size<loco::DataType::S32>(input_elements);
86 for (uint32_t i = 0; i < input_elements; ++i)
87 {
88 const_node->at<loco::DataType::S32>(i) = node->at<loco::DataType::S32>(i);
89 }
90 break;
91 }
92 case loco::DataType::FLOAT32:
93 {
94 uint32_t input_elements = node->size<loco::DataType::FLOAT32>();
95 const_node->size<loco::DataType::FLOAT32>(input_elements);
96 for (uint32_t i = 0; i < input_elements; ++i)
97 {
98 const_node->at<loco::DataType::FLOAT32>(i) = node->at<loco::DataType::FLOAT32>(i);
99 }
100 break;
101 }
102 default:
103 throw oops::UserExn("Const has unsupported data type", node->name());
104 }
105
106 // update graph
107 replace(node).with(const_node);
108
109 INFO(l) << "TFNodeCanonicalize TFConst done";
110
111 return true;
112}
