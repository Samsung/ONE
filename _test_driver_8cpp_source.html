32uint32_t tensor_size_of(const luci::CircleNode *node)
33{
34 uint32_t tensor_size = loco::size(node->dtype());
35 for (uint32_t i = 0; i < node->rank(); ++i)
36 tensor_size *= node->dim(i).value();
37 return tensor_size;
38}

40std::vector<uint8_t> random_data_for(const luci::CircleInput *node)
41{
42 // allocate data buffer
43 std::vector<uint8_t> inputs_data(tensor_size_of(node));
44 auto *buffer = inputs_data.data();
45
46 // define size of buffer in elements
47 const auto dtype = node->dtype();
48 assert(inputs_data.size() % loco::size(dtype) == 0); // FIX ME UNLESS
49 const auto element_count = inputs_data.size() / loco::size(dtype);
50
51 // random generator engine
52 std::random_device device;
53 std::mt19937 engine{device()};
54
55 // fill buffer with random data
56 switch (node->dtype())
57 {
58 case loco::DataType::FLOAT32:
59 {
60 auto element_buffer = reinterpret_cast<float *>(buffer);
61
62 std::uniform_real_distribution<float> distrib(-3, 3);
63 const auto generator = [&distrib, &engine]() { return distrib(engine); };
64 std::generate(element_buffer, element_buffer + element_count, generator);
65
66 break;
67 }
68 case loco::DataType::U8:
69 {
70 auto element_buffer = buffer;
71
72 std::uniform_int_distribution<uint8_t> distrib(100, 200);
73 const auto generator = [&distrib, &engine]() { return distrib(engine); };
74 std::generate(element_buffer, element_buffer + element_count, generator);
75
76 break;
77 }
78 case loco::DataType::S16:
79 {
80 auto element_buffer = reinterpret_cast<int16_t *>(buffer);
81
82 std::uniform_int_distribution<int16_t> distrib(0, 100);
83 const auto generator = [&distrib, &engine]() { return distrib(engine); };
84 std::generate(element_buffer, element_buffer + element_count, generator);
85
86 break;
87 }
88 case loco::DataType::S32:
89 {
90 auto element_buffer = reinterpret_cast<int32_t *>(buffer);
91
92 std::uniform_int_distribution<int32_t> distrib(0, 100);
93 const auto generator = [&distrib, &engine]() { return distrib(engine); };
94 std::generate(element_buffer, element_buffer + element_count, generator);
95
96 break;
97 }
98 case loco::DataType::BOOL:
99 {
100 // num of bool data type is equivalent to uint8_t num in [0, 1] range
101 auto element_buffer = buffer;
102
103 std::uniform_int_distribution<uint8_t> distrib(0, 1);
104 const auto generator = [&distrib, &engine]() { return distrib(engine); };
105 std::generate(element_buffer, element_buffer + element_count, generator);
106
107 break;
108 }
109 default:
110 // TODO Support other dtypes
111 throw std::runtime_error("Unsupported data type, yet!");
112 }
113
114 return inputs_data;
115}

119int entry(int argc, char **argv)
120{
121 // check arguments
122 if (argc != 3 || std::string(argv[1]) != "--model")
123 {
124 std::cerr << "Usage: " << argv[0] << " --model <path/to/model>" << std::endl;
125 return EXIT_FAILURE;
126 }
127
128 // open file with model
129 const auto model_file = std::string(argv[2]);
130 std::ifstream fs(model_file, std::ifstream::binary);
131 if (fs.fail())
132 {
133 std::cerr << "Cannot open model file \"" << model_file << "\"." << std::endl;
134 return EXIT_FAILURE;
135 }
136
137 // create constant circle model
138 const std::vector<char> model_buffer((std::istreambuf_iterator<char>(fs)),
139 std::istreambuf_iterator<char>());
140 const auto circle_model = circle::GetModel(model_buffer.data());
141
142 // create random model's inputs
143 std::vector<std::vector<uint8_t>> inputs_data;
144 {
145 // model inputs
146 auto model = luci::Importer(nullptr).importModule(circle_model);
147 const auto inputs = loco::input_nodes(model->graph());
148
149 // create random data for each input
150 for (const auto *input : inputs)
151 {
152 const auto input_node = loco::must_cast<const luci::CircleInput *>(input);
153 inputs_data.emplace_back(random_data_for(input_node));
154 }
155 }
156
157 // interpret given module
158 const auto interpret_module_and_compute_output =
159 [&](const std::unique_ptr<luci::Module> &module) {
160 // create interpreter
161 luci_interpreter::Interpreter interpreter(module.get());
162
163 // model's input and output nodes
164 const auto input_nodes = loco::input_nodes(module->graph());
165 const auto output_nodes = loco::output_nodes(module->graph());
166
167 // set inputs
168 for (uint32_t i = 0; i < input_nodes.size(); ++i)
169 {
170 const auto input_node = loco::must_cast<const luci::CircleInput *>(input_nodes[i]);
171 const auto &data = inputs_data.at(i);
172 interpreter.writeInputTensor(input_node, data.data(), data.size());
173 }
174
175 // do inference
176 interpreter.interpret();
177
178 // read outputs
179 std::vector<std::vector<uint8_t>> outputs_data;
180 for (const auto *node : output_nodes)
181 {
182 const auto output_node = loco::must_cast<const luci::CircleOutput *>(node);
183
184 // allocate output buffer
185 outputs_data.emplace_back(tensor_size_of(output_node));
186
187 auto &data = outputs_data.back();
188 interpreter.readOutputTensor(output_node, data.data(), data.size());
189 }
190
191 return outputs_data;
192 };
193
194 // import with copying, execute and save
195 std::vector<std::vector<uint8_t>> outputs_data_1;
196 {
197 const auto default_source = &luci::GraphBuilderRegistry::get();
198 const auto module = luci::Importer(default_source).importModule(circle_model);
199 if (not module)
200 {
201 std::cerr << "Fail to import model with constant copying." << std::endl;
202 return EXIT_FAILURE;
203 }
204
205 outputs_data_1 = interpret_module_and_compute_output(module);
206 }
207
208 // import without copying, execute and save
209 std::vector<std::vector<uint8_t>> outputs_data_2;
210 {
211 const auto optimized_source = luci_interpreter::source_without_constant_copying();
212 const auto module = luci::Importer(optimized_source.get()).importModule(circle_model);
213 if (not module)
214 {
215 std::cerr << "Fail to import model without constant copying." << std::endl;
216 return EXIT_FAILURE;
217 }
218
219 outputs_data_2 = interpret_module_and_compute_output(module);
220 }
221
222 // check all tensors are equal
223 assert(outputs_data_1.size() == outputs_data_2.size());
224 for (uint32_t n = 0; n < outputs_data_1.size(); ++n)
225 {
226 const auto &output_1 = outputs_data_1.at(n);
227 const auto &output_2 = outputs_data_2.at(n);
228 assert(output_1.size() == output_2.size());
229
230 for (uint32_t o = 0; o < output_1.size(); ++o)
231 {
232 if (output_1[o] != output_2[o])
233 {
234 std::cerr << "Values mismatch in model's output number " << n << std::endl;
235 return EXIT_FAILURE;
236 }
237 }
238 }
239
240 std::cout << "[TEST PASSED]" << std::endl;
241 return EXIT_SUCCESS;
242}
