37bool valid_axis_range(int32_t output_rank, int32_t pack_axis)
38{
39 // check axis range in [-r-1, r+1)
40 assert(output_rank > 0);
41 return (-output_rank <= pack_axis) && (pack_axis < output_rank);
42}

44bool constantfold_pack(moco::TFPack *node)
45{
46 // check if all the inputs are Const
47 std::vector<moco::TFConst *> input_nodes;
48 uint32_t num = node->N();
49
50 for (uint32_t index = 0; index < num; ++index)
51 {
52 auto in = dynamic_cast<moco::TFConst *>(node->values(index));
53 if (in == nullptr)
54 return false;
55
56 input_nodes.push_back(in);
57 }
58 assert(input_nodes.size() == num);
59
60 // check if all inputs have same shape and dtype
61 auto input_0 = input_nodes.at(0);
62 auto shape_0 = moco::tensor_shape(input_0);
63 auto dtype_0 = input_0->dtype();
64 if (dtype_0 != loco::DataType::S32 && dtype_0 != loco::DataType::FLOAT32)
65 {
66 // TODO support other types
67 assert(false);
68 return false;
69 }
70 for (uint32_t index = 1; index < num; ++index)
71 {
72 auto input_i = input_nodes.at(index);
73 auto shape_i = moco::tensor_shape(input_i);
74 auto dtype_i = input_i->dtype();
75 if (!(shape_0 == shape_i))
76 return false;
77 if (dtype_0 != dtype_i)
78 return false;
79 }
80
81 int32_t output_rank = static_cast<int32_t>(shape_0.rank() + 1);
82 int32_t pack_axis = node->axis();
83 if (!valid_axis_range(output_rank, pack_axis))
84 {
85 throw oops::UserExn("axis is out of range: ", node->name());
86 }
87
88 if (pack_axis < 0)
89 {
90 pack_axis = output_rank + pack_axis;
91 }
92
93 // define output shape
94 loco::TensorShape output_shape;
95 output_shape.rank(output_rank);
96
97 for (int32_t r = 0, s = 0; r < output_rank; ++r)
98 {
99 if (r == pack_axis)
100 {
101 output_shape.dim(r).set(num);
102 }
103 else
104 {
105 output_shape.dim(r).set(shape_0.dim(s++).value());
106 }
107 }
108
109 auto graph = node->graph();
110
111 // create new constant
112 auto output_const = moco::new_const(graph, output_shape, input_0->dtype());
113
114 moco::TensorPackEnumerator etor;
115
116 etor.shape(shape_0, output_shape);
117 etor.axis(pack_axis);
118 for (etor.start(); etor.valid(); etor.advance())
119 {
120 uint32_t inp_num = etor.inp_num();
121 uint32_t inp_element = etor.inp_element();
122 uint32_t out_element = etor.out_element();
123
124 auto inp_const = input_nodes[inp_num];
125
126 if (input_0->dtype() == loco::DataType::S32)
127 {
128 int32_t val = inp_const->at<loco::DataType::S32>(inp_element);
129 output_const->at<loco::DataType::S32>(out_element) = val;
130 }
131 else if (input_0->dtype() == loco::DataType::FLOAT32)
132 {
133 float val = inp_const->at<loco::DataType::FLOAT32>(inp_element);
134 output_const->at<loco::DataType::FLOAT32>(out_element) = val;
135 }
136 }
137
138 // replace
139 loco::replace(node).with(output_const);
140
141 return true;
142}

165bool ConstantFoldPack::run(loco::Graph *graph)
166{
167 bool changed = false;
168 for (auto node : loco::active_nodes(loco::output_nodes(graph)))
169 {
170 if (auto pack_node = as<moco::TFPack>(node))
171 {
172 if (constantfold_pack(pack_node))
173 changed = true;
174 }
175 }
176
177 return changed;
178}
