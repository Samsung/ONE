35void MirrorPad::configure()
36{
37 const Shape &input_shape = input()->shape();
38 const int num_dims = input_shape.num_dims();
39
40 if (num_dims > 4)
41 throw std::runtime_error("Unsupported number of dimensions.");
42
43 assert(output()->element_type() == input()->element_type());
44 assert(paddings()->element_type() == DataType::S32);
45 // Paddings shape should be [N, 2].
46 assert(paddings()->shape().num_dims() == 2);
47 assert(paddings()->shape().dim(0) == num_dims);
48 assert(paddings()->shape().dim(1) == 2);
49
50 Shape output_shape(num_dims);
51 const auto *paddings_data = getTensorData<int32_t>(paddings());
52 for (int i = 0; i < num_dims; ++i)
53 {
54 const int32_t padding_before = paddings_data[i * 2];
55 const int32_t padding_after = paddings_data[i * 2 + 1];
56 assert(padding_before >= 0 && padding_after >= 0);
57 output_shape.dim(i) = input_shape.dim(i) + padding_before + padding_after;
58 }
59
60 output()->resize(output_shape);
61}

67void MirrorPad::execute() const
68{
69 switch (input()->element_type())
70 {
71 case DataType::FLOAT32:
72 {
73 MirrorPadImpl<float>(*input(), *paddings(), params().mode, *output());
74 break;
75 }
76 case DataType::U8:
77 {
78 assert(output()->zero_point() >= std::numeric_limits<uint8_t>::min());
79 assert(output()->zero_point() <= std::numeric_limits<uint8_t>::max());
80
81 MirrorPadImpl<uint8_t>(*input(), *paddings(), params().mode, *output());
82 break;
83 }
84 default:
85 throw std::runtime_error("Unsupported type.");
86 }
87}

89template <typename T>
90inline void MirrorPadImpl(const Tensor &input, const Tensor &paddings, MirrorPadMode mode,
91 Tensor &output)
92{
93 auto const input_dims = input.shape().num_dims();
94 auto const input_data = input.data<T>();
95 auto const paddings_data = paddings.data<int32_t>();
96 auto const output_data = output.data<T>();
97
98 auto const input_b = input_dims > 3 ? input.shape().dim(input_dims - 4) : 1;
99 auto const input_h = input_dims > 2 ? input.shape().dim(input_dims - 3) : 1;
100 auto const input_w = input_dims > 1 ? input.shape().dim(input_dims - 2) : 1;
101 auto const input_d = input.shape().dim(input_dims - 1);
102
103 auto const input_h_offset = input_d * input_w;
104 auto const input_b_offset = input_h_offset * input_h;
105
106 auto const output_b = input_dims > 3 ? output.shape().dim(input_dims - 4) : 1;
107 auto const output_h = input_dims > 2 ? output.shape().dim(input_dims - 3) : 1;
108 auto const output_w = input_dims > 1 ? output.shape().dim(input_dims - 2) : 1;
109 auto const output_d = output.shape().dim(input_dims - 1);
110
111 auto const left_b_pad = paddings_data[2 * (input_dims - 4)];
112 auto const left_h_pad = paddings_data[2 * (input_dims - 3)];
113 auto const left_w_pad = paddings_data[2 * (input_dims - 2)];
114 auto const left_d_pad = paddings_data[2 * (input_dims - 1)];
115
116 auto const right_b_pad = paddings_data[2 * (input_dims - 4) + 1];
117 auto const right_h_pad = paddings_data[2 * (input_dims - 3) + 1];
118 auto const right_w_pad = paddings_data[2 * (input_dims - 2) + 1];
119 auto const right_d_pad = paddings_data[2 * (input_dims - 1) + 1];
120
121 const auto positive_mod = [](auto a, auto b) { return (a % b + b) % b; };
122 const auto offset_index = [input_d, input_h_offset, input_b_offset](auto d, auto w, auto h,
123 auto b) {
124 return d + w * input_d + h * input_h_offset + b * input_b_offset;
125 };
126
127 const auto symmetric_dim = [&positive_mod](auto i, auto left_pad, auto input) {
128 bool reflected = (((i < left_pad ? i + 1 - input : i) - left_pad) / input & 1) == 1;
129 return positive_mod(reflected ? input + left_pad - i - 1 : i - left_pad, input);
130 };
131
132 const T *in_ptr = input_data;
133 T *out_ptr = output_data;
134
135 for (int32_t b = 0; b < output_b; ++b)
136 {
137 for (int32_t h = 0; h < output_h; ++h)
138 {
139 for (int32_t w = 0; w < output_w; ++w)
140 {
141 for (int32_t d = 0; d < output_d; ++d)
142 {
143 if (b < left_b_pad || b >= output_b - right_b_pad || //
144 h < left_h_pad || h >= output_h - right_h_pad || //
145 w < left_w_pad || w >= output_w - right_w_pad || //
146 d < left_d_pad || d >= output_d - right_d_pad)
147 {
148 if (mode == MirrorPadMode::REFLECT)
149 {
150 *out_ptr++ = input_data[offset_index(
151 positive_mod(d - left_d_pad, input_d), positive_mod(w - left_w_pad, input_w),
152 positive_mod(h - left_h_pad, input_h), positive_mod(b - left_b_pad, input_b))];
153 }
154 else
155 {
156 *out_ptr++ = input_data[offset_index(
157 symmetric_dim(d, left_d_pad, input_d), symmetric_dim(w, left_w_pad, input_w),
158 symmetric_dim(h, left_h_pad, input_h), symmetric_dim(b, left_b_pad, input_b))];
159 }
160 }
161 else
162 {
163 *out_ptr++ = *in_ptr++;
164 }
165 }
166 }
167 }
168 }
169}
