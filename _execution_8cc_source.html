29Execution::Execution(const std::shared_ptr<IExecutors> &executors) : _executors{executors}
30{
31 assert(executors != nullptr);
32 assert(executors->entryExecutor() != nullptr);
33
34 // Initialize I/O description
35 for (uint32_t i = 0; i < _executors->inputSize(); ++i)
36 _ctx.desc.inputs.emplace_back(_executors->inputInfo(ir::IOIndex(i)));
37
38 for (uint32_t i = 0; i < _executors->outputSize(); ++i)
39 _ctx.desc.outputs.emplace_back(_executors->outputInfo(ir::IOIndex(i)));
40 _ctx.shape_updated = false;
41
42 _is_internal_output_tensor.resize(_executors->outputSize());
43 for (uint32_t i = 0; i < _executors->outputSize(); ++i)
44 {
45 const auto output_tensor =
46 dynamic_cast<const backend::builtin::IOTensor *>(executors->outputTensor(ir::IOIndex{i}));
47 if (!output_tensor)
48 throw std::runtime_error("Output tensor must be IOTensor");
49
50 _is_internal_output_tensor.at(i) = output_tensor->hasBackendTensor();
51 }
52
53 // Initialize options
54 ExecutionOptions::fromGlobalConfig(_ctx.options);
55}

57void Execution::changeInputShape(const ir::IOIndex &index, const ir::Shape &new_shape)
58{
59 // This will be used later to set input tensor dynamic
60 // Note that 'compiled' model will not be updated with new_shape
61 // but new_shape will change model input shape while 'running' the model
62 auto &input_desc = _ctx.desc.inputs.at(index.value());
63 if (new_shape != input_desc.info.shape())
64 {
65 input_desc.info.shape(new_shape);
66 _ctx.shape_updated = true;
67
68 VERBOSE(Execution) << "Model input shape will be changed at the start of execute()"
69 << "(index: " << index << ")" << std::endl;
70 }
71}

92void Execution::execute()
93{
94 VERBOSE(Execution) << "Start execution" << std::endl;
95
96 // Input length validation check
97 for (const auto &input : _ctx.desc.inputs)
98 {
99 if (input.info.total_size() > input.size)
100 throw std::runtime_error{"Too small input buffer length"};
101 }
102
103 // Output length validation check
104 if (!_ctx.shape_updated)
105 {
106 assert(_ctx.desc.outputs.size() == _is_internal_output_tensor.size());
107 for (uint32_t i = 0; i < _ctx.desc.outputs.size(); ++i)
108 {
109 const bool is_managed_internally = _is_internal_output_tensor.at(i);
110 const auto &output = _ctx.desc.outputs.at(i);
111 if (!is_managed_internally && output.info.total_size() > output.size)
112 throw std::runtime_error{"Too small output buffer length"};
113 if (is_managed_internally && output.buffer != nullptr)
114 VERBOSE(Execution) << "Warning: Output buffer was set from API even though the output "
115 "tensor was allocated internally"
116 << std::endl;
117 }
118 }
119
120 _executors->execute(_ctx);
121 finished = true;
122
123 VERBOSE(Execution) << "Execution finished" << std::endl;
124}
