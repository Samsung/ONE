29Execution::Execution(const std::shared_ptr<IExecutors> &executors) : _executors{executors}
30{
31 assert(executors != nullptr);
32 assert(executors->entryExecutor() != nullptr);
33
34 // Initialize I/O description
35 _ctx.desc.inputs.resize(_executors->inputSize());
36 for (uint32_t i = 0; i < _executors->inputSize(); ++i)
37 _ctx.desc.inputs.at(i) = std::make_unique<InputDesc>(_executors->inputInfo(ir::IOIndex(i)));
38
39 _ctx.desc.outputs.resize(_executors->outputSize());
40 for (uint32_t i = 0; i < _executors->outputSize(); ++i)
41 _ctx.desc.outputs.at(i) = std::make_unique<OutputDesc>(_executors->outputInfo(ir::IOIndex(i)));
42 _ctx.shape_updated = false;
43
44 _is_internal_output_tensor.resize(_executors->outputSize());
45 for (uint32_t i = 0; i < _executors->outputSize(); ++i)
46 {
47 const auto output_tensor =
48 dynamic_cast<const backend::builtin::IOTensor *>(executors->outputTensor(ir::IOIndex{i}));
49 if (!output_tensor)
50 throw std::runtime_error("Output tensor must be IOTensor");
51
52 _is_internal_output_tensor.at(i) = output_tensor->hasBackendTensor();
53 }
54
55 // Initialize options
56 ExecutionOptions::fromGlobalConfig(_ctx.options);
57}

59void Execution::changeInputShape(const ir::IOIndex &index, const ir::Shape &new_shape)
60{
61 // This will be used later to set input tensor dynamic
62 // Note that 'compiled' model will not be updated with new_shape
63 // but new_shape will change model input shape while 'running' the model
64 auto &input_desc = _ctx.desc.inputs.at(index.value());
65 if (new_shape != input_desc->info.shape())
66 {
67 input_desc->info.shape(new_shape);
68 _ctx.shape_updated = true;
69
70 VERBOSE(Execution) << "Model input shape will be changed at the start of execute()"
71 << "(index: " << index << ")" << std::endl;
72 }
73}

132void Execution::execute()
133{
134 VERBOSE(Execution) << "Start execution" << std::endl;
135
136 // Input length validation check
137 for (const auto &input : _ctx.desc.inputs)
138 {
139 if (input->info.total_size() > input->size)
140 throw std::runtime_error{"Too small input buffer length"};
141 }
142
143 // Output length validation check
144 if (!_ctx.shape_updated)
145 {
146 assert(_ctx.desc.outputs.size() == _is_internal_output_tensor.size());
147 for (uint32_t i = 0; i < _ctx.desc.outputs.size(); ++i)
148 {
149 const bool is_managed_internally = _is_internal_output_tensor.at(i);
150 const auto &output = _ctx.desc.outputs.at(i);
151 if (!is_managed_internally && output->info.total_size() > output->size)
152 throw std::runtime_error{"Too small output buffer length"};
153 if (is_managed_internally && output->buffer != nullptr)
154 VERBOSE(Execution) << "Warning: Output buffer was set from API even though the output "
155 "tensor was allocated internally"
156 << std::endl;
157 }
158 }
159
160 _executors->execute(_ctx);
161 finished = true;
162
163 VERBOSE(Execution) << "Execution finished" << std::endl;
164}
