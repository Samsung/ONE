33void change_single_input_shape(luci::CircleInput *circle_input, const Shape &new_shape)
34{
35 circle_input->rank(new_shape.rank());
36 for (uint32_t i = 0; i < new_shape.rank(); ++i)
37 {
38 if (new_shape[i].is_dynamic())
39 {
40 circle_input->dim(i) = loco::Dimension(); // empty ctor means dynamic dimension
41 }
42 else
43 {
44 // a value here can be in range (0, std::numeric_limits<int32_t>::max()) so the cast is safe
45 circle_input->dim(i) = loco::Dimension(static_cast<uint32_t>(new_shape[i].value()));
46 }
47 }
48}

50void change_inputs_shapes(loco::Graph *graph, const std::vector<Shape> &new_inputs_shapes)
51{
52 auto graph_inputs = loco::input_nodes(graph);
53 if (graph_inputs.size() != new_inputs_shapes.size())
54 {
55 throw std::runtime_error("Expected " + std::to_string(graph_inputs.size()) +
56 " shapes but provided " + std::to_string(new_inputs_shapes.size()));
57 }
58 for (size_t in_idx = 0; in_idx < new_inputs_shapes.size(); ++in_idx)
59 {
60 auto circle_input = loco::must_cast<luci::CircleInput *>(graph_inputs[in_idx]);
61 change_single_input_shape(circle_input, new_inputs_shapes[in_idx]);
62 }
63}

72ModelEditor &ModelEditor::resize_inputs(const std::vector<Shape> &new_inputs_shapes)
73{
74 auto graph = _circle_model->module()->graph();
75 change_inputs_shapes(graph, new_inputs_shapes);
76
77 logo::Phase phase;
78 phase.emplace_back(std::make_unique<logo::RemoveDeadNodeWithQueryPass>());
79 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
80 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
81
82 logo::PhaseRunner<logo::PhaseStrategy::Restart> phase_runner{graph};
83 try
84 {
85 phase_runner.run(phase);
86 }
87 catch (const std::exception &e)
88 {
89 throw std::runtime_error("Exception during resizing with message: " + std::string{e.what()});
90 }
91
92 return *this;
93}
