58inline void FullyConnected16x1Float32(const FullyConnectedParams &params, const Shape &input_shape,
59 const float *input_data, const Shape &weights_shape,
60 const float *weights_data, const Shape &,
61 const float *bias_data, const Shape &, float *output_data)
62{
63 int total_input_size = input_shape.FlatSize();
64 int input_size = weights_shape.Dims(1);
65 const int batch_size = total_input_size / input_size;
66 const int num_units = weights_shape.Dims(0);
67
68 float *out = output_data;
69 const float *weights = weights_data;
70 int rows = num_units;
71 int cols = input_size;
72 int col_stride = input_size;
73 const float *x = input_data;
74
75 // Output = bias if bias tensor exists.
76 if (bias_data)
77 {
78 VectorBatchVectorAssign(bias_data, num_units, batch_size, output_data);
79 }
80 else
81 {
82 ZeroVector(output_data, batch_size * num_units);
83 }
84
85 // rows : out, cols : in
86 int i, j;
87 for (i = 0; i < rows; i += 16)
88 {
89 const float *w = &weights[i * col_stride];
90
91 /* keep y[0..15] in registers for duration of inner loop */
92 float *__restrict y = &out[i];
93
94 float32x4_t y0_3 = vld1q_f32(&y[0]);
95 float32x4_t y4_7 = vld1q_f32(&y[4]);
96 float32x4_t y8_11 = vld1q_f32(&y[8]);
97 float32x4_t y12_15 = vld1q_f32(&y[12]);
98
99 for (j = 0; j < cols; j++)
100 {
101 float32x4_t wvec0_3, wvec4_7, wvec8_11, wvec12_15;
102 float32x4_t xj;
103
104 xj = vld1q_dup_f32(&x[j]);
105
106 wvec0_3 = vld1q_f32(&w[0]);
107 y0_3 = vmlaq_f32(y0_3, wvec0_3, xj);
108 wvec4_7 = vld1q_f32(&w[4]);
109 y4_7 = vmlaq_f32(y4_7, wvec4_7, xj);
110 wvec8_11 = vld1q_f32(&w[8]);
111 y8_11 = vmlaq_f32(y8_11, wvec8_11, xj);
112 wvec12_15 = vld1q_f32(&w[12]);
113 y12_15 = vmlaq_f32(y12_15, wvec12_15, xj);
114
115 w += 16;
116 }
117
118 /* save y[0..15] back to memory */
119
120 vst1q_f32(&y[0], y0_3);
121 vst1q_f32(&y[4], y4_7);
122 vst1q_f32(&y[8], y8_11);
123 vst1q_f32(&y[12], y12_15);
124 }
125 if (params.activation != FusedActivationFunctionType::kNone)
126 {
127 // Apply activation function
128 ApplyActivationToVector(output_data, batch_size * num_units, params.activation, output_data);
129 }
130}
