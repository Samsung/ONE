26inline void Softmax(const float *input_data, const Dims<4> &input_dims, float beta,
27 float *output_data, const Dims<4> &output_dims)
28{
29 MatchingArraySize(input_dims, 3, output_dims, 3);
30 MatchingArraySize(input_dims, 2, output_dims, 2);
31 MatchingArraySize(input_dims, 1, output_dims, 1);
32 MatchingArraySize(input_dims, 0, output_dims, 0);
33
34 const auto in_mat = MapAsMatrixWithFirstDimAsRows(input_data, input_dims);
35 auto out_mat = MapAsMatrixWithFirstDimAsRows(output_data, output_dims);
36 // Compute the exponential first, removing the max coefficient for numerical
37 // stability.
38 out_mat = (in_mat.rowwise() - in_mat.colwise().maxCoeff()).array() * beta;
39 // We are separating out the exp function so that exp can be vectorized.
40 out_mat = out_mat.array().exp();
41 // Normalize to get the activations.
42 Eigen::Array<float, 1, Eigen::Dynamic> scale = out_mat.array().colwise().sum().inverse();
43 out_mat.array().rowwise() *= scale;
44}

46bool softmaxFloat32(const float *inputData, const Shape &inputShape, const float beta,
47 float *outputData, const Shape &outputShape)
48{
49 Dims<4> dim;
50 if (getNumberOfDimensions(inputShape) == 2)
51 {
52 uint32_t batch_size = getSizeOfDimension(inputShape, 0);
53 uint32_t input_size = getNumberOfElements(inputShape) / batch_size;
54
55 Shape shapeIn4D;
56 shapeIn4D.dimensions = {batch_size, 1, 1, input_size};
57 dim = convertShapeToDims(shapeIn4D);
58 }
59 else if (getNumberOfDimensions(inputShape) == 4)
60 {
61 dim = convertShapeToDims(inputShape);
62 }
63 else
64 {
65 LOG(ERROR) << "only 2D and 4D tensors supported";
66 return false;
67 }
68
69 Softmax(inputData, dim, beta, outputData, dim);
70 return true;
71}
