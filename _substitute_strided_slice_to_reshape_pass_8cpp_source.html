50luci::CircleConst *build_rank1_const(loco::Graph *graph, const std::vector<uint32_t> &values)
51{
52 auto const_node = graph->nodes()->create<luci::CircleConst>();
53 const_node->dtype(loco::DataType::S32);
54 const_node->size<loco::DataType::S32>(values.size());
55 const_node->shape_status(luci::ShapeStatus::VALID);
56 const_node->rank(1);
57 const_node->dim(0) = values.size();
58
59 for (size_t i = 0; i < values.size(); i++)
60 {
61 const_node->at<loco::DataType::S32>(i) = values.at(i);
62 }
63
64 return const_node;
65}

93int64_t value_from_circle_const(const luci::CircleConst *node, uint32_t idx)
94{
95 assert(node->rank() == 1 && node->dim(0).value() > idx);
96 assert(node->dtype() == loco::DataType::S64 || node->dtype() == loco::DataType::S32);
97
98 if (node->dtype() == loco::DataType::S64)
99 return node->at<loco::DataType::S64>(idx);
100 return static_cast<int64_t>(node->at<loco::DataType::S32>(idx));
101}

103bool substitute_strided_slice_to_reshape(luci::CircleStridedSlice *ss_node)
104{
105 if (ss_node->shrink_axis_mask() == 0)
106 return false;
107
108 // TODO Consider cases with ellipsis_mask and new_axis_mask
109 // NOT YET SUPPORTED
110 if (ss_node->ellipsis_mask() != 0 or ss_node->new_axis_mask() != 0)
111 return false;
112
113 auto begin_const = dynamic_cast<luci::CircleConst *>(ss_node->begin());
114 auto strides_const = dynamic_cast<luci::CircleConst *>(ss_node->strides());
115 auto end_const = dynamic_cast<luci::CircleConst *>(ss_node->end());
116
117 if (not(begin_const && strides_const && end_const))
118 return false;
119
120 auto input_node = loco::must_cast<luci::CircleNode *>(ss_node->input());
121
122 // condition check
123 std::bitset<32> begin_mask(ss_node->begin_mask());
124 std::bitset<32> end_mask(ss_node->end_mask());
125 std::bitset<32> shrink_axis_mask(ss_node->shrink_axis_mask());
126
127 uint32_t input_rank = input_node->rank();
128 for (uint32_t i = 0; i < input_rank; i++)
129 {
130 if (!input_node->dim(i).known())
131 return false;
132
133 auto begin_dim = value_from_circle_const(begin_const, i);
134 if (begin_dim != 0 and begin_mask.test(i) == false)
135 return false;
136
137 // NOTE:
138 // In Tensorflow and TFLite, e.g., if input_shape = [2,3],
139 // strided_slice.end = [10,20] (larger value than actual dim)
140 // is treated as strided_slice.end = [2,3]
141 int64_t end_dim = value_from_circle_const(end_const, i);
142 if (end_dim < input_node->dim(i).value() and end_mask.test(i) == false)
143 return false;
144
145 int64_t strides_value = value_from_circle_const(strides_const, i);
146 if (strides_value != 1)
147 return false;
148
149 if (shrink_axis_mask.test(i) && input_node->dim(i).value() != 1)
150 return false;
151 }
152
153 // build shape for Reshape op
154 bool found = false;
155 std::vector<uint32_t> shrunk_shape;
156 for (uint32_t i = 0; i < input_rank; i++)
157 {
158 if (input_node->dim(i) == 1 and shrink_axis_mask.test(i))
159 found = true;
160 else
161 shrunk_shape.emplace_back(input_node->dim(i).value());
162 }
163
164 if (not found)
165 return false;
166
167 auto reshape_node = build_reshape(input_node->graph(), ss_node->name(), luci::get_origin(ss_node),
168 input_node, shrunk_shape);
169
170 replace(ss_node).with(reshape_node);
171 return true;
172}

199bool SubstituteStridedSliceToReshapePass::run(loco::Graph *g)
200{
201 bool changed = false;
202 for (auto node : loco::active_nodes(loco::output_nodes(g)))
203 {
204 if (auto circle_node = dynamic_cast<luci::CircleStridedSlice *>(node))
205 {
206 if (substitute_strided_slice_to_reshape(circle_node))
207 {
208 changed = true;
209 }
210 }
211 }
212 return changed;
213}
