102class FixInterGraph final
103{
104public:
105 void run(loco::Graph *g, const luci::Module *m, const luci::CircleReader &r)
106 {
107 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
108 {
109 if (recognize(node->dialect()))
110 {
111 auto cn = loco::must_cast<luci::CircleNode *>(node);
112
113 fix(cn, m, r);
114 }
115 }
116 }
117
118private:
119 bool recognize(const loco::Dialect *dialect) { return (dialect == luci::CircleDialect::get()); }

122 {
123 FixInterGraphNodes fix(module, reader);

138class ValidateNodeProp final : public luci::CircleNodeMutableVisitor<void>

140public:
141 ValidateNodeProp(const luci::Module *m, const luci::CircleReader &r) : _module(m), _reader(r) {}

147 void visit(luci::CircleIf *node) final
148 {
149 LOGGER(l);
150 INFO(l) << "CircleIf " << node->name() << std::endl;
151
152 auto then_graph = node->then_graph();
153 auto else_graph = node->else_graph();
154 assert(then_graph != nullptr);
155 assert(else_graph != nullptr);
156
157 // TODO support for differnt shape; but how?
158 // NODE Shape/Type inference assume below conditions
159
160 // Check both "then" and "else" subgraph outputs are same in count
161 auto then_outputs = loco::output_nodes(then_graph); // CircleOutput nodes
162 auto else_outputs = loco::output_nodes(else_graph);
163 if (then_outputs.size() != else_outputs.size())
164 {
165 INTERNAL_EXN("CircleIf THEN and ELSE Graph are not same in size");
166 }
167
168 // check outputs have same shape and dtype
169 auto then_graph_outputs = then_graph->outputs(); // loco::GraphOutput items
170 auto else_graph_outputs = else_graph->outputs();
171 for (size_t idx = 0; idx < then_outputs.size(); ++idx)
172 {
173 auto then_out = loco::must_cast<luci::CircleOutput *>(then_outputs.at(idx));
174 auto else_out = loco::must_cast<luci::CircleOutput *>(else_outputs.at(idx));
175
176 auto then_graph_output = then_graph_outputs->at(then_out->index());
177 auto else_graph_output = else_graph_outputs->at(else_out->index());
178 if (then_graph_output->shape()->rank() != else_graph_output->shape()->rank())
179 {
180 INTERNAL_EXN_V("CircleIf THEN and ELSE Graph Output rank mismatch ", idx);
181 }
182 for (uint32_t i = 0; i < then_graph_output->shape()->rank(); ++i)
183 {
184 if (then_graph_output->shape()->dim(i).known() &&
185 else_graph_output->shape()->dim(i).known() &&
186 then_graph_output->shape()->dim(i).value() !=
187 else_graph_output->shape()->dim(i).value())
188 {
189 INTERNAL_EXN_V("CircleIf THEN and ELSE Graph Output dimension mismatch ", idx);
190 }
191 }
192 if (then_graph_output->dtype() != else_graph_output->dtype())
193 {
194 INTERNAL_EXN_V("CircleIf THEN and ELSE Graph Output type mismatch ", idx);
195 }
196 }
197 }

203 void visit(luci::CircleWhile *node) final
204 {
205 LOGGER(l);
206 INFO(l) << "CircleWhile " << node->name() << std::endl;
207
208 auto cond_graph = node->cond_graph();
209 auto body_graph = node->body_graph();
210 assert(cond_graph != nullptr);
211 assert(body_graph != nullptr);
212
213 // Check input of "cond" and input/output of "body" subgraph have the same size
214 auto cond_inputs = loco::input_nodes(cond_graph);
215 auto cond_outputs = loco::output_nodes(cond_graph);
216 auto body_inputs = loco::input_nodes(body_graph);
217 auto body_outputs = loco::output_nodes(body_graph);
218 if (cond_inputs.size() != body_outputs.size())
219 {
220 INTERNAL_EXN("CircleWhile COND input and BODY output have different sizes");
221 }
222 if (cond_inputs.size() != body_inputs.size())
223 {
224 INTERNAL_EXN("CircleWhile COND input and BODY input have different sizes");
225 }
226 if (cond_outputs.size() != 1)
227 {
228 INTERNAL_EXN("CircleWhile COND output must have size 1");
229 }
230 auto cond_out = loco::must_cast<luci::CircleOutput *>(cond_outputs.at(0));
231 if (cond_out->dtype() != loco::DataType::BOOL)
232 {
233 INTERNAL_EXN("CircleWhile COND output must have bool type");
234 }
235
236 // input of "cond" and input/output of "body" subgraph must have the same shape and type
237 // First we compare input of "cond" with input of "body"
238 auto cond_graph_inputs = cond_graph->inputs();
239 auto body_graph_inputs = body_graph->inputs();
240 for (size_t idx = 0; idx < cond_inputs.size(); ++idx)
241 {
242 auto cond_in = loco::must_cast<luci::CircleInput *>(cond_inputs.at(idx));
243 auto body_in = loco::must_cast<luci::CircleInput *>(body_inputs.at(idx));
244
245 auto cond_graph_input = cond_graph_inputs->at(cond_in->index());
246 auto body_graph_input = body_graph_inputs->at(body_in->index());
247 if (cond_graph_input->shape()->rank() != body_graph_input->shape()->rank())

257 INTERNAL_EXN_V("CircleWhile COND input and BODY input dimension mismatch ", idx);
258 }
259 }
260 if (cond_graph_input->dtype() != body_graph_input->dtype())
261 {
262 INTERNAL_EXN_V("CircleWhile COND input and BODY input type mismatch ", idx);
263 }
264 }
265
266 // Next we compare input of "cond" with output of "body"
267 auto body_graph_outputs = body_graph->outputs();
268 for (size_t idx = 0; idx < cond_inputs.size(); ++idx)

342void post_import_graph(luci::Module *module, const luci::CircleReader &reader)
343{
344 LOGGER(l);
345
346 auto count = module->size();
347
348 for (size_t s = 0; s < count; ++s)
349 {
350 auto g = module->graph(s);
351 assert(g != nullptr);
352
353 INFO(l) << "--- FixInterGraph " << g->name() << "-------------------------";
354 FixInterGraph fix;
355 fix.run(g, module, reader);
356 }
357
358 for (size_t s = 0; s < count; ++s)
359 {
360 auto g = module->graph(s);
361 assert(g != nullptr);
362
363 INFO(l) << "--- ValidateGraphProp " << g->name() << "---------------------";
364 ValidateGraphProp prop;
365 prop.run(g, module, reader);
366 }
367
368 INFO(l) << "--- post_import_graph done -------------------------------------";
