41luci::CircleConst *create_fused_indices(luci::CircleConst *indices,
42 const std::set<uint32_t> &indices_set)
43{
44 auto name = indices->name();
45
46 auto fused_indices_const = indices->graph()->nodes()->create<luci::CircleConst>();
47 fused_indices_const->dtype(indices->dtype());
48 fused_indices_const->rank(1);
49 fused_indices_const->dim(0) = indices_set.size();
50 fused_indices_const->size<loco::DataType::S32>(indices_set.size());
51 fused_indices_const->shape_status(luci::ShapeStatus::VALID);
52 fused_indices_const->name(name);
53
54 auto curr_index = 0;
55 for (auto it = indices_set.begin(); it != indices_set.end(); it++)
56 {
57 fused_indices_const->at<loco::DataType::S32>(curr_index) = *it;
58 curr_index++;
59 }
60
61 return fused_indices_const;
62}

64bool fuse_mean_with_mean(luci::CircleMean *mean)
65{
66 // Get reduction indices of current CircleMean operation.
67 auto indices = dynamic_cast<luci::CircleConst *>(mean->reduction_indices());
68 if (not indices)
69 return false;
70 assert(indices->dtype() == loco::DataType::S32);
71
72 // Check whether previous node is CircleMean operation or not.
73 auto prev_mean = dynamic_cast<luci::CircleMean *>(mean->input());
74 if (not prev_mean)
75 return false;
76
77 // Check whether input rank of previous CircleMean operation is less 2 or not.
78 // This optimization works only if doesn't.
79 auto input = loco::must_cast<luci::CircleNode *>(prev_mean->input());
80 if (input->shape_status() != luci::ShapeStatus::VALID)
81 return false;
82 auto input_rank = input->rank();
83 if (input_rank < 2)
84 return false;
85
86 // Check whether current CircleMean and previous CircleMean
87 // has the same keep_dims parameter or not.
88 // If it doesn't, keep the graph unchanged.
89 if (mean->keep_dims() != prev_mean->keep_dims())
90 return false;
91
92 // Get reduction indices of previous CircleMean operation.
93 auto prev_indices = dynamic_cast<luci::CircleConst *>(prev_mean->reduction_indices());
94 if (not prev_indices)
95 return false;
96 assert(prev_indices->dtype() == loco::DataType::S32);
97
98 // Get sizes of indices of current CircleMean operation and previous CircleMean operation.
99 auto indices_size = indices->size<loco::DataType::S32>();
100 auto prev_indices_size = prev_indices->size<loco::DataType::S32>();
101
102 // Get set of indices of previous CircleMean operation.
103 std::set<uint32_t> indices_set;
104 for (uint32_t i = 0; i < prev_indices_size; i++)
105 {
106 auto index = prev_indices->at<loco::DataType::S32>(i);
107 if (index < 0)
108 index += input_rank;
109 indices_set.insert(index);
110 }
111
112 // Get the vector of input indexes, that remained untouched
113 // after the current CircleMean operation.
114 std::vector<uint32_t> input_indices_vector;
115 for (uint32_t i = 0; i < input_rank; i++)
116 {
117 if (indices_set.find(i) == indices_set.end())
118 input_indices_vector.push_back(i);
119 }
120
121 // Get final set of merged indices.
122 for (uint32_t i = 0; i < indices_size; i++)
123 {
124 auto index = indices->at<loco::DataType::S32>(i);
125 if (index < 0)
126 index += input_rank;
127 indices_set.insert(input_indices_vector.at(index));
128 }
129
130 // Create merged indices.
131 auto fused_indices_const = create_fused_indices(indices, indices_set);
132
133 auto name = mean->name();
134 assert(name.length() > 0);
135
136 // Create and configure new CircleMean operation.
137 auto fused_mean = mean->graph()->nodes()->create<luci::CircleMean>();
138 fused_mean->reduction_indices(fused_indices_const);
139 fused_mean->input(prev_mean->input());
140 fused_mean->keep_dims(mean->keep_dims());
141 fused_mean->name(name + "/Mean");
142
143 // Replace old CircleMean operations with new CircleMean operation with merged indices.
144 replace(mean).with(fused_mean);
145 luci::add_origin(fused_mean,
146 luci::composite_origin({luci::get_origin(mean), luci::get_origin(prev_mean)}));
147
148 return true;
149}

156bool FuseMeanWithMeanPass::run(loco::Graph *g)
157{
158 bool changed = false;
159 for (auto node : loco::active_nodes(loco::output_nodes(g)))
160 {
161 auto mean = dynamic_cast<luci::CircleMean *>(node);
162 if (not mean)
163 continue;
164
165 if (fuse_mean_with_mean(mean))
166 changed = true;
167 }
168
169 return changed;
170}
