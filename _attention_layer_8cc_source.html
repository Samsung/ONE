38void KernelGenerator::visit(const ir::operation::Attention &node)
39{
40 using ir::operation::Attention;
41
42 const auto input_index{node.getInputs().at(Attention::Input::INPUT)};
43 const auto wq_index{node.getInputs().at(Attention::Input::WQ)};
44 const auto wk_index{node.getInputs().at(Attention::Input::WK)};
45 const auto wv_index{node.getInputs().at(Attention::Input::WV)};
46 const auto wo_index{node.getInputs().at(Attention::Input::WO)};
47 const auto cos_index = node.getInputs().at(Attention::Input::COS);
48 const auto sin_index = node.getInputs().at(Attention::Input::SIN);
49 const auto mask_index = node.getInputs().at(Attention::Input::MASK);
50 const auto k_cache_index = node.getInputs().at(Attention::Input::K_CACHE);
51 const auto v_cache_index = node.getInputs().at(Attention::Input::V_CACHE);
52 const auto pos_index = node.getInputs().at(Attention::Input::POS);
53
54 const auto output_index{node.getOutputs().at(0)};
55 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
56
57 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
58 auto wq_tensor = _tensor_reg->getPortableTensor(wq_index);
59 auto wk_tensor = _tensor_reg->getPortableTensor(wk_index);
60 auto wv_tensor = _tensor_reg->getPortableTensor(wv_index);
61 auto wo_tensor = _tensor_reg->getPortableTensor(wo_index);
62 auto cos_tensor = cos_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(cos_index);
63 auto sin_tensor = sin_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(sin_index);
64 auto mask_tensor = mask_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(mask_index);
65 auto k_cache_tensor =
66 k_cache_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(k_cache_index);
67 auto v_cache_tensor =
68 v_cache_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(v_cache_index);
69 auto pos_tensor = pos_index.undefined() ? nullptr : _tensor_reg->getPortableTensor(pos_index);
70
71 auto fn = std::make_unique<ops::AttentionLayer>();
72
73 fn->configure(input_tensor, wq_tensor, wk_tensor, wv_tensor, wo_tensor, cos_tensor, sin_tensor,
74 mask_tensor, k_cache_tensor, v_cache_tensor, pos_tensor, output_tensor);
75
76 _return_fn = std::move(fn);
77}

94void AttentionLayer::configure(const IPortableTensor *input, const IPortableTensor *wq,
95 const IPortableTensor *wk, const IPortableTensor *wv,
96 const IPortableTensor *wo, const IPortableTensor *cos,
97 const IPortableTensor *sin, const IPortableTensor *mask,
98 IPortableTensor *k_cache, IPortableTensor *v_cache,
99 const IPortableTensor *pos, IPortableTensor *output)
100{
101 _input = input;
102 _wq = wq;
103 _wk = wk;
104 _wv = wv;
105 _wo = wo;
106 _cos = cos;
107 _sin = sin;
108 _mask = mask;
109 _k_cache = k_cache;
110 _v_cache = v_cache;
111 _cache_pos = pos;
112 _output = output;
113
114 // 0. Read and check inputs and params
115 const auto n_batch = getShape(_input).Dims(0);
116 assert(n_batch == 1); // Multi-batch is not supported.
117 const auto d_model = getShape(_input).Dims(2);
118
119 if (_cos == nullptr || _sin == nullptr || _cache_pos == nullptr)
120 throw std::runtime_error{"Attention: input tensors cannot be nullptr"};
121
122 const auto k_cache_shape = getShape(_k_cache);
123 if (k_cache_shape.DimensionsCount() != 4)
124 throw std::runtime_error{"K cache tensor must be 4D"};
125
126 // 0.1 Param - Read n_head from K cache 3rd dimension
127 const int32_t n_head = k_cache_shape.Dims(2);
128 if (d_model % n_head != 0)
129 throw std::runtime_error{"d_model must be divisible by n_head"};
130
131 const int32_t d_head = d_model / n_head;
132 const auto k_cache_dims = k_cache_shape.DimsData();
133 const int32_t k_cache_n_batch = k_cache_dims[0];
134 const int32_t k_cache_n_head = k_cache_dims[2];
135 const int32_t k_cache_d_head = k_cache_dims[3];
136
137 if (n_batch != k_cache_n_batch || n_head != k_cache_n_head || d_head != k_cache_d_head)
138 throw std::runtime_error{"Attention: shape mismatch between inputs"};
139}

157int32_t blockSizeFor(int32_t index, int32_t block_size = 32)
158{
159 // We need to include index, so we calculate for index + 1 elements
160 // Then round up to the nearest multiple of block_size
161 const int32_t elements_needed = index + 1;
162 return ((elements_needed + block_size - 1) / block_size) * block_size;
163}

165void AttentionLayer::attentionFloat32()
166{
167 // 0. Read and check inputs and params
168
169 const auto n_batch = getShape(_input).Dims(0);
170 assert(n_batch == 1); // Multi-batch is not supported.
171 const auto n_tokens = getShape(_input).Dims(1); // tokens to decode. Expects 1
172 const auto d_model = getShape(_input).Dims(2);
173
174 const int32_t n_head = getShape(_k_cache).Dims(2);
175 const int32_t d_head = d_model / n_head;
176
177 // 1. Q, K Projection
178
179 // Input tensor: _input
180 // Shape: [n_batch, n_tokens, d_model]
181 // Data: float*
182 // Weight tensors: _wq, _wk, _wv, _wo
183 // Shape: [d_model, d_model] (assuming d_q = d_k = d_v = d_model)
184 // Data: float*
185
186 // Define the output shape for Q and K projections
187 nnfw::cker::Shape proj_output_shape({n_batch, n_tokens, d_model});
188 nnfw::cker::FullyConnectedParams fc_params{};
189 fc_params.float_activation_min = std::numeric_limits<float>::lowest();
190 fc_params.float_activation_max = std::numeric_limits<float>::max();
191 fc_params.lhs_cacheable = true;
192
193 // Q Projection
194 std::vector<float> q_proj_buf(n_batch * n_tokens * d_model);
195 nnfw::cker::FullyConnected(fc_params, getShape(_input), getBuffer<float>(_input), getShape(_wq),
196 getBuffer<float>(_wq), /*bias_shape=*/getShape(nullptr),
197 /*bias_data=*/nullptr, proj_output_shape, q_proj_buf.data());
198
199 // K Projection
200 std::vector<float> k_proj_buf(n_batch * n_tokens * d_model);
201 nnfw::cker::FullyConnected(fc_params, getShape(_input), getBuffer<float>(_input), getShape(_wk),
202 getBuffer<float>(_wk), /*bias_shape=*/getShape(nullptr),
203 /*bias_data=*/nullptr, proj_output_shape, k_proj_buf.data());
204
205 // V Projection
206 std::vector<float> v_proj_buf(n_batch * n_tokens * d_model);
207 nnfw::cker::FullyConnected(fc_params, getShape(_input), getBuffer<float>(_input), getShape(_wv),
208 getBuffer<float>(_wv), /*bias_shape=*/getShape(nullptr),
209 /*bias_data=*/nullptr, proj_output_shape, v_proj_buf.data());
210
211 // 2. RoPE
212
213 // _cos, _sin
214 // Shape: [ n_batch, n_tokens, d_head ] (assume n_batch=1, n_tokens=1)
215 // DataType: float*
216
217 // 2.1 nullcheck (Validation moved to configure())
218
219 // 2.2 inputs: (input, cos, sin)
220 // RoPE expects 4D tensor for input and sin/cos tables.
221 // The projection output (proj_out) is logically reinterpreted:
222 // from [ n_batch, n_tokens, d_model ]
223 // to [ n_batch, n_head, n_tokens, d_head ]
224 // This reinterpretation is valid as d_model = n_head * d_head and n_tokens = 1.
225 nnfw::cker::Shape rope_in_shape({n_batch, n_head, n_tokens, d_head});
226 // Extend cos, sin to 4D
227 nnfw::cker::Shape sin_cos_shape({1, n_batch, n_tokens, d_head});
228
229 // 2.3 output buffer:
230 nnfw::cker::Shape rope_out_shape({n_batch, n_head, n_tokens, d_head});
231 std::vector<float> q_rope_buf(rope_out_shape.FlatSize());
232 std::vector<float> k_rope_buf(rope_out_shape.FlatSize());
233
234 nnfw::cker::RoPEMode rope_mode = nnfw::cker::RoPEMode::kGptNeox;
235
236 // 2.4 Call cker::RoPE
237 nnfw::cker::RoPE<float>(rope_mode, rope_in_shape, k_proj_buf.data(), sin_cos_shape,
238 getBuffer<float>(_sin), sin_cos_shape, getBuffer<float>(_cos),
239 rope_out_shape, k_rope_buf.data());
240 nnfw::cker::RoPE<float>(rope_mode, rope_in_shape, q_proj_buf.data(), sin_cos_shape,
241 getBuffer<float>(_sin), sin_cos_shape, getBuffer<float>(_cos),
242 rope_out_shape, q_rope_buf.data());
243
244 // 3. K cache
245
246 // _k_cache
247 // Shape: [ n_batch, cache_size, n_head, d_head ]
248 // DataType: float
249
250 // 3.1 Transpose K tensor to match k_cache memory layout
251 //
252 // from [ n_batch, n_head, n_tokens, d_head ]
253 // to [ n_batch, n_tokens, n_head, d_head ]
254 //
255 // Since both n_tokens and n_batch are 1, the K tensor is already effectively
256 // a contiguous memory block of size n_head * d_head, making an explicit transpose
257 // operation unnecessary.
258
259 // 3.2 Put K in k_cache[cache_pos]
260 const auto k_cache_shape = getShape(_k_cache);
261 // _cache_pos is expected to be a 1D tensor with one element, the current sequence position.
262 auto cache_pos = static_cast<int64_t>(getBuffer<int64_t>(_cache_pos)[0]);
263 if (cache_pos < 0 || cache_pos >= k_cache_shape.Dims(1))
264 throw std::runtime_error{"Attention: Current position is out of cache bounds"};
265
266 auto n_k = k_rope_buf.size();
267 auto sz_k = sizeof(k_rope_buf[0]) * n_k;
268 memcpy(getBuffer<float>(_k_cache) + cache_pos * n_k, k_rope_buf.data(), sz_k);
269
270 // 4. Attension Score
271
272 // 4.1 Transpose K
273 const int32_t block_size = 32;
274 auto alignedSize = [](int32_t index) -> int32_t {
275 // We need to include index, so we calculate for index + 1 elements
276 // Then round up to the nearest multiple of block_size
277 const int32_t elements_needed = index + 1;
278 return ((elements_needed + block_size - 1) / block_size) * block_size;
279 };
280
281 const int32_t aligned_ctx_size = alignedSize(cache_pos + 1);
282 nnfw::cker::Shape k_cache_tr_in_shape({n_batch, aligned_ctx_size, n_head, d_head});
283 nnfw::cker::Shape k_cache_tr_out_shape({n_batch, n_head, aligned_ctx_size, d_head});
284 std::vector<float> k_cache_tr_buf(k_cache_tr_out_shape.FlatSize());
285 nnfw::cker::TransposeParams tr_params{4, {0, 2, 1, 3}};
286 nnfw::cker::Transpose<float>(tr_params, k_cache_tr_in_shape, getBuffer<float>(_k_cache),
287 k_cache_tr_out_shape, k_cache_tr_buf.data());
288
289 // 4.2 Compute attention logits via batch matrix multiplication
290 // Inputs:
291 // q = [n_batch, n_head, n_tokens, d_head]
292 // (transposed) k = [n_batch, n_head, aligned_ctx_size, d_head]
293 // Output:
294 // qk = [n_batch, n_head, n_tokens, aligned_ctx_size]
295 //
296 // Note: Use adj_y=true to avoid additional transpose inside BMM kernel.
297 // With adj_y=true, the kernel directly computes Q K^T without extra
298 // memory operations, improving performance by eliminating redundant
299 // transpose steps that would otherwise occur internally.
300 nnfw::cker::Shape qk_shape({n_batch, n_head, n_tokens, aligned_ctx_size});
301 std::vector<float> qk_buf(qk_shape.FlatSize());
302 nnfw::cker::BatchMatMul bmm_op;
303 bool adj_x = false;
304 bool adj_y = true;
305 bmm_op.prepare(rope_out_shape, k_cache_tr_out_shape, adj_x, adj_y,
306 /*rhs_const=*/false);
307 bmm_op(rope_out_shape, q_rope_buf.data(), k_cache_tr_out_shape, k_cache_tr_buf.data(), adj_x,
308 adj_y, qk_shape, qk_buf.data());
309
310 // 4.3 Scale attention logits by 1/sqrt(d_model)
311 const float scaling_factor = 1.0f / std::sqrt(static_cast<float>(d_head));
312 // NOTE: TICO believes in duality: each tensor seeks its own sqrt(scaling_factor)
313 // truth, manifesting as (sqrt(scaling_factor) Q) * (sqrt(scaling_factor) K).
314 // We embrace unity: one scaling_factor for all, scaling_factor (Q * K)
315 for (size_t i = 0; i < qk_buf.size(); ++i)
316 qk_buf[i] *= scaling_factor;
317
318 // 4.4 Add attention
319
320 // mask tensor
321 //
322 // - shape: [n_batch, n_head, n_tokens, alignged_ctx_sz]
323 // - n_batch = 1
324 // - n_head = 1 : mask vector (last axis) is shared across heads
325 // - n_tokens = 1 : Process 1 token per 1 decode
326 // - aligned_ctx_size : multiple of block_size
327 // - type: float32
328 // - data:
329 // index 0 1 ... cache_pos ... {block_size-1}
330 // mask 0 0 ... 0 0 -inf -inf ... -inf
331 // | | | |
332 // +--past--+ new +-----future-----+
333 // tokens token tokens
334
335 const float *mask_data = getBuffer<float>(_mask);
336 for (size_t i = 0; i < qk_buf.size(); ++i)
337 {
338 // The mask repeats every 'aligned_ctx_size' elements.
339 qk_buf[i] += mask_data[i % aligned_ctx_size];
340 }
341
342 // 4.5 Softmax
343 std::vector<float> attn_score_buf(qk_shape.FlatSize());
344 nnfw::cker::SoftmaxParams softmax_params;
345 softmax_params.beta = 1.0f; // Standard beta for attention softmax
346 nnfw::cker::Softmax(softmax_params, qk_shape, qk_buf.data(), qk_shape, attn_score_buf.data());
347
348 // 5. V cache
349
350 // _v_cache
351 // Shape: [n_batch, cache_size, n_head, d_head]
352 // DataType: float
353
354 // 5.1 Transpose V to align v_cache (NOP)
355 // 5.2 Put V in v_cache[cache_pos]
356 const auto v_cache_shape = getShape(_v_cache);
357 if (cache_pos >= v_cache_shape.Dims(1))
358 throw std::runtime_error{"Attention: Current position is out of cache bounds"};
359
360 auto n_v = v_proj_buf.size();
361 auto sz_v = sizeof(v_proj_buf[0]) * n_v;
362 memcpy(getBuffer<float>(_v_cache) + cache_pos * n_v, v_proj_buf.data(), sz_v);
363
364 // 7. Attention Output Calculation
365 // 7.1. Compute attention output: attn_scores * V_cache -> attn_out
366 nnfw::cker::Shape v_cache_in_shape({n_batch, aligned_ctx_size, n_head, d_head});
367 nnfw::cker::Shape v_cache_tr_out_shape({n_batch, n_head, aligned_ctx_size, d_head});
368 std::vector<float> v_cache_tr_buf(v_cache_tr_out_shape.FlatSize());
369 nnfw::cker::TransposeParams v_tr_params{4, {0, 2, 1, 3}};
370 nnfw::cker::Transpose<float>(v_tr_params, v_cache_in_shape, getBuffer<float>(_v_cache),
371 v_cache_tr_out_shape, v_cache_tr_buf.data());
372
373 nnfw::cker::Shape attn_out_shape({n_batch, n_head, n_tokens, d_head});
374 std::vector<float> attn_out_buf(attn_out_shape.FlatSize());
375 nnfw::cker::BatchMatMul attn_bmm_op;
376 attn_bmm_op.prepare(qk_shape, v_cache_tr_out_shape, /*adj_x=*/false, /*adj_y=*/false,
377 /*rhs_const=*/false);
378 attn_bmm_op(qk_shape, attn_score_buf.data(), v_cache_tr_out_shape, v_cache_tr_buf.data(),
379 /*adj_x=*/false, /*adj_y=*/false, attn_out_shape, attn_out_buf.data());
380
381 // 7.2. Output Projection:
382 // When n_batch = n_tokens = 1, attn_out_buf layout already matches [1,1,d_model]
383 nnfw::cker::FullyConnected(fc_params, proj_output_shape, attn_out_buf.data(), getShape(_wo),
384 getBuffer<float>(_wo), /*bias_shape=*/getShape(nullptr),
385 /*bias_data=*/nullptr, getShape(_output), getBuffer<float>(_output));
386}
