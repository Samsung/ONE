38void TensorPlanner::planNonConstTensors(TensorBuilder *tensor_builder)
39{
40 VERBOSE(BackendContext) << "Start planning non-constant tensors" << std::endl;
41
42 const auto &training_usedefs = _tgraph.trainingUseDefs();
43
44 // NOTE The uses_map and defs_map must have the size of only registered tensors
45 std::unordered_map<ir::train::TrainingOperandIndex, uint32_t> uses_map;
46 std::unordered_map<ir::train::TrainingOperandIndex, uint32_t> defs_map;
47
48 // Prepare scanning
49 // This assumes TrainingOperationIndex in forwarding are always used
50 for (const auto &[operand_index, operand_usedefs] : training_usedefs)
51 {
52 const auto &operand = operand_usedefs.operand();
53
54 if (_external_operands.contains(operand_index.index()))
55 continue;
56
57 if (!operand_index.is_forward() || operand.isConstant())
58 continue;
59
60 uses_map[operand_index] = operand_usedefs.getTrainingUses().size();
61 defs_map[operand_index] = operand_usedefs.getTrainingDefs().size();
62 }
63
64 // Start scanning to do notify{First|Last}Use for each tensor
65 // TODO Remove this or find the reason why it is needed
66 // Q. Why is notifyFirstUse() called if operand's def count is 0?
67 // It's neither an external operand or a constant operand
68 // What does it mean when def count is 0?
69 // A. Not yet found the reason to need it yet.
70 for (const auto &[operand_index, def_count] : defs_map)
71 {
72 if (def_count == 0)
73 tensor_builder->notifyFirstUse(operand_index.index());
74 }
75
76 // This is a workaround to keep the operands over the execution
77 // (the operands look like they are unused)
78 std::vector<ir::train::TrainingOperandIndex> operands_last_until_end;
79 for (const auto &[operand_index, use_count] : uses_map)
80 {
81 if (use_count == 0)
82 operands_last_until_end.push_back(operand_index);
83 }
84
85 // Plan used or defined tensors in forwarding nodes
86 // At each operation,
87 // 1. Scan DEF of outputs. If the DEF, allocate it
88 // 2. Scan DEF of inputs. If variable tensor, throw an exception (not supported yet)
89 // 3. Scan USE of inputs/outputs. Decrease the USE and deallocate if the USE is 0
90 const auto order = _tgraph.topolSortOperations();
91 for (const auto &op_index : order)
92 {
93 const auto &op = _tgraph.operations().at(op_index);
94 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
95 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
96
97 // Define outputs
98 for (const auto &output : op_outputs)
99 {
100 if (_external_operands.contains(output))
101 continue;
102 if (!tensor_builder->isRegistered(output))
103 continue;
104
105 const auto output_index = ir::train::TrainingOperandIndex{output, true};
106 assert(defs_map.find(output_index) != defs_map.end());
107 assert(defs_map.at(output_index) == 1);
108 defs_map[output_index] = 0;
109 tensor_builder->notifyFirstUse(output_index.index());
110 }
111
112 // Scan variable tensors
113 // This tensor has features like constant. But OperandInfo and LowerInfo treat them as
114 // non-constant because of less memory usage by memory planning in here
115 // However, train backend does not support variable tensors yet
116 for (const auto &input : op_inputs)
117 {
118 if (_external_operands.contains(input))
119 continue;
120 if (!tensor_builder->isRegistered(input))
121 continue;
122
123 const auto input_index = ir::train::TrainingOperandIndex{input, true};
124 const auto &operand = training_usedefs.at(input_index).operand();
125 if (operand.isConstant())
126 continue;
127
128 assert(training_usedefs.find(input_index) != training_usedefs.end());
129 if (operand.info().isVariable())
130 throw std::runtime_error("The train backend does not support variable tensors");
131 }
132
133 for (const auto &input : op_inputs)
134 {
135 if (_external_operands.contains(input))
136 continue;
137 if (!tensor_builder->isRegistered(input))
138 continue;
139
140 const auto input_index = ir::train::TrainingOperandIndex{input, true};
141 const auto &operand = training_usedefs.at(input_index).operand();
142 if (operand.isConstant())
143 continue;
144
145 assert(uses_map.find(input_index) != uses_map.end());
146 assert(uses_map[input_index] > 0);
147 uses_map[input_index]--;
148 if (uses_map[input_index] == 0)
149 {
150 // plan for deallocation of static tensor node
151 tensor_builder->notifyLastUse(input_index.index());
152 }
153 }
154 }
155
156 // Plan used tensors in backwarding nodes
157 const auto border = _tgraph.essentialBackwardOrder();
158 for (const auto &op_index : border)
159 {
160 const auto &op = _tgraph.operations().at(op_index);
161 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
162 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
163
164 for (const auto &index : op_inputs + op_outputs)
165 {
166 if (_external_operands.contains(index))
167 continue;
168 if (!tensor_builder->isRegistered(index))
169 continue;
170
171 const auto operand_index = ir::train::TrainingOperandIndex{index, true};
172 assert(training_usedefs.find(operand_index) != training_usedefs.end());
173 const auto &operand_usedefs = training_usedefs.at(operand_index);
174 const auto &operand = operand_usedefs.operand();
175 if (operand.isConstant())
176 continue;
177
178 const auto &training_op_index = ir::train::TrainingOperationIndex{op_index, false};
179 assert(operand_usedefs.getTrainingDefs().find(training_op_index) ==
180 operand_usedefs.getTrainingDefs().end());
181
182 const auto &uses = operand_usedefs.getTrainingUses();
183 if (uses.find(training_op_index) != uses.end())
184 {
185 assert(uses_map.find(operand_index) != uses_map.end());
186 assert(uses_map[operand_index] > 0);
187 uses_map[operand_index]--;
188 if (uses_map[operand_index] == 0)
189 {
190 // plan for deallocation of static tensor node
191 tensor_builder->notifyLastUse(operand_index.index());
192 }
193 }
194 }
195 }
196
197 for (const auto &operand_index : operands_last_until_end)
198 {
199 tensor_builder->notifyLastUse(operand_index.index());
200 }
201
202 assert(std::all_of(
203 uses_map.begin(), uses_map.end(),
204 [](std::pair<const ir::train::TrainingOperandIndex, uint32_t> it) { return it.second == 0; }));
205
206 assert(std::all_of(
207 defs_map.begin(), defs_map.end(),
208 [](std::pair<const ir::train::TrainingOperandIndex, uint32_t> it) { return it.second == 0; }));
209
210 VERBOSE(BackendContext) << "Finish planning non-constant tensors" << std::endl;
211}

213void TensorPlanner::planTrainableTensors(TensorBuilder *tensor_builder)
214{
215 VERBOSE(BackendContext) << "Start planning constant tensors" << std::endl;
216
217 const auto &training_usedefs = _tgraph.trainingUseDefs();
218
219 std::unordered_map<ir::train::TrainingOperandIndex, uint32_t> uses_map;
220 std::unordered_map<ir::train::TrainingOperandIndex, uint32_t> defs_map;
221 std::vector<ir::train::TrainingOperandIndex> constants;
222
223 // Prepare scanning
224 for (const auto &pair : training_usedefs)
225 {
226 const auto &operand_index = pair.first;
227 const auto &operand_usedefs = pair.second;
228 const auto &operand = operand_usedefs.operand();
229
230 if (!operand_index.valid())
231 continue;
232
233 if (operand.isConstant() && operand_index.is_forward())
234 {
235 uses_map[operand_index] = 0;
236 const auto &defs = operand_usedefs.getTrainingDefs();
237 defs_map[operand_index] = defs.size(); // It means def_map's values are 0
238 constants.emplace_back(operand_index);
239 }
240 }
241
242 // Start scanning to do notify{First|Last}Use for each tensor
243 // If a tensor is a constant, increase the use of the tensor and allocate it first.
244 // Increasing use count here makes the tensor never be deallocated, i.e it they will be
245 // deallocated last.
246 for (const auto &index : constants)
247 {
248 assert(index.is_forward());
249 if (tensor_builder->isRegistered(index.index()))
250 {
251 uses_map[index]++;
252 tensor_builder->notifyFirstUse(index.index());
253 }
254 }
255
256 // Dispose and validate
257 for (const auto &index : constants)
258 {
259 assert(index.is_forward());
260 if (tensor_builder->isRegistered(index.index()))
261 {
262 uses_map[index]--;
263 tensor_builder->notifyLastUse(index.index());
264 }
265 }
266
267 assert(std::all_of(
268 uses_map.begin(), uses_map.end(),
269 [](std::pair<const ir::train::TrainingOperandIndex, uint32_t> it) { return it.second == 0; }));
270
271 assert(std::all_of(
272 defs_map.begin(), defs_map.end(),
273 [](std::pair<const ir::train::TrainingOperandIndex, uint32_t> it) { return it.second == 0; }));
274
275 VERBOSE(BackendContext) << "Finish planning constant tensors" << std::endl;
276}
