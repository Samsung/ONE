74std::shared_ptr<basic::Allocator> DynamicMemoryManager::allocate(const ITensor *tensor,
75 uint32_t capacity)
76{
77 auto find = _mem_alloc_map.find(tensor);
78 if (find != _mem_alloc_map.end())
79 throw std::runtime_error("Cannot allocate memory for a tensor. It was already allocated.");
80
81 _mem_alloc_map[tensor] = std::make_shared<basic::Allocator>(capacity);
82 return _mem_alloc_map[tensor];
83}

85void DynamicMemoryManager::deallocate(const ITensor *tensor)
86{
87 auto find = _mem_alloc_map.find(tensor);
88 if (find == _mem_alloc_map.end())
89 throw std::runtime_error("Cannot find Allocator for the requested index");
90
91 find->second->release(); // explicitly erase memory
92 _mem_alloc_map.erase(find); // remove tensor and alloc
93}
94
95void DynamicMemoryManager::deallocate(void)
96{
97 for (auto &mem_alloc : _mem_alloc_map)
98 {
99 // Release memory buffer of mem_alloc
100 mem_alloc.second->release();
101 }
102
103 _mem_alloc_map.clear();
104}
