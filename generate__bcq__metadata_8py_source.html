64def load_graph(model_file):
65 graph = tf.Graph()
66 graph_def = tf.compat.v1.GraphDef()
67
68 with open(model_file, "rb") as f:
69 graph_def.ParseFromString(f.read())
70 with graph.as_default():
71 tf.import_graph_def(graph_def, name="")
72
73 return graph

76def generate_metadata_header(original_graph, bcq_version, output_arrays):
77 # Generating metadata starts
78 metadata_values = np.array([ONE_START_MAGICNUM])
79
80 # Append BCQ version
81 metadata_values = np.append(metadata_values, bcq_version)
82
83 # Append original output count
84 output_cnt = output_arrays.count(',') + 1
85 metadata_values = np.append(metadata_values, output_cnt)

90def generate_bcq_metadata_v1(flags):
91 """
92 BCQv1 contains following metadata.
93 - The number of each BCQ information set
94 """
95
96 is_valid = True
97 allowed_info_names = [
98 "bcqinfo_do_w_x", "bcqinfo_alpha", "bcqinfo_packed_binary_code",
99 "bcqinfo_number_of_clusters", "bcqinfo_size_of_clusters",
100 "bcqinfo_qbits_of_clusters", "bcqinfo_dequant_weight"
101 ]
102
103 original_graph = load_graph(flags.input_path)
104 original_graph_def = original_graph.as_graph_def()
105
106 prefix_infonames_dict = {}
107
108 for node in original_graph_def.node:
109 if node.op == "Const" and "/bcqinfo_" in node.name:
110 prefix_index = node.name.index("/bcqinfo_")
111 prefix = node.name[:prefix_index]
112 infoname = node.name[prefix_index + 1:]
113
114 if infoname not in allowed_info_names:
115 is_valid = False
116 break
117
118 if prefix not in prefix_infonames_dict:
119 prefix_infonames_dict[prefix] = set()
120
121 prefix_infonames_dict[prefix].add(infoname)
122
123 # All the number of BCQ information should be same
124 num_of_bcqinfo = -1
125 for key in prefix_infonames_dict:
126 infonames = prefix_infonames_dict[key]
127 if num_of_bcqinfo == -1:
128 num_of_bcqinfo = len(infonames)
129 elif num_of_bcqinfo != len(infonames):
130 is_valid = False
131
132 # The number of BCQv1 information should be 6 or 7
133 if num_of_bcqinfo != 6 and num_of_bcqinfo != 7:
134 is_valid = False
135
136 # If BCQ information is invalid, return original model
137 if is_valid == False:
138 return original_graph_def
139
140 new_graph_def = tf.compat.v1.GraphDef()
141 for node in original_graph_def.node:
142 new_node = new_graph_def.node.add()
143 new_node.CopyFrom(node)
144
145 # Generate metadata header
146 metadata_values = generate_metadata_header(original_graph, 1, flags.output_arrays)
147
148 # Append metadata of BCQv1
149 metadata_values = np.append(metadata_values, num_of_bcqinfo + 1)
150
151 # Finish generating metadata
152 metadata_values = np.append(metadata_values, ONE_END_MAGICNUM)
153
154 # Generate metadata tensor
155 metadata_tensor = tf.make_tensor_proto(metadata_values, tf.int32)
156
157 new_node = new_graph_def.node.add()
158 new_node.op = "Const"
159 new_node.name = "one_compiler/bcqinfo_one_metadata"
160 new_node.attr["dtype"].CopyFrom(
161 tf.compat.v1.AttrValue(type=tf.int32.as_datatype_enum))
162 new_node.attr["value"].tensor.CopyFrom(metadata_tensor)
163 return new_graph_def

166def determine_bcq_version(flags):
167 """
168 CAUTION : For now, BCQ has only one version and thus always returns 1 when BCQ
169 information nodes are included. If new BCQ version is introduced,
170 this function must be updated accordingly.
171
172 When BCQ information does not exist, -1 is returned.
173 """
174 bcq_version = -1
175
176 original_graph = load_graph(flags.input_path)
177 original_graph_def = original_graph.as_graph_def()
178
179 for node in original_graph_def.node:
180 if node.op == "Const" and "/bcqinfo_" in node.name:
181 bcq_version = 1
182 break
183
184 return bcq_version

187def generate_bcq_metadata(flags):
188 """
189 Basic format of metadata is as following.
190 - Magic number indicating start
191 - Version of BCQ Format
192 - The number of original outputs
193 - Metadata based on each BCQ format
194 - Magic number indicating end
195 """
196 program_version = 1
197 model_version = determine_bcq_version(flags)
198
199 if model_version == 1:
200 result_graph_def = generate_bcq_metadata_v1(flags)
201 elif model_version == -1:
202 # When there is no BCQ information, do nothing
203 result_graph_def = load_graph(flags.input_path)
204 else:
205 err_msg = "BCQ version of the model(v{}) ".format(model_version)
206 err_msg += "is higher than "
207 err_msg += "the version supported by this program(v{})".format(program_version)
208 raise SystemExit(err_msg)
209
210 tf.io.write_graph(result_graph_def, '.', flags.output_path, False)
