30// `UntrainableOperation` wraps operations that are not yet supported for training.
31// This class can be removed if all operations are supported for training.
32template <typename OperationType,
33 std::enable_if_t<std::is_base_of_v<Operation, OperationType>, bool> = true>
34class UntrainableOperation : public OperationType, public TrainableOperation
35{
36public:
37 UntrainableOperation(const OperationType &operation) : OperationType{operation} {}
38 virtual ~UntrainableOperation() = default;
39
40public:
41 std::unique_ptr<ITrainableOperation> clone() const override
42 {
43 return std::make_unique<UntrainableOperation<OperationType>>(*this);
44 }
45 void accept(OperationVisitor &v) const override { v.visit(*this); }
46 void accept(TrainableOperationVisitor &) const override
47 {
48 throw std::runtime_error(OperationType::name() + " operation is not trainable yet");
49 }
50 bool hasTrainableParameter() const override { return false; }
51};
