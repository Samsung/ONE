62// Allow below edges only
63// m1 < m2, s1 == 0 and s2 == 0 if m1:s1:o1 -> m2:s2:o2'
64void Executors::checkSupportedMultimodel() const
65{
66 // If package includes no-connection model, model_count is less than real model count in package.
67 // Then this method will throw exception based on model index
68 // 1st model: input assumption
69 // Otherwise: edges assumption
70
71 // Assumption: edges
72 // m1 < m2, s1 == 0 and s2 == 0 if edge 'm1:s1:o1 -> m2:s2:o2'
73 for (auto edge : _model_edges->edges)
74 {
75 auto const model_from = std::get<ir::ModelIndex>(edge.from);
76 auto const model_to = std::get<ir::ModelIndex>(edge.to);
77 auto const subg_from = std::get<ir::SubgraphIndex>(edge.from);
78 auto const subg_to = std::get<ir::SubgraphIndex>(edge.to);
79
80 if (model_from.value() == model_to.value())
81 {
82 throw std::runtime_error{"Multi model's edge set has invalid edge"};
83 }
84
85 if ((model_from.value() > model_to.value()) || (subg_from != ir::SubgraphIndex{0}) ||
86 (subg_to != ir::SubgraphIndex{0}))
87 throw std::runtime_error{"NYI: Multi model execution for this edge set is not supported yet"};
88 }
89
90 // Assumption: package inputs
91 // All 1st model inputs come from package input if always m1 < m2
92 {
93 auto first_executor = at(ir::ModelIndex{0}, ir::SubgraphIndex{0});
94 auto search_first_model = [&](const ir::IOIndex &input_index) {
95 for (auto &input : _model_edges->pkg_inputs)
96 {
97 if ((std::get<ir::ModelIndex>(input) == ir::ModelIndex{0}) ||
98 (std::get<ir::SubgraphIndex>(input) == ir::SubgraphIndex{0}) ||
99 (std::get<ir::IOIndex>(input) == input_index))
100 return true;
101 }
102
103 return false;
104 };
105
106 for (uint32_t i = 0; i < first_executor->graph().getInputs().size(); i++)
107 {
108 if (!search_first_model(ir::IOIndex{i}))
109 throw std::runtime_error{"Cannot find 1st model's input buffer"};
110 }
111 }
112}

114void Executors::execute(const IODescription &desc)
115{
116 // Check supported multi model package
117 checkSupportedMultimodel();
118
119 // TODO Find better way to schedule order of executors
120 std::vector<std::unique_ptr<backend::builtin::IOTensor>> pkgs_inputs(desc.inputs.size());
121 std::vector<std::unique_ptr<backend::builtin::IOTensor>> pkgs_outputs(desc.outputs.size());
122 const auto layout = ir::Layout::NHWC;
123 auto const model_count = modelCount();
124
125 auto find_input_index = [&](const ir::ModelIndex &model_index,
126 const ir::SubgraphIndex &subg_index, const ir::IOIndex &io_index) {
127 for (size_t i = 0; i < _model_edges->pkg_inputs.size(); i++)
128 {
129 auto &input_desc = _model_edges->pkg_inputs[i];
130 if ((std::get<ir::ModelIndex>(input_desc) == model_index) &&
131 (std::get<ir::SubgraphIndex>(input_desc) == subg_index) &&
132 (std::get<ir::IOIndex>(input_desc) == io_index))
133 return static_cast<int32_t>(i);
134 }
135 return -1;
136 };
137
138 auto find_output_index = [&](const ir::ModelIndex &model_index,
139 const ir::SubgraphIndex &subg_index, const ir::IOIndex &io_index) {
140 for (size_t i = 0; i < _model_edges->pkg_outputs.size(); i++)
141 {
142 auto &input_desc = _model_edges->pkg_outputs[i];
143 if ((std::get<ir::ModelIndex>(input_desc) == model_index) &&
144 (std::get<ir::SubgraphIndex>(input_desc) == subg_index) &&
145 (std::get<ir::IOIndex>(input_desc) == io_index))
146 return static_cast<int32_t>(i);
147 }
148 return -1;
149 };
150
151 auto find_from = [&](const ir::ModelIndex &model_index, const ir::SubgraphIndex &subg_index,
152 const ir::IOIndex &io_index) {
153 for (auto &edge : _model_edges->edges)
154 {
155 if ((std::get<ir::ModelIndex>(edge.to) == model_index) &&
156 (std::get<ir::SubgraphIndex>(edge.to) == subg_index) &&
157 (std::get<ir::IOIndex>(edge.to) == io_index))
158 return edge.from;
159 }
160
161 throw std::runtime_error{"Cannot find edge for model input"};
162 };
163
164 // TODO Find better way to share buffers and tensors between executors
165 // NOTE Executor's inputs/outputs are always IPortableTensor. If backend of inputs/outputs
166 // is using tensor that does not inherit IPortableTensor, Permute operation is added
167 // and all inputs/outputs become IPortableTensor at compile stage.
168 // This allows user's buffers to be set to inputs/outputs of executors.
169 // Shared buffers and tensors between executors
170 std::vector<std::unique_ptr<backend::builtin::IOTensor>> edge_tensors;
171 std::unordered_map<ir::IODesc, std::unique_ptr<uint8_t[]>> internal_bufs;
172
173 // Reference count for each `from` of edges to reduce peak memory
174 std::unordered_map<ir::IODesc, int32_t> internal_tensor_ref_counts(_model_edges->edges.size());
175 for (const auto &edge : _model_edges->edges)
176 {
177 if (internal_tensor_ref_counts.find(edge.from) == internal_tensor_ref_counts.end())
178 internal_tensor_ref_counts[edge.from] = 1;
179 else
180 internal_tensor_ref_counts[edge.from]++;
181 }
182
183 // Execute each model
184 // NOTE May be better to use vector instead of unordered_map for _executors
185 for (auto model_index = ir::ModelIndex{0}; model_index.value() < model_count; model_index++)
186 {
187 // Find executor
188 auto executor = at(model_index, ir::SubgraphIndex{0});
189
190 // Set IOTensors
191 std::vector<backend::IPortableTensor *> inputs_inter;
192 std::vector<backend::IPortableTensor *> outputs_inter;
193 auto const input_size = executor->graph().getInputs().size();
194 auto const output_size = executor->graph().getOutputs().size();
195 inputs_inter.resize(input_size);
196 outputs_inter.resize(output_size);
197
198 // Set inputs of executor
199 for (uint32_t i = 0; i < input_size; i++)
200 {
201 auto const &index = executor->graph().getInputs().at(ir::IOIndex{i});
202 auto const &info = executor->graph().operands().at(index).info();
203
204 auto input_pkg_index = find_input_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
205 if (input_pkg_index != -1)
206 {
207 auto input_desc = desc.inputs[input_pkg_index].get();
208 pkgs_inputs[input_pkg_index] = std::make_unique<backend::builtin::IOTensor>(info, layout);
209 // TODO Remove const_cast (we need const_cast as ITensor is writable)
210 pkgs_inputs[input_pkg_index]->setUserTensor(
211 reinterpret_cast<uint8_t *>(const_cast<void *>(input_desc->buffer)), input_desc->size);
212
213 inputs_inter[i] = pkgs_inputs[input_pkg_index].get();
214 }
215 else
216 {
217 auto from_iodesc = find_from(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
218 const auto &from_model_index = std::get<ir::ModelIndex>(from_iodesc);
219 const auto &from_subg_index = std::get<ir::SubgraphIndex>(from_iodesc);
220 const auto &from_ioindex = std::get<ir::IOIndex>(from_iodesc).value();
221
222 // Supported only sequantial execution of models
223 assert(from_model_index.value() < model_index.value());
224 assert(from_subg_index.value() == 0);
225
226 // TODO Add check if from_executor has already been executed
227 const auto from_executor = _executors.at({from_model_index, from_subg_index}).get();
228 inputs_inter[i] = from_executor->getOutputTensors().at(from_ioindex);
229 }
230 }
231
232 // Set outputs of executor
233 for (uint32_t i = 0; i < output_size; i++)
234 {
235 auto const &index = executor->graph().getOutputs().at(ir::IOIndex{i});
236 auto const &info = executor->graph().operands().at(index).info();
237
238 auto output_pkg_index = find_output_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
239 if (output_pkg_index != -1)
240 {
241 auto output_desc = desc.outputs[output_pkg_index].get();
242 pkgs_outputs[output_pkg_index] = std::make_unique<backend::builtin::IOTensor>(info, layout);
243 pkgs_outputs[output_pkg_index]->setUserTensor(
244 reinterpret_cast<uint8_t *>(output_desc->buffer), output_desc->size);
245
246 outputs_inter[i] = pkgs_outputs[output_pkg_index].get();
247 }
248 else
249 {
250 const auto from_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
251 const auto temp_index = edge_tensors.size();
252 internal_bufs[from_iodesc] = std::make_unique<uint8_t[]>(info.total_size());
253 edge_tensors.emplace_back(std::make_unique<backend::builtin::IOTensor>(info, layout));
254 edge_tensors.at(temp_index)
255 ->setUserTensor(internal_bufs[from_iodesc].get(), info.total_size());
256 outputs_inter[i] = edge_tensors[temp_index].get();
257 }
258 }
259
260 executor->execute(inputs_inter, outputs_inter);
261
262 // Release internal buffers that are no longer needed
263 for (uint32_t i = 0; i < input_size; i++)
264 {
265 const auto input_pkg_index =
266 find_input_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
267 if (input_pkg_index == -1)
268 {
269 const auto from_iodesc = find_from(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
270 internal_tensor_ref_counts[from_iodesc]--;
271
272 assert(internal_tensor_ref_counts[from_iodesc] >= 0);
273 if (internal_tensor_ref_counts[from_iodesc] == 0)
274 {
275 internal_bufs[from_iodesc].reset();
276 }
277 }
278 }
279 }
280}

282// modelCount() iterates _executors.
283// It assumes that Compiler will generate Executor for all models and _executors includes all
284// generated Executor.
285// If nnpackage includes model(s) which has no connection and Compiler does not
286// generate Executor for them, modelCount() return less value than real model count.
287uint16_t Executors::modelCount() const
288{
289 uint16_t model_count = 0;
290 for (; _executors.find(std::make_pair(ir::ModelIndex{model_count}, ir::SubgraphIndex{0})) !=
291 _executors.end();
292 model_count++)
293 ;
294
295 return model_count;
296}
