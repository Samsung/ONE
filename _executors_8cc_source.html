62// Allow below edges only
63// m1 < m2, s1 == 0 and s2 == 0 if m1:s1:o1 -> m2:s2:o2'
64void Executors::checkSupportedMultimodel() const
65{
66 // If package includes no-connection model, model_count is less than real model count in package.
67 // Then this method will throw exception based on model index
68 // 1st model: input assumption
69 // Otherwise: edges assumption
70
71 // Assumption: edges
72 // m1 < m2, s1 == 0 and s2 == 0 if edge 'm1:s1:o1 -> m2:s2:o2'
73 for (auto edge : _model_edges->edges)
74 {
75 auto const model_from = std::get<ir::ModelIndex>(edge.from);
76 auto const model_to = std::get<ir::ModelIndex>(edge.to);
77 auto const subg_from = std::get<ir::SubgraphIndex>(edge.from);
78 auto const subg_to = std::get<ir::SubgraphIndex>(edge.to);
79
80 if (model_from.value() == model_to.value())
81 {
82 throw std::runtime_error{"Multi model's edge set has invalid edge"};
83 }
84
85 if ((model_from.value() > model_to.value()) || (subg_from != ir::SubgraphIndex{0}) ||
86 (subg_to != ir::SubgraphIndex{0}))
87 throw std::runtime_error{"NYI: Multi model execution for this edge set is not supported yet"};
88 }
89
90 // Assumption: package inputs
91 // All 1st model inputs come from package input if always m1 < m2
92 {
93 auto first_executor = at(ir::ModelIndex{0}, ir::SubgraphIndex{0});
94 auto search_first_model = [&](const ir::IOIndex &input_index) {
95 for (auto &input : _model_edges->pkg_inputs)
96 {
97 if ((std::get<ir::ModelIndex>(input) == ir::ModelIndex{0}) ||
98 (std::get<ir::SubgraphIndex>(input) == ir::SubgraphIndex{0}) ||
99 (std::get<ir::IOIndex>(input) == input_index))
100 return true;
101 }
102
103 return false;
104 };
105
106 for (uint32_t i = 0; i < first_executor->getInputTensors().size(); i++)
107 {
108 if (!search_first_model(ir::IOIndex{i}))
109 throw std::runtime_error{"Cannot find 1st model's input buffer"};
110 }
111 }
112}

114// TODO Support type-aware quantization of nnpkg inputs/outputs
115void Executors::createTypeAwareQuantLayers()
116{
117 if (_is_created_type_quant_layers)
118 {
119 return;
120 }
121
122 // Initialize layers for type-ware quantization as empty layer
123 for (const auto &pair : _executors)
124 {
125 const auto &model_index = pair.first.first;
126 const auto &subg_index = pair.first.second;
127
128 std::vector<backend::ITensor *> inputs;
129 std::vector<backend::ITensor *> outputs;
130 _type_aware_quant_layers[{model_index, subg_index}] =
131 std::make_unique<PermuteLayer>(inputs, outputs);
132 }
133
134 for (const auto &pair : _edge_map)
135 {
136 const auto &from_iodesc = pair.first;
137 const auto &from_model_index = std::get<ir::ModelIndex>(from_iodesc);
138 const auto &from_subg_index = std::get<ir::SubgraphIndex>(from_iodesc);
139 const auto &from_io_index = std::get<ir::IOIndex>(from_iodesc);
140
141 const auto from_executor = _executors.at({from_model_index, from_subg_index}).get();
142 const auto from_tensor = from_executor->getOutputTensors().at(from_io_index.value());
143
144 const auto &to_list = pair.second;
145 std::vector<backend::ITensor *> inputs;
146 std::vector<backend::ITensor *> outputs;
147 for (const auto &to_iodesc : to_list)
148 {
149 const auto &to_model_index = std::get<ir::ModelIndex>(to_iodesc);
150 const auto &to_subg_index = std::get<ir::SubgraphIndex>(to_iodesc);
151 const auto &to_io_index = std::get<ir::IOIndex>(to_iodesc);
152
153 const auto to_executor = _executors.at({to_model_index, to_subg_index}).get();
154 const auto to_tensor = to_executor->getInputTensors().at(to_io_index.value());
155
156 // TODO Unify tensors with the same `from` tensor and same type
157 if (from_tensor->data_type() != to_tensor->data_type())
158 {
159 assert(inputs.size() == outputs.size());
160 const auto &to_info = to_executor->getInputTensors().at(to_io_index.value())->orig_info();
161 const auto to_layout = to_tensor->orig_layout();
162 inputs.emplace_back(from_tensor);
163
164 auto type_aware_quant_tensor =
165 std::make_unique<backend::builtin::IOTensor>(to_info, to_layout);
166 outputs.emplace_back(type_aware_quant_tensor.get());
167
168 _type_aware_quant_tensors[to_iodesc] = std::move(type_aware_quant_tensor);
169 }
170 }
171
172 auto layer = std::make_unique<PermuteLayer>(inputs, outputs);
173 layer->prepare();
174 _type_aware_quant_layers[{from_model_index, from_subg_index}] = std::move(layer);
175 }
176
177 _is_created_type_quant_layers = true;
178}

180void Executors::execute(const IODescription &desc)
181{
182 // Check supported multi model package
183 checkSupportedMultimodel();
184
185 // TODO Move creating layers for type-aware quantization in compilation stage
186 createTypeAwareQuantLayers();
187
188 // TODO Find better way to schedule order of executors
189 std::vector<std::unique_ptr<backend::builtin::IOTensor>> pkgs_inputs(desc.inputs.size());
190 std::vector<std::unique_ptr<backend::builtin::IOTensor>> pkgs_outputs(desc.outputs.size());
191 auto const model_count = modelCount();
192
193 auto find_input_index = [&](const ir::ModelIndex &model_index,
194 const ir::SubgraphIndex &subg_index, const ir::IOIndex &io_index) {
195 for (size_t i = 0; i < _model_edges->pkg_inputs.size(); i++)
196 {
197 auto &input_desc = _model_edges->pkg_inputs[i];
198 if ((std::get<ir::ModelIndex>(input_desc) == model_index) &&
199 (std::get<ir::SubgraphIndex>(input_desc) == subg_index) &&
200 (std::get<ir::IOIndex>(input_desc) == io_index))
201 return static_cast<int32_t>(i);
202 }
203 return -1;
204 };
205
206 auto find_output_index = [&](const ir::ModelIndex &model_index,
207 const ir::SubgraphIndex &subg_index, const ir::IOIndex &io_index) {
208 for (size_t i = 0; i < _model_edges->pkg_outputs.size(); i++)
209 {
210 auto &input_desc = _model_edges->pkg_outputs[i];
211 if ((std::get<ir::ModelIndex>(input_desc) == model_index) &&
212 (std::get<ir::SubgraphIndex>(input_desc) == subg_index) &&
213 (std::get<ir::IOIndex>(input_desc) == io_index))
214 return static_cast<int32_t>(i);
215 }
216 return -1;
217 };
218
219 auto find_from = [&](const ir::ModelIndex &model_index, const ir::SubgraphIndex &subg_index,
220 const ir::IOIndex &io_index) {
221 for (auto &edge : _model_edges->edges)
222 {
223 if ((std::get<ir::ModelIndex>(edge.to) == model_index) &&
224 (std::get<ir::SubgraphIndex>(edge.to) == subg_index) &&
225 (std::get<ir::IOIndex>(edge.to) == io_index))
226 return edge.from;
227 }
228
229 throw std::runtime_error{"Cannot find edge for model input"};
230 };
231
232 // TODO Find better way to share buffers and tensors between executors
233 // NOTE Executor's inputs/outputs are always IPortableTensor. If backend of inputs/outputs
234 // is using tensor that does not inherit IPortableTensor, Permute operation is added
235 // and all inputs/outputs become IPortableTensor at compile stage.
236 // This allows user's buffers to be set to inputs/outputs of executors.
237 // Shared buffers and tensors between executors
238 // TODO Introduce a new class for edge tensors
239 std::vector<std::unique_ptr<backend::builtin::IOTensor>> edge_tensors;
240 std::unordered_map<ir::IODesc, std::unique_ptr<uint8_t[]>> internal_bufs;
241 std::unordered_map<ir::IODesc, std::unique_ptr<uint8_t[]>> type_aware_bufs;
242
243 // Reference count for each `from` of edges to reduce peak memory
244 std::unordered_map<ir::IODesc, int32_t> internal_tensor_ref_counts(_model_edges->edges.size());
245 for (const auto &edge : _model_edges->edges)
246 {
247 if (internal_tensor_ref_counts.find(edge.from) == internal_tensor_ref_counts.end())
248 internal_tensor_ref_counts[edge.from] = 1;
249 else
250 internal_tensor_ref_counts[edge.from]++;
251 }
252
253 // Execute each model
254 // NOTE May be better to use vector instead of unordered_map for _executors
255 for (auto model_index = ir::ModelIndex{0}; model_index.value() < model_count; model_index++)
256 {
257 // Find executor
258 auto executor = at(model_index, ir::SubgraphIndex{0});
259
260 // Set IOTensors
261 std::vector<backend::IPortableTensor *> inputs_inter;
262 std::vector<backend::IPortableTensor *> outputs_inter;
263 const auto &input_tensors = executor->getInputTensors();
264 const auto &output_tensors = executor->getOutputTensors();
265 auto const input_size = input_tensors.size();
266 auto const output_size = output_tensors.size();
267 inputs_inter.resize(input_size);
268 outputs_inter.resize(output_size);
269
270 // Set inputs of executor
271 for (uint32_t i = 0; i < input_size; i++)
272 {
273 auto const &info = input_tensors.at(i)->orig_info();
274
275 auto input_pkg_index = find_input_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
276 if (input_pkg_index != -1)
277 {
278 auto input_desc = desc.inputs[input_pkg_index].get();
279 const auto orig_layout = input_tensors.at(i)->orig_layout();
280 if (input_desc->layout != orig_layout)
281 {
282 throw std::runtime_error("Executors.cc: Changing layout is not supported");
283 }
284 pkgs_inputs[input_pkg_index] =
285 std::make_unique<backend::builtin::IOTensor>(info, input_desc->layout);
286 // TODO Remove const_cast (we need const_cast as ITensor is writable)
287 pkgs_inputs[input_pkg_index]->setUserTensor(
288 reinterpret_cast<uint8_t *>(const_cast<void *>(input_desc->buffer)), input_desc->size);
289
290 inputs_inter[i] = pkgs_inputs[input_pkg_index].get();
291 }
292 else
293 {
294 auto from_iodesc = find_from(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
295 const auto &from_model_index = std::get<ir::ModelIndex>(from_iodesc);
296 const auto &from_subg_index = std::get<ir::SubgraphIndex>(from_iodesc);
297 const auto &from_ioindex = std::get<ir::IOIndex>(from_iodesc).value();
298
299 // Supported only sequantial execution of models
300 assert(from_model_index.value() < model_index.value());
301 assert(from_subg_index.value() == 0);
302
303 // TODO Add check if from_executor has already been executed
304 const auto from_executor = _executors.at({from_model_index, from_subg_index}).get();
305 const auto to_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
306 if (_type_aware_quant_tensors.find(to_iodesc) == _type_aware_quant_tensors.end())
307 {
308 inputs_inter[i] = from_executor->getOutputTensors().at(from_ioindex);
309 }
310 else
311 {
312 inputs_inter[i] = _type_aware_quant_tensors.at(to_iodesc).get();
313 }
314 }
315 }
316
317 // Set outputs of executor
318 for (uint32_t i = 0; i < output_size; i++)
319 {
320 const auto &output_tensor = output_tensors.at(i);
321 auto const &info = output_tensor->orig_info();
322
323 auto output_pkg_index = find_output_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
324 if (output_pkg_index != -1)
325 {
326 auto output_desc = desc.outputs[output_pkg_index].get();
327 const auto orig_layout = output_tensors.at(i)->orig_layout();
328 if (output_desc->layout != orig_layout)
329 {
330 throw std::runtime_error("Executors.cc: Changing layout is not supported");
331 }
332 pkgs_outputs[output_pkg_index] =
333 std::make_unique<backend::builtin::IOTensor>(info, output_desc->layout);
334 pkgs_outputs[output_pkg_index]->setUserTensor(
335 reinterpret_cast<uint8_t *>(output_desc->buffer), output_desc->size);
336
337 outputs_inter[i] = pkgs_outputs[output_pkg_index].get();
338 }
339 else
340 {
341 // Allocate buffer and set it as buffer of `from` tensors
342 const auto from_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
343 const auto from_layout = output_tensor->orig_layout();
344
345 const auto temp_index = edge_tensors.size();
346 internal_bufs[from_iodesc] = std::make_unique<uint8_t[]>(info.total_size());
347 edge_tensors.emplace_back(std::make_unique<backend::builtin::IOTensor>(info, from_layout));
348 edge_tensors.at(temp_index)
349 ->setUserTensor(internal_bufs[from_iodesc].get(), info.total_size());
350 outputs_inter[i] = edge_tensors[temp_index].get();
351
352 // Allocate buffer and set it as buffer of tensors for type-aware quantization
353 for (const auto &to_iodesc : _edge_map[from_iodesc])
354 {
355 if (_type_aware_quant_tensors.find(to_iodesc) != _type_aware_quant_tensors.end())
356 {
357 const auto &to_model_index = std::get<ir::ModelIndex>(to_iodesc);
358 const auto &to_subg_index = std::get<ir::SubgraphIndex>(to_iodesc);
359 const auto &to_io_index = std::get<ir::IOIndex>(to_iodesc);
360
361 const auto to_executor = _executors.at({to_model_index, to_subg_index}).get();
362 auto type_aware_quant_tensor = _type_aware_quant_tensors.at(to_iodesc).get();
363
364 const auto to_tensor = to_executor->getInputTensors().at(to_io_index.value());
365 const auto total_size = to_tensor->orig_info().total_size();
366 type_aware_bufs[to_iodesc] = std::make_unique<uint8_t[]>(total_size);
367 type_aware_quant_tensor->setUserTensor(type_aware_bufs[to_iodesc].get(), total_size);
368 }
369 }
370 }
371 }
372
373 executor->execute(inputs_inter, outputs_inter);
374 _type_aware_quant_layers[{model_index, ir::SubgraphIndex{0}}]->run();
375
376 // Release input buffers that are no longer needed
377 for (uint32_t i = 0; i < input_size; i++)
378 {
379 const auto input_pkg_index =
380 find_input_index(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
381
382 const auto to_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
383 if (input_pkg_index == -1)
384 {
385 if (type_aware_bufs.find(to_iodesc) != type_aware_bufs.end())
386 {
387 // Relase buffer if input tensor is using buffer for type-aware quantization
388 const auto to_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
389 if (type_aware_bufs.find(to_iodesc) != type_aware_bufs.end())
390 {
391 type_aware_bufs[to_iodesc].reset();
392 }
393 }
394 else
395 {
396 // Decrease reference count `from` tensor if input tensor is using the buffer of the
397 // `from` tensor
398 const auto from_iodesc = find_from(model_index, ir::SubgraphIndex{0}, ir::IOIndex{i});
399 assert(internal_bufs.find(from_iodesc) != internal_bufs.end());
400 if (internal_tensor_ref_counts[from_iodesc] > 0)
401 {
402 internal_tensor_ref_counts[from_iodesc]--;
403
404 assert(internal_tensor_ref_counts[from_iodesc] >= 0);
405 if (internal_tensor_ref_counts[from_iodesc] == 0)
406 {
407 internal_bufs[from_iodesc].reset();
408 }
409 }
410 }
411 }
412 }
413
414 // Release output buffers if those buffers are no longer used other executors because of
415 // type-aware quantization
416 // FIXME if tensors for type-aware quantization unified for the same `from` tensor and same type
417 for (uint32_t i = 0; i < output_size; i++)
418 {
419 auto from_iodesc = ir::IODesc{model_index, ir::SubgraphIndex{0}, ir::IOIndex{i}};
420
421 // Check if other executors will use the buffer of internal_bufs[from_iodesc]
422 const auto &to_list = _edge_map[from_iodesc];
423 if (to_list.size() == 0)
424 {
425 // This condition means `from_iodesc` tensor is an output of nnpkg
426 continue;
427 }
428
429 bool to_be_release =
430 !std::any_of(to_list.begin(), to_list.end(), [&](const ir::IODesc &to_iodesc) {
431 // This condition means another executor uses the buffer of internal_bufs[from_iodesc]
432 return type_aware_bufs.find(to_iodesc) == type_aware_bufs.end();
433 });
434
435 if (to_be_release)
436 {
437 internal_tensor_ref_counts[from_iodesc]--;
438
439 assert(internal_tensor_ref_counts[from_iodesc] >= 0);
440 if (internal_tensor_ref_counts[from_iodesc] == 0)
441 {
442 internal_bufs[from_iodesc].reset();
443 }
444 }
445 }
446 }
447}

449// modelCount() iterates _executors.
450// It assumes that Compiler will generate Executor for all models and _executors includes all
451// generated Executor.
452// If nnpackage includes model(s) which has no connection and Compiler does not
453// generate Executor for them, modelCount() return less value than real model count.
454uint16_t Executors::modelCount() const
455{
456 uint16_t model_count = 0;
457 for (; _executors.find(std::make_pair(ir::ModelIndex{model_count}, ir::SubgraphIndex{0})) !=
458 _executors.end();
459 model_count++)
460 ;
461
462 return model_count;
463}
