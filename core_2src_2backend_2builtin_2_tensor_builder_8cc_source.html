38void TensorBuilder::registerTensorInfo(const ir::OperandIndex &ind, const ir::OperandInfo &info,
39 ir::Layout backend_layout)
40{
41 _tensor_info_map.emplace(ind, info);
42
43 VERBOSE_F() << "cpucommon REGISTER!! " << ind << std::endl;
44 if (info.isDynamic())
45 {
46 _dynamic_tensor_mgr->buildTensor(ind, info, backend_layout);
47 }
48 else
49 {
50 _static_tensor_mgr->buildTensor(ind, info, backend_layout, info.isConstant());
51 }
52}

54void TensorBuilder::notifyFirstUse(const ir::OperandIndex &ind)
55{
56 // TODO Enhance the way of checking user tensors
57 if (_tensor_info_map.find(ind) == _tensor_info_map.end()) // Do not proceed for user tensors
58 return;
59
60 const auto tensor_info = _tensor_info_map.at(ind);
61
62 if (!nativeOwnTensorAt(ind)->is_dynamic())
63 {
64 const auto size = tensor_info.total_size();
65 _static_tensor_mgr->claimPlan(ind, size);
66 }
67}

69void TensorBuilder::notifyLastUse(const ir::OperandIndex &ind)
70{
71 // TODO Enhance the way of checking user tensors
72 if (_tensor_info_map.find(ind) == _tensor_info_map.end()) // Do not proceed for user tensors
73 return;
74
75 if (!nativeOwnTensorAt(ind)->is_dynamic())
76 {
77 _static_tensor_mgr->releasePlan(ind);
78 }
79}

81bool TensorBuilder::isRegistered(const ir::OperandIndex &ind) const
82{
83 // User tensors are not registered in _tensor_info_map but objects for them are exist
84 // in the tensor registry.
85 // TODO Enhance the way of checking user tensors
86 if (_tensor_reg->getITensor(ind))
87 return true;
88 return _tensor_info_map.find(ind) != _tensor_info_map.end();
