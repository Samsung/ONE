27template <FusedActivationFunctionType Ac>
28void Div(const float *input1_data, const Dims<4> &input1_dims, const float *input2_data,
29 const Dims<4> &input2_dims, float *output_data, const Dims<4> &output_dims)
30{
31 MatchingArraySize(input1_dims, 3, input2_dims, 3, output_dims, 3);
32 MatchingArraySize(input1_dims, 2, input2_dims, 2, output_dims, 2);
33 MatchingArraySize(input1_dims, 1, input2_dims, 1, output_dims, 1);
34 MatchingArraySize(input1_dims, 0, input2_dims, 0, output_dims, 0);
35 DCHECK(IsPackedWithoutStrides(input1_dims));
36 DCHECK(IsPackedWithoutStrides(input2_dims));
37 DCHECK(IsPackedWithoutStrides(output_dims));
38
39 const int size = input1_dims.sizes[3] * input1_dims.strides[3];
40
41 for (int i = 0; i < size; i++)
42 {
43 auto x = input1_data[i] / input2_data[i];
44 output_data[i] = ActivationFunction<Ac>(x);
45 }
46}

48// From optimized_ops.h in TensorFlow Lite
49//
50// TODO: We can implement BroadcastDiv on buffers of arbitrary
51// dimensionality if the runtime code does a single loop over one dimension
52// that handles broadcasting as the base case. The code generator would then
53// generate max(D1, D2) nested for loops.
54// TODO: BroadcastDiv is intentionally duplicated from
55// reference_ops.h. Once an optimized version is implemented and NdArrayDesc<T>
56// is no longer referenced in this file, move NdArrayDesc<T> from types.h to
57// reference_ops.h.
58template <FusedActivationFunctionType Ac>
59void BroadcastDiv(const float *input1_data, const Dims<4> &input1_dims, const float *input2_data,
60 const Dims<4> &input2_dims, float *output_data, const Dims<4> &output_dims)
61{
62 NdArrayDesc<4> desc1;
63 NdArrayDesc<4> desc2;
64 NdArrayDescsForElementwiseBroadcast(input1_dims, input2_dims, &desc1, &desc2);
65
66 // In Tensorflow, the dimensions are canonically named (batch_number, row,
67 // col, channel), with extents (batches, height, width, depth), with the
68 // trailing dimension changing most rapidly (channels has the smallest stride,
69 // typically 1 element).
70 //
71 // In generated C code, we store arrays with the dimensions reversed. The
72 // first dimension has smallest stride.
73 //
74 // We name our variables by their Tensorflow convention, but generate C code
75 // nesting loops such that the innermost loop has the smallest stride for the
76 // best cache behavior.
77 for (int b = 0; b < ArraySize(output_dims, 3); ++b)
78 {
79 for (int y = 0; y < ArraySize(output_dims, 2); ++y)
80 {
81 for (int x = 0; x < ArraySize(output_dims, 1); ++x)
82 {
83 for (int c = 0; c < ArraySize(output_dims, 0); ++c)
84 {
85 output_data[Offset(output_dims, c, x, y, b)] =
86 ActivationFunction<Ac>(input1_data[SubscriptToIndex(desc1, c, x, y, b)] /
87 input2_data[SubscriptToIndex(desc2, c, x, y, b)]);
88 }
89 }
90 }
91 }
92}
