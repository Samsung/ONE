65 void assignTensor(const ir::OperandIndex index, std::shared_ptr<ITensor> tensor)

67 assert(tensor->bufferRO() != nullptr);

77 const ITensor *tensorAt(const ir::OperandIndex index, bool can_optional = false) const
78 {
79 if (_tensors.find(index) == _tensors.end())
80 {
81 // It may optional input,
82 // otherwise input is not set by runtime user
83 if (can_optional)
84 {
85 return nullptr;
86 }
87
88 throw std::runtime_error{"ExecEnv: Input is not set"};
89 }
90
91 return _tensors.at(index).get();
92 }
93
99 bool contains(const ir::OperandIndex index) const
100 {
101 return (_tensors.find(index) != _tensors.end());
102 }
103
111 void allocateIfNeeded(const ir::OperandIndex index, const ir::OperandInfo &info)
112 {
113 // already allocated, or constant
114 if (contains(index))
115 {
116 return;
117 }
118
119 // Buffer from external (ex. model output)
120 auto tensor = std::make_shared<Tensor>(info);
121 if (isExtBuffer(index))
122 {
123 tensor->setBuffer(_external_buffers.at(index));
124 assignTensor(index, tensor);
125
126 return;
127 }
128
129 tensor->setBuffer(std::make_shared<InternalBuffer>(tensor->total_size()));
130 assignTensor(index, tensor);
131 _buffers.insert(index);
132 }
133
140 void allocateAndShareIfNeeded(const ir::OperandIndex index, const ir::OperandInfo &info,
141 const ir::OperandIndex index_to_share)
142 {
143 if (!contains(index_to_share))
144 {
145 throw std::runtime_error{"Cannot find tensor to share data"};
146 }
147
148 // already allocated
149 if (contains(index))
150 {
151 return;
152 }
153
154 if (isExtBuffer(index))
155 {
156 auto tensor = std::make_shared<Tensor>(info);
157 tensor->setBuffer(_external_buffers.at(index));
158 assignTensor(index, tensor);
159 }
160 else
161 {
162 auto tensor = std::make_shared<ROTensor>(info);
163 tensor->setData(tensorAt(index_to_share)->shareData());
164 assignTensor(index, tensor);
165 _buffers.insert(index);
166 }
167 }
168
174 void freeIfAllocated(const ir::OperandIndex index)
175 {
176 if (_buffers.find(index) != _buffers.end())
177 {
178 _tensors.at(index)->releaseData();
179 }
180 }
181
187 void assignExternalBuffer(const ir::OperandIndex index, std::shared_ptr<ExternalBuffer> buffer)
188 {
189 _external_buffers.emplace(index, buffer);
190 }
191
192private:
193 bool isExtBuffer(const ir::OperandIndex index)
194 {
195 return (_external_buffers.find(index) != _external_buffers.end());
196 }
197
198private:
199 const ir::Graph &_graph;
200 // Tensor map to use in interpreter
201 // It should map tensors that have allocated or assigned buffer pointer
202 std::unordered_map<ir::OperandIndex, std::shared_ptr<ITensor>> _tensors;
203 // Tensors allocated by allocateIfNeed (buffer)
204 std::unordered_set<ir::OperandIndex> _buffers;
205 // Tensor buffer from external
206 std::unordered_map<ir::OperandIndex, std::shared_ptr<ExternalBuffer>> _external_buffers;
207};
