206template <typename LoaderDomain>
207void BaseLoader<LoaderDomain>::BaseLoader::loadFromFile(const std::string &file_path)
208{
209 _fd = open(file_path.c_str(), O_RDONLY);
210 if (_fd < 0)
211 {
212 throw std::runtime_error("Failed to open file " + file_path);
213 }
214
215 struct stat file_stat;
216 if (fstat(_fd, &file_stat) != 0)
217 {
218 throw std::runtime_error("Fstat failed or file " + file_path + " is not a regular file");
219 }
220 int size = file_stat.st_size;
221
222 // Map model file into memory region
223 _base = static_cast<uint8_t *>(mmap(NULL, size, PROT_READ, MAP_PRIVATE, _fd, 0));
224 if (_base == MAP_FAILED)
225 {
226 close(_fd);
227 throw std::runtime_error("mmap failed - " + std::string(strerror(errno)));
228 }
229
230 _verifier = std::make_unique<Verifier>(reinterpret_cast<const std::uint8_t *>(_base), size);
231
232 loadModel();
233 munmap(_base, size);
234
235 close(_fd);
236}

316template <typename T> bool Copy(const T *data_ptr, std::vector<uint16_t> &arr)
317{
318 if (data_ptr->values() == nullptr)
319 {
320 return false;
321 }
322
323 int size = data_ptr->values()->size();
324 arr.reserve(size);
325 for (int i = 0; i < size; i++)
326 {
327 arr.emplace_back(static_cast<uint16_t>(data_ptr->values()->Get(i)));
328 }
329 return true;
330}

332template <typename LoaderDomain>
333ir::OperandIndex BaseLoader<LoaderDomain>::loadOperand(const Tensor *tensor, ir::Graph &subg)
334{
335 ir::Shape shape;
336 // Shape
337 const auto *tensor_shape = tensor->shape();
338 if (tensor_shape != nullptr)
339 {
340 for (const auto &dim : *tensor_shape)
341 {
342 shape.append(dim);
343 }
344 }
345
346 // Note for tensor->shape_signature()
347 // We don't handle shape signature
348 // How we handle:
349 // If shape_signature[k] == -1, we will use tensor->shape()[k] == 1
350 // If app wants to change the input shape, call nnfw_apply_input_tensorinfo() can
351 // be used.
352
353 // TypeInfo
354 ir::TypeInfo type_info(tensorTypeToDataType(tensor->type()));
355 loadQuantization(tensor, type_info);
356 loadSparsity(tensor, type_info);
357
358 // Create operand
359 const auto operand_index = subg.addOperand(shape, type_info);
360
361 // Constant tensors are indicated by non-empty data.
362 const auto *data = _domain_model->buffers()->Get(tensor->buffer())->data();
363 if (data != nullptr)
364 {
365 using std::ptrdiff_t;
366 std::shared_ptr<ir::Data> data_obj;
367
368 if (_fd == -1) // Model is from memory
369 {
370 data_obj = std::make_shared<ir::ExternalData>(data->data(), data->size());
371 }
372 else // Model is loaded(mmap'd) from a file
373 {
374 size_t data_size = data->size();
375 ptrdiff_t unaligned_offset_start = data->data() - _base;
376 ptrdiff_t offset_end = unaligned_offset_start + data_size;
377
378 // Calculated aligned offset from base address of mapped region
379 // munmap accepts memory address which is a multiple of the pagesize
380 ptrdiff_t aligned_offset_start = (unaligned_offset_start / _pagesize) * _pagesize;
381 size_t mmap_size = offset_end - aligned_offset_start;
382
383 uint32_t buf_idx = tensor->buffer();
384 auto buffer_found = _buf_to_data.find(buf_idx);
385
386 if (buffer_found != _buf_to_data.end())
387 {
388 // Another tensor points this buffer and its matching Data(either CachedData or MMapedData)
389 // was already created. Let's reuse the Data
390 data_obj = buffer_found->second;
391 }
392 else if (_use_mmaped_data)
393 {
394 data_obj = std::make_shared<ir::MMapedData>(_fd, aligned_offset_start, mmap_size,
395 unaligned_offset_start, data_size);
396 _buf_to_data[buf_idx] = data_obj;
397 }
398 else
399 {
400 size_t offset = unaligned_offset_start - aligned_offset_start;
401 uint8_t *mmap_base = static_cast<uint8_t *>(
402 mmap(NULL, mmap_size, PROT_READ, MAP_PRIVATE, _fd, aligned_offset_start));
403
404 data_obj = std::make_shared<ir::CachedData>(mmap_base + offset, data_size);
405 _buf_to_data[buf_idx] = data_obj;
406
407 munmap(mmap_base, mmap_size);
408 }
409 }
410 subg.setOperandValue(operand_index, std::move(data_obj));
411 }
412
413 _tensor_names.emplace(operand_index, tensor->name()->str());
414
415 // Variable
416 if (tensor->is_variable())
417 {
418 if (data != nullptr)
419 throw std::runtime_error("Variable tensor with buffer is not supported!");
420
421 subg.operands().at(operand_index).info().setAsVariable();
422 }
423
424 return operand_index;
425}

427template <typename LoaderDomain>
428void BaseLoader<LoaderDomain>::loadQuantization(const Tensor *tensor, ir::TypeInfo &typeInfo)
429{
430 auto q_params = tensor->quantization();
431 if (q_params == nullptr || q_params->scale() == nullptr || q_params->scale()->size() == 0)
432 {
433 typeInfo.quantization(0., 0);
434 return;
435 }
436 if (q_params->zero_point() == nullptr)
437 {
438 throw std::runtime_error("Quantization params: scale is not null, but zero_point is null.");
439 }
440 const size_t num_scales = q_params->scale()->size();
441 if (num_scales != q_params->zero_point()->size())
442 {
443 throw std::runtime_error("Quantization params: scale size != zero_point size");
444 }
445 std::vector<float> scales;
446 std::vector<int32_t> zero_points;
447 scales.resize(num_scales);
448 zero_points.resize(num_scales);
449 for (size_t i = 0; i < num_scales; ++i)
450 {
451 scales[i] = q_params->scale()->Get(i);
452 // zero_point is defined as long (i64) in schema while TypeInfo's zero_point is int32_t.
453 // int64_t is used instead of long because long is 4 byte in most 32bit architecture.
454 int64_t zero_point = q_params->zero_point()->Get(i);
455 if (zero_point < std::numeric_limits<int32_t>::min() ||
456 zero_point > std::numeric_limits<int32_t>::max())
457 throw std::runtime_error("Zero_point is out of int32 range.");
458 zero_points[i] = static_cast<int32_t>(zero_point);
459 }
460 auto details = q_params->details_as_CustomQuantization();
461 if (details != nullptr)
462 throw std::runtime_error("Custom Quantization is not supported");
463 typeInfo.quantization(std::move(scales), std::move(zero_points));
464}

466template <typename LoaderDomain>
467void BaseLoader<LoaderDomain>::loadSparsity(const Tensor *tensor, ir::TypeInfo &typeInfo)
468{
469 auto src_sparsity = tensor->sparsity();
470 if (src_sparsity != nullptr)
471 {
472 std::vector<uint16_t> w1_segments;
473 std::vector<uint16_t> w1_indices;
474 // check traversal_order
475 if (src_sparsity->traversal_order())
476 {
477 const int traversal_order_size = src_sparsity->traversal_order()->size();
478 for (int i = 0; i < traversal_order_size; ++i)
479 {
480 if (i != src_sparsity->traversal_order()->Get(i))
481 throw std::runtime_error("traversal_order [0, 1, ..., n-1] is only supported.");
482 }
483 }
484 // check block_map
485 int block_rank = 0;
486 if (src_sparsity->block_map())
487 {
488 block_rank = src_sparsity->block_map()->size();
489 for (int i = 0; i < block_rank; ++i)
490 {
491 if (i != src_sparsity->block_map()->Get(i))
492 throw std::runtime_error("block_map [0, 1, ..., n-1] is only supported.");
493 }
494 }
495 // load metadata
496 const auto dim_metadata_size = src_sparsity->dim_metadata()->size();
497 const auto dense_rank = tensor->shape() ? tensor->shape()->size() : 0;
498 if (dense_rank + block_rank != dim_metadata_size)
499 throw std::runtime_error("sparsity dim_metadata length is wrong.");
500 bool random_sparsity = dim_metadata_size == 2 && block_rank == 0;
501 bool block2D_sparsity = dim_metadata_size == 4 && block_rank == 2;
502 if (dim_metadata_size != !random_sparsity && !block2D_sparsity)
503 throw std::runtime_error(
504 "sparsity is supported only for 2D tensor with random or 16x1 block sparsity.");
505
506 const auto *src_metadata = src_sparsity->dim_metadata()->Get(0);
507 if (src_metadata->format() != DimensionType::DimensionType_DENSE)
508 throw std::runtime_error("sparse tensor dim[0] is not DENSE");
509 src_metadata = src_sparsity->dim_metadata()->Get(1);
510 if (src_metadata->format() != DimensionType::DimensionType_SPARSE_CSR)
511 throw std::runtime_error("sparse tensor dim[0] is not SPARSE_CSR");
512 auto ParseSparseIndexVector = [src_metadata, &w1_segments, &w1_indices]() {
513 if (src_metadata->array_segments() == nullptr || src_metadata->array_indices() == nullptr)
514 return false;
515 bool status = true;
516 switch (src_metadata->array_segments_type())
517 {
518 case SparseIndexVector::SparseIndexVector_Int32Vector:
519 status = Copy(src_metadata->array_segments_as_Int32Vector(), w1_segments);
520 break;
521 case SparseIndexVector::SparseIndexVector_Uint16Vector:
522 status = Copy(src_metadata->array_segments_as_Uint16Vector(), w1_segments);
523 break;
524 case SparseIndexVector::SparseIndexVector_Uint8Vector:
525 status = Copy(src_metadata->array_segments_as_Uint8Vector(), w1_segments);
526 break;
527 default:
528 return false;
529 }
530 if (status != true)
531 return false;
532 switch (src_metadata->array_indices_type())
533 {
534 case SparseIndexVector::SparseIndexVector_Int32Vector:
535 return Copy(src_metadata->array_indices_as_Int32Vector(), w1_indices);
536 case SparseIndexVector::SparseIndexVector_Uint16Vector:
537 return Copy(src_metadata->array_indices_as_Uint16Vector(), w1_indices);
538 case SparseIndexVector::SparseIndexVector_Uint8Vector:
539 return Copy(src_metadata->array_indices_as_Uint8Vector(), w1_indices);
540 default:
541 break;
542 }
543 return false;
544 };
545 if (ParseSparseIndexVector() == false)
546 throw std::runtime_error("Error during parsing sparsity index information");
547 // Get block size
548 std::vector<int32_t> block_size;
549 for (int i = 0; i < block_rank; ++i)
550 {
551 auto block_metadata = src_sparsity->dim_metadata()->Get(dense_rank + i);
552 if (block_metadata->format() != DimensionType::DimensionType_DENSE)
553 throw std::runtime_error("block dimension must be DENSE.");
554 block_size.push_back(block_metadata->dense_size());
555 }
556 typeInfo.sparsity(std::make_shared<ir::Sparsity>(std::move(w1_segments), std::move(w1_indices),
557 std::move(block_size)));
558 }
559}

561template <typename LoaderDomain>
562void BaseLoader<LoaderDomain>::loadOperationIO(const Operator *op, ir::OperandIndexSequence &inputs,
563 ir::OperandIndexSequence &outputs)
564{
565 for (const std::int32_t idx : *op->inputs())
566 {
567 // Optional tensors are not supported yet except for FULLY_CONNECTED and BCQ_FULLY_CONNECTED
568 auto check_optional_input = [&]() {
569 auto builtin_code = getBuiltinOperator(op);
570 if (isOptionalInputTensor(idx) && !allowOptionalInputTensor(builtin_code))
571 throw std::runtime_error(
572 std::string("loader doesn't support optional input tensor yet for ")
573 .append(EnumNameBuiltinOperator(builtin_code)));
574 };
575 check_optional_input();
576 inputs.append(tensorIdxToOperandIdx(idx));
577 }
578
579 for (const std::int32_t idx : *op->outputs())
580 {
581 outputs.append(tensorIdxToOperandIdx(idx));
582 }
583}

607template <typename LoaderDomain>
608template <typename Param>
609void BaseLoader<LoaderDomain>::loadPool2DOptions(Param &param, const Pool2DOptions *options)
610{
611 // Strides and Paddings
612 if (options->stride_h() <= 0 || options->stride_w() <= 0)
613 throw std::runtime_error{"Invalid stride vertical or horizontal - both must be bigger than 0"};
614 loadStridesAndPaddings(param, options);
615 // Filter width and height
616 // Strides
617 if (options->filter_width() <= 0 || options->filter_height() <= 0)
618 throw std::runtime_error{"Invalid filter width or height - both must be bigger than 0"};
619 param.kw = options->filter_width();
620 param.kh = options->filter_height();
621 // Activation
622 param.activation = convertActivation(options->fused_activation_function());
623}

1032template <typename LoaderDomain>
1033void BaseLoader<LoaderDomain>::loadCustom(const Operator *op, ir::Graph &subg)
1034{
1035 ir::OperandIndexSequence inputs;
1036 ir::OperandIndexSequence outputs;
1037
1038 assert(op->custom_options_format() == CustomOptionsFormat::CustomOptionsFormat_FLEXBUFFERS &&
1039 "Unsupported custom operation options format");
1040
1041 auto *op_code = _domain_model->operator_codes()->Get(op->opcode_index());
1042 auto custom_op_name = op_code->custom_code()->str();
1043
1044 enum class BuiltinOP
1045 {
1046 AddV2,
1047 ReduceAll,
1048 MatrixBandPart,
1049 BatchMatMul,
1050 Einsum,
1051 BroadcastTo,
1052 FusedBatchNorm,
1053 StatelessRandomUniform,
1054 Erf,
1055 DetectionPostProcess
1056 };
1057
1058 // Mapping from custom op name string to BuiltinOP enum
1059 std::map<std::string, BuiltinOP> builtin_map = {
1060 {"AddV2", BuiltinOP::AddV2},
1061 {"All", BuiltinOP::ReduceAll},
1062 {"MatrixBandPart", BuiltinOP::MatrixBandPart},
1063 {"BatchMatMulV2", BuiltinOP::BatchMatMul},
1064 {"Einsum", BuiltinOP::Einsum},
1065 {"FusedBatchNormV3", BuiltinOP::FusedBatchNorm},
1066 {"BroadcastTo", BuiltinOP::BroadcastTo},
1067 {"StatelessRandomUniform", BuiltinOP::StatelessRandomUniform},
1068 {"Erf", BuiltinOP::Erf},
1069 {"TFLite_Detection_PostProcess", BuiltinOP::DetectionPostProcess},
1070 };
1071
1072 try
1073 {
1074 // Throw out_of_range if it is unknown custom op
1075 auto custom_op_id = builtin_map.at(custom_op_name);
1076 switch (custom_op_id)
1077 {
1078 case BuiltinOP::AddV2:
1079 loadAddV2(op, subg);
1080 break;
1081 case BuiltinOP::ReduceAll:
1082 loadReduceAll(op, subg);
1083 break;
1084 case BuiltinOP::MatrixBandPart:
1085 loadOperationTo<ir::operation::MatrixBandPart>(op, subg);
1086 break;
1087 case BuiltinOP::BatchMatMul:
1088 loadBatchMatMul(op, subg);
1089 break;
1090 case BuiltinOP::Einsum:
1091 loadEinsum(op, subg);
1092 break;
1093 case BuiltinOP::BroadcastTo:
1094 loadOperationTo<ir::operation::BroadcastTo>(op, subg);
1095 break;
1096 case BuiltinOP::FusedBatchNorm:
1097 loadFusedBatchNorm(op, subg);
1098 break;
1099 case BuiltinOP::StatelessRandomUniform:
1100 loadOperationTo<ir::operation::StatelessRandomUniform>(op, subg);
1101 break;
1102 case BuiltinOP::Erf:
1103 loadElementwiseUnary(op, subg, ir::operation::ElementwiseUnary::Type::ERF);
1104 break;
1105 case BuiltinOP::DetectionPostProcess:
1106 loadDetectionPostProcess(op, subg);
1107 break;
1108 default:
1109 throw std::runtime_error{
1110 "Loader: Custom OP map is defined but operation loader function is not defined"};
1111 }
1112
1113 return;
1114 }
1115 catch (...)
1116 {
1117 loadOperationIO(op, inputs, outputs);
1118
1119 auto constraint = ir::OperandConstraint::createExact(inputs.size());
1120
1121 size_t custom_op_data_size = op->custom_options()->size();
1122 auto custom_op_data = new char[custom_op_data_size];
1123 std::copy(op->custom_options()->begin(), op->custom_options()->end(), custom_op_data);
1124
1125 ir::operation::Custom::Userdata userdata{};
1126 userdata.data = custom_op_data;
1127 userdata.size = custom_op_data_size;
1128
1129 auto new_op = std::make_unique<ir::operation::Custom>(constraint, inputs, outputs,
1130 custom_op_name, userdata);
1131
1132 subg.addOperation(std::move(new_op));
1133 }
1134}
