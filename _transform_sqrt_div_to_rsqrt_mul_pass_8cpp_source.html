48bool transform_sqrtdiv_to_rsqrtmul(luci::CircleDiv *div)
49{
50 assert(div != nullptr);
51
52 // skip if x is const, for FuseRsqrtPass
53 auto *const_node = dynamic_cast<luci::CircleConst *>(div->x());
54 if (const_node != nullptr)
55 return false;
56
57 auto *sqrt = dynamic_cast<luci::CircleSqrt *>(div->y());
58 if (sqrt == nullptr)
59 return false;
60
61 auto *graph = div->graph();
62
63 auto *rsqrt = graph->nodes()->create<luci::CircleRsqrt>();
64 rsqrt->x(sqrt->x());
65 rsqrt->name(sqrt->name() + "_rsqrt");
66 luci::add_origin(rsqrt, luci::get_origin(sqrt));
67
68 auto *mul = graph->nodes()->create<luci::CircleMul>();
69 mul->x(div->x());
70 mul->y(rsqrt);
71 mul->fusedActivationFunction(div->fusedActivationFunction());
72 mul->name(div->name() + "_mul");
73 luci::add_origin(mul, luci::get_origin(div));
74
75 replace(div).with(mul);
76
77 return true;
78}
