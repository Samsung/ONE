34template <>
35bool prepare_const_gen<float>(loco::ConstGen *const_node, const loco::TensorShape &tensorshape,
36 float value)
37{
38 LOGGER(l);
39
40 uint32_t const_num_elements = 1;
41
42 auto dtype = loco::DataType::FLOAT32;
43 const_node->dtype(dtype);
44
45 auto rank = tensorshape.rank();
46 const_node->rank(rank);
47 for (uint32_t r = 0; r < rank; ++r)
48 {
49 if (tensorshape.dim(r).known())
50 const_node->dim(r) = tensorshape.dim(r);
51 else
52 return false;
53
54 assert(tensorshape.dim(r).value() > 0);
55
56 const_num_elements *= tensorshape.dim(r).value();
57 }
58
59 INFO(l) << "prepare_const_gen : Elements = " << const_num_elements;
60
61 const_node->size<loco::DataType::FLOAT32>(const_num_elements);
62 for (uint32_t i = 0; i < const_num_elements; ++i)
63 {
64 const_node->at<loco::DataType::FLOAT32>(i) = value;
65 }
66
67 return true;
68}

70bool canonicalize_rsqrt(loco::Graph *graph, moco::TFRsqrt *node)
71{
91 auto nodeshape = moco::node_shape(node);
92 if (nodeshape.domain() == loco::Domain::Unknown)
93 {
94 // We need this shape information
95 assert(false); // this shouldn't happen, let's add an alarm
96 return false;
97 }
98 auto tensorshape = nodeshape.as<loco::TensorShape>();
99
100 if (!loco::dtype_known(node))
101 {
102 // We need type of this node
103 return false;
104 }
105
106 auto sqrt_node = graph->nodes()->create<loco::EltwiseSqrt>();
107 auto eltdiv_node = graph->nodes()->create<loco::EltwiseDiv>();
108 auto const_node = graph->nodes()->create<loco::ConstGen>();
109
110 auto dtype = loco::dtype_get(node);
111
112 switch (dtype)
113 {
114 case loco::DataType::FLOAT32:
115 if (!prepare_const_gen<float>(const_node, tensorshape, 1.0f))
116 throw oops::UserExn("Cannot handle unknown shape", node->name());
117 break;
118
119 default:
120 throw oops::UserExn("Unsupported data type", node->name());
121 }
122
123 auto node_A = node->x();
124
125 // update connections
126 sqrt_node->input(node_A);
127 eltdiv_node->lhs(const_node);
128 eltdiv_node->rhs(sqrt_node);
129
130 // replace node
131 replace(node).with(eltdiv_node);
132
133 return true;
134}
