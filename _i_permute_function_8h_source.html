41inline void UpdateOffsets(::onert::backend::ITensor *src, ::onert::backend::ITensor *dst,
42 const ::onert::ir::Shape &loop_shape, std::vector<size_t> &src_offsets,
43 std::vector<size_t> &dst_offsets)
44{
45 ShapeLoop(loop_shape, [&](const onert::ir::Coordinates &coords) {
46 src_offsets.emplace_back(src->calcOffset(coords));
47 dst_offsets.emplace_back(dst->calcOffset(coords));
48 });
49}

51inline void CopyStatic(const uint8_t *src_buffer, uint8_t *dst_buffer,
52 const std::vector<size_t> &src_offsets,
53 const std::vector<size_t> &dst_offsets, size_t copy_len)
54{
55 assert(src_offsets.size() == dst_offsets.size());
56 for (size_t i = 0; i < src_offsets.size(); ++i)
57 {
58 memcpy(dst_buffer + dst_offsets.at(i), src_buffer + src_offsets.at(i), copy_len);
59 }
60}

62inline void CopyDynamic(const ::onert::backend::ITensor *src, const ::onert::backend::ITensor *dst,
63 uint8_t *dst_buffer, const ::onert::ir::Shape &loop_shape, size_t copy_len)
64{
65 ShapeLoop(loop_shape, [&](const onert::ir::Coordinates &coords) {
66 // Copy src tensor's data to dst_buffer with calculated offset of dst tensor
67 memcpy(dst_buffer + dst->calcOffset(coords), src->buffer() + src->calcOffset(coords), copy_len);
68 });
69}

82 virtual void run() override
83 {
84 // TODO Optimization : Make control does not reach here? when (_src_tensors.size() == 0)
85 assert(_src_tensors.size() == _dst_tensors.size());
86 if (_src_tensors_offsets.size() == 0)
87 {
88 _src_tensors_offsets.resize(_src_tensors.size());
89 _dst_tensors_offsets.resize(_dst_tensors.size());
90 }
91 assert(_src_tensors.size() == _src_tensors_offsets.size());
92 assert(_src_tensors_offsets.size() == _dst_tensors_offsets.size());
93
94 for (size_t i = 0; i < _src_tensors.size(); ++i)
95 {
96 auto src_tensor = _src_tensors.at(i);
97 auto dst_tensor = _dst_tensors.at(i);
98 auto &src_offsets = _src_tensors_offsets.at(i);
99 auto &dst_offsets = _dst_tensors_offsets.at(i);
100 if (src_tensor != dst_tensor)
101 {
102 const auto rank = src_tensor->getShape().rank();
103 permute(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
104 }
105 }
106 }

113 void permute(backend::ITensor *src_tensor, backend::ITensor *dst_tensor, size_t rank,
114 std::vector<size_t> &src_offsets, std::vector<size_t> &dst_offsets)
115 {
116 if (src_tensor->total_size() == 0)
117 {
118 assert(dst_tensor->total_size() == 0);
119 return;
120 }
121
122 assert(src_tensor != dst_tensor);
123 if (underlying_type(src_tensor->data_type()) != underlying_type(dst_tensor->data_type()))
124 throw std::runtime_error("data type does not match");
125 switch (src_tensor->data_type())
126 {
127 case ir::DataType::FLOAT32:
128 permute<float>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
129 break;
130 case ir::DataType::INT32:
131 permute<int32_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
132 break;
133 case ir::DataType::UINT32:
134 permute<uint32_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
135 break;
136 case ir::DataType::BOOL8:
137 case ir::DataType::QUANT_UINT8_ASYMM:
138 case ir::DataType::UINT8:
139 permute<uint8_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
140 break;
141 case ir::DataType::QUANT_INT8_ASYMM:
142 case ir::DataType::QUANT_INT8_SYMM:
143 permute<int8_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
144 break;
145 case ir::DataType::INT64:
146 permute<int64_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
147 break;
148 case ir::DataType::QUANT_INT16_SYMM:
149 permute<int16_t>(src_tensor, dst_tensor, rank, src_offsets, dst_offsets);
150 break;
151 default:
152 throw std::runtime_error("IPermuteFunction: Not supported data type");
153 break;
154 }
155 }

159 template <class T>
160 void permute(backend::ITensor *src, backend::ITensor *dst, size_t rank,
161 std::vector<size_t> &src_offsets, std::vector<size_t> &dst_offsets)
162 {
163 assert(src->total_size() != 0 && dst->total_size() != 0);
164 // If dst is subtensor, we have to use clEnqueueMapBuffer instead of clEnqueueWirteBuffer
165 if (dst->needMemoryMap() && !dst->is_subtensor())
166 {
167 // A assertion to check mapping without calling map()
168 // Now there is no case where both src and dst have cl buffer.
169 assert(!src->needMemoryMap());
170
171 if (!src->has_padding() && !dst->has_padding() && src->layout() == dst->layout())
172 {
173 src->access([&](backend::ITensor &) { dst->enqueueWriteBuffer(src->buffer(), false); });
174 }
175 else
176 {
177 // TODO Optimize this block in case of that padding size of dst is big.
178 _buffers_map[dst].reserve(dst->total_size());
179 auto dst_buffer = _buffers_map[dst].data();
180 src->access([&](backend::ITensor &) {
181 permute<T>(src, dst, rank, dst_buffer, dst->total_size(), src_offsets, dst_offsets);
182 });
183 dst->enqueueWriteBuffer(dst_buffer, false);
184 }
185 }
186 else if (src->needMemoryMap() && !src->is_subtensor() && !src->has_padding() &&
187 !dst->has_padding() && src->layout() == dst->layout())
188 {
189 assert(!dst->needMemoryMap());
190 dst->access([&](backend::ITensor &) { src->enqueueReadBuffer(dst->buffer(), true); });
191 }
192 else
193 {
194 auto fn = [&](backend::ITensor &) {
195 dst->access([&](backend::ITensor &) {
196 permute<T>(src, dst, rank, dst->buffer(), dst->total_size(), src_offsets, dst_offsets);
197 });
198 };
199 src->access(fn);
200 }
201 }

203 template <class T>
204 void permute(backend::ITensor *src, backend::ITensor *dst, size_t rank, uint8_t *dst_buffer,
205 size_t dst_size, std::vector<size_t> &src_offsets, std::vector<size_t> &dst_offsets)
206 {
207 assert(dst_buffer != nullptr);
208 assert(dst_size == dst->total_size());
209
210 const auto permute_type = [&]() -> PermuteType {
211 if (src->layout() == ir::Layout::NHWC && dst->layout() == ir::Layout::NCHW)
212 {
213 return PermuteType::NHWC_TO_NCHW;
214 }
215 else if (src->layout() == ir::Layout::NCHW && dst->layout() == ir::Layout::NHWC)
216 {
217 return PermuteType::NCHW_TO_NHWC;
218 }
219 else
220 {
221 return PermuteType::COPY;
222 }
223 }();
224 if (rank == 4 && permute_type != PermuteType::COPY)
225 {
226 switch (permute_type)
227 {
228 case PermuteType::NHWC_TO_NCHW:
229 {
230 ir::FeatureShape shape;
231 auto dst_shape = dst->getShape();
232 shape.N = dst_shape.dim(0);
233 shape.C = dst_shape.dim(1);
234 shape.H = dst_shape.dim(2);
235 shape.W = dst_shape.dim(3);
236
237 typename feature::nchw::View<T>::Strides strides;
238 const auto start_offset = dst->calcOffset({0, 0, 0, 0});
239 strides.W = dst_shape.dim(3) == 1 ? 0 : dst->calcOffset({0, 0, 0, 1}) - start_offset;
240 strides.H = dst_shape.dim(2) == 1 ? 0 : dst->calcOffset({0, 0, 1, 0}) - start_offset;
241 strides.C = dst_shape.dim(1) == 1 ? 0 : dst->calcOffset({0, 1, 0, 0}) - start_offset;
242 strides.N = dst_shape.dim(0) == 1 ? 0 : dst->calcOffset({1, 0, 0, 0}) - start_offset;
243
244 const feature::nhwc::Reader<T> from(src);
245 feature::nchw::View<T> into(shape, strides,
246 reinterpret_cast<T *>(dst_buffer + start_offset), dst_size);
247 feature::iterate(shape) << [&](uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) {
248 const auto value = from.at(batch, row, col, ch);
249 into.at(batch, ch, row, col) = value;
250 };
251 break;
252 }
253 case PermuteType::NCHW_TO_NHWC:
254 {
255 ir::FeatureShape shape;
256 auto dst_shape = dst->getShape();
257 shape.N = dst_shape.dim(0);
258 shape.H = dst_shape.dim(1);
259 shape.W = dst_shape.dim(2);
260 shape.C = dst_shape.dim(3);
261
262 typename feature::nhwc::View<T>::Strides strides;
263 const auto start_offset = dst->calcOffset({0, 0, 0, 0});
264 strides.C = dst_shape.dim(3) == 1 ? 0 : dst->calcOffset({0, 0, 0, 1}) - start_offset;
265 strides.W = dst_shape.dim(2) == 1 ? 0 : dst->calcOffset({0, 0, 1, 0}) - start_offset;
266 strides.H = dst_shape.dim(1) == 1 ? 0 : dst->calcOffset({0, 1, 0, 0}) - start_offset;
267 strides.N = dst_shape.dim(0) == 1 ? 0 : dst->calcOffset({1, 0, 0, 0}) - start_offset;
268
269 const feature::nchw::Reader<T> from(src);
270 feature::nhwc::View<T> into(shape, strides,
271 reinterpret_cast<T *>(dst_buffer + start_offset), dst_size);
272 feature::iterate(shape) << [&](uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) {
273 const auto value = from.at(batch, ch, row, col);
274 into.at(batch, row, col, ch) = value;
275 };
276 break;
277 }
278 default:
279 {
280 throw std::runtime_error("Unsupported Permutation");
281 break;
282 }
283 }
284 }
285 else if (!src->has_padding() && !dst->has_padding())
286 {
287 auto src_size = src->total_size();
288 assert(src_size <= dst->total_size());
289 memcpy(dst_buffer, src->buffer(), src_size);
290 }
291 else
292 {
293 auto loop_shape = src->getShape();
294 const auto copy_axis = loop_shape.rank() - 1;
295 const auto copy_len = loop_shape.dim(copy_axis) * sizeof(T);
296 loop_shape.dim(copy_axis) = 1;
297
298 if (src->is_dynamic())
299 {
300 assert(dst->is_dynamic());
301 CopyDynamic(src, dst, dst_buffer, loop_shape, copy_len);
302 }
303 else
304 {
305 // TODO Uncomment the assertion below
306 // assert(!dst->is_dynamic() || dst is output of graph);
307 if (src_offsets.size() == 0)
308 {
309 assert(dst_offsets.size() == 0);
310
311 auto loop_shape = src->getShape();
312 const auto copy_axis = loop_shape.rank() - 1;
313 loop_shape.dim(copy_axis) = 1;
314 UpdateOffsets(src, dst, loop_shape, src_offsets, dst_offsets);
315 }
316 CopyStatic(src->buffer(), dst_buffer, src_offsets, dst_offsets, copy_len);
317 }
318 }
319 }
