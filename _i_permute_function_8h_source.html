37inline void UpdateOffsets(::onert::backend::ITensor *src, ::onert::backend::ITensor *dst,
38 const ::onert::ir::Shape &loop_shape, std::vector<size_t> &src_offsets,
39 std::vector<size_t> &dst_offsets)
40{
41 ShapeLoop(loop_shape, [&](const onert::ir::Coordinates &coords) {
42 src_offsets.emplace_back(src->calcOffset(coords));
43 dst_offsets.emplace_back(dst->calcOffset(coords));
44 });
45}

47inline void CopyStatic(const uint8_t *src_buffer, uint8_t *dst_buffer,
48 const std::vector<size_t> &src_offsets,
49 const std::vector<size_t> &dst_offsets, size_t copy_len)
50{
51 assert(src_offsets.size() == dst_offsets.size());
52 for (size_t i = 0; i < src_offsets.size(); ++i)
53 {
54 memcpy(dst_buffer + dst_offsets.at(i), src_buffer + src_offsets.at(i), copy_len);
55 }
56}

58inline void CopyDynamic(const ::onert::backend::ITensor *src, const ::onert::backend::ITensor *dst,
59 uint8_t *dst_buffer, const ::onert::ir::Shape &loop_shape, size_t copy_len)
60{
61 ShapeLoop(loop_shape, [&](const onert::ir::Coordinates &coords) {
62 // Copy src tensor's data to dst_buffer with calculated offset of dst tensor
63 memcpy(dst_buffer + dst->calcOffset(coords), src->buffer() + src->calcOffset(coords), copy_len);
64 });
65}

89 // TODO make src const by proving const access()
90 template <class T>
91 void permute(backend::ITensor *src, backend::ITensor *dst, size_t rank,
92 std::vector<size_t> &src_offsets, std::vector<size_t> &dst_offsets)
93 {
94 assert(src->total_size() != 0 && dst->total_size() != 0);
95 // If dst is subtensor, we have to use clEnqueueMapBuffer instead of clEnqueueWirteBuffer
96 if (dst->needMemoryMap() && !dst->is_subtensor())
97 {
98 // A assertion to check mapping without calling map()
99 // Now there is no case where both src and dst have cl buffer.
100 assert(!src->needMemoryMap());
101
102 if (!src->has_padding() && !dst->has_padding() && src->layout() == dst->layout())
103 {
104 src->access([&](backend::ITensor &) { dst->enqueueWriteBuffer(src->buffer(), false); });
105 }
106 else
107 {
108 // TODO Optimize this block in case of that padding size of dst is big.
109 _buffers_map[dst].reserve(dst->total_size());
110 auto dst_buffer = _buffers_map[dst].data();
111 src->access([&](backend::ITensor &) {
112 permute<T>(src, dst, rank, dst_buffer, dst->total_size(), src_offsets, dst_offsets);
113 });
114 dst->enqueueWriteBuffer(dst_buffer, false);
115 }
116 }
117 else if (src->needMemoryMap() && !src->is_subtensor() && !src->has_padding() &&
118 !dst->has_padding() && src->layout() == dst->layout())
119 {
120 assert(!dst->needMemoryMap());
121 dst->access([&](backend::ITensor &) { src->enqueueReadBuffer(dst->buffer(), true); });
122 }
123 else
124 {
125 auto fn = [&](backend::ITensor &) {
126 dst->access([&](backend::ITensor &) {
127 permute<T>(src, dst, rank, dst->buffer(), dst->total_size(), src_offsets, dst_offsets);
128 });
129 };
130 src->access(fn);
131 }
132 }

134 template <class T>
135 void permute(backend::ITensor *src, backend::ITensor *dst, size_t rank, uint8_t *dst_buffer,
136 size_t dst_size, std::vector<size_t> &src_offsets, std::vector<size_t> &dst_offsets)
137 {
138 assert(dst_buffer != nullptr);
139 assert(dst_size == dst->total_size());
140
141 const auto permute_type = [&]() -> PermuteType {
142 if (src->layout() == ir::Layout::NHWC && dst->layout() == ir::Layout::NCHW)
143 {
144 return PermuteType::NHWC_TO_NCHW;
145 }
146 else if (src->layout() == ir::Layout::NCHW && dst->layout() == ir::Layout::NHWC)
147 {
148 return PermuteType::NCHW_TO_NHWC;
149 }
150 else
151 {
152 return PermuteType::COPY;
153 }
154 }();
155 if (rank == 4 && permute_type != PermuteType::COPY)
156 {
157 switch (permute_type)
158 {
159 case PermuteType::NHWC_TO_NCHW:
160 {
161 ir::FeatureShape shape;
162 auto dst_shape = dst->getShape();
163 shape.N = dst_shape.dim(0);
164 shape.C = dst_shape.dim(1);
165 shape.H = dst_shape.dim(2);
166 shape.W = dst_shape.dim(3);
167
168 typename feature::nchw::View<T>::Strides strides;
169 const auto start_offset = dst->calcOffset({0, 0, 0, 0});
170 strides.W = dst_shape.dim(3) == 1 ? 0 : dst->calcOffset({0, 0, 0, 1}) - start_offset;
171 strides.H = dst_shape.dim(2) == 1 ? 0 : dst->calcOffset({0, 0, 1, 0}) - start_offset;
172 strides.C = dst_shape.dim(1) == 1 ? 0 : dst->calcOffset({0, 1, 0, 0}) - start_offset;
173 strides.N = dst_shape.dim(0) == 1 ? 0 : dst->calcOffset({1, 0, 0, 0}) - start_offset;
174
175 const feature::nhwc::Reader<T> from(src);
176 feature::nchw::View<T> into(shape, strides,
177 reinterpret_cast<T *>(dst_buffer + start_offset), dst_size);
178 feature::iterate(shape) << [&](uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) {
179 const auto value = from.at(batch, row, col, ch);
180 into.at(batch, ch, row, col) = value;
181 };
182 break;
183 }
184 case PermuteType::NCHW_TO_NHWC:
185 {
186 ir::FeatureShape shape;
187 auto dst_shape = dst->getShape();
188 shape.N = dst_shape.dim(0);
189 shape.H = dst_shape.dim(1);
190 shape.W = dst_shape.dim(2);
191 shape.C = dst_shape.dim(3);
192
193 typename feature::nhwc::View<T>::Strides strides;
194 const auto start_offset = dst->calcOffset({0, 0, 0, 0});
195 strides.C = dst_shape.dim(3) == 1 ? 0 : dst->calcOffset({0, 0, 0, 1}) - start_offset;
196 strides.W = dst_shape.dim(2) == 1 ? 0 : dst->calcOffset({0, 0, 1, 0}) - start_offset;
197 strides.H = dst_shape.dim(1) == 1 ? 0 : dst->calcOffset({0, 1, 0, 0}) - start_offset;
198 strides.N = dst_shape.dim(0) == 1 ? 0 : dst->calcOffset({1, 0, 0, 0}) - start_offset;
199
200 const feature::nchw::Reader<T> from(src);
201 feature::nhwc::View<T> into(shape, strides,
202 reinterpret_cast<T *>(dst_buffer + start_offset), dst_size);
203 feature::iterate(shape) << [&](uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) {
204 const auto value = from.at(batch, ch, row, col);
205 into.at(batch, row, col, ch) = value;
206 };
207 break;
208 }
209 default:
210 {
211 throw std::runtime_error("Unsupported Permutation");
212 break;
213 }
214 }
215 }
216 else if (!src->has_padding() && !dst->has_padding())
217 {
218 auto src_size = src->total_size();
219 assert(src_size <= dst->total_size());
220 memcpy(dst_buffer, src->buffer(), src_size);
221 }
222 else
223 {
224 auto loop_shape = src->getShape();
225 const auto copy_axis = loop_shape.rank() - 1;
226 const auto copy_len = loop_shape.dim(copy_axis) * sizeof(T);
227 loop_shape.dim(copy_axis) = 1;
228
229 if (src->is_dynamic())
230 {
231 assert(dst->is_dynamic());
232 CopyDynamic(src, dst, dst_buffer, loop_shape, copy_len);
233 }
234 else
235 {
236 // TODO Uncomment the assertion below
237 // assert(!dst->is_dynamic() || dst is output of graph);
238 if (src_offsets.size() == 0)
239 {
240 assert(dst_offsets.size() == 0);
241
242 auto loop_shape = src->getShape();
243 const auto copy_axis = loop_shape.rank() - 1;
244 loop_shape.dim(copy_axis) = 1;
245 UpdateOffsets(src, dst, loop_shape, src_offsets, dst_offsets);
246 }
247 CopyStatic(src->buffer(), dst_buffer, src_offsets, dst_offsets, copy_len);
248 }
249 }
250 }
