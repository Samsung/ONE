82#define CHECK_OR_FALSE(condition) \
83 if (not(condition)) \
84 return false;

86bool PReluPattern::matched()
87{
88 // check pattern
89 CHECK_OR_FALSE(luci::fill(&_relu, &_mul_half).with_commutative_args_of(_add_ofm));
90 CHECK_OR_FALSE(luci::fill(&_mul_alpha, &_const_half).with_commutative_args_of(_mul_half));
91 CHECK_OR_FALSE(luci::fill(&_sub, &_const_alpha).with_commutative_args_of(_mul_alpha));
92
93 CHECK_OR_FALSE(luci::fill(&_ifm, &_abs).with_args_of(_sub));
94
95 CHECK_OR_FALSE(_relu->features() == _ifm);
96 CHECK_OR_FALSE(_abs->x() == _ifm);
97
98 // Check Activation to be NONE
99 CHECK_OR_FALSE(_sub->fusedActivationFunction() == luci::FusedActFunc::NONE);
100 CHECK_OR_FALSE(_mul_alpha->fusedActivationFunction() == luci::FusedActFunc::NONE);
101 CHECK_OR_FALSE(_mul_half->fusedActivationFunction() == luci::FusedActFunc::NONE);
102 CHECK_OR_FALSE(_add_ofm->fusedActivationFunction() == luci::FusedActFunc::NONE);
103
104 // TODO support other types?
105 // check if _const_half is really FLOAT32 & 0.5
106 CHECK_OR_FALSE(_const_half->dtype() == loco::DataType::FLOAT32);
107 CHECK_OR_FALSE(_const_half->size<loco::DataType::FLOAT32>() == 1);
108 CHECK_OR_FALSE(_const_half->at<loco::DataType::FLOAT32>(0) == 0.5);
109
110 // check _const_alpha condition
111 CHECK_OR_FALSE(_const_alpha->dtype() == loco::DataType::FLOAT32);
112 // TODO add more if needed
113
114 return true;
115}

134luci::CirclePRelu *FusePRelu::create_prelu(loco::Graph *graph)
135{
136 assert(graph);
137
138 auto prelu = graph->nodes()->create<luci::CirclePRelu>();
139 prelu->input(_p._ifm);
140 prelu->alpha(_p._const_alpha);
141 prelu->name(_p._add_ofm->name() + "_prelu");
142 return prelu;
143}
144
145void FusePRelu::apply()
146{
147 auto graph = _p._add_ofm->graph();
148
149 auto prelu = create_prelu(graph);
150
151 // set origin
152 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> origin_vec{
153 luci::get_origin(_p._relu), luci::get_origin(_p._abs), luci::get_origin(_p._sub),
154 luci::get_origin(_p._mul_alpha), luci::get_origin(_p._mul_half), luci::get_origin(_p._add_ofm)};
155
156 luci::add_origin(prelu, luci::composite_origin(origin_vec));
157
158 replace(_p._add_ofm).with(prelu);
159}

166bool fuse_prelu(luci::CircleAdd *add)
167{
168 assert(add);
169
170 PReluPattern pattern(add);
171 if (pattern.matched())
172 {
173 FusePRelu fuse(pattern);
174 fuse.apply();
175 return true;
176 }
177 return false;
178}

185bool FusePReluPass::run(loco::Graph *g)
186{
187 bool changed = false;
188
189 for (auto node : loco::active_nodes(loco::output_nodes(g)))
190 {
191 auto add = dynamic_cast<luci::CircleAdd *>(node);
192 if (not add)
193 continue;
194
195 if (fuse_prelu(add))
196 changed = true;
197 }
198
199 return changed;
200}
