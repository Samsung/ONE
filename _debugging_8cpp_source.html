149 }

152 return "(unknown)";
153 }
154 }
155
156 return "(none)";

304pp::LinearDocument describe(const OpTree &t, bool verbose = false)
305{
306 pp::LinearDocument doc;
307
308 struct Frame
309 {
310 public:
311 Frame(const coco::Op *op) : _op{op}, _indicator{0}
312 {
313 // op SHOULD BE valid
314 assert(_op != nullptr);
315 }
316
317 public:
321 const coco::Op *op(void) const { return _op; }
322
333 uint32_t indicator(void) const { return _indicator; }
334
335 public:
336 void advance(void) { _indicator += 1; }
337
338 private:
339 const coco::Op *_op;
340 uint32_t _indicator;
341 };
342
343 std::stack<Frame> stack;
344
345 stack.emplace(t.root());
346
347 while (stack.size() > 0)
348 {
349 auto op = stack.top().op();
350 uint32_t indicator = stack.top().indicator();
351
352 if (indicator == 0)
353 {
354 doc.append(op_kind(op), " (", op, ")");
355
356 doc.indent();
357 stack.top().advance();
358
359 // TODO Need to update it to better design for verbose flag
360 if (verbose)
361 {
362 auto set = [op](pp::LinearDocument &doc) { doc.append(describe(op)); };
363 auto desc = section("op").build(set);
364 doc.append(desc);
365 }
366 }
367 else if (indicator < op->arity() + 1)
368 {
369 stack.top().advance();
370 stack.emplace(op->arg(indicator - 1));
371 }
372 else
373 {
374 assert(indicator == op->arity() + 1);
375 doc.unindent();
376 stack.pop();
377 }
378 }
379
380 return doc;
381}

435{
436 pp::LinearDocument doc;
437
438 doc.append("addr: ", ins);
439 doc.append("kind: ", kind(ins));
440 doc.append("parent: ", ins->parent());
441
442 // TODO Need to update it to better design for verbose flag
443 if (verbose)
444 {
445 if (auto eval = ins->asEval())
446 {
447 auto optset = [eval, verbose](pp::LinearDocument &doc) {
448 doc.append(describe(OpTree(eval->op()), verbose));
449 };
450 auto optdesc = section("op").build(optset);
451 doc.append(optdesc);
452
453 auto outset = [eval](pp::LinearDocument &doc) { doc.append(describe(eval->out())); };
454 auto outdesc = section("out").build(outset);
455 doc.append(outdesc);
456 }
457 else if (auto copy = ins->asCopy())
458 {
459 auto from = [copy](pp::LinearDocument &doc) { doc.append(describe(copy->from())); };
460 auto into = [copy](pp::LinearDocument &doc) { doc.append(describe(copy->into())); };
461
462 auto fdesc = section("from").build(from);
463 doc.append(fdesc);
464
465 auto idesc = section("into").build(into);
466 doc.append(idesc);
467 }
468 }
469
470 return doc;
