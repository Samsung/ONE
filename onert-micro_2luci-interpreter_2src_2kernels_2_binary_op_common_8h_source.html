43template <typename T>
44void fillArithmeticActivationRange(luci_interpreter_pal::ArithmeticParams &p, Activation act)
45{
46 static_assert(one_of_types<T, float, int32_t, int64_t>(), "Unsupported dtype");
47
48 if (std::is_same<T, float>::value)
49 calculateActivationRange(act, &p.float_activation_min, &p.float_activation_max);
50 if (std::is_same<T, int32_t>::value)
51 calculateActivationRange(act, &p.quantized_activation_min, &p.quantized_activation_max);
52 else
53 calculateActivationRange(act, &p.int64_activation_min, &p.int64_activation_max);
54}

58template <typename T, typename TISOFunc = nullptr_t, typename TISOBroadcastFunc = nullptr_t,
59 typename Options = nullptr_t>
60void evalTISOKernel(TISOFunc tiso_func, TISOBroadcastFunc tiso_broadcast_func,
61 kernels::TISOKernel *kernel, kernels::TISOData *kernel_data,
62 const Options *options, RuntimeShape &&input_shape_1,
63 RuntimeShape &&input_shape_2)
64{
65 const auto *output = kernel->output();
66
67 luci_interpreter_pal::ArithmeticParams params{};
68 fillArithmeticActivationRange<T>(params, luci_actfunc(options->fused_activation_function()));
69
70 const bool need_broadcast =
71 luci_interpreter_pal::ProcessBroadcastShapes(input_shape_1, input_shape_2, &params);
72
73 if (need_broadcast)
74 {
75 tiso_broadcast_func(params, input_shape_1, kernels::getTensorData<T>(kernel_data->input1_data),
76 input_shape_2, kernels::getTensorData<T>(kernel_data->input2_data),
77 kernels::getTensorShape(output),
78 kernels::getTensorData<T>(kernel_data->output_data));
79 }
80 else
81 {
82 const int flat_size = input_shape_1.flatSize();
83 tiso_func(params, flat_size, kernels::getTensorData<T>(kernel_data->input1_data),
84 kernels::getTensorData<T>(kernel_data->input2_data),
85 kernels::getTensorData<T>(kernel_data->output_data));
86 }
87}

89template <typename T, typename TISOFunc = nullptr_t, typename TISOBroadcastFunc = nullptr_t,
90 typename Options = nullptr_t>
91void evalTISOInplaceKernel(TISOFunc tiso_func, TISOBroadcastFunc tiso_broadcast_func,
92 kernels::TISOKernel *kernel, const Options *options,
93 RuntimeShape &&input_shape_1, RuntimeShape &&input_shape_2)
94{
95 uint8_t *inplace_data_ptr = nullptr;
96 circle::Tensor *input_inplace_tensor = nullptr;
97
98 kernels::TISOData kernel_data = kernel->readInplaceData(inplace_data_ptr, input_inplace_tensor);
99
100 evalTISOKernel<T, TISOFunc, TISOBroadcastFunc, Options>(
101 tiso_func, tiso_broadcast_func, kernel, &kernel_data, options, std::move(input_shape_1),
102 std::move(input_shape_2));
103
104 BaseRuntimeGraph *runtime_graph = kernel->runtime_graph();
105
106 runtime_graph->makeInplaceOperation(input_inplace_tensor, kernel->output());
107 if (input_inplace_tensor == kernel->input1())
108 {
109 runtime_graph->makeInplaceOperation(kernel->input2(), nullptr);
110 }
111 else
112 {
113 runtime_graph->makeInplaceOperation(kernel->input1(), nullptr);
114 }
115}

118template <typename T, typename TISOFunc = nullptr_t, typename TISOBroadcastFunc = nullptr_t,
119 typename Options = nullptr_t>
120void evalTISOQuantizedKernel(TISOFunc tiso_func, TISOBroadcastFunc tiso_broadcast_func,
121 kernels::TISOKernel *kernel, kernels::TISOData *kernel_data,
122 const Options *options)
123{
124 const auto *input1 = kernel->input1();
125 const auto *input2 = kernel->input2();
126 const auto *output = kernel->output();
127
128 const auto input1_scale = static_cast<double>(Tensor::scale(input1));
129 const auto input2_scale = static_cast<double>(Tensor::scale(input2));
130 const auto output_scale = static_cast<double>(Tensor::scale(output));
131
132 const int left_shift = 20;
133 const double twice_max_input_scale = 2 * std::max(input1_scale, input2_scale);
134 const double real_input1_multiplier = input1_scale / twice_max_input_scale;
135 const double real_input2_multiplier = input2_scale / twice_max_input_scale;
136 const double real_output_multiplier = twice_max_input_scale / ((1 << left_shift) * output_scale);
137
138 int32_t input1_multiplier{}, input2_multiplier{}, output_multiplier{};
139 int input1_shift{}, input2_shift{}, output_shift{};
140 kernels::quantizeMultiplierSmallerThanOneExp(real_input1_multiplier, &input1_multiplier,
141 &input1_shift);
142 kernels::quantizeMultiplierSmallerThanOneExp(real_input2_multiplier, &input2_multiplier,
143 &input2_shift);
144 kernels::quantizeMultiplierSmallerThanOneExp(real_output_multiplier, &output_multiplier,
145 &output_shift);
146
147 int32_t activation_min{};
148 int32_t activation_max{};
149 kernels::calculateActivationRangeQuantized(luci_actfunc(options->fused_activation_function()),
150 output, &activation_min, &activation_max);
151
152 luci_interpreter_pal::ArithmeticParams params{};
153 params.left_shift = left_shift;
154 // The kernel expects inputs' zero points to be negated.
155 params.input1_offset = -Tensor::zero_point(input1); // Note the '-'.
156 params.input1_multiplier = input1_multiplier;
157 params.input1_shift = input1_shift;
158 params.input2_offset = -Tensor::zero_point(input2); // Note the '-'.
159 params.input2_multiplier = input2_multiplier;
160 params.input2_shift = input2_shift;
161 params.output_offset = Tensor::zero_point(output);
162 params.output_multiplier = output_multiplier;
163 params.output_shift = output_shift;
164 params.quantized_activation_min = activation_min;
165 params.quantized_activation_max = activation_max;
166
167 const bool need_broadcast = luci_interpreter_pal::ProcessBroadcastShapes(
168 kernels::getTensorShape(input1), kernels::getTensorShape(input2), &params);
169
170 if (need_broadcast)
171 {
172 tiso_broadcast_func(
173 params, kernels::getTensorShape(input1), kernels::getTensorData<T>(kernel_data->input1_data),
174 kernels::getTensorShape(input2), kernels::getTensorData<T>(kernel_data->input2_data),
175 kernels::getTensorShape(output), kernels::getTensorData<T>(kernel_data->output_data));
176 }
177 else
178 {
179 tiso_func(params, kernels::getTensorShape(input1),
180 kernels::getTensorData<uint8_t>(kernel_data->input1_data),
181 kernels::getTensorShape(input2), kernels::getTensorData<T>(kernel_data->input2_data),
182 kernels::getTensorShape(output), kernels::getTensorData<T>(kernel_data->output_data));
183 }
184}
