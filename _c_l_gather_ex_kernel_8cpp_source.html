59inline Status validate_arguments(const ITensorInfo *input, const ITensorInfo *indices,
60 const ITensorInfo *output, int axis)
61{
62 const uint32_t actual_axis = wrap_around(axis, static_cast<int>(input->num_dimensions()));
63 ARM_COMPUTE_RETURN_ERROR_ON(indices->num_dimensions() > 3);
64 ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 4);
65 ARM_COMPUTE_ERROR_ON(input->num_dimensions() + indices->num_dimensions() - 1 > 4);
66 ARM_COMPUTE_RETURN_ERROR_ON(actual_axis >= input->num_dimensions());
67 ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output);
68 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(
69 input, 1, DataType::U8, DataType::S8, DataType::QASYMM8, DataType::U16, DataType::S16,
70 DataType::U32, DataType::S32, DataType::F16, DataType::F32);
71
72 if (output->total_size() != 0)
73 {
74 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
75 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
76 TensorShape output_shape = arm_compute::misc::shape_calculator::compute_gather_shape_ex(
77 input->tensor_shape(), indices->tensor_shape(), actual_axis);
78 ARM_COMPUTE_RETURN_ERROR_ON(output_shape.total_size() != output->tensor_shape().total_size());
79 }
80
81 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(indices, 1, DataType::U32, DataType::S32);
82
83 return Status{};
84}

86std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *indices,
87 ITensorInfo *output, int axis)
88{
89 ARM_COMPUTE_ERROR_ON_NULLPTR(input, output, indices);
90 const uint32_t actual_axis = wrap_around(axis, static_cast<int>(input->num_dimensions()));
91 std::unique_ptr<ITensorInfo> output_info = input->clone();
92 output_info->set_tensor_shape(arm_compute::misc::shape_calculator::compute_gather_shape_ex(
93 input->tensor_shape(), indices->tensor_shape(), actual_axis));
94 // Output auto initialization if not yet initialized
95 auto_init_if_empty((*output), output_info->tensor_shape(), 1, input->data_type());
96
97 // Create window
98 Window win = calculate_max_window(*output, Steps());
99 output->set_valid_region(ValidRegion(Coordinates(), output->tensor_shape()));
100
101 return std::make_pair(Status{}, win);
102}

111void CLGatherExKernel::configure(const ICLTensor *input, const ICLTensor *indices,
112 ICLTensor *output, int axis)
113{
114 ARM_COMPUTE_ERROR_ON_NULLPTR(input, output, indices);
115 ARM_COMPUTE_ERROR_THROW_ON(
116 validate_arguments(input->info(), indices->info(), output->info(), axis));
117
118 // Configure kernel window
119 auto win_config =
120 validate_and_configure_window(input->info(), indices->info(), output->info(), axis);
121 ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
122
123 _input = input;
124 _output = output;
125 _indices = indices;
126 _axis = wrap_around(axis, static_cast<int>(input->info()->num_dimensions()));
127
128 // Set build options
129 CLBuildOptions build_opts;
130 build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
131 build_opts.add_option("-DOUTPUT_DIM_Z=" +
132 support::cpp11::to_string(output->info()->dimension(2)));
133 build_opts.add_option("-DINPUT_DIM_Z=" + support::cpp11::to_string(input->info()->dimension(2)));
134 build_opts.add_option("-DAXIS=" + support::cpp11::to_string(_axis));
135 build_opts.add_option("-DINDICES_DIM=" +
136 support::cpp11::to_string(indices->info()->num_dimensions()));
137
138 // Create kernel
139 _kernel = static_cast<cl::Kernel>(
140 CLKernelLibraryEx::get().create_kernel("gather_ex", build_opts.options()));
141 ICLKernel::configure_internal(win_config.second);
142}

144Status CLGatherExKernel::validate(const ITensorInfo *input, const ITensorInfo *indices,
145 const ITensorInfo *output, int axis)
146{
147 ARM_COMPUTE_RETURN_ON_ERROR(validate_arguments(input, indices, output, axis));
148 ARM_COMPUTE_RETURN_ON_ERROR(validate_and_configure_window(input->clone().get(),
149 indices->clone().get(),
150 output->clone().get(), axis)
151 .first);
152 return Status{};

155void CLGatherExKernel::run(const Window &window, cl::CommandQueue &queue)
156{
157 ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
158 ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
159
160 Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ, 4);
161 unsigned int idx = 0;
162 add_4D_tensor_argument(idx, _input, window_collapsed);
163 add_3D_tensor_argument(idx, _indices, window_collapsed);
164 add_4D_tensor_argument(idx, _output, window_collapsed);
165 enqueue(queue, *this, window_collapsed, lws_hint());
166}
