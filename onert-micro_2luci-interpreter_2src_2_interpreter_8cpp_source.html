38// Construct interpreter with configurations
39Interpreter::Interpreter(const char *model_data_raw, const InterpreterConfigure &configuration)
40{
41 _runtime_module = std::make_unique<RuntimeModule>();
42
43 // Note:
44 // configuration._input_buf_size, configuration._temp_buf_size, configuration._output_buf_size
45 // will be removed and will be read from circle file
46 if (configuration.isStaticManager())
47 {
48 _memory_manager = std::make_unique<StaticMemoryManager>(
49 configuration._input_buf_size, configuration._temp_buf_size, configuration._output_buf_size);
50 }
51 else
52 {
53 _memory_manager = std::make_unique<SimpleMemoryManager>();
54 }
55
56 _memory_manager->is_allocate_input(configuration.getAllocateInputValue());
57
58 ModuleLoader loader(model_data_raw, _runtime_module.get(), _memory_manager.get());
59 loader.load(configuration.isStaticManager());
60}
