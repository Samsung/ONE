52void validate_tensor_name(const std::string &tensor_name, const char *path)
53{
54 // Note that Tensorflow tensor name format is
55 // operation name ":" index, e.g., "in/placeholder:0"
56 int pos = tensor_name.find(":");
57 if (pos == std::string::npos)
58 throw oops::UserExn("Missing index separator, ':'", "name", tensor_name, "file", path);
59
60 if (tensor_name.length() == pos + 1) // ':' is the last char
61 throw oops::UserExn("Missing tensor index after ':'", "name", tensor_name, "file", path);
62
63 // 1. Validating operation name.
64 // for naming format, refer to https://www.tensorflow.org/api_docs/python/tf/Operation#__init__
65 // First char is in the form of "[A-Za-z0-9.]"
66 // and the rest chars are in the form of "[A-Za-z0-9_.\\-/]*"
67 std::string op_name = tensor_name.substr(0, pos);
68
69 // first character
70 if (!(isalnum(op_name[0]) || op_name[0] == '.'))
71 throw oops::UserExn("Wrong tensor name format", "name", tensor_name, "file", path);
72
73 // and the rest chars
74 for (int i = 1; i < op_name.length(); i++)
75 if (!(isalnum(op_name[i]) || std::string("_.\\-/").find(op_name[i]) != std::string::npos))
76 throw oops::UserExn("Wrong tensor name format", "name", tensor_name, "file", path);
77
78 // 2. validating index after ":"
79 std::string index = tensor_name.substr(pos + 1, op_name.length() - pos - 1);
80
81 if (!is_num(index))
82 throw oops::UserExn("Wrong tensor name format", "name", tensor_name, "file", path);
83}

85void check_duplicate(tfinfo::v2::TensorSignatures &tensors, const char *path)
86{
87 std::map<std::string, bool> tool;
88 for (auto &tensor : tensors)
89 {
90 if (tool.find(tensor->name()) != tool.end())
91 throw oops::UserExn("Duplicate tensor definition", "name", tensor->name(), "file", path);
92 else
93 tool[tensor->name()] = true;
94 }
95}

97void convert(tfinfo_v2_proto::InfoDef &info_def, tfinfo::v2::TensorSignatures &tensors,
98 const char *path)
99{
100 // processing input. Note that there could be no input.
101 if (auto input_size = info_def.input_size())
102 {
103 for (int i = 0; i < input_size; i++)
104 {
105 auto input_def = info_def.input().Get(i);
106
107 auto name = input_def.name();
108 validate_tensor_name(name, path);
109
110 auto tensor = std::make_unique<tfinfo::v2::TensorSignature>(
111 tfinfo::v2::TensorSignature::Kind::Input, name);
112
113 // when there is dim attribute for unknown shape
114 if (input_def.dim_size() > 0)
115 {
116 for (int d = 0; d < input_def.dim().size(); d++)
117 {
118 auto dim = input_def.dim(d);
119 tensor->shapeHint().add(dim.axis(), dim.size());
120 }
121 }
122
123 tensors.emplace_back(std::move(tensor));
124 }
125 }
126
127 // processing output
128 auto output_size = info_def.output_size();
129 if (output_size == 0)
130 throw oops::UserExn("Missing output node. At least 1 output node must exist", "file", path);
131
132 if (auto output_node_size = info_def.output_size())
133 {
134 for (int i = 0; i < output_node_size; i++)
135 {
136 auto name = info_def.output().Get(i).name();
137 validate_tensor_name(name, path);
138
139 auto tensor = std::make_unique<tfinfo::v2::TensorSignature>(
140 tfinfo::v2::TensorSignature::Kind::Output, name);
141 tensors.emplace_back(std::move(tensor));
142 }
143 }
144
145 check_duplicate(tensors, path);
146}

162TensorSignatures load(std::istream *stream, const char *path_for_error_msg)
163{
164 tfinfo_v2_proto::InfoDef info_def;
165
166 if (!load_text(stream, info_def))
167 {
168 throw oops::UserExn("Cannot parse the info file", "path", path_for_error_msg);
169 }
170
171 TensorSignatures tensors;
172
173 convert(info_def, tensors, path_for_error_msg);
174
175 return tensors;
176}
