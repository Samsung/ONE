30// Update u8 node to i16
31// Qparam of i16 is inferred from the qparam of u8
32void update_u8_to_i16(luci::CircleNode *node)
33{
34 assert(node->dtype() == loco::DataType::U8); // FIX_CALLER_UNLESS
35
36 node->dtype(loco::DataType::S16);
37
38 auto qparam = node->quantparam();
39 assert(qparam);
40 assert(qparam->scale.size() == 1);
41 assert(qparam->zerop.size() == 1);
42
43 auto u8_scale = qparam->scale[0];
44 auto u8_zerop = qparam->zerop[0];
45
46 auto min = u8_scale * (-u8_zerop);
47 auto max = u8_scale * (255 - u8_zerop);
48
49 float s16_scale{0};
50 int64_t s16_zerop{0};
51 float nudged_min{0};
52 float nudged_max{0};
53
54 compute_sym_scale(min, max, s16_scale, nudged_min, nudged_max);
55
56 auto quantparam = std::make_unique<CircleQuantParam>();
57 quantparam->scale.push_back(s16_scale);
58 quantparam->zerop.push_back(s16_zerop);
59
60 node->quantparam(std::move(quantparam));
61}

63// Update i16 node to u8 node
64// Qparam of u8 is inferred from the qparam of i16
65void update_i16_to_u8(luci::CircleNode *node)
66{
67 assert(node->dtype() == loco::DataType::S16); // FIX_CALLER_UNLESS
68
69 node->dtype(loco::DataType::U8);
70
71 auto qparam = node->quantparam();
72 assert(qparam);
73 assert(qparam->scale.size() == 1);
74 assert(qparam->zerop.size() == 1);
75
76 auto s16_scale = qparam->scale[0];
77 assert(qparam->zerop[0] == 0);
78
79 auto max = s16_scale * std::numeric_limits<int16_t>::max();
80 auto min = -max;
81
82 float u8_scale{0};
83 int64_t u8_zerop{0};
84 float nudged_min{0};
85 float nudged_max{0};
86
87 compute_asym_scale_zp(min, max, u8_scale, u8_zerop, nudged_min, nudged_max);
88
89 auto quantparam = std::make_unique<CircleQuantParam>();
90 quantparam->scale.push_back(u8_scale);
91 quantparam->zerop.push_back(u8_zerop);
92
93 node->quantparam(std::move(quantparam));
94}

96// Create a Quantize Op which has the same
97// dtype, shape, and qparam with node
98luci::CircleQuantize *create_quantize_op(luci::CircleNode *node)
99{
100 auto quantize = node->graph()->nodes()->create<CircleQuantize>();
101 quantize->name(node->name() + "_Quantize");
102 quantize->dtype(node->dtype());
103 quantize->rank(node->rank());
104 for (uint32_t i = 0; i < node->rank(); i++)
105 quantize->dim(i).set(node->dim(i).value());
106
107 quantize->shape_status(luci::ShapeStatus::VALID);
108
109 assert(node->quantparam()); // FIX_CALLER_UNLESS
110 copy_quantparam(node, quantize);
111
112 luci::add_origin(quantize, luci::get_origin(node));
113
114 return quantize;
115}

122// Add Quantize Op before Transpose
123//
124// Before
125//
126// [Node] s16
127// |
128// [Transpose] u8
129//
130// After
131//
132// [Node] s16
133// |
134// [Quantize] s16->u8
135// |
136// [Transpose] u8
137void InsertQuantizeOpOnDTypeMismatch::visit(luci::CircleTranspose *node)
138{
139 auto input = loco::must_cast<luci::CircleNode *>(node->a());
140
141 // Input dtype == Output dtype. No problem
142 if (input->dtype() == node->dtype())
143 return;
144
145 // Only cares quantized case
146 if (not is_quantized(input))
147 return;
148
149 if (not is_quantized(node))
150 return;
151
152 // Let's support limited case
153 // TODO Extend this to another dtype
154 if (input->dtype() != loco::DataType::S16)
155 return;
156
157 if (node->dtype() != loco::DataType::U8)
158 return;
159
160 // Create Quantize Op
161 auto quant_op = create_quantize_op(node);
162 quant_op->input(input);
163
164 node->a(quant_op);
165}

167void InsertQuantizeOpOnDTypeMismatch::visit(luci::CircleFullyConnected *node)
168{
169 auto input = loco::must_cast<luci::CircleNode *>(node->input());
170
171 // Input dtype == Output dtype. No problem
172 if (input->dtype() == node->dtype())
173 return;
174
175 // Skip if node has bias
176 if (dynamic_cast<luci::CircleOutputExclude *>(node->bias()) == nullptr)
177 return;
178
179 if (node->fusedActivationFunction() != luci::FusedActFunc::NONE)
180 return;
181
182 // Only cares quantized case
183 if (not is_quantized(input))
184 return;
185
186 if (not is_quantized(node))
187 return;
188
189 // Let's support limited case
190 // TODO Extend this to another dtype
191 if (input->dtype() != loco::DataType::U8)
192 return;
193
194 if (node->dtype() != loco::DataType::S16)
195 return;
196
197 // Create Quantize Op
198 auto quant_op = create_quantize_op(node);
199
200 // Insert Quantize Op after node
201 loco::replace(node).with(quant_op);
202 quant_op->input(node);
203
204 // Update node's dtype and qparam from i16 to u8
205 // NOTE This would severely degrade accuracy. It is
206 // important to mitigate this accuracy drop in backend.
207 update_i16_to_u8(node);
208}
