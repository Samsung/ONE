68bool compatible(const coco::FeatureObject *lhs, const coco::FeatureObject *rhs)

99class FeatureGroup
100{
101public:
102 explicit FeatureGroup(coco::FeatureObject *feature) { insert(feature); }
103
104public:
105 uint32_t size(void) const { return _uses.size() + (_def ? 1 : 0); }
106
107public:
108 void insert(coco::FeatureObject *feature)
109 {
110 if (feature->def() != nullptr)
111 {
112 assert(_def == nullptr);
113 _def = feature;
114 }
115 else
116 {
117 _uses.insert(feature);
118 }
119 }
120
121public:
122 coco::FeatureObject *parent(void) const
123 {
124 if (_def)
125 {
126 return _def;
127 }
128
129 assert(_uses.size() > 0);
130 return *(_uses.begin());
131 }
132
133public:
134 std::set<coco::FeatureObject *> children(void) const
135 {
136 auto res = _uses;
137 res.erase(parent());
138 return res;
139 }
140
141private:
142 coco::FeatureObject *_def = nullptr;
143 std::set<coco::FeatureObject *> _uses;
144};

151void unify_feature(enco::Code *code)
152{
153 auto m = code->module();
154
155 for (uint32_t n = 0; n < m->entity()->bag()->size(); ++n)
156 {
157 std::vector<std::unique_ptr<FeatureGroup>> groups;
158
159 auto assign_group = [&](coco::FeatureObject *feature) {
160 // Find a compatible FeatureGroup
161 FeatureGroup *group = nullptr;
162
163 for (const auto &g : groups)
164 {
165 FeatureGroup *candidate = g.get();
166
167 if (!compatible(candidate->parent(), feature))
168 {
169 continue;
170 }
171
172 group = candidate;
173 break;
174 }
175
176 if (group == nullptr)
177 {
178 // Insert FeatureObject into a new FeatureGroup
179 groups.emplace_back(make_unique<FeatureGroup>(feature));
180 }
181 else
182 {
183 // Insert FeatureObject into the compatible FeatureGroup
184 group->insert(feature);
185 }
186 };
187
188 auto bag = m->entity()->bag()->at(n);
189
190 for (auto o : coco::dependent_objects(bag))
191 {
192 if (auto feature = o->asFeature())
193 {
194 if (candidate(feature))
195 {
196 assign_group(feature);
197 }
198 }
199 }
200
201 for (const auto &g : groups)
202 {
203 auto group = g.get();
204 for (const auto child : group->children())
205 {
206 subst(child, group->parent());
207 assert(child->def() == nullptr);
208 assert(child->uses()->size() == 0);
209 m->entity()->object()->destroy(child);
210 }
211 }
212 }
213}
