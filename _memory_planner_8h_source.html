93template <typename Index> class FirstFitPlanner : public IMemoryPlanner<Index>
94{
95public:
101 void claim(const Index &, size_t) override;
106 void release(const Index &) override;
111 uint32_t capacity() override { return _capacity; }
116 std::unordered_map<Index, Block> &memory_plans() override { return _mem_plans; }
117
118private:
119 uint32_t _capacity = 0;
120 std::unordered_map<Index, Block> _mem_plans;
121 // Use std::map because claim() assumes that _claim_table is sorted by uint32_t(base_offset)
122 std::map<uint32_t, Index> _claim_table;
123};
124
125// There are some assumptions for claiming memory(== making a reservation for memory).
126// 1. About _claim_table(std::map).
127// - The table's data structure is std::map so that it always sorts
128// value(Index) by key(base_offset).
129// - This claim() inserts key/value into _claim_table and the release() removes the key/value from
130// _claim_table.
131// - _claim_table shows the memory status at a certain point in time. Therefore,
132// - If _claim_table has an offset and a certain size at a certain point in time,
133// it means the place at the offset has been already claimed(== can't claim now. need to find
134// someplace new).
135// - If _claim_table doesn't have any element for an offset and a certain size at a certain
136// point in time, it means the place at the offset can be claimed.
137// 2. In the loop for _claim_table, we can assume the current claim_base_offset value is bigger than
138// the previous claim_base_offset.
139template <typename Index> inline void FirstFitPlanner<Index>::claim(const Index &ind, size_t size)
140{
141 // Find the right position for claiming
142 uint32_t next_offset = 0;
143 for (const auto &mem_claim : _claim_table)
144 {
145 auto claimed_base_offset = mem_claim.first;
146 auto claimed_size = _mem_plans[mem_claim.second].size;
147 if (next_offset + size <= claimed_base_offset)
148 {
149 break;
150 }
151 else
152 {
153 next_offset = claimed_base_offset + claimed_size;
154 }
155 }
156
157 // Now next_offset is set to the proper offset
158 // std::map's operator[] requires default constructor of value
159 // _claim_table[next_offset] = ind;
160 _claim_table.emplace(std::make_pair(next_offset, ind));
161 _mem_plans[ind] = {next_offset, size};
162
163 VERBOSE(FF_PLANNER) << "claim(" << ind << "): [+" << next_offset << ", " << size << "sz]"
164 << std::endl;
165
166 if (_capacity < next_offset + size)
167 {
168 _capacity = next_offset + size;
169 }
170}
171
172template <typename Index> inline void FirstFitPlanner<Index>::release(const Index &ind)
173{
174 for (auto it = _claim_table.cbegin(); it != _claim_table.cend(); ++it)
175 {
176 if (it->second == ind)
177 {
178 uint32_t offset = it->first;
179 uint32_t size = _mem_plans[ind].size;
180
181 _claim_table.erase(it);
182
183 VERBOSE(FF_PLANNER) << "release(" << ind << "): [+" << offset << ", " << size << "sz]"
184 << std::endl;
185 return;
186 }
187 }
188 assert(!"Cannot release for given index. It has been not claimed or released already.");
189}

195{
196public:

246template <typename Index> inline void WICPlanner<Index>::claim(const Index &ind, size_t size)
247{
248 _indices.emplace(size, ind);
249 _interference_graph[ind].insert(_interference_graph[ind].end(), _live_indices.cbegin(),
250 _live_indices.cend());
251 for (const auto &live_operand : _live_indices)
252 {
253 _interference_graph[live_operand].emplace_back(ind);
254 }
255 _live_indices.emplace(ind);
256
257 VERBOSE(WIC_PLANNER) << "claim(" << ind << "): [" << size << "sz]" << std::endl;
258}

260template <typename Index> inline void WICPlanner<Index>::release(const Index &ind)
261{
262 _live_indices.erase(ind);
263 VERBOSE(WIC_PLANNER) << "release(" << ind << ")" << std::endl;
264}

266/*
267 * Build memory plans using liveness and size of operands
268 * 1. Build inference graph at claim
269 * - Two operands interfere if they have overlapped live range
270 * 2. Sort operands in descending order of size
271 * - Use std::multimap to sort operands
272 * 3. Allocate memory block for sorted operands
273 * - Find free memory block which does not overlap with interfered operands
274 */
275template <typename Index> inline void WICPlanner<Index>::buildMemoryPlans()
276{
277 for (const auto &operand : _indices)
278 {
279 uint32_t size = operand.first;
280 const Index &ind = operand.second;
281 VERBOSE(WIC_PLANNER) << "build_plan(" << ind << "): [" << size << "sz]" << std::endl;
282
283 uint32_t next_offset = 0;
284 if (_interference_graph.count(ind))
285 {
286 // Find interfered memory plans and sort them by offset
287 std::multimap<uint32_t, uint32_t> interfered_plans;
288 for (const auto &interference : _interference_graph[ind])
289 {
290 if (_mem_plans.count(interference))
291 interfered_plans.emplace(_mem_plans[interference].offset, _mem_plans[interference].size);
292 }
293
294 // Find free memory block in first-fit manner
295 for (const auto &interfered_plan : interfered_plans)
296 {
297 auto claimed_base_offset = interfered_plan.first;
298 auto claimed_size = interfered_plan.second;
299 VERBOSE(WIC_PLANNER) << "interfere : [+" << claimed_base_offset << ", " << claimed_size
300 << "sz]" << std::endl;
301 if (next_offset + size <= claimed_base_offset)
302 {
303 break;
304 }
305 else if (next_offset < claimed_base_offset + claimed_size)
306 {
307 next_offset = claimed_base_offset + claimed_size;
308 }
309 }
310 }
311 else
312 {
313 VERBOSE(WIC_PLANNER) << "No interference" << std::endl;
314 }
315
316 _mem_plans[ind] = {next_offset, size};
317 VERBOSE(WIC_PLANNER) << "alloc(" << ind << "): [+" << next_offset << ", " << size << "sz]"
318 << std::endl;
319
320 if (_capacity < next_offset + size)
321 {
322 _capacity = next_offset + size;
323 }
324 }
325 _initialized = true;
326 _interference_graph.clear();
327 _indices.clear();
328}
329
330template <typename Index> inline std::unordered_map<Index, Block> &WICPlanner<Index>::memory_plans()
331{
332 if (!_initialized)
333 buildMemoryPlans();
334 return _mem_plans;
335}
