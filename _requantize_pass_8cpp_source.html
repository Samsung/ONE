35// Check if the node is the bias of Conv2D, DepthwiseConv2D, or FullyConnected layer
36bool is_bias(CircleConst *node)
37{
38 if (node == nullptr)
39 return false;
40
41 auto succs = loco::succs(node);
42 if (succs.size() != 1) // assume bias is used by only one node
43 return false;
44
45 for (auto out : succs)
46 {
47 auto conv = dynamic_cast<CircleConv2D *>(out);
48 if (conv != nullptr && conv->bias() == node)
49 return true;
50
51 auto dw_conv = dynamic_cast<CircleDepthwiseConv2D *>(out);
52 if (dw_conv != nullptr && dw_conv->bias() == node)
53 return true;
54
55 auto fc = dynamic_cast<CircleFullyConnected *>(out);
56 if (fc != nullptr && fc->bias() == node)
57 return true;
58
59 auto tconv = dynamic_cast<CircleTransposeConv *>(out);
60 if (tconv != nullptr && tconv->bias() == node)
61 return true;
62 }
63 return false;
64}

66void requant_nonconst_int8_to_uint8(CircleNode *circle_node)
67{
68 assert(circle_node->dtype() == loco::DataType::S8);
69
70 auto quantparam = circle_node->quantparam();
71 assert(quantparam != nullptr);
72 for (size_t i = 0; i < quantparam->zerop.size(); ++i)
73 {
74 quantparam->zerop[i] += 128;
75 }
76 circle_node->dtype(loco::DataType::U8);
77}

79// Requantize CircleConst from symmetric int8 to asymmetric uint8
80// Original values: -127 ~ 127
81// After requantization: 1 ~ 255 (zp <- zp + 128)
82void requant_const_int8_to_uint8(CircleConst *node)
83{
84 assert(node->dtype() == loco::DataType::S8);
85
86 uint32_t size = node->size<loco::DataType::S8>();
87 std::vector<int32_t> requantized_values(size);
88 for (uint32_t i = 0; i < size; ++i)
89 {
90 int32_t data = node->at<loco::DataType::S8>(i);
91 requantized_values[i] = data + 128;
92 }
93
94 node->dtype(loco::DataType::U8); // change the type of tensor
95 node->size<loco::DataType::U8>(size);
96 for (uint32_t i = 0; i < size; ++i)
97 {
98 assert(1 <= requantized_values[i] && requantized_values[i] <= 255);
99 node->at<loco::DataType::U8>(i) = requantized_values[i];
100 }
101
102 auto quantparam = node->quantparam();
103 assert(quantparam != nullptr);
104 for (size_t i = 0; i < quantparam->zerop.size(); ++i)
105 {
106 quantparam->zerop[i] += 128;
107 }
108}

113struct RequantizeNonConst final : public luci::CircleNodeMutableVisitor<bool>
114{
115 RequantizeNonConst(loco::DataType input, loco::DataType output)
116 : _input_type(input), _output_type(output)
117 {
118 }
119
120 loco::DataType _input_type;
121 loco::DataType _output_type;
122
123 // Requantize input tensors of each node
124 bool visit(luci::CircleNode *node)
125 {
126 LOGGER(l);
127 INFO(l) << "RequantizeNonConst visit node: " << node->name() << std::endl;
128 auto arity = node->arity();
129 for (uint32_t i = 0; i < arity; i++)
130 {
131 auto input_node = node->arg(i);
132 auto circle_node = loco::must_cast<luci::CircleNode *>(input_node);
133
134 // Check if this was quantized (only quantized tensors are requantized)
135 if (circle_node->quantparam() == nullptr)
136 continue;
137
138 // Check if this is already requantized
139 if (circle_node->dtype() == _output_type)
140 continue;
141
142 // Check if this is not const (only non-const is requantized in this function)
143 auto circle_const = dynamic_cast<CircleConst *>(circle_node);
144 if (circle_const != nullptr)
145 continue;
146
147 if (_input_type == loco::DataType::S8 && _output_type == loco::DataType::U8)
148 requant_nonconst_int8_to_uint8(circle_node);
149 }
150 return false;
151 }
152};

179 if (circle_node->quantparam() == nullptr)
180 continue;
181
182 // Check if this is already requantized
183 if (circle_node->dtype() == _output_type)
184 continue;
185
186 // Check if this is const (only const is requantized in this function)
187 auto circle_const = dynamic_cast<CircleConst *>(circle_node);
188 if (circle_const == nullptr)

198 }

205bool RequantizePass::run(loco::Graph *g)
206{
207 LOGGER(l);
208 INFO(l) << "RequantizePass Start" << std::endl;
209
210 // Requantize non-const (activations)
211 for (auto node : loco::active_nodes(loco::output_nodes(g)))
212 {
213 RequantizeNonConst rqnc(_input_dtype, _output_dtype);
214 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
215 circle_node->accept(&rqnc);
216 }
217
218 // Requantize const (including weights, constants)
219 for (auto node : loco::active_nodes(loco::output_nodes(g)))
220 {
221 RequantizeConst rqc(_input_dtype, _output_dtype);
222 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
223 circle_node->accept(&rqc);
224 }
225
226 // Update output dtype
227 auto graph_outputs = g->outputs();
228 for (auto node : loco::output_nodes(g))
229 {
230 auto circle_node = loco::must_cast<luci::CircleOutput *>(node);
231 if (static_cast<luci::CircleNode *>(circle_node->from())->dtype() == _output_dtype)
232 {
233 circle_node->dtype(_output_dtype);
234 auto graph_output = graph_outputs->at(circle_node->index());
235 graph_output->dtype(_output_dtype);
236 }
237 }
238
239 INFO(l) << "RequantizePass End" << std::endl;
240 return false; // one time run
241}
