66void Subst<SubstQualifier::Default>::with(Node *into) const
67{
68 if (_from == into)
69 {
70 return;
71 }
72
73 auto *uses = &(_from->_uses);
74
75 while (!uses->empty())
76 {
77 auto use = *(uses->begin());
