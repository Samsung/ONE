31void OddOutputPass::run()
32{
33 auto &outputs = _graph.getOutputs();
34
35 VERBOSE(OddOutputPass) << "Case 1 : An operand which is a model output and a model input"
36 << std::endl;
37 for (auto &ind : outputs)
38 {
39 if (_graph.getInputs().contains(ind))
40 {
41 auto permute_output_ind = insertPermute(ind);
42 // Update the output to be newly added operand
43 _graph.getOutputs().replace(ind, permute_output_ind);
44 }
45 }
46
47 VERBOSE(OddOutputPass) << "Case 2 : Two or more duplicated outputs" << std::endl;
48 std::unordered_set<ir::OperandIndex> occurence;
49 for (auto &ind : outputs)
50 {
51 auto &obj = _graph.operands().at(ind);
52 if (occurence.count(ind) == 0)
53 {
54 occurence.insert(ind);
55 continue;
56 }
57
58 // Panic when it is const, it must have been handled earlier in another pass
59 UNUSED_RELEASE(obj);
60 assert(!obj.isConstant());
61
62 auto permute_output_ind = insertPermute(ind);
63 ind = permute_output_ind; // Replace output index to fix output duplication
64 }
65}

67ir::OperandIndex OddOutputPass::insertPermute(ir::OperandIndex ind)
68{
69 auto &obj = _graph.operands().at(ind);
70 auto output_ind = _graph.addOperand(obj.shape(), obj.typeInfo());
71 auto &output_obj = _graph.operands().at(output_ind);
72
73 using ir::operation::Permute;
74 auto permute_obj = std::make_unique<Permute>(ind, output_ind, Permute::Type::COPY);
75 auto permute_ind = _graph.operations().push(std::move(permute_obj));
76
77 output_obj.setDef(permute_ind);
78 obj.insertUse(permute_ind);
79
80 VERBOSE(OddOutputPass) << "Permute Op inserted for a constant output, node index : "
81 << permute_ind << std::endl;
82 VERBOSE(OddOutputPass) << " - Input (original) Operand : " << ind << std::endl;
83 VERBOSE(OddOutputPass) << " - Output(inserted) Operand : " << output_ind << std::endl;
84
85 return output_ind;
86}
