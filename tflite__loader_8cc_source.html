89 std::unique_ptr<ir::Graph> loadSubgraph(const onert_tflite::SubGraph *tflite_subg) override
90 {
91 auto subg = std::make_unique<ir::Graph>();
92 // Load tensors
93 _tensor_to_operand.resize(tflite_subg->tensors()->size());
94 for (flatbuffers::uoffset_t i = 0; i < tflite_subg->tensors()->size(); ++i)
95 {
96 _tensor_to_operand[i] = loadOperand(tflite_subg->tensors()->Get(i), *subg);
97 }
98 // Set inputs
99 for (const std::int32_t input_ind : *tflite_subg->inputs())
100 {
101 subg->addInput(tensorIdxToOperandIdx(input_ind),
102 _tensor_names.at(_tensor_to_operand[input_ind]));
103 }
104 // Set outputs
105 for (const std::int32_t output_ind : *tflite_subg->outputs())
106 {
107 subg->addOutput(tensorIdxToOperandIdx(output_ind),
108 _tensor_names.at(_tensor_to_operand[output_ind]));
109 }
110 // Create operations
111 for (const auto *op : *tflite_subg->operators())
112 {
113 loadOperation(op, *subg);
114 }
115
116 subg->verify();
117
118 return subg;
119 }

121 void loadOperation(const onert_tflite::Operator *op, ir::Graph &subg)
122 {
123 auto const builtin_op = getBuiltinOperator(op);
124
125 switch (builtin_op)
126 {
127 case onert_tflite::BuiltinOperator::BuiltinOperator_BATCH_MATMUL:
128 loadBatchMatMul(op, subg);
129 return;
130 default:
131 BaseLoader::loadOperation(op, subg);
132 return;
133 }
134 }

137void TFLiteLoader::loadBatchMatMul(const Operator *op, ir::Graph &subg)
138{
139 ir::OperandIndexSequence inputs;
140 ir::OperandIndexSequence outputs;
141
142 loadOperationIO(op, inputs, outputs);
143
144 ir::operation::BatchMatMul::Param param;
145 const auto *options = op->builtin_options_as_BatchMatMulOptions();
146
147 param.adj_x = options->adj_x();
148 param.adj_y = options->adj_y();
149
150 std::unique_ptr<ir::Operation> new_op(new ir::operation::BatchMatMul(inputs, outputs, param));
151 subg.addOperation(std::move(new_op));
152}
