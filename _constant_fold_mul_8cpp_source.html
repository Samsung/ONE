35bool constantfold_mul(moco::TFMul *node)
36{
37 auto x_const = moco::as<moco::TFConst>(node->x());
38 auto y_const = moco::as<moco::TFConst>(node->y());
39 if (x_const == nullptr || y_const == nullptr)
40 return false;
41
42 if (x_const->dtype() != y_const->dtype())
43 return false;
44 // TODO support other types
45 if (x_const->dtype() != loco::DataType::S32 && x_const->dtype() != loco::DataType::FLOAT32)
46 return false;
47
48 // NOTE we support limited shape of elementwise mul or multiply with a scalar.
49 // valid_shape_for_constfold_binary_op() explains limited shape.
50 auto x_shape = moco::tensor_shape(x_const);
51 auto y_shape = moco::tensor_shape(y_const);
52 if (!moco::valid_shape_for_constfold_binary_op(x_shape, y_shape))
53 return false;
54
55 loco::TensorShape output_shape;
56 if (y_shape.rank() == 0 || y_shape.rank() == 1)
57 output_shape = x_shape;
58 else
59 output_shape = y_shape;
60
61 auto graph = node->graph();
62 auto output_const = moco::new_const(graph, output_shape, x_const->dtype());
63 Func f;
64
65 if (x_const->dtype() == loco::DataType::S32)
66 {
67 moco::apply_binary<int32_t>(x_const, y_const, output_const, f);
68 }
69 else if (x_const->dtype() == loco::DataType::FLOAT32)
70 {
71 moco::apply_binary<float>(x_const, y_const, output_const, f);
72 }
73
74 // replace
75 loco::replace(node).with(output_const);
76
77 return true;
78}

101bool ConstantFoldMul::run(loco::Graph *graph)
102{

107 {
108 if (constantfold_mul(mul_node))
109 changed = true;
110 }
111 }
112
113 return changed;
114}
