26// **CAUTION** Child SHOULD inherit DLinkedList<Child, Parent>::Node
27template <typename Child, typename Parent> struct DLinkedList
28{
30 static void joined(Parent *, Child *);
32 static void leaving(Parent *, Child *);
33
34 class Head
35 {
36 public:
37 Head(Parent *parent) : _parent{parent}
38 {
39 _head = nullptr;
40 _tail = nullptr;
41 }
42
43 public:
44 Head(const Head &) = delete;
45 Head(Head &&) = delete;
46
47 public:
48 Child *head(void) const { return _head; }
49 Child *tail(void) const { return _tail; }
50
51 public:
52 bool empty(void) const
53 {
54 if (_head == nullptr)
55 {
56 assert(_head == _tail);
57 return true;
58 }
59
60 assert(_head != nullptr);
61 assert(_tail != nullptr);
62 return false;
63 }
64
65 public:
66 void enlist(Child *child)
67 {
68 assert((child->prev() == nullptr) || (child->prev()->parent() == _parent));
69 assert((child->next() == nullptr) || (child->next()->parent() == _parent));
70
71 if (empty())
72 {
73 _head = child;
74 _tail = child;
75 }
76 else
77 {
78 if (child->next() == _head)
79 {
80 // _child is a new head
81 assert(child->prev() == nullptr);
82 _head = child;
83 }
84
85 if (child->prev() == _tail)
86 {
87 // _child is a new tail
88 assert(child->next() == nullptr);
89 _tail = child;
90 }
91 }
92
93 // Update parent-child relation
94 child->parent(_parent);
95
96 // Notify Child-Joining event
97 joined(_parent, child);
98 }
99
100 public:
101 void delist(Child *child)
102 {
103 assert(child->parent() == _parent);
104 assert(!empty());
105
106 // Notify Child-Leaving event
107 leaving(_parent, child);
108
109 if (child == _head)
110 {
111 _head = child->next();
112 }
113
114 if (child == _tail)
115 {
116 _tail = child->prev();
117 }
118
119 // Update parent-child relation
120 child->parent(nullptr);
121 }
122
123 public:
124 void prepend(Child *child)
125 {
126 if (empty())
127 {
128 enlist(child);
129 }
130 else
131 {
132 child->insertBefore(_head);
133 }
134 }
135
136 public:
137 void append(Child *child)
138 {
139 if (empty())
140 {
141 enlist(child);
142 }
143 else
144 {
145 child->insertAfter(_tail);
146 }
147 }
148
149 private:
150 Parent *const _parent;
151
152 private:
153 Child *_head;
154 Child *_tail;
155 };
156
157 // NOTE Client SHOULD implement this static method
158 static Head *head(Parent *);

160 class Node
161 {
162 public:
163 friend class Head;
164
165 public:
166 Node()
167 {
168 static_assert(std::is_base_of<Node, Child>::value,
169 "Type `Child` must be subclass of `Node`.");
170
171 _prev = nullptr;
172 _next = nullptr;
173 }
174
175 public:
176 virtual ~Node()
177 {
178 // Each Child should unlink itself on destruction
179 //
180 // NOTE detach invokes "leaving" hook which may access the internal of each Child,
181 // so it is not safe to invoke detach here
182 assert(parent() == nullptr);
183 }
184
185 public:
186 Parent *parent(void) const { return _parent; }
187
188 private:
189 Child *curr(void) { return reinterpret_cast<Child *>(this); }
190 const Child *curr(void) const { return reinterpret_cast<const Child *>(this); }
191
192 public:
193 Child *prev(void) const { return _prev; }
194 Child *next(void) const { return _next; }
195
196 public:
197 void insertBefore(Node *next)
198 {
199 assert(next != nullptr);
200 assert(next->parent() != nullptr);
201 assert(head(next->parent()) != nullptr);
202
203 assert(_prev == nullptr);
204 assert(_next == nullptr);
205
206 // Update the link of the current node
207 _prev = next->prev();
208 _next = next->curr();
209
210 if (auto prev = next->prev())
211 {
212 prev->_next = curr();
213 }
214 next->_prev = curr();
215
216 // Update parent-child relation
217 assert(parent() == nullptr);
218 head(next->parent())->enlist(curr());
219 assert(parent() == next->parent());
220 }
221
222 public:
223 void insertAfter(Node *prev)
224 {
225 assert(prev != nullptr);
226 assert(prev->parent() != nullptr);
227 assert(head(prev->parent()) != nullptr);
228
229 assert(_prev == nullptr);
230 assert(_next == nullptr);
231
232 // Update the link of the current node
233 _prev = prev->curr();
234 _next = prev->next();
235
236 // Update the link of the sibling nodes
237 if (auto next = prev->next())
238 {
239 next->_prev = curr();
240 }
241 prev->_next = curr();
242
243 // Update parent-child relation
244 assert(parent() == nullptr);
245 head(prev->parent())->enlist(curr());
246 assert(parent() == prev->parent());
247 };
248
249 public:
250 void detach(void)
251 {
252 // Update parent-child relation
253 assert(parent() != nullptr);
254 assert(head(parent()) != nullptr);
255 head(parent())->delist(curr());
256 assert(parent() == nullptr);
257
258 // Update the link of sibling nodes
259 if (prev())
260 {
261 prev()->_next = next();
262 }
263
264 if (next())
265 {
266 next()->_prev = prev();
267 }
268
269 // Update the link of the current node
270 _prev = nullptr;
271 _next = nullptr;
272 }
273
274 private:
275 // WARN Do NOT invoke this method outside Head::enlist
276 void parent(Parent *p) { _parent = p; }
277
278 private:
279 // WARN Do NOT modify this field inside Node.
280 Parent *_parent = nullptr;
281 Child *_prev;
282 Child *_next;
283 };
284};
