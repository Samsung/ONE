27// DO NOT USE THIS STRUCT FOR NEW FUNCTIONALITY BEYOND IMPLEMENTING ELEMENT-WISE

30// NdArrayDesc<N> describes the shape and memory layout of an N-dimensional
31// rectangular array of numbers.
32//
33// NdArrayDesc<N> is basically identical to Dims<N> defined in types.h.
34// However, as Dims<N> is to be deprecated, this class exists as an adaptor
35// to enable simple unoptimized implementations of element-wise broadcasting
36// operations.
37template <int N> struct NdArrayDesc
38{
39 // The "extent" of each dimension. Indices along dimension d must be in the
40 // half-open interval [0, extents[d]).
41 int extents[N];
42
43 // The number of *elements* (not bytes) between consecutive indices of each
44 // dimension.
45 int strides[N];
46};

54inline int SubscriptToIndex(const NdArrayDesc<4> &desc, int i0, int i1, int i2, int i3)
55{
56 DCHECK(i0 >= 0 && i0 < desc.extents[0]);
57 DCHECK(i1 >= 0 && i1 < desc.extents[1]);
58 DCHECK(i2 >= 0 && i2 < desc.extents[2]);
59 DCHECK(i3 >= 0 && i3 < desc.extents[3]);
60 return i0 * desc.strides[0] + i1 * desc.strides[1] + i2 * desc.strides[2] + i3 * desc.strides[3];
61}

63// From optimized_ops.h in TensorFlow Lite
64//
65// Given the dimensions of the operands for an element-wise binary broadcast,
66// adjusts them so that they can be directly iterated over with simple loops.
67// Returns the adjusted dims as instances of NdArrayDesc in 'desc0_out' and
68// 'desc1_out'. 'desc0_out' and 'desc1_out' cannot be nullptr.
69//
70// This function assumes that the two input shapes are compatible up to
71// broadcasting and the shorter one has already been prepended with 1s to be the
72// same length. E.g., if shape0 is (1, 16, 16, 64) and shape1 is (1, 64),
73// shape1 must already have been prepended to be (1, 1, 1, 64). Recall that
74// Dims<N> refer to shapes in reverse order. In this case, input0_dims will be
75// (64, 16, 16, 1) and input1_dims will be (64, 1, 1, 1).
76//
77// When two shapes are compatible up to broadcasting, for each dimension d,
78// the input extents are either equal, or one of them is 1.
79//
80// This function performs the following for each dimension d:
81// - If the extents are equal, then do nothing since the loop that walks over
82// both of the input arrays is correct.
83// - Otherwise, one (and only one) of the extents must be 1. Say extent0 is 1
84// and extent1 is e1. Then set extent0 to e1 and stride0 *to 0*. This allows
85// array0 to be referenced *at any index* in dimension d and still access the
86// same slice.
87template <int N>
88inline void
89NdArrayDescsForElementwiseBroadcast(const Dims<N> &input0_dims, const Dims<N> &input1_dims,
90 NdArrayDesc<N> *desc0_out, NdArrayDesc<N> *desc1_out)
91{
92 DCHECK(desc0_out != nullptr);
93 DCHECK(desc1_out != nullptr);
94
95 // Copy dims to desc.
96 for (int i = 0; i < N; ++i)
97 {
98 desc0_out->extents[i] = input0_dims.sizes[i];
99 desc0_out->strides[i] = input0_dims.strides[i];
100 desc1_out->extents[i] = input1_dims.sizes[i];
101 desc1_out->strides[i] = input1_dims.strides[i];
102 }
103
104 // Walk over each dimension. If the extents are equal do nothing.
105 // Otherwise, set the desc with extent 1 to have extent equal to the other and
106 // stride 0.
107 for (int i = 0; i < N; ++i)
108 {
109 const int extent0 = ArraySize(input0_dims, i);
110 const int extent1 = ArraySize(input1_dims, i);
111 if (extent0 != extent1)
112 {
113 if (extent0 == 1)
114 {
115 desc0_out->strides[i] = 0;
116 desc0_out->extents[i] = extent1;
117 }
118 else
119 {
120 DCHECK_EQ(extent1, 1);
121 desc1_out->strides[i] = 0;
122 desc1_out->extents[i] = extent0;
123 }
124 }
125 }
126}
