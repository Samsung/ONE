94template <loco::DataType DT> void fill_conv_weights(luci::CircleConst *weights)
95{
96 assert(weights->rank() == 4);
97
98 auto const kn = weights->dim(0).value();
99 auto const kh = weights->dim(1).value();
100 auto const kw = weights->dim(2).value();
101
102 auto elements_size = kn * kh * kw * 1;
103 weights->size<DT>(elements_size);
104
105 for (uint32_t b = 0; b < kn; ++b)
106 {
107 for (uint32_t y = 0; y < kh; ++y)
108 {
109 for (uint32_t x = 0; x < kw; ++x)
110 {
111 auto const idx = (b * kh + y) * kw + x;
112 weights->at<DT>(idx) = (y * kw + x == b) ? 1 : 0;
113 }
114 }
115 }
116}

229int32_t compute_full_padding(int32_t input_size, int32_t output_size, int32_t stride,
230 int32_t filter_size)
231{
232 int32_t effective_input = (output_size - 1) * stride + filter_size;
233 int32_t full = effective_input - input_size;
234 // some extreme cases when part of input was not used in computations
235 if (full < 0)
236 full = 0;
237 return full;
238}

240template <loco::DataType DT>
241void fill_coords_addition(luci::Padding padding, const luci::Stride &stride,
242 const luci::Filter &filter, uint32_t input_height, uint32_t input_width,
243 uint32_t depth, luci::CircleConst *cords)
244{
245 assert(cords->rank() == 4);
246
247 auto const output_height = static_cast<int32_t>(cords->dim(1).value());
248 auto const output_width = static_cast<int32_t>(cords->dim(2).value());
249 {
250 auto const element_counts = 1 * output_height * output_width * 1;
251 cords->size<DT>(element_counts);
252 }
253
254 assert(padding != luci::Padding::UNDEFINED);
255
256 // For VALID padding:
257 int32_t start_y = 0;
258 int32_t start_x = 0;
259
260 // For SAME padding:
261 if (padding == luci::Padding::SAME)
262 {
263 start_y = -compute_full_padding(input_height, output_height, stride.h(), filter.h()) / 2;
264 start_x = -compute_full_padding(input_width, output_width, stride.w(), filter.w()) / 2;
265 }
266
267 auto const step_y = static_cast<int32_t>(stride.h());
268 auto const step_x = static_cast<int32_t>(stride.w());
269
270 for (int32_t y_o = 0, y_i = start_y; y_o < output_height; ++y_o, y_i += step_y)
271 {
272 for (int32_t x_o = 0, x_i = start_x; x_o < output_width; ++x_o, x_i += step_x)
273 {
274 auto const output_idx = y_o * output_width + x_o;
275 auto const input_idx = y_i * static_cast<int32_t>(input_width) + x_i;
276
277 // Add small adjustment value to fix cast operation result that follows "coord addition"
278 // in generated subgraph.
279 //
280 // Cast operation discards fractional part of value, so 1.9996 will be transformed to 1
281 // This is not a problem when working with float32, because it represents integers precisely,
282 // but leads to wrong results, when working with quantized numbers.
283 //
284 // This value is larger than quantization error,
285 // and small enough to not affect following computations
286 // (in particular multiplication with depth)
287 const float round_adjustment = 1.0f / (depth + 1);
288
289 cords->at<DT>(output_idx) = input_idx + round_adjustment;
290 }
291 }
292}

330luci::CircleNode *max_pool_branch(luci::Padding padding, const luci::Stride &stride,
331 const luci::Filter filter, luci::CircleCustom *cop)
332{
333 auto graph = cop->graph();
334 auto input = cop->inputs(0);
335
336 auto origin = luci::get_origin(cop);
337 auto name = cop->name() + "/Argmax";
338
339 // Create MaxPool
340 auto maxpool = none_act_func(graph->nodes()->create<luci::CircleMaxPool2D>());
341 {
342 init_name_and_origin(maxpool, name + "/MaxPool2D", origin);
343
344 set_stride(maxpool, stride);

348 maxpool->value(input);
349 }

354luci::CircleNode *window_flattened_coord(const std::string &name, luci::Padding padding,
355 const luci::Stride &stride, const luci::Filter filter,
356 int32_t input_height, int32_t input_width,
357 uint32_t output_height, uint32_t output_width,
358 luci::CircleNode *input)
359{
360 auto const graph = input->graph();
361 auto const origin = luci::get_origin(input);
362
363 auto const depth_dimension = 3;
364
365 // Create pad in case of SAME padding
366 luci::CircleNode *conv_input = input;
367 if (padding == luci::Padding::SAME)
368 {
369 // Create redundant add to combine two nodes with special quantization restrictions:
370 // PadV2 and Split in this case
371 // TODO Introduce special requantize node and fix quantizer?
372 auto requantize = none_act_func(graph->nodes()->create<luci::CircleMul>());
373 init_name_and_origin(requantize, name + "/Requantize", origin);
374 auto zero_const = create_scalar<loco::DataType::FLOAT32>(graph, 1.0f);
375 init_name_and_origin(zero_const, name + "Requantize_const", origin);
376
377 requantize->x(input);
378 requantize->y(zero_const);
379
380 auto pad = graph->nodes()->create<luci::CirclePadV2>();
381 init_name_and_origin(pad, name + "/Pad", origin);
382
383 pad->input(requantize);
384
385 int32_t full_w_pad = compute_full_padding(input_width, output_width, stride.w(), filter.w());
386 int32_t full_h_pad = compute_full_padding(input_height, output_height, stride.h(), filter.h());
387 int32_t left_pad = full_w_pad / 2;
388 int32_t right_pad = full_w_pad - left_pad;
389 int32_t top_pad = full_h_pad / 2;
390 int32_t bottom_pad = full_h_pad - top_pad;
391 auto padding_const = create_padding_const(graph, left_pad, right_pad, top_pad, bottom_pad);
392 init_name_and_origin(padding_const, name + "/Pad_shape", origin);
393 pad->paddings(padding_const);
394
395 auto padding_value =
396 create_scalar<loco::DataType::FLOAT32, float>(graph, std::numeric_limits<float>::lowest());
397 init_name_and_origin(padding_value, name + "/Pad_value", origin);
398 pad->constant_values(padding_value);
399
400 conv_input = pad;
401 }
402 // Create Conv2D to move spatial dimensions to depth
403 auto conv = none_act_func(graph->nodes()->create<luci::CircleConv2D>());
404 {
405 init_name_and_origin(conv, name + "/Conv2D", origin);
406
407 // Padding, Stride and kernel size equal to MaxPool's
408 set_stride(conv, stride);
409 conv->padding(luci::Padding::VALID);
410
411 // depth of kernel is equal to square size
412 auto const kh = filter.h();
413 auto const kw = filter.w();
414 auto const kd = kh * kw;
415
416 // use zero bias
417 auto bias = create_zero_bias(graph, kd);
418 init_name_and_origin(bias, conv->name() + "/Bias", origin);
419
420 // create filter
421 // TODO make shared
422 auto weights = create_conv_filter(graph, kh, kw, kd);
423 init_name_and_origin(weights, conv->name() + "/Weights", origin);
424
425 conv->bias(bias);
426 conv->filter(weights);
427 conv->input(conv_input);
428 }
429
430 // Create ArgMax
431 auto argmax = graph->nodes()->create<luci::CircleArgMax>();
432 {
433 init_name_and_origin(argmax, name + "/ArgMax", origin);
434
435 argmax->output_type(loco::DataType::S32);
436
437 // Create argmax_dim
438 auto argmax_dim = create_scalar<loco::DataType::S32>(graph, depth_dimension);
439 init_name_and_origin(argmax_dim, argmax->name() + "/Dimension", origin);
440
441 argmax->dimension(argmax_dim);
442 argmax->input(conv);
443 }
444
445 // Create Reshape to 4-rank back, because argmax decrease rank of tensor by 1
446 auto reshape = graph->nodes()->create<luci::CircleReshape>();
447 {
448 init_name_and_origin(reshape, name + "/Reshape", origin);
449
450 auto shape = create_shape_tensor(graph, {1, output_height, output_width, 1});
451 init_name_and_origin(shape, reshape->name() + "/Shape", origin);
452
453 reshape->tensor(argmax);
454 reshape->shape(shape);
455 }
456
457 // Create Cast to use float32 instead int32
458 auto argmax_cast = create_cast(reshape, loco::DataType::S32, loco::DataType::FLOAT32);
459 init_name_and_origin(argmax_cast, argmax->name() + "/Cast", origin);
460
461 return argmax_cast;
462}

464// Creates "identity operation" after Floor
465// to force circle-quantizer requantize output tensor with scale << 1.
466//
467// Dealing with values of extremely different scales
468// in following binary operations hurts backend precision.
469luci::CircleNode *create_post_floor_requantize_node(luci::CircleFloor *floor)
470{
471 auto graph = floor->graph();
472 auto const origin = luci::get_origin(floor);
473 auto name = floor->name();
474
475 // Use DepthwiseConv2D with identity filter as an "identity operation".
476 //
477 // This operation do not change values, but forces circle-quantizer to use
478 // statistics to compute qparam scale instead of fixed scale == 1.0 after floor.
479 // DepthwiseConv2d is not eliminated by optimizations,
480 // so desired scale will reach backend.
481 auto requantizer = none_act_func(graph->nodes()->create<luci::CircleDepthwiseConv2D>());
482 init_name_and_origin(requantizer, name + "/Requantizer", origin);
483
484 requantizer->input(floor);
485
486 auto requantizer_filter = create_scalar<loco::DataType::FLOAT32>(graph, 1.0f);
487 init_name_and_origin(requantizer_filter, name + "/Requantizer/filter", origin);
488 requantizer_filter->rank(4);
489 for (uint32_t i = 0; i < 4; ++i)
490 {
491 requantizer_filter->dim(i) = 1;
492 }
493 requantizer->filter(requantizer_filter);
494
495 auto requantizer_bias = create_zero_bias(graph, 1);
496 init_name_and_origin(requantizer_bias, name + "/Requantizer/bias", origin);
497 requantizer->bias(requantizer_bias);
498
499 requantizer->padding(luci::Padding::VALID);
500 requantizer->stride()->w(1);
501 requantizer->stride()->h(1);
502 requantizer->depthMultiplier(1);
503 requantizer->dilation()->w(1);
504 requantizer->dilation()->h(1);
505
506 return requantizer;
507}

509luci::CircleNode *window_y_coord(const std::string &name, const luci::Filter &filter,
510 luci::CircleNode *flattened)
511{
512 auto const graph = flattened->graph();
513 auto const origin = luci::get_origin(flattened);
514
515 auto div = none_act_func(graph->nodes()->create<luci::CircleMul>());
516 {
517 init_name_and_origin(div, name + "/Div", origin);
518
519 // Adjustment_coeff is needed to fix computation of quantized tensors
520 //
521 // For example float32 value 2.0 could be quantized to 1.996
522 // after floor it will be transformed to 1.0, but desired answer is still something close to 2.0
523 //
524 // rounding_adjustment is chosen so it is small enough to not affect float32 computations,
525 // but "Div" change is larger then potential quantization error.
526 //
527 // This computation exploits the fact that div is an x coord in maxpool window,
528 // and lies in defined range [0, filter.h())
529 const float rounding_adjustment = 1.0f / (filter.w() * filter.h());
530 const float divider_value = filter.w() - rounding_adjustment;
531 auto divider = create_scalar<loco::DataType::FLOAT32>(graph, 1.0f / divider_value);
532 init_name_and_origin(divider, div->name() + "/Divider", origin);
533
534 div->x(flattened);
535 div->y(divider);
536 }
537
538 auto floor = graph->nodes()->create<luci::CircleFloor>();
539 {
540 init_name_and_origin(floor, name + "/Floor", origin);
541 floor->x(div);
542 }
543
544 auto requantizer = create_post_floor_requantize_node(floor);
545
546 return requantizer;
547}

549luci::CircleNode *window_x_coord(const std::string &name, float filter_width,
550 luci::CircleNode *flattened, luci::CircleNode *y_coord)

554

558

560 {
561 init_name_and_origin(neg, mod->name() + "/Neg", origin);
562
563 auto mul = none_act_func(graph->nodes()->create<luci::CircleMul>());
564 {
565 init_name_and_origin(mul, neg->name() + "/Neg", origin);

572 }
573
574 neg->x(mul);
575 }
576
577 mod->x(flattened);
578 mod->y(neg);
579 }
580
581 return mod;
582}

660luci::CircleNode *argmax_branch(luci::Padding padding, const luci::Stride &stride,
661 const luci::Filter filter, luci::CircleCustom *cop)
662{
663 auto graph = cop->graph();
664 auto input = loco::must_cast<luci::CircleNode *>(cop->inputs(0));
665 auto output = get_custom_output(cop, 1);
666
667 auto const depth_dimension = 3;
668 auto const input_depth = input->dim(depth_dimension).value();
669 auto const input_height = input->dim(1).value();
670 auto const input_width = input->dim(2).value();
671
672 assert(output->rank() == 4);
673 auto const output_height = output->dim(1).value();
674 auto const output_width = output->dim(2).value();
675
676 auto origin = luci::get_origin(cop);
677 auto name = cop->name() + "/Argmax";
678
679 // Create Split
680 auto split = graph->nodes()->create<luci::CircleSplit>();
681 {
682 init_name_and_origin(split, name + "/Split", origin);
683
684 // Create split_dim
685 auto split_dim = create_scalar<loco::DataType::S32>(graph, depth_dimension);
686 init_name_and_origin(split_dim, split->name() + "/Dim", origin);
687
688 split->num_split(int32_t(input_depth));
689
690 split->split_dim(split_dim);
691 split->input(input);
692 }
693
699 std::vector<luci::CircleNode *> branch_outputs(input_depth);
700
701 for (uint32_t br_n = 0; br_n < input_depth; ++br_n)
702 {
703 auto const branch_name = name + "/depth_" + std::to_string(br_n);
704
705 // Create CircleSplitOut
706 auto split_out = graph->nodes()->create<luci::CircleSplitOut>();
707 init_name_and_origin(split_out, branch_name + "/SplitOut", origin);
708 split_out->index(int32_t(br_n));
709 split_out->input(split);
710
711 // Define idx of max element in Window:
712 auto window_coords =
713 window_flattened_coord(branch_name + "/WindowFlat", padding, stride, filter, input_height,
714 input_width, output_height, output_width, split_out);
715
716 auto const window_y = window_y_coord(branch_name + "/WindowY", filter, window_coords);
717 auto const window_x =
718 window_x_coord(branch_name + "/WindowX", filter.w(), window_coords, window_y);
719
720 // Define idx of max element in Plane
721 // This tensor contains coords of left top corners for each window from input tensor
722 auto corners = create_coords_addition(graph, padding, stride, filter, input_height, input_width,
723 input_depth, output_height, output_width);
724 init_name_and_origin(corners, branch_name + "/Const", origin);
725
726 auto plane_coord =
727 plane_flattened_coord(branch_name + "/PlaneFlat", input_width, window_y, window_x, corners);
728
729 // Define volume coords as final value
730 branch_outputs[br_n] =
731 volume_flattened_coords(branch_name + "/VolumeFlat", br_n, input_depth, plane_coord);
732 }
733
734 // Create Concatenation
735 auto concat = none_act_func(graph->nodes()->create<luci::CircleConcatenation>(input_depth));
736 {
737 init_name_and_origin(concat, name + "/Concatenation", origin);
738 concat->axis(depth_dimension);
739
740 for (uint32_t i = 0; i < input_depth; ++i)
741 {
742 concat->values(i, branch_outputs[i]);
743 }
744 }
745
746 // Output of argmax_with_maxpool should be S64 or S32
747 loco::DataType output_dtype = get_custom_output(cop, 1)->dtype();
748 auto output_cast = create_cast(concat, loco::DataType::FLOAT32, output_dtype);
749 init_name_and_origin(output_cast, name + "/Cast", origin);
750
751 return output_cast;
752}

754bool resolve_max_pool_with_argmax(luci::CircleCustom *cop)
755{
756#define CHECK_OR_FALSE(condition) \
757 if (not(condition)) \
758 return false;
759
760 const std::vector<uint8_t> custom_options = cop->custom_options();
761 auto map = flexbuffers::GetRoot(custom_options).AsMap();
762
763 // Define params
764 // Note: Only `Targmax` equal to DT_INT64 is supported by tflite converter
765 // Note: Only `data_format` equal to "NHWC" is supported by tflite converter
766 // TODO add support of `include_batch_in_index` param
767 auto ksize_param = to_vector<uint32_t>(map["ksize"].AsTypedVector());
768 auto strides_param = to_vector<uint32_t>(map["strides"].AsTypedVector());
769 auto padding_param = map["padding"].As<std::string>();
770
771 // Batch size and depth of ksize more than 1 is not supported.
772 CHECK_OR_FALSE(ksize_param.size() == 4);
773 CHECK_OR_FALSE(ksize_param[0] == 1 && ksize_param[3] == 1);
774
775 CHECK_OR_FALSE(strides_param.size() == 4);
776 CHECK_OR_FALSE(strides_param[0] == 1 && strides_param[3] == 1);
777
778 // define Padding
779 auto padding = string_to_padding(padding_param);
780
781 // define Filter
782 luci::Filter filter;
783 filter.h(ksize_param[1]);
784 filter.w(ksize_param[2]);
785
786 // define Stride
787 luci::Stride stride;
788 stride.h(strides_param[1]);
789 stride.w(strides_param[2]);
790
791 // input node
792 auto const input = loco::must_cast<luci::CircleNode *>(cop->inputs(0));
793 CHECK_OR_FALSE(input->dtype() == loco::DataType::FLOAT32);
794 CHECK_OR_FALSE(input->rank() == 4);
795
796 // TODO support batch size > 1 and `include_batch_in_index` option
797 CHECK_OR_FALSE(input->dim(0).value() == 1);
798
799 // output nodes
800 auto const outputs = loco::succs(cop);
801 CHECK_OR_FALSE(outputs.size() == 2);
802 assert(outputs.size() == cop->numOutputs());
803
804 auto output0 = get_custom_output(cop, 0);
805 auto output1 = get_custom_output(cop, 1);
806
807 // From TF documentation: output of maxpool must has same type as input
808 assert(output0->dtype() == input->dtype());
809 assert(output1->dtype() == loco::DataType::S64 || output1->dtype() == loco::DataType::S32);
810
811 // Create MaxPool
812 auto maxpool = max_pool_branch(padding, stride, filter, cop);
813 auto argmax = argmax_branch(padding, stride, filter, cop);
814
815 // last argmax branch op is cast, it should have dtype initialized
816 assert(argmax->dtype() == output1->dtype());
817
818 // replace old node with new subgraph
819 cop->inputs(0, nullptr);
820 loco::replace(output0).with(maxpool);
821 loco::replace(output1).with(argmax);
822
823 return true;
824}

884bool ResolveCustomOpMaxPoolWithArgmaxPass::run(loco::Graph *g)
885{
886 bool changed = false;
887 for (auto node : loco::active_nodes(loco::output_nodes(g)))
888 {
889 auto cop = dynamic_cast<luci::CircleCustom *>(node);
890 if (not cop)
