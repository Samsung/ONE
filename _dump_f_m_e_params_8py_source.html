28# Recursively visit items and round floats with ndigits
29def _pretty_float(item, ndigits=6):
30 if isinstance(item, list):
31 return [_pretty_float(x, ndigits) for x in item]
32 if isinstance(item, float):
33 return round(item, ndigits)
34 return item

37# Dump scale parameters for FM equalization
38# The parameters are saved in the json file given as an input
39#
40# Contents of the json file ('<model_name>.fme_patterns.json')
41#
42# Before
43# [
44# {
45# “front”: <tensor_name>,
46# “back”: <tensor_name>,
47# “type”: ScaleOnly,
48# },
49# …
50# ]
51#
52# After
53# [
54# {
55# “front”: <tensor_name>,
56# “back”: <tensor_name>,
57# “type”: ScaleOnly,
58# “act_scale”: [..],
59# },
60# …
61# ]
62class DumpFMEParams:
63
64 # Return path to the data
65 # self._dir_path/<tid>.<data_idx>.npy
66 def _data_path(self, tid: int, data_idx: int):
67 assert (self._dir_path != None) # FIX_CALLER_UNLESS
68 return self._dir_path + '/' + str(tid) + '.' + str(data_idx) + '.npy'

70 def record_activaion_max(self, tensor: dict):
71 tensor_name = tensor['name']
72
73 act_max = getActivationMax(tensor['data'])
74 old_act_max = self.activation_max.get(tensor_name, act_max)
75 self.activation_max[tensor_name] = np.maximum.reduce([old_act_max, act_max])
76
77 def StartAnalysis(self, args: str):
78 """Called when the analysis starts"""
79 self._json_path = args
80 self._dir_path = os.path.dirname(args)
81 # Data structure to save tensor information
82 # {
83 # <tensor_name>: <activation max>
84 # }
85 self.activation_max: Dict[str, float] = {}
86
87 # Names of target tensors ('back' of equalization pattern)
88 self._target_tensors = []
89
90 with open(args, 'r') as f:
91 patterns = json.load(f)
92 for pattern in patterns:
93 self._target_tensors.append(pattern['front'])
94 self._patterns = patterns
95
96 def DefaultOpPost(self, name, opcode, inputs, output):
97 # DO NOTHING
98 pass
99
100 def Conv2DPost(self, name, input, filter, bias, padding, stride, dilation, output,
101 fused_act):
102 if name in self._target_tensors:
103 self.record_activaion_max(output)
104
105 def DepthwiseConv2DPost(self, name, input, filter, bias, padding, stride,
106 depthMultiplier, dilation, output, fused_act):
107 if name in self._target_tensors:
108 self.record_activaion_max(output)
109
110 def FullyConnectedPost(self, name, input, weights, bias, output, fused_act):
111 if name in self._target_tensors:
112 self.record_activaion_max(output)
113
114 def InstanceNormPost(self, name, input, gamma, beta, epsilon, output, fused_act):
115 if name in self._target_tensors:
116 self.record_activaion_max(output)
117
118 def TransposeConvPost(self, name, input, filter, output_shape, bias, padding, stride,
119 output):
120 if name in self._target_tensors:
121 self.record_activaion_max(output)
122
123 def EndAnalysis(self):
124 """Called when the analysis ends"""
125
126 res = []
127 for pattern in self._patterns:
128 # pattern['front'] is the input of pattern['back']
129 tensor_name = pattern['front']
130 eq_type = pattern['type']
131
132 act_max = self.activation_max[tensor_name]
133
134 if eq_type == 'ScaleOnly':
135 pattern['act_scale'] = _pretty_float(act_max).tolist()
136 else:
137 raise ValueError('Unknown equalization type: ' + eq_type)
138
139 res.append(pattern)
140
141 # Overwrite
142 with open(self._json_path, 'w') as json_file:
143 json.dump(res, json_file)
