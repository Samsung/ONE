70template <uint32_t N, typename Base> class FixedArityNode : public Base
71{
72public:
73 FixedArityNode()
74 {
75 _args.resize(N);
76 for (uint32_t n = 0; n < N; ++n)
77 {
78 _args[n] = std::make_unique<loco::Use>(this);
79 }
80 }
81
82 virtual ~FixedArityNode() = default;
83
84public:
85 uint32_t arity(void) const final { return N; }
86
87 loco::Node *arg(uint32_t n) const final { return _args.at(n)->node(); }
88
89 void drop(void) final
90 {
91 for (uint32_t n = 0; n < N; ++n)
92 {
93 _args.at(n)->node(nullptr);
94 }
95 }
96
97protected:
98 // This API allows inherited classes to access "_args" field.
99 loco::Use *at(uint32_t n) const { return _args.at(n).get(); }

101private:
102 std::vector<std::unique_ptr<loco::Use>> _args{};
103};
