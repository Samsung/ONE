36void calculateGRU(const float *input_data, const float *weight_input_data,
37 const float *weight_hidden_data, const float *bias_input_data,
38 const float *bias_hidden_data, float *output_data,
39 const core::OMRuntimeShape &input_shape, const core::OMRuntimeShape &output_shape,
40 const core::OMRuntimeShape &weight_input_shape,
41 const core::OMRuntimeShape &weight_hidden_shape, float *output_input_data,
42 float *output_hidden_data, const core::OMRuntimeShape &output_shape_fc,
43 float *intermediate_buffer)
44{
45 core::FullyConnectedParams op_params{};
46 // As FC nodes doesn't have any activations inside GRU, let' use just numeric limits
47 op_params.float_activation_min = std::numeric_limits<float>::lowest();
48 op_params.float_activation_max = std::numeric_limits<float>::max();
49 // If intermediate_buffer != nullptr - then it is train mode and we need save intermediate inform
50 bool is_train_mode = intermediate_buffer != nullptr;
51 if (is_train_mode)
52 {
53 // Copy input for FC Input to calculate weights gradients
54 std::memcpy(intermediate_buffer, output_data, output_shape.flatSize() * sizeof(float));
55 // Move intermediate_buffer pointer
56 intermediate_buffer += output_shape.flatSize();
57 }
58 // FC Input
59 FullyConnected(op_params, output_data, weight_input_shape, weight_input_data, bias_input_data,
60 output_shape_fc, output_input_data);
61
62 // FC Hidden
63 // Note: input for this FC node will be saved without intermediate buffer
64 FullyConnected(op_params, input_data, weight_hidden_shape, weight_hidden_data, bias_hidden_data,
65 output_shape_fc, output_hidden_data);
66
67 int num_elements = output_shape_fc.dims(1) / 3;
68
69 float *second_hidden_part = output_hidden_data + num_elements;
70 float *second_input_part = output_input_data + num_elements;
71
72 float *third_hidden_part = second_hidden_part + num_elements;
73 float *third_input_part = second_input_part + num_elements;
74
75 // Calculate Left part
76 for (int i = 0; i < num_elements; ++i)
77 {
78 output_input_data[i] += output_hidden_data[i];
79 }
80
81 // If train mode - save logistic input
82 if (is_train_mode)
83 {
84 std::memcpy(intermediate_buffer, output_input_data, output_shape.flatSize() * sizeof(float));
85 // Move intermediate_buffer pointer
86 intermediate_buffer += output_shape.flatSize();
87 }
88 Logistic(num_elements, output_input_data, output_input_data);
89
90 // If train mode - save most left mul input (right input)
91 if (is_train_mode)
92 {
93 std::memcpy(intermediate_buffer, output_input_data, output_shape.flatSize() * sizeof(float));
94 // Move intermediate_buffer pointer
95 intermediate_buffer += output_shape.flatSize();
96 }
97 // Calculate most left mul
98 float *most_left_part_final = output_input_data;
99 float *first_part = output_input_data;
100 for (int i = 0; i < num_elements; ++i)
101 {
102 output_data[i] *= most_left_part_final[i];
103 first_part[i] = 1.0f - first_part[i];
104 }
105
106 // Calc second part
107 for (int i = 0; i < num_elements; ++i)
108 {
109 second_hidden_part[i] += second_input_part[i];
110 }
111 // If train mode - save logistic input
112 if (is_train_mode)
113 {
114 std::memcpy(intermediate_buffer, second_hidden_part, output_shape.flatSize() * sizeof(float));
115 // Move intermediate_buffer pointer
116 intermediate_buffer += output_shape.flatSize();
117 }
118 Logistic(num_elements, second_hidden_part, second_hidden_part);
119
120 // If train mode - save mul input (left and right)
121 if (is_train_mode)
122 {
123 // Left input
124 std::memcpy(intermediate_buffer, second_hidden_part, output_shape.flatSize() * sizeof(float));
125 // Move intermediate_buffer pointer
126 intermediate_buffer += output_shape.flatSize();
127
128 // Right input
129 std::memcpy(intermediate_buffer, third_input_part, output_shape.flatSize() * sizeof(float));
130 // Move intermediate_buffer pointer
131 intermediate_buffer += output_shape.flatSize();
132 }
133 for (int i = 0; i < num_elements; ++i)
134 {
135 second_hidden_part[i] *= third_input_part[i];
136 second_hidden_part[i] += third_hidden_part[i];
137 }
138 // If train mode - save tanh input
139 if (is_train_mode)
140 {
141 std::memcpy(intermediate_buffer, second_hidden_part, output_shape.flatSize() * sizeof(float));
142 // Move intermediate_buffer pointer
143 intermediate_buffer += output_shape.flatSize();
144 }
145 for (int i = 0; i < num_elements; ++i)
146 {
147 if (second_hidden_part[i] > 19)
148 {
149 second_hidden_part[i] = 1;
150 }
151 else if (second_hidden_part[i] < -19)
152 {
153 second_hidden_part[i] = -1;
154 }
155 else
156 {
157 second_hidden_part[i] = std::tanh(second_hidden_part[i]);
158 }
159 }
160
161 // If train mode - save mul input (left and right)
162 if (is_train_mode)
163 {
164 // Left input
165 std::memcpy(intermediate_buffer, first_part, output_shape.flatSize() * sizeof(float));
166 // Move intermediate_buffer pointer
167 intermediate_buffer += output_shape.flatSize();
168
169 // Right input
170 std::memcpy(intermediate_buffer, second_hidden_part, output_shape.flatSize() * sizeof(float));
171 // Move intermediate_buffer pointer
172 intermediate_buffer += output_shape.flatSize();
173 }
174 for (int i = 0; i < num_elements; ++i)
175 {
176 second_hidden_part[i] *= first_part[i];
177 output_data[i] += second_hidden_part[i];
178 }
179}

183OMStatus GRU(const float *input_data, const float *weight_input_data,
184 const float *weight_hidden_data, const float *bias_input_data,
185 const float *bias_hidden_data, const float *hidden_state_data, float *output_data,
186 float *output_input_data, float *output_hidden_data,
187 const core::OMRuntimeShape &input_shape, const core::OMRuntimeShape &output_shape,
188 const core::OMRuntimeShape &weight_input_shape,
189 const core::OMRuntimeShape &weight_hidden_shape, const size_t intermediate_buffer_size,
190 float *intermediate_buffer)
191{
192 const int32_t time = input_shape.dims(0);
193
194 core::OMRuntimeShape output_shape_fc(2);
195 output_shape_fc.setDim(0, 1);
196 output_shape_fc.setDim(1, weight_hidden_shape.dims(0));
197
198 std::memcpy(output_data, hidden_state_data, output_shape.flatSize() * sizeof(float));
199
200 for (int i = 0; i < time; ++i)
201 {
202 calculateGRU(input_data, weight_input_data, weight_hidden_data, bias_input_data,
203 bias_hidden_data, output_data, input_shape, output_shape, weight_input_shape,
204 weight_hidden_shape, output_input_data, output_hidden_data, output_shape_fc,
205 intermediate_buffer);
206 input_data += input_shape.dims(2);
207 if (intermediate_buffer_size != 0)
208 {
209 assert(intermediate_buffer != nullptr);
210 intermediate_buffer += intermediate_buffer_size;
211 }
212 }
213 return Ok;
214}
