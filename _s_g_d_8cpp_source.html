37/*
38 * Update internal states according to calculated gradients using Adam theory
39 * grad(t) = grad(t - 1) + calculated_grad(t)
40 */
41OMStatus SGD::handle(core::OMRuntimeStorage &backward_storage, core::OMRuntimeContext &context)
42{
43 auto &backward_tensor_to_data = backward_storage.getTensorIndexToData();
44 // Check is allocated or not helper buffers
45 if (_tensor_index_to_gradient.empty())
46 {
47 // If not - let's just move it with calculations
48 // Goes over all calculated gradients
49 // Warning: assume that backward storage at this moment contains only weigths gradients -
50 // This should be done due to execution plan work
51 for (auto &tensor_to_data : backward_tensor_to_data)
52 {
53 // Move data
54 _tensor_index_to_gradient[tensor_to_data.first] = tensor_to_data.second;
55 tensor_to_data.second = nullptr;
56 }
57 backward_tensor_to_data.clear();
58 }
59 else
60 {
61 // Goes over all calculated gradients
62 // Warning: assume that backward storage at this moment contains only weigths gradients -
63 // This should be done due to execution plan work
64 for (auto &tensor_to_data : backward_tensor_to_data)
65 {
66 auto tensor = context.getTensorByIndex(tensor_to_data.first);
67 core::OMRuntimeShape shape(tensor);
68
69 const auto flat_size = shape.flatSize();
70
71 auto *grad_data = reinterpret_cast<float *>(_tensor_index_to_gradient[tensor_to_data.first]);
72 auto *calculated_data = reinterpret_cast<float *>(tensor_to_data.second);
73
74 for (uint32_t i = 0; i < flat_size; ++i)
75 {
76 grad_data[i] += calculated_data[i];
77 }
78 }
79 }
80
81 return Ok;
82}

84/*
85 * Update weights according to Adam theory:
86 *
87 * w(t + 1) = w(t) - lambda * grad(t) / batch_size
88 */
89OMStatus SGD::updateWeights(const onert_micro::OMTrainingContext &training_config,
90 core::OMRuntimeContext &context)
91{
92 assert(!_tensor_index_to_gradient.empty());
93 if (_tensor_index_to_gradient.empty())
94 return UnknownError;
95
96 for (auto &tensor_to_data : _tensor_index_to_gradient)
97 {
98 auto tensor = context.getTensorByIndex(tensor_to_data.first);
99 core::OMRuntimeShape shape(tensor);
100
101 const auto flat_size = shape.flatSize();
102
103 auto *grad_data = reinterpret_cast<float *>(tensor_to_data.second);
104 uint8_t *weight_data = nullptr;
105 if (context.getConstDataByTensorIndex(&weight_data, tensor_to_data.first) != Ok)
106 return UnknownError;
107
108 assert(weight_data != nullptr);
109 if (weight_data == nullptr)
110 return UnknownError;
111
112 auto *f_weight_data = reinterpret_cast<float *>(weight_data);
113 float lambda = training_config.learning_rate;
114 const uint32_t batch_size = training_config.batch_size;
115
116 assert(batch_size != 0);
117
118 for (uint32_t i = 0; i < flat_size; ++i)
119 {
120 f_weight_data[i] -= (lambda * grad_data[i]) / (static_cast<float>(batch_size));
121 }
122 }
123 return Ok;
124}
