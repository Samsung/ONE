42bool inputs_dtype_ready(loco::Node *node)
43{
44 assert(node != nullptr);
45
46 for (uint32_t arity = 0; arity < node->arity(); ++arity)
47 {
48 if (!loco::TypeInference::known(node->arg(arity)))
49 {
50 return false;
51 }
52 }
53 return true;

61bool TypeInferenceSession::to(Graph *g) const
62{
63 bool changed = false;
64
65 for (auto node : postorder_traversal(output_nodes(g)))
66 {
67 if (_rule->recognize(node->dialect()))
68 {
69 DataType dtype = DataType::Unknown;
70
71 if (!dtype_known(node) && inputs_dtype_ready(node))
72 {
73 if (_rule->infer(node, dtype))
74 {
75 node->annot(std::make_unique<DataTypeAnnotation>(dtype));
76 changed = true;
77 }
78 }
79 }
80 }
81
82 return changed;
83}

172bool CanonicalTypeInferenceRule::recognize(const Dialect *d) const
173{
174 // This rule recognizes only "loco.canonical" dialect!
175 return CanonicalDialect::get() == d;
176}
177
178bool CanonicalTypeInferenceRule::infer(const Node *node, DataType &dtype) const
179{
180 assert(node->dialect() == loco::CanonicalDialect::get());
181 assert(dynamic_cast<const loco::CanonicalNode *>(node) != nullptr);
182
183 CanonicalTypeForwardAlgorithm alg;
184 dtype = loco::must_cast<const loco::CanonicalNode *>(node)->accept(&alg);
185
186 return true;
187}

189bool MultiDialectTypeInferenceRule::recognize(const Dialect *d) const
190{
191 const auto found = _rules.find(d);
192
193 if (found == _rules.cend())
194 return false;
195
196 auto rule = found->second;
197 auto result = rule->recognize(d);
198
199 return result;
200}
201
202bool MultiDialectTypeInferenceRule::infer(const Node *node, DataType &dtype) const
203{
204 const auto found = _rules.find(node->dialect());
205
206 if (found == _rules.cend())
207 return false;
208
209 auto rule = found->second;
210 if (rule->infer(node, dtype))
211 return true;
212
213 return false;
214}
215
216MultiDialectTypeInferenceRule &MultiDialectTypeInferenceRule::bind(const Dialect *d,
217 const TypeInferenceRule *rule)
218{
219 assert(_rules.find(d) == _rules.end());
220 assert(rule->recognize(d));
221
222 _rules[d] = rule;
223
224 return (*this);
225}
