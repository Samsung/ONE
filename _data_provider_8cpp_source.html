29// Check the type and the shape of input_node
30// Throw an exception if type or shape does not match
31void verifyTypeShape(const luci::CircleNode *input_node, const loco::DataType &dtype,
32 const Shape &shape)
33{
34 assert(input_node != nullptr); // FIX_CALLER_UNLESS
35
36 // Type check
37 if (dtype != input_node->dtype())
38 throw std::runtime_error("Wrong input type.");
39
40 if (shape.size() != input_node->rank())
41 throw std::runtime_error("Input rank mismatch.");
42
43 for (uint32_t i = 0; i < shape.size(); i++)
44 {
45 if (not(shape.at(i) == input_node->dim(i)))
46 throw std::runtime_error("Input shape mismatch.");
47 }
48}

52H5FileDataProvider::H5FileDataProvider(const std::string &h5file, const std::string &module_path)

73void H5FileDataProvider::getSampleInput(uint32_t sample, uint32_t input, InputData &data) const
74{
75 if (_is_raw_data)
76 {
77 _importer.readTensor(sample, input, data.data().data(), data.data().size());
78 }
79 else
80 {
81 loco::DataType dtype;
82 Shape shape;
83 _importer.readTensor(sample, input, &dtype, &shape, data.data().data(), data.data().size());
84
85 // Check the type and the shape of the input data is valid
86 auto input_node = loco::must_cast<luci::CircleNode *>(_input_nodes.at(input));
87 verifyTypeShape(input_node, dtype, shape);
88 }
