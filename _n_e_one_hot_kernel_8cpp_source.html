66template <typename U> void validate_depth(const ITensor *depth, const ITensor *output, int axis)
67{
68 ARM_COMPUTE_ERROR_ON(*(reinterpret_cast<U *>(depth->buffer())) < 0);
69 ARM_COMPUTE_ERROR_ON(static_cast<U>(output->info()->tensor_shape()[axis]) !=
70 *(reinterpret_cast<U *>(depth->buffer())));
71}
72
73Status validate_arguments(const ITensorInfo *indices, const ITensorInfo *depth,
74 const ITensorInfo *on_value, const ITensorInfo *off_value,
75 const ITensorInfo *output, int axis)
76{
77 ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(indices, depth, on_value, off_value, output);
78 const int actual_axis = wrap_around(axis, static_cast<int>(output->num_dimensions()));
79 ARM_COMPUTE_RETURN_ERROR_ON(output->num_dimensions() > 4);
80 ARM_COMPUTE_RETURN_ERROR_ON(on_value->tensor_shape().total_size() != 1);
81 ARM_COMPUTE_RETURN_ERROR_ON(0 > actual_axis ||
82 actual_axis >= static_cast<int>(output->num_dimensions()));
83 ARM_COMPUTE_RETURN_ERROR_ON(on_value->data_type() == DataType::UNKNOWN);
84 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(on_value, 1, DataType::U8, DataType::S8,
85 DataType::U16, DataType::S16, DataType::F16,
86 DataType::U32, DataType::S32, DataType::F32);
87 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(indices, 1, DataType::U32, DataType::S32);
88
89 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(on_value, off_value);
90 if (output->total_size() != 0)
91 {
92 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(on_value, output);
93 }
94
95 return Status{};
96}

100template <typename U, std::enable_if_t<std::is_integral<U>::value, int> = 0>

103 return index >= 0 && index < depth;

113template <typename U>
114void NEOneHotKernel::onehot_0_axis(const Window &window, const ThreadInfo &info)
115{
116 ARM_COMPUTE_UNUSED(info);
117 // Validate that the depth are not negative
118 validate_depth<U>(_depth, _output, _axis);
119 Window output_window{window};
120 output_window.set(Window::DimX, Window::Dimension(0, 1, 1));
121 Iterator output_it(_output, output_window);
122 const U off_value = *reinterpret_cast<U *>(_off_value->buffer());
123 execute_window_loop(
124 output_window,
125 [&](const Coordinates &id) {
126 std::fill_n(output_it.ptr(), _output->info()->dimension(0) * _output->info()->element_size(),
127 off_value);
128 Coordinates indices_id(id);
129 indices_id.remove(0);
130 const U new_index = *(reinterpret_cast<U *>(_indices->ptr_to_element(indices_id)));
131 if (isOnValue(new_index, *(reinterpret_cast<U *>(_depth->buffer()))))
132 {
133 Coordinates onehot_id(id);
134 onehot_id.set(0, new_index);
135 std::copy_n(_on_value->buffer(), _output->info()->element_size(),
136 _output->ptr_to_element(onehot_id));
137 }
138 },
139 output_it);
140}

142template <typename U>
143inline void NEOneHotKernel::onehot_n_axis(const Window &window, const ThreadInfo &info)
144{
145 ARM_COMPUTE_UNUSED(info);
146 // Validate that the indices are not negative
147 validate_depth<U>(_depth, _output, _axis);
148 Iterator output_it(_output, window);
149 execute_window_loop(
150 window,
151 [&](const Coordinates &id) {
152 Coordinates indices_id(id);
153 indices_id.remove(_axis);
154 const U new_index = *(reinterpret_cast<U *>(_indices->ptr_to_element(indices_id)));
155 if (isOnValue(new_index, *(reinterpret_cast<U *>(_depth->buffer()))))
156 {
157 Coordinates onehot_id(id);
158 onehot_id.set(_axis, new_index);
159 std::copy_n(static_cast<U>(id[_axis]) == new_index ? _on_value->buffer()
160 : _off_value->buffer(),
161 _output->info()->element_size(), output_it.ptr());
162 }
163 },
164 output_it);
165}

167void NEOneHotKernel::configure(const ITensor *indices, const ITensor *depth,
168 const ITensor *on_value, const ITensor *off_value, ITensor *output,
169 int axis)
170{
171 ARM_COMPUTE_ERROR_ON_NULLPTR(indices, depth, on_value, off_value, output);
172 ARM_COMPUTE_ERROR_ON(output->info()->total_size() == 0);
173 ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(indices->info(), depth->info(), on_value->info(),
174 off_value->info(), output->info(), axis));
175 _indices = indices;
176 _depth = depth;
177 _on_value = on_value;
178 _off_value = off_value;
179 _output = output;
180 _axis = wrap_around(axis, static_cast<int>(output->info()->num_dimensions()));
181 if (0 == _axis)
182 {
183 switch (_indices->info()->data_type())
184 {
185 case DataType::U32:
186 _func = &NEOneHotKernel::onehot_0_axis<uint32_t>;
187 break;
188 case DataType::S32:
189 _func = &NEOneHotKernel::onehot_0_axis<int32_t>;
190 break;
191 default:
192 ARM_COMPUTE_ERROR("Not supported");
193 break;
194 }
195 }
196 else
197 {
198 switch (_indices->info()->data_type())
199 {
200 case DataType::U32:
201 _func = &NEOneHotKernel::onehot_n_axis<uint32_t>;
202 break;
203 case DataType::S32:
204 _func = &NEOneHotKernel::onehot_n_axis<int32_t>;
205 break;
206 default:
207 ARM_COMPUTE_ERROR("Not supported");
208 break;
209 }
210 }
211 // Create window
212 Window win = calculate_max_window(*output->info(), Steps());
213 output->info()->set_valid_region(ValidRegion(Coordinates(), output->info()->tensor_shape()));
214 INEKernel::configure(win);
215}
