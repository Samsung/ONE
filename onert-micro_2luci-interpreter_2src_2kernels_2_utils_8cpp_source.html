29template <typename T>
30void calculateActivationRange(Activation activation, T *activation_min, T *activation_max)
31{
32 switch (activation)
33 {
34 case Activation::NONE:
35 *activation_min = std::numeric_limits<T>::lowest();
36 *activation_max = std::numeric_limits<T>::max();
37 break;
38 case Activation::RELU:
39 *activation_min = 0;
40 *activation_max = std::numeric_limits<T>::max();
41 break;
42 case Activation::RELU_N1_TO_1:
43 *activation_min = -1;
44 *activation_max = 1;
45 break;
46 case Activation::RELU6:
47 *activation_min = 0;
48 *activation_max = 6;
49 break;
50 default:
51 assert(false && "Unsupported activation.");
52 }
53}

62static void calculateActivationRangeQuantizedImpl(Activation activation, int32_t qmin, int32_t qmax,
63 const Tensor *output, int32_t *activation_min,
64 int32_t *activation_max)
65{
66 const float scale = output->scale();
67 const int32_t zero_point = output->zero_point();
68
69 auto quantize = [scale, zero_point](float x) {
70 return zero_point + static_cast<int32_t>(std::round(x / scale));
71 };
72
73 switch (activation)
74 {
75 case Activation::NONE:
76 case Activation::TANH:
77 *activation_min = qmin;
78 *activation_max = qmax;
79 break;
80 case Activation::RELU:
81 *activation_min = std::max(qmin, quantize(0.0f));
82 *activation_max = qmax;
83 break;
84 case Activation::RELU_N1_TO_1:
85 *activation_min = std::max(qmin, quantize(-1.0f));
86 *activation_max = std::min(qmax, quantize(1.0f));
87 break;
88 case Activation::RELU6:
89 *activation_min = std::max(qmin, quantize(0.0f));
90 *activation_max = std::min(qmax, quantize(6.0f));
91 break;
92 default:
93 assert(false && "Unsupported activation.");
94 }
95}

97void calculateActivationRangeQuantized(Activation activation, const Tensor *output,
98 int32_t *activation_min, int32_t *activation_max)
99{
100 assert(output->zero_points().size() == 1);
101 int32_t qmin{};
102 int32_t qmax{};
103 switch (output->element_type())
104 {
105 case DataType::U8:
106 qmin = 0;
107 qmax = std::numeric_limits<uint8_t>::max();
108 break;
109 case DataType::S8:
110 qmin = -std::numeric_limits<int8_t>::max();
111 qmax = std::numeric_limits<int8_t>::max();
112 break;
113 case DataType::S16:
114 // For now, assume that signed int16 type implies signed symmetric quantization.
115 assert(output->zero_point() == 0);
116 qmin = -std::numeric_limits<int16_t>::max();
117 qmax = std::numeric_limits<int16_t>::max();
118 break;
119 default:
120 assert(false && "Unsupported type.");
121 }
122
123 calculateActivationRangeQuantizedImpl(activation, qmin, qmax, output, activation_min,
124 activation_max);
125}

127void quantizeMultiplier(double double_multiplier, int32_t *quantized_multiplier, int *shift)
128{
129 if (double_multiplier == 0.0)
130 {
131 *quantized_multiplier = 0;
132 *shift = 0;
133 return;
134 }
135
136 const double q = std::frexp(double_multiplier, shift);
137 auto q_fixed = static_cast<int64_t>(std::round(q * (int64_t(1) << 31)));
138
139 if (q_fixed == (int64_t(1) << 31))
140 {
141 q_fixed /= 2;
142 ++*shift;
143 }
144 assert(q_fixed <= std::numeric_limits<int32_t>::max());
145 // A shift amount smaller than -31 would cause all bits to be shifted out
146 // and thus all results would be zero. We implement that instead with
147 // q_fixed==0, so as to avoid hitting issues with right-shift
148 // operations with shift amounts greater than 31. Note that this happens
149 // roughly when abs(double_multiplier) < 2^-31 and the present handling means
150 // that we're effectively flushing tiny double_multiplier's to zero.
151 // We could conceivably handle values in the range (roughly) [32, 63]
152 // as 'denormals' i.e. (shift==0, q_fixed < 2^30). In that point of view
153 // the present handling is just doing 'flush denormals to zero'. We could
154 // reconsider and actually generate nonzero denormals if a need arises.
155 if (*shift < -31)
156 {
157 *shift = 0;
158 q_fixed = 0;
159 }
160 *quantized_multiplier = static_cast<int32_t>(q_fixed);
161}

163void quantizeMultiplierSmallerThanOneExp(double double_multiplier, int32_t *quantized_multiplier,
164 int *left_shift)
165{
166 assert(double_multiplier < 1.0);
167 assert(double_multiplier > 0.0);
168 int shift;
169 quantizeMultiplier(double_multiplier, quantized_multiplier, &shift);
170 assert(shift <= 0);
171 *left_shift = shift;
172}

174Shape calculateShapeForBroadcast(const Shape &input1_shape, const Shape &input2_shape)
175{
176 const int num_input1_dims = input1_shape.num_dims();
177 const int num_input2_dims = input2_shape.num_dims();
178 const int num_out_dims = std::max(num_input1_dims, num_input2_dims);
179 Shape output_shape(num_out_dims);
180
181 for (int i = 0; i < num_out_dims; ++i)
182 {
183 const int32_t input1_dim = i < num_input1_dims ? input1_shape.dim(num_input1_dims - i - 1) : 1;
184 const int32_t input2_dim = i < num_input2_dims ? input2_shape.dim(num_input2_dims - i - 1) : 1;
185
186 bool need_broadcast = input1_dim != input2_dim;
187 bool can_broadcast = input1_dim == 1 || input2_dim == 1;
188 LUCI_INTERPRETER_CHECK(!need_broadcast || can_broadcast);
189
190 output_shape.dim(num_out_dims - i - 1) = std::max(input1_dim, input2_dim);
191 }
192
193 return output_shape;
194}
