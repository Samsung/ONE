54// Helper functions for tensor size calculation
55uint64_t num_elems(const nnfw_tensorinfo *tensor_info)
56{
57 uint64_t n = 1;
58 for (int32_t i = 0; i < tensor_info->rank; ++i)
59 n *= tensor_info->dims[i];
60 return n;
61}

63uint64_t bufsize_for(const nnfw_tensorinfo *ti)
64{
65 static int elmsize[] = {
66 sizeof(float), /* NNFW_TYPE_TENSOR_FLOAT32 */
67 sizeof(int), /* NNFW_TYPE_TENSOR_INT32 */
68 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM */
69 sizeof(bool), /* NNFW_TYPE_TENSOR_BOOL = 3 */
70 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_UINT8 = 4 */
71 sizeof(int64_t), /* NNFW_TYPE_TENSOR_INT64 = 5 */
72 sizeof(int8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED = 6 */
73 sizeof(int16_t), /* NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED = 7 */
74 };
75 return elmsize[ti->dtype] * num_elems(ti);
76}

84GGMA_STATUS context::from_package(ggma_context **session, const char *package_path)
85{
86 if (session == nullptr)
87 return GGMA_STATUS_UNEXPECTED_NULL;
88 try
89 {
90 auto new_session = std::unique_ptr<context>(new context(package_path));
91 *session = reinterpret_cast<ggma_context *>(new_session.release());
92 }
93 catch (const std::bad_alloc &e)
94 {
95 std::cerr << "Error during session creation" << std::endl;
96 *session = nullptr; // Set nullptr on error to keep the old behavior
97 return GGMA_STATUS_OUT_OF_MEMORY;
98 }
99 catch (const std::exception &e)
100 {
101 std::cerr << "Error during session initialization : " << e.what() << std::endl;
102 *session = nullptr; // Set nullptr on error to keep the old behavior
103 return GGMA_STATUS_ERROR;
104 }
105 return GGMA_STATUS_NO_ERROR;
106}

119void context::prefill(ggma_token *tokens, size_t n_tokens, std::vector<uint8_t> &hidden_state)
120{
121 std::filesystem::path nnpkg_path = std::filesystem::path(_package_path) / "prefill";
122 nnfw_session *session = create_and_prepare_session(nnpkg_path.string());
123
124 nnfw_tensorinfo ti;
125
126 // Input 0: token_id
127 // shape = [n_batch, n_seq]
128 // n_batch = 1
129 NNFW_ENSURE_STATUS(nnfw_input_tensorinfo(session, 0, &ti));
130 if (ti.rank != 2 || ti.dims[0] != 1)
131 throw std::runtime_error("prefill : invalid input shape");
132
133 // TODO: Check ubatch from model is same to runtime config
134 int ubatch = ti.dims[1]; // Number of tokens after padding to align to 32 multiples
135 // Use tokens as input without copying (zero-copy)
136 NNFW_ENSURE_STATUS(nnfw_set_input(session, 0, ti.dtype, tokens, ubatch * sizeof(ggma_token)));
137
138 // Expected Output:
139 //
140 // Index | Name | Description
141 // ------|----------|---------------------------
142 // 0 | k0 | key cache for layer 0
143 // 1 | v0 | value cache for layer 0
144 // ... | ... | ...
145 // 2n-2 | k{n-1} | key cache for layer n-1
146 // 2n-1 | v{n-1} | value cache for layer n-1
147 // 2n | hidden | hidden state
148 //
149 // where n = number of layers
150
151 uint32_t num_outputs;
152 NNFW_ENSURE_STATUS(nnfw_output_size(session, &num_outputs));
153 if (num_outputs != _cfg.model.n_layers * 2 + 1)
154 throw std::runtime_error("prefill : number of outputs mismatch");
155
156 // Output 0~2n-1: KV caches
157 for (int i = 0; i < _cfg.model.n_layers; ++i)
158 {
159 if (!_cache.v[i].empty())
160 NNFW_ENSURE_STATUS(nnfw_set_output(session, 2 * i, _cache.to_nnfw_type(), _cache.v[i].data(),
161 _cache.v[i].size()));
162 if (!_cache.k[i].empty())
163 NNFW_ENSURE_STATUS(nnfw_set_output(session, 2 * i + 1, _cache.to_nnfw_type(),
164 _cache.k[i].data(), _cache.k[i].size()));
165 }
166
167 // Output 2n: hidden_state
168 // shape = [n_batch, n_seq, n_emb]
169 // n_batch = 1
170
171 NNFW_ENSURE_STATUS(nnfw_output_tensorinfo(session, 2 * _cfg.model.n_layers, &ti));
172 if (ti.rank != 3 || ti.dims[0] != 1)
173 throw std::runtime_error("prefill : invalid hidden shape");
174
175 // Allocate output buffer
176 hidden_state.resize(bufsize_for(&ti), 0);
177 // Output buffer setup - use externally allocated hidden_state (single output for single token)
178 NNFW_ENSURE_STATUS(
179 nnfw_set_output(session, num_outputs - 1, ti.dtype, hidden_state.data(), hidden_state.size()));
180
181 NNFW_ENSURE_STATUS(nnfw_run(session));
182 nnfw_close_session(session);
183}

185void context::unemb(std::vector<uint8_t> &hidden_state, size_t n_tokens, std::vector<float> &logits)
186{
187 std::filesystem::path nnpkg_path = std::filesystem::path(_package_path) / "unemb";
188 nnfw_session *session = create_and_prepare_session(nnpkg_path.string());
189
190 // Input buffer setup - use externally allocated hidden_state
191 nnfw_tensorinfo ti;
192 NNFW_ENSURE_STATUS(nnfw_input_tensorinfo(session, 0, &ti));
193 // ti[0] : n_batch
194 // ti[1] : n_seq = ubatch if padded
195 // = n_tokens if not padded
196 if (ti.rank != 3 || ti.dims[0] != 1)
197 throw std::runtime_error("unemb : invalid input shape");
198 assert(ti.dims[1] == _cfg.ubatch); // Previously, it was padded to ubatch.
199 // Handle effective (actual) tokens only.
200 ti.dims[1] = n_tokens;
201 // Update buffer and nnfw input tensor info as sequence length is adjusted.
202 hidden_state.resize(bufsize_for(&ti), 0);
203 NNFW_ENSURE_STATUS(nnfw_set_input_tensorinfo(session, 0, &ti));
204 NNFW_ENSURE_STATUS(
205 nnfw_set_input(session, 0, ti.dtype, hidden_state.data(), hidden_state.size()));
206
207 // Output buffer setup - use externally allocated logits
208 NNFW_ENSURE_STATUS(nnfw_output_tensorinfo(session, 0, &ti));
209 // Check if output data type is float
210 if (ti.dtype != NNFW_TYPE_TENSOR_FLOAT32)
211 throw std::runtime_error("unemb: output tensor must be float type");
212 // Allocate output buffer
213 // ti[0] : n_batch
214 // ti[1] : n_seq = ubatch if padded
215 // = n_tokens if not padded
216 if (ti.rank != 3 || ti.dims[0] != 1)
217 throw std::runtime_error("unemb : invalid output shape");
218 assert(ti.dims[1] == _cfg.ubatch); // Previously, it was padded to ubatch.
219 // Handle effective (actual) tokens only.
220 ti.dims[1] = n_tokens;
221 logits.resize(num_elems(&ti), 0);
222 NNFW_ENSURE_STATUS(
223 nnfw_set_output(session, 0, ti.dtype, logits.data(), logits.size() * sizeof(logits[0])));
224
225 NNFW_ENSURE_STATUS(nnfw_run(session));
226 nnfw_close_session(session);
227}

229// Template implementation to eliminate code duplication
230template <bool ReturnLogits, typename OutputType>
231void context::decode_impl(ggma_token token_id, OutputType &output)
232{
233 std::filesystem::path nnpkg_path = std::filesystem::path(_package_path) / "decode";
234 nnfw_session *session = create_and_prepare_session(nnpkg_path.string());
235
236 // Expected Input:
237 //
238 // Index | Name | Description
239 // ------|-----------|------------------------
240 // 0 | token_id | input token to decode
241 // 1 | k0 | key cache for layer 0
242 // 2 | k1 | key cache for layer 1
243 // ... | ... | ...
244 // n | k{n-1} | key cache for layer n-1
245 // n+1 | v0 | value cache for layer 0
246 // n+2 | v1 | value cache for layer 1
247 // ... | ... | ...
248 // 2n | v{n-1} | value cache for layer n-1
249 // 2n+1 | cache_pos | current cache position
250 //
251 // where n = number of layers
252
253 // Input 0: Token ID - use directly without copying
254 nnfw_tensorinfo token_ti;
255 NNFW_ENSURE_STATUS(nnfw_input_tensorinfo(session, 0, &token_ti));
256 NNFW_ENSURE_STATUS(nnfw_set_input(session, 0, token_ti.dtype, &token_id, sizeof(ggma_token)));
257
258 // Input 1~2n: KV caches - use directly without copying
259 for (int i = 0; i < _cfg.model.n_layers; ++i)
260 {
261 NNFW_ENSURE_STATUS(nnfw_set_input(session, 1 + i, _cache.to_nnfw_type(), _cache.k[i].data(),
262 _cache.k[i].size()));
263 }
264 for (int i = 0; i < _cfg.model.n_layers; ++i)
265 {
266 NNFW_ENSURE_STATUS(nnfw_set_input(session, 1 + _cfg.model.n_layers + i, _cache.to_nnfw_type(),
267 _cache.v[i].data(), _cache.v[i].size()));
268 }
269
270 // Input 2n+1: Cache position - current position in KV cache
271 int64_t cache_pos = _cache.pos();
272 NNFW_ENSURE_STATUS(nnfw_set_input(session, 1 + 2 * _cfg.model.n_layers, NNFW_TYPE_TENSOR_INT64,
273 &cache_pos, sizeof(cache_pos)));
274
275 // Output buffer setup - mode dependent
276 nnfw_tensorinfo ti;
277 NNFW_ENSURE_STATUS(nnfw_output_tensorinfo(session, 0, &ti));
278 if constexpr (ReturnLogits)
279 {
280 // Logits mode: float vector, type validation required
281 if (ti.dtype != NNFW_TYPE_TENSOR_FLOAT32)
282 throw std::runtime_error("decode: output tensor must be float type for logits");
283 output.resize(bufsize_for(&ti) / sizeof(float), 0);
284 NNFW_ENSURE_STATUS(
285 nnfw_set_output(session, 0, ti.dtype, output.data(), output.size() * sizeof(float)));
286 }
287 else
288 {
289 // Hidden state mode: uint8_t vector, no type validation (preserve existing behavior)
290 output.resize(bufsize_for(&ti), 0);
291 NNFW_ENSURE_STATUS(nnfw_set_output(session, 0, ti.dtype, output.data(), output.size()));
292 }
293
294 NNFW_ENSURE_STATUS(nnfw_run(session));
295 nnfw_close_session(session);
296 _cache.advance_pos();
297}

316// Sample token from logits using greedy sampling
317// Input shape: [n_seq, vocab_size], sample from last token
318ggma_token context::sample(const std::vector<float> &logits)
319{
320 if (logits.empty())
321 throw std::runtime_error("Empty logits tensor");
322
323 // Calculate total number of float elements in logits tensor
324 size_t total_elements = logits.size();
325
326 if (total_elements % _cfg.model.vocab_size != 0)
327 throw std::runtime_error("Invalid sequence length in logits tensor");
328
329 const float *last_logits = logits.data() + (total_elements - _cfg.model.vocab_size);
330
331 // Find the token with maximum logit value from the last token's logits
332 const float *max_elem_iter = std::max_element(last_logits, last_logits + _cfg.model.vocab_size);
333
334 return std::distance(last_logits, max_elem_iter);
335}
