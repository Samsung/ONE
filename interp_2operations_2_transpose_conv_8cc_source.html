32void prepareTransposeConv(ExecEnv *env, const ir::Operation &node)
33{
34 const auto ifm_index = node.getInputs().at(ir::operation::TransposeConv::INPUT);
35 const auto ker_index = node.getInputs().at(ir::operation::TransposeConv::KERNEL);
36 const auto ofm_shape_index = node.getInputs().at(ir::operation::TransposeConv::OUTPUT_SHAPE);
37 const auto ofm_index = node.getOutputs().at(0);
38
39 const auto ifm_tensor = env->tensorAt(ifm_index);
40 const auto ker_tensor = env->tensorAt(ker_index);
41 const auto ofm_shape_tensor = env->tensorAt(ofm_shape_index);
42
43 assert(ifm_tensor->getShape().rank() == 4);
44 assert(ker_tensor->getShape().rank() == 4);
45 assert(ofm_shape_tensor->getShape().rank() == 1);
46
47 UNUSED_RELEASE(ifm_tensor);
48 UNUSED_RELEASE(ker_tensor);
49 UNUSED_RELEASE(ofm_shape_tensor);
50
51 const auto output_info = env->graph().operands().at(ofm_index).info();
52 if (output_info.total_size() == 0)
53 {
54 // TODO: Handle unspecified output shape
55 throw std::runtime_error{"Interp(TConv): NYI unspecified output shape"};
56 }
57 else
58 {
59 env->allocateIfNeeded(ofm_index, output_info);
60 }
61
62 auto ofm_tensor = env->tensorAt(ofm_index);
63 UNUSED_RELEASE(ofm_tensor);
64
65 // Handle same ifm & ofm data type only
66 if (ifm_tensor->data_type() != ofm_tensor->data_type())
67 {
68 throw std::runtime_error{"Interp(TConv): Different I/O data dype"};
69 }
70
71 if (ofm_tensor->getShape().rank() != 4)
72 {
73 throw std::runtime_error{"Interp(TConv): Invalid output rank"};
74 }
75}

77void invoke(const ITensor *ifm_tensor, const ITensor *ker_tensor, const ITensor *ofm_tensor,
78 const ir::operation::TransposeConv::Param &param)
79{
80 const auto ifm_shape = ifm_tensor->tensorInfo().shape().asFeature(ir::Layout::NHWC);
81 const auto ofm_shape = ofm_tensor->tensorInfo().shape().asFeature(ir::Layout::NHWC);
82 // Kernel format is [depth_out, kernel_height, kernel_width, depth_in].
83 const auto ker_shape = ker_tensor->tensorInfo().shape();
84 const auto ker_height = ker_shape.dim(1);
85 const auto ker_width = ker_shape.dim(2);
86 const auto padding =
87 ir::calculatePadding(param.padding, ofm_shape, ifm_shape, param.stride, ker_width, ker_height);
88
89 nnfw::cker::TransposeConvParams cker_param;
90 cker_param.padding_values.width = padding.left;
91 cker_param.padding_values.height = padding.top;
92 cker_param.stride_width = param.stride.horizontal;
93 cker_param.stride_height = param.stride.vertical;
94 cker_param.dilation_width_factor = 1;
95 cker_param.dilation_height_factor = 1;
96
97 const auto cker_ifm_shape = convertShape(ifm_tensor->tensorInfo().shape());
98 const auto cker_ker_shape = convertShape(ker_tensor->tensorInfo().shape());
99 const auto cker_ofm_shape = convertShape(ofm_tensor->tensorInfo().shape());
100 const float *ifm_ptr = reinterpret_cast<const float *>(ifm_tensor->bufferRO());
101 const float *ker_ptr = reinterpret_cast<const float *>(ker_tensor->bufferRO());
102 float *ofm_ptr = reinterpret_cast<float *>(ofm_tensor->buffer());
103
104 nnfw::cker::TransposeConv(cker_param, cker_ifm_shape, ifm_ptr, cker_ker_shape, ker_ptr,
105 cker_ofm_shape, ofm_ptr);
106}

108void invokeTransposeConv(const ExecEnv *env, const ir::Operation &node)
109{
110 const auto &tconv_node =
111 nnfw::misc::polymorphic_downcast<const ir::operation::TransposeConv &>(node);
112
113 const auto ifm_index = node.getInputs().at(ir::operation::TransposeConv::INPUT);
114 const auto ker_index = node.getInputs().at(ir::operation::TransposeConv::KERNEL);
115 const auto ofm_index = node.getOutputs().at(0);
116
117 const auto ifm_tensor = env->tensorAt(ifm_index);
118 const auto ker_tensor = env->tensorAt(ker_index);
119 const auto ofm_tensor = env->tensorAt(ofm_index);
120
121 const auto data_type = ifm_tensor->data_type();
122 if (data_type == ir::DataType::FLOAT32)
123 {
124 invoke(ifm_tensor, ker_tensor, ofm_tensor, tconv_node.param());
125 }
126 else
127 {
128 throw std::runtime_error{"Interp(TConv): Support float32 only"};
129 }
130}
