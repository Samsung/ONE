63void BinaryArithmeticLayer::backward()
64{
65 // Calculate gradient for activation
66 if (_back_prop_output->data_type() != OperandType::FLOAT32)
67 throw std::runtime_error{"Unsupported Data Type"};
68
69 const IPortableTensor *backprop_act;
70 try
71 {
72 backprop_act =
73 backpropActivation(_activation, _output, _back_prop_output, _act_back_prop_output.get());
74 }
75 catch (const std::exception &e)
76 {
77 throw std::runtime_error{"train BinaryArithmeticLayer: " + std::string(e.what())};
78 }
79 assert(backprop_act != nullptr);
80
81 nnfw::cker::train::BinaryArithmeticGrad(
82 getShape(_lhs), getBuffer<float>(_lhs), getShape(_rhs), getBuffer<float>(_rhs),
83 getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_back_prop_lhs),
84 getBuffer<float>(_back_prop_lhs), getShape(_back_prop_rhs), getBuffer<float>(_back_prop_rhs),
85 static_cast<nnfw::cker::train::ArithmeticType>(_arithmetic_type));
86}
