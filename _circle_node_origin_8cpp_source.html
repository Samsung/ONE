78class CompositeOrigin final : public luci::CircleNodeOrigin
79{
80public:
81 CompositeOrigin() = delete;
82
83 template <typename T> CompositeOrigin(T origins)
84 {
85 if (origins.size() == 0)
86 throw std::invalid_argument("No origins provided");
87
88 for (auto &origin : origins)
89 {
90 if (origin != nullptr)
91 _origins.emplace_back(origin);
92 }
93 }
94
95public:
96 std::set<const Source *> sources(void) const final
97 {
98 std::set<const Source *> res;
99
100 for (auto &origin : _origins)
101 {
102 for (auto source : origin->sources())
103 {
104 res.emplace(source);
105 }
106 }
107
108 return res;
109 }
110
111private:
112 std::vector<std::shared_ptr<CircleNodeOrigin>> _origins;
113};

154bool has_origin(const luci::CircleNode *circle_node)
155{
156 if (circle_node->annot<CircleNodeOriginAnnotation>() == nullptr)
157 return false;
158
159 assert(!circle_node->annot<CircleNodeOriginAnnotation>()->origin()->sources().empty());
160
161 return true;
162}

169void add_origin(luci::CircleNode *circle_node, const std::shared_ptr<CircleNodeOrigin> origin)
170{
171 // Nothing to add
172 if (origin == nullptr)
173 return;
174
175 auto new_origin = composite_origin({get_origin(circle_node), origin});
176 circle_node->annot<CircleNodeOriginAnnotation>(nullptr);
177 circle_node->annot(std::make_unique<CircleNodeOriginAnnotation>(new_origin));
178}

180const std::shared_ptr<luci::CircleNodeOrigin> get_origin(const luci::CircleNode *circle_node)
181{
182 if (!has_origin(circle_node))
183 return nullptr;
184
185 assert(circle_node->annot<CircleNodeOriginAnnotation>()->origin() != nullptr);
186 return circle_node->annot<CircleNodeOriginAnnotation>()->origin();
