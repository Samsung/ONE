34void SplitV::configure()
35{
36 assert(axis()->shape().num_elements() == 1);
37 _axis_value = getTensorData<int32_t>(axis())[0];
38 if (_axis_value < 0)
39 _axis_value += input()->shape().num_dims();
40 assert(_axis_value >= 0 && _axis_value < input()->shape().num_dims());
41
42 auto num_split = static_cast<int32_t>(_outputs.size());
43 auto sizes_data = getTensorData<int32_t>(size_splits());
44
45 assert(size_splits()->shape().num_dims() == 1);
46
47 int32_t sum = 0;
48 const auto num_dims_size_spits = size_splits()->shape().dim(0);
49 int32_t count_neg_dim = 0;
50
51 for (int32_t i = 0; i < num_dims_size_spits - 1; ++i)
52 {
53 if (sizes_data[i] != -1)
54 {
55 sum += sizes_data[i];
56 }
57 else
58 {
59 count_neg_dim++;
60 }
61 }
62 assert(count_neg_dim < 2);
63 assert(size_splits()->shape().num_elements() == num_split);
64
65 // TODO: enable it only if kernel with dynamic shapes
66 auto output_shape = input()->shape();
67 for (int32_t i = 0; i < num_split; ++i)
68 {
69 if (sizes_data[i] == -1)
70 {
71 output_shape.dim(_axis_value) = input()->shape().dim(_axis_value) - sum;
72 }
73 else
74 {
75 output_shape.dim(_axis_value) = sizes_data[i];
76 }
77 _outputs[i]->resize(output_shape);
78 }
79}

81void SplitV::execute() const
82{
83 tflite::SplitParams params{};
84 params.num_split = _outputs.size();
85 params.axis = _axis_value;
86
87#define TF_LITE_SPLIT(scalar) \
88 { \
89 VectorOfTensors<scalar, false> all_outputs(_outputs); \
90 tflite::optimized_ops::Split(params, getTensorShape(input()), getTensorData<scalar>(input()), \
91 all_outputs.shapes(), all_outputs.data()); \
92 }
93
94 switch (input()->element_type())
95 {
96 case DataType::FLOAT32:
97 TF_LITE_SPLIT(float);
98 break;
99 case DataType::U8:
100 TF_LITE_SPLIT(uint8_t);
101 break;
102 case DataType::S16:
103 TF_LITE_SPLIT(int16_t);
104 break;
105 default:
106 assert(false && "Unsupported type.");
107 }
108#undef TF_LITE_SPLIT
109}
