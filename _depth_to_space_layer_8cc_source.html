50void DepthToSpaceLayer::run()
51{
52 switch (_input->data_type())
53 {
54 case OperandType::FLOAT32:
55 depthToSpace<float>();
56 break;
57 case OperandType::INT32:
58 depthToSpace<int32_t>();
59 break;
60 case OperandType::INT64:
61 depthToSpace<int64_t>();
62 break;
63 case OperandType::QUANT_UINT8_ASYMM:
64 depthToSpace<uint8_t>();
65 break;
66 case OperandType::QUANT_INT8_ASYMM:
67 depthToSpace<int8_t>();
68 break;
69 default:
70 throw std::runtime_error{"DepthToSpace: unsupported data type"};
71 }
