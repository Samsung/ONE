37ir::OperandInfo createBackwardTensorInfo(const ir::Operand &operand)
38{
39 // TODO Use different shape of back-propagated tensor if it exists
40 return ir::OperandInfo{operand.shape(), operand.typeInfo(), operand.info().memAllocType(),
41 operand.isConstant()};
42}

44// NOTE Even if there are duplicate indices, the duplicate back-propagated tensors may need
45// to be updated respectively. So we use a sequence instead of a set.
46ir::OperandIndexSequence getBackPropSeq(const ir::train::TrainableGraph &tgraph,
47 const ir::OperationIndex &op_index)
48{
49 ir::OperandIndexSequence ret;
50
51 const auto &op = tgraph.operations().at(op_index);
52 for (const auto &input : (op.getInputs() | ir::Remove::UNDEFINED))
53 {
54 const auto &operand = tgraph.operands().at(input);
55 // TODO Remove other inputs that are not back-propagated
56 if (!operand.isConstant() && !tgraph.getInputs().contains(input))
57 ret.append(input);
58 }
59
60 return ret;
61}

63void AddBackPropInitializers(const ir::train::TrainableGraph &tgraph, TensorRegistry &tensor_reg,
64 FunctionMap &fn_map)
65{
66 util::Set<ir::OperandIndex> unvisited;
67 tgraph.operands().iterate([&](const ir::OperandIndex &index, const ir::Operand &operand) {
68 // TODO Consider not adding BackPropInitializer if the coresponding BackPropTensors don't
69 // require initilization (i.g. BackPropTensors that are not back-propagated)
70 if (!tgraph.getInputs().contains(index) && !operand.isConstant())
71 unvisited.add(index);
72 });
73
74 for (const auto &op_index : tgraph.btopolSortOperations())
75 {
76 assert(fn_map.find(op_index) != fn_map.end());
77
78 auto &tn_seq = fn_map.at(op_index);
79
80 // The function added lastest is executed first in a sequence during backwarding.
81 std::vector<BackPropTensor *> back_props;
82 const auto &op = tgraph.operations().at(op_index);
83 for (const auto &back_prop_index :
84 op.getInputs() | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED)
85 {
86 if (unvisited.contains(back_prop_index))
87 {
88 auto back_prop_tensor = tensor_reg.getBackPropTensor(back_prop_index);
89 assert(back_prop_tensor != nullptr);
90 back_props.emplace_back(back_prop_tensor);
91 unvisited.remove(back_prop_index);
92 }
93 }
94
95 if (back_props.size() != 0)
96 {
97 auto initializer = std::make_unique<ops::BackPropInitializer>(back_props);
98 tn_seq->append(std::move(initializer));
99 }
100 }
101}

109backend::train::ITensorRegistry *BackendContext::genTrainingTensors()
110{
111 const ir::train::TrainableGraph &tgraph = *trainable_graph();
112 auto tensor_builder = _tensor_builder;
113
114 tgraph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
115 if (external_operands().contains(ind))
116 return;
117 // NOTE Assuming there is no layout changes (Always assume NHWC or UNKNOWN)
118 assert(tgraph.layout() != ir::Layout::NCHW);
119
120 tensor_builder->registerBackwardTensorInfo(ind, createBackwardTensorInfo(obj),
121 ir::Layout::NHWC);
122 });
123
124 // TODO Plan tensor builds to reduce peak memory usage
125 tgraph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &) {
126 if (tensor_builder->isRegisteredBackward(ind))
127 tensor_builder->notifyBackwardFirstUse(ind);
128 });
129
130 for (const auto &op_index : tgraph.btopolSortOperations())
131 {
132 const auto back_prop_seq = getBackPropSeq(tgraph, op_index);
133 for (const auto &back_prop_index : back_prop_seq)
134 {
135 DisposableTensorIndex disposable_index{op_index, back_prop_index};
136 const auto &operand = tgraph.operands().at(back_prop_index);
137 tensor_builder->registerDisposableBackwardTensorInfo(
138 disposable_index, createBackwardTensorInfo(operand), ir::Layout::NHWC);
139 }
140 }
141
142 planDisposableBackPropTensors();
143
144 tensor_builder->allocateBackward();
145
146 return _tensor_registry.get();
147}

149void BackendContext::planDisposableBackPropTensors()
150{
151 const ir::train::TrainableGraph &tgraph = *trainable_graph();
152 auto tensor_builder = _tensor_builder;
153
154 std::vector<DisposableTensorIndex> prev_seq;
155 for (const auto &op_index : tgraph.btopolSortOperations())
156 {
157 for (const auto &index : prev_seq)
158 {
159 tensor_builder->notifyDisposableBackPropLastUse(index);
160 }
161
162 std::vector<DisposableTensorIndex> cur_seq;
163 const auto back_prop_indices = getBackPropSeq(tgraph, op_index);
164 for (const auto &back_prop_index : back_prop_indices)
165 {
166 DisposableTensorIndex cur_index{op_index, back_prop_index};
167 tensor_builder->notifyDisposableBackPropFirstUse(cur_index);
168
169 cur_seq.emplace_back(cur_index);
170 }
171
172 prev_seq = cur_seq;
173 }
174}

176FunctionMap BackendContext::genKernels()
177{
178 auto ret = generateFunctionMap();
179
180 // Initialize TrainableTensors
181 trainable_graph()->operands().iterate(
182 [&](const ir::OperandIndex &ind, const ir::Operand &operand) {
183 if (external_operands().contains(ind) || !operand.isConstant())
184 return;
185
186 auto tensor = tensor_registry()->getNativeITensor(ind);
187 assert(tensor != nullptr);
188
189 VERBOSE(FillOperandData) << "Fill data for " << ind << std::endl;
190
191 auto data = operand.shareData();
192 assert(data && data->base());
193 auto trainable_tensor = dynamic_cast<TrainableTensor *>(tensor);
194
195 if (trainable_tensor == nullptr)
196 throw std::runtime_error{"This tensor is not trainable tensor"};
197
198 trainable_tensor->fillBuffer(data);
199 });
200
201 // NOTE For memory optimization, we want to free some operand data
202 const_cast<ir::train::TrainableGraph &>(*_tdata->tgraph)
203 .operands()
204 .iterate([&](const ir::OperandIndex &, ir::Operand &obj) { obj.releaseData(); });
205
206 // TODO Enable
207 // for (auto &&it : ret)
208 // {
209 // auto &fn_seq = it.second;
210 // fn_seq->iterate([&](exec::IFunction &ifunc) { ifunc.prepare(); });
211 // }
212
213 return ret;
214}

216FunctionMap BackendContext::generateFunctionMap()
217{
218 train::FunctionMap ret;
219
220 for (const auto &op_ind : _tdata->op_order)
221 {
222 auto fn_seq = kernel_gen->generate(op_ind);
223 ret.emplace(op_ind, std::move(fn_seq));
224 }
225
226 // NOTE Each BackPropInitializer should be called first in each op node during backwarding
227 const auto &tgraph = *_tdata->tgraph;
228 auto tensor_reg = nnfw::misc::polymorphic_downcast<TensorRegistry *>(_tensor_registry.get());
229 AddBackPropInitializers(tgraph, *tensor_reg, ret);
230
231 return ret;
232}
