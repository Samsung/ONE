28template <loco::DataType DT>
29bool is_scalar_with_value(luci::CircleConst *node, typename loco::DataTypeImpl<DT>::Type val)
30{
31 if (node->dtype() != DT)
32 return false;
33 if (node->rank() != 0)
34 return false;
35 if (node->size<DT>() != 1)
36 return false;
37 if (node->at<DT>(0) != static_cast<typename loco::DataTypeImpl<DT>::Type>(val))
38 return false;
39
40 return true;
41}

63template <loco::DataType DT> bool transform_min_relu_pattern(luci::CircleRelu *relu)
64{
65 if (not relu)
66 return false;
67
68 if (relu->dtype() != DT)
69 return false;
70
71 auto *mini = dynamic_cast<luci::CircleMinimum *>(relu->features());
72 if (not mini)
73 return false;
74
75 luci::CircleConst *mini_const = nullptr;
76 loco::Node *mini_input = nullptr;
77
78 // There are two ways Miminum takes inputs.
79 // 1. Miminum(x = CircleNode, y = CircleConst)
80 // 2. Miminum(x = CircleConst, y = CircleNode)
81 if (not luci::fill(&mini_const, &mini_input).with_commutative_args_of(mini))
82 return false;
83
84 // Miminum constant should be scalar whose value is 6.
85 if (not is_scalar_with_value<DT>(mini_const,
86 static_cast<typename loco::DataTypeImpl<DT>::Type>(6)))
87 return false;
88
89 auto name = relu->name();
90 assert(name.length() > 0);
91
92 // Create Relu6 op
93 auto relu6 = mini->graph()->nodes()->create<luci::CircleRelu6>();
94 relu6->features(mini_input);
95 relu6->name(name + "/Relu6");
96 luci::add_origin(relu6, luci::composite_origin({luci::get_origin(relu), luci::get_origin(mini)}));
97
98 replace(relu).with(relu6);
99
100 return true;
101}

108bool TransformMinReluToRelu6Pass::run(loco::Graph *g)
109{
110 bool changed = false;
111
112 for (auto node : loco::active_nodes(loco::output_nodes(g)))
113 {
114 if (auto relu = dynamic_cast<luci::CircleRelu *>(node))
115 {
116 if (transform_min_relu_pattern<loco::DataType::FLOAT32>(relu))
117 changed = true;
118 }
119 }
120
121 return changed;
122}
