184std::unique_ptr<CircleQuantParam>
185luci_quantparam(const circle::QuantizationParametersT *quantization)
186{
187 const auto &min = quantization->min;
188 const auto &max = quantization->max;
189 const auto &scale = quantization->scale;
190 const auto &zero_point = quantization->zero_point;
191 const auto &quantized_dimension = quantization->quantized_dimension;
192
193 if ((!min.empty() && !max.empty()) || (!scale.empty() && !zero_point.empty()))
194 {
195 auto quantparam = std::make_unique<CircleQuantParam>();
196
197 quantparam->min = min;
198 quantparam->max = max;
199 quantparam->scale = scale;
200 quantparam->zerop = zero_point;
201 quantparam->quantized_dimension = quantized_dimension;
202
203 return quantparam;
204 }
205
206 return nullptr;
207}

219std::unique_ptr<SparsityParam> luci_sparsityparam(const circle::SparsityParametersT *sparsity)
220{
221 assert(sparsity);
222 const auto &traversal_order = sparsity->traversal_order;
223 const auto &block_map = sparsity->block_map;
224 const auto &dim_metadata = sparsity->dim_metadata;
225
226 // TODO find a condition that should return nullptr
227 auto sparsityparam = std::make_unique<SparsityParam>();
228
229 sparsityparam->traversal_order = traversal_order;
230 sparsityparam->block_map = block_map;
231 for (const auto &dm : dim_metadata)
232 {
233 sparsityparam->dim_metadata.emplace_back(luci_dim_type(dm->format), dm->dense_size,
234 luci_sparse_index_vector(dm->array_segments),
235 luci_sparse_index_vector(dm->array_indices));
236 }
237
238 return sparsityparam;
239}

251void copy_tensor_attributes(const circle::Tensor *tensor, CircleNode *node)
252{
253 assert(tensor != nullptr);
254
255 node->name(tensor_name(tensor));
256 node->dtype(luci_datatype(tensor->type()));
257
258 const auto tensor_shape_signature = wrap(tensor->shape_signature());
259 const auto tensor_shape = wrap(tensor->shape());
260 assert(tensor_shape_signature.size() == 0 ||
261 tensor_shape_signature.size() == tensor_shape.size());
262
263 const auto dims = tensor_shape; // in NHWC
264 node->rank(dims.size());
265 for (uint32_t r = 0; r < dims.size(); ++r)
266 {
267 if (tensor_shape_signature.size() > 0 && tensor_shape_signature.at(r) == -1)
268 node->dim(r).unset();
269 else
270 node->dim(r).set(dims[r]);
271 }
272
273 const auto quantization = tensor->quantization();
274 if (quantization != nullptr)
275 {
276 auto quantparam = luci_quantparam(quantization);
277 if (quantparam)
278 node->quantparam(std::move(quantparam));
279 }
280
281 const auto sparsity = tensor->sparsity();
282 if (sparsity != nullptr)
283 {
284 auto sparsityparam = luci_sparsityparam(sparsity);
285 if (sparsityparam)
286 node->sparsityparam(std::move(sparsityparam));
287 }
288}
