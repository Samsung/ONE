185bool TaggedShapeAnalyzer::analyze_reshape(const luci::CircleReshape *reshape_node)
186{
187 // At least one element must be in reshape's output-tensor.
188 if (reshape_node->rank() <= 0)
189 return false;
190
191 // Create new_shape based on reshape_node
192 TagShape new_shape;
193 for (uint32_t i = 0; i < reshape_node->rank(); i++)
194 {
195 if (not reshape_node->dim(i).known())
196 return false;
197
198 TagDim dim;
199 dim.value = reshape_node->dim(i).value();
200 new_shape.push_back(dim);
201 }
202
203 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
204 uint32_t old_shape_start_idx = 0;
205 uint32_t old_shape_end_idx = 1;
206 auto old_shape_product = _shape[old_shape_start_idx].value;
207
208 auto expand_range = [&]() -> bool {
209 if (old_shape_end_idx >= _shape.size())
210 return false;
211
212 old_shape_product *= _shape[old_shape_end_idx].value;
213 old_shape_end_idx++;
214 return true;
215 };
216
217 auto move_to_next_range = [&]() -> bool {
218 if (old_shape_end_idx >= _shape.size())
219 return false;
220
221 old_shape_start_idx = old_shape_end_idx;
222 old_shape_end_idx++;
223 old_shape_product = _shape[old_shape_start_idx].value;
224 return true;
225 };
226
227 // Add tags from '_shape' to the 'new_shape'
228 uint32_t new_shape_idx = 0;
229 while (new_shape_idx < new_shape.size())
230 {
231 TagDim &target_dim = new_shape[new_shape_idx];
232
233 // Ignore dim == 1
234 if (target_dim.value == 1)
235 {
236 new_shape_idx++;
237 continue;
238 }
239
240 while (old_shape_product < target_dim.value)
241 {
242 if (expand_range() == false)
243 break;
244 }
245
246 if (old_shape_product != target_dim.value)
247 return false;
248
249 assert(old_shape_product == target_dim.value);
250 for (uint32_t idx = old_shape_start_idx; idx < old_shape_end_idx; idx++)
251 {
252 const auto &old_tags = _shape[idx].tags;
253 target_dim.tags.insert(target_dim.tags.end(), old_tags.begin(), old_tags.end());
254 }
255
256 new_shape_idx++;
257 move_to_next_range();
258 }
259 _shape = new_shape;
260 return true;
261}

436bool remove_unnecessary_transpose(luci::CircleTranspose *node)
437{
438 // find 'front_transpose - mid_reshape - back_transpose' pattern
439 const auto back_transpose = node;
440 const auto mid_reshape = dynamic_cast<luci::CircleReshape *>(back_transpose->a());
441 {
442 if (mid_reshape == nullptr)
443 return false;
444 }
445 const auto front_transpose = dynamic_cast<luci::CircleTranspose *>(mid_reshape->tensor());
446 {
447 if (not front_transpose)
448 return false;
449 }
450
451 TaggedShapeAnalyzer analyzer;
452
453 if (not analyzer.init(front_transpose, mid_reshape, back_transpose))
454 return false;
455
456 if (not analyzer.can_remove_transposes<loco::DataType::S32>())
457 return false;
458
459 // repalce with new_node
460 luci::CircleReshape *new_node = create_reshape_node<loco::DataType::S32>(
461 node->graph(), front_transpose, mid_reshape, back_transpose);
462
463 replace(node).with(new_node);
464
465 return true;
