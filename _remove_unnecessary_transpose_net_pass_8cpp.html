205bool TaggedShapeAnalyzer::analyze_reshape(const std::vector<int32_t> &new_shape)
206{
207 if (new_shape.size() <= 0)
208 return false;
209
210 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
211 uint32_t old_shape_start_idx = 0;
212 uint32_t old_shape_end_idx = 1;
213 auto old_shape_product = _shape[old_shape_start_idx].value;
214
215 auto expand_range = [&]() -> bool {
216 if (old_shape_end_idx >= _shape.size())
217 return false;
218
219 old_shape_product *= _shape[old_shape_end_idx].value;
220 old_shape_end_idx++;

225 if (old_shape_end_idx >= _shape.size())
226 return false;
227

229 old_shape_end_idx++;
230 old_shape_product = _shape[old_shape_start_idx].value;
231 return true;

241

245 new_tagged_shape.emplace_back(1);
246 target_idx++;
247 continue;
248 }

266 }
