28template <typename T> class ExplicitDataChef final : public DataChef
29{
30public:
31 ExplicitDataChef()
32 {
33 // DO NOTHING
34 }
35
36public:
37 std::vector<uint8_t> generate(int32_t count) const override
38 {
39 std::vector<uint8_t> res;
40
41 for (uint32_t n = 0; n < count; ++n)
42 {
43 T const value = (n < _values.size()) ? _values.at(n) : T{};
44 const uint8_t *arr = reinterpret_cast<const uint8_t *>(&value);
45
46 for (uint32_t b = 0; b < sizeof(T); ++b)
47 {
48 res.emplace_back(arr[b]);
49 }
50 }
51
52 return res;
53 }
54
55public:
56 void insert(const T &value) { _values.emplace_back(value); }
57
58private:
59 std::vector<T> _values;
60};
