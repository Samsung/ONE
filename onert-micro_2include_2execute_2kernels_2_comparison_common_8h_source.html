42template <typename T> void evalComparisonGeneric(OMRuntimeKernel *runtime_kernel, bool F(T, T))
43{
44 const circle::Tensor *input1 = nullptr;
45 const circle::Tensor *input2 = nullptr;
46 const circle::Tensor *output = nullptr;
47
48 uint8_t *input_data1 = nullptr;
49 uint8_t *input_data2 = nullptr;
50 uint8_t *output_data = nullptr;
51
52 input1 = runtime_kernel->inputs[input1TensorIdx];
53 input2 = runtime_kernel->inputs[input2TensorIdx];
54 output = runtime_kernel->outputs[outputTensorIdx];
55
56 assert(input1 != nullptr);
57 assert(input2 != nullptr);
58 assert(output != nullptr);
59
60 input_data1 = runtime_kernel->inputs_data[input1TensorIdx];
61 input_data2 = runtime_kernel->inputs_data[input2TensorIdx];
62 output_data = runtime_kernel->outputs_data[outputTensorIdx];
63
64 assert(input_data1 != nullptr);
65 assert(input_data2 != nullptr);
66 assert(output_data != nullptr);
67
68 core::OMRuntimeShape input1_shape(input1);
69 core::OMRuntimeShape input2_shape(input2);
70 core::OMRuntimeShape output_shape(output);
71
72 auto cast_input_data1 = core::utils::castInputData<T>(input_data1);
73
74 assert(cast_input_data1 != nullptr);
75
76 auto cast_input_data2 = core::utils::castInputData<T>(input_data2);
77
78 assert(cast_input_data2 != nullptr);
79
80 auto cast_output_data = core::utils::castOutputData<bool>(output_data);
81
82 onert_micro::core::ComparisonParams op_params;
83 op_params.is_broadcast = input1_shape.flatSize() != input2_shape.flatSize();
84
85 if (op_params.is_broadcast)
86 {
87 onert_micro::execute::pal::BroadcastComparison4DSlowNoScaling<T>(
88 op_params, input1_shape, cast_input_data1, input2_shape, cast_input_data2, output_shape,
89 cast_output_data, F);
90 }
91 else
92 {
93 const int64_t flat_size = input1_shape.flatSize();
94 onert_micro::execute::pal::ComparisonNoScaling<T>(flat_size, cast_input_data1, cast_input_data2,
95 cast_output_data, F);
96 }
97}
