54static inline void SetupScratchpadTensor(
55 const luci_interpreter::DataType &input_data_type,
56 const luci_interpreter::DataType &weight_feature_data_type,
57 luci_interpreter::Tensor *scratchpad_1, luci_interpreter::Tensor *scratchpad_2,
58 luci_interpreter::Tensor *scratchpad_3, luci_interpreter::Tensor *scratchpad_4,
59 luci_interpreter::Tensor *scratchpad_5, luci_interpreter::Tensor *scratchpad_6,
60 const luci_interpreter::Shape input_shape, const luci_interpreter::Shape weight_time_shape,
61 const int32_t batch_size, const int32_t num_filters, const int32_t num_units)
62{
63
64 if (input_data_type == luci_interpreter::DataType::FLOAT32 &&
65 (weight_feature_data_type == luci_interpreter::DataType::S8 ||
66 weight_feature_data_type == luci_interpreter::DataType::U8))
67 {
68 (void)input_shape;
69 (void)weight_time_shape;
70 (void)scratchpad_3;
71 (void)scratchpad_4;
72 (void)scratchpad_5;
73 (void)scratchpad_6;
74
75 assert(false && "Hybrid type is not currently supported for linux platform");
76 }
77
78 // Resize scratchpad_1 tensor
79 scratchpad_1->resize({batch_size, num_filters});
80
81 if (input_data_type == luci_interpreter::DataType::S8)
82 {
83 // Resize scratchpad_2 for full_integer op
84 scratchpad_2->resize({batch_size, num_units});
85 }
86}
