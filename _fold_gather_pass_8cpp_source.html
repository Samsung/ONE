42template <loco::DataType InputT, loco::DataType IndexT>
43bool fold_gather(luci::CircleGather *gather_node)
44{
45 const auto params = loco::must_cast<luci::CircleConst *>(gather_node->params());
46 const auto indices = loco::must_cast<luci::CircleConst *>(gather_node->indices());
47
48 const auto rank = params->rank();
49 auto axis = gather_node->axis();
50 if (axis < 0)
51 {
52 axis += static_cast<int32_t>(rank);
53 }
54
55 if (axis < 0 or axis >= static_cast<int32_t>(rank))
56 throw std::runtime_error("Unsupported axis value");
57
58 const auto name = gather_node->name();
59 assert(name.length() > 0);
60
61 auto constant = gather_node->graph()->nodes()->create<luci::CircleConst>();
62 constant->dtype(InputT);
63 constant->name(name + "_folded");
64
65 constant->rank(rank + indices->rank() - 1);
66
67 assert(constant->rank() > 0);
68
69 std::vector<uint32_t> shape;
70 for (uint32_t i = 0; i < rank; ++i)
71 {
72 if (i != static_cast<uint32_t>(axis))
73 {
74 const auto dim = params->dim(i).value();
75 shape.push_back(dim);
76 }
77 else
78 {
79 for (uint32_t j = 0; j < indices->rank(); ++j)
80 {
81 const auto dim = indices->dim(j).value();
82 shape.push_back(dim);
83 }
84 }
85 }
86
87 uint32_t size = 1;
88 for (uint32_t i = 0; i < shape.size(); ++i)
89 {
90 constant->dim(i).set(shape.at(i));
91 size *= shape.at(i);
92 }
93
94 constant->size<InputT>(size);
95
96 uint32_t outer_size = 1;
97 for (uint32_t i = 0; i < static_cast<uint32_t>(axis); ++i)
98 {
99 outer_size *= params->dim(i).value();
100 }
101
102 uint32_t inner_size = 1;
103 for (uint32_t i = axis + 1; i < rank; ++i)
104 {
105 inner_size *= params->dim(i).value();
106 }
107
108 uint32_t coord_size = 1;
109 for (uint32_t i = 0; i < indices->rank(); ++i)
110 {
111 coord_size *= indices->dim(i).value();
112 }
113
114 const auto axis_size = params->dim(axis).value();
115
116 for (uint32_t outer = 0; outer < outer_size; ++outer)
117 {
118 for (uint32_t i = 0; i < coord_size; ++i)
119 {
120 constant->at<InputT>((outer * coord_size + i) * inner_size) =
121 params->at<InputT>((outer * axis_size + indices->at<IndexT>(i)) * inner_size);
122 }
123 }
124 loco::replace(gather_node).with(constant);
125
126 return true;
127}

129bool fold_gather(luci::CircleGather *gather_node)
130{
131 const auto params = dynamic_cast<luci::CircleConst *>(gather_node->params());
132 if (not params)
133 return false;
134
135 const auto indices = dynamic_cast<luci::CircleConst *>(gather_node->indices());
136 if (not indices)
137 return false;
138
139 // TODO: support more types
140 if (params->dtype() != loco::DataType::S32 and params->dtype() != loco::DataType::S64)
141 return false;
142
143 if (indices->dtype() != loco::DataType::S32 and indices->dtype() != loco::DataType::S64)
144 throw std::runtime_error("Unsupported type");
145
146 if (params->dtype() == loco::DataType::S64)
147 {
148 if (indices->dtype() == loco::DataType::S64)
149 return fold_gather<loco::DataType::S64, loco::DataType::S64>(gather_node);
150 else
151 return fold_gather<loco::DataType::S64, loco::DataType::S32>(gather_node);
152 }
153 else
154 {
155 if (indices->dtype() == loco::DataType::S64)
156 return fold_gather<loco::DataType::S32, loco::DataType::S64>(gather_node);
157 else
158 return fold_gather<loco::DataType::S32, loco::DataType::S32>(gather_node);
159 }
160}

170bool FoldGatherPass::run(loco::Graph *g)
171{
172 bool changed = false;
173 for (auto node : loco::active_nodes(loco::output_nodes(g)))
174 {
175 if (auto gather_node = dynamic_cast<luci::CircleGather *>(node))
176 {
177 if (fold_gather(gather_node))
178 changed = true;
179 }
180 }
181
182 return changed;
183}
