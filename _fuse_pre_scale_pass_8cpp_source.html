31// Fuse CircleCustom(PreScale) + Op
32struct FusePreScale final : public luci::CircleNodeMutableVisitor<bool>
33{
34 bool visit(luci::CircleNode *) { return false; }
35
36 bool visit(luci::CircleConv2D *node)
37 {
38 auto pre_scale = to_scale(node->input());
39 if (not pre_scale)
40 return false;
41
42 auto param = loco::must_cast<luci::CircleConst *>(pre_scale->inputs(1)); // FIX_PreScale_UNLESS
43 auto filter = loco::must_cast<luci::CircleConst *>(node->filter());
44
45 uint32_t filter_o = filter->dim(0).value();
46 uint32_t filter_h = filter->dim(1).value();
47 uint32_t filter_w = filter->dim(2).value();
48 uint32_t filter_i = filter->dim(3).value();
49
50 if (filter_i != param->size<loco::DataType::FLOAT32>())
51 {
52 throw std::runtime_error(
53 "Mismatch between scale size and filter input channel size: " + std::to_string(filter_i) +
54 " != " + std::to_string(param->size<loco::DataType::FLOAT32>()));
55 }
56
57 auto fused_filter = luci::clone(filter);
58 fused_filter->name(filter->name() + "_fused");
59 add_origin(fused_filter, luci::get_origin(filter));
60
61 // Multiply param to weights
62 for (uint32_t c = 0; c < filter_o; c++)
63 {
64 for (uint32_t h = 0; h < filter_h; h++)
65 {
66 for (uint32_t w = 0; w < filter_w; w++)
67 {
68 for (uint32_t b = 0; b < filter_i; b++)
69 {
70 uint32_t offset =
71 c * filter_h * filter_w * filter_i + h * filter_w * filter_i + w * filter_i + b;
72 float scale = param->at<loco::DataType::FLOAT32>(b);
73 assert(scale > 0.0); // Defensive guard
74
75 fused_filter->at<loco::DataType::FLOAT32>(offset) =
76 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
77 }
78 }
79 }
80 }
81
82 node->input(pre_scale->inputs(0));
83 node->filter(fused_filter);
84
85 return true;
86 }
87
88 bool visit(luci::CircleDepthwiseConv2D *node)
89 {
90 auto pre_scale = to_scale(node->input());
91 if (not pre_scale)
92 return false;
93
94 auto param = loco::must_cast<luci::CircleConst *>(pre_scale->inputs(1)); // FIX_PreScale_UNLESS
95 auto filter = loco::must_cast<luci::CircleConst *>(node->filter());
96 auto bias = loco::must_cast<luci::CircleConst *>(node->bias());
97
98 auto depth_multiplier = node->depthMultiplier();
99 auto in_channel = pre_scale->dim(3).value();
100 auto out_channel = node->dim(3).value();
101
102 if (in_channel != param->size<loco::DataType::FLOAT32>())
103 {
104 throw std::runtime_error(
105 "Mismatch between scale size and filter input channel size: " + std::to_string(in_channel) +
106 " != " + std::to_string(param->size<loco::DataType::FLOAT32>()));
107 }
108
109 assert(in_channel * depth_multiplier == out_channel); // FIX_ME_UNLESS
110
111 uint32_t filter_i = filter->dim(0).value();
112 uint32_t filter_h = filter->dim(1).value();
113 uint32_t filter_w = filter->dim(2).value();
114 uint32_t filter_o = filter->dim(3).value();
115
116 assert(filter_i == 1); // FIX_ME_UNLESS
117 assert(filter_o == out_channel); // FIX_ME_UNLESS
118 assert(filter_o == bias->size<loco::DataType::FLOAT32>()); // FIX_ME_UNLESS
119
120 auto fused_filter = luci::clone(filter);
121 fused_filter->name(filter->name() + "_prescale");
122 add_origin(fused_filter, luci::get_origin(filter));
123
124 // Multiply param to weights
125 for (uint32_t b = 0; b < filter_i; b++)
126 {
127 for (uint32_t h = 0; h < filter_h; h++)
128 {
129 for (uint32_t w = 0; w < filter_w; w++)
130 {
131 for (uint32_t c = 0; c < filter_o; c++)
132 {
133 uint32_t offset =
134 b * filter_h * filter_w * filter_o + h * filter_w * filter_o + w * filter_o + c;
135 float scale = param->at<loco::DataType::FLOAT32>(c % in_channel);
136 assert(scale > 0.0); // Defensive guard
137
138 fused_filter->at<loco::DataType::FLOAT32>(offset) =
139 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
140 }
141 }
142 }
143 }
144
145 node->input(pre_scale->inputs(0));
146 node->filter(fused_filter);
147
148 return true;
149 }

252bool FusePreScalePass::run(loco::Graph *g)
253{
254 bool changed = false;
255 for (auto node : loco::active_nodes(loco::output_nodes(g)))
256 {
257 FusePreScale fps;
258 auto cnode = loco::must_cast<luci::CircleNode *>(node);
259 if (cnode->accept(&fps))
260 changed = true;
261 }
262 return changed;
263}
