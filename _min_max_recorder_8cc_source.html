35std::pair<float, float> minmaxFrom(const backend::ITensor *tensor)
36{
37 const auto data = reinterpret_cast<float *>(tensor->buffer());
38 const auto num_elements = tensor->total_size() / sizeof(float);
39
40 float max = std::numeric_limits<float>::lowest();
41 float min = std::numeric_limits<float>::max();
42
43 bool all_nan = true;
44 for (size_t i = 0; i < num_elements; ++i)
45 {
46 const float number = data[i];
47 if (std::isnan(number))
48 continue;
49
50 if (number == std::numeric_limits<float>::lowest())
51 continue;
52
53 all_nan = false;
54
55 if (number > max)
56 max = number;
57
58 if (number < min)
59 min = number;
60 }
61
62 if (all_nan)
63 throw std::runtime_error("All values are NaN(Not a Number)");
64
65 return {min, max};
66}

68void MinMaxRecorder::handleJobEnd(IExecutor *, ir::SubgraphIndex subg_idx,
69 ir::OperationIndex op_idx, const backend::Backend *backend)
70{
71 const auto &tensor_reg = _backend_contexts.at(backend)->tensor_registry;
72 const auto &op = _graph.operations().at(op_idx);
73 const auto &outputs = op.getOutputs();
74 // TODO: Support multiple output
75 if (outputs.size() != 1)
76 throw std::runtime_error("Only 1 output operator is supported for recording minmax.");
77
78 auto tensor = tensor_reg->getITensor(outputs.at(0));
79
80 // Logic copied from MinMaxObserver.cpp.
81
82 // Filter Ops
83 if (tensor->is_constant())
84 return;
85
86 if (tensor->data_type() != ir::DataType::FLOAT32)
87 return;
88
89 switch (op.opcode())
90 {
91 // Operators with multiple outputs
92 case ir::OpCode::If:
93 case ir::OpCode::Split:
94 case ir::OpCode::SplitV:
95 case ir::OpCode::TopKV2:
96 case ir::OpCode::Unpack:
97 case ir::OpCode::While:
98 return;
99 // NOTE: Sin, Cos, Tanh's output is in [-1, 1]
100 // We may not need to dump those operators.
101 default:; // Do Nothing
102 }
103
104 // Otherwise, dump!
105 assert(tensor->data_type() == ir::DataType::FLOAT32);
106 auto minmax = minmaxFrom(tensor);
107 _op_minmax.append({subg_idx, op_idx}, minmax.first, minmax.second);
108}

110void MinMaxRecorder::handleSubgraphBegin(ir::SubgraphIndex subg_idx)
111{
112 // Make sure there is only cpu backend except for builtin backend
113 std::set<std::string> backend_names;
114 backend::ITensorRegistry *tensor_reg = nullptr;
115 for (const auto &pair : _backend_contexts)
116 {
117 backend_names.insert(pair.first->config()->id());
118 if (pair.first->config()->id() == "cpu")
119 {
120 tensor_reg = pair.second->tensor_registry.get();
121 }
122 }
123 if (backend_names != std::set<std::string>{"builtin", "cpu"})
124 throw std::runtime_error("MinMaxRecorder must have cpu backend only.");
125
126 const auto &inputs = _graph.getInputs(); //.at(op_idx);
127 for (uint32_t i = 0; i < inputs.size(); ++i)
128 {
129 auto input_idx = inputs.at(i);
130 auto tensor = tensor_reg->getITensor(input_idx);
131
132 if (tensor->is_constant())
133 return;
134 if (tensor->data_type() != ir::DataType::FLOAT32)
135 return;
136
137 auto minmax = minmaxFrom(tensor);
138 _input_minmax.append({subg_idx, ir::IOIndex{i}}, minmax.first, minmax.second);
139 }
140}
