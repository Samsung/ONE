36// Naive implementation of transpose for floats. Could be optimized to be more
37// cache friendly, but for now it's a one-time cost on first run, and we would
38// prefer to remove the need to do this at all eventually.
39inline void TransposeFloatTensor(const float *input_data, const nnfw::cker::Shape &output_shape,
40 float *output_data)
41{
42 const int rows = output_shape.Dims(1);
43 const int cols = output_shape.Dims(0);
44 for (int i = 0; i < rows; ++i)
45 {
46 for (int j = 0; j < cols; ++j)
47 {
48 const float in_value = input_data[i * cols + j];
49 output_data[j * rows + i] = in_value;
50 }
51 }
52}

60 void prepareF32(const Shape &filter_shape, const float *filter_data, PaddingType padding_type,
61 bool &is_replaced_weights, uint32_t dilationWidthFactor,
62 uint32_t dilationHeightFactor)
63 {
64 if (!_prepared)
65 {
66 if (usableMultiThreaded(padding_type, dilationWidthFactor, dilationHeightFactor))
67 {
68 transposeFilter(filter_shape, filter_data, is_replaced_weights);
69 }
70 _prepared = true;
71 }

74 void prepareQ8uPerTensor(const Shape &input_shape, const Shape &kernel_shape,
75 const Shape &output_shape, uint32_t stride_width, uint32_t stride_height,

86 void operator()(const ConvParams &params, const Shape &input_shape, const float *input_data,
87 const Shape &filter_shape, const float *filter_data, const Shape &bias_shape,
88 const float *bias_data, const Shape &output_shape, float *output_data)
89 {
90 if (usableMultiThreaded(params.padding_type, params.dilation_width_factor,
91 params.dilation_height_factor))
92 {
93 bool transposed_in_execution = false;
94 if (!_prepared)
95 {
96 // This means that filter is not constant
97 // TODO Apply optimized kernel if multithreaded kernel is slower than optimized kernel by
98 // transposing filter data
99 transposeFilter(filter_shape, filter_data, transposed_in_execution);
100 }
101 multithreaded::Conv(params, input_shape, input_data, filter_shape, &_modified_filter_data[0],
102 bias_shape, bias_data, output_shape, output_data);
103 }
104 else
105 {
106 // TODO Support optimized kernel
107 reference::Conv(params, input_shape, input_data, filter_shape, filter_data, bias_shape,
108 bias_data, output_shape, output_data);
109 }
110 }

112 void operator()(const ConvParams &params, const Shape &input_shape, const uint8_t *input_data,
113 const Shape &filter_shape, const uint8_t *filter_data, const Shape &bias_shape,
114 const int32_t *bias_data, const Shape &output_shape, uint8_t *output_data)
115 {
116 if (!_prepared)
117 {
118 // This means that input or output are dynamic or filter is not constant
119 IsRequiredIm2col(input_shape, filter_shape, output_shape, params.stride_width,
120 params.stride_height, params.dilation_width_factor,
121 params.dilation_height_factor);
122 }
123
124 int im2col_size = _need_im2col ? _im2col_shape.FlatSize() : 1;
125
126 // Use heap if size is larger than 8MB
127 if (im2col_size > 8 * 1024 * 1024)
128 {
129 std::unique_ptr<uint8_t[]> im2col_data = std::make_unique<uint8_t[]>(im2col_size);
130 optimized::Conv(params, input_shape, input_data, filter_shape, filter_data, bias_shape,
131 bias_data, output_shape, output_data, _im2col_shape, im2col_data.get());
132 }
133 else
134 {
135 uint8_t im2col_data[im2col_size];
136 optimized::Conv(params, input_shape, input_data, filter_shape, filter_data, bias_shape,
137 bias_data, output_shape, output_data, _im2col_shape, im2col_data);
138 }
139 }

164private:
165 bool usableMultiThreaded(PaddingType padding_type, uint32_t dilation_width_factor,
166 int32_t dilation_height_factor)
167 {
168 return padding_type != PaddingType::kNone && std::thread::hardware_concurrency() > 1 &&
169 dilation_width_factor == 1 && dilation_height_factor == 1;
170 }

172 void transposeFilter(const Shape &filter_shape, const float *filter_data,
173 bool &is_replaced_weights)
174 {
175 const auto output_depth = filter_shape.Dims(0);
176 const Shape hwcn_filter_shape{filter_shape.FlatSize() / output_depth, output_depth};
177 _modified_filter_data.resize(hwcn_filter_shape.FlatSize());
178 TransposeFloatTensor(filter_data, hwcn_filter_shape, &_modified_filter_data[0]);
179 is_replaced_weights = true;
180 }
181
182 void IsRequiredIm2col(const Shape &input_shape, const Shape &kernel_shape,
183 const Shape &output_shape, uint32_t stride_width, uint32_t stride_height,
184 uint32_t dilation_width_factor, uint32_t dilation_height_factor)
185 {
186 const bool need_dilated_im2col = dilation_width_factor != 1 || dilation_height_factor != 1;
187 const bool need_non_dilated_im2col = stride_width != 1 || stride_height != 1 ||
188 kernel_shape.Dims(1) != 1 || kernel_shape.Dims(2) != 1;
189
190 _need_im2col = need_dilated_im2col || need_non_dilated_im2col;
191
192 if (_need_im2col)
193 {
194 _im2col_shape.SetDim(0, output_shape.Dims(0));
195 _im2col_shape.SetDim(1, output_shape.Dims(1));
196 _im2col_shape.SetDim(2, output_shape.Dims(2));
197 _im2col_shape.SetDim(3, input_shape.Dims(3) * kernel_shape.Dims(1) * kernel_shape.Dims(2));
198 }
199 }
200
201private:
202 std::vector<float> _modified_filter_data;
203 Shape _im2col_shape;
204 bool _need_im2col;
205 bool _prepared;
206 // Per channel output multiplier and shift.
207 std::vector<int32_t> _per_channel_output_multiplier;
208 std::vector<int> _per_channel_output_shift;
209};
