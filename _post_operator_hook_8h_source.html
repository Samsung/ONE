36// Invoke a user-written Python hook after an operator is executed
37class PostOperatorHook final : public luci::CircleNodeVisitor<void>
38{
39
40// This macro creates three variables used for post-operator hooks.
41// 1. hook: Python function to be invoked (type: py::object)
42// 2. inputs: input data (type: std::vector of numpy array)
43// 3. output: output data (type: numpy array)
44#define POST_OPERATOR_HOOK_PROLOGUE(OP_NAME) \
45 if (!py::hasattr(_analysis, #OP_NAME "Post")) \
46 { \
47 visit(loco::must_cast<const luci::CircleNode *>(node)); \
48 return; \
49 } \
50 py::object hook = _analysis.attr(#OP_NAME "Post"); \
51 auto inputs = inputsPyArray(node, _interpreter); \
52 auto output = outputPyArray(node, _interpreter);
53
54// Multi-output version of POST_OPERATOR_HOOK_PROLOGUE
55#define POST_OPERATOR_HOOK_PROLOGUE_MULTI_OUTS(OP_NAME) \
56 if (!py::hasattr(_analysis, #OP_NAME "Post")) \
57 { \
58 visit(loco::must_cast<const luci::CircleNode *>(node)); \
59 return; \
60 } \
61 py::object hook = _analysis.attr(#OP_NAME "Post"); \
62 auto inputs = inputsPyArray(node, _interpreter); \
63 auto outputs = outputsPyArray(node, _interpreter);
64
65private:
66 py::object _analysis;
67 luci_interpreter::Interpreter *_interpreter{nullptr};
68
69public:
70 explicit PostOperatorHook(py::object analysis, luci_interpreter::Interpreter *interpreter)
71 : _analysis(analysis), _interpreter(interpreter)
72 {
73 // Do nothing
74 }
75
76 // default
77 void visit(const luci::CircleNode *node)
78 {
79 if (not py::hasattr(_analysis, "DefaultOpPost"))
80 return;
81
82 py::object hook = _analysis.attr("DefaultOpPost");
83 auto inputs = inputsPyArray(node, _interpreter);
84 auto output = outputPyArray(node, _interpreter);
85
86 py::list input_list;
87 for (uint32_t i = 0; i < inputs.size(); i++)
88 {
89 input_list.append(inputs[i]);
90 }
91
92 pySafeCall(hook,
93 node->name(), // name
94 toString(node->opcode()), // opcode
95 input_list, // list of inputs
96 output // output
97 );
98 }
99
100 void visit(const luci::CircleConv2D *node)
101 {
102 POST_OPERATOR_HOOK_PROLOGUE(Conv2D)
103
104 auto padding = node->padding();
105 auto stride = node->stride();
106 auto dilation = node->dilation();
107
108 auto py_stride = py::dict("w"_a = stride->w(), "h"_a = stride->h());
109 auto py_dilation = py::dict("w"_a = dilation->w(), "h"_a = dilation->h());
110
111 auto fused_act = node->fusedActivationFunction();
112
113 pySafeCall(hook,
114 node->name(), // name
115 inputs[0], // input
116 inputs[1], // filter
117 inputs[2], // bias
118 padding == luci::Padding::SAME ? "SAME" : "VALID", // padding
119 py_stride, // stride
120 py_dilation, // dilation
121 output, // output
122 toString(fused_act) // fused activation
123 );
124 }

185 void visit(const luci::CircleTransposeConv *node)
186 {
187 POST_OPERATOR_HOOK_PROLOGUE(TransposeConv)
188
189 auto padding = node->padding();
190 auto stride = node->stride();
191
192 auto py_stride = py::dict("w"_a = stride->w(), "h"_a = stride->h());
193
194 pySafeCall(hook,
195 node->name(), // name
196 inputs[2], // input
197 inputs[1], // filter
198 inputs[0], // output shape
199 inputs.size() == 4 ? inputs[3] : none(), // bias
200 padding == luci::Padding::SAME ? "SAME" : "VALID", // padding
201 py_stride, // stride
202 output // output
203 );
204 }

206 void visit(const luci::CircleInstanceNorm *node)
207 {
208 POST_OPERATOR_HOOK_PROLOGUE(InstanceNorm)
209
210 auto epsilon = node->epsilon();
211
212 auto fused_act = node->fusedActivationFunction();
213
214 pySafeCall(hook,
215 node->name(), // name
216 inputs[0], // input
217 inputs[1], // gamma
218 inputs[2], // beta
219 epsilon, // epsilon
220 output, // output
221 toString(fused_act) // fused activation
222 );
223 }

225 void visit(const luci::CircleSplit *node)
226 {
227 POST_OPERATOR_HOOK_PROLOGUE_MULTI_OUTS(Split)
228
229 py::list output_list;
230 for (uint32_t i = 0; i < outputs.size(); i++)
231 {
232 output_list.append(outputs[i]);
233 }
234
235 auto num_split = node->num_split();
236
237 pySafeCall(hook,
238 node->name(), // name
239 inputs[0], // split_dim
240 inputs[1], // input
241 num_split, // num_split
242 output_list // list of outputs
243 );
244 }

246#undef POST_OPERATOR_HOOK_PROLOGUE_MULTI_OUTS
247};
