66void loadModelFromBinaryFile(const std::string &filename, onnx::ModelProto *model)
67{
68 GOOGLE_PROTOBUF_VERIFY_VERSION;
69
70 int file_handle = open(filename.c_str(), O_RDONLY);
71
72 if (file_handle == -1)
73 throw std::runtime_error("Couldn't open file \"" + filename + "\": " + std::strerror(errno) +
74 ".");
75
76 google::protobuf::io::FileInputStream file_stream(file_handle);
77 file_stream.SetCloseOnDelete(true);
78
79 google::protobuf::io::CodedInputStream coded_stream(&file_stream);
80 coded_stream.SetTotalBytesLimit(INT_MAX, INT_MAX);
81
82 if (!model->ParseFromCodedStream(&coded_stream))
83 throw std::runtime_error("Couldn't parse file \"" + filename + "\".");
84
85 // If the file has not been consumed entirely, assume that the file is in the wrong format.
86 if (!coded_stream.ConsumedEntireMessage())
87 throw std::runtime_error("File \"" + filename + "\" has not been consumed entirely.");
88}

102

104 throw std::runtime_error("Couldn't parse file \"" + filename + "\".");

125void ONNXImporterImpl::collectUnsupportedOps()
126{
127 std::set<std::pair<std::string, int64_t>> problems_op_set;
128
129 for (int i = 0; i < _model->graph().node_size(); i++)
130 {
131 const auto &onnx_node = _model->graph().node(i);
132 assert(onnx_node.has_op_type());
133 const auto &op_type = onnx_node.op_type();
134 auto opset = _modelCtx->getDomainOpsetVersion(onnx_node.domain());
135
136 NodeConverterRegistry::ConverterFunc converter =
137 NodeConverterRegistry::getInstance().lookup(op_type, opset);
138
139 if (converter == nullptr)
140 problems_op_set.emplace(op_type, opset);
141 }
142 if (!problems_op_set.empty())
143 {
144 std::cerr << "The following operators are not supported:\n";
145 for (const auto &op : problems_op_set)
146 std::cerr << op.first << " opset " << op.second << std::endl;
147 throw std::runtime_error("Unsupported operators found");
148 }
149}

151void ONNXImporterImpl::createGraphInputs()
152{
153 const auto &graph = _model->graph();
154 const auto &initializer = graph.initializer();
155
156 // Create all initializer Tensors
157 for (const auto &tensor : initializer)
158 {
159 const auto mir_tensor = createTensor(&tensor);
160 auto *op = _graph->create<mir::ops::ConstantOp>(mir_tensor);
161 _converterCtx->setOutput(tensor.name(), op->getOutput(0));
162 }
163
164 for (const auto &input : graph.input())
165 {
166 assert(input.has_name());
167
168 if (_converterCtx->getOutput(input.name()) == nullptr)
169 {
170 const auto &onnx_input_shape = input.type().tensor_type().shape();
171 mir::Shape shape(onnx_input_shape.dim_size());
172 for (int i = 0; i < onnx_input_shape.dim_size(); i++)
173 {
174 assert(onnx_input_shape.dim(i).has_dim_value());
175 shape.dim(i) = static_cast<int32_t>(onnx_input_shape.dim(i).dim_value());
176 }
177
178 auto elem_type = onnxDataTypeToMirDataType(
179 (onnx::TensorProto_DataType)input.type().tensor_type().elem_type());
180 mir::TensorType type{elem_type, shape};
181 auto *op = _graph->create<mir::ops::InputOp>(type);
182 _converterCtx->setOutput(input.name(), op->getOutput(0));
183 }
184 }

187std::unique_ptr<mir::Graph> ONNXImporterImpl::createIR()
188{
189 _graph = std::make_unique<mir::Graph>();
190 _converterCtx = std::make_unique<ConverterContext>(_graph.get());
191
192 createGraphInputs();
193
194 // Forming partially ordered computation graph
195 for (const auto &onnx_node : _model->graph().node())
196 {
197 assert(onnx_node.has_op_type());
198 auto &op_type = onnx_node.op_type();
199 auto opset = _modelCtx->getDomainOpsetVersion(onnx_node.domain());
200 // Get converter
201 NodeConverterRegistry::ConverterFunc converter =
202 NodeConverterRegistry::getInstance().lookup(op_type, opset);
203 assert(converter != nullptr);
204 converter(onnx_node, _converterCtx.get());
205 }
206 // Set graph outputs
207 const auto &outputs = _model->graph().output();
208 for (const auto &output : outputs)
209 {
210 assert(output.has_name());
211 auto mir_output = _converterCtx->getOutput(output.name());
212 if (mir_output == nullptr)
213 throw std::runtime_error("Bad output name!");
214
215 _graph->create<mir::ops::OutputOp>(mir_output);
216 }
217
218 return std::move(_graph);
219}
