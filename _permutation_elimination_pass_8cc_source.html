34void PermutationEliminationPass::visit(const ir::operation::Permute &node)
35{
36 auto in_operand = node.getInputs().at(0);
37 auto out_operand = node.getOutputs().at(0);
38
39 // Check if two tensors are both portable if not, we can't eliminate the node
40 {
41 auto &operand_li_map = _lowered_graph.lower_info().operand;
42 auto in_def_factor = operand_li_map.getRawPtr(in_operand)->def_factors().getOnlyElement();
43 auto out_def_factor = operand_li_map.getRawPtr(out_operand)->def_factors().getOnlyElement();
44
45 auto in_config = in_def_factor.backend()->config();
46 auto out_config = out_def_factor.backend()->config();
47
48 // FIXME Supporting dynamic tensor does not exactly mean those are portable.
49 // It may need to have another config option for checking if each uses `IPortableTensor`.
50 if (!(in_config->supportDynamicTensor() && out_config->supportDynamicTensor()))
51 return;
52 }
53
54 if (_graph.getOutputs().contains(out_operand))
55 {
56 // If the input is a const, we cannot remove it since we cannot put the constant data in the
57 // output buffer during prepare phase.
58 auto permute_input = node.getInputs().at(0);
59 if (_graph.operands().at(permute_input).isConstant())
60 return;
61 // If the input is a model input, we cannot remove it since our API lets users to set different
62 // buffers for inputs and outputs even though one tensor is both at the same time.
63 auto permute_output = node.getOutputs().at(0);
64 if (_graph.getInputs().contains(permute_input) && _graph.getOutputs().contains(permute_output))
65 return;
66 // Likewise, if copying between outputs to outputs, keep it.
67 if (_graph.getOutputs().contains(permute_input) && _graph.getOutputs().contains(permute_output))
68 return;
69
70 // Exceptional case : When the output operand is a model output
71 // In this case we keep the output and remove the input
72
73 auto &out_operand_obj = _graph.operands().at(out_operand);
74 assert(out_operand_obj.getDef() == _op_ind);
75 out_operand_obj.unsetDef();
76 _graph.operations().iterate([&](const ir::OperationIndex &op_ind, ir::Operation &op) {
77 if (!op.getOutputs().contains(in_operand))
78 return;
79 // Update Operation and Operand edges
80 op.replaceOutputs(in_operand, out_operand);
81 out_operand_obj.setDef(op_ind);
82 });
83
84 // Remove Permute operation and the operand
85 {
86 _graph.removeOperand(in_operand);
87 _graph.operations().remove(_op_ind);
88 }
89
90 _graph.operations().iterate([&](const ir::OperationIndex &op_ind, ir::Operation &op) {
91 if (!op.getInputs().contains(in_operand))
92 return;
93 op.replaceInputs(in_operand, out_operand);
94 out_operand_obj.insertUse(op_ind);
95 });
96
97 VERBOSE(removePermute) << "Permute Op removed, node index : " << _op_ind << std::endl;
98 VERBOSE(removePermute) << " - Input (removed) Operand : " << in_operand << std::endl;
99 VERBOSE(removePermute) << " - Output(kept) Operand : " << out_operand << std::endl;
100 }
101 else
102 {
103 // Otherwise keep the input and remove the output
104
105 auto &in_operand_obj = _graph.operands().at(in_operand);
106 in_operand_obj.removeUse(_op_ind);
107
108 // Make operations(that use the output) use the input
109 _graph.operations().iterate([&](const ir::OperationIndex &op_ind, ir::Operation &op) {
110 if (!op.getInputs().contains(out_operand))
111 return;
112 op.replaceInputs(out_operand, in_operand);
113 in_operand_obj.insertUse(op_ind);
114 });
115
116 // Remove the Permute operation and out_operand
117 {
118 _graph.removeOperand(out_operand);
119 _graph.operations().remove(_op_ind);
120 }
121
122 VERBOSE(removePermute) << "Permute Op removed : " << _op_ind << std::endl;
123 VERBOSE(removePermute) << " - Input (kept) Operand : " << in_operand << std::endl;
124 VERBOSE(removePermute) << " - Output(removed) Operand : " << out_operand << std::endl;
125 }
126}
