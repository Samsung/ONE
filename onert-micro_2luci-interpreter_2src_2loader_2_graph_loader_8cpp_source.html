53bool GraphLoader::isCouldBeEmplaceTensor(const int32_t tensor_index)
54{
55 uint32_t usage_count = 0;
56 for (uint32_t i = 0; i < _reader->operators().size(); ++i)
57 {
58 const auto op = _reader->operators().at(i);
59 assert(op != nullptr);
60
61 for (int32_t j = 0; j < op->inputs()->size(); ++j)
62 {
63 const auto input_index = op->inputs()->operator[](j);
64 if (input_index == tensor_index)
65 usage_count++;
66
67 if (usage_count > 1)
68 return false;
69 }
70 }
71 return true;
72}

74void GraphLoader::loadTensors(bool use_static_memory_manager)
75{
76 for (uint32_t i = 0; i < _reader->tensors().size(); ++i)
77 {
78 const auto const_tensor = _reader->tensors().at(i);
79
80 // TODO: handle variable tensors with static memory manager
81 if (const_tensor->is_variable() and use_static_memory_manager)
82 assert(false && "Not supported now");
83
84 auto const buffer = wrap(_reader->buffers()[const_tensor->buffer()]->data());
85 auto const const_dims = wrap(const_tensor->shape()); // in NHWC
86 if (const_dims.empty() && buffer.empty())
87 {
88 // unknown shape tensor and scalar tensor
89 continue;
90 }
91
92 uint32_t size = 1;
93 for (int const_dim : const_dims)
94 {
95 size *= const_dim;
96 }
97
98 if (buffer.empty() && size > 0 && not const_tensor->is_variable())
99 {
100 // normal empty tensor
101 continue;
102 }
103
104 Shape shape(static_cast<int>(const_dims.size()));
105 for (int j = 0; j < const_dims.size(); ++j)
106 {
107 shape.dim(j) = const_dims.at(j);
108 }
109
110 // Create dtype
111 const auto dtype = luci_datatype(const_tensor->type());
112
113 AffineQuantization *quantization = nullptr;
114 const auto quant_params = const_tensor->quantization();
115 if (quant_params)
116 {
117 auto unique_ptr_quantization = std::make_unique<AffineQuantization>();
118 assert(quant_params->zero_point()->size() == quant_params->scale()->size());
119 unique_ptr_quantization->scale.assign(quant_params->scale()->cbegin(),
120 quant_params->scale()->cend());
121 unique_ptr_quantization->zero_point.assign(quant_params->zero_point()->cbegin(),
122 quant_params->zero_point()->cend());
123 unique_ptr_quantization->quantized_dimension = quant_params->quantized_dimension();
124
125 quantization = _runtime_graph->addAffineQuantization(std::move(unique_ptr_quantization));
126 }
127
128 // Let's check if tensor with zero size have quantization params
129 if (size == 0)
130 {
131 if (quantization != nullptr)
132 _runtime_graph->addIntermediateTensorAffineQuantization(quantization);
133
134 continue;
135 }
136
137 // Get pointer to data from buffer
138 auto data_ptr = const_cast<unsigned char *>(buffer.data());
139
140 auto tensor = std::make_unique<Tensor>(dtype, std::move(shape), quantization);
141
142 assert(data_ptr != nullptr or const_tensor->is_variable());
143
144 // Save pointer to const data
145 if (data_ptr)
146 tensor->writeDataWithoutCopy(static_cast<void *>(data_ptr));
147
148 _index_to_tensor->emplace(i, tensor.get());
149 _runtime_graph->addTensor(std::move(tensor));
150 }
151}

153void GraphLoader::initInputTensors(bool use_static_memory_manager) const
154{
155 for (const auto input_ind : _reader->inputs())
156 {
157 const auto tensor = _reader->tensors()[input_ind];
158 const auto dtype = luci_datatype(tensor->type());
159 const auto tensor_shape = wrap(tensor->shape());
160
161 Shape shape(static_cast<int>(tensor_shape.size()));
162 for (int i = 0; i < tensor_shape.size(); ++i)
163 {
164 shape.dim(i) = tensor_shape.at(i);
165 }
166
167 AffineQuantization *quantization = nullptr;
168 const auto quant_params = tensor->quantization();
169 if (quant_params)
170 {
171 auto unique_ptr_quantization = std::make_unique<AffineQuantization>();
172 assert(quant_params->zero_point()->size() == quant_params->scale()->size());
173 unique_ptr_quantization->scale.assign(quant_params->scale()->cbegin(),
174 quant_params->scale()->cend());
175 unique_ptr_quantization->zero_point.assign(quant_params->zero_point()->cbegin(),
176 quant_params->zero_point()->cend());
177 unique_ptr_quantization->quantized_dimension = quant_params->quantized_dimension();
178
179 quantization = _runtime_graph->addAffineQuantization(std::move(unique_ptr_quantization));
180 }
181
182 auto tensor_interpreter = std::make_unique<Tensor>(dtype, std::move(shape), quantization);
183
184 tensor_interpreter->set_allocatable(_memory_manager->is_allocate_input());
185 if (not use_static_memory_manager)
186 {
187 // Using Dynamic Allocations
188 _memory_manager->allocate_memory(*tensor_interpreter);
189 }
190 else
191 {
192 // Using static allocations
193 _memory_manager->allocate_input_buf();
194 }
195
196 _runtime_graph->addInputTensor(tensor_interpreter.get());
197 _index_to_tensor->emplace(input_ind, tensor_interpreter.get());
198 _runtime_graph->addTensor(std::move(tensor_interpreter));
199 }

202void GraphLoader::loadOperators(bool use_static_memory_manager)
203{
204 ExecutionPlanTable execution_plan;
205 // Set execution plan for static memory manager
206 if (use_static_memory_manager)
207 {
208 // Read metadata
209 const auto metadata = _reader->metadata();
210 for (auto meta : metadata)
211 {
212 assert(meta != nullptr);
213
214 assert(meta->buffer() < _reader->buffers().size());
215 assert(_reader->buffers()[meta->buffer()] != nullptr);
216 const auto buffer = wrap(_reader->buffers()[meta->buffer()]->data());
217
218 if (meta->name()->str().compare("ONE_execution_plan_table") == 0)
219 {
220 execution_plan = read_metadata::decode_execution_plan(buffer);
221 }
222 }
223 if (execution_plan.empty())
224 assert(false && "Static Memory Manager should be used with circle-execution-planner");
225 }
226
227 KernelBuilder kernel_builder(_runtime_graph, _reader);
228 const uint32_t input_size = _runtime_graph->getInputTensors().size();
229 const uint32_t output_size = _reader->outputs().size();
230
231 if (use_static_memory_manager)
232 {
233 // Set offset for input tensors
234 for (int32_t input_ind = 0; input_ind < input_size; ++input_ind)
235 {
236 auto input_tensor = _runtime_graph->getInputTensors().at(input_ind);
237 input_tensor->set_offset(execution_plan.at(input_ind)[0]);
238
239 _memory_manager->allocate_memory_for_input(*input_tensor);
240 }
241 }
242
243 for (uint32_t i = 0; i < _reader->operators().size(); ++i)
244 {
245 const auto op = _reader->operators().at(i);
246 assert(op != nullptr);
247
248 std::vector<const Tensor *> input_tensors(op->inputs()->size());
249 std::vector<Tensor *> output_tensors(op->outputs()->size());
250
251 bool is_inplace = false;
252 bool is_graph_input = false;
253 for (int32_t j = 0; j < op->inputs()->size(); ++j)
254 {
255 const auto input_index = op->inputs()->operator[](j);
256 if (_index_to_tensor->find(input_index) != _index_to_tensor->end())
257 {
258 auto input_tensor = _index_to_tensor->at(input_index);
259 input_tensors.at(j) = input_tensor;
260 const auto &graph_input_tensors = _runtime_graph->getInputTensors();
261
262 is_graph_input = (std::find(graph_input_tensors.begin(), graph_input_tensors.end(),
263 input_tensor) != graph_input_tensors.end()) or
264 is_graph_input;
265
266 // TODO: handle Inplace Optimization with Static Memory Manager
267 if (not use_static_memory_manager and
268 isCouldBeEmplaceOperation(_reader->builtin_code(op)) and op->outputs()->size() == 1 and
269 isCouldBeEmplaceTensor(input_index) and not is_graph_input)
270 {
271 is_inplace = true;
272 }
273 }
274 else
275 {
276 input_tensors.at(j) = nullptr;
277 }
278 }
279
280 for (int32_t j = 0; j < op->outputs()->size(); ++j)
281 {
282 const auto output_index = op->outputs()->operator[](j);
283
284 const auto tensor = _reader->tensors()[output_index];
285 const auto dtype = luci_datatype(tensor->type());
286 const auto tensor_shape = wrap(tensor->shape());
287
288 Shape shape(static_cast<int>(tensor_shape.size()));
289 for (int k = 0; k < tensor_shape.size(); ++k)
290 {
291 shape.dim(k) = tensor_shape.at(k);
292 }
293
294 AffineQuantization *quantization = nullptr;
295 const auto quant_params = tensor->quantization();
296 if (quant_params)
297 {
298 auto unique_ptr_quantization = std::make_unique<AffineQuantization>();
299 assert(quant_params->zero_point()->size() == quant_params->scale()->size());
300 unique_ptr_quantization->scale.assign(quant_params->scale()->cbegin(),
301 quant_params->scale()->cend());
302 unique_ptr_quantization->zero_point.assign(quant_params->zero_point()->cbegin(),
303 quant_params->zero_point()->cend());
304 unique_ptr_quantization->quantized_dimension = quant_params->quantized_dimension();
305
306 quantization = _runtime_graph->addAffineQuantization(std::move(unique_ptr_quantization));
307 }
308
309 auto tensor_interpreter = std::make_unique<Tensor>(dtype, std::move(shape), quantization);
310
311 if (use_static_memory_manager and
312 (std::find(_reader->outputs().begin(), _reader->outputs().end(), output_index) ==
313 _reader->outputs().end()))
314 {
315 tensor_interpreter->set_offset(execution_plan.at(i + input_size + output_size).at(0));
316 }
317
318 _index_to_tensor->emplace(output_index, tensor_interpreter.get());
319 output_tensors.at(j) = tensor_interpreter.get();
320
321 if (std::find(_reader->outputs().begin(), _reader->outputs().end(), output_index) !=
322 _reader->outputs().end())
323 _runtime_graph->addOutputTensor(tensor_interpreter.get());
324
325 _runtime_graph->addTensor(std::move(tensor_interpreter));
326 }
327
328 const auto opcode = _reader->builtin_code(op);
329 std::unique_ptr<Kernel> kernel =
330 kernel_builder.build(std::move(input_tensors), std::move(output_tensors), opcode, i);
331 kernel->setInplaceValue(is_inplace);
332
333 _runtime_graph->addKernel(std::move(kernel));
334 }
335 if (use_static_memory_manager)
336 {
337 for (int32_t ind = 0; ind < output_size; ++ind)
338 {
339 auto output_tensor = _runtime_graph->getOutputTensors().at(ind);
340 output_tensor->set_offset(execution_plan.at(input_size + ind)[0]);
341 }
342 }
343}
