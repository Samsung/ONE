38bool isCouldBeEmplaceTensor(CircleReader *reader, const int32_t tensor_index)
39{
40 uint32_t usage_count = 0;
41 for (uint32_t i = 0; i < reader->operators().size(); ++i)
42 {
43 const auto op = reader->operators().at(i);
44 assert(op != nullptr);
45
46 for (int32_t j = 0; j < op->inputs()->size(); ++j)
47 {
48 const auto input_index = op->inputs()->operator[](j);
49 if (input_index == tensor_index)
50 usage_count++;
51
52 if (usage_count > 1)
53 return false;
54 }
55 }
56 return true;
57}

61void GraphLoader::checkInplaceOps(CircleReader *reader, RuntimeGraph *runtime_graph)
62{
63 for (uint32_t i = 0; i < reader->operators().size(); ++i)
64 {
65 const auto *op = reader->operators().at(i);
66 assert(op != nullptr);
67
68 bool is_graph_input = false;
69 for (int32_t j = 0; j < op->inputs()->size(); ++j)
70 {
71 const auto input_index = op->inputs()->operator[](j);
72 if (input_index == -1)
73 continue;
74
75 const auto &inputs_indexes = reader->inputs();
76
77 is_graph_input = (std::find(inputs_indexes.begin(), inputs_indexes.end(), input_index) !=
78 inputs_indexes.end()) or
79 is_graph_input;
80
81 if (not is_graph_input and isCouldBeEmplaceOperation(reader->builtin_code(op)) and
82 op->outputs()->size() == 1 and isCouldBeEmplaceTensor(reader, input_index))
83 {
84 runtime_graph->addInplaceOpIndex(i);
85 }
86 }
87 }
88}
