24bool isInplaceOperation(const circle::BuiltinOperator &op)
25{
26 switch (op)
27 {
28 case circle::BuiltinOperator_LOGISTIC:
29 case circle::BuiltinOperator_RESHAPE:
30 case circle::BuiltinOperator_EXPAND_DIMS:
31 case circle::BuiltinOperator_TANH:
32 case circle::BuiltinOperator_ADD:
33 case circle::BuiltinOperator_MUL:
34 case circle::BuiltinOperator_SUB:
35 case circle::BuiltinOperator_WHILE:
36 return true;
37 default:
38 return false;
39 }
40}

42bool isSingleUsageOfTensor(CircleReader *reader, const int32_t tensor_index)
43{
44 uint32_t usage_count = 0;
45
46 const auto operators = reader->operators();
47 for (uint32_t i = 0; i < operators.size(); ++i)
48 {
49 const auto *op = operators.at(i);
50 assert(op != nullptr);
51
52 const auto *op_inputs = op->inputs();
53 for (int32_t j = 0; j < op_inputs->size(); ++j)
54 {
55 const auto input_index = op_inputs->operator[](j);
56 if (input_index == tensor_index)
57 {
58 if (++usage_count > 1)
59 return false;
60 }
61 }
62 }
63
64 // Let's check that it is not graph output
65 if (usage_count == 1)
66 {
67 const auto &outputs_indexes = reader->outputs();
68 bool is_graph_output = (std::find(outputs_indexes.begin(), outputs_indexes.end(),
69 tensor_index) != outputs_indexes.end());
70 if (is_graph_output)
71 return false;
72 }
73
74 return true;
75}

79void GraphLoader::checkInplaceOps(CircleReader *reader, RuntimeGraph *runtime_graph)
80{
81 const auto operators = reader->operators();
82 const auto graph_outputs = reader->outputs();
83 for (uint32_t i = 0; i < operators.size(); ++i)
84 {
85 const auto *op = operators.at(i);
86 assert(op != nullptr);
87
88 // Check inplace optimization for operation with single input and single output
89 if (isInplaceOperation(reader->builtin_code(op)))
90 {
91 const auto *op_inputs = op->inputs();
92 const auto *op_outputs = op->outputs();
93
94 bool is_inplace = true;
95 auto non_const_input_it = op_inputs->begin();
96 while (true)
97 {
98 non_const_input_it =
99 std::find_if(non_const_input_it, op_inputs->end(), [&reader](const auto input_idx) {
100 if (input_idx == -1)
101 return false;
102
103 return not Tensor::is_constant_tensor(reader, reader->tensors()[input_idx]);
104 });
105
106 if (non_const_input_it == op_inputs->end())
107 break;
108
109 auto dist = std::distance(op_inputs->begin(), non_const_input_it);
110
111 const auto non_const_input_idx = *non_const_input_it;
112
113 // Check single usage of input tensor
114 if (not isSingleUsageOfTensor(reader, non_const_input_idx))
115 {
116 is_inplace = false;
117 break;
118 }
119
120 // Let's check single usage of output tensor
121 if (dist >= op_outputs->size() and op_outputs->size() == 1)
122 dist = 0;
123 assert(dist < op_outputs->size());
124 const auto output_index = op_outputs->operator[](dist);
125 if (not isSingleUsageOfTensor(reader, output_index))
126 {
127 is_inplace = false;
128 break;
129 }
130
131 // Check that num elements are equal
132 {
133 const auto *input_non_const_tensor = reader->tensors().at(non_const_input_idx);
134 const auto *output_tensor = reader->tensors().at(output_index);
135 if (Tensor::num_elements(input_non_const_tensor) != Tensor::num_elements(output_tensor))
136 {
137 is_inplace = false;
138 break;
139 }
140 }
141
142 // Let's check that output is not a graph output tensor
143 {
144 if (std::find(graph_outputs.begin(), graph_outputs.end(), output_index) !=
145 graph_outputs.end())
146 {
147 is_inplace = false;
148 break;
149 }
150 }
151
152 non_const_input_it++;
153 }
154
155 if (is_inplace)
156 runtime_graph->addInplaceOpIndex(op);
157 }
158 }
159}
