322// Accumulate PEIR (Peak Error to Interval Ratio)
323// PEIR = max(|a - b|) / (max(a) - min(a))
324// PEIR >= 0 (lower is better)
325void MPEIRPrinter::accum_peir(uint32_t output_idx, const std::shared_ptr<Tensor> &a,
326 const std::shared_ptr<Tensor> &b)
327{
328 assert(a->dtype() == loco::DataType::FLOAT32 and
329 b->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
330 assert(same_shape(a.get(), b.get())); // FIX_CALLER_UNLESS
331 assert(output_idx < _intermediate.size()); // FIX_CALLER_UNLESS
332
333 float min = std::numeric_limits<float>::max();
334 float max = std::numeric_limits<float>::lowest();
335
336 for (uint32_t i = 0; i < a->size<loco::DataType::FLOAT32>(); i++)
337 {
338 const auto a_val = a->at<loco::DataType::FLOAT32>(i);
339 min = std::min(a_val, min);
340 max = std::max(a_val, max);
341 }
342
343 float interval = max - min;
344
345 // Corner case: All values are the same. We set interval = 1 in this case
346 if (interval == 0)
347 interval = 1.0;
348
349 float peak_error = std::numeric_limits<float>::lowest();
350
351 for (uint32_t i = 0; i < a->size<loco::DataType::FLOAT32>(); i++)
352 {
353 const auto a_val = a->at<loco::DataType::FLOAT32>(i);
354 const auto b_val = b->at<loco::DataType::FLOAT32>(i);
355 const auto error = std::abs(a_val - b_val);
356 peak_error = std::max(error, peak_error);
357 }
358
359 _intermediate.at(output_idx) += peak_error / interval;
360}

435void TopKMatchPrinter::accum_topk_accuracy(uint32_t output_idx, const std::shared_ptr<Tensor> &a,
436 const std::shared_ptr<Tensor> &b)
437{
438 assert(a->dtype() == loco::DataType::FLOAT32 and
439 b->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
440 assert(same_shape(a.get(), b.get())); // FIX_CALLER_UNLESS
441 assert(output_idx < _intermediate.size()); // FIX_CALLER_UNLESS
442
443 // Find Top-k largest elements
444 // This implementation is a variant of "Method 2 (Use temporary array)" in
445 // https://www.geeksforgeeks.org/k-largestor-smallest-elements-in-an-array/
446 // We sort top-k elements by value and index to ensure that the element with an earlier
447 // index comes first if multiple elements have the same value.
448 auto find_topk = [this](const std::shared_ptr<Tensor> &tensor) {
449 assert(_k <= tensor->size<loco::DataType::FLOAT32>()); // FIX_CALLER_UNLESS
450
451 // first: value, second: index
452 std::vector<std::pair<float, uint32_t>> topk;
453 topk.resize(_k);
454
455 // Initialize
456 for (uint32_t i = 0; i < _k; i++)
457 {
458 topk[i] = std::make_pair(tensor->at<loco::DataType::FLOAT32>(i), i);
459 }
460
461 // Input pair: (value, index)
462 // Return true if a has smaller value than b. If a and b have the same value,
463 // return true if a has larger index.
464 auto compare = [](const std::pair<float, uint32_t> &a, const std::pair<float, uint32_t> &b) {
465 if (a.first == b.first)
466 return a.second > b.second;
467
468 return a.first < b.first;
469 };
470
471 for (uint32_t i = _k; i < tensor->size<loco::DataType::FLOAT32>(); i++)
472 {
473 auto val = std::make_pair(tensor->at<loco::DataType::FLOAT32>(i), i);
474
475 auto min = std::min_element(topk.begin(), topk.end(), compare);
476 if (compare(*min, val))
477 {
478 // val is larger than min. Replace min with val.
479 auto min_index = std::distance(topk.begin(), min);
480 topk[min_index] = val;
481 }
482 }
483
484 return topk;
485 };
486
487 auto first_topk = find_topk(a);
488 auto second_topk = find_topk(b);
489
490 uint32_t matched = 0;
491 for (uint32_t i = 0; i < _k; i++)
492 {
493 for (uint32_t j = 0; j < _k; j++)
494 {
495 if (first_topk[i].second == second_topk[j].second)
496 {
497 matched++;
498 break;
499 }
500 }
501 }
502
503 float matched_ratio = static_cast<float>(matched) / _k;
504
505 _intermediate.at(output_idx) += matched_ratio;
506}
