62template <typename TFNode> bool canonicalize_eltwise_binary_node(TFNode *node)
63{
64 auto graph = node->graph();
65
88 using CanonicalNodeT = typename EltwiseBinaryCanonicalizationRule<TFNode>::CanonicalNode;
89
90 auto node_A = node->x();
91 auto node_B = node->y();
92
93 if (!loco::shape_known(node_A) || !loco::shape_known(node_B))
94 return false;
95 if (!loco::shape_known(node))
96 return false;
97
98 auto out_shape = loco::shape_get(node).template as<loco::TensorShape>();
99
100 // Create a node
101 auto canonical_node = graph->nodes()->template create<CanonicalNodeT>();
102
103 using moco::tf::eltwise::binary::connect_to;
104 using moco::tf::broadcast_to;
105
106 // update connections
107 std::make_pair(node_A, node_B) | broadcast_to(out_shape) | connect_to(canonical_node);
108
109 // replace node
110 replace(node).with(canonical_node);
111
112 return true;
113}
