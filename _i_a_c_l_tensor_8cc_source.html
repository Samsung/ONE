28size_t IACLTensor::calcOffset(const ir::Coordinates &coords) const
29{
30 auto rank = _rank;
31 rank = rank == 0 ? 1 : rank;
32 assert(static_cast<size_t>(rank) == coords.size());
33
34 ::arm_compute::Coordinates acl_coords;
35 for (size_t i = 0; i < rank; ++i)
36 {
37 const ARMComputeAxis reversed{static_cast<uint32_t>((rank - i) - 1)};
38 acl_coords.set(reversed.value(), coords[i]);
39 }
40
41 return info()->offset_element_in_bytes(acl_coords);
42}
