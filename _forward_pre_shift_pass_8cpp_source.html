29class ForwardPreShift final : public luci::CircleNodeMutableVisitor<bool>
30{
31protected:
32 bool visit(luci::CircleNode *node) { return false; }
33
34 bool visit(luci::CircleSlice *node)
35 {
36 auto pre_shift = to_pre_shift(node->input());
37 if (not pre_shift)
38 return false;
39
40 if (loco::succs(pre_shift).size() != 1)
41 return false;
42
43 node->input(pre_shift->inputs(0));
44 loco::replace(node).with(pre_shift);
45 pre_shift->inputs(0, node);
46
47 // Shape should be copied, because
48 // shape inference does not work well for Custom Op (PreShift)
49 copy_shape(node, pre_shift);
50
51 return true;
52 }
53};

60bool ForwardPreShiftPass::run(loco::Graph *g)
61{
62 bool changed = false;
63
64 ForwardPreShift fps;
65 for (auto node : loco::active_nodes(loco::output_nodes(g)))
66 {
67 auto cnode = loco::must_cast<luci::CircleNode *>(node);
68 if (cnode->accept(&fps))
69 changed = true;
70 }
71
72 return changed;
73}
