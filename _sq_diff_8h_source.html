29#define SQDIFF(N) \
30 do \
31 { \
32 NdArrayDesc<N> input1_desc; \
33 NdArrayDesc<N> input2_desc; \
34 NdArrayDesc<N> output_desc; \
35 SqDiffImpl<T, N>(input1_shape, input1_data, input2_shape, input2_data, output_shape, \
36 output_data, &input1_desc, &input2_desc, &output_desc); \
37 } while (0);

39template <typename T, int N>
40void SqDiffImpl(const Shape &input1_shape, const T *input1_data, const Shape &input2_shape,
41 const T *input2_data, const Shape &output_shape, T *output_data,
42 NdArrayDesc<N> *desc1_in, NdArrayDesc<N> *desc2_in, NdArrayDesc<N> *desc_out)
43{
44 std::vector<int> input_iter;
45 input_iter.resize(N);
46 const auto output_dims = output_shape.DimsData();
47
48 // Copy dims to desc, calculating strides.
49 CopyDimsToDesc<N>(output_shape, desc_out);
50 NdArrayDescsForElementwiseBroadcast<N>(input1_shape, input2_shape, desc1_in, desc2_in);
51
52 do
53 {
54 int input1_indx = SubscriptToIndexGeneric(desc1_in, input_iter.data());
55 int input2_indx = SubscriptToIndexGeneric(desc2_in, input_iter.data());
56 int output_indx = SubscriptToIndexGeneric(desc_out, input_iter.data());
57 output_data[output_indx] = (input1_data[input1_indx] - input2_data[input2_indx]) *
58 (input1_data[input1_indx] - input2_data[input2_indx]);
59 } while (NextIndex(N, output_dims, input_iter.data()));
60}

62template <typename T>
63void SqDiff(const Shape &input1_shape, const T *input1_data, const Shape &input2_shape,
64 const T *input2_data, const Shape &output_shape, T *output_data)
65{
66 UNUSED_RELEASE(output_shape);
67 assert(input1_shape.DimensionsCount() > 0 && input2_shape.DimensionsCount() > 0 &&
68 output_shape.DimensionsCount() > 0);
69 int outRank = output_shape.DimensionsCount();
70
71 switch (outRank)
72 {
73 case 4:
74 SQDIFF(4);
75 break;
76
77 case 3:
78 SQDIFF(3);
79 break;
80
81 case 2:
82 SQDIFF(2);
83 break;
84
85 case 1:
86 SQDIFF(1);
87 break;
88
89 default:
90 throw std::runtime_error("Support up to 4-D tensors at present");
91 break;
92 }
93}
