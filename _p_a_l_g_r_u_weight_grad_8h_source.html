37void calculateGRUWeightGrads(
38 const float *output_grad_data, const float *weight_input_data, float *weight_input_grad_data,
39 const float *weight_hidden_data, float *weight_hidden_grad_data, const float *bias_input_data,
40 float *bias_input_grad_data, const float *bias_hidden_data, float *bias_hidden_grad_data,
41 const float *input_data, float *input_grad_data, float *state_grad_data,
42 const core::OMRuntimeShape &input_shape, const core::OMRuntimeShape &output_fc_shape,
43 const core::OMRuntimeShape &output_shape, const core::OMRuntimeShape &weight_input_shape,
44 const core::OMRuntimeShape &weight_hidden_shape, float *output_data, float *left_logistic_data,
45 float *left_mul_data, float *right_logistic_data, const float *right_mul_left_input_data,
46 const float *right_mul_right_input_data, float *tanh_data, const float *middle_mul_left_input,
47 const float *middle_mul_right_input, float *left_fc_output_grad_buffer,
48 float *right_fc_output_grad_buffer)
49{
50 int num_elements = output_shape.flatSize();
51 for (int i = 0; i < num_elements; ++i)
52 {
53 // Middle Mul left input grad
54 float left_middle_mul = output_grad_data[i];
55 left_middle_mul *= middle_mul_right_input[i];
56
57 // Middle Mul right input grad
58 float right_middle_mul = output_grad_data[i];
59 right_middle_mul *= middle_mul_left_input[i];
60
61 // Tanh` = 1 / (cos(x) ^ 2)
62 float tanh_grad_value;
63 {
64 float tanh = std::tanh(tanh_data[i]);
65 if (tanh_data[i] > 19)
66 {
67 tanh = 1;
68 }
69 else if (tanh_data[i] < -19)
70 {
71 tanh = -1;
72 }
73
74 tanh_grad_value = (1 - tanh * tanh) * right_middle_mul;
75 }
76
77 // Left mul
78 float left_mul_grad_value = output_grad_data[i] * output_data[i];
79
80 // Sub` = -1
81 // Left Logistic: Logistic` = (exp(-x) * (1 / (1 + exp(-x))) ^ 2)
82 float left_logistic_grad_value;
83 {
84 float log_value = (1 / (1 + std::exp(-left_logistic_data[i])));
85 left_logistic_grad_value =
86 log_value * (1 - log_value) * (left_middle_mul + left_mul_grad_value);
87 }
88
89 // Right mul left input
90 float right_mul_left_input = tanh_grad_value;
91 right_mul_left_input *= right_mul_right_input_data[i];
92
93 // Right mul right input
94 float right_mul_right_input = tanh_grad_value;
95 right_mul_right_input *= right_mul_left_input_data[i];
96
97 // Right logistic
98 float right_logistic_grad_value;
99 {
100 float log_value = (1 / (1 + std::exp(-right_logistic_data[i])));
101 right_logistic_grad_value = log_value * (1 - log_value) * right_mul_left_input;
102 }
103
104 // Left concatenation
105 left_fc_output_grad_buffer[i] = left_logistic_grad_value;
106 left_fc_output_grad_buffer[i + num_elements] = right_logistic_grad_value;
107 left_fc_output_grad_buffer[i + 2 * num_elements] = right_mul_right_input;
108
109 // Right concatenation
110 right_fc_output_grad_buffer[i] = left_logistic_grad_value;
111 right_fc_output_grad_buffer[i + num_elements] = right_logistic_grad_value;
112 right_fc_output_grad_buffer[i + 2 * num_elements] = tanh_grad_value;
113 }
114
115 // Left fc weight grad
116 FullyConnectedWeightGrad(left_fc_output_grad_buffer, output_fc_shape, output_data, output_shape,
117 weight_input_grad_data, weight_input_shape,
118 core::OpTrainableRankType::ALL);
119 // Right fc weight grad
120 FullyConnectedWeightGrad(right_fc_output_grad_buffer, output_fc_shape, input_data, input_shape,
121 weight_hidden_grad_data, weight_hidden_shape,
122 core::OpTrainableRankType::ALL);
123
124 // Set state grad to zero
125 std::memset(state_grad_data, 0, output_shape.flatSize() * sizeof(float));
126}

130OMStatus GRUWeightGrads(
131 const float *output_grad_data, const float *weight_input_data, float *weight_input_grad_data,
132 const float *weight_hidden_data, float *weight_hidden_grad_data, const float *bias_input_data,
133 float *bias_input_grad_data, const float *bias_hidden_data, float *bias_hidden_grad_data,
134 const float *input_data, float *input_grad_data, float *state_grad_data,
135 const core::OMRuntimeShape &input_shape, const core::OMRuntimeShape &output_shape,
136 const core::OMRuntimeShape &weight_input_shape, const core::OMRuntimeShape &weight_hidden_shape,
137 const core::OMRuntimeShape &output_shape_fc, float *intermediate_buffer,
138 float *left_fc_output_grad_buffer, float *right_fc_output_grad_buffer)
139{
140 const int32_t time = input_shape.dims(0);
141
142 // Init pointers to intermediate values
143 size_t offset = output_shape.flatSize();
144
145 size_t data_type_size = sizeof(float);
146 const int32_t num_of_intermediate_tensors = 9;
147 size_t time_offset = num_of_intermediate_tensors * offset;
148
149 core::OMRuntimeShape two_dim_input_shape(2);
150 auto dim_count = input_shape.dimensionsCount();
151 if (dim_count < 2)
152 return UnsupportedType;
153 two_dim_input_shape.setDim(0, input_shape.dims(dim_count - 2));
154 two_dim_input_shape.setDim(1, input_shape.dims(dim_count - 1));
155
156 core::OMRuntimeShape two_dim_output_shape(2);
157 dim_count = output_shape.dimensionsCount();
158 if (dim_count < 2)
159 return UnsupportedType;
160 two_dim_output_shape.setDim(0, output_shape.dims(dim_count - 2));
161 two_dim_output_shape.setDim(1, output_shape.dims(dim_count - 1));
162
163 std::memset(weight_input_grad_data, 0, weight_input_shape.flatSize() * sizeof(float));
164 std::memset(weight_hidden_grad_data, 0, weight_hidden_shape.flatSize() * sizeof(float));
165
166 for (int i = 0; i < time; ++i)
167 {
168 float *output_data = intermediate_buffer;
169 float *left_logistic_data = output_data + offset;
170 float *left_mul_data = left_logistic_data + offset;
171 float *right_logistic_data = left_mul_data + offset;
172 float *right_mul_left_input_data = right_logistic_data + offset;
173 float *right_mul_right_input_data = right_mul_left_input_data + offset;
174 float *tanh_data = right_mul_right_input_data + offset;
175 float *middle_mul_left_input = tanh_data + offset;
176 float *middle_mul_right_input = middle_mul_left_input + offset;
177
178 calculateGRUWeightGrads(
179 output_grad_data, weight_input_data, weight_input_grad_data, weight_hidden_data,
180 weight_hidden_grad_data, bias_input_data, bias_input_grad_data, bias_hidden_data,
181 bias_hidden_grad_data, input_data, input_grad_data, state_grad_data, two_dim_input_shape,
182 output_shape_fc, two_dim_output_shape, weight_input_shape, weight_hidden_shape, output_data,
183 left_logistic_data, left_mul_data, right_logistic_data, right_mul_left_input_data,
184 right_mul_right_input_data, tanh_data, middle_mul_left_input, middle_mul_right_input,
185 left_fc_output_grad_buffer, right_fc_output_grad_buffer);
186 input_data += input_shape.dims(2);
187 intermediate_buffer += time_offset;
188 }
189 return Ok;
190}
