33inline uint32_t cal_offset(const luci::CircleConst *node, const std::vector<uint32_t> &indices)
34{
35 // sanity check for node's rank
36 assert(node != nullptr && node->rank() == 4);
37
38 // sanity check for indices
39 assert(indices.size() == 4);
40
41 return indices[0] * node->dim(1).value() * node->dim(2).value() * node->dim(3).value() +
42 indices[1] * node->dim(2).value() * node->dim(3).value() +
43 indices[2] * node->dim(3).value() + indices[3];
44}

65bool fuse_mul_with_conv(luci::CircleMul *mul)
66{
67 // sanity check
68 RETURN_FALSE_UNLESS(mul->dtype() == loco::DataType::FLOAT32);
69
70 luci::CircleConst *const_mul_operand = nullptr;
71 luci::CircleConv2D *conv = nullptr;
72 RETURN_FALSE_UNLESS(luci::fill(&const_mul_operand, &conv).with_commutative_args_of(mul));
73
74 // sanity check
75 RETURN_FALSE_UNLESS(conv->dtype() == loco::DataType::FLOAT32 &&
76 const_mul_operand->dtype() == loco::DataType::FLOAT32);
77
78 // NOTE for general activation function: S * Act(A * B) != Act(A*(SB))
79 if (conv->fusedActivationFunction() != luci::FusedActFunc::NONE)
80 return false;
81
82 // check that const_mul_operand is channel-wise constant or just a scalar
83 RETURN_FALSE_UNLESS(const_mul_operand->rank() == 4 || const_mul_operand->rank() == 1 ||
84 const_mul_operand->rank() == 0);
85
86 std::vector<float> mul_values;
87 if (const_mul_operand->rank() == 4)
88 {
89 // check channel-wise broadcasting
90 RETURN_FALSE_UNLESS(const_mul_operand->dim(0).value() == 1 &&
91 const_mul_operand->dim(1).value() == 1 &&
92 const_mul_operand->dim(2).value() == 1);
93 }
94 else if (const_mul_operand->rank() == 1 || const_mul_operand->rank() == 0)
95 {
96 // sanity check
97 RETURN_FALSE_UNLESS(const_mul_operand->size<loco::DataType::FLOAT32>() != 0);
98 }
99
100 mul_values.resize(const_mul_operand->size<loco::DataType::FLOAT32>());
101 for (uint32_t idx = 0; idx < mul_values.size(); idx++)
102 {
103 mul_values[idx] = const_mul_operand->at<loco::DataType::FLOAT32>(idx);
104 }
105
106 // filter
107 auto const conv_filter = dynamic_cast<luci::CircleConst *>(conv->filter());
108
109 // sanity check
110 RETURN_FALSE_UNLESS(conv_filter != nullptr && conv_filter->rank() == 4 &&
111 conv_filter->dtype() == loco::DataType::FLOAT32);
112
113 auto const out_channels = conv_filter->dim(0).value();
114
115 // multiplier is either channelwise constant or scalar
116 RETURN_FALSE_UNLESS(out_channels == mul_values.size() || mul_values.size() == 1);
117
118 // bias
119 auto const conv_bias = dynamic_cast<luci::CircleConst *>(conv->bias());
120
121 RETURN_FALSE_UNLESS(conv_bias == nullptr ||
122 (conv_bias->rank() == 1 && conv_bias->dim(0) == out_channels &&
123 conv_bias->dtype() == loco::DataType::FLOAT32));
124
125 luci::CircleConst *fused_conv_filter = nullptr;
126 {
127 // fused filter
128 fused_conv_filter = luci::clone(conv_filter);
129 // set values of conv filter multiplied by constant channel-wise
130 for (uint32_t out_chan = 0; out_chan < out_channels; out_chan++)
131 {
132 // for scalar - first element, otherwise - channelwise
133 float mult = mul_values[out_chan % mul_values.size()];
134 for (uint32_t out_height = 0; out_height < fused_conv_filter->dim(1).value(); out_height++)
135 {
136 for (uint32_t out_width = 0; out_width < fused_conv_filter->dim(2).value(); out_width++)
137 {
138 for (uint32_t in_chan = 0; in_chan < fused_conv_filter->dim(3).value(); in_chan++)
139 {
140 std::vector<uint32_t> indices = {out_chan, out_height, out_width, in_chan};
141 auto const data =
142 conv_filter->at<loco::DataType::FLOAT32>(cal_offset(conv_filter, indices));
143 fused_conv_filter->at<loco::DataType::FLOAT32>(cal_offset(fused_conv_filter, indices)) =
144 mult * data;
145 }
146 }
147 }
148 }
149 fused_conv_filter->name(conv_filter->name() + "/FusedMul");
150 luci::add_origin(fused_conv_filter, luci::get_origin(conv_filter));
151 }
152
153 luci::CircleConst *fused_conv_bias = nullptr;
154 if (conv_bias != nullptr)
155 {
156 // fused bias
157 fused_conv_bias = luci::clone(conv_bias);
158 // update bias values
159 for (uint32_t c = 0; c < conv_bias->size<loco::DataType::FLOAT32>(); c++)
160 {
161 // for scalar - first element, otherwise - channelwise
162 float mult = mul_values[c % mul_values.size()];
163 auto const data = conv_bias->at<loco::DataType::FLOAT32>(c);
164 fused_conv_bias->at<loco::DataType::FLOAT32>(c) = mult * data;
165 }
166
167 fused_conv_bias->name(conv_bias->name() + "/FusedMul");
168 luci::add_origin(fused_conv_bias, luci::get_origin(conv_bias));
169 }
170
171 // Configure new CircleConv2D operation.
172 auto *fused_conv = loco::must_cast<luci::CircleConv2D *>(luci::clone_node(conv, mul->graph()));
173 fused_conv->input(conv->input());
174 fused_conv->filter(fused_conv_filter);
175 fused_conv->bias(fused_conv_bias);
176 fused_conv->name(conv->name() + "/FusedMul");
177 fused_conv->fusedActivationFunction(mul->fusedActivationFunction());
178 luci::add_origin(fused_conv,
179 luci::composite_origin({luci::get_origin(conv), luci::get_origin(mul)}));
180
181 // Replace old mul operation with new fused_conv with updated kernel/bias
182 replace(mul).with(fused_conv);
183
184 return true;
185}

192bool FuseMulWithConvPass::run(loco::Graph *g)
193{
194 bool changed = false;
195 for (auto node : loco::active_nodes(loco::output_nodes(g)))
196 {
197 auto mul = dynamic_cast<luci::CircleMul *>(node);
198 if (not mul)
199 continue;
200
201 if (fuse_mul_with_conv(mul))
202 changed = true;
203 }
204
205 return changed;
206}
