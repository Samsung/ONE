29void unfold(std::vector<float> &buf, uint32_t input_n, uint32_t input_h, uint32_t input_w,
30 uint32_t input_c, uint32_t stride_h, uint32_t stride_w, uint32_t dilation_h,
31 uint32_t dilation_w, uint32_t kernel_h, uint32_t kernel_w, uint32_t kernel_ic)
32{
33 assert(input_n > 0 && input_h > 0 && input_w > 0 && input_c > 0);
34 assert(stride_h > 0 && stride_w > 0);
35 assert(kernel_h > 0 && kernel_w > 0 && kernel_ic > 0);
36
37 if (input_c != kernel_ic)
38 throw std::runtime_error("RecordHessian: Input channels do not match kernel channels.");
39 uint32_t out_height = (input_h - dilation_h * (kernel_h - 1) - 1) / stride_h + 1;
40 uint32_t out_width = (input_w - dilation_w * (kernel_w - 1) - 1) / stride_w + 1;
41 uint32_t patch_size = kernel_h * kernel_w * kernel_ic;
42 std::vector<float> unfolded_buf(input_n * out_height * out_width * patch_size, 0.0f);
43
44 uint32_t index = 0;
45 uint32_t in_y, in_x;
46 for (uint32_t n = 0; n < input_n; ++n)
47 {
48 for (uint32_t y = 0; y < out_height; ++y)
49 {
50 for (uint32_t x = 0; x < out_width; ++x)
51 {
52 for (uint32_t in_c = 0; in_c < input_c; ++in_c)
53 {
54 for (uint32_t ky = 0; ky < kernel_h; ++ky)
55 {
56 for (uint32_t kx = 0; kx < kernel_w; ++kx)
57 {
58 in_y = y * stride_h + ky * dilation_h;
59 in_x = x * stride_w + kx * dilation_w;
60 if (in_y < input_h && in_x < input_w)
61 {
62 unfolded_buf[index] = buf[((n * input_h + in_y) * input_w + in_x) * input_c + in_c];
63 }
64 index++;
65 }
66 }
67 }
68 }
69 }
70 }
71
72 buf.swap(unfolded_buf);
73}

75void HessianComputer::recordHessianForFullyConnected(const luci::CircleNode *node)
76{
77 assert(_input_tensor->shape().num_dims() < 4);
78 assert(_input_tensor->element_type() == luci_interpreter::DataType::FLOAT32);
79
80 uint32_t size_in_ch;
81 uint32_t length;
82
83 const auto data = _input_tensor->data<float>();
84 const auto num_elements = _input_tensor->shape().num_elements();
85 std::vector<float> buf(data, data + num_elements);
86
87 if (_input_tensor->shape().num_dims() == 3)
88 {
89 size_in_ch = _input_tensor->shape().dim(2); // input_tensor [batch, length, channel]
90 }
91 else if (_input_tensor->shape().num_dims() == 2)
92 {
93 size_in_ch = _input_tensor->shape().dim(1); // input_tensor [length, channel]
94 }
95 else
96 {
97 throw std::runtime_error("RecordHessian: Unsupported node rank");
98 }
99 assert(size_in_ch != 0);
100 length = num_elements / size_in_ch;
101
102 std::vector<float> hessian(size_in_ch * size_in_ch, 0);
103
104 for (uint32_t i = 0; i < size_in_ch; ++i)
105 {
106 for (uint32_t j = 0; j < size_in_ch; ++j)
107 {
108 float sum = 0;
109 for (uint32_t k = 0; k < length; ++k)
110 {
111 sum += buf[i + k * size_in_ch] * buf[j + k * size_in_ch];
112 }
113 hessian[i * size_in_ch + j] = 2 * sum;
114 }
115 }
116
117 HessianVector &vector = _hessian_map[node];
118 vector.update(hessian);
119}

121void HessianComputer::recordHessianForConv2D(const luci::CircleNode *node)
122{
123 assert(_input_tensor->shape().num_dims() == 4);
124 assert(_input_tensor->element_type() == luci_interpreter::DataType::FLOAT32);
125
126 const auto circle_conv2d = loco::must_cast<const luci::CircleConv2D *>(node);
127 const auto node_filter = loco::must_cast<luci::CircleConst *>((circle_conv2d)->filter());
128 assert(circle_conv2d->rank() >= 4);
129 assert(node_filter->dtype() == loco::DataType::FLOAT32);
130 assert(node_filter->rank() == 4);
131
132 uint32_t size_in_ch =
133 node_filter->size<loco::DataType::FLOAT32>() / circle_conv2d->dim(3).value();
134
135 uint32_t input_n = _input_tensor->shape().dim(0);
136 uint32_t input_h = _input_tensor->shape().dim(1);
137 uint32_t input_w = _input_tensor->shape().dim(2);
138 uint32_t input_c = _input_tensor->shape().dim(3);
139
140 uint32_t stride_h = circle_conv2d->stride()->h();
141 uint32_t stride_w = circle_conv2d->stride()->w();
142 uint32_t dilation_h = circle_conv2d->dilation()->h();
143 uint32_t dilation_w = circle_conv2d->dilation()->w();
144
145 uint32_t kernel_h = node_filter->dim(1).value();
146 uint32_t kernel_w = node_filter->dim(2).value();
147 uint32_t kernel_ic = node_filter->dim(3).value();
148
149 const auto data = _input_tensor->data<float>();
150 const auto num_elements = _input_tensor->shape().num_elements();
151 assert(data != 0);
152 assert(num_elements != 0);
153 std::vector<float> buf(data, data + num_elements);
154
155 unfold(buf, input_n, input_h, input_w, input_c, stride_h, stride_w, dilation_h, dilation_w,
156 kernel_h, kernel_w, kernel_ic);
157 assert(size_in_ch != 0);
158 uint32_t length = buf.size() / size_in_ch;
159
160 std::vector<float> hessian(size_in_ch * size_in_ch, 0);
161 for (uint32_t i = 0; i < size_in_ch; ++i)
162 {
163 for (uint32_t j = 0; j < size_in_ch; ++j)
164 {
165 float sum = 0;
166 for (uint32_t k = 0; k < length; ++k)
167 {
168 sum += buf[i + k * size_in_ch] * buf[j + k * size_in_ch];
169 }
170 hessian[i * size_in_ch + j] = 2 * sum;
171 }
172 }
173
174 HessianVector &vector = _hessian_map[node];
175 vector.update(hessian);
176}

178void HessianComputer::recordHessian(const luci::CircleNode *node,
179 const luci_interpreter::Tensor *input_tensor)
180{
181 if (node == nullptr || input_tensor == nullptr)
182 throw std::invalid_argument("RecordHessian: node or input_tensor is null.");
183
184 if (input_tensor->element_type() != luci_interpreter::DataType::FLOAT32)
185 throw std::runtime_error("RecordHessian: Unsupported dtype: only FLOAT32 is supported.");
186
187 _input_tensor = input_tensor;
188
189 switch (node->opcode())
190 {
191 case luci::CircleOpcode::FULLY_CONNECTED:
192 recordHessianForFullyConnected(node);
193 break;
194 case luci::CircleOpcode::CONV_2D:
195 recordHessianForConv2D(node);
196 break;
197 default:
198 throw std::runtime_error("RecordHessian: " + node->name() + " is unsupported op.");
199 }
200}
