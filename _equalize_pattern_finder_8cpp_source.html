131struct Forwardable
132{
133 bool scale_forwardable = false;
134 bool shift_forwardable = false;
135};

137// Return Forwardable of node
138Forwardable forwardable(luci::CircleNode *node)
139{
140 if (node == nullptr)
141 throw std::invalid_argument("node");
142
143 switch (node->opcode())
144 {
145 case luci::CircleOpcode::PAD:
146 return {true, false};
147 case luci::CircleOpcode::MAX_POOL_2D:
148 // Assumption: all scale values are positive.
149 return {true, true};
150 case luci::CircleOpcode::SLICE:
151 return {true, true};
152 default:
153 return {false, false};
154 }
155}

157// Save matching EqualizePattern to res
158void match(luci::CircleNode *front, std::vector<EqualizePattern> &res)
159{
160 if (front == nullptr)
161 throw std::invalid_argument("front");
162
163 auto front_fusability = fusability(front);
164
165 for (auto succ : loco::succs(front))
166 {
167 // Check succ fusability
168 auto back = loco::must_cast<luci::CircleNode *>(succ);
169 auto back_fusability = fusability(back);
170
171 // If 'back' is not fusable with PreScale/Shift, we check if PreScale/Shift
172 // can forward across 'back'
173 // TODO Generalize this code to support multiple forwardable Ops
174 if ((not back_fusability.pre_scale) and (not back_fusability.pre_shift))
175 {
176 auto f = forwardable(back);
177 if (f.scale_forwardable or f.shift_forwardable)
178 {
179 auto succ_succs = loco::succs(back);
180 // Only support single successor for simplicity
181 if (succ_succs.size() != 1)
182 continue;
183 auto next_succ = *succ_succs.begin();
184 auto next_back = loco::must_cast<luci::CircleNode *>(next_succ);
185 back_fusability = fusability(next_back);
186 back_fusability.pre_scale &= f.scale_forwardable;
187 back_fusability.pre_shift &= f.shift_forwardable;
188 }
189 }
190
191 // Create EqualizePattern based on fusability
192 // ScaleShift
193 // front: fusable_post_shift and fusable_post_scale
194 // back: fusable_pre_shift and fusable_post_shift
195 if (front_fusability.post_shift and front_fusability.post_scale and
196 back_fusability.pre_shift and back_fusability.pre_scale)
197 {
198 res.emplace_back(front->name(), back->name(), EqualizePattern::Type::ScaleShift);
199 }
200 // ShiftOnly
201 // front: fusable_post_shift
202 // back: fusable_pre_shift
203 else if (front_fusability.post_shift and back_fusability.pre_shift)
204 {
205 res.emplace_back(front->name(), back->name(), EqualizePattern::Type::ShiftOnly);
206 }
207 // ScaleOnly
208 // front: fusable_post_scale
209 // back: fusable_pre_scale
210 else if (front_fusability.post_scale and back_fusability.pre_scale)
211 {
212 res.emplace_back(front->name(), back->name(), EqualizePattern::Type::ScaleOnly);
213 }
214 }
215}

222std::vector<EqualizePattern> EqualizePatternFinder::find(loco::Graph *g) const
223{
224 if (g == nullptr)
225 throw std::invalid_argument("g");
226
227 std::vector<EqualizePattern> res;
228 for (auto node : loco::active_nodes(loco::output_nodes(g)))
229 {
230 const auto cnode = loco::must_cast<luci::CircleNode *>(node);
231 if (not _ctx._allow_dup_op)
232 {
233 // We create new Ops (front, back) for each matched pattern, so duplicate
234 // Ops can be created. This condition prevents pattern matching for a node
235 // (front) with multiple successors (backs) to avoid duplicate computation.
236 // TODO Duplicate Ops can be removed if all (pred-succ) pairs are matched
237 // as the same pattern. For example, if (pred-succ1) is ScaleOnly and
238 // (pred-succ2) is ScaleOnly, pred does not need to be duplicated.
239 if (loco::succs(cnode).size() != 1)
240 continue;
241 }
242
243 ::match(cnode, res);
244 }
245
246 return res;
247}
