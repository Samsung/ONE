136template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
137void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::registerTensorInfo(
138 const ir::OperandIndex &ind, const ir::OperandInfo &info)
139{
140 assert(_tensor_mgr->constTensors().size() == 0);
141 assert(_tensor_mgr->nonconstTensors().size() == 0);
142
143 _uses_count_map[ind] = _operands.at(ind).getUses().size();
144
145 // SubTensors
146 assert((_parent_map.count(ind) == 0 || !info.isConstant()) &&
147 "Subtensors of constants are not supported yet.");
148
149 _tensor_info_map.emplace(ind, info);
150}

177template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
178void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::allocate(void)
179{
180 auto lifetime_map = cl_common::createLifetimeMap(_lifetime_seq, _parent_map);
181
182 for (const auto &entry : lifetime_map)
183 {
184 const auto &[use_type, use_index] = entry.second;
185 assert(use_index.valid());
186 if (use_type == UsesType::FIRST)
187 _tensor_mgr->startLifetime(use_index);
188 else
189 _tensor_mgr->finishLifetime(use_index);
190 }
191
192 _tensor_mgr->allocateConsts();
193
194 // TODO Since `_parent_map` is filled for all Concat nodes even if the node this backend uses
195 // After refactoring BackendContext we can uncomment this
196 // assert(_tensor_info_map.size() ==
197 // _tensor_mgr->nonconstTensors().size() + num of constants of _tensor_info_map +
198 // _parent_map.size());
199 _tensor_mgr->allocateNonconsts();
200
201 _tensor_mgr->allocateInternalBufferManager();
202}

210template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
211void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::buildTensors(void)
212{
213 assert(_tensor_mgr->constTensors().size() == 0);
214 assert(_tensor_mgr->nonconstTensors().size() == 0);
215
216 // Normal tensors
217 for (const auto &[ind, info] : _tensor_info_map)
218 {
219 if (_parent_map.count(ind) > 0)
220 continue;
221
222 auto tensor_info = asTensorInfo(info.shape(), info.typeInfo(), true);
223 _tensor_mgr->buildTensor(ind, tensor_info, info.shape().rank(), info.isConstant(),
224 _uses_count_map[ind]);
225 }
226
227 // Subtensors
228 assert(_tensor_mgr->nonconstSubtensors().size() == 0);
229 // TODO Iterate `_parent_map` instead, once the optimizer bug is fixed
230 // `Optimizer` iterates the entire Operations, so there is a bug if iterating _parent_map
231 for (const auto &entry : _tensor_info_map)
232 {
233 const auto &ind = entry.first;
234 if (_parent_map.count(ind) == 0)
235 continue;
236
237 // To make subtensor, parent tensor must be made first
238 // For this condition, use stack
239 // 1) Push one subtensor index to stack (iterate subtensors)
240 // 2) If tensor at stack top is already made, pop and go to 4)
241 // 3) If tensor pushed at 1) is not made, check parent tensor
242 // 3-1) If parent tensor is already made, we can make child tensor
243 // Make child tensor and pop, go to 4)
244 // 3-2) If parent tensor is not made, we can't make child tensor yet
245 // Push parent tensor index to stack and return to 4)
246 // 4) If stack is empty, return to 1), else return to 2)
247 auto &subtensors = _tensor_mgr->nonconstSubtensors();
248
249 std::stack<ir::OperandIndex> stack;
250 stack.push(ind);
251
252 while (!stack.empty())
253 {
254 const auto current = stack.top();
255 const auto &tensor_info = _tensor_info_map.at(current);
256 const auto &parent_info = _parent_map.at(current);
257
258 // Already generated SubTensor
259 if (subtensors.find(current) != subtensors.end())
260 {
261 stack.pop();
262 continue;
263 }
264
265 auto parent = parent_info.parent;
266 std::shared_ptr<T_ITensor> parent_tensor = _tensor_mgr->findTensorAsParent(parent);
267 if (!parent_tensor)
268 {
269 // Cannot find allocated parent tensor: allocate parent first
270 assert(_parent_map.count(parent) > 0);
271 stack.push(parent);
272 continue;
273 }
274 assert(parent_tensor != nullptr);
275
276 // Child's type should be same with parent
277 assert(tensor_info.typeInfo().zero_point() ==
278 parent_tensor->info()->quantization_info().uniform().offset);
279 assert(tensor_info.typeInfo().scale() ==
280 parent_tensor->info()->quantization_info().uniform().scale);
281 assert(tensor_info.typeInfo().type() == parent_tensor->data_type());
282
283 auto shape = asTensorShape(tensor_info.shape(), true);
284 ::arm_compute::Coordinates coordinates = asTensorCoordinate(parent_info.coordinates);
285 _tensor_mgr->buildSubtensor(parent, current, shape, coordinates, tensor_info.shape().rank(),
286 true);
287 stack.pop();
288 }
289 }

292template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
293bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::areSubTensorsOf(
294 const ir::OperandIndex &parent, const ir::OperandIndexSequence &seq)
295{
296 for (const auto &cand : seq)
297 {
298 if (!isSubTensorOf(parent, cand))
299 {
300 return false;
301 }
302 }
303 return true;
304}
305
306template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
307bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::isSubTensorOf(
308 const ir::OperandIndex &parent, const ir::OperandIndex &child)
309{
310 auto itr = _parent_map.find(child);
311 if (itr == _parent_map.end())
312 {
313 return false;
314 }
315
316 return itr->second.parent == parent;
317}

319template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
320ir::OperandIndex
321AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::findRootParent(ir::OperandIndex ind)
322{
323 if (_parent_map.find(ind) == _parent_map.end())
324 return ind;
325
326 auto parent_ind = _parent_map.at(ind).parent;
327 return findRootParent(parent_ind);
328}
