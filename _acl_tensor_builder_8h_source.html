138template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
139void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::registerTensorInfo(
140 const ir::OperandIndex &ind, const ir::OperandInfo &info, ir::Layout backend_layout)
141{
142 assert(_tensor_mgr->constTensors().size() == 0);
143 assert(_tensor_mgr->nonconstTensors().size() == 0);
144
145 _uses_count_map[ind] = _operands.at(ind).getUses().size();
146
147 if (_parent_map.count(ind) == 0)
148 {
149 // Normal Tensors
150 _tensor_info_map.emplace(ind, info);
151 _tensor_layout_map.insert({ind, backend_layout});
152 }
153 else
154 {
155 // SubTensors
156 assert(!info.isConstant() && "Subtensors of constants are not supported yet.");
157
158 // Update offset info and emplace
159 auto &parent_info = _parent_map[ind];
160 const auto &obj = _operands.at(ind);
161 auto parent_index = parent_info.parent;
162 auto &offset = parent_info.coordinates;
163 auto frontend_layout = parent_info.frontend_layout;
164
165 assert(obj.shape().rank() <= ir::Shape::kMaxRank);
166 auto shape = obj.shape();
167 if (_operands.at(parent_index).shape().rank() >= 4 && frontend_layout == ir::Layout::NHWC &&
168 backend_layout == ir::Layout::NCHW)
169 {
170 // Permutation changing layout beyond 4-D is not supported yet
171 const auto parent_rank = _operands.at(parent_index).shape().rank();
172 assert(parent_rank == 4);
173 shape.extendRank(parent_rank);
174 offset = {offset[0], offset[3], offset[1], offset[2]};
175 }
176 else if (_operands.at(parent_index).shape().rank() >= 4 &&
177 frontend_layout == ir::Layout::NCHW && backend_layout == ir::Layout::NHWC)
178 {
179 // Permutation changing layout beyond 4-D is not supported yet
180 const auto parent_rank = _operands.at(parent_index).shape().rank();
181 assert(parent_rank == 4);
182 shape.extendRank(parent_rank);
183 offset = {offset[0], offset[2], offset[3], offset[1]};
184 }
185 auto new_shape = permuteShape(shape, frontend_layout, backend_layout);
186 auto oi = ir::OperandInfo::createStaticInfo(new_shape, obj.typeInfo());
187 _tensor_info_map.emplace(ind, oi);
188 }
189}

216template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
217void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::allocate(void)
218{
219 auto lifetime_map = cl_common::createLifetimeMap(_lifetime_seq, _parent_map);
220
221 for (auto &entry : lifetime_map)
222 {
223 auto &use = entry.second;
224 auto use_type = use.first;
225 auto use_index = use.second;
226 assert(use_index.valid());
227 if (use_type == UsesType::FIRST)
228 _tensor_mgr->startLifetime(use_index);
229 else
230 _tensor_mgr->finishLifetime(use_index);
231 }
232
233 _tensor_mgr->allocateConsts();
234
235 // TODO Since `_parent_map` is filled for all Concat nodes even if the node this backend uses
236 // After refactoring BackendContext we can uncomment this
237 // assert(_tensor_info_map.size() ==
238 // _tensor_mgr->nonconstTensors().size() + num of constants of _tensor_info_map +
239 // _parent_map.size());
240 _tensor_mgr->allocateNonconsts();
241
242 _tensor_mgr->allocateInternalBufferManager();
243}

251template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
252void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::buildTensors(void)
253{
254 assert(_tensor_mgr->constTensors().size() == 0);
255 assert(_tensor_mgr->nonconstTensors().size() == 0);
256
257 // Normal tensors
258 for (auto &entry : _tensor_info_map)
259 {
260 auto ind = entry.first;
261 if (_parent_map.count(ind) > 0)
262 continue;
263
264 const auto &info = entry.second;
265 const auto &backend_layout = _tensor_layout_map[ind];
266 auto tensor_info =
267 asTensorInfo(info.shape(), info.typeInfo(), ir::Layout::UNKNOWN, backend_layout, true);
268 _tensor_mgr->buildTensor(ind, tensor_info, info.shape().rank(), info.isConstant(),
269 _uses_count_map[ind]);
270 }
271
272 // Subtensors
273 assert(_tensor_mgr->nonconstSubtensors().size() == 0);
274 // TODO Iterate `_parent_map` instead, once the optimizer bug is fixed
275 // `Optimizer` iterates the entire Operations, so there is a bug if iterating _parent_map
276 for (auto &entry : _tensor_info_map)
277 {
278 auto ind = entry.first;
279 if (_parent_map.count(ind) == 0)
280 continue;
281
282 // To make subtensor, parent tensor must be made first
283 // For this condition, use stack
284 // 1) Push one subtensor index to stack (iterate subtensors)
285 // 2) If tensor at stack top is already made, pop and go to 4)
286 // 3) If tensor pushed at 1) is not made, check parent tensor
287 // 3-1) If parent tensor is already made, we can make child tensor
288 // Make child tensor and pop, go to 4)
289 // 3-2) If parent tensor is not made, we can't make child tensor yet
290 // Push parent tensor index to stack and return to 4)
291 // 4) If stack is empty, return to 1), else return to 2)
292 auto &subtensors = _tensor_mgr->nonconstSubtensors();
293
294 std::stack<ir::OperandIndex> stack;
295 stack.push(ind);
296
297 while (!stack.empty())
298 {
299 const auto current = stack.top();
300 const auto &tensor_info = _tensor_info_map.at(current);
301 const auto &parent_info = _parent_map.at(current);
302
303 // Already generated SubTensor
304 if (subtensors.find(current) != subtensors.end())
305 {
306 stack.pop();
307 continue;
308 }
309
310 auto parent = parent_info.parent;
311 std::shared_ptr<T_ITensor> parent_tensor = _tensor_mgr->findTensorAsParent(parent);
312 if (!parent_tensor)
313 {
314 // Cannot find allocated parent tensor: allocate parent first
315 assert(_parent_map.count(parent) > 0);
316 stack.push(parent);
317 continue;
318 }
319 assert(parent_tensor != nullptr);
320
321 // Child's type should be same with parent
322 assert(tensor_info.typeInfo().zero_point() ==
323 parent_tensor->info()->quantization_info().uniform().offset);
324 assert(tensor_info.typeInfo().scale() ==
325 parent_tensor->info()->quantization_info().uniform().scale);
326 assert(tensor_info.typeInfo().type() == parent_tensor->data_type());
327
328 // NOTE SubTensor's layout must be the same with layout of parent tensor
329 const auto &root_parent = findRootParent(parent);
330 const auto &backend_layout = _tensor_layout_map[root_parent];
331
332 auto shape = asTensorShape(tensor_info.shape(), ir::Layout::UNKNOWN, backend_layout, true);
333 ::arm_compute::Coordinates coordinates =
334 asTensorCoordinate(parent_info.coordinates, ir::Layout::UNKNOWN, backend_layout);
335 _tensor_mgr->buildSubtensor(parent, current, shape, coordinates, tensor_info.shape().rank(),
336 true);
337 stack.pop();
338 }
339 }

342template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
343bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::areSubTensorsOf(
344 const ir::OperandIndex &parent, const ir::OperandIndexSequence &seq)
345{
346 for (auto &cand : seq)
347 {
348 if (!isSubTensorOf(parent, cand))
349 {
350 return false;
351 }
352 }
353 return true;
354}

356template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
357bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::isSubTensorOf(
358 const ir::OperandIndex &parent, const ir::OperandIndex &child)
359{
360 auto itr = _parent_map.find(child);
361 if (itr == _parent_map.end())
362 {
363 return false;
364 }
365
366 return itr->second.parent == parent;
367}

369template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
370ir::OperandIndex
371AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::findRootParent(ir::OperandIndex ind)
372{
373 if (_parent_map.find(ind) == _parent_map.end())
374 return ind;
375
376 auto parent_ind = _parent_map.at(ind).parent;
377 return findRootParent(parent_ind);
378}
