125template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
126void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::registerTensorInfo(
127 const ir::OperandIndex &ind, const ir::OperandInfo &info)
128{
129 assert(_tensor_mgr->constTensors().size() == 0);
130 assert(_tensor_mgr->nonconstTensors().size() == 0);
131
132 _uses_count_map[ind] = _operands.at(ind).getUses().size();
133
134 // SubTensors
135 assert((_parent_map.count(ind) == 0 || !info.isConstant()) &&
136 "Subtensors of constants are not supported yet.");
137
138 _tensor_info_map.emplace(ind, info);
139}

166template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
167void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::allocate(void)
168{
169 auto lifetime_map = cl_common::createLifetimeMap(_lifetime_seq, _parent_map);
170
171 for (const auto &entry : lifetime_map)
172 {
173 const auto &[use_type, use_index] = entry.second;
174 assert(use_index.valid());
175 if (use_type == UsesType::FIRST)
176 _tensor_mgr->startLifetime(use_index);
177 else
178 _tensor_mgr->finishLifetime(use_index);
179 }
180
181 _tensor_mgr->allocateConsts();
182
183 // TODO Since `_parent_map` is filled for all Concat nodes even if the node this backend uses
184 // After refactoring BackendContext we can uncomment this
185 // assert(_tensor_info_map.size() ==
186 // _tensor_mgr->nonconstTensors().size() + num of constants of _tensor_info_map +
187 // _parent_map.size());
188 _tensor_mgr->allocateNonconsts();
189
190 _tensor_mgr->allocateInternalBufferManager();
191}

199template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
200void AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::buildTensors(void)
201{
202 assert(_tensor_mgr->constTensors().size() == 0);
203 assert(_tensor_mgr->nonconstTensors().size() == 0);
204
205 // Normal tensors
206 for (const auto &[ind, info] : _tensor_info_map)
207 {
208 if (_parent_map.count(ind) > 0)
209 continue;
210
211 auto tensor_info = asTensorInfo(info.shape(), info.typeInfo(), true);
212 _tensor_mgr->buildTensor(ind, tensor_info, info.shape().rank(), info.isConstant(),
213 _uses_count_map[ind]);
214 }
215
216 // Subtensors
217 assert(_tensor_mgr->nonconstSubtensors().size() == 0);
218 // TODO Iterate `_parent_map` instead, once the optimizer bug is fixed
219 // `Optimizer` iterates the entire Operations, so there is a bug if iterating _parent_map
220 for (const auto &entry : _tensor_info_map)
221 {
222 const auto &ind = entry.first;
223 if (_parent_map.count(ind) == 0)
224 continue;
225
226 // To make subtensor, parent tensor must be made first
227 // For this condition, use stack
228 // 1) Push one subtensor index to stack (iterate subtensors)
229 // 2) If tensor at stack top is already made, pop and go to 4)
230 // 3) If tensor pushed at 1) is not made, check parent tensor
231 // 3-1) If parent tensor is already made, we can make child tensor
232 // Make child tensor and pop, go to 4)
233 // 3-2) If parent tensor is not made, we can't make child tensor yet
234 // Push parent tensor index to stack and return to 4)
235 // 4) If stack is empty, return to 1), else return to 2)
236 auto &subtensors = _tensor_mgr->nonconstSubtensors();
237
238 std::stack<ir::OperandIndex> stack;
239 stack.push(ind);
240
241 while (!stack.empty())
242 {
243 const auto current = stack.top();
244 const auto &tensor_info = _tensor_info_map.at(current);
245 const auto &parent_info = _parent_map.at(current);
246
247 // Already generated SubTensor
248 if (subtensors.find(current) != subtensors.end())
249 {
250 stack.pop();
251 continue;
252 }
253
254 auto parent = parent_info.parent;
255 std::shared_ptr<T_ITensor> parent_tensor = _tensor_mgr->findTensorAsParent(parent);
256 if (!parent_tensor)
257 {
258 // Cannot find allocated parent tensor: allocate parent first
259 assert(_parent_map.count(parent) > 0);
260 stack.push(parent);
261 continue;
262 }
263 assert(parent_tensor != nullptr);
264
265 // Child's type should be same with parent
266 assert(tensor_info.typeInfo().zero_point() ==
267 parent_tensor->info()->quantization_info().uniform().offset);
268 assert(tensor_info.typeInfo().scale() ==
269 parent_tensor->info()->quantization_info().uniform().scale);
270 assert(tensor_info.typeInfo().type() == parent_tensor->data_type());
271
272 auto shape = asTensorShape(tensor_info.shape(), true);
273 ::arm_compute::Coordinates coordinates = asTensorCoordinate(parent_info.coordinates);
274 _tensor_mgr->buildSubtensor(parent, current, shape, coordinates, tensor_info.shape().rank(),
275 true);
276 stack.pop();
277 }
278 }

281template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
282bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::areSubTensorsOf(
283 const ir::OperandIndex &parent, const ir::OperandIndexSequence &seq)
284{
285 for (const auto &cand : seq)
286 {
287 if (!isSubTensorOf(parent, cand))
288 {
289 return false;
290 }
291 }
292 return true;
293}
294
295template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
296bool AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::isSubTensorOf(
297 const ir::OperandIndex &parent, const ir::OperandIndex &child)
298{
299 auto itr = _parent_map.find(child);
300 if (itr == _parent_map.end())
301 {
302 return false;
303 }
304
305 return itr->second.parent == parent;
306}

308template <typename T_ITensor, typename T_Tensor, typename T_SubTensor>
309ir::OperandIndex
310AclTensorBuilder<T_ITensor, T_Tensor, T_SubTensor>::findRootParent(ir::OperandIndex ind)
311{
312 if (_parent_map.find(ind) == _parent_map.end())
313 return ind;
314
315 auto parent_ind = _parent_map.at(ind).parent;
316 return findRootParent(parent_ind);
317}
