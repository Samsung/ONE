36template <typename T> class Reader : public feature::Reader<T>
37{
38public:
39 using Strides = ir::FeatureShape;
40 // Construct for buffer and strides
41 Reader(const ir::FeatureShape &shape, const Strides &strides, const T *ptr, size_t len)
42 : _shape{shape}, _strides{strides}, _ptr{reinterpret_cast<const uint8_t *>(ptr)}, _len{len}
43 {
44 UNUSED_RELEASE(len); // Workaround for unused variable in release mode
45 assert(len == static_cast<size_t>(strides.N != 0
46 ? shape.N * strides.N
47 : strides.C != 0 ? shape.C * strides.C
48 : strides.H != 0 ? shape.H * strides.H
49 : shape.W * strides.W));
50 }
51
52 // Construct for backend tensor
53 Reader(backend::ITensor *tensor)
54 : _ptr{tensor->buffer() + tensor->calcOffset({0, 0, 0, 0})}, _len{tensor->total_size()}
55 {
56 assert(tensor->layout() == ir::Layout::NCHW);
57
58 const auto start_offset = tensor->calcOffset({0, 0, 0, 0});
59 auto shape = tensor->getShape();
60 _strides.W = shape.dim(3) == 1 ? 0 : tensor->calcOffset({0, 0, 0, 1}) - start_offset;
61 _strides.H = shape.dim(2) == 1 ? 0 : tensor->calcOffset({0, 0, 1, 0}) - start_offset;
62 _strides.C = shape.dim(1) == 1 ? 0 : tensor->calcOffset({0, 1, 0, 0}) - start_offset;
63 _strides.N = shape.dim(0) == 1 ? 0 : tensor->calcOffset({1, 0, 0, 0}) - start_offset;
64
65 _shape.W = shape.dim(3);
66 _shape.H = shape.dim(2);
67 _shape.C = shape.dim(1);
68 _shape.N = shape.dim(0);
69 }
70
71public:
72 T at(uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) const final
73 {
74 return getRef(batch, ch, row, col);
75 }
76 T at(uint32_t ch, uint32_t row, uint32_t col) const final { return getRef(0, ch, row, col); }
77
78protected:
79 const T &getRef(uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) const
80 {
81 const auto offset = feature_index_to_byte_offset(batch, ch, row, col);
82
83 const T *ptr = reinterpret_cast<const T *>(_ptr + offset);
84
85 return *ptr;
86 }
87
88private:
89 size_t feature_index_to_byte_offset(uint32_t batch, uint32_t ch, uint32_t row, uint32_t col) const
90 {
91 assert(1u * _shape.N > batch); // shape.N > batch
92 assert(1u * _shape.C > ch); // shape.C > ch
93 assert(1u * _shape.H > row); // shape.H > row
94 assert(1u * _shape.W > col); // shape.W > col
95
96 uint32_t res = 0;
97 res += batch * _strides.N;
98 res += ch * _strides.C;
99 res += row * _strides.H;
100 res += col * _strides.W;
101
102 return res;
103 }
104
105private:
106 // TODO Remove _shape
107 ir::FeatureShape _shape;
108 Strides _strides;
109 const uint8_t *_ptr;
110 size_t _len;
111};
