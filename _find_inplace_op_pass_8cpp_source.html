29OMStatus isInplaceOperation(const circle::Operator *op, core::OMRuntimeContext &runtime_context,
30 bool &is_inplace)
31{
32 OMStatus status = Ok;
33
34 const auto op_codes = runtime_context.getCircleOpcodes();
35 uint32_t index = op->opcode_index();
36
37 assert(index < op_codes->size());
38
39 const auto opcode = op_codes->operator[](index);
40
41 switch (opcode->builtin_code())
42 {
43 case circle::BuiltinOperator_ABS:
44 case circle::BuiltinOperator_CEIL:
45 case circle::BuiltinOperator_LOGISTIC:
46 case circle::BuiltinOperator_RESHAPE:
47 case circle::BuiltinOperator_ELU:
48 case circle::BuiltinOperator_EXPAND_DIMS:
49 case circle::BuiltinOperator_EXP:
50 case circle::BuiltinOperator_TANH:
51 case circle::BuiltinOperator_LEAKY_RELU:
52 case circle::BuiltinOperator_LOG:
53 case circle::BuiltinOperator_RELU:
54 case circle::BuiltinOperator_RELU6:
55 case circle::BuiltinOperator_ROUND:
56 case circle::BuiltinOperator_ADD:
57 case circle::BuiltinOperator_MUL:
58 case circle::BuiltinOperator_SUB:
59 case circle::BuiltinOperator_ZEROS_LIKE:
60 {
61 is_inplace = true;
62 break;
63 }
64#if 0 // FIXME: Enable after custom operation is introduced
65 case circle::BuiltinOperator_CUSTOM:
66 {
67 core::OMBuilderCustomID custom_id;
68
69 core::OMBuilderID builderID = core::OMBuilderID::Size;
70 status = core::getCustomOperatorBuilderId(opcode->custom_code(), builderID);
71
72 if (status != Ok)
73 return status;
74
75 status = core::getCustomOperatorByBuilderId(builderID, custom_id);
76
77 if (status != Ok)
78 return status;
79
80 switch (custom_id)
81 {
82 case onert_micro::core::CUSTOM_custom_gru:
83 is_inplace = true;
84 break;
85 default:
86 is_inplace = false;
87 break;
88 }
89 }
90#endif
91 default:
92 is_inplace = false;
93 break;
94 }
95 return status;
96}

98bool isSingleUsageOfTensor(core::OMRuntimeContext &context, const int32_t tensor_index)
99{
100 uint32_t usage_count = 0;
101
102 const auto operators = context.getCircleOperators();
103 for (uint32_t i = 0; i < operators->size(); ++i)
104 {
105 const auto *op = operators->operator[](i);
106 assert(op != nullptr);
107
108 const auto *op_inputs = op->inputs();
109 for (int32_t j = 0; j < op_inputs->size(); ++j)
110 {
111 const auto input_index = op_inputs->operator[](j);
112 if (input_index == tensor_index)
113 {
114 if (++usage_count > 1)
115 return false;
116 }
117 }
118 }
119
120 // Let's check that it is not graph output
121 if (usage_count == 1)
122 {
123 const auto &outputs_indexes = context.getCircleOutputs();
124 bool is_graph_output = (std::find(outputs_indexes->begin(), outputs_indexes->end(),
125 tensor_index) != outputs_indexes->end());
126 if (is_graph_output)
127 return false;
128 }
129
130 return true;
131}

133OMStatus checkInplaceOp(core::OMRuntimeContext &context, const circle::Operator *cur_op,
134 bool &is_inplace)
135{
136 const auto graph_outputs = context.getCircleOutputs();
137 const auto *op_inputs = cur_op->inputs();
138 const auto *op_outputs = cur_op->outputs();
139
140 auto non_const_input_it = op_inputs->begin();
141 while (non_const_input_it != op_inputs->end())
142 {
143 auto dist = std::distance(op_inputs->begin(), non_const_input_it);
144
145 const auto non_const_input_idx = *non_const_input_it;
146
147 if (context.isConstTensor(non_const_input_idx))
148 {
149 non_const_input_it++;
150 continue;
151 }
152
153 // Check single usage of input tensor
154 if (not isSingleUsageOfTensor(context, non_const_input_idx))
155 {
156 is_inplace = false;
157 break;
158 }
159
160 // Let's check single usage of output tensor
161 if (dist >= op_outputs->size() and op_outputs->size() == 1)
162 dist = 0;
163
164 assert(dist < op_outputs->size());
165 if (dist >= op_outputs->size())
166 return UnknownError;
167
168 const auto output_index = op_outputs->operator[](dist);
169 if (not isSingleUsageOfTensor(context, output_index))
170 {
171 is_inplace = false;
172 break;
173 }
174
175 // Check that num elements are equal
176 {
177 const auto *input_non_const_tensor = context.getTensorByIndex(non_const_input_idx);
178 const auto *output_tensor = context.getTensorByIndex(output_index);
179 if (input_non_const_tensor->shape()->size() != output_tensor->shape()->size())
180 {
181 is_inplace = false;
182 break;
183 }
184 }
185
186 // Let's check that output is not a graph output tensor
187 // TODO: check this statement
188 {
189 if (std::find(graph_outputs->begin(), graph_outputs->end(), output_index) !=
190 graph_outputs->end())
191 {
192 is_inplace = false;
193 break;
194 }
195 }
196
197 non_const_input_it++;
198 }
199
200 return Ok;
201}

203OMStatus findInplaceOp(core::OMRuntimeStorage &storage, core::OMRuntimeContext &context,
204 const OMConfig &configs, bool &is_changed)
205{
206 OMStatus status = Ok;
207
208 const core::reader::CircleOperators *operators = context.getCircleOperators();
209
210 for (uint32_t i = 0; i < operators->size(); ++i)
211 {
212 auto kernel_type = storage.getKernelType(i);
213 if (kernel_type == onert_micro::core::Inplace)
214 continue;
215
216 auto cur_op = operators->operator[](i);
217
218 bool is_inplace = false;
219 isInplaceOperation(cur_op, context, is_inplace);
220
221 if (is_inplace == false)
222 continue;
223
224 is_inplace = true;
225
226 status = checkInplaceOp(context, cur_op, is_inplace);
227
228 if (status != Ok)
229 return status;
230
231 if (not is_inplace)
232 continue;
233
234 is_changed = true;
235 storage.setKernelType(i, onert_micro::core::Inplace);
236 }
237
238 return status;
239}

243optimize::OMGraphStatus optimize::onert_micro_FindInplaceOpPass(core::OMRuntimeStorage &storage,
244 core::OMRuntimeContext &context,
245 const OMConfig &configs)
246{
247 bool changed = false;
248
249 OMGraphStatus graph_status = {Unchanged, Ok};
250
251 // If it is train mode, skip inplace optimization
252 // We need all tensors
253 if (configs.train_mode)
254 return graph_status;
255
256 do
257 {
258 changed = false;
259 graph_status.main_status = findInplaceOp(storage, context, configs, changed);
260
261 if (graph_status.main_status != Ok)
262 return graph_status;
263
264 if (changed)
265 graph_status.graph_status = Changed;
266
267 } while (changed);
268
269 return graph_status;
270}
