29OMStatus isInplaceOperation(const circle::Operator *op, core::OMRuntimeContext &runtime_context,
30 bool &is_inplace)
31{
32 OMStatus status = Ok;
33
34 const auto op_codes = runtime_context.getCircleOpcodes();
35 uint32_t index = op->opcode_index();
36
37 assert(index < op_codes->size());
38
39 const auto opcode = op_codes->operator[](index);
40
41 switch (opcode->builtin_code())
42 {
43 case circle::BuiltinOperator_ABS:
44 case circle::BuiltinOperator_CEIL:
45 case circle::BuiltinOperator_LOGISTIC:
46 case circle::BuiltinOperator_RESHAPE:
47 case circle::BuiltinOperator_ELU:
48 case circle::BuiltinOperator_EXPAND_DIMS:
49 case circle::BuiltinOperator_EXP:
50 case circle::BuiltinOperator_TANH:
51 case circle::BuiltinOperator_LEAKY_RELU:
52 case circle::BuiltinOperator_LOG:
53 case circle::BuiltinOperator_RELU:
54 case circle::BuiltinOperator_RELU6:
55 case circle::BuiltinOperator_ROUND:
56 case circle::BuiltinOperator_ADD:
57 case circle::BuiltinOperator_MUL:
58 case circle::BuiltinOperator_SUB:
59 case circle::BuiltinOperator_WHILE:
60 case circle::BuiltinOperator_ZEROS_LIKE:
61 {
62 is_inplace = true;
63 break;
64 }
65 case circle::BuiltinOperator_CUSTOM:
66 {
67 core::OMBuilderCustomID custom_id;
68
69 core::OMBuilderID builderID = core::OMBuilderID::Size;
70 status = core::getCustomOperatorBuilderId(opcode->custom_code(), builderID);
71
72 if (status != Ok)
73 return status;
74
75 status = core::getCustomOperatorByBuilderId(builderID, custom_id);
76
77 if (status != Ok)
78 return status;
79
80 switch (custom_id)
81 {
82 case onert_micro::core::CUSTOM_custom_gru:
83 is_inplace = true;
84 break;
85 default:
86 is_inplace = false;
87 }
88 }
89 default:
90 is_inplace = false;
91 }
92 return status;
93}

95bool isSingleUsageOfTensor(core::OMRuntimeContext &context, const int32_t tensor_index)
96{
97 uint32_t usage_count = 0;
98
99 const auto operators = context.getCircleOperators();
100 for (uint32_t i = 0; i < operators->size(); ++i)
101 {
102 const auto *op = operators->operator[](i);
103 assert(op != nullptr);
104
105 const auto *op_inputs = op->inputs();
106 for (int32_t j = 0; j < op_inputs->size(); ++j)
107 {
108 const auto input_index = op_inputs->operator[](j);
109 if (input_index == tensor_index)
110 {
111 if (++usage_count > 1)
112 return false;
113 }
114 }
115 }
116
117 // Let's check that it is not graph output
118 if (usage_count == 1)
119 {
120 const auto &outputs_indexes = context.getCircleOutputs();
121 bool is_graph_output = (std::find(outputs_indexes->begin(), outputs_indexes->end(),
122 tensor_index) != outputs_indexes->end());
123 if (is_graph_output)
124 return false;
125 }
126
127 return true;
128}

130OMStatus checkInplaceOp(core::OMRuntimeContext &context, const circle::Operator *cur_op,
131 bool &is_inplace)
132{
133 const auto graph_outputs = context.getCircleOutputs();
134 const auto *op_inputs = cur_op->inputs();
135 const auto *op_outputs = cur_op->outputs();
136
137 auto non_const_input_it = op_inputs->begin();
138 while (non_const_input_it != op_inputs->end())
139 {
140 auto dist = std::distance(op_inputs->begin(), non_const_input_it);
141
142 const auto non_const_input_idx = *non_const_input_it;
143
144 if (context.isConstTensor(non_const_input_idx))
145 {
146 non_const_input_it++;
147 continue;
148 }
149
150 // Check single usage of input tensor
151 if (not isSingleUsageOfTensor(context, non_const_input_idx))
152 {
153 is_inplace = false;
154 break;
155 }
156
157 // Let's check single usage of output tensor
158 if (dist >= op_outputs->size() and op_outputs->size() == 1)
159 dist = 0;
160
161 assert(dist < op_outputs->size());
162 if (dist >= op_outputs->size())
163 return UnknownError;
164
165 const auto output_index = op_outputs->operator[](dist);
166 if (not isSingleUsageOfTensor(context, output_index))
167 {
168 is_inplace = false;
169 break;
170 }
171
172 // Check that num elements are equal
173 {
174 const auto *input_non_const_tensor = context.getTensorByIndex(non_const_input_idx);
175 const auto *output_tensor = context.getTensorByIndex(output_index);
176 if (input_non_const_tensor->shape()->size() != output_tensor->shape()->size())
177 {
178 is_inplace = false;
179 break;
180 }
181 }
182
183 // Let's check that output is not a graph output tensor
184 // TODO: check this statement
185 {
186 if (std::find(graph_outputs->begin(), graph_outputs->end(), output_index) !=
187 graph_outputs->end())
188 {
189 is_inplace = false;
190 break;
191 }
192 }
193
194 non_const_input_it++;
195 }
196
197 return Ok;
198}

200OMStatus findInplaceOp(core::OMRuntimeStorage &storage, core::OMRuntimeContext &context,
201 const OMConfig &configs, bool &is_changed)
202{
203 OMStatus status;
204
205 const core::reader::CircleOperators *operators = context.getCircleOperators();
206
207 for (uint32_t i = 0; i < operators->size(); ++i)
208 {
209 auto kernel_type = storage.getKernelType(i);
210 if (kernel_type == onert_micro::core::Inplace)
211 continue;
212
213 auto cur_op = operators->operator[](i);
214
215 bool is_inplace = false;
216 status = isInplaceOperation(cur_op, context, is_inplace);
217 if (status != Ok)
218 return status;
219
220 if (is_inplace == false)
221 continue;
222
223 is_inplace = true;
224
225 status = checkInplaceOp(context, cur_op, is_inplace);
226
227 if (status != Ok)
228 return status;
229
230 if (not is_inplace)
231 continue;
232
233 is_changed = true;
234 storage.setKernelType(i, onert_micro::core::Inplace);
235 }
236
237 return status;
238}

242optimize::OMGraphStatus optimize::onert_micro_FindInplaceOpPass(core::OMRuntimeStorage &storage,
243 core::OMRuntimeContext &context,
244 const OMConfig &configs)
245{
246 bool changed = false;
247
248 OMGraphStatus graph_status = {Unchanged, Ok};
249 do
250 {
251 changed = false;
252 graph_status.main_status = findInplaceOp(storage, context, configs, changed);
253
254 if (graph_status.main_status != Ok)
255 return graph_status;
256
257 if (changed)
258 graph_status.graph_status = Changed;
259
260 } while (changed);
261
262 return graph_status;
263}
