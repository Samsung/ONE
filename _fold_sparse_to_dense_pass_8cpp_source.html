37template <loco::DataType IndexT, loco::DataType ValueT>
38bool fold_sparse_to_dense(luci::CircleSparseToDense *stod)
39{
40 const auto indices = loco::must_cast<luci::CircleNode *>(stod->indices());
41 const auto default_value = loco::must_cast<luci::CircleConst *>(stod->default_value());
42 const auto output_shape = loco::must_cast<luci::CircleConst *>(stod->output_shape());
43
44 bool has_zero = false;
45 for (uint32_t i = 0; i < indices->rank(); i++)
46 {
47 if (indices->dim(i).known() && indices->dim(i).value() == 0)
48 has_zero = true;
49 }
50 if (!has_zero)
51 return false;
52
53 if (default_value->rank() != 0 || default_value->size<ValueT>() != 1)
54 return false;
55
56 auto rank = output_shape->size<IndexT>();
57 std::vector<uint32_t> shape;
58 for (uint32_t i = 0; i < rank; i++)
59 {
60 auto dim = output_shape->at<IndexT>(i);
61 assert(dim >= 0 && dim <= std::numeric_limits<uint32_t>::max());
62 if (!(dim >= 0 && dim <= std::numeric_limits<uint32_t>::max()))
63 return false;
64
65 shape.push_back(dim);
66 }
67
68 auto name = stod->name();
69 assert(name.length() > 0);
70 auto constant = stod->graph()->nodes()->create<luci::CircleConst>();
71 constant->dtype(default_value->dtype());
72 constant->rank(rank);
73 uint32_t dim_size = 1;
74 for (uint32_t i = 0; i < rank; i++)
75 {
76 constant->dim(i).set(shape[i]);
77 dim_size *= shape[i];
78 }
79
80 constant->size<ValueT>(dim_size);
81 const auto value = default_value->scalar<ValueT>();
82 for (uint32_t i = 0; i < dim_size; i++)
83 constant->at<ValueT>(i) = value;
84
85 constant->shape_status(luci::ShapeStatus::VALID);
86 constant->name(name + "_D");
87
88 loco::replace(stod).with(constant);
89
90 return true;
91}

93bool fold_sparse_to_dense(luci::CircleSparseToDense *stod)
94{
95 auto indices = loco::must_cast<luci::CircleNode *>(stod->indices());
96 auto default_value = dynamic_cast<luci::CircleConst *>(stod->default_value());
97 if (not default_value)
98 return false;
99
100 auto output_shape = dynamic_cast<luci::CircleConst *>(stod->output_shape());
101 if (not output_shape)
102 return false;
103
104 // Illegal input check
105 if (indices->dtype() != output_shape->dtype())
106 throw std::runtime_error("indices and output_shape of SparseToDense must have the same dtype");
107
108 // TODO: Support more data types
109 if (indices->dtype() == loco::DataType::S64)
110 {
111 if (default_value->dtype() == loco::DataType::S64)
112 {
113 return fold_sparse_to_dense<loco::DataType::S64, loco::DataType::S64>(stod);
114 }
115 }
116 return false;
117}

127bool FoldSparseToDensePass::run(loco::Graph *g)
128{
129 bool changed = false;
130 for (auto node : loco::active_nodes(loco::output_nodes(g)))
131 {
132 if (auto stod = dynamic_cast<luci::CircleSparseToDense *>(node))
133 {
134 if (fold_sparse_to_dense(stod))
135 changed = true;
136 }
137 }
138
139 return changed;
140}
