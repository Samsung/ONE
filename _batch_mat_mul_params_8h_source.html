28 BatchMatMulParams(const Shape &lhs_shape, const Shape &rhs_shape)
29 {
30 const Shape extended_lhs_shape = Shape::ExtendedShape(5, lhs_shape);

34 batch_dim1 = broadcast_dim(extended_lhs_shape.Dims(1), extended_rhs_shape.Dims(1));

67private:
68 // Determines which dimension is the broadcast dimension.
69 int32_t broadcast_dim(int32_t lhs_dim, int32_t rhs_dim)
70 {
71 if (lhs_dim == rhs_dim)
72 return lhs_dim;
73 if (lhs_dim == 1)
74 return rhs_dim;
75 assert(rhs_dim == 1);
76 return lhs_dim;
77 };
78
79 // Computes the "extent" for iterating on this dimension.
80 // If we are broadcasting, then don't advance (i.e return 0).
81 int extent(const Shape &shape, int x)
82 {
83 if (shape.Dims(x) == 1)
84 {
85 return 0;
86 }
87 int prod = 1;
88 for (int i = x + 1; i < shape.DimensionsCount(); ++i)
89 {
90 prod *= shape.Dims(i);
91 }
92 return prod;
93 };
94};
