55const circle::Operator *operator_match_output(mio::circle::Reader &reader, const int32_t tensor)
56{
57 auto ops = reader.operators();
58
59 for (uint32_t i = 0; i < ops->Length(); ++i)
60 {
61 const auto op = ops->Get(i);
62
63 const std::vector<int32_t> &outputs = mio::circle::as_index_vector(op->outputs());
64
65 for (auto output : outputs)
66 {
67 if (output == tensor)
68 return op;
69 }
70 }
71 return nullptr;
72}

74size_t tensor_buffer_size(mio::circle::Reader &reader, const int32_t tensor_id)
75{
76 auto tensors = reader.tensors();
77
78 if (tensor_id < 0 || tensor_id >= tensors->Length())
79 {
80 throw std::runtime_error("Invalid Tensor ID");
81 }
82
83 auto tensor = tensors->Get(tensor_id);
84 auto buffer_id = tensor->buffer();
85
86 size_t size = reader.buffer_info(buffer_id, nullptr);
87
88 return size;

96void DumpConv2DWeight::run(std::ostream &os, const circle::Model *model)
97{
98 mio::circle::Reader reader(model);
99
100 const uint32_t subgraph_size = reader.num_subgraph();
101
102 for (uint32_t g = 0; g < subgraph_size; g++)
103 {
104 reader.select_subgraph(g);
105 auto ops = reader.operators();
106
107 // dump Conv2D, DepthwiseConv2D and its weight input operator
108 for (uint32_t i = 0; i < ops->Length(); ++i)
109 {
110 const auto op = ops->Get(i);
111 auto bc = reader.builtin_code(op);
112
113 if (bc == circle::BuiltinOperator_CONV_2D || bc == circle::BuiltinOperator_DEPTHWISE_CONV_2D)
114 {
115 const std::vector<int32_t> &inputs = mio::circle::as_index_vector(op->inputs());
116 if (inputs.size() < 2)
117 {
118 throw std::runtime_error("Operator has invalid input");
119 }
120 auto weight_input = inputs[1]; // Tensor ID of weight input
121
122 const auto op_weight = operator_match_output(reader, weight_input);
123 const auto buffer_size = tensor_buffer_size(reader, weight_input);
124
125 std::string weight_op_name = "?";
126
127 if (op_weight == nullptr && buffer_size > 0)
128 {
129 weight_op_name = "CONST";
130 }
131 else if (op_weight != nullptr)
132 {
133 weight_op_name = reader.opcode_name(op_weight);
134 }
135
136 auto op_name = reader.opcode_name(op);
137 os << op_name << "," << weight_op_name << std::endl;
138 }
139 }
140 }
141}

148void DumpOperatorVersion::run(std::ostream &os, const circle::Model *model)
149{
150 std::map<std::string, int32_t> op_version_map;
151
152 mio::circle::Reader reader(model);
153
154 // This assert is subject to be changed later
155 assert(reader.num_subgraph() == 1);
156 reader.select_subgraph(0);
157
158 auto ops = reader.operators();
159
160 // Dump operators' version
161 for (uint32_t i = 0; i < ops->Length(); ++i)
162 {
163 const auto op = ops->Get(i);
164
165 auto op_name = reader.opcode_name(op);
166 auto op_version = reader.opcodes().at(op->opcode_index())->version();
167
168 if (op_version_map.find(op_name) == op_version_map.end() ||
169 op_version_map[op_name] < op_version)
170 op_version_map[op_name] = op_version;
171 }
172
173 for (auto op : op_version_map)
174 {
175 os << op.first << "," << op.second << std::endl;
176 }
177}
