35template <typename T>
36class OMTensorData
37{
38 using QuantData = OMQuantizationData<T>;
39 using QuantDataPtr = std::unique_ptr<QuantData>;
40
41private:
42 T *_data = nullptr;
43 size_t _size = 0;
44 QuantDataPtr _quant = {};
45 std::map<size_t, float> _out_float_data = {};
46
47public:
48 OMTensorData(T *data, const circle::Tensor *tensor)
49 : _data(data)
50 {
51 assert(data != nullptr);
52 assert(tensor != nullptr);
53
54 _size = OMRuntimeShape(tensor).flatSize();
55
56 if (IsQuantized<T>)
57 {
58 _quant = std::make_unique<QuantData>(_data, tensor);
59 }
60 }
61
62public:
63 template <typename U = T>
64 constexpr static bool IsInt8 = std::is_same<std::decay_t<U>, int8_t>::value;
65
66 template <typename U = T>
67 constexpr static bool IsUInt8 = std::is_same<std::decay_t<U>, uint8_t>::value;
68
69 template <typename U = T>
70 constexpr static bool IsQuantized = IsInt8<U> || IsUInt8<U>;
71
72public:
73 template <typename U = T>
74 constexpr static bool HasConstData = std::is_const<U>::value;
75
76 template <typename U = T>
77 constexpr static bool HasNonConstData = !HasConstData<U>;
78
79 template <typename U = T>
80 constexpr static bool HasQuantizedConstData = IsQuantized<U> && HasConstData<U>;
81
82 template <typename U = T>
83 constexpr static bool HasQuantizedNonConstData = IsQuantized<U> && HasNonConstData<U>;
84
85public:
86 bool IsNull() const noexcept
87 {
88 return (_data == nullptr) || (_size == 0);
89 }
90
91public:
92 T* Get() const noexcept
93 {
94 return _data;
95 }
96
97 T& At(size_t idx) const
98 {
99 CheckIndex(idx);
100 return _data[idx];
101 }
102
103 void SetAt(size_t idx, T value)
104 {
105 CheckIndex(idx);
106 _data[idx] = value;
107 }
108
109 template <typename U = T>
110 std::enable_if_t<IsQuantized<U>> SetAt(size_t idx, float value)
111 {
112 CheckIndex(idx);
113 _quant->SetDataAt(idx, value);
114 }
115
116public:
117 template <typename U = T>
118 std::enable_if_t<!IsQuantized<U>, T&> ValueAt(size_t idx) const
119 {
120 return At(idx);
121 }
122
123 template <typename U = T>
124 std::enable_if_t<HasQuantizedConstData<U>, float> ValueAt(size_t idx) const
125 {
126 CheckIndex(idx);
127 return _quant->DataAt(idx);
128 }
129
130 template <typename U = T>
131 std::enable_if_t<HasQuantizedNonConstData<U>, float> ValueAt(size_t idx) const
132 {
133 CheckIndex(idx);
134
135 if (_out_float_data.count(idx) == 0)
136 return 0.f;
137
138 return _out_float_data.at(idx);
139 }
140
141public:
142 template <typename U = T>
143 std::enable_if_t<!IsQuantized<U>> SetValueAt(size_t idx, T value)
144 {
145 SetAt(idx, value);
146 }
147
148 template <typename U = T>
149 std::enable_if_t<HasQuantizedNonConstData<U>> SetValueAt(size_t idx, float value)
150 {
151 CheckIndex(idx);
152 _out_float_data[idx] = value;
153 }
154
155private:
156 bool CheckIndex(size_t idx) const
157 {
158 assert(idx < _size);
159 return idx < _size;
160 }
161};

165template <class T, class RuntimeKernel>
166OMTensorData<const T> MakeInputData(RuntimeKernel& rtk, size_t input_idx)
167{
168 const T *data = utils::castInputData<T>(rtk.inputs_data[input_idx]);
169 const circle::Tensor *tensor = rtk.inputs[input_idx];
170
171 assert(data != nullptr);
172 assert(tensor != nullptr);
173
174 OMTensorData<const T> result(data, tensor);
175 return result;
176}
177
178template <class T, class RuntimeKernel>
179OMTensorData<T> MakeOutputData(RuntimeKernel& rtk, size_t output_idx)
180{
181 T *data = utils::castOutputData<T>(rtk.outputs_data[output_idx]);
182 const circle::Tensor *tensor = rtk.outputs[output_idx];
183
184 assert(data != nullptr);
185 assert(tensor != nullptr);
186
187 OMTensorData<T> result(data, tensor);
188 return result;
189}
