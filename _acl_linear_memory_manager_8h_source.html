33template <typename T_MemoryManager, typename T_PoolManager, typename T_LifetimeManager>
34std::shared_ptr<T_MemoryManager> createMemoryManager()
35{
36 std::shared_ptr<T_LifetimeManager> lifetime_mgr = std::make_shared<T_LifetimeManager>();
37 std::shared_ptr<T_PoolManager> pool_mgr = std::make_shared<T_PoolManager>();
38
39 std::shared_ptr<T_MemoryManager> mem_mgr =
40 std::make_shared<T_MemoryManager>(lifetime_mgr, pool_mgr);
41 return mem_mgr;
42}

44template <typename T_ITensor, typename T_Tensor, typename T_SubTensor, typename T_MemoryManager,
45 typename T_PoolManager, typename T_LifetimeManager, typename T_Allocator,
46 typename T_MemoryGroup>
47class AclLinearMemoryManager : public AclMemoryManager<T_ITensor, T_Tensor, T_SubTensor>
48{
49public:
50 AclLinearMemoryManager()
51 : _allocator{nullptr},
52 _io_manager{createMemoryManager<T_MemoryManager, T_PoolManager, T_LifetimeManager>()},
53 _io_group{std::make_shared<T_MemoryGroup>(_io_manager)}
54 {
55 // DO NOTHING
56 }
57
58 virtual ~AclLinearMemoryManager() = default;
59
60 void allocate(void) override
61 {
62 _allocator = std::make_shared<T_Allocator>();
63 _io_manager->populate(*_allocator, 1);
64 _io_group->acquire();
65 }
66
67 void deallocate(void) override
68 {
69 _io_group->release();
70 _io_manager->clear();
71 }
72
73 void startLifetime(const ir::OperandIndex &ind) override
74 {
75 auto &tensors = this->tensors();
76 assert(tensors.find(ind) != tensors.end());
77
78 auto tensor = tensors[ind];
79 assert(tensor->handle());
80
81 _io_group->manage(tensor->handle());
82 }
83
84 void finishLifetime(const ir::OperandIndex &ind) override
85 {
86 auto &tensors = this->tensors();
87 assert(tensors.find(ind) != tensors.end());
88
89 auto tensor = tensors[ind];
90 assert(tensor->allocator());
91
92 tensor->allocator()->allocate();
93 }
94
95private:
96 std::shared_ptr<T_Allocator> _allocator;
97 std::shared_ptr<T_MemoryManager> _io_manager;
98 std::shared_ptr<T_MemoryGroup> _io_group;
99};
