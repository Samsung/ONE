32constexpr uint32_t node_not_assigned = std::numeric_limits<int32_t>::max();
33
34bool isExecutableNode(const luci::CircleNode *node)
35{
36 switch (node->opcode())
37 {
38 // The following nodes denote outputs of multiple-output nodes.
39 // The list is synchronized with the same list from luci-interpreter/src/loader/GraphLoader.cpp
40 case luci::CircleOpcode::CIRCLEIFOUT:
41 case luci::CircleOpcode::CIRCLESPLITOUT:
42 case luci::CircleOpcode::CIRCLESPLITVOUT:
43 case luci::CircleOpcode::CIRCLEUNPACKOUT:
44 case luci::CircleOpcode::CIRCLEWHILEOUT:
45 return false;
46 default:
47 return true;
48 }
49}
50
51bool isTensorProducingNode(const luci::CircleNode *node)
52{
53 switch (node->opcode())
54 {
55 // The following nodes are multiple-output nodes. They do not produce tensors, the tensors
56 // are produced by the corresponding *Out nodes instead.
57 // The list is synchronized with the same list from luci-interpreter/src/loader/GraphLoader.cpp
58 case luci::CircleOpcode::IF:
59 case luci::CircleOpcode::SPLIT:
60 case luci::CircleOpcode::UNPACK:
61 return false;
62 default:
63 return true;
64 }
65}

67// Create allocation node part for current circle node for json allocation info file
68void create_allocation_node(Json::Value &allocations_node,
69 AllocationNodeInformation &alloca_node_inform, uint32_t alive_till_max,
70 luci::CircleNode *circle_node)
71{
72 Json::Value allocation_node;
73 if (alloca_node_inform.size == 0)
74 return;
75
76 allocation_node["offset"] = alloca_node_inform.offset;
77 allocation_node["size"] = alloca_node_inform.size;
78 allocation_node["alive_from"] = alloca_node_inform.first_node;
79
80 if (alloca_node_inform.last_node == node_not_assigned)
81 allocation_node["alive_till"] = alive_till_max + 1;
82 else
83 allocation_node["alive_till"] = alloca_node_inform.last_node;
84
85 allocation_node["origin"] = circle_node->name();
86
87 allocations_node.append(allocation_node);
88}

121void ExecutionPlanner::write_execution_plan(uint32_t order_offset)
122{
123 _required_size = get_offsets_with_greedy_by_size();
124
125 int32_t counter_ops = 0;
126 for (uint32_t i = 0; i < _ordered_nodes.size(); i++)
127 {
128 const auto circle_node = dynamic_cast<luci::CircleNode *>(_ordered_nodes[i]);
129 if (circle_node->opcode() != luci::CircleOpcode::CIRCLECONST and
130 circle_node->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
131 {
132 luci::CircleNodeExecutionPlan execution_plan(counter_ops + order_offset, _offsets[i]);
133 luci::add_execution_plan(loco::must_cast<luci::CircleNode *>(_ordered_nodes[i]),
134 execution_plan);
135 counter_ops++;
136 }
137 }

140void ExecutionPlanner::make_execution_plan_onert_micro_split_buffer()
141{
142 LOGGER(l);
143
144 const auto input_size = _graph->inputs()->size();
145 const auto output_size = _graph->outputs()->size();
146
147 // Make execution plan for inputs
148 _ordered_nodes = loco::input_nodes(_graph);
149 write_execution_plan(0);
150 dump_inform();
151 VERBOSE(l, 0) << "Input graph buffer required memory = " << _required_size << std::endl;
152
153 // Clear structures for next buffer
154 _ordered_nodes.clear();
155 _alloc_node_inform_vector.clear();
156 _dealloc_node.clear();
157 _alloc_node.clear();
158 _offsets.clear();
159 _required_size = 0;
160
161 // Make execution plan for outputs
162 _ordered_nodes = loco::output_nodes(_graph);
163 write_execution_plan(input_size);
164 dump_inform();
165 VERBOSE(l, 0) << "Output graph buffer required memory = " << _required_size << std::endl;
166
167 // Clear structures for next buffer
168 _ordered_nodes.clear();
169 _alloc_node_inform_vector.clear();
170 _dealloc_node.clear();
171 _alloc_node.clear();
172 _offsets.clear();
173 _required_size = 0;
174
175 // Make execution plan for intermediates calculations
176 get_default_execution_order_plan_without_inputs_and_outputs();
177 write_execution_plan(input_size + output_size);
178 dump_inform();
179 VERBOSE(l, 0) << "Main graph buffer required memory = " << _required_size << std::endl;
180}

182void ExecutionPlanner::make_execution_plan_onert_micro_common_buffer()
183{
184 LOGGER(l);
185
186 get_default_execution_order_plan();
187 _required_size = get_offsets_with_greedy_by_size();
188
189 // Find prev nodes for output nodes (actual graph output node, not luci::CircleOutput)
190 const auto output_nodes = loco::output_nodes(const_cast<loco::Graph *>(_graph));
191 std::vector<loco::Node *> output_prev_nodes;
192 for (const auto output_node : output_nodes)
193 {
194 const auto prev_nodes = loco::preds(output_node);
195 std::copy(prev_nodes.begin(), prev_nodes.end(), std::back_inserter(output_prev_nodes));
196 }
197 const auto output_nodes_size = output_prev_nodes.size();
198
199 const auto inputs_nodes = loco::input_nodes(_graph);
200 const auto input_nodes_size = inputs_nodes.size();
201
202 int32_t counter_ops = 0;
203 for (uint32_t i = 0; i < _ordered_nodes.size(); i++)
204 {
205 const auto circle_node = dynamic_cast<luci::CircleNode *>(_ordered_nodes[i]);
206 // First write to input nodes
207 if (circle_node->opcode() == luci::CircleOpcode::CIRCLEINPUT)
208 {
209 // Find input_position for proper position in execution order
210 const auto input_position = std::distance(
211 inputs_nodes.begin(), std::find(inputs_nodes.begin(), inputs_nodes.end(), circle_node));
212 luci::CircleNodeExecutionPlan execution_plan(input_position, _offsets[i]);
213 luci::add_execution_plan(loco::must_cast<luci::CircleNode *>(_ordered_nodes[i]),
214 execution_plan);
215 }
216 // Second write to actual output nodes (not luci::CircleOutput)
217 else if (std::find(output_prev_nodes.begin(), output_prev_nodes.end(), circle_node) !=
218 output_prev_nodes.end())
219 {
220 // Find output_position for proper position in execution order
221 const auto output_position =
222 std::distance(output_prev_nodes.begin(),
223 std::find(output_prev_nodes.begin(), output_prev_nodes.end(), circle_node));
224 luci::CircleNodeExecutionPlan execution_plan(input_nodes_size + output_position, _offsets[i]);
225 luci::add_execution_plan(loco::must_cast<luci::CircleNode *>(_ordered_nodes[i]),
226 execution_plan);
227 }
228 // Finally write to all intermediate nodes
229 else if (circle_node->opcode() != luci::CircleOpcode::CIRCLECONST and
230 circle_node->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
231 {
232 luci::CircleNodeExecutionPlan execution_plan(
233 counter_ops + input_nodes_size + output_nodes_size, _offsets[i]);
234 luci::add_execution_plan(loco::must_cast<luci::CircleNode *>(_ordered_nodes[i]),
235 execution_plan);
236 counter_ops++;
237 }
238 }
239
240 dump_inform();
241 VERBOSE(l, 0) << "Buffer required memory = " << _required_size << std::endl;
242}

279void ExecutionPlanner::create_json_allocation_file(const std::string &json_path)
280{
281 Json::Value main_tree;
282 Json::Value segments_node;
283 Json::Value allocations_node;
284
285 uint32_t alive_till_max = 0;
286
287 // Find max dealloc value to assign to nodes with node_not_assigned value
288 for (const auto elem : _dealloc_node)
289 {
290 if (alive_till_max < elem and elem != node_not_assigned)
291 alive_till_max = elem;
292 }
293
294 for (auto &alloc_node_inform : _alloc_node_inform_vector)
295 {
296 const auto node_num = alloc_node_inform.node_num;
297 const auto circle_node = loco::must_cast<luci::CircleNode *>(_ordered_nodes[node_num]);
298
299 create_allocation_node(allocations_node, alloc_node_inform, alive_till_max, circle_node);
300 }
301
302 // Create segment part
303 Json::Value segment_node;
304 segment_node["name"] = "Segment1";
305 segment_node["allocations"] = allocations_node;
306 segments_node.append(segment_node);
307
308 main_tree["schema_version"] = 1;
309 main_tree["segments"] = segments_node;
310
311 Json::StreamWriterBuilder builder;
312 const std::unique_ptr<Json::StreamWriter> writer(builder.newStreamWriter());
313
314 // Write to json file
315 std::ofstream out;
316 out.open(json_path);
317 if (out.is_open())
318 {
319 writer->write(main_tree, &out);
320 }
321}

329void ExecutionPlanner::get_default_execution_order_plan_without_inputs_and_outputs()
330{
331 // Get all nodes
332 _ordered_nodes = loco::postorder_traversal(loco::output_nodes(const_cast<loco::Graph *>(_graph)));
333
334 // Get real output nodes (not luci::CircleOutput)
335 const auto output_nodes = loco::output_nodes(const_cast<loco::Graph *>(_graph));
336 std::vector<loco::Node *> output_prev_nodes;
337 for (const auto output_node : output_nodes)
338 {
339 const auto prev_nodes = loco::preds(output_node);
340 std::copy(prev_nodes.begin(), prev_nodes.end(), std::back_inserter(output_prev_nodes));
341 }
342
343 // Remove input and real output nodes from _ordered_nodes
344 _ordered_nodes.erase(
345 std::remove_if(_ordered_nodes.begin(), _ordered_nodes.end(),
346 [&output_prev_nodes](auto node) {
347 const auto circle_node = dynamic_cast<luci::CircleNode *>(node);
348
349 return circle_node->opcode() == luci::CircleOpcode::CIRCLEINPUT or
350 circle_node->opcode() == luci::CircleOpcode::CIRCLEOUTPUT or
351 std::find(output_prev_nodes.begin(), output_prev_nodes.end(), node) !=
352 output_prev_nodes.end();
353 }),
354 _ordered_nodes.end());
355}

357void ExecutionPlanner::get_usage_interval()
358{
359 // Initialize vectors of first and last nodes for usage interval
360 _alloc_node.assign(_ordered_nodes.size(), node_not_assigned);
361 _dealloc_node.assign(_ordered_nodes.size(), node_not_assigned);
362
363 // Vector for count usages
364 std::vector<int> usages_counts(_ordered_nodes.size(), 0);
365
366 auto allocate = [this](uint32_t node, uint32_t tensor) {
367 if (_alloc_node[tensor] != node_not_assigned)
368 {
369 return;
370 }
371 assert(_dealloc_node[tensor] == node_not_assigned);
372 _alloc_node[tensor] = node;
373 };
374
375 auto deallocate = [this](uint32_t node, uint32_t tensor) {
376 assert(_dealloc_node[tensor] == node_not_assigned);
377 _dealloc_node[tensor] = node;
378 };
379
380 // Increase refcounts for graph outputs and inputs nodes
381 for (auto &output_node : output_nodes(_graph))
382 {
383 auto it = std::find(_ordered_nodes.begin(), _ordered_nodes.end(), output_node);
384 if (it == _ordered_nodes.end())
385 continue;
386 size_t index = std::distance(_ordered_nodes.begin(), it);
387 usages_counts[index]++;
388 }
389
390 for (auto &input_node : input_nodes(_graph))
391 {
392 auto it = std::find(_ordered_nodes.begin(), _ordered_nodes.end(), input_node);
393 if (it == _ordered_nodes.end())
394 continue;
395 size_t index = std::distance(_ordered_nodes.begin(), it);
396 usages_counts[index]++;
397 allocate(0, index);
398 }
399
400 // Increase refcounts of usage for all nodes in _ordered_nodes vector
401 for (uint32_t i = 0; i < _ordered_nodes.size(); i++)
402 {
403 const auto node = _ordered_nodes.at(i);
404 auto prev_nodes = preds(node);
405 for (auto &prev_node : prev_nodes)
406 {
407 auto it = std::find(_ordered_nodes.begin(), _ordered_nodes.end(), prev_node);
408 size_t index = std::distance(_ordered_nodes.begin(), it);
409 usages_counts[index]++;
410 }
411 }
412
413 for (uint32_t i = 0; i < _ordered_nodes.size(); i++)
414 {
415 const auto node = _ordered_nodes.at(i);
416 auto prev_nodes = preds(node);
417 if (const auto *const_node = dynamic_cast<const luci::CircleConst *>(node))
418 {
419 allocate(0, i);
420 }
421 else if (!isExecutableNode(loco::must_cast<luci::CircleNode *>(node)))
422 {
423 // If current node is multi output node than begin life time for current node should start
424 // when prev node start live
425 auto it = std::find(_ordered_nodes.begin(), _ordered_nodes.end(), *prev_nodes.begin());
426 size_t index = std::distance(_ordered_nodes.begin(), it);
427 allocate(index, i);
428 }
429 else
430 {
431 allocate(i, i);
432 }
433
434 for (auto &prev_node : prev_nodes)
435 {
436 auto it = std::find(_ordered_nodes.begin(), _ordered_nodes.end(), prev_node);
437 size_t index = std::distance(_ordered_nodes.begin(), it);
438 usages_counts[index]--;
439 if (usages_counts[index] == 0)
440 {
441 deallocate(i, index);
442 }
443 }
444 }
445}

447uint32_t ExecutionPlanner::get_offsets_with_greedy_by_size()
448{
449 get_usage_interval();
450 auto required_size = greedy_by_size_approach();
451
452 _offsets.resize(_ordered_nodes.size());
453 for (const auto &alloc : _alloc_node_inform_vector)
454 {
455 // Fill offsets vector: first go offset for current node and then should go offsets for
456 // temporaries tensors
457 if (alloc.is_temp)
458 {
459 _offsets[alloc.node_num].push_back(alloc.offset);
460 }
461 else
462 {
463 _offsets[alloc.node_num].insert(_offsets[alloc.node_num].begin(), alloc.offset);
464 }
465 }
466 return required_size;
467}

469uint32_t ExecutionPlanner::greedy_by_size_approach()
470{
471 size_t result_size = 0;
472 create_alloc_node_inform_vector();
473 std::vector<AllocationNodeInformation> ordered_alloc_inform;
474 for (auto &current_node : _alloc_node_inform_vector)
475 {
476 if (current_node.size == 0)
477 {
478 current_node.offset = 0;
479 continue;
480 }
481 const uint32_t offsetNotAssigned = std::numeric_limits<uint32_t>::max();
482 size_t best_offset = offsetNotAssigned;
483 uint32_t best_offset_fit = offsetNotAssigned;
484
485 uint32_t current_offset = 0;
486
487 for (const auto &alloc_inform : ordered_alloc_inform)
488 {
489 if ((alloc_inform.last_node < current_node.first_node ||
490 alloc_inform.first_node > current_node.last_node))
491 {
492 continue;
493 }
494
495 if (current_offset + current_node.size <= alloc_inform.offset &&
496 alloc_inform.offset - current_offset < best_offset_fit)
497 {
498 best_offset = current_offset;
499 best_offset_fit = alloc_inform.offset - current_offset;
500 }
501 current_offset = std::max(current_offset, alloc_inform.offset + alloc_inform.size);
502 }
503 if (best_offset == offsetNotAssigned)
504 {
505 best_offset = current_offset;
506 }
507
508 result_size = std::max(result_size, best_offset + current_node.size);
509 current_node.offset = best_offset;
510
511 auto insertion_it =
512 std::upper_bound(ordered_alloc_inform.begin(), ordered_alloc_inform.end(), current_node);
513 ordered_alloc_inform.insert(insertion_it, current_node);
514 }
515 return result_size;
516}

518void ExecutionPlanner::create_alloc_node_inform_vector()
519{
520 auto node_compare = [this](const AllocationNodeInformation &alloc_1,
521 const AllocationNodeInformation &alloc_2) {
522 auto idx1 = alloc_1.node_num;
523 auto idx2 = alloc_2.node_num;
524
525 if (this->_alloc_node[idx1] == 0 && this->_dealloc_node[idx1] == node_not_assigned)
526 {
527 if (this->_alloc_node[idx2] == 0 && this->_dealloc_node[idx2] == node_not_assigned)
528 {
529 return idx1 < idx2;
530 }
531 return true;
532 }
533 if (this->_alloc_node[idx2] == 0 && this->_dealloc_node[idx2] == node_not_assigned)
534 {
535 return false;
536 }
537
538 auto size_1 = alloc_1.size;
539 auto size_2 = alloc_2.size;
540
541 if (size_1 != size_2)
542 {
543 return size_1 > size_2;
544 }
545 return this->_alloc_node[idx1] < this->_alloc_node[idx2];
546 };
547
548 _alloc_node_inform_vector.resize(_ordered_nodes.size());
549
550 for (size_t i = 0; i < _ordered_nodes.size(); i++)
551 {
552 auto circle_node = loco::must_cast<luci::CircleNode *>(_ordered_nodes[i]);
553 auto node_size = 1;
554 for (uint32_t axis = 0; axis < circle_node->rank(); ++axis)
555 {
556 node_size *= circle_node->dim(axis).value();
557 }
558 node_size *= size(circle_node->dtype());
559
560 _alloc_node_inform_vector[i].node_num = i;
561 _alloc_node_inform_vector[i].first_node = _alloc_node[i];
562 _alloc_node_inform_vector[i].last_node = _dealloc_node[i];
563
564 const auto *const_node = dynamic_cast<const luci::CircleConst *>(circle_node);
565 if (circle_node->opcode() == luci::CircleOpcode::CIRCLEINPUT && not _is_allocate_inputs)
566 {
567 _alloc_node_inform_vector[i].size = 0;
568 }
569 else if (circle_node->opcode() == luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
570 {
571 _alloc_node_inform_vector[i].size = 0;
572 }
573 else if (const_node && not _is_allocate_consts)
574 {
575 _alloc_node_inform_vector[i].size = 0;
576 }
577 else if (!isTensorProducingNode(circle_node))
578 {
579 _alloc_node_inform_vector[i].size = 0;
580 }
581 else
582 {
583 _alloc_node_inform_vector[i].size = node_size;
584 }
585
586 // Scratchpad If needed
587 std::vector<uint32_t> scratchpad_sizes;
588 if (_is_allocate_scratchpads)
589 {
590 switch (circle_node->opcode())
591 {
592 case luci::CircleOpcode::AVERAGE_POOL_2D:
593 {
594 const auto avg_pool = loco::must_cast<const luci::CircleAveragePool2D *>(circle_node);
595 scratchpad_sizes.push_back(
596 _scratchpad_helper->ComputeScratchpadSizeAveragePool2d(avg_pool));
597 break;
598 }
599 case luci::CircleOpcode::BATCH_MATMUL:
600 {
601 const auto batch_mat_mul = loco::must_cast<const luci::CircleBatchMatMul *>(circle_node);
602 scratchpad_sizes = _scratchpad_helper->ComputeScratchpadSizeBatchMatMul(batch_mat_mul);
603 break;
604 }
605 case luci::CircleOpcode::CONV_2D:
606 {
607 const auto conv = loco::must_cast<const luci::CircleConv2D *>(circle_node);
608 scratchpad_sizes.push_back(_scratchpad_helper->ComputeScratchpadSizeConv2d(conv));
609 break;
610 }
611 case luci::CircleOpcode::DEPTHWISE_CONV_2D:
612 {
613 const auto depthwise_conv =
614 loco::must_cast<const luci::CircleDepthwiseConv2D *>(circle_node);
615 scratchpad_sizes.push_back(
616 _scratchpad_helper->ComputeScratchpadSizeDepthwiseConv2d(depthwise_conv));
617 break;
618 }
619 case luci::CircleOpcode::SVDF:
620 {
621 const auto svdf = loco::must_cast<const luci::CircleSVDF *>(circle_node);
622 scratchpad_sizes = _scratchpad_helper->ComputeScratchpadSizeSVDF(svdf);
623 break;
624 }
625 default:
626 break;
627 }
628 }
629
630 for (const auto scratchpad_size : scratchpad_sizes)
631 {
632 if (scratchpad_size > 0)
633 {
634 AllocationNodeInformation temp_alloc;
635
636 temp_alloc.size = scratchpad_size;
637 temp_alloc.first_node = i - 1;
638 temp_alloc.last_node = i + 1;
639 temp_alloc.node_num = i;
640 temp_alloc.is_temp = true;
641
642 _alloc_node_inform_vector.push_back(temp_alloc);
643 _alloc_node.push_back(i);
644 _dealloc_node.push_back(i);
645 }
646 }
647 }
648 // Sort _alloc_node_inform_vector with node_compare for the greedy by size approach.
649 std::sort(_alloc_node_inform_vector.begin(), _alloc_node_inform_vector.end(), node_compare);
650}

652void ExecutionPlanner::dump_inform()
653{
654 LOGGER(l);
655 uint32_t max_breadth = 0;
656
657 for (uint32_t i = 0; i < _ordered_nodes.size(); i++)
658 {
659 auto current_node_it = std::find_if(
660 _alloc_node_inform_vector.begin(), _alloc_node_inform_vector.end(),
661 [i](const AllocationNodeInformation &x) { return x.node_num == i && !x.is_temp; });
662 for (uint32_t j = 0; j < _ordered_nodes.size(); j++)
663 {
664 auto first_node = _alloc_node[j];
665 auto last_node = _dealloc_node[j];
666
667 auto it = std::find_if(
668 _alloc_node_inform_vector.begin(), _alloc_node_inform_vector.end(),
669 [j](const AllocationNodeInformation &x) { return x.node_num == j && !x.is_temp; });
670 if (i >= first_node && i <= last_node)
671 {
