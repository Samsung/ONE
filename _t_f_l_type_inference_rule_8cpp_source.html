28struct TypeInferenceAlgorithm final : public locoex::TFLNodeVisitor<loco::DataType>
29{
30 loco::DataType visit(const locoex::TFLAdd *node) final { return loco::dtype_get(node->x()); }
31
32 loco::DataType visit(const locoex::TFLAveragePool2D *node) final

34 return loco::dtype_get(node->value());
35 }
36
37 loco::DataType visit(const locoex::TFLConcatenation *node) final
38 {
39 // TODO Support when TFLConcatenation has 0 input
40 assert(node->numValues() > 0);
41
42 for (uint32_t i = 1; i < node->numValues(); ++i)
43 assert(loco::dtype_get(node->values(i - 1)) == loco::dtype_get(node->values(i)));

129bool TFLTypeInferenceRule::infer(const loco::Node *node, loco::DataType &dtype) const
130{
131 assert(node->dialect() == TFLDialect::get());
132
133 TypeInferenceAlgorithm alg;
134
135 dtype = loco::must_cast<const TFLNode *>(node)->accept(&alg);
136 assert(dtype != loco::DataType::Unknown);
137
138 return true;
139}
