31luci::CircleConst *create_div_const_with_new_value(luci::CircleConst *div_const,
32 luci::CircleConst *mul_const, float new_value)

34 assert(div_const); // FIX_CALLER_UNLESS
35 assert(div_const->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
36 assert(div_const->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
37
38 auto new_div_const = div_const->graph()->nodes()->create<luci::CircleConst>();
39 new_div_const->dtype(loco::DataType::FLOAT32);
40 new_div_const->size<loco::DataType::FLOAT32>(1);
41 new_div_const->rank(1);
42 new_div_const->dim(0) = 1;
43 new_div_const->at<loco::DataType::FLOAT32>(0) = new_value;
44 new_div_const->shape_status(luci::ShapeStatus::VALID);
45 new_div_const->name(div_const->name() + ";" + mul_const->name());
46
47 luci::add_origin(new_div_const, luci::composite_origin(
48 {luci::get_origin(div_const), luci::get_origin(mul_const)}));
49
50 return new_div_const;
51}

101bool fuse_mul_with_div(luci::CircleDiv *div)
102{
103 auto div_const = dynamic_cast<luci::CircleConst *>(div->x());
104 if (not div_const)
105 return false;
106
107 if (div_const->dtype() != loco::DataType::FLOAT32)
108 return false;
109
110 if (div_const->size<loco::DataType::FLOAT32>() != 1)
111 return false;
112
113 auto mul = dynamic_cast<luci::CircleMul *>(div->y());
114 if (not mul)
115 return false;
116
117 auto mul_const = dynamic_cast<luci::CircleConst *>(mul->y());
118 if (not mul_const)
119 return false;
120
121 if (mul_const->dtype() != loco::DataType::FLOAT32)
122 return false;
123
124 if (mul_const->size<loco::DataType::FLOAT32>() != 1)
125 return false;
126
127 const auto div_value = div_const->at<loco::DataType::FLOAT32>(0);
128 const auto mul_value = mul_const->at<loco::DataType::FLOAT32>(0);
129
130 if (mul_value == 0)
131 return false;
132
133 const auto new_value = div_value / mul_value;
134
135 auto new_div_const = create_div_const_with_new_value(div_const, mul_const, new_value);
136
137 div->x(new_div_const);
138
139 div->y(mul->x());
140
141 return true;
142}

170bool fuse_mul_with_div_to_mul(luci::CircleDiv *div)
171{
172 if (div->fusedActivationFunction() != luci::FusedActFunc::NONE)
173 return false;
174
175 luci::CircleMul *mul = nullptr;
176 luci::CircleConst *div_const = nullptr;
177 if (not luci::fill(&mul, &div_const).with_args_of(div))
178 return false;
179
180 if (mul->fusedActivationFunction() != luci::FusedActFunc::NONE)
181 return false;
182
183 if (div_const->dtype() != loco::DataType::FLOAT32)
184 return false;
185 // TODO support other shape
186 if (div_const->size<loco::DataType::FLOAT32>() != 1)
187 return false;
188
189 luci::CircleNode *mul_input = nullptr;
190 luci::CircleConst *mul_const = nullptr;
191 if (not luci::fill(&mul_input, &mul_const).with_commutative_args_of(mul))
192 return false;
193
194 if (mul_const->dtype() != loco::DataType::FLOAT32)
195 return false;
196 // TODO support other shape
197 if (mul_const->size<loco::DataType::FLOAT32>() != 1)
198 return false;
199
200 const auto mul_value = mul_const->at<loco::DataType::FLOAT32>(0);
201 const auto div_value = div_const->at<loco::DataType::FLOAT32>(0);
202 const auto new_value = mul_value / div_value;
203 auto new_mul_const = create_mul_const_with_new_value(mul_const, div_const, new_value);
204
205 auto new_mul = div->graph()->nodes()->create<luci::CircleMul>();
206 new_mul->fusedActivationFunction(luci::FusedActFunc::NONE);
207 new_mul->x(mul_input);
208 new_mul->y(new_mul_const);
209 new_mul->name(mul->name());
210 luci::add_origin(new_mul, luci::composite_origin({luci::get_origin(div), luci::get_origin(mul)}));
211
212 replace(div).with(new_mul);
213
214 return true;
215}

219bool FuseMulWithDivPass::run(loco::Graph *g)
220{
221 bool changed = false;
222 for (auto node : loco::active_nodes(loco::output_nodes(g)))
223 {
224 auto div = dynamic_cast<luci::CircleDiv *>(node);
225 if (not div)
226 continue;
227
228 if (fuse_mul_with_div(div))
229 changed = true;
230
231 if (fuse_mul_with_div_to_mul(div))
232 changed = true;
233 }
234
235 return changed;
236}
