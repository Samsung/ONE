30// Return a new CircleConst with a new value
31luci::CircleConst *create_div_const_with_new_value(luci::CircleConst *div_const,
32 luci::CircleConst *mul_const, float new_value)
33{
34 assert(div_const); // FIX_CALLER_UNLESS
35 assert(div_const->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
36 assert(div_const->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
37
38 auto new_div_const = div_const->graph()->nodes()->create<luci::CircleConst>();
39 new_div_const->dtype(loco::DataType::FLOAT32);
40 new_div_const->size<loco::DataType::FLOAT32>(1);
41 new_div_const->rank(1);
42 new_div_const->dim(0) = 1;
43 new_div_const->at<loco::DataType::FLOAT32>(0) = new_value;
44 new_div_const->shape_status(luci::ShapeStatus::VALID);
45 new_div_const->name(div_const->name() + ";" + mul_const->name());
46
47 luci::add_origin(new_div_const, luci::composite_origin(
48 {luci::get_origin(div_const), luci::get_origin(mul_const)}));
49
50 return new_div_const;
51}

101bool fuse_mul_with_div_to_div(luci::CircleDiv *div)
102{
103 if (div->fusedActivationFunction() != luci::FusedActFunc::NONE)
104 return false;
105
106 luci::CircleConst *div_const = nullptr;
107 luci::CircleMul *mul = nullptr;
108 if (not luci::fill(&div_const, &mul).with_args_of(div))
109 return false;
110
111 if (div_const->dtype() != loco::DataType::FLOAT32)
112 return false;
113
114 if (div_const->size<loco::DataType::FLOAT32>() != 1)
115 return false;
116
117 if (mul->fusedActivationFunction() != luci::FusedActFunc::NONE)
118 return false;
119
120 luci::CircleNode *mul_input = nullptr;
121 luci::CircleConst *mul_const = nullptr;
122 if (not luci::fill(&mul_input, &mul_const).with_commutative_args_of(mul))
123 return false;
124
125 if (mul_const->dtype() != loco::DataType::FLOAT32)
126 return false;
127
128 if (mul_const->size<loco::DataType::FLOAT32>() != 1)
129 return false;
130
131 const auto div_value = div_const->at<loco::DataType::FLOAT32>(0);
132 const auto mul_value = mul_const->at<loco::DataType::FLOAT32>(0);
133
134 if (mul_value == 0)
135 return false;
136
137 const auto new_value = div_value / mul_value;
138
139 auto new_div_const = create_div_const_with_new_value(div_const, mul_const, new_value);
140 auto new_div = div->graph()->nodes()->create<luci::CircleDiv>();
141 new_div->fusedActivationFunction(luci::FusedActFunc::NONE);
142 new_div->x(new_div_const);
143 new_div->y(mul_input);
144 new_div->name(div->name());
145 luci::add_origin(new_div, luci::composite_origin({luci::get_origin(div), luci::get_origin(mul)}));
146
147 replace(div).with(new_div);
148
149 return true;
150}

178bool fuse_mul_with_div_to_mul(luci::CircleDiv *div)
179{
180 if (div->fusedActivationFunction() != luci::FusedActFunc::NONE)
181 return false;
182
183 luci::CircleMul *mul = nullptr;
184 luci::CircleConst *div_const = nullptr;
185 if (not luci::fill(&mul, &div_const).with_args_of(div))
186 return false;
187
188 if (mul->fusedActivationFunction() != luci::FusedActFunc::NONE)
189 return false;
190
191 if (div_const->dtype() != loco::DataType::FLOAT32)
192 return false;
193 // TODO support other shape
194 if (div_const->size<loco::DataType::FLOAT32>() != 1)
195 return false;
196
197 luci::CircleNode *mul_input = nullptr;
198 luci::CircleConst *mul_const = nullptr;
199 if (not luci::fill(&mul_input, &mul_const).with_commutative_args_of(mul))
200 return false;
201
202 if (mul_const->dtype() != loco::DataType::FLOAT32)
203 return false;
204 // TODO support other shape
205 if (mul_const->size<loco::DataType::FLOAT32>() != 1)
206 return false;
207
208 const auto mul_value = mul_const->at<loco::DataType::FLOAT32>(0);
209 const auto div_value = div_const->at<loco::DataType::FLOAT32>(0);
210 const auto new_value = mul_value / div_value;
211 auto new_mul_const = create_mul_const_with_new_value(mul_const, div_const, new_value);
212

218 luci::add_origin(new_mul, luci::composite_origin({luci::get_origin(div), luci::get_origin(mul)}));

231 {

233 if (not div)
234 continue;
235
236 if (fuse_mul_with_div_to_div(div))
237 changed = true;
238
239 if (fuse_mul_with_div_to_mul(div))
240 changed = true;
241 }
242
243 return changed;
244}
