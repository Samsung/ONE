29void PermutationIOPass::run()
30{
31 if (_options.input_layout.size() == 0 && _options.output_layout.size() == 0 &&
32 _options.input_type.size() == 0 && _options.output_type.size() == 0)
33 return;
34
35 for (auto i = ir::IOIndex{0}; i < ir::IOIndex{_graph.getInputs().size()}; i++)
36 {
37 if (_options.input_layout.count(i) == 0 && _options.input_type.count(i) == 0)
38 continue;
39
40 const auto index = _graph.getInputs().at(i);
41 const auto &type = _options.input_type.count(i) > 0 ? _options.input_type.at(i)
42 : _graph.operands().at(index).typeInfo();
43 const auto layout =
44 _options.input_layout.count(i) > 0 ? _options.input_layout.at(i) : ir::Layout::NHWC;
45
46 insertInputPermute(index, type, layout);
47 }
48
49 for (auto i = ir::IOIndex{0}; i < ir::IOIndex{_graph.getOutputs().size()}; i++)
50 {
51 if (_options.output_layout.count(i) == 0 && _options.output_type.count(i) == 0)
52 continue;
53
54 const auto index = _graph.getOutputs().at(i);
55 const auto &type = _options.output_type.count(i) > 0 ? _options.output_type.at(i)
56 : _graph.operands().at(index).typeInfo();
57 const auto layout =
58 _options.output_layout.count(i) > 0 ? _options.output_layout.at(i) : ir::Layout::NHWC;
59
60 insertOutputPermute(index, type, layout);
61 }
62}

64void PermutationIOPass::insertInputPermute(const ir::OperandIndex &index, const ir::TypeInfo &type,
65 const ir::Layout &from_layout)
66{
67 assert(from_layout == ir::Layout::NCHW || from_layout == ir::Layout::NHWC);
68 const auto &origin_operand = _graph.operands().at(index);
69
70 // Check nothing to permute
71 if (origin_operand.typeInfo() == type && from_layout == ir::Layout::NHWC)
72 return;
73
74 // Update graph operand
75 auto input_shape = from_layout == ir::Layout::NCHW
76 ? ir::convertShape(origin_operand.shape(), ir::PermuteType::NHWC_TO_NCHW)
77 : origin_operand.shape();
78 auto input_operand_index = _graph.addOperand(input_shape, type);
79 _graph.getInputs().replace(index, input_operand_index);
80
81 // Update graph operation
82 using Permute = ir::operation::Permute;
83 auto permute_type =
84 from_layout == ir::Layout::NCHW ? ir::PermuteType::NCHW_TO_NHWC : ir::PermuteType::SAME;
85 auto permute_node = std::make_unique<Permute>(input_operand_index, index, permute_type);
86 auto node_index = _graph.operations().push(std::move(permute_node));
87
88 // Update use/def info
89 auto &new_input = _graph.operands().at(input_operand_index);
90 new_input.setDef(origin_operand.getDef());
91 new_input.insertUse(node_index);
92 _graph.operands().at(index).setDef(node_index);
93
94 VERBOSE(PermuteIOPass) << "Permute Op inserted for an input, node index : " << node_index
95 << std::endl;
96 VERBOSE(PermuteIOPass) << " - Input (inserted) Operand : " << input_operand_index << std::endl;
97 VERBOSE(PermuteIOPass) << " - Output(original) Operand : " << index << std::endl;
98}

100void PermutationIOPass::insertOutputPermute(const ir::OperandIndex &index, const ir::TypeInfo &type,
101 const ir::Layout &to_layout)
102{
103 assert(to_layout == ir::Layout::NCHW || to_layout == ir::Layout::NHWC);
104 auto &origin_operand = _graph.operands().at(index);
105
106 // Check nothing to permute
107 if (origin_operand.typeInfo() == type && to_layout == ir::Layout::NHWC)
108 return;
109
110 // Update graph operand
111 auto output_shape = to_layout == ir::Layout::NCHW
112 ? ir::convertShape(origin_operand.shape(), ir::PermuteType::NHWC_TO_NCHW)
113 : origin_operand.shape();
114 auto output_operand_index = _graph.addOperand(output_shape, type);
115 _graph.getOutputs().replace(index, output_operand_index);
116
117 // Update graph operation
118 using Permute = ir::operation::Permute;
119 auto permute_type =
120 to_layout == ir::Layout::NCHW ? ir::PermuteType::NHWC_TO_NCHW : ir::PermuteType::SAME;
121 auto permute_node = std::make_unique<Permute>(index, output_operand_index, permute_type);
122 auto node_index = _graph.operations().push(std::move(permute_node));
123
124 // Update use/def info
125 auto &new_output = _graph.operands().at(output_operand_index);
126 new_output.setDef(node_index);
127 assert(_graph.operands().at(index).getUses().size() == 0);
128 origin_operand.insertUse(node_index);
129
130 VERBOSE(PermuteIOPass) << "Permute Op inserted for an output, node index : " << node_index
131 << std::endl;
132 VERBOSE(PermuteIOPass) << " - Input (original) Operand : " << index << std::endl;
133 VERBOSE(PermuteIOPass) << " - Output(inserted) Operand : " << output_operand_index << std::endl;
134}
