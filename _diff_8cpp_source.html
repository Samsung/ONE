56void DiffSummary::notify(const nnfw::misc::tensor::Index &index, float expected, float obtained)
57{
58 const auto abs_diff_value = std::fabs(expected - obtained);
59
60 if (max_abs_diff_value < abs_diff_value)
61 {
62 max_abs_diff_index = index;
63 max_abs_diff_value = abs_diff_value;
64 max_abs_diff_expected = expected;
65 max_abs_diff_obtained = obtained;
66 }
67
68 const auto rel_diff_value = nnfw::misc::fp32::relative_diff(expected, obtained);
69
70 if (max_rel_diff_value < rel_diff_value)
71 {
72 max_rel_diff_index = index;
73 max_rel_diff_value = rel_diff_value;
74 max_rel_diff_expected = expected;
75 max_rel_diff_obtained = obtained;
76 }
77}

79template <typename T>
80bool TfLiteInterpMatchApp::compareSingleTensorView(const nnfw::tflite::TensorView<T> &expected,
81 const nnfw::tflite::TensorView<T> &obtained,
82 int id) const
83{
84 std::vector<nnfw::misc::tensor::Diff<T>> diffs;
85 assert(expected.shape() == obtained.shape());
86
87 using nnfw::misc::tensor::Index;
88 using nnfw::misc::tensor::zip;
89
90 zip(expected.shape(), expected, obtained)
91 << [&](const Index &index, T expected_value, T obtained_value) {
92 if (expected_value != obtained_value)
93 {
94 diffs.emplace_back(index, expected_value, obtained_value);
95 }
96 };
97
98 // TODO Unify summary generation code
99 if (diffs.size() == 0)
100 {
101 std::cout << " Tensor #" << id << ": MATCHED" << std::endl;
102 }
103 else
104 {
105 std::cout << " Tensor #" << id << ": UNMATCHED" << std::endl;
106 std::cout << " " << diffs.size() << " diffs are detected" << std::endl;
107 }
108
109 if (diffs.size() > 0 && _verbose != 0)
110 {
111 std::cout << " ---- Details ---" << std::endl;
112 for (const auto &diff : diffs)
113 {
114 std::cout << " Diff at [" << nnfw::misc::tensor::IndexFormatter(diff.index) << "]"
115 << std::endl;
116 std::cout << " expected: " << diff.expected << std::endl;
117 std::cout << " obtained: " << diff.obtained << std::endl;
118 }
119 }
120
121 return diffs.size() == 0;
122}

195bool TfLiteInterpMatchApp::run(TfLiteInterpreter &expected, TfLiteInterpreter &obtained) const
196{
197 auto output_count = TfLiteInterpreterGetOutputTensorCount(&expected);
198 assert(output_count == TfLiteInterpreterGetOutputTensorCount(&obtained));
199
200 bool all_matched = true;
201
202 using Comparator = std::function<bool(int32_t, const TfLiteTensor *, const TfLiteTensor *)>;
203
204 std::map<TfLiteType, Comparator> comparators;
205
206 comparators[kTfLiteUInt8] = [this](int32_t id, const TfLiteTensor *expected_tensor,
207 const TfLiteTensor *obtained_tensor) {
208 const auto expected_view = nnfw::tflite::TensorView<uint8_t>::make(expected_tensor);
209 const auto obtained_view = nnfw::tflite::TensorView<uint8_t>::make(obtained_tensor);
210
211 return compareSingleTensorView(expected_view, obtained_view, id);
212 };
213
214 comparators[kTfLiteInt32] = [this](int32_t id, const TfLiteTensor *expected_tensor,
215 const TfLiteTensor *obtained_tensor) {
216 const auto expected_view = nnfw::tflite::TensorView<int32_t>::make(expected_tensor);
217 const auto obtained_view = nnfw::tflite::TensorView<int32_t>::make(obtained_tensor);
218
219 return compareSingleTensorView(expected_view, obtained_view, id);
220 };
221
222 comparators[kTfLiteFloat32] = [this](int32_t id, const TfLiteTensor *expected_tensor,
223 const TfLiteTensor *obtained_tensor) {
224 const auto expected_view = nnfw::tflite::TensorView<float>::make(expected_tensor);
225 const auto obtained_view = nnfw::tflite::TensorView<float>::make(obtained_tensor);
226
227 return compareSingleTensorView(expected_view, obtained_view, id);
228 };
229
230 comparators[kTfLiteBool] = [this](int32_t id, const TfLiteTensor *expected_tensor,
231 const TfLiteTensor *obtained_tensor) {
232 const auto expected_view = nnfw::tflite::TensorView<bool>::make(expected_tensor);
233 const auto obtained_view = nnfw::tflite::TensorView<bool>::make(obtained_tensor);
234
235 return compareSingleTensorView(expected_view, obtained_view, id);
236 };
237
238 for (int32_t idx = 0; idx < output_count; idx++)
239 {
240 auto const expected_tensor = TfLiteInterpreterGetOutputTensor(&expected, idx);
241 auto const obtained_tensor = TfLiteInterpreterGetOutputTensor(&obtained, idx);
242 auto const tensor_type = TfLiteTensorType(expected_tensor);
243 assert(tensor_type == TfLiteTensorType(obtained_tensor));
244
245 auto it = comparators.find(tensor_type);
246
247 if (it == comparators.end())
248 {
249 throw std::runtime_error{"Not supported output type"};
250 }
251
252 const auto &comparator = it->second;
253
254 if (!comparator(idx, expected_tensor, obtained_tensor))
255 {
256 all_matched = false;
257 }
258 }
259
260 return all_matched;
261}
