73void ParallelExecutor::executeImpl()
74{
75 bool dynamic_input_exists = hasDynamicInput();
76
77 // Init scheduler
78 // TODO Consider to have distinct backend set in GraphLowerInfo
79 BackendSet backends;
80 _lowered_graph->lower_info().operation.iterate(
81 [&](const ir::OperationIndex &, const compiler::OperationLowerInfo &lower_info) {
82 backends.add(lower_info.backend());
83 });
84 _scheduler = std::make_unique<ParallelScheduler>(backends);
85
86 assert(noWaitingJobs());
87
88 // Execution setup
89 _waiting_jobs.swap(_finished_jobs); // Move finished jobs to waiting jobs
90
91 for (uint32_t i = 0; i < _waiting_jobs.size(); ++i)
92 {
93 VERBOSE(ParallelExecutor) << i << ": " << _input_info[i] << std::endl;
94 if (_input_info[i] == 0)
95 {
96 emplaceToReadyJobs(i);
97 }
98 }
99 assert(!_ready_jobs.empty()); // Cannot begin if there is no initial jobs
100
101 VERBOSE(ParallelExecutor) << "INITIAL JOBS : " << _ready_jobs.size() << std::endl;
102
103 auto profiling_subg_index = _tracing_ctx->getSubgraphIndex(&_graph);
104
105 _subject.notifySubgraphBegin(profiling_subg_index);
106
107 while (true)
108 {
109 std::unique_lock<std::mutex> lock{_mu_jobs};
110
111 if (_ready_jobs.empty())
112 {
113 _cv_jobs.wait(lock, [this] { return !_ready_jobs.empty() || noWaitingJobs(); });
114 // Check finish condition
115 if (_ready_jobs.empty() && noWaitingJobs())
116 {
117 break;
118 }
119 }
120
121 auto job = std::move(_ready_jobs.begin()->second);
122 _ready_jobs.erase(_ready_jobs.begin());
123
124 lock.unlock();
125
126 VERBOSE(ParallelExecutor) << "Assigning fn " << job->index() << std::endl;
127
128 auto job_index = job->index();
129 auto op_ind = _job_to_op[job_index];
130 auto backend = _lowered_graph->lower_info().operation.at(op_ind).backend();
131 auto setup = [&, op_ind, backend]() {
132 _subject.notifyJobBegin(this, profiling_subg_index, op_ind, backend);
133 };
134 auto teardown = [&, job_index, op_ind, backend]() {
135 _subject.notifyJobEnd(this, profiling_subg_index, op_ind, backend);
136 notify(job_index);
137 };
138
139 job->fn_seq()->initRunning();
140
141 // dynamic tensor setting
142 bool handle_dynamic_tensor =
143 _lowered_graph->getHasDynamicTensor(op_ind) || dynamic_input_exists;
144 job->fn_seq()->enableDynamicShapeInferer(handle_dynamic_tensor);
145
146 _scheduler->assign(std::make_unique<HookFunction>(job->fn_seq(), setup, teardown), backend);
147 _finished_jobs[job_index] = std::move(job);
148 }
149
150 assert(noWaitingJobs());
151
152 // Wait for all the jobs done
153 _scheduler->finish();
154 _subject.notifySubgraphEnd(profiling_subg_index);
155
156 // Reset input info for the next execution
157 _input_info = _initial_input_info;
158}
