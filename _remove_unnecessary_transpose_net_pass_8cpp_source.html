100void TaggedShapeAnalyzer::init_shape_with_tag(const luci::CircleNode *in_tensor)
101{
102 _shape.clear();
103 uint8_t tag = START_TAG;
104
105 for (uint32_t i = 0; i < in_tensor->rank(); i++)
106 {
107 TaggedShapeAnalyzer::Dim dim;
108 {
109 dim.value = in_tensor->dim(i).value();
110 if (dim.value != 1)
111 dim.tags.push_back(tag++);
112 }
113 _shape.push_back(dim);
114 }
115}

132template <loco::DataType PermType>
133void TaggedShapeAnalyzer::analyze_transpose(const luci::CircleTranspose *transpose_node)
134{
135 const luci::CircleConst *perm_node = loco::must_cast<luci::CircleConst *>(transpose_node->perm());
136 assert(perm_node->dtype() == PermType);
137
138 TaggedShapeAnalyzer::Shape new_shape;
139 const auto size = perm_node->size<PermType>();
140 for (uint32_t i = 0; i < size; i++)
141 {
142 auto perm_idx = perm_node->at<PermType>(i);
143 new_shape.push_back(_shape.at(perm_idx));
144 }
145 _shape = new_shape;
146}

163template <loco::DataType ReshapeType>
164bool TaggedShapeAnalyzer::analyze_reshape(const luci::CircleReshape *reshape_node)
165{
166 const luci::CircleConst *shape_node = loco::must_cast<luci::CircleConst *>(reshape_node->shape());
167 assert(shape_node->dtype() == ReshapeType);
168
169 // At least one element must be in reshape's output-tensor.
170 if (shape_node->size<ReshapeType>() <= 0)
171 return false;
172
173 // Create new_shape based on reshape_node/shape
174 Shape new_shape;
175 for (uint32_t i = 0; i < shape_node->size<ReshapeType>(); i++)
176 {
177 TaggedShapeAnalyzer::Dim dim;
178 dim.value = shape_node->at<ReshapeType>(i);
179
180 new_shape.push_back(dim);
181 }
182
183 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
184 uint32_t old_shape_start_idx = 0;
185 uint32_t old_shape_end_idx = 1;
186 uint32_t old_shape_product = _shape[old_shape_start_idx].value;
187
188 auto expand_range = [&]() -> bool {
189 if (old_shape_end_idx >= _shape.size())
190 return false;
191
192 old_shape_product *= _shape[old_shape_end_idx].value;
193 old_shape_end_idx++;
194 return true;
195 };
196
197 auto move_to_next_range = [&]() -> bool {
198 if (old_shape_end_idx >= _shape.size())
199 return false;
200
201 old_shape_start_idx = old_shape_end_idx;
202 old_shape_end_idx++;
203 old_shape_product = _shape[old_shape_start_idx].value;
204 return true;
205 };
206
207 // Add tags from '_shape' to the 'new_shape'
208 uint32_t new_shape_idx = 0;
209 while (new_shape_idx < new_shape.size())
210 {
211 Dim &target_dim = new_shape[new_shape_idx];
212
213 // Ignore dim == 1
214 if (target_dim.value == 1)
215 {
216 new_shape_idx++;
217 continue;
218 }
219
220 while (old_shape_product < target_dim.value)
221 {
222 if (expand_range() == false)
223 break;
224 }
225
226 if (old_shape_product != target_dim.value)
227 return false;
228
229 assert(old_shape_product == target_dim.value);
230 for (uint32_t idx = old_shape_start_idx; idx < old_shape_end_idx; idx++)
231 {
232 const auto &old_tags = _shape[idx].tags;
233 target_dim.tags.insert(target_dim.tags.end(), old_tags.begin(), old_tags.end());
234 }
235
236 new_shape_idx++;
237 move_to_next_range();
238 }
239 _shape = new_shape;
240 return true;
241}

243bool TaggedShapeAnalyzer::verify_tag() const
244{
245 // check whether tags in _shape are incremental
246 uint8_t tag = START_TAG;
247 for (const auto &dim : _shape)
248 {
249 for (const auto &t : dim.tags)
250 {
251 if (t == tag)
252 tag++;
253 else
254 return false;
255 }
256 }
257 return true;
258}

261template <loco::DataType DType>
262bool TaggedShapeAnalyzer::can_remove_transposes(const luci::CircleTranspose *f_tr,
263 const luci::CircleReshape *m_rs,
264 const luci::CircleTranspose *b_tr)
265{
266 assert(loco::must_cast<luci::CircleConst *>(f_tr->perm())->dtype() == DType);
267 assert(loco::must_cast<luci::CircleConst *>(m_rs->shape())->dtype() == DType);
268 assert(loco::must_cast<luci::CircleConst *>(b_tr->perm())->dtype() == DType);
269
270 const luci::CircleNode *in_tensor = loco::must_cast<luci::CircleNode *>(f_tr->a());
271
272 init_shape_with_tag(in_tensor);
273
274 analyze_transpose<DType>(f_tr);
275
276 if (not analyze_reshape<DType>(m_rs))
277 return false;
278
279 analyze_transpose<DType>(b_tr);
280
281 if (not verify_tag())
282 return false;
283
284 return true;
285}

291template <loco::DataType ShapeType>
292luci::CircleReshape *create_reshape_node(loco::Graph *graph,
293 const luci::CircleTranspose *front_transpose,
294 const luci::CircleReshape *mid_reshape,
295 const luci::CircleTranspose *back_transpose)
296{
297 std::string composed_name =
298 front_transpose->name() + ";" + mid_reshape->name() + ";" + back_transpose->name();
299
300 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> src_origin{luci::get_origin(front_transpose),
301 luci::get_origin(mid_reshape),
302 luci::get_origin(back_transpose)};
303 auto const composed_origin = luci::composite_origin(src_origin);
304
305 auto shape_node = graph->nodes()->create<luci::CircleConst>();
306 {
307 shape_node->dtype(ShapeType);
308 shape_node->rank(1);
309 shape_node->dim(0).set(back_transpose->rank());
310
311 shape_node->size<ShapeType>(back_transpose->rank());
312 for (uint32_t i = 0; i < back_transpose->rank(); i++)
313 {
314 shape_node->at<ShapeType>(i) = back_transpose->dim(i).value();
315 }
316 shape_node->shape_status(luci::ShapeStatus::VALID);
317 shape_node->name(composed_name + "/shape");
318 luci::add_origin(shape_node, composed_origin);
319 }
320
321 auto reshape_node = graph->nodes()->create<luci::CircleReshape>();
322 {
323 reshape_node->name(composed_name);
324 reshape_node->tensor(front_transpose->a());
325 reshape_node->shape(shape_node);
326 luci::add_origin(reshape_node, composed_origin);
327 }
328 return reshape_node;
329}

331bool remove_unnecessary_transpose(luci::CircleTranspose *node)
332{
333 // find 'front_transpose - mid_reshape - back_transpose' pattern
334 const auto back_transpose = node;
335 const auto mid_reshape = dynamic_cast<luci::CircleReshape *>(back_transpose->a());
336 {
337 if (mid_reshape == nullptr)
338 return false;
339 }
340 const auto front_transpose = dynamic_cast<luci::CircleTranspose *>(mid_reshape->tensor());
341 {
342 if (not front_transpose)
343 return false;
344 }
345
346 // check perm and shape are CircleConst node and its' datatype is S32
347 const auto back_perm = dynamic_cast<luci::CircleConst *>(back_transpose->perm());
348 {
349 if (back_perm == nullptr)
350 return false;
351
352 if (back_perm->dtype() != loco::DataType::S32)
353 return false;
354 }
355 const auto shape = dynamic_cast<luci::CircleConst *>(mid_reshape->shape());
356 {
357 if (shape == nullptr)
358 return false;
359
360 if (shape->dtype() != loco::DataType::S32)
361 return false;
362 }
363 const auto front_perm = dynamic_cast<luci::CircleConst *>(front_transpose->perm());
364 {
365 if (front_perm == nullptr)
366 return false;
367
368 if (front_perm->dtype() != loco::DataType::S32)
369 return false;
370 }
371
372 // for now, handle only rank reduction equal (not expansion) cases
373 const auto output_rank = back_transpose->rank();
374 const auto input_rank = front_transpose->rank();
375 if (input_rank < output_rank)
376 return false;
377
378 // analyze pattern to check this pass is applicable
379 TaggedShapeAnalyzer analyzer;
380 if (not analyzer.can_remove_transposes<loco::DataType::S32>(front_transpose, mid_reshape,
381 back_transpose))
382 return false;
383
384 // repalce with new_node
385 luci::CircleReshape *new_node = create_reshape_node<loco::DataType::S32>(
386 node->graph(), front_transpose, mid_reshape, back_transpose);
387
388 replace(node).with(new_node);
389
390 return true;
391}

434bool RemoveUnnecessaryTransposeNetPass::run(loco::Graph *g)
435{
436 bool changed = false;
437
438 for (auto node : loco::active_nodes(loco::output_nodes(g)))
439 {
440 if (auto transpose_node = dynamic_cast<luci::CircleTranspose *>(node))
441 {
442 if (remove_unnecessary_transpose(transpose_node))
443 changed = true;
444 }
445 }
446
447 return changed;
448}
