32bool extract_shape(const luci::CircleNode *node, std::vector<int32_t> &shape)
33{
34 uint32_t max_i32 = static_cast<uint32_t>(std::numeric_limits<int32_t>::max());
35
36 auto rank = node->rank();
37 for (auto i = 0u; i < rank; ++i)
38 {
39 uint32_t v = node->dim(i).value();
40 CHECK_OR_FALSE(v <= max_i32)
41 shape.push_back(static_cast<int32_t>(v));
42 }
43 return true;
44};

46bool extract_const(const luci::CircleConst *const_node, std::vector<int32_t> &values)
47{
48 auto dtype = const_node->dtype();
49
50 if (dtype == loco::DataType::S32)
51 {
52 auto size = const_node->size<loco::DataType::S32>();
53 for (auto i = 0u; i < size; ++i)
54 {
55 int32_t v = const_node->at<loco::DataType::S32>(i);
56 values.push_back(v);
57 }
58 }
59 else if (dtype == loco::DataType::S64)
60 {
61 int64_t max_i32 = static_cast<int64_t>(std::numeric_limits<int32_t>::max());
62 int64_t min_i32 = static_cast<int64_t>(std::numeric_limits<int32_t>::lowest());
63
64 auto size = const_node->size<loco::DataType::S64>();
65 for (auto i = 0u; i < size; ++i)
66 {
67 int64_t v = const_node->at<loco::DataType::S64>(i);
68 CHECK_OR_FALSE(min_i32 <= v && v <= max_i32);
69 values.push_back(static_cast<int32_t>(v));
70 }
71 }
72 else
73 return false;
74
75 return true;
76};

123void TaggedShapeAnalyzer::init_shape_with_tag(const luci::CircleNode *in_tensor)
124{
125 _shape.clear();
126 uint8_t tag = START_TAG;
127
128 for (uint32_t i = 0; i < in_tensor->rank(); i++)
129 {
130 TagDim dim;
131 {
132 dim.value = in_tensor->dim(i).value();
133 if (dim.value != 1)
134 dim.tags.push_back(tag++);
135 }
136 _shape.push_back(dim);
137 }

155template <loco::DataType PermType>
156void TaggedShapeAnalyzer::analyze_transpose(const luci::CircleTranspose *transpose_node)
157{
158 const luci::CircleConst *perm_node = loco::must_cast<luci::CircleConst *>(transpose_node->perm());
159 assert(perm_node->dtype() == PermType);
160
161 TagShape new_shape;
162 const auto size = perm_node->size<PermType>();
163 for (uint32_t i = 0; i < size; i++)
164 {
165 auto perm_idx = perm_node->at<PermType>(i);
166 new_shape.push_back(_shape.at(perm_idx));
167 }
168 _shape = new_shape;

186bool TaggedShapeAnalyzer::analyze_reshape(const luci::CircleReshape *reshape_node)
187{
188 // At least one element must be in reshape's output-tensor.
189 if (reshape_node->rank() <= 0)
190 return false;
191
192 // Create new_shape based on reshape_node
193 TagShape new_shape;
194 for (uint32_t i = 0; i < reshape_node->rank(); i++)
195 {
196 if (not reshape_node->dim(i).known())
197 return false;
198
199 TagDim dim;
200 dim.value = reshape_node->dim(i).value();
201 new_shape.push_back(dim);
202 }
203
204 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
205 uint32_t old_shape_start_idx = 0;
206 uint32_t old_shape_end_idx = 1;
207 auto old_shape_product = _shape[old_shape_start_idx].value;
208
209 auto expand_range = [&]() -> bool {
210 if (old_shape_end_idx >= _shape.size())
211 return false;
212
213 old_shape_product *= _shape[old_shape_end_idx].value;
214 old_shape_end_idx++;
215 return true;
216 };
217
218 auto move_to_next_range = [&]() -> bool {
219 if (old_shape_end_idx >= _shape.size())
220 return false;
221
222 old_shape_start_idx = old_shape_end_idx;
223 old_shape_end_idx++;
224 old_shape_product = _shape[old_shape_start_idx].value;
225 return true;
226 };
227
228 // Add tags from '_shape' to the 'new_shape'
229 uint32_t new_shape_idx = 0;
230 while (new_shape_idx < new_shape.size())
231 {
232 TagDim &target_dim = new_shape[new_shape_idx];
233
234 // Ignore dim == 1
235 if (target_dim.value == 1)
236 {
237 new_shape_idx++;
238 continue;
239 }
240
241 while (old_shape_product < target_dim.value)
242 {
243 if (expand_range() == false)
244 break;
245 }
246
247 if (old_shape_product != target_dim.value)
248 return false;
249
250 assert(old_shape_product == target_dim.value);
251 for (uint32_t idx = old_shape_start_idx; idx < old_shape_end_idx; idx++)
252 {
253 const auto &old_tags = _shape[idx].tags;
254 target_dim.tags.insert(target_dim.tags.end(), old_tags.begin(), old_tags.end());
255 }
256
257 new_shape_idx++;
258 move_to_next_range();
259 }
260 _shape = new_shape;
261 return true;
262}

264bool TaggedShapeAnalyzer::verify_tag() const
265{
266 // check whether tags in _shape are incremental
267 uint8_t tag = START_TAG;
268 for (const auto &dim : _shape)
269 {
270 for (const auto &t : dim.tags)
271 {
272 if (t == tag)
273 tag++;
274 else
275 return false;
276 }
277 }
278 return true;
279}

292bool TaggedShapeAnalyzer::init(const luci::CircleTranspose *front_transpose,
293 const luci::CircleReshape *mid_reshape,
294 const luci::CircleTranspose *back_transpose)
295{
296 _in = loco::must_cast<luci::CircleNode *>(front_transpose->a());
297 _front_transpose = front_transpose;
298 _mid_reshape = mid_reshape;
299 _back_transpose = back_transpose;
300
301 // check c1
302 CHECK_OR_FALSE(_in->rank() >= _back_transpose->rank());
303
304 const auto front_perm = dynamic_cast<luci::CircleConst *>(_front_transpose->perm());
305 const auto back_perm = dynamic_cast<luci::CircleConst *>(_back_transpose->perm());
306
307 // check c2
308 CHECK_OR_FALSE(front_perm != nullptr);
309 CHECK_OR_FALSE(back_perm != nullptr);
310
311 CHECK_OR_FALSE(extract_shape(_in, _in_shape_v));
312 CHECK_OR_FALSE(extract_const(front_perm, _front_perm_v));
313 CHECK_OR_FALSE(extract_shape(_mid_reshape, _mid_shape_v));
314 CHECK_OR_FALSE(extract_const(back_perm, _back_perm_v));
315
316 auto all_known = [](const std::vector<int32_t> &v) -> bool {
317 for (auto i : v)
318 if (i <= 0)
319 return false;
320 return true;
321 };
322
323 // check c3
324 CHECK_OR_FALSE(all_known(_in_shape_v));
325 CHECK_OR_FALSE(all_known(_mid_shape_v));
326
327 return true;
328}

372template <loco::DataType DType> bool TaggedShapeAnalyzer::can_remove_transposes()
373{
374 assert(_in != nullptr && _front_transpose != nullptr && _mid_reshape != nullptr &&
375 _back_transpose != nullptr);
376
377 // TODO: Update under methods to use std::vector<int32_t&> intead of CircleNode*
378 init_shape_with_tag(_in);
379
380 analyze_transpose<DType>(_front_transpose);
381
382 if (not analyze_reshape(_mid_reshape))
383 return false;
384
385 analyze_transpose<DType>(_back_transpose);
386
387 if (not verify_tag())
388 return false;
389
390 return true;
391}

397template <loco::DataType ShapeType>
398luci::CircleReshape *create_reshape_node(loco::Graph *graph,
399 const luci::CircleTranspose *front_transpose,
400 const luci::CircleReshape *mid_reshape,
401 const luci::CircleTranspose *back_transpose)
402{
403 std::string composed_name =
404 front_transpose->name() + ";" + mid_reshape->name() + ";" + back_transpose->name();
405
406 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> src_origin{luci::get_origin(front_transpose),
407 luci::get_origin(mid_reshape),
408 luci::get_origin(back_transpose)};
409 auto const composed_origin = luci::composite_origin(src_origin);
410
411 auto shape_node = graph->nodes()->create<luci::CircleConst>();
412 {
413 shape_node->dtype(ShapeType);
414 shape_node->rank(1);
415 shape_node->dim(0).set(back_transpose->rank());
416
417 shape_node->size<ShapeType>(back_transpose->rank());
418 for (uint32_t i = 0; i < back_transpose->rank(); i++)
419 {
420 shape_node->at<ShapeType>(i) = back_transpose->dim(i).value();
421 }
422 shape_node->shape_status(luci::ShapeStatus::VALID);
423 shape_node->name(composed_name + "/shape");
424 luci::add_origin(shape_node, composed_origin);
425 }
426
427 auto reshape_node = graph->nodes()->create<luci::CircleReshape>();
428 {
429 reshape_node->name(composed_name);
430 reshape_node->tensor(front_transpose->a());
431 reshape_node->shape(shape_node);
432 luci::add_origin(reshape_node, composed_origin);
433 }
434 return reshape_node;
435}

437bool remove_unnecessary_transpose(luci::CircleTranspose *node)
438{
439 // find 'front_transpose - mid_reshape - back_transpose' pattern
440 const auto back_transpose = node;
441 const auto mid_reshape = dynamic_cast<luci::CircleReshape *>(back_transpose->a());
442 {
443 if (mid_reshape == nullptr)
444 return false;
445 }
446 const auto front_transpose = dynamic_cast<luci::CircleTranspose *>(mid_reshape->tensor());
447 {
448 if (not front_transpose)
449 return false;
450 }
451
452 TaggedShapeAnalyzer analyzer;
453
454 if (not analyzer.init(front_transpose, mid_reshape, back_transpose))
455 return false;
456
457 if (not analyzer.can_remove_transposes<loco::DataType::S32>())
458 return false;
459
460 // repalce with new_node
461 luci::CircleReshape *new_node = create_reshape_node<loco::DataType::S32>(
462 node->graph(), front_transpose, mid_reshape, back_transpose);
463
464 replace(node).with(new_node);
465
466 return true;
467}

510bool RemoveUnnecessaryTransposeNetPass::run(loco::Graph *g)
511{
512 bool changed = false;
513
514 for (auto node : loco::active_nodes(loco::output_nodes(g)))
515 {
516 if (auto transpose_node = dynamic_cast<luci::CircleTranspose *>(node))
517 {
518 if (remove_unnecessary_transpose(transpose_node))
519 changed = true;
520 }
521 }
522
523 return changed;
524}
