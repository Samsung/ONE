32bool extract_shape(const luci::CircleNode *node, std::vector<int32_t> &shape)
33{
34 uint32_t max_i32 = static_cast<uint32_t>(std::numeric_limits<int32_t>::max());
35
36 auto rank = node->rank();
37 for (auto i = 0u; i < rank; ++i)
38 {
39 uint32_t v = node->dim(i).value();
40 CHECK_OR_FALSE(v <= max_i32)
41 shape.push_back(static_cast<int32_t>(v));
42 }
43 return true;
44};

46bool extract_const(const luci::CircleConst *const_node, std::vector<int32_t> &values)
47{
48 auto dtype = const_node->dtype();
49
50 if (dtype == loco::DataType::S32)
51 {
52 auto size = const_node->size<loco::DataType::S32>();
53 for (auto i = 0u; i < size; ++i)
54 {
55 int32_t v = const_node->at<loco::DataType::S32>(i);
56 values.push_back(v);
57 }
58 }
59 else if (dtype == loco::DataType::S64)
60 {
61 int64_t max_i32 = static_cast<int64_t>(std::numeric_limits<int32_t>::max());
62 int64_t min_i32 = static_cast<int64_t>(std::numeric_limits<int32_t>::lowest());
63
64 auto size = const_node->size<loco::DataType::S64>();
65 for (auto i = 0u; i < size; ++i)
66 {
67 int64_t v = const_node->at<loco::DataType::S64>(i);
68 CHECK_OR_FALSE(min_i32 <= v && v <= max_i32);
69 values.push_back(static_cast<int32_t>(v));
70 }
71 }
72 else
73 return false;
74
75 return true;
76};

146void TaggedShapeAnalyzer::init_shape_with_tag(const std::vector<int32_t> &in_shape)
147{
148 _shape.clear();
149 uint8_t tag = START_TAG;
150
151 for (auto i : in_shape)
152 {
153 TagDim dim(i);
154
155 if (dim.value != 1)
156 dim.tags.push_back(tag++);
157
158 _shape.push_back(dim);
159 }

206bool TaggedShapeAnalyzer::analyze_reshape(const std::vector<int32_t> &new_shape)
207{
208 if (new_shape.size() <= 0)
209 return false;
210
211 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
212 uint32_t old_shape_start_idx = 0;
213 uint32_t old_shape_end_idx = 1;
214 auto old_shape_product = _shape[old_shape_start_idx].value;
215
216 auto expand_range = [&]() -> bool {
217 if (old_shape_end_idx >= _shape.size())
218 return false;
219
220 old_shape_product *= _shape[old_shape_end_idx].value;
221 old_shape_end_idx++;
222 return true;
223 };
224
225 auto move_to_next_range = [&]() -> bool {
226 if (old_shape_end_idx >= _shape.size())
227 return false;
228
229 old_shape_start_idx = old_shape_end_idx;
230 old_shape_end_idx++;
231 old_shape_product = _shape[old_shape_start_idx].value;
232 return true;
233 };
234
235 // Create 'new_tagged_shape' based on 'new_shape'
236 TagShape new_tagged_shape;
237
238 uint32_t target_idx = 0;
239 while (target_idx < new_shape.size())
240 {
241 auto target_dim = new_shape[target_idx];
242
243 // Ignore dim == 1
244 if (target_dim == 1)
245 {
246 new_tagged_shape.emplace_back(1);
247 target_idx++;
248 continue;
249 }
250
251 while (old_shape_product < target_dim)
252 {
253 if (expand_range() == false)
254 break;
255 }
256
257 if (old_shape_product != target_dim)
258 return false;
259
260 assert(old_shape_product == target_dim);
261
262 TagDim dim(target_dim);
263 for (uint32_t idx = old_shape_start_idx; idx < old_shape_end_idx; idx++)
264 {
265 const auto &old_tags = _shape[idx].tags;
266 dim.tags.insert(dim.tags.end(), old_tags.begin(), old_tags.end());
267 }
268 new_tagged_shape.push_back(dim);
269
270 target_idx++;
271 move_to_next_range();
272 }
273 _shape = new_tagged_shape;
274 return true;
275}

277bool TaggedShapeAnalyzer::verify_tag() const
278{
279 // check whether tags in _shape are incremental
280 uint8_t tag = START_TAG;
281 for (const auto &dim : _shape)
282 {
283 for (const auto &t : dim.tags)
284 {
285 if (t == tag)
286 tag++;
287 else
288 return false;
289 }
290 }
291 return true;
292}

304bool TaggedShapeAnalyzer::init(const luci::CircleTranspose *front_transpose,
305 const luci::CircleReshape *mid_reshape,
306 const luci::CircleTranspose *back_transpose)
307{
308 _in = loco::must_cast<luci::CircleNode *>(front_transpose->a());
309 _front_transpose = front_transpose;
310 _mid_reshape = mid_reshape;
311 _back_transpose = back_transpose;
312
313 const auto front_perm = dynamic_cast<luci::CircleConst *>(_front_transpose->perm());
314 const auto back_perm = dynamic_cast<luci::CircleConst *>(_back_transpose->perm());
315
316 // check c1
317 CHECK_OR_FALSE(front_perm != nullptr);
318 CHECK_OR_FALSE(back_perm != nullptr);
319
320 CHECK_OR_FALSE(extract_shape(_in, _in_shape_v));
321 CHECK_OR_FALSE(extract_const(front_perm, _front_perm_v));
322 CHECK_OR_FALSE(extract_shape(_front_transpose, _front_shape_v));
323 CHECK_OR_FALSE(extract_shape(_mid_reshape, _mid_shape_v));
324 CHECK_OR_FALSE(extract_const(back_perm, _back_perm_v));
325 CHECK_OR_FALSE(extract_shape(_back_transpose, _out_shape_v));
326
327 auto all_known = [](const std::vector<int32_t> &v) -> bool {
328 for (auto i : v)
329 if (i <= 0)
330 return false;
331 return true;
332 };
333
334 // check c2
335 CHECK_OR_FALSE(all_known(_in_shape_v));
336 CHECK_OR_FALSE(all_known(_front_shape_v));
337 CHECK_OR_FALSE(all_known(_mid_shape_v));
338 CHECK_OR_FALSE(all_known(_out_shape_v));
339
340 return true;
341}

385bool TaggedShapeAnalyzer::can_remove_transposes()
386{
387 assert(_in != nullptr && _front_transpose != nullptr && _mid_reshape != nullptr &&
388 _back_transpose != nullptr);
389
390 auto count_not_1_dim = [](const std::vector<int32_t> &shape) -> int {
391 int count = 0;
392 for (auto i : shape)
393 {
394 if (i != 1)
395 count++;
396 }
397 return count;
398 };
399
400 auto in_rank = count_not_1_dim(_in_shape_v);
401 auto out_rank = count_not_1_dim(_out_shape_v);
402
403 if (in_rank >= out_rank)
404 {
405 init_shape_with_tag(_in_shape_v);
406 analyze_transpose(_front_perm_v);
407 CHECK_OR_FALSE(analyze_reshape(_mid_shape_v));
408 analyze_transpose(_back_perm_v);
409 }
410 else
411 {
412 init_shape_with_tag(_out_shape_v);
413 analyze_transpose(inverse_perm(_back_perm_v));
414 CHECK_OR_FALSE(analyze_reshape(_front_shape_v));
415 analyze_transpose(inverse_perm(_front_perm_v));
416 }
417
418 if (not verify_tag())
419 return false;
420
421 return true;
422}

428template <loco::DataType ShapeType>
429luci::CircleReshape *create_reshape_node(loco::Graph *graph,
430 const luci::CircleTranspose *front_transpose,
431 const luci::CircleReshape *mid_reshape,
432 const luci::CircleTranspose *back_transpose)
433{
434 std::string composed_name =
435 front_transpose->name() + ";" + mid_reshape->name() + ";" + back_transpose->name();
436
437 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> src_origin{luci::get_origin(front_transpose),
438 luci::get_origin(mid_reshape),
439 luci::get_origin(back_transpose)};
440 auto const composed_origin = luci::composite_origin(src_origin);
441
442 auto shape_node = graph->nodes()->create<luci::CircleConst>();
443 {
444 shape_node->dtype(ShapeType);
445 shape_node->rank(1);
446 shape_node->dim(0).set(back_transpose->rank());
447
448 shape_node->size<ShapeType>(back_transpose->rank());
449 for (uint32_t i = 0; i < back_transpose->rank(); i++)
450 {
451 shape_node->at<ShapeType>(i) = back_transpose->dim(i).value();
452 }
453 shape_node->shape_status(luci::ShapeStatus::VALID);
454 shape_node->name(composed_name + "/shape");
455 luci::add_origin(shape_node, composed_origin);
456 }
457
458 auto reshape_node = graph->nodes()->create<luci::CircleReshape>();
459 {
460 reshape_node->name(composed_name);
461 reshape_node->tensor(front_transpose->a());
462 reshape_node->shape(shape_node);
463 luci::add_origin(reshape_node, composed_origin);
464 }
465 return reshape_node;
466}

468bool remove_unnecessary_transpose(luci::CircleTranspose *node)
469{
470 // find 'front_transpose - mid_reshape - back_transpose' pattern
471 const auto back_transpose = node;
472
473 const auto mid_reshape = dynamic_cast<luci::CircleReshape *>(back_transpose->a());
474 CHECK_OR_FALSE(mid_reshape != nullptr);
475
476 const auto front_transpose = dynamic_cast<luci::CircleTranspose *>(mid_reshape->tensor());
477 CHECK_OR_FALSE(front_transpose != nullptr);
478
479 TaggedShapeAnalyzer analyzer;
480 CHECK_OR_FALSE(analyzer.init(front_transpose, mid_reshape, back_transpose));
481 CHECK_OR_FALSE(analyzer.can_remove_transposes());
482
483 // repalce with new_node
484 luci::CircleReshape *new_node = create_reshape_node<loco::DataType::S32>(
485 node->graph(), front_transpose, mid_reshape, back_transpose);
486
487 replace(node).with(new_node);
488
489 return true;
490}

534bool RemoveUnnecessaryTransposeNetPass::run(loco::Graph *g)
535{
536 bool changed = false;
537
538 for (auto node : loco::active_nodes(loco::output_nodes(g)))
539 {
540 if (auto transpose_node = dynamic_cast<luci::CircleTranspose *>(node))
541 {
542 if (remove_unnecessary_transpose(transpose_node))

545 }
546
