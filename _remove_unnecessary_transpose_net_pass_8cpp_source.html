98int32_t TaggedShapeAnalyzer::flatsize(const Shape &shape) const
99{
100 int32_t size = 1;
101 for (const auto &dim : shape)
102 {
103 if (dim.value >= 1)
104 size *= dim.value;
105 }
106 return size;

115bool TaggedShapeAnalyzer::inference_incomplete_shape(const Shape &src, Shape &dst)
116{
117 std::vector<size_t> incomplete_indexes;
118 for (size_t i = 0; i < dst.size(); i++)
119 {
120 if (dst[i].value == -1)
121 incomplete_indexes.push_back(i);
122 }
123
124 if (incomplete_indexes.size() == 0)
125 return true;
126 else if (incomplete_indexes.size() == 1)
127 {
128 if (flatsize(src) % flatsize(dst) == 0)
129 dst[incomplete_indexes[0]].value = flatsize(src) / flatsize(dst);
130 else
131 return false;
132 }
133 else // incomplete_indexes.size() >= 2
134 return false;
135
136 return true;
137}

144void TaggedShapeAnalyzer::init_shape_with_tag(const luci::CircleNode *in_tensor)
145{
146 _shape.clear();
147 uint8_t tag = START_TAG;
148
149 for (uint32_t i = 0; i < in_tensor->rank(); i++)
150 {
151 TaggedShapeAnalyzer::Dim dim;
152 {
153 dim.value = in_tensor->dim(i).value();
154 if (dim.value != 1)
155 dim.tags.push_back(tag++);
156 }
157 _shape.push_back(dim);
158 }

176template <loco::DataType PermType>
177void TaggedShapeAnalyzer::analyze_transpose(const luci::CircleTranspose *transpose_node)
178{
179 const luci::CircleConst *perm_node = loco::must_cast<luci::CircleConst *>(transpose_node->perm());
180 assert(perm_node->dtype() == PermType);
181
182 TaggedShapeAnalyzer::Shape new_shape;
183 const auto size = perm_node->size<PermType>();
184 for (uint32_t i = 0; i < size; i++)
185 {
186 auto perm_idx = perm_node->at<PermType>(i);
187 new_shape.push_back(_shape.at(perm_idx));
188 }
189 _shape = new_shape;
190}

207template <loco::DataType ReshapeType>
208bool TaggedShapeAnalyzer::analyze_reshape(const luci::CircleReshape *reshape_node)
209{
210 const luci::CircleConst *shape_node = loco::must_cast<luci::CircleConst *>(reshape_node->shape());
211 assert(shape_node->dtype() == ReshapeType);
212
213 // At least one element must be in reshape's output-tensor.
214 if (shape_node->size<ReshapeType>() <= 0)
215 return false;
216
217 // Create new_shape based on reshape_node/shape
218 Shape new_shape;
219 for (uint32_t i = 0; i < shape_node->size<ReshapeType>(); i++)
220 {
221 TaggedShapeAnalyzer::Dim dim;
222 dim.value = shape_node->at<ReshapeType>(i);
223
224 new_shape.push_back(dim);
225 }
226
227 // inference new_shape dim with -1 value
228 if (inference_incomplete_shape(_shape, new_shape) == false)
229 return false;
230
231 // indexing for _shape [old_shape_start_idx, old_shape_end_idx)
232 uint32_t old_shape_start_idx = 0;
233 uint32_t old_shape_end_idx = 1;
234 auto old_shape_product = _shape[old_shape_start_idx].value;
235
236 auto expand_range = [&]() -> bool {
237 if (old_shape_end_idx >= _shape.size())
238 return false;
239
240 old_shape_product *= _shape[old_shape_end_idx].value;
241 old_shape_end_idx++;
242 return true;
243 };
244
245 auto move_to_next_range = [&]() -> bool {
246 if (old_shape_end_idx >= _shape.size())
247 return false;
248
249 old_shape_start_idx = old_shape_end_idx;
250 old_shape_end_idx++;
251 old_shape_product = _shape[old_shape_start_idx].value;
252 return true;
253 };
254
255 // Add tags from '_shape' to the 'new_shape'
256 uint32_t new_shape_idx = 0;
257 while (new_shape_idx < new_shape.size())
258 {
259 Dim &target_dim = new_shape[new_shape_idx];
260
261 // Ignore dim == 1
262 if (target_dim.value == 1)
263 {
264 new_shape_idx++;
265 continue;
266 }
267
268 while (old_shape_product < target_dim.value)
269 {
270 if (expand_range() == false)
271 break;
272 }
273
274 if (old_shape_product != target_dim.value)
275 return false;
276
277 assert(old_shape_product == target_dim.value);
278 for (uint32_t idx = old_shape_start_idx; idx < old_shape_end_idx; idx++)
279 {
280 const auto &old_tags = _shape[idx].tags;
281 target_dim.tags.insert(target_dim.tags.end(), old_tags.begin(), old_tags.end());
282 }
283
284 new_shape_idx++;
285 move_to_next_range();
286 }
287 _shape = new_shape;
288 return true;
289}

291bool TaggedShapeAnalyzer::verify_tag() const
292{
293 // check whether tags in _shape are incremental
294 uint8_t tag = START_TAG;
295 for (const auto &dim : _shape)
296 {
297 for (const auto &t : dim.tags)
298 {
299 if (t == tag)
300 tag++;
301 else
302 return false;
303 }
304 }
305 return true;
306}

309template <loco::DataType DType>
310bool TaggedShapeAnalyzer::can_remove_transposes(const luci::CircleTranspose *f_tr,
311 const luci::CircleReshape *m_rs,
312 const luci::CircleTranspose *b_tr)
313{
314 assert(loco::must_cast<luci::CircleConst *>(f_tr->perm())->dtype() == DType);
315 assert(loco::must_cast<luci::CircleConst *>(m_rs->shape())->dtype() == DType);
316 assert(loco::must_cast<luci::CircleConst *>(b_tr->perm())->dtype() == DType);
317
318 const luci::CircleNode *in_tensor = loco::must_cast<luci::CircleNode *>(f_tr->a());
319
320 init_shape_with_tag(in_tensor);
321
322 analyze_transpose<DType>(f_tr);
323
324 if (not analyze_reshape<DType>(m_rs))
325 return false;
326
327 analyze_transpose<DType>(b_tr);
328
329 if (not verify_tag())
330 return false;
331
332 return true;
333}

339template <loco::DataType ShapeType>
340luci::CircleReshape *create_reshape_node(loco::Graph *graph,
341 const luci::CircleTranspose *front_transpose,
342 const luci::CircleReshape *mid_reshape,
343 const luci::CircleTranspose *back_transpose)
344{
345 std::string composed_name =
346 front_transpose->name() + ";" + mid_reshape->name() + ";" + back_transpose->name();
347
348 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> src_origin{luci::get_origin(front_transpose),
349 luci::get_origin(mid_reshape),
350 luci::get_origin(back_transpose)};
351 auto const composed_origin = luci::composite_origin(src_origin);
352
353 auto shape_node = graph->nodes()->create<luci::CircleConst>();
354 {
355 shape_node->dtype(ShapeType);
356 shape_node->rank(1);
357 shape_node->dim(0).set(back_transpose->rank());
358
359 shape_node->size<ShapeType>(back_transpose->rank());
360 for (uint32_t i = 0; i < back_transpose->rank(); i++)
361 {
362 shape_node->at<ShapeType>(i) = back_transpose->dim(i).value();
363 }
364 shape_node->shape_status(luci::ShapeStatus::VALID);
365 shape_node->name(composed_name + "/shape");
366 luci::add_origin(shape_node, composed_origin);
367 }
368
369 auto reshape_node = graph->nodes()->create<luci::CircleReshape>();
370 {
371 reshape_node->name(composed_name);
372 reshape_node->tensor(front_transpose->a());
373 reshape_node->shape(shape_node);
374 luci::add_origin(reshape_node, composed_origin);
375 }
376 return reshape_node;
377}

379bool remove_unnecessary_transpose(luci::CircleTranspose *node)
380{
381 // find 'front_transpose - mid_reshape - back_transpose' pattern
382 const auto back_transpose = node;
383 const auto mid_reshape = dynamic_cast<luci::CircleReshape *>(back_transpose->a());
384 {
385 if (mid_reshape == nullptr)
386 return false;
387 }
388 const auto front_transpose = dynamic_cast<luci::CircleTranspose *>(mid_reshape->tensor());
389 {
390 if (not front_transpose)
391 return false;
392 }
393
394 // check perm and shape are CircleConst node and its' datatype is S32
395 const auto back_perm = dynamic_cast<luci::CircleConst *>(back_transpose->perm());
396 {
397 if (back_perm == nullptr)
398 return false;
399
400 if (back_perm->dtype() != loco::DataType::S32)
401 return false;
402 }
403 const auto shape = dynamic_cast<luci::CircleConst *>(mid_reshape->shape());
404 {
405 if (shape == nullptr)
406 return false;
407
408 if (shape->dtype() != loco::DataType::S32)
409 return false;
410 }
411 const auto front_perm = dynamic_cast<luci::CircleConst *>(front_transpose->perm());
412 {
413 if (front_perm == nullptr)
414 return false;
415
416 if (front_perm->dtype() != loco::DataType::S32)
417 return false;
418 }
419
420 // for now, handle only rank reduction equal (not expansion) cases
421 const auto output_rank = back_transpose->rank();
422 const auto input_rank = front_transpose->rank();
423 if (input_rank < output_rank)
424 return false;
425
426 // analyze pattern to check this pass is applicable
427 TaggedShapeAnalyzer analyzer;
428 if (not analyzer.can_remove_transposes<loco::DataType::S32>(front_transpose, mid_reshape,
429 back_transpose))
430 return false;
431
432 // repalce with new_node
433 luci::CircleReshape *new_node = create_reshape_node<loco::DataType::S32>(
434 node->graph(), front_transpose, mid_reshape, back_transpose);
435
436 replace(node).with(new_node);
437
438 return true;
439}

482bool RemoveUnnecessaryTransposeNetPass::run(loco::Graph *g)
483{
484 bool changed = false;
485
486 for (auto node : loco::active_nodes(loco::output_nodes(g)))
487 {
488 if (auto transpose_node = dynamic_cast<luci::CircleTranspose *>(node))
489 {
490 if (remove_unnecessary_transpose(transpose_node))
491 changed = true;
492 }
493 }
494
495 return changed;
496}
