42public:
43 AclBackendContext(const Backend *backend, ContextData &&data,
44 std::shared_ptr<ITensorRegistry> tensor_registry = nullptr,
45 std::shared_ptr<T_TensorBuilder> tensor_builder = nullptr,
46 std::shared_ptr<T_ConstantInitializer> constant_initializer = nullptr,
47 std::shared_ptr<T_KernelGenerator> kernel_gen = nullptr)
48 : onert::backend::cl_common::BackendContext<T_TensorBuilder, T_ConstantInitializer,
49 T_KernelGenerator>(
50 backend, std::move(data), tensor_registry, tensor_builder, constant_initializer, kernel_gen)
51 {
52 // DO NOTHING
53 }
54
55 ITensorRegistry *genTensors() override
56 {
57 optimizer->optimize();
58
59 this->graph()->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
60 if (this->external_operands().contains(ind))
61 return;
62
63 const auto frontend_layout = this->graph()->layout();
64 const auto backend_layout = this->operand_layouts().at(ind);
65 ir::OperandInfo backend_info{permuteShape(obj.shape(), frontend_layout, backend_layout),
66 obj.typeInfo(), obj.info().memAllocType(), obj.isConstant()};
67 this->tensor_builder->registerTensorInfo(ind, backend_info, backend_layout);
68 });
69
70 // TODO Get compiler options from compiler, and use it rather than getting it from Env
71 if (util::getConfigString(util::config::EXECUTOR) == "Linear")
72 {
73 this->planTensors();
74 }
75 else
76 {
77 // For the executors that does not have fixed linear execution order:
78 // To make tensors never be deallocated, this is a workaround to use static memory planner
79 this->graph()->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &) {
80 if (this->tensor_builder->isRegistered(ind))
81 this->tensor_builder->notifyFirstUse(ind);
82 });
83 }
84
85 this->tensor_builder->prepare();
86
87 return this->tensor_registry.get();
88 }
89
90protected:
91 void registerTensorInfo(const ir::OperandIndex &ind, const ir::OperandInfo &info,
92 ir::Layout backend_layout) override
93 {
94 this->tensor_builder->registerTensorInfo(ind, info, backend_layout);
95 }
96
97public:
98 // TODO Make it private
99 std::shared_ptr<T_Optimizer> optimizer;
100};
