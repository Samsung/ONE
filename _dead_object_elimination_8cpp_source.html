24std::set<coco::Object *> dead_objects(const coco::Module *m)
25{
26 std::set<coco::Object *> res;
27
28 for (uint32_t n = 0; n < m->entity()->object()->size(); ++n)
29 {
30 auto obj = m->entity()->object()->at(n);
31
32 if (auto bag = obj->bag())
33 {
34 if (coco::readers(bag).empty() && !(bag->isOutput()))
35 {
36 res.insert(obj);
37 }
38 }
39 else
40 {
41 // NOTE Just in case if there are Objects not related to Bags
42 if (obj->uses()->size() == 0)
43 {
44 res.insert(obj);
45 }
46 }
47 }
48
49 return res;
50}

57void eliminate_dead_object(enco::Code *code)
58{
59 auto m = code->module();
60
61 // Destroy a dead object and its producer
62 for (auto obj : dead_objects(m))
63 {
64 if (auto producer = coco::producer(obj))

66 auto ins = producer->loc();
67 assert(ins != nullptr);
68
69 ins->detach();

71 }
72
