30template <size_t from, size_t to, typename Enable = void> struct ForEachDimension
31{
32 template <typename L, typename... Args>
33 static void unroll(const onert::ir::Shape &shape, onert::ir::Coordinates &coords,
34 L &&lambda_function, Args &&... args)
35 {
36 static_assert(from < to, "from must not be less than to");
37 assert(static_cast<int>(to) <= shape.rank());
38 const auto &d = shape.dim(from);
39
40 for (auto v = 0; v < d; v++)
41 {
42 coords.set(from, v);
43 ForEachDimension<from + 1, to>::unroll(shape, coords, std::forward<L>(lambda_function),
44 std::forward<Args>(args)...);
45 }
46 }
47};

49template <size_t from, size_t to>

62template <typename L, typename... Args>
63inline void ShapeLoop(const onert::ir::Shape &shape, L &&lambda_function, Args &&... args)
64{
65 assert(shape.rank() > 0);
66 for (auto i = 0; i < shape.rank(); ++i)
67 {
68 assert(shape.dim(i) > 0);
69 }
70
71 onert::ir::Coordinates coords;
72 switch (shape.rank())
73 {
74 case 0:
75 coords.set(0, 0);
76 ForEachDimension<0, 0>::unroll(shape, coords, std::forward<L>(lambda_function),
77 std::forward<Args>(args)...);
78 break;
79 case 1:
80 ForEachDimension<0, 1>::unroll(shape, coords, std::forward<L>(lambda_function),
81 std::forward<Args>(args)...);
82 break;
83 case 2:
84 ForEachDimension<0, 2>::unroll(shape, coords, std::forward<L>(lambda_function),
85 std::forward<Args>(args)...);
86 break;
87 case 3:
88 ForEachDimension<0, 3>::unroll(shape, coords, std::forward<L>(lambda_function),
89 std::forward<Args>(args)...);
90 break;
91 case 4:
92 ForEachDimension<0, 4>::unroll(shape, coords, std::forward<L>(lambda_function),
93 std::forward<Args>(args)...);
94 break;
95 case 5:
96 ForEachDimension<0, 5>::unroll(shape, coords, std::forward<L>(lambda_function),
97 std::forward<Args>(args)...);
98 break;
99 case 6:
100 ForEachDimension<0, 6>::unroll(shape, coords, std::forward<L>(lambda_function),
101 std::forward<Args>(args)...);
102 break;
103 default:
104 assert(false && "ShapeLoop, 1 <= Shape'rank <= 6");
105 break;
106 }
107}
