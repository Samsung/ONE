37void write_file(const std::string &filename, const char *data, size_t data_size)
38{
39 std::ofstream fs(filename, std::ofstream::binary);
40 if (fs.fail())
41 throw std::runtime_error("Cannot open file \"" + filename + "\".\n");
42 if (fs.write(data, data_size).fail())
43 {
44 throw std::runtime_error("Failed to write data to file \"" + filename + "\".\n");
45 }
46}

48std::unique_ptr<luci::Module> import_circle(const std::string &filename)
49{
50 std::ifstream fs(filename, std::ifstream::binary);
51 if (fs.fail())
52 {
53 throw std::runtime_error("Cannot open model file \"" + filename + "\".\n");
54 }
55 std::vector<char> model_data((std::istreambuf_iterator<char>(fs)),
56 std::istreambuf_iterator<char>());
57
58 return luci::Importer().importModule(circle::GetModel(model_data.data()));
59}

61void save_shape(const std::string &shape_filename, const luci::CircleOutput *output_node)
62{
63 if (output_node->rank() == 0)
64 {
65 write_file(shape_filename, "1", 1);
66 }
67 else
68 {
69 auto shape_str = std::to_string(output_node->dim(0).value());
70 for (uint32_t j = 1; j < output_node->rank(); j++)
71 {
72 shape_str += ",";
73 shape_str += std::to_string(output_node->dim(j).value());
74 }
75 write_file(shape_filename, shape_str.c_str(), shape_str.size());
76 }
77}

79template <typename NodeT> size_t tensor_size(const NodeT *node)
80{
81 uint32_t tsize = loco::size(node->dtype());
82 for (uint32_t i = 0; i < node->rank(); ++i)
83 {
84 assert(node->dim(i).known());
85 tsize *= node->dim(i).value();
86 }
87 return tsize;
88}

133bool PModelsRunner::is_input_ready(const RunModel &model)
134{
135 for (auto &part : _pconfig.parts)
136 {
137 if (part.model_file != model)
138 continue;
139
140 for (auto &input : part.inputs)
141 {
142 auto it = _data_stage.find(input);
143 if (it == _data_stage.end())
144 return false;
145 }
146 }

150bool PModelsRunner::run(void)
151{
152 LOGGER(l);
153
154 // for each partitioned model, if the inputs of the model are ready, run the model
155 do
156 {
157 bool found_model = false;
158
159 for (auto it = _models_to_run.begin(); it != _models_to_run.end(); ++it)
160 {
161 auto model_fname = *it;
162
163 INFO(l) << "Check model input ready: " << model_fname << std::endl;
164 if (is_input_ready(model_fname))
165 {
166 found_model = true;
167
168 INFO(l) << "Run model: " << model_fname << std::endl;
169 auto module = import_circle(model_fname);
170
171 luci_interpreter::Interpreter interpreter(module.get());
172
173 // Set input
174 const auto input_nodes = loco::input_nodes(module->graph());
175 int32_t num_inputs = static_cast<int32_t>(input_nodes.size());
176 for (int32_t i = 0; i < num_inputs; i++)
177 {
178 const auto *input_node = loco::must_cast<const luci::CircleInput *>(input_nodes[i]);
179
180 auto input_name = input_node->name();
181 assert(_data_stage.find(input_name) != _data_stage.end());
182
183 auto input_data = _data_stage[input_name];
184
185 interpreter.writeInputTensor(input_node, input_data.data(), input_data.size());
186 }
187
188 // Run interpreter
189 interpreter.interpret();
190 INFO(l) << "Run model: " << model_fname << " done" << std::endl;
191
192 // Get output.
193 const auto output_nodes = loco::output_nodes(module->graph());
194 for (uint32_t i = 0; i < module->graph()->outputs()->size(); i++)
195 {
196 const auto *output_node = loco::must_cast<const luci::CircleOutput *>(output_nodes[i]);
197 auto output_name = output_node->name();
198
199 Buffer output_data(tensor_size(output_node));
200
201 interpreter.readOutputTensor(output_node, output_data.data(), output_data.size());
202
203 // There should not exist same output names
204 // TODO check with multiple virtual outputs
205 assert(_data_stage.find(output_name) == _data_stage.end());
206 _data_stage[output_name] = output_data;
207 }
208
209 // We've ran this model, remove from the model list
210 _models_to_run.erase(it);
211 break;
212 }
213 }
214
215 if (not found_model)
216 {
217 std::cerr << "ERROR: model partition or configuration has problems" << std::endl;
218 return false;
219 }
220 } while (not _models_to_run.empty());
221
222 return true;
223}

225void PModelsRunner::save_outputs(const std::string &output_file)
226{
227 LOGGER(l);
228
229 // load source model as we need to get both shape and node name
230 // TODO check for unknown shape
231 auto source_fname = _pconfig.source.model_file;
232
233 INFO(l) << "save_outputs() loading file: " << source_fname << std::endl;
234 auto module = import_circle(source_fname);
235
236 const auto output_nodes = loco::output_nodes(module->graph());
237 for (uint32_t i = 0; i < module->graph()->outputs()->size(); i++)
238 {
239 const auto *output_node = loco::must_cast<const luci::CircleOutput *>(output_nodes[i]);
240
241 auto output_name = output_node->name();
242 INFO(l) << "save_outputs() save output node: " << output_name << std::endl;
243 assert(_data_stage.find(output_name) != _data_stage.end());
244
245 auto tensor_data = _data_stage[output_name];
246 auto output_filename = output_file + std::to_string(i);
247
248 write_file(output_filename, tensor_data.data(), tensor_data.size());
249 save_shape(output_filename + ".shape", output_node);
250 }
251}
