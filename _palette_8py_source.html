16# Class to save colorscheme
17class Palette:
18 # Child class must implement __init__ to fill the below members
19 def __init__(self):
20 # Element of self._slots has [lower bound, upper bound] of qerrors to decide a color
21 self._slots = []
22 # Element of self._colors has rgb values in string format
23 self._colors = []
24 raise NotImplementedError('Child class must implement __init__')
25
26 # Return color scheme as a list of objects
27 # Each object has the following attributes
28 # b: begin qerror
29 # e: end qerror
30 # c: color (in RGB string)
31 def colorscheme(self):
32 cs = []
33 for slot, color in zip(self._slots, self._colors):
34 cs.append({"b": slot[0], "e": slot[1], "c": color})
35 return cs

38# Ranges of slots are defined by qerror_min/qerror_max
39# Each slot has a uniform range
40# For example, if qerror_min = 0.0, qerror_max = 1.0, number of colors = 10
41# Ranges of slots will be as follows.
42# [0.0, 0.1], [0.1, 0.2], [0.2, 0.3] ... [0.8, 0.9], [0.9, 1.0]
43class UniformPalette(Palette):
44 def __init__(self, qerror_min, qerror_max, colors):
45 self._colors_colors = colors
46 self._slots_slots = []
47 qerror_range = qerror_max - qerror_min
48 num_colors = len(self._colors_colors)
49 for i in range(num_colors):
50 lower_bound = qerror_min + i * (qerror_range / num_colors)
51 upper_bound = qerror_min + (i + 1) * (qerror_range / num_colors)
52
53 self._slots_slots.append([lower_bound, upper_bound])
54
55 # Invariant
56 assert len(self._slots_slots) == num_colors
