40template <typename T>
41static void Init(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj, const bool copy,
42 const onert::ir::Layout frontend_layout = onert::ir::Layout::UNKNOWN)
43{
44 const auto shape = model_obj.shape();
45 assert(model_obj.data());
46 auto base = reinterpret_cast<const T *>(model_obj.data()->base());
47
48 obj.access([&](::onert::backend::ITensor &tensor) {
49 switch (shape.rank())
50 {
51 case 0:
52 {
53 assert(model_obj.data()->size() == sizeof(T));
54 const auto value = *reinterpret_cast<const T *>(base);
55 T *into = reinterpret_cast<T *>(tensor.buffer());
56 *into = value;
57 break;
58 }
59 case 1:
60 {
61 auto vec_size = shape.dim(0);
62 for (int32_t n = 0; n < vec_size; ++n)
63 {
64 const T *from = reinterpret_cast<const T *>(base) + n;
65 const auto value = *from;
66
67 T *into = reinterpret_cast<T *>(tensor.buffer()) + n;
68
69 *into = value;
70 }
71 break;
72 }
73 case 2:
74 {
75 const int32_t copy_len = shape.dim(1);
76
77 for (auto i = 0; i < shape.dim(0); ++i)
78 {
79 ::onert::ir::Coordinates coords{i, 0};
80 memcpy(tensor.buffer() + tensor.calcOffset(coords), base + i * copy_len,
81 copy_len * sizeof(T));
82 }
83 break;
84 }
85 case 3:
86 {
87 const int32_t width = shape.dim(1);
88 const int32_t copy_len = shape.dim(2);
89
90 for (auto i = 0; i < shape.dim(0); ++i)
91 {
92 for (auto j = 0; j < shape.dim(1); ++j)
93 {
94 ::onert::ir::Coordinates coords{i, j, 0};
95 memcpy(tensor.buffer() + tensor.calcOffset(coords),
96 base + i * width * copy_len + j * copy_len, copy_len * sizeof(T));
97 }
98 }
99 break;
100 }
101 case 4:
102 {
103 const int32_t height = shape.dim(1);
104 const int32_t width = shape.dim(2);
105 const int32_t copy_len = shape.dim(3);
106 for (auto i = 0; i < shape.dim(0); ++i)
107 {
108 for (auto j = 0; j < shape.dim(1); ++j)
109 {
110 for (auto k = 0; k < shape.dim(2); ++k)
111 {
112 if (copy)
113 {
114 ::onert::ir::Coordinates coords{i, j, k, 0};
115 memcpy(tensor.buffer() + tensor.calcOffset(coords),
116 base + i * height * width * copy_len + j * width * copy_len + k * copy_len,
117 copy_len * sizeof(T));
118 }
119 else
120 {
121 for (auto l = 0; l < shape.dim(3); ++l)
122 {
123 const auto coords =
124 ::onert::ir::convertCoordinates({i, j, k, l}, frontend_layout, tensor.layout());
125 T *into = reinterpret_cast<T *>(tensor.buffer() + tensor.calcOffset(coords));
126 T value = *(base + i * height * width * copy_len + j * width * copy_len +
127 k * copy_len + l);
128 *into = value;
129 }
130 }
131 }
132 }
133 }
134 break;
135 }
136 default:
137 throw std::runtime_error{"Not yet supported"};
138 }
139 });
140}

142template <typename T>
143void copyInit(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj)
144{
145 Init<T>(model_obj, obj, true);
146}

148template <typename T>
149void permuteInit(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj,
150 const onert::ir::Layout frontend_layout)
151{
152 const bool copy = frontend_layout == obj.layout();
153 Init<T>(model_obj, obj, copy, frontend_layout);

156// Pre-defined initializer - fill reverse order
157template <typename T> void initReverseOrder(const ir::Operand &model_obj, backend::ITensor &obj)
158{
159 assert(model_obj.data());
160 const auto &shape = model_obj.shape();
161 const auto base = reinterpret_cast<const T *>(model_obj.data()->base());
162 assert(model_obj.shape().rank() == 1);
163 obj.access([&](ITensor &tensor) {
164 for (size_t i = 0; i < shape.num_elements(); ++i)
165 {
166 const T value = base[shape.num_elements() - i - 1];
167 T *into = reinterpret_cast<T *>(tensor.buffer() + tensor.calcOffset({static_cast<T>(i)}));
168 *into = value;
169 }
170 });
171}

173class AclConstantInitializer : public ir::OperationVisitor
174{
175public:
176 void run()
177 {
178 assert(_tensor_reg);
179 for (const auto &it : _init_map)
180 {
181 const auto &ind = it.first;
182 const auto &fn = it.second;
183
184 const auto &model_obj = _operands.at(ind);
185 auto tensor_obj = _tensor_reg->getNativeITensor(ind);
186 assert(tensor_obj != nullptr);
187 fn(model_obj, *tensor_obj);
188 VERBOSE(FillOperandData) << "Fill data for operand " << ind << std::endl;
189 }
190 _init_map.clear();
191 }
192
193public:
194 AclConstantInitializer(const ir::Operands &operands,
195 const std::shared_ptr<ITensorRegistry> &tensor_reg);
196
197public:
198 using Initializer = std::function<void(const ir::Operand &, backend::ITensor &)>;
199
200public:
201 void registerDefaultInitializer(const ir::OperandIndex &index, const ir::Operand &obj)
202 {
203 registerPermuteInitializer(index, obj);
204 }
205 void registerCopyInitializer(const ir::OperandIndex &index, const ir::Operand &obj);
206 void registerPermuteInitializer(const ir::OperandIndex &index, const ir::Operand &obj);
207
208public:
209 void setLayout(ir::Layout layout) { _current_layout = layout; }
210 bool exist(const ir::OperandIndex &ind) { return _init_map.find(ind) != _init_map.end(); }
211
212public:
213 void visit(const ir::operation::BatchToSpaceND &) override;
214 void visit(const ir::operation::Conv2D &) override;
215 void visit(const ir::operation::DepthwiseConv2D &) override;
216 void visit(const ir::operation::FullyConnected &) override;
217 void visit(const ir::operation::LSTM &) override;
218 void visit(const ir::operation::RNN &) override;
219 void visit(const ir::operation::TransposeConv &) override;
220
221protected:
222 void copyInputInitialize(const ir::Operation &node, uint32_t index);
223 void permuteInputInitialize(const ir::Operation &node, uint32_t index);
224
225protected:
226 const ir::Operands &_operands;
227 std::shared_ptr<ITensorRegistry> _tensor_reg;
228 std::unordered_map<ir::OperandIndex, Initializer> _init_map;
229 ir::Layout _current_layout;
230};
