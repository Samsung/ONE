40template <typename T>
41static void Init(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj)
42{
43 const auto shape = model_obj.shape();
44 assert(model_obj.data());
45 auto base = reinterpret_cast<const T *>(model_obj.data()->base());
46
47 obj.access([&](::onert::backend::ITensor &tensor) {
48 switch (shape.rank())
49 {
50 case 0:
51 {
52 assert(model_obj.data()->size() == sizeof(T));
53 const auto value = *reinterpret_cast<const T *>(base);
54 T *into = reinterpret_cast<T *>(tensor.buffer());
55 *into = value;
56 break;
57 }
58 case 1:
59 {
60 auto vec_size = shape.dim(0);
61 for (int32_t n = 0; n < vec_size; ++n)
62 {
63 const T *from = reinterpret_cast<const T *>(base) + n;
64 const auto value = *from;
65
66 T *into = reinterpret_cast<T *>(tensor.buffer()) + n;
67
68 *into = value;
69 }
70 break;
71 }
72 case 2:
73 {
74 const int32_t copy_len = shape.dim(1);
75
76 for (auto i = 0; i < shape.dim(0); ++i)
77 {
78 ::onert::ir::Coordinates coords{i, 0};
79 memcpy(tensor.buffer() + tensor.calcOffset(coords), base + i * copy_len,
80 copy_len * sizeof(T));
81 }
82 break;
83 }
84 case 3:
85 {
86 const int32_t width = shape.dim(1);
87 const int32_t copy_len = shape.dim(2);
88
89 for (auto i = 0; i < shape.dim(0); ++i)
90 {
91 for (auto j = 0; j < shape.dim(1); ++j)
92 {
93 ::onert::ir::Coordinates coords{i, j, 0};
94 memcpy(tensor.buffer() + tensor.calcOffset(coords),
95 base + i * width * copy_len + j * copy_len, copy_len * sizeof(T));
96 }
97 }
98 break;
99 }
100 case 4:
101 {
102 const int32_t height = shape.dim(1);
103 const int32_t width = shape.dim(2);
104 const int32_t copy_len = shape.dim(3);
105 for (auto i = 0; i < shape.dim(0); ++i)
106 {
107 for (auto j = 0; j < shape.dim(1); ++j)
108 {
109 for (auto k = 0; k < shape.dim(2); ++k)
110 {
111 ::onert::ir::Coordinates coords{i, j, k, 0};
112 memcpy(tensor.buffer() + tensor.calcOffset(coords),
113 base + i * height * width * copy_len + j * width * copy_len + k * copy_len,
114 copy_len * sizeof(T));
115 }
116 }
117 }
118 break;
119 }
120 default:
121 throw std::runtime_error{"Not yet supported"};
122 }
123 });
124}

132// Pre-defined initializer - fill reverse order
133template <typename T> void initReverseOrder(const ir::Operand &model_obj, backend::ITensor &obj)
134{
135 assert(model_obj.data());
136 const auto &shape = model_obj.shape();
137 const auto base = reinterpret_cast<const T *>(model_obj.data()->base());
138 assert(model_obj.shape().rank() == 1);
139 obj.access([&](ITensor &tensor) {
140 for (size_t i = 0; i < shape.num_elements(); ++i)
141 {
142 const T value = base[shape.num_elements() - i - 1];
143 T *into = reinterpret_cast<T *>(tensor.buffer() + tensor.calcOffset({static_cast<T>(i)}));
144 *into = value;
145 }
146 });
147}

149class AclConstantInitializer : public ir::OperationVisitor
150{
151public:
152 void run()
153 {
154 assert(_tensor_reg);
155 for (const auto &[ind, fn] : _init_map)
156 {
157 const auto &model_obj = _operands.at(ind);
158 auto tensor_obj = _tensor_reg->getNativeITensor(ind);
159 assert(tensor_obj != nullptr);
160 fn(model_obj, *tensor_obj);
161 VERBOSE(FillOperandData) << "Fill data for operand " << ind << std::endl;
162 }
163 _init_map.clear();
164 }
165
166public:
167 AclConstantInitializer(const ir::Operands &operands,
168 const std::shared_ptr<ITensorRegistry> &tensor_reg);
169
170public:
171 using Initializer = std::function<void(const ir::Operand &, backend::ITensor &)>;
172
173public:
174 void registerDefaultInitializer(const ir::OperandIndex &index, const ir::Operand &obj)
175 {
176 registerCopyInitializer(index, obj);
177 }
178 void registerCopyInitializer(const ir::OperandIndex &index, const ir::Operand &obj);
179
180public:
181 bool exist(const ir::OperandIndex &ind) { return _init_map.find(ind) != _init_map.end(); }
182
183public:
184 void visit(const ir::operation::BatchToSpaceND &) override;
185 void visit(const ir::operation::Conv2D &) override;
186 void visit(const ir::operation::DepthwiseConv2D &) override;
187 void visit(const ir::operation::FullyConnected &) override;
188 void visit(const ir::operation::LSTM &) override;
189 void visit(const ir::operation::RNN &) override;
190 void visit(const ir::operation::TransposeConv &) override;
191
192protected:
193 void copyInputInitialize(const ir::Operation &node, uint32_t index);
194
195protected:
196 const ir::Operands &_operands;
197 std::shared_ptr<ITensorRegistry> _tensor_reg;
198 std::unordered_map<ir::OperandIndex, Initializer> _init_map;
199};
