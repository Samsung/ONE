36template <typename T>
37static void Init(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj)
38{
39 const auto shape = model_obj.shape();
40 assert(model_obj.data());
41 auto base = reinterpret_cast<const T *>(model_obj.data()->base());
42
43 obj.access([&](::onert::backend::ITensor &tensor) {
44 switch (shape.rank())
45 {
46 case 0:
47 {
48 assert(model_obj.data()->size() == sizeof(T));
49 const auto value = *reinterpret_cast<const T *>(base);
50 T *into = reinterpret_cast<T *>(tensor.buffer());
51 *into = value;
52 break;
53 }
54 case 1:
55 {
56 auto vec_size = shape.dim(0);
57 for (int32_t n = 0; n < vec_size; ++n)
58 {
59 const T *from = reinterpret_cast<const T *>(base) + n;
60 const auto value = *from;
61
62 T *into = reinterpret_cast<T *>(tensor.buffer()) + n;
63
64 *into = value;
65 }
66 break;
67 }
68 case 2:
69 {
70 const int32_t copy_len = shape.dim(1);
71
72 for (auto i = 0; i < shape.dim(0); ++i)
73 {
74 ::onert::ir::Coordinates coords{i, 0};
75 memcpy(tensor.buffer() + tensor.calcOffset(coords), base + i * copy_len,
76 copy_len * sizeof(T));
77 }
78 break;
79 }
80 case 3:
81 {
82 const int32_t width = shape.dim(1);
83 const int32_t copy_len = shape.dim(2);
84
85 for (auto i = 0; i < shape.dim(0); ++i)
86 {
87 for (auto j = 0; j < shape.dim(1); ++j)
88 {
89 ::onert::ir::Coordinates coords{i, j, 0};
90 memcpy(tensor.buffer() + tensor.calcOffset(coords),
91 base + i * width * copy_len + j * copy_len, copy_len * sizeof(T));
92 }
93 }
94 break;
95 }
96 case 4:
97 {
98 const int32_t height = shape.dim(1);
99 const int32_t width = shape.dim(2);
100 const int32_t copy_len = shape.dim(3);
101 for (auto i = 0; i < shape.dim(0); ++i)
102 {
103 for (auto j = 0; j < shape.dim(1); ++j)
104 {
105 for (auto k = 0; k < shape.dim(2); ++k)
106 {
107 ::onert::ir::Coordinates coords{i, j, k, 0};
108 memcpy(tensor.buffer() + tensor.calcOffset(coords),
109 base + i * height * width * copy_len + j * width * copy_len + k * copy_len,
110 copy_len * sizeof(T));
111 }
112 }
113 }
114 break;
115 }
116 default:
117 throw std::runtime_error{"Not yet supported"};
118 }
119 });
120}

128// Pre-defined initializer - fill reverse order
129template <typename T> void initReverseOrder(const ir::Operand &model_obj, backend::ITensor &obj)
130{
131 assert(model_obj.data());
132 const auto &shape = model_obj.shape();
133 const auto base = reinterpret_cast<const T *>(model_obj.data()->base());
134 assert(model_obj.shape().rank() == 1);
135 obj.access([&](ITensor &tensor) {
136 for (size_t i = 0; i < shape.num_elements(); ++i)
137 {
138 const T value = base[shape.num_elements() - i - 1];
139 T *into = reinterpret_cast<T *>(tensor.buffer() + tensor.calcOffset({static_cast<T>(i)}));
140 *into = value;
141 }
142 });
143}

145class AclConstantInitializer : public ir::OperationVisitor
146{
147public:
148 void run()
149 {
150 assert(_tensor_reg);
151 for (const auto &[ind, fn] : _init_map)
152 {
153 const auto &model_obj = _operands.at(ind);
154 auto tensor_obj = _tensor_reg->getNativeITensor(ind);
155 assert(tensor_obj != nullptr);
156 fn(model_obj, *tensor_obj);
157 VERBOSE(FillOperandData) << "Fill data for operand " << ind << std::endl;
158 }
159 _init_map.clear();
160 }
161
162public:
163 AclConstantInitializer(const ir::Operands &operands,
164 const std::shared_ptr<ITensorRegistry> &tensor_reg);
165
166public:
167 using Initializer = std::function<void(const ir::Operand &, backend::ITensor &)>;
168
169public:
170 void registerDefaultInitializer(const ir::OperandIndex &index, const ir::Operand &obj)
171 {
172 registerCopyInitializer(index, obj);
173 }
174 void registerCopyInitializer(const ir::OperandIndex &index, const ir::Operand &obj);
175
176public:
177 bool exist(const ir::OperandIndex &ind) { return _init_map.find(ind) != _init_map.end(); }
178
179public:
180 void visit(const ir::operation::BatchToSpaceND &) override;
181 void visit(const ir::operation::Conv2D &) override;
182 void visit(const ir::operation::DepthwiseConv2D &) override;
183 void visit(const ir::operation::FullyConnected &) override;
184 void visit(const ir::operation::LSTM &) override;
185 void visit(const ir::operation::RNN &) override;
186 void visit(const ir::operation::TransposeConv &) override;
187
188protected:
189 void copyInputInitialize(const ir::Operation &node, uint32_t index);
190
191protected:
192 const ir::Operands &_operands;
193 std::shared_ptr<ITensorRegistry> _tensor_reg;
194 std::unordered_map<ir::OperandIndex, Initializer> _init_map;
195};
