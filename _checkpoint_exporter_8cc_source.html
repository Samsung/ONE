36class CheckpointExporter
37{
38public:
39 CheckpointExporter(const ir::train::TrainingInfo *const train_info,
40 const exec::Execution *const exec)
41 {
42 std::memset(&_header, 0, sizeof(_header));
43 _header.magic = checkpoint::MAGIC_NUMBER;
44 _header.schema = checkpoint::SCHEMA_VERSION;
45
46 uint32_t offset = sizeof(_header);
47 // TODO Store tensor and optimizer data
48 UNUSED_RELEASE(exec);
49 _header.other_offset = offset;
50
51 std::memset(&_footer, 0, sizeof(_footer));
52 _footer.cur_step = train_info->trainingStep();
53 }
54
55 void save(const std::string &path)
56 {
57 std::ofstream dst(path.c_str(), std::ios::binary | std::ios::trunc);
58 if (!dst.is_open())
59 throw std::runtime_error{"Failed to save checkpoint: " + path};
60
61 dst.write(reinterpret_cast<const char *>(&_header), sizeof(_header));
62 // TODO Write tensor and optimizer data
63 dst.write(reinterpret_cast<const char *>(&_footer), sizeof(_footer));
64 dst.close();
65 }
66
67private:
68 checkpoint::Header _header;
69 checkpoint::Footer _footer;
70};
