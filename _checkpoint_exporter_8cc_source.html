36struct DataBuffer
37{
38 void setSize(const std::vector<uint32_t> &sizes)
39 {
40 _offset.resize(sizes.size());
41 uint32_t total = std::accumulate(sizes.begin(), sizes.end(), 0);
42 _data.resize(total);
43
44 _offset_it = _offset.begin();
45 _data_ptr = _data.data();
46 }
47
48 void setOffset(uint32_t offset) { _start_offset = offset + _offset.size() * sizeof(uint32_t); }
49
50 // This function should be called after executing the setSize() and setOffset() functions.
51 void setData(const char *data, uint32_t size)
52 {
53 assert(_offset_it != _offset.end());
54 assert(_data_ptr - _data.data() + size <= _data.size());
55
56 *_offset_it++ = _start_offset;
57 if (data && size > 0)
58 std::memcpy(_data_ptr, data, size);
59 _data_ptr += size;
60 _start_offset += size;
61 }
62
63 uint32_t size() const { return sizeof(uint32_t) * _offset.size() + _data.size(); }
64
65 void write(std::ofstream &ostream)
66 {
67 ostream.write(reinterpret_cast<const char *>(&_offset[0]),
68 static_cast<std::streamsize>(sizeof(_offset[0]) * _offset.size()));
69 ostream.write(reinterpret_cast<const char *>(&_data[0]),
70 static_cast<std::streamsize>(_data.size()));
71 }
72
73private:
74 std::vector<uint32_t> _offset;
75 std::vector<char> _data;
76 uint32_t _start_offset = 0;
77 std::vector<uint32_t>::iterator _offset_it;
78 char *_data_ptr = nullptr;
79};

81class CheckpointExporter
82{
83public:
84 CheckpointExporter(const ir::train::TrainingInfo *const train_info,
85 const exec::Execution *const exec)
86 {
87 std::memset(&_header, 0, sizeof(_header));
88 _header.magic = checkpoint::MAGIC_NUMBER;
89 _header.schema = checkpoint::SCHEMA_VERSION;
90
91 uint32_t offset = sizeof(_header);
92
93 auto length = 0;
94 exec->iterateTrainableTensors(
95 [&](const ir::OperandIndex &, const backend::train::ITrainableTensor *) { length++; });
96 _header.length = length;
97
98 setTensorData(offset, exec);
99 offset += _tensor_data.size();
100
101 _header.opt1_offset = offset;
102 setOptimizerData(offset, train_info, exec);
103 if (_optimizer_data.size() > 2)
104 throw std::runtime_error{"Do not support optimizer data more than 2."};
105 if (_optimizer_data.size() > 0)
106 {
107 offset += _optimizer_data[0].size();
108 if (_optimizer_data.size() > 1)
109 {
110 _header.opt2_offset = offset;
111 offset += _optimizer_data[1].size();
112 }
113 }
114
115 _header.other_offset = offset;
116
117 std::memset(&_footer, 0, sizeof(_footer));
118 _footer.cur_step = train_info->trainingStep();
119 }
120
121 void save(const std::string &path)
122 {
123 std::ofstream dst(path.c_str(), std::ios::binary | std::ios::trunc);
124 if (!dst.is_open())
125 throw std::runtime_error{"Failed to save checkpoint: " + path};
126
127 dst.write(reinterpret_cast<const char *>(&_header), sizeof(_header));
128 _tensor_data.write(dst);
129 for (auto &opt : _optimizer_data)
130 opt.write(dst);
131 dst.write(reinterpret_cast<const char *>(&_footer), sizeof(_footer));
132 dst.close();
133 }
134
135private:
136 void setTensorData(uint32_t start_offset, const exec::Execution *const exec)
137 {
138 std::vector<uint32_t> sizes;
139 exec->iterateTrainableTensors([&](const ir::OperandIndex &,
140 const backend::train::ITrainableTensor *tensor) {
141 assert(tensor);
142 if (tensor->total_size() >= std::numeric_limits<uint32_t>::max())
143 {
144 throw std::runtime_error{"Tensor size exceeds the uint32_t max value. This model does not "
145 "support saving as a checkpoint file."};
146 }
147 sizes.emplace_back(tensor->total_size());
148 });
149
150 assert(_header.length == sizes.size());
151
152 _tensor_data.setSize(sizes);
153 _tensor_data.setOffset(start_offset);
154 [[maybe_unused]] auto vindex = 0;
155 exec->iterateTrainableTensors([&](const ir::OperandIndex &,
156 const backend::train::ITrainableTensor *tensor) {
157 assert(tensor);
158 assert(sizes[vindex++] == tensor->total_size());
159 _tensor_data.setData(reinterpret_cast<const char *>(tensor->buffer()), tensor->total_size());
160 });
161 }
162
163 void setOptimizerData(uint32_t start_offset, const ir::train::TrainingInfo *const train_info,
164 const exec::Execution *const exec)
165 {
166 // TODO Support multiple optimizer
167 switch (train_info->optimizerInfo().optim_code)
168 {
169 case onert::ir::train::OptimizerCode::Adam:
170 setAdamOptimizerData(start_offset, exec);
171 break;
172 default:
173 break;
174 }
175 }
176
177 void setAdamOptimizerData(uint32_t start_offset, const exec::Execution *const exec)
178 {
179 // Adam optimizer has two optimizer variables. (mean, variance)
180 constexpr auto ADAM_VARIABLE_COUNT = 2;
181
182 std::vector<uint32_t> sizes;
183 exec->iterateTrainableTensors(
184 [&](const ir::OperandIndex &, const backend::train::ITrainableTensor *tensor) {
185 assert(tensor);
186 auto trainable_tensor = const_cast<backend::train::ITrainableTensor *>(tensor);
187 const auto &opt_vars = trainable_tensor->optVars();
188
189 // Untrainable tensor should not have any optimizer variables.
190 assert(opt_vars.size() == ADAM_VARIABLE_COUNT || opt_vars.size() == 0);
191
192 uint32_t size = 0;
193 if (opt_vars.size() == ADAM_VARIABLE_COUNT)
194 {
195 assert(opt_vars[0]->total_size() == opt_vars[1]->total_size());
196 size = opt_vars[0]->total_size();
197 }
198
199 sizes.emplace_back(size);
200 });
201
202 assert(_header.length == sizes.size());
203
204 _optimizer_data.resize(ADAM_VARIABLE_COUNT);
205 for (auto &opt : _optimizer_data)
206 {
207 opt.setSize(sizes);
208 opt.setOffset(start_offset);
209 start_offset += opt.size();
210 }
211
212 [[maybe_unused]] auto vindex = 0;
213 exec->iterateTrainableTensors(
214 [&](const ir::OperandIndex &, const backend::train::ITrainableTensor *tensor) {
215 assert(tensor);
216 auto trainable_tensor = const_cast<backend::train::ITrainableTensor *>(tensor);
217 const auto &opt_vars = trainable_tensor->optVars();
218
219 for (auto i = 0; i < ADAM_VARIABLE_COUNT; ++i)
220 {
221 if (opt_vars.size() == ADAM_VARIABLE_COUNT)
222 {
223 assert(opt_vars[i]->total_size() == sizes[vindex]);
224 _optimizer_data[i].setData(reinterpret_cast<const char *>(opt_vars[i]->buffer()),
225 opt_vars[i]->total_size());
226 }
227 else
228 _optimizer_data[i].setData(nullptr, 0);
229 }
230 vindex++;
231 });
232 }
233
234private:
235 checkpoint::Header _header;
236 checkpoint::Footer _footer;
237 DataBuffer _tensor_data;
238 std::vector<DataBuffer> _optimizer_data;
239};
