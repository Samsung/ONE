30void KernelGenerator::visit(const ir::operation::TopKV2 &node)
31{
32 const auto outputValues_index{node.getOutputs().at(ir::operation::TopKV2::Output::OUTPUT_VALUES)};
33 const auto outputIndices_index{
34 node.getOutputs().at(ir::operation::TopKV2::Output::OUTPUT_INDICES)};
35
36 const auto inputData_index{node.getInputs().at(ir::operation::TopKV2::Input::INPUT)};
37
38 const auto k = node.param().k;
39
40 auto values_tensor = _tensor_reg->getPortableTensor(outputValues_index);
41 auto indices_tensor = _tensor_reg->getPortableTensor(outputIndices_index);
42 auto input_tensor = _tensor_reg->getPortableTensor(inputData_index);
43
44 auto fn = std::make_unique<ops::TopKV2Layer>();
45
46 fn->configure(input_tensor, values_tensor, indices_tensor, k);
47
48 _return_fn = std::move(fn);
49}

71void TopKV2Layer::run()
72{
73 if (_output_indices->data_type() != OperandType::INT32)
74 throw std::runtime_error{"TopKV2: unsupported output indices type"};
75
76 if (_input->data_type() == OperandType::FLOAT32)
77 {
78 nnfw::cker::TopKV2<float, int32_t>(getShape(_input), getBuffer<float>(_input), _k,
79 getBuffer<float>(_output_value),
80 getBuffer<int32_t>(_output_indices));
81 }
82 else if (_input->data_type() == OperandType::INT32)
83 {
84 nnfw::cker::TopKV2<int32_t, int32_t>(getShape(_input), getBuffer<int32_t>(_input), _k,
85 getBuffer<int32_t>(_output_value),
86 getBuffer<int32_t>(_output_indices));
87 }
88 else if (_input->data_type() == OperandType::QUANT_UINT8_ASYMM)
89 {
90 nnfw::cker::TopKV2<uint8_t, int32_t>(getShape(_input), getBuffer<uint8_t>(_input), _k,
91 getBuffer<uint8_t>(_output_value),
92 getBuffer<int32_t>(_output_indices));
93 }
94 else
95 {
96 throw std::runtime_error{"TopKV2: unsupported data type"};
97 }
98}
