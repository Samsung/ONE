25static inline void Conv(const tflite::ConvParams &params, const tflite::RuntimeShape &input_shape,
26 const float *input_data, const tflite::RuntimeShape &filter_shape,
27 const float *filter_data, const tflite::RuntimeShape &bias_shape,
28 const float *bias_data, const tflite::RuntimeShape &output_shape,
29 float *output_data, const tflite::RuntimeShape &scratchpad_shape,
30 float *scratchpad_data)
31{
32 (void)scratchpad_shape;
33 if (scratchpad_data)
34 {
35 const int32_t batches = tflite::MatchingDim(input_shape, 0, output_shape, 0);
36 const int32_t input_depth = tflite::MatchingDim(input_shape, 3, filter_shape, 3);
37 const int32_t output_height = output_shape.Dims(1);
38 const int32_t output_width = output_shape.Dims(2);
39 const int32_t filter_height = filter_shape.Dims(1);
40 const int32_t filter_width = filter_shape.Dims(2);
41 tflite::RuntimeShape im2col_shape{batches, output_height, output_width,
42 input_depth * filter_height * filter_width};
43
44 tflite::optimized_ops::Conv(params, input_shape, input_data, filter_shape, filter_data,
45 bias_shape, bias_data, output_shape, output_data, im2col_shape,
46 scratchpad_data);
47 }
48 else
49 tflite::reference_ops::Conv(params, input_shape, input_data, filter_shape, filter_data,
50 bias_shape, bias_data, output_shape, output_data,
51 tflite::RuntimeShape(), nullptr);
52}

87static inline void SetupScratchpadTensor(luci_interpreter::Tensor *scratchpad,
88 const luci_interpreter::DataType &input_data_type,
89 const tflite::ConvParams &params,
90 const tflite::RuntimeShape &input_shape,
91 const tflite::RuntimeShape &filter_shape,
92 const tflite::RuntimeShape &output_shape)
93{
94 const int32_t filter_height = filter_shape.Dims(1);
95 const int32_t filter_width = filter_shape.Dims(2);
96
97 // Allocate tensor for scratchpad, if needed.
98 // The checks here should be aligned with the actual implementation.
99 const bool need_dilated_scratchpad =
100 params.dilation_height_factor != 1 || params.dilation_width_factor != 1;
101 const bool need_non_dilated_scratchpad = params.stride_height != 1 || params.stride_width != 1 ||
102 filter_height != 1 || filter_width != 1;
103 auto _need_scratchpad = input_data_type != luci_interpreter::DataType::S16 &&
104 (need_dilated_scratchpad || need_non_dilated_scratchpad);
105
106 if (_need_scratchpad)
107 {
108 const int32_t batches = tflite::MatchingDim(input_shape, 0, output_shape, 0);
109 const int32_t input_depth = tflite::MatchingDim(input_shape, 3, filter_shape, 3);
110 const int32_t output_height = output_shape.Dims(1);
111 const int32_t output_width = output_shape.Dims(2);
112
113 auto data_type_size = static_cast<int32_t>(luci_interpreter::getDataTypeSize(input_data_type));
114 int32_t scratchpad_size = batches * output_width * output_height * input_depth * filter_height *
115 filter_width * data_type_size;
116 luci_interpreter::Shape scratchpad_shape{scratchpad_size};
117 scratchpad->resize(scratchpad_shape);
118 }
119 else
120 {
121 scratchpad->set_allocatable(false);
122 }
123}
