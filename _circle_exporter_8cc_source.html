31CircleExporter::CircleExporter(const std::string &source, const std::string &path)
32 : _path{path}, _data{}, _model{nullptr}
33{
34 // make sure the architecture is little endian before direct access to flatbuffers
35 assert(FLATBUFFERS_LITTLEENDIAN);
36
37 std::ifstream src(source.c_str(), std::ios::binary);
38 if (src.is_open())
39 {
40 src.seekg(0, std::ios::end);
41 _data.resize(src.tellg());
42 src.seekg(0, std::ios::beg);
43 src.read(&_data[0], _data.size());
44 src.close();
45 }
46
47 if (_data.size() == 0)
48 throw std::runtime_error("Invalid source file");
49
50 const auto model = ::circle::GetModel(_data.data());
51 if (!model)
52 throw std::runtime_error("Failed to load original circle file");
53 _model.reset(model->UnPack());
54}

58void CircleExporter::updateWeight(const std::unique_ptr<exec::Execution> &exec)
59{
60 exec->iterateTrainableTensors(
61 [&](const ir::OperandIndex &idx, const backend::train::ITrainableTensor *tensor) {
62 const auto &subgs = _model->subgraphs;
63 if (subgs.size() != 1)
64 throw std::runtime_error("Circle does not has valid subgraph or has multiple subgraphs");
65
66 const auto &subg = subgs.at(0); // Get 1st subgraph
67 if (!idx.valid() || idx.value() >= subg->tensors.size())
68 throw std::runtime_error("Trainable tensor index is out of range");
69
70 const auto buf_idx = subg->tensors.at(idx.value())->buffer;
71 if (buf_idx >= _model->buffers.size())
72 throw std::runtime_error("Buffer for trainable tensors is invalid");
73
74 const auto &buffer = _model->buffers.at(buf_idx);
75
76 auto org_buf_sz = buffer->data.size();
77 if (org_buf_sz != tensor->total_size())
78 throw std::runtime_error("Trained tensor buffer size does not match original tensor's one");
79
80 memcpy(&buffer->data[0], tensor->buffer(), org_buf_sz);
81 });
82}

84void CircleExporter::finish()
85{
86 flatbuffers::FlatBufferBuilder builder(1024);
87 builder.Finish(::circle::Model::Pack(builder, _model.get()), ::circle::ModelIdentifier());
88
89 std::ofstream dst(_path.c_str(), std::ios::binary);
90 dst.write(reinterpret_cast<const char *>(builder.GetBufferPointer()), builder.GetSize());
91 dst.close();
92}
