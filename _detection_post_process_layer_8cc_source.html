119// sort indices in decreasing order of first `k` scores
120void PartialArgSort(const ContiguousSpan<float, true> &scores,
121 const ContiguousSpan<int, false> &indices, int k)
122{
123 std::iota(indices.begin(), indices.begin() + k, 0);
124 std::partial_sort(indices.begin(), indices.begin() + k, indices.begin() + scores.size(),
125 [&scores](const int i, const int j) { return scores[i] > scores[j]; });
126}

133Array<const CornerBox> decodeBoxes(const Array<float> &raw_boxes, const Array<float> &raw_anchors,
134 bool center_box_format, const CenterSizeBox &scales)
135{
136 auto nbatches = raw_boxes.shape().dim(0);
137 auto num_boxes = raw_boxes.shape().dim(1);
138
139 auto anchors = array_cast<const CenterSizeBox>(raw_anchors, {num_boxes});
140
141 if (!center_box_format)
142 {
143 auto boxes_p = reinterpret_cast<const CornerBox *>(raw_boxes.flat().data());
144 return {boxes_p, {num_boxes}};
145 }
146 else
147 {
148 // TODO support box center-width encoding correctly
149 // i.e anchors
150 auto boxes_p = reinterpret_cast<const CenterSizeBox *>(raw_boxes.flat().data());
151 Array<const CenterSizeBox> in_boxes{boxes_p, {num_boxes}};
152
153 auto decoded_boxes_p = new CornerBox[nbatches * num_boxes];
154 Array<CornerBox> decoded_boxes_a{decoded_boxes_p, {num_boxes}};
155
156 for (size_t i = 0; i < num_boxes; ++i)
157 {
158 const auto &anchor = anchors.at(i);
159 auto &box = decoded_boxes_a.at(i);
160 float yc = in_boxes.at(i).y / scales.y * anchor.h + anchor.y;
161 float xc = in_boxes.at(i).x / scales.x * anchor.w + anchor.x;
162 float halfh = 0.5f * std::exp(in_boxes.at(i).h / scales.h) * anchor.h;
163 float halfw = 0.5f * std::exp(in_boxes.at(i).w / scales.w) * anchor.w;
164 box.x1 = xc - halfw;
165 box.x2 = xc + halfw;
166 box.y1 = yc - halfh;
167 box.y2 = yc + halfh;
168
169 assert(box.x2 > box.x1);
170 assert(box.y2 > box.y1);
171 }
172
173 const auto &decoded_boxes_a_shape = decoded_boxes_a.shape();
174
175 return array_cast<const CornerBox>(std::move(decoded_boxes_a), decoded_boxes_a_shape);
176 }
177}

179float computeIOU(const CornerBox &box1, const CornerBox &box2)
180{
181 float area_i = (box1.y2 - box1.y1) * (box1.x2 - box1.x1);
182 float area_j = (box2.y2 - box2.y1) * (box2.x2 - box2.x1);
183 if (area_i <= 0 || area_j <= 0)
184 {
185 return 0.0;
186 }
187 float in_ymin = std::max<float>(box1.y1, box2.y1);
188 float in_xmin = std::max<float>(box1.x1, box2.x1);
189 float in_ymax = std::min<float>(box1.y2, box2.y2);
190 float in_xmax = std::min<float>(box1.x2, box2.x2);
191 float in_area = std::max<float>(in_ymax - in_ymin, 0.0) * std::max<float>(in_xmax - in_xmin, 0.0);
192
193 return in_area / (area_i + area_j - in_area);
194}

196int doSingleClass(const Array<const CornerBox> &boxes, const std::vector<float> &scores,
197 const NonMaxSuppressionParam &param, TemporaryArrays &temps,
198 size_t max_detections)
199{
200 auto num_boxes = boxes.shape().dim(0);
201
202 std::vector<int> sorted_box_indices(num_boxes);
203 PartialArgSort(ContiguousSpan<float, true>(scores.data(), num_boxes),
204 vecToSpan(sorted_box_indices), num_boxes);
205
206 // TODO move to temp allocations
207 std::vector<int> process_box(num_boxes, 1);
208
209 size_t selected_count = 0;
210 for (size_t i = 0; i < num_boxes; ++i)
211 {
212 auto box_index = sorted_box_indices[i];
213
214 if (!process_box[box_index] || scores[box_index] < param.score_threshold)
215 {
216 continue;
217 }
218
219 temps.selections.at(selected_count) = box_index;
220 selected_count++;
221
222 if (selected_count >= max_detections)
223 {
224 break;
225 }
226
227 for (size_t j = i + 1; j < num_boxes; ++j)
228 {
229 if (!process_box[sorted_box_indices[j]])
230 {
231 continue;
232 }
233
234 float IOU = computeIOU(boxes.at(box_index), boxes.at(sorted_box_indices[j]));
235 if (IOU > param.iou_threshold)
236 {
237 process_box[sorted_box_indices[j]] = 0;
238 }
239 }
240 }
241
242 return selected_count;
243}

245void collectBoxes(TemporaryArrays &temporary, const Array<const CornerBox> &decoded_boxes,
246 std::vector<float> &scores, int num_selected, OutputArrays &output,
247 const Array<int> &sorted_classes, int detections_per_box)
248{
249 auto &selections = temporary.selections;
250
251 size_t output_box_count = 0;
252
253 for (int i = 0; i < num_selected; ++i)
254 {
255 int selected_box = selections.at(output_box_count);
256
257 for (int c = 0; c < detections_per_box; ++c)
258 {
259 output.classes.at(output_box_count) = sorted_classes.at(selected_box, c);
260 output.scores.at(output_box_count) = scores[selected_box];
261 output.coords.at(output_box_count) = decoded_boxes.at(selected_box);
262 output_box_count++;
263 }
264 }
265}

267void DetectionPostProcess(const Array<float> &boxes_a, const Array<float> &scores_a,
268 Array<float> &num_selected_a, const NonMaxSuppressionParam &param,
269 const Allocations &allocations, OutputArrays &outputs)
270{
271 TemporaryArrays temporary(allocations.selections_buffer, param.max_detections);
272
273 // Only batch of 1 is supported atm
274 auto num_boxes = boxes_a.shape().dim(1);
275 size_t num_classes = param.num_classes;
276 size_t num_classes_with_background = scores_a.shape().dim(2);
277 bool have_background = num_classes_with_background != num_classes;
278
279 size_t max_classes_per_box = std::min<size_t>(num_classes, param.max_classes_per_detection);
280
281 // TODO move this to allocations
282 std::vector<int> sorted_class_indices(num_boxes * num_classes);
283
284 Array<int> class_indices(sorted_class_indices.data(), {num_boxes, num_classes});
285
286 // TODO move to allocations
287 std::vector<float> max_scores(num_boxes);
288
289 for (size_t row = 0; row < num_boxes; row++)
290 {
291 auto box_scores = scores_a.slice(0, row).offset(have_background ? 1 : 0);
292 auto indices = class_indices.slice(row);
293
294 PartialArgSort(box_scores, indices, num_classes);
295
296 max_scores[row] = box_scores[indices[0]];
297 }
298
299 auto anchors_a =
300 Array<float>(reinterpret_cast<float *>(param.anchors_input->buffer()), {num_boxes, 4});
301 auto decoded_boxes = decodeBoxes(boxes_a, anchors_a, param.center_box_format, param.scales);
302
303 int num_selected =
304 doSingleClass(decoded_boxes, max_scores, param, temporary, param.max_detections);
305
306 collectBoxes(temporary, decoded_boxes, max_scores, num_selected, outputs, class_indices,
307 max_classes_per_box);
308
309 num_selected_a.at(0) = num_selected;
310}

330void DetectionPostProcessLayer::run()
331{
332 auto nbatches = (unsigned int)_parameters.boxes_descr[0];
333 // no suport for batch other than 1( it's fine since tflite does not support
334 // batch for postprocess either )
335 assert(nbatches == 1);
336
337 auto boxes_a = toArray<float>(_parameters.boxes_input->buffer(), _parameters.boxes_descr);
338 auto scores_a = toArray<float>(_parameters.scores_input->buffer(), _parameters.scrores_descr);
339
340 auto num_selected_a = ndarray::Array<float>(
341 reinterpret_cast<float *>(_parameters.num_selections_output->buffer()), {nbatches});
342
343 OutputArrays outputArrays(reinterpret_cast<CornerBox *>(_parameters.box_coords_output->buffer()),
344 reinterpret_cast<float *>(_parameters.box_scores_output->buffer()),
345 reinterpret_cast<float *>(_parameters.box_classes_output->buffer()),
346 reinterpret_cast<int *>(_parameters.num_selections_output->buffer()),
347 _parameters.max_detections);
348
349 DetectionPostProcess(boxes_a, scores_a, num_selected_a, _parameters, _allocations, outputArrays);
350}
