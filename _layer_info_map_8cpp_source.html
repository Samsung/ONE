28bool is_multiple_output_node(const luci::CircleNode *node)
29{
30 switch (node->opcode())
31 {
32 // The following nodes have multiple outputs. Output tensors are not produced by themselves but
33 // by the corresponding *Out nodes.
34 case luci::CircleOpcode::SPLIT:
35 case luci::CircleOpcode::SPLIT_V:
36 case luci::CircleOpcode::TOPK_V2:
37 case luci::CircleOpcode::UNIQUE:
38 case luci::CircleOpcode::UNPACK:
39 return true;
40 // TODO: Support ops
41 case luci::CircleOpcode::BIDIRECTIONAL_SEQUENCE_LSTM:
42 case luci::CircleOpcode::CUSTOM:
43 case luci::CircleOpcode::IF:
44 case luci::CircleOpcode::NON_MAX_SUPPRESSION_V4:
45 case luci::CircleOpcode::NON_MAX_SUPPRESSION_V5:
46 case luci::CircleOpcode::WHILE:
47 throw std::runtime_error("Unsupported op now");
48 default:
49 return false;
50 }
51}

53const luci::CircleNode *get_multi_output_node(const luci::CircleNode *node)
54{
55 if (is_multiple_output_node(node))
56 return node;
57
58 switch (node->opcode())
59 {
60 // The following nodes denote outputs of multiple-output nodes.
61 case luci::CircleOpcode::CIRCLESPLITOUT:
62 {
63 const auto split_out = loco::must_cast<const CircleSplitOut *>(node);
64 return loco::must_cast<luci::CircleNode *>(split_out->input());
65 }
66 case luci::CircleOpcode::CIRCLESPLITVOUT:
67 {
68 const auto splitv_out = loco::must_cast<const CircleSplitVOut *>(node);
69 return loco::must_cast<luci::CircleNode *>(splitv_out->input());
70 }
71 case luci::CircleOpcode::CIRCLETOPKV2OUT:
72 {
73 const auto top_kv2_out = loco::must_cast<const CircleTopKV2Out *>(node);
74 return loco::must_cast<luci::CircleNode *>(top_kv2_out->input());
75 }
76 case luci::CircleOpcode::CIRCLEUNIQUEOUT:
77 {
78 const auto unique_out = loco::must_cast<const CircleUniqueOut *>(node);
79 return loco::must_cast<luci::CircleNode *>(unique_out->input());
80 }
81 case luci::CircleOpcode::CIRCLEUNPACKOUT:
82 {
83 const auto unpack_out = loco::must_cast<const CircleUnpackOut *>(node);
84 return loco::must_cast<luci::CircleNode *>(unpack_out->input());
85 }
86 // TODO: Support these ops
87 case luci::CircleOpcode::CIRCLEBIDIRECTIONAL_SEQUENCE_LSTM_OUT:
88 case luci::CircleOpcode::CIRCLECUSTOMOUT:
89 case luci::CircleOpcode::CIRCLEIFOUT:
90 case luci::CircleOpcode::CIRCLENONMAXSUPPRESSIONV4OUT:
91 case luci::CircleOpcode::CIRCLENONMAXSUPPRESSIONV5OUT:
92 case luci::CircleOpcode::CIRCLEWHILEOUT:
93 throw std::runtime_error("Unsupported op now");
94 default:
95 return nullptr;
96 }
97}

104void add_multi_output_node(LayerInfoMap &info_by_name, LayerInfo &layer_info,
105 const luci::CircleNode *node)
106{
107 assert(is_multiple_output_node(node)); // FIX_CALLER_UNLESS
108
109 const auto succs_nodes = loco::succs(node);
110 const auto name = node->name();
111
112 if (info_by_name.find(name) != info_by_name.end())
113 {
114 // Check that all outputs have equal dtype and granularity
115 for (const auto succs_node : succs_nodes)
116 {
117 const auto succs_circle_node = loco::must_cast<luci::CircleNode *>(succs_node);
118
119 const auto it = info_by_name.find(succs_circle_node->name());
120 if (it != info_by_name.end() and not same_setting(layer_info, (it->second)))
121 throw std::runtime_error("Outputs of multiple-output nodes should have equal dtype and "
122 "granularity. Check the quantization configuration file");
123 }
124 return;
125 }
126
127 // Add multiple output node to info_by_name
128 info_by_name[name] = {name, layer_info.dtype, layer_info.granularity};
129
130 // Add outputs node to info_by_name
131 for (const auto succs_node : succs_nodes)
132 {
133 const auto succs_circle_node = loco::must_cast<luci::CircleNode *>(succs_node);
134 const auto succs_circle_node_name = succs_circle_node->name();
135 info_by_name[succs_circle_node_name] = {succs_circle_node_name, layer_info.dtype,
136 layer_info.granularity};
137 }
138}

142LayerInfoMap layer_info_map(loco::Graph *g, std::vector<LayerInfo> &layers_info)
143{
144 LayerInfoMap info_by_name;
145
146 for (auto &&info : layers_info)
147 {
148 auto name = info.name;
149 bool found = false;
150 for (auto node : loco::active_nodes(loco::output_nodes(g)))
151 {
152 auto cnode = loco::must_cast<luci::CircleNode *>(node);
153 if (cnode->opcode() == luci::CircleOpcode::CIRCLEOUTPUT)
154 continue;
155
156 if (cnode->name() == name)
157 {
158 // Check and add multiple-output node and its outputs to info_by_name
159 if (const auto multi_output = get_multi_output_node(cnode))
160 {
161 add_multi_output_node(info_by_name, info, multi_output);
162 found = true;
163 continue;
164 }
165
166 if (info_by_name.find(name) != info_by_name.end())
167 {
168 throw std::runtime_error("Duplicate layer name " + name +
169 ". Check layer names in the quantization configuration file.");
170 }
171
172 info_by_name[name] = info;
173 found = true;
174 continue;
175 }
176 }
177
178 if (not found)
179 throw std::runtime_error("No such layer named " + name +
180 ". Check layer names in the quantization configuration file.");
181 }
182
183 // TODO Check all names in layers_info exist in the info_by_name
184 // TODO Check names in info_by_name but not in layers_info are from virtual outputs
185
186 return info_by_name;
187}
