27bool satisfy_precondition(luci::CircleFullyConnected *fc)
28{
29 // check if it's already been shuffled
30 if (fc->weights_format() != luci::CircleFullyConnected::WeightsFormat::DEFAULT)
31 return false;
32
33 // check if its data type is FLOAT32
34 if (fc->dtype() != loco::DataType::FLOAT32)
35 return false;
36
37 auto weights = loco::must_cast<luci::CircleConst *>(fc->weights());
38 // rank must be 2
39 if (weights->rank() != 2)
40 return false;
41
42 // check if it has sparsity parameter
43 if (weights->sparsityparam())
44 return false;
45
46 // check if the number of row of FullyConnected's weight is a multiple of 16
47 const uint32_t MULTIPLE = 16;
48 uint32_t rows = weights->dim(0).value();
49 if (rows % MULTIPLE)
50 return false;
51
52 return true;
53}

56void get_FCs_having_same_tensor(std::vector<luci::CircleFullyConnected *> &fc_vec, loco::Graph *g,
57 luci::CircleFullyConnected *fc)
58{
59 auto the_tensor = fc->weights();

61 {
62 auto fc = dynamic_cast<luci::CircleFullyConnected *>(node);
63 if (not fc)
64 continue;
65
66 if (fc->weights() == the_tensor)
67 fc_vec.push_back(fc);
68 }

71luci::CircleConst *shuffle_weight(luci::CircleFullyConnected *fc)
72{
73 auto the_weights = loco::must_cast<luci::CircleConst *>(fc->weights());
74
75 auto name = fc->name();
76 assert(name.length() > 0);
77
78 // create CircleConst where shuffled data will be stored
79 luci::CircleConst *new_weights = fc->graph()->nodes()->create<luci::CircleConst>();
80 new_weights->dtype(loco::DataType::FLOAT32);
81 new_weights->size<loco::DataType::FLOAT32>(the_weights->size<loco::DataType::FLOAT32>());
82 new_weights->rank(the_weights->rank());
83 new_weights->shape_status(the_weights->shape_status());
84 for (uint32_t r = 0; r < new_weights->rank(); r++)
85 {
86 new_weights->dim(r).set(the_weights->dim(r).value());
87 }
88 new_weights->name(name + "/shuffle_weight");
89
90 // suffle weight
91 const uint32_t MULTIPLE = 16;
92 const uint32_t rows = the_weights->dim(0).value();
93 const uint32_t cols = the_weights->dim(1).value();
94 const uint32_t r_step = rows / MULTIPLE;
95 uint32_t index = 0;
96 for (uint32_t r = 0; r < r_step; r++)
97 {
98 for (uint32_t c = 0; c < cols; c++)
99 {
100 for (uint32_t i = 0; i < MULTIPLE; i++)
101 {
102 new_weights->at<loco::DataType::FLOAT32>(index++) =
103 the_weights->at<loco::DataType::FLOAT32>((r * MULTIPLE + i) * cols + c);
104 }
105 }
106 }
107
108 return new_weights;
109}

116bool ShuffleWeightTo16x1Float32Pass::run(loco::Graph *g)
117{
118 bool changed = false;
119 for (auto node : loco::active_nodes(loco::output_nodes(g)))
120 {
121 auto fc = dynamic_cast<luci::CircleFullyConnected *>(node);
122 if (not fc)
123 continue;
124
125 if (not satisfy_precondition(fc))
126 continue;
127
128 std::vector<luci::CircleFullyConnected *> fc_vec;
129 get_FCs_having_same_tensor(fc_vec, g, fc);
130 auto new_weights = shuffle_weight(fc);
131
132 // replace to new weights
133 for (const auto fc : fc_vec)
134 {
135 fc->weights(new_weights);
136 fc->weights_format(luci::CircleFullyConnected::WeightsFormat::SHUFFLED16x1FLOAT32);
137 }
138
139 changed = true;
140 }
141
142 return changed;
143}
