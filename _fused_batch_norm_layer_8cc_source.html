29void KernelGenerator::visit(const ir::operation::FusedBatchNorm &node)
30{
31 const auto ofm_index{node.getOutputs().at(0)};
32
33 auto output_tensor = _tensor_reg->getPortableTensor(ofm_index);
34 std::vector<const IPortableTensor *> input_tensors;
35 for (const auto &ifm_idx : node.getInputs())
36 input_tensors.emplace_back(_tensor_reg->getPortableTensor(ifm_idx));
37
38 const auto epsilon = node.param().epsilon;
39 const auto is_training = node.param().is_training;
40 const auto &data_format = node.param().data_format;
41
42 auto fn = std::make_unique<ops::FusedBatchNormLayer>();
43
44 fn->configure(input_tensors, epsilon, is_training, data_format, output_tensor);
45
46 _return_fn = std::move(fn);
47}

63void FusedBatchNormLayer::fusedbatchnormFloat32()
64{
65 uint32_t num_inputs = _inputs.size();
66 nnfw::cker::FusedBatchNorm &kernel = *_fusedbatchnorm_kernel;
67
68 kernel.prepare();
69
70 std::vector<nnfw::cker::Shape> inputShapes;
71 std::vector<const float *> inputFloatPtrs;
72
73 for (uint32_t i = 0; i < num_inputs; i++)
74 {
75 inputShapes.emplace_back(getShape(_inputs[i]));
76 inputFloatPtrs.emplace_back(getBuffer<float>(_inputs[i]));
77 }
78
79 nnfw::cker::FusedBatchNormParams param;
80
81 param.epsilon = _epsilon;
82 param.is_training = _is_training;
83 param.data_format = _data_format;
84
85 kernel(inputShapes, inputFloatPtrs, getShape(_output), getBuffer<float>(_output), param);
86}
