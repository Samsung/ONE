33void FusedBatchNormLayer::fusedbatchnormFloat32()
34{
35 uint32_t num_inputs = _inputs.size();
36 nnfw::cker::FusedBatchNorm &kernel = *_fusedbatchnorm_kernel;
37
38 kernel.prepare();
39
40 std::vector<nnfw::cker::Shape> inputShapes;
41 std::vector<const float *> inputFloatPtrs;
42
43 for (uint32_t i = 0; i < num_inputs; i++)
44 {
45 inputShapes.emplace_back(getShape(_inputs[i]));
46 inputFloatPtrs.emplace_back(getBuffer<float>(_inputs[i]));
47 }
48
49 nnfw::cker::FusedBatchNormParams param;
50
51 param.epsilon = _epsilon;
52 param.is_training = _is_training;
53 param.data_format = _data_format;
54
55 kernel(inputShapes, inputFloatPtrs, getShape(_output), getBuffer<float>(_output), param);
56}

58void FusedBatchNormLayer::run()
59{

62 fusedbatchnormFloat32();
