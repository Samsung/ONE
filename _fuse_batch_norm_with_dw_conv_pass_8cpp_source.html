58bool is_scale_shift_shape(luci::CircleConst *node)
59{
60 auto rank = node->rank();
61 if (rank != 1 && rank != 4)
62 return false;
63 for (uint32_t r = 0; r < rank - 1; ++r)
64 {
65 if (node->dim(r).value() != 1)
66 return false;
67 }
68 return true;
69}

71bool fused_batch_norm_with_dwconv(luci::CircleAdd *add)
72{
73 assert(add != nullptr);
74
75 // Find the pattern of CircleDepthwiseConv2D - CircleMul - CircleAdd
76 luci::CircleConst *scale = nullptr;
77 luci::CircleConst *shift = nullptr;
78 luci::CircleDepthwiseConv2D *dwconv = nullptr;
79 luci::CircleMul *mul = nullptr;
80 if (not luci::fill(&shift, &mul).with_commutative_args_of(add))
81 return false;
82 if (not luci::fill(&scale, &dwconv).with_commutative_args_of(mul))
83 return false;
84
85 // check scale and shift constant attributes
86 // scale and shift can be [x] or [1, 1, 1, x]
87 if (not is_scale_shift_shape(scale))
88 return false;
89 if (not is_scale_shift_shape(shift))
90 return false;
91
92 // check mul, add attributes
93 if (mul->dtype() != loco::DataType::FLOAT32)
94 return false;
95 if (mul->fusedActivationFunction() != luci::FusedActFunc::NONE)
96 return false;
97 if (add->dtype() != loco::DataType::FLOAT32)
98 return false;
99
100 // get weight of dwconv
101 auto filter = dynamic_cast<luci::CircleConst *>(dwconv->filter());
102 if (not filter)
103 return false;
104 if (filter->dtype() != loco::DataType::FLOAT32)
105 return false;
106 if (filter->rank() != 4)
107 return false;
108
109 // check attributes of dwconv
110 if (dwconv->fusedActivationFunction() != luci::FusedActFunc::NONE)
111 return false;
112 if (dwconv->depthMultiplier() < 0) // can this happen?
113 return false;
114
115 // get bias of dwconv
116 auto bias = dynamic_cast<luci::CircleConst *>(dwconv->bias());
117 if (not bias)
118 return false;
119 if (bias->dtype() != loco::DataType::FLOAT32)
120 return false;
121 if (bias->rank() != 1)
122 return false;
123
124 // filter represents as [1, H, W, C*M] where M is multiplier.
125 auto filter_out_chn = filter->dim(3).value();
126 auto multiplier = static_cast<uint32_t>(dwconv->depthMultiplier());
127 auto srank = scale->rank(); // as rank can be 1 or 4
128 if (filter_out_chn != scale->dim(srank - 1).value() * multiplier)
129 return false;
130 srank = shift->rank();
131 if (filter_out_chn != shift->dim(srank - 1).value() * multiplier)
132 return false;
133 auto channel = filter_out_chn / multiplier;
134
135 auto name = add->name();
136 assert(name.length() > 0);
137
138 loco::Graph *graph = add->graph();
139 luci::CircleDepthwiseConv2D *fused_dwconv = graph->nodes()->create<luci::CircleDepthwiseConv2D>();
140 luci::CircleConst *fused_filter = graph->nodes()->create<luci::CircleConst>();
141 luci::CircleConst *fused_bias = graph->nodes()->create<luci::CircleConst>();
142
143 auto filter_in_chn = filter->dim(0).value();
144 auto filter_height = filter->dim(1).value();
145 auto filter_width = filter->dim(2).value();
146 assert(filter_in_chn == 1);
147
148 // Copy filter shape
149 fused_filter->dtype(filter->dtype());
150 fused_filter->size<loco::DataType::FLOAT32>(filter->size<loco::DataType::FLOAT32>());
151 fused_filter->rank(4);
152 fused_filter->dim(0).set(filter_in_chn);
153 fused_filter->dim(1).set(filter_height);
154 fused_filter->dim(2).set(filter_width);
155 fused_filter->dim(3).set(filter_out_chn);
156 fused_filter->shape_status(luci::ShapeStatus::VALID);
157 fused_filter->name(name + "/DepthwiseConv2D/filter");
158
159 // fused filter weight = filter weight * mul(scale) + add(shift)
160 for (uint32_t b = 0; b < filter_in_chn; b++)
161 {
162 for (uint32_t h = 0; h < filter_height; h++)
163 {
164 for (uint32_t w = 0; w < filter_width; w++)
165 {
166 for (uint32_t c = 0; c < filter_out_chn; c++)
167 {
168 uint32_t offset = b * filter_height * filter_width * filter_out_chn +
169 h * filter_width * filter_out_chn + w * filter_out_chn + c;
170 uint32_t chn = c / multiplier;
171 fused_filter->at<loco::DataType::FLOAT32>(offset) =
172 filter->at<loco::DataType::FLOAT32>(offset) * scale->at<loco::DataType::FLOAT32>(chn);
173 }
174 }
175 }
176 }
177
178 // Fuse bias with scale and shift
179 fused_bias->dtype(shift->dtype());
180 fused_bias->size<loco::DataType::FLOAT32>(shift->size<loco::DataType::FLOAT32>());
181 fused_bias->rank(1);
182 fused_bias->dim(0).set(channel);
183 fused_bias->shape_status(luci::ShapeStatus::VALID);
184 for (uint32_t c = 0; c < channel; ++c)
185 {
186 fused_bias->at<loco::DataType::FLOAT32>(c) =
187 bias->at<loco::DataType::FLOAT32>(c) * scale->at<loco::DataType::FLOAT32>(c) +
188 shift->at<loco::DataType::FLOAT32>(c);
189 }
190 fused_bias->name(name + "/DepthwiseConv2D/bias");
191
192 // set new tconv properties
193 fused_dwconv->input(dwconv->input());
194 fused_dwconv->filter(fused_filter);
195 fused_dwconv->bias(fused_bias);
196 fused_dwconv->fusedActivationFunction(add->fusedActivationFunction());
197 fused_dwconv->padding(dwconv->padding());
198 fused_dwconv->stride()->h(dwconv->stride()->h());
199 fused_dwconv->stride()->w(dwconv->stride()->w());
200 fused_dwconv->depthMultiplier(dwconv->depthMultiplier());
201 fused_dwconv->dilation()->h(dwconv->dilation()->h());
202 fused_dwconv->dilation()->w(dwconv->dilation()->w());
203 fused_dwconv->name(name + "/DepthwiseConv2D");
204 luci::add_origin(fused_dwconv,
205 luci::composite_origin(
206 {luci::get_origin(add), luci::get_origin(mul), luci::get_origin(dwconv)}));
207
208 replace(add).with(fused_dwconv);
209
210 return true;
211}

218bool FuseBatchNormWithDwConvPass::run(loco::Graph *g)
219{
220 bool changed = false;
221 for (auto node : loco::active_nodes(loco::output_nodes(g)))
222 {
223 if (auto add = dynamic_cast<luci::CircleAdd *>(node))
224 {
225 if (fused_batch_norm_with_dwconv(add))
226 changed = true;
227 }
228 }
229
230 return changed;
231}
