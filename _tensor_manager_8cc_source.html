26template <typename MemoryManager, typename TensorMap>
27void allocateMemory(MemoryManager *mgr, const TensorMap &tensors, const std::string tensor_type)
28{
29 mgr->allocate();
30
31 for (auto &&[index, tensor] : tensors)
32 {
33 assert(!tensor->is_dynamic());
34
35 auto *buffer = mgr->getBuffer(index);
36 tensor->setBuffer(buffer);
37 VERBOSE(TensorManager) << tensor_type << index << " : " << static_cast<void *>(buffer)
38 << std::endl;
39 }
40}

42inline size_t alignedSize(const size_t size, const uint64_t align)
43{
44 return (((size) + ((align)-1)) & ~((align)-1));

72void TensorManager::allocateTrainableTensors()
73{
74 allocateMemory(_trainable_mgr.get(), _tensors->trainable_tensors(),
75 std::string{" TRAINABLE TENSOR "});
76
77 // Set buffers of optimizer variables
78 // Allocating optimizer variables has been done when calling allocate() of TrainableMemoryManager
79 for (const auto &[index, trainable_tensor] : _tensors->trainable_tensors())
80 {
81 for (uint32_t var_pos = 0; var_pos < trainable_tensor->optVars().size(); ++var_pos)
82 {
83 auto *buffer = _trainable_mgr.get()->getOptVarBuffer(index, var_pos);
84 trainable_tensor->setOptVarBuffer(buffer, var_pos);
85 VERBOSE(TensorManager) << std::string{" OPTIM_VAR TENSOR "} << index << " : "
86 << static_cast<void *>(buffer) << std::endl;
87 }
88 }
89}

109void TensorManager::claimNonConstPlan(const ir::OperandIndex &index)
110{
111 auto tensor = _tensors->getNonConstTensor(index);
112 assert(tensor && !tensor->is_dynamic());
113
114 auto size = alignedSize(tensor->total_size(), _align);
115 _nonconst_mgr->claimPlan(index, size);
