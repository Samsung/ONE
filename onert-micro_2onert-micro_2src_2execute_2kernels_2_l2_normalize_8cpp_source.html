38OMStatus onert_micro::execute::execute_kernel_CircleL2Normalize(const OMExecuteArgs &execute_args)
39{
40 core::OMRuntimeContext &runtime_context = execute_args.runtime_context;
41 core::OMRuntimeStorage &runtime_storage = execute_args.runtime_storage;
42 uint16_t op_index = execute_args.kernel_index;
43
44 const circle::Tensor *input = nullptr;
45 const circle::Tensor *output = nullptr;
46
47 uint8_t *input_data = nullptr;
48 uint8_t *output_data = nullptr;
49
50 OMStatus status = Ok;
51
52 {
53 OMRuntimeKernel runtime_kernel;
54 runtime_kernel.readKernel(op_index, runtime_context);
55
56 input = runtime_kernel.inputs[inputTensorIdx];
57 output = runtime_kernel.outputs[outputTensorIdx];
58
59 assert(input != nullptr);
60 assert(output != nullptr);
61
62 status = runtime_kernel.getDataFromStorage(op_index, runtime_storage, runtime_context);
63 if (status != Ok)
64 return status;
65
66 input_data = runtime_kernel.inputs_data[inputTensorIdx];
67 output_data = runtime_kernel.outputs_data[outputTensorIdx];
68 }
69
70 assert(input_data != nullptr);
71 assert(output_data != nullptr);
72
73 switch (input->type())
74 {
75#ifndef DIS_FLOAT
76 case circle::TensorType_FLOAT32:
77 {
78
79 core::OMRuntimeShape inputs_shape(input);
80 core::OMRuntimeShape outputs_shape(output);
81
82 const auto trailing_dim = inputs_shape.dimensionsCount() - 1;
83
84 core::L2NormalizationParams params;
85 params.num_rows =
86 pal::flatSizeSkipDim(inputs_shape.dimsData(), trailing_dim, inputs_shape.dimensionsCount());
87
88 assert(inputs_shape.dims(trailing_dim) == outputs_shape.dims(trailing_dim));
89 params.row_size = inputs_shape.dims(trailing_dim);
90
91 status = pal::L2Normalization(params, core::utils::castInputData<float>(input_data),
92 core::utils::castOutputData<float>(output_data));
93 }
94 break;
95#endif // DIS_FLOAT
96 default:
97 {
98 status = UnsupportedType;
99 assert(false && "Unsupported type.");
100 }
101 }
102
103 return status;
104}
