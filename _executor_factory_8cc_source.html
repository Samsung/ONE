49class SyncFunction final : public exec::IFunction
50{
51public:
52 virtual ~SyncFunction() = default;
53 SyncFunction(std::unique_ptr<exec::IFunction> fn, const std::shared_ptr<backend::IConfig> config)
54 : _fn{std::move(fn)}, _config{config}
55 {
56 assert(_fn);
57 assert(_config);
58 }
59
60 void run() override
61 {
62 _fn->run();
63 _config->sync();
64 }
65
66 void prepare() override { _fn->prepare(); }
67
68private:
69 std::unique_ptr<exec::IFunction> _fn;
70 std::shared_ptr<backend::IConfig> _config;
71};

73using DeallocList = std::vector<backend::ITensor *>;
74// Deallocation after execution of an operation used by Linear Executor
75class DeallocFunction final : public exec::IFunction
76{
77public:
78 DeallocFunction(const DeallocList &tensors) : _dealloc_list{tensors} {}
79
80 void run() override
81 {
82 for (auto &&tensor : _dealloc_list)
83 {
84 if (!tensor->is_dynamic())
85 continue;
86 tensor->deallocBuffer();
87 }
88 }
89
90private:
91 DeallocList _dealloc_list;
92};

146backend::BackendContexts
147createBackendContexts(compiler::ILoweredGraph &lgraph, bool linear_executor,
148 std::shared_ptr<backend::custom::IKernelBuilder> custom_kernel_builder)
149{
150 backend::BackendContexts contexts;
151 std::unordered_map<const backend::Backend *, backend::ContextData> context_data_map;
152
153 // Generate partial graphs for each backend
154 auto init_context_data = [&](const backend::Backend *backend) {
155 auto &data = context_data_map[backend];
156 auto graph = std::make_unique<ir::Graph>();
157 data.graph = std::move(graph);
158 };
159
160 auto &whole_graph = lgraph.graph();
161 // Separate operands into partial graphs
162 whole_graph.operands().iterate([&](const ir::OperandIndex &operand_ind, ir::Operand &operand) {
163 auto &operand_li = lgraph.lower_info().operand;
164 const auto &def_backends = operand_li.at(operand_ind).def_backends();
165 if (def_backends.size() == 0) // Ignore unused tensor
166 return;
167 const auto backend = def_backends.getOnlyElement();
168 if (context_data_map.find(backend) == context_data_map.end())
169 init_context_data(backend);
170
171 auto &partial_graph = *context_data_map[backend].graph;
172
173 // Copy the operand and insert it to the partial graph
174 auto new_operand = std::make_unique<ir::Operand>(operand);
175 new_operand->clearDefUse();
176 operand.releaseData(); // Deref data of LoweredGraph
177 [[maybe_unused]] auto new_operand_ind =
178 partial_graph.addOperand(operand_ind, std::move(new_operand));
179 assert(new_operand_ind == operand_ind);
180 });
181 // Separate operations into partial graphs
182 whole_graph.operations().iterate(
183 [&](const ir::OperationIndex &op_ind, const ir::IOperation &operation) {
184 auto &op_li = lgraph.lower_info().operation;
185 const auto backend = op_li.at(op_ind);
186 if (context_data_map.find(backend) == context_data_map.end())
187 init_context_data(backend);
188
189 auto &partial_graph = *context_data_map[backend].graph;
190 auto &external_operands = context_data_map[backend].external_operands;
191
192 {
193 // Add missing operands (externals)
194 auto io_list = (operation.getInputs() + operation.getOutputs()) | ir::Remove::DUPLICATED |
195 ir::Remove::UNDEFINED;
196 for (auto &&operand_ind : io_list)
197 {
198 if (partial_graph.operands().exist(operand_ind))
199 continue;
200
201 // Copy the operand and insert it to the partial graph
202 const auto &operand = whole_graph.operands().at(operand_ind);
203 auto new_operand = std::make_unique<ir::Operand>(operand);
204 new_operand->clearDefUse();
205 [[maybe_unused]] auto new_operand_ind =
206 partial_graph.addOperand(operand_ind, std::move(new_operand));
207 assert(new_operand_ind == operand_ind);
208
209 external_operands.add(operand_ind);
210 }
211
212 [[maybe_unused]] auto new_op_ind = partial_graph.addOperation(op_ind, clone(operation));
213 assert(new_op_ind == op_ind);
214 }
215 });
216
217 // Create contexts
218 auto whole_op_order = lgraph.graph().topolSortOperations();
219 for (auto &&[backend, data] : context_data_map)
220 {
221 auto graph = data.graph.get();
222 auto &external_operands = data.external_operands;
223 // Handle graph input/outputs or external tensors
224 graph->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &operand) {
225 if (whole_graph.getInputs().contains(ind) || whole_graph.getOutputs().contains(ind))
226 external_operands.add(ind);
227 // Inputs are either "graph input" or "no def op and non-constant"
228 if (whole_graph.getInputs().contains(ind) ||
229 (!operand.getDef().valid() && !operand.isConstant()))
230 // Outputs are either "graph output" or "no uses"
231 graph->addInput(ind);
232 if (whole_graph.getOutputs().contains(ind) || operand.getUses().size() == 0)
233 graph->addOutput(ind);
234 });
235 VERBOSE(ExecutorFactory) << "createBackendContexts: partial graph for backend="
236 << backend->config()->id() << std::endl;
237 dumper::text::dumpGraph(*data.graph);
238
239 auto &op_order = data.op_order;
240 std::copy_if(whole_op_order.begin(), whole_op_order.end(), std::back_inserter(op_order),
241 [&](const auto &ind) { return graph->operations().exist(ind); });
242 data.is_linear_executor = linear_executor;
243 data.custom_kernel_builder = custom_kernel_builder;
244 contexts.emplace(backend, backend->newContext(std::move(data)));
245 }
246 return contexts;
247}

249template <typename Context>
250std::deque<std::pair<const backend::Backend *, Context *>> orderBackendContext(
251 const std::unordered_map<const backend::Backend *, std::unique_ptr<Context>> &tbackend_contexts)
252{
253 std::deque<std::pair<const backend::Backend *, Context *>> ordered_contexts;
254
255 for (auto &&[backend, context] : tbackend_contexts)
256 {
257 // NOTE builtin backend must be processed lastly.
258 // This is because of Permute layer's specialty which is the only operation that could have
259 // different ITensor objects for the input and the output. And it requires all other backends'
260 // tensors are ready to use.
261 if (backend->config()->id() == "builtin")
262 ordered_contexts.emplace_back(backend, context.get());
263 else
264 ordered_contexts.emplace_front(backend, context.get());
265 }
266
267 return ordered_contexts;
268}

316void ExecutorFactory::prepareMigrantTensors(compiler::ILoweredGraph &lowered_graph,
317 const backend::BackendContexts &backend_contexts)
318{
319 TensorRegistries tensor_regs{backend_contexts, true};
320
321 lowered_graph.graph().operations().iterate(
322 [&](const ir::OperationIndex &op_ind, const ir::IOperation &op) {
323 const auto backend = lowered_graph.lower_info().operation.at(op_ind);
324 auto &backend_ctx = backend_contexts.at(backend);
325 for (auto &&ind :
326 (op.getInputs() + op.getOutputs()) | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED)
327 {
328 // If an Operation's input/output tensor does not have an own tensor object,
329 // it must be using migrant tensors, so find the tensor from other tensor registries and
330 // register it to the current tensor registry if it is portable
331 if (!backend_ctx->tensor_registry->getITensor(ind))
332 {
333 auto tensor = tensor_regs.getITensor(ind);
334 assert(tensor); // The tensor must have been registered
335 auto ptensor = dynamic_cast<backend::IPortableTensor *>(tensor);
336 if (ptensor)
337 backend_ctx->tensor_registry->setMigrantTensor(ind, ptensor);
338 }
339 }
340 });
341}

379exec::IExecutor *
380ExecutorFactory::createLinearExecutor(std::unique_ptr<compiler::LoweredGraph> lowered_graph,
381 const std::shared_ptr<exec::IExecutors> &executors,
382 const ExecutorFactoryArgs &args)
383{
384 const auto options = args.options;
385 const auto &model_index = args.model_index;
386 const auto tracing_ctx = args.tracing_ctx;
387 auto custom_kernel_builder = args.custom_kernel_builder;
388 auto &graph = lowered_graph->graph();
389
390 backend::BackendContexts backend_contexts =
391 createBackendContexts(*lowered_graph, options->executor == "Linear", custom_kernel_builder);
392
393 TensorRegistries tensor_regs{backend_contexts, true};
394
395 initializeSubgraphIOTensors(
396 *lowered_graph, backend_contexts,
397 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
398 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
399
400 // linearize
401 auto order = Linear::linearize(*lowered_graph);
402 Linear::dump(*lowered_graph, order);
403
404 for (auto &&pair : backend_contexts)
405 {
406 pair.second->genTensors();
407 }
408
409 prepareMigrantTensors(*lowered_graph, backend_contexts);
410
411 // Give some runtime objects to builtin KernelGenerator
412 prepareBuiltinBackend(tensor_regs, executors, backend_contexts, model_index);
413
414 ExecutionBuilder builder;
415
416 // Adjust the order of backends for the upcoming iteration
417 auto ordered_contexts = orderBackendContext(backend_contexts);
418
419 // Simulate the execution for deallocation of tensors
420 std::unordered_map<ir::OperationIndex, DeallocList> dealloc_list_map;
421 {
422 ir::OperandIndexMap<uint32_t> uses_map;
423 ir::OperandIndexSequence constants;
424
425 auto model_io =
426 (graph.getInputs() + graph.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
427
428 // Prepare scanning
429 graph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
430 uses_map[ind] = obj.getUses().size();
431
432 if (obj.isConstant())
433 constants.append(ind);
434 });
435
436 // A trick to consider constants as an execption
437 for (const auto &ind : constants)
438 {
439 uses_map[ind]++;
440 }
441
442 for (const auto &op_ind : order)
443 {
444 const auto &op = graph.operations().at(op_ind);
445 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
446 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
447
448 for (const auto &ind : op_inputs)
449 {
450 const auto &operand = graph.operands().at(ind);
451 assert(uses_map.find(ind) != uses_map.end());
452 assert(uses_map[ind] > 0);
453 uses_map[ind]--;
454 if (uses_map[ind] == 0 && !operand.info().isVariable() && !model_io.contains(ind))
455 {
456 dealloc_list_map[op_ind].emplace_back(tensor_regs.getITensor(ind));
457 }
458 }
459 }
460
461 // Dispose and validate
462 for (const auto &ind : constants)
463 {
464 --uses_map[ind];
465 }
466
467 assert(
468 std::all_of(uses_map.begin(), uses_map.end(),
469 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
470 }
471
472 // Generate kernels
473 for (auto &&pair : ordered_contexts)
474 {
475 auto codes = pair.second->genKernels();
476 for (auto &&[op_ind, fn_seq] : codes)
477 {
478 auto &op = lowered_graph->graph().operations().at(op_ind);
479 const auto backend = lowered_graph->lower_info().operation.at(op_ind);
480 if (options->he_profiling_mode)
481 fn_seq->wrap<SyncFunction>(backend->config());
482 if (!dealloc_list_map[op_ind].empty())
483 fn_seq->append(std::make_unique<DeallocFunction>(dealloc_list_map[op_ind]));
484 builder.append(op_ind, {op_ind, &op, backend, std::move(fn_seq)});
485 }
486 }
487
488 auto code_map = builder.releaseCodeMap();
489
490 auto exec = new exec::LinearExecutor{std::move(lowered_graph),
491 std::move(backend_contexts),
492 tensor_regs,
493 std::move(code_map),
494 order,
495 tracing_ctx};
496
497 if (!options->workspace_dir.empty())
498 {
499 exec->addObserver(
500 std::make_unique<exec::TracingObserver>(options->workspace_dir, exec->graph(), tracing_ctx));
501 exec->addObserver(std::make_unique<exec::MinMaxRecorder>(options->workspace_dir, exec->graph(),
502 exec->getBackendContexts()));
503 }
504
505 return exec;
506}

508exec::IExecutor *
509ExecutorFactory::createDataflowExecutor(std::unique_ptr<compiler::LoweredGraph> lowered_graph,
510 const std::shared_ptr<exec::IExecutors> &executors,
511 const ExecutorFactoryArgs &args, bool parallel)
512{
513 const auto options = args.options;
514 const auto &model_index = args.model_index;
515 const auto tracing_ctx = args.tracing_ctx;
516 auto custom_kernel_builder = args.custom_kernel_builder;
517
518 backend::BackendContexts backend_contexts =
519 createBackendContexts(*lowered_graph, options->executor == "Linear", custom_kernel_builder);
520
521 TensorRegistries tensor_regs{backend_contexts, true};
522
523 initializeSubgraphIOTensors(
524 *lowered_graph, backend_contexts,
525 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
526 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
527
528 for (auto &&pair : backend_contexts)
529 {
530 pair.second->genTensors();
531 }
532
533 prepareMigrantTensors(*lowered_graph, backend_contexts);
534
535 // Give some runtime objects to builtin KernelGenerator
536 prepareBuiltinBackend(tensor_regs, executors, backend_contexts, model_index);
537
538 ExecutionBuilder builder;
539
540 // Adjust the order of backends for the upcoming iteration
541 auto ordered_contexts = orderBackendContext(backend_contexts);
542
543 // Generate kernels
544 for (auto &&pair : ordered_contexts)
545 {
546 auto codes = pair.second->genKernels();
547 for (auto &&[op_ind, fn_seq] : codes)
548 {
549 auto &op = lowered_graph->graph().operations().at(op_ind);
550 const auto backend = lowered_graph->lower_info().operation.at(op_ind);
551 if (options->he_profiling_mode)
552 fn_seq->wrap<SyncFunction>(backend->config());
553 builder.append(op_ind, {op_ind, &op, backend, std::move(fn_seq)});
554 }
555 }
556
557 auto code_map = builder.releaseCodeMap();
558
559 exec::ExecutorBase *exec = nullptr;
560 if (parallel)
561 {
562 exec = new exec::ParallelExecutor{std::move(lowered_graph), std::move(backend_contexts),
563 tensor_regs, std::move(code_map), tracing_ctx};
564 }
565 else
566 {
567 auto dataflow_exec =
568 new exec::DataflowExecutor{std::move(lowered_graph), std::move(backend_contexts), tensor_regs,
569 std::move(code_map), tracing_ctx};
570 if (options->he_profiling_mode)
571 {
572 std::vector<const backend::Backend *> backends;
573 for (const auto &pair : backend_contexts)
574 {
575 backends.push_back(pair.first);
576 }
577 auto et = std::make_shared<exec::ExecTime>(backends);
578 std::unique_ptr<exec::IExecutionObserver> obs =
579 std::make_unique<exec::ProfileObserver>(et, dataflow_exec->graph());
580 dataflow_exec->addObserver(std::move(obs));
581 }
582 exec = dataflow_exec;
583 }
584
585 if (!options->workspace_dir.empty())
586 {
587 exec->addObserver(
588 std::make_unique<exec::TracingObserver>(options->workspace_dir, exec->graph(), tracing_ctx));
589 }
590
591 return exec;
592}

636exec::IExecutor *ExecutorFactory::createTrainableExecutor(
637 std::unique_ptr<compiler::train::LoweredTrainableGraph> lowered_graph,
638 const std::shared_ptr<exec::IExecutors> &, const ExecutorFactoryArgs &args,
639 const ir::train::TrainingInfo &training_info)
640{
641 const auto options = args.options;
642 const auto tracing_ctx = args.tracing_ctx;
643 auto custom_kernel_builder = args.custom_kernel_builder;
644
645 auto &graph = lowered_graph->graph();
646
647 lowered_graph->trainable_graph().operations().iterate([](const onert::ir::OperationIndex &,
648 const onert::ir::IOperation &op) {
649 try
650 {
651 [[maybe_unused]] const auto &casted_op =
652 dynamic_cast<const ir::train::ITrainableOperation &>(op);
653 }
654 catch (std::bad_cast &)
655 {
656 throw std::runtime_error("ExecutorFactory: " + op.name() + " is not trainable operation yet");
657 }
658 });
659
660 // TODO Create context only once instead of replacing
661 backend::train::TrainableBackendContexts tbackend_contexts;
662 backend::BackendContexts base_backend_contexts =
663 createBackendContexts(*lowered_graph, true, custom_kernel_builder);
664
665 // Replace BackendContext with TrainbleBackendContext
666 for (auto &&pair : base_backend_contexts)
667 {
668 auto ctx = pair.second.get();
669 const auto &data = ctx->data();
670
671 // Create partial and trainable graphs
672 auto tgraph = std::make_unique<ir::train::TrainableGraph>(*data.graph);
673 data.graph->operations().iterate(
674 [&](const onert::ir::OperationIndex &op_index, const onert::ir::IOperation &) {
675 const auto &orig_tgraph = lowered_graph->trainable_graph();
676 const auto &trainable_op = orig_tgraph.operation(op_index);
677 [[maybe_unused]] auto gen_index = tgraph->replaceOperation(op_index, trainable_op.clone());
678 assert(gen_index == op_index);
679 });
680 data.graph->operands().iterate([&](const ir::OperandIndex &index, const ir::Operand &) {
681 const auto &orig_tgraph = lowered_graph->trainable_graph();
682 if (orig_tgraph.backward_operands().exist(index))
683 {
684 const auto &bwd_operand = orig_tgraph.backward_operands().at(index);
685 auto new_bwd_operand = std::make_unique<ir::Operand>(bwd_operand);
686 [[maybe_unused]] auto gen_index =
687 tgraph->addBackwardOperand(index, std::move(new_bwd_operand));
688 assert(gen_index == index);
689 }
690 });
691
692 // Remove outputs of whole graph from external_operands
693 auto external_operands = data.external_operands;
694 for (const auto &index : lowered_graph->trainable_graph().getOutputs())
695 {
696 if (external_operands.contains(index))
697 external_operands.remove(index);
698 }
699
700 const auto backend = pair.first;
701 // NOTE The builtin backend does not yet support initializing UseDefs for training
702 // because it's graph does not have loss operation
703 // Without loss opeartion, we cannot call btopolSortOperations() or
704 // getEssentialBackwardOrder()
705 // TODO Modify checking the condition to check whether loss op exists
706 if (backend->config()->id() != "builtin")
707 {
708 // Initialize training def-uses
709 tgraph->updateGraphDependency();
710
711 tgraph->verify();
712 }
713
714 // Set trainable context data
715 backend::train::TrainableContextData tdata;
716 tdata.tgraph = std::move(tgraph);
717 tdata.op_order = std::move(data.op_order);
718 tdata.external_operands = std::move(external_operands);
719 tdata.custom_kernel_builder = std::move(data.custom_kernel_builder);
720 tdata.is_linear_executor = data.is_linear_executor;
721 tdata.optim_info = training_info.optimizerInfo();
722
723 // TODO Remove dynamic_cast
724 const auto tbackend = dynamic_cast<const backend::train::ITrainableBackend *>(backend);
725 if (!tbackend)
726 {
727 throw std::runtime_error("ExecutorFactory: Invalid backend - TrainableExecutor does not "
728 "support non-trainble backends");
729 }
730 tbackend_contexts.emplace(backend, tbackend->newContext(std::move(tdata)));
731 }
732 base_backend_contexts.clear();
733
734 train::TensorRegistries tensor_regs{tbackend_contexts, true};
735
736 initializeSubgraphIOTensors(
737 *lowered_graph, tbackend_contexts,
738 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
739 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
740
741 // linearize for forwarding
742 auto order = Linear::linearize(*lowered_graph);
743 VERBOSE(ExecutorFactory) << "Linearize for forwarding order" << std::endl;
744 Linear::dump(*lowered_graph, order);
745
746 // linearize for backwarding
747 auto backward_order = lowered_graph->trainable_graph().essentialBackwardOrder();
748 VERBOSE(ExecutorFactory) << "Linearize for backwarding order" << std::endl;
749 Linear::dump(*lowered_graph, backward_order);
750
751 train::TrainableCodeMap code_map;
752 // Generate tensors and kernels
753 for (auto &&[backend, context] : tbackend_contexts)
754 {
755 // builtin backend's kernel generator requires access to tensors in other backends.
756 // So, the other backends must be generated first.
757 if (backend->config()->id() == "builtin")
758 continue;
759
760 auto fn_map = context->gen();
761 generateCodes(fn_map, lowered_graph.get(), code_map);
762 }
763
764 prepareMigrantTensors(*lowered_graph, tbackend_contexts);
765
766 // Give some runtime objects to builtin KernelGenerator
767 for (auto &&pair : tbackend_contexts)
768 {
769 auto builtin_context =
770 dynamic_cast<backend::builtin::train::BackendContext *>(pair.second.get());
771 if (builtin_context != nullptr)
772 {
773 auto builtin_kernel_gen = builtin_context->kernel_gen;
774 builtin_kernel_gen->setTensorRegistries(tensor_regs);
775 builtin_kernel_gen->setWholeGraphOutputs(lowered_graph->trainable_graph().getOutputs());
776 }
777 }
778
779 // Generate tensors and kernels for only builtin backend
780 for (auto &&[backend, context] : tbackend_contexts)
781 {
782 if (backend->config()->id() == "builtin")
783 {
784 auto fn_map = context->gen();
785 generateCodes(fn_map, lowered_graph.get(), code_map);
786 }
787 }
788
789 // Adjust the order of backends for the upcoming iteration
790 auto ordered_contexts =
791 onert::orderBackendContext<backend::train::TrainableBackendContext>(tbackend_contexts);
792
793 // TODO Remove this simulation
794 // Simulate the execution for deallocation of tensors
795 std::unordered_map<ir::OperationIndex, DeallocList> dealloc_list_map;
796 {
797 ir::OperandIndexMap<uint32_t> uses_map;
798 ir::OperandIndexSequence constants;
799
800 auto model_io =
801 (graph.getInputs() + graph.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
802
803 // Prepare scanning
804 graph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
805 uses_map[ind] = obj.getUses().size();
806
807 if (obj.isConstant())
808 constants.append(ind);
809 });
810
811 // A trick to consider constants as an execption
812 for (const auto &ind : constants)
813 {
814 uses_map[ind]++;
815 }
816
817 for (const auto &op_ind : order)
818 {
819 const auto &op = graph.operations().at(op_ind);
820 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
821 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
822
823 for (const auto &ind : op_inputs)
824 {
825 const auto &operand = graph.operands().at(ind);
826 assert(uses_map.find(ind) != uses_map.end());
827 assert(uses_map[ind] > 0);
828 uses_map[ind]--;
829 if (uses_map[ind] == 0 && !operand.info().isVariable() && !model_io.contains(ind))
830 {
831 dealloc_list_map[op_ind].emplace_back(tensor_regs.getITensor(ind));
832 }
833 }
834 }
835
836 // Dispose and validate
837 for (const auto &ind : constants)
838 {
839 --uses_map[ind];
840 }
841
842 assert(
843 std::all_of(uses_map.begin(), uses_map.end(),
844 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
845 }
846
847 // Check back propagation tensors
848 {
849 // TODO Support multiple subgraphs
850 // Check if the back propagation tensors corresponding to inputs of model are nullptr
851 // NOTE The back propagation tensors corresponding to inputs of model are for inputs of
852 // PermuteLayers
853 // and they are nullptr and because they are meaningless.
854 assert(std::all_of(
855 lowered_graph->trainable_graph().getInputs().begin(),
856 lowered_graph->trainable_graph().getInputs().end(),
857 [&](const auto &input_idx) { return tensor_regs.getBackPropITensor(input_idx) == nullptr; }));
858
859 // Check if the back propagation tensors corresponding to outputs of model exist
860 assert(std::all_of(lowered_graph->trainable_graph().getOutputs().begin(),
861 lowered_graph->trainable_graph().getOutputs().end(),
862 [&](const auto &output_idx) {
863 return tensor_regs.getBackPropITensor(output_idx) == nullptr;
864 }));
865 }
866
867 if (order.size() != code_map.size())
868 {
869 throw std::runtime_error("ExecutorFactory: Some kernels are not generated");
870 }
871
872 auto exec = new exec::train::TrainableExecutor{std::move(lowered_graph),
873 std::move(tbackend_contexts),
874 tensor_regs,
875 std::move(code_map),
876 order,
877 backward_order,
878 tracing_ctx,
879 training_info.lossInfo()};
880
881 if (!options->workspace_dir.empty())
882 {
883 exec->addObserver(
884 std::make_unique<exec::TracingObserver>(options->workspace_dir, exec->graph(), tracing_ctx));
885 }
886 // TODO Support MINMAX_DUMPER
887
888 return exec;
889}
