51class SyncFunction final : public exec::IFunction
52{
53public:
54 virtual ~SyncFunction() = default;
55 SyncFunction(std::unique_ptr<exec::IFunction> fn, const std::shared_ptr<backend::IConfig> config)
56 : _fn{std::move(fn)}, _config{config}
57 {
58 assert(_fn);
59 assert(_config);
60 }
61
62 void run() override
63 {
64 _fn->run();
65 _config->sync();
66 }
67
68 void prepare() override { _fn->prepare(); }
69
70private:
71 std::unique_ptr<exec::IFunction> _fn;
72 std::shared_ptr<backend::IConfig> _config;
73};

75using DeallocList = std::vector<backend::ITensor *>;
76// Deallocation after execution of an operation used by Linear Executor
77class DeallocFunction final : public exec::IFunction
78{
79public:
80 DeallocFunction(const DeallocList &tensors) : _dealloc_list{tensors} {}
81
82 void run() override
83 {
84 for (auto &&tensor : _dealloc_list)
85 {
86 if (!tensor->is_dynamic())
87 continue;
88 tensor->deallocBuffer();
89 }
90 }
91
92private:
93 DeallocList _dealloc_list;
94};

158backend::BackendContexts
159createBackendContexts(compiler::ILoweredGraph &lgraph, bool linear_executor,
160 std::shared_ptr<backend::custom::IKernelBuilder> custom_kernel_builder)
161{
162 backend::BackendContexts contexts;
163 auto &backend_manager = compiler::BackendManager::get();
164
165 std::unordered_map<const backend::Backend *, backend::ContextData> context_data_map;
166
167 // Generate partial graphs for each backend
168 for (auto &&backend : backend_manager.getAll())
169 {
170 auto &data = context_data_map[backend];
171 auto graph = std::make_unique<ir::Graph>();
172 graph->setLayout(lgraph.graph().layout());
173 data.graph = std::move(graph);
174 }
175
176 auto &whole_graph = lgraph.graph();
177 // Separate operands into partial graphs
178 whole_graph.operands().iterate([&](const ir::OperandIndex &operand_ind, ir::Operand &operand) {
179 auto &operand_li = lgraph.lower_info().operand;
180 const auto &def_factors = operand_li.at(operand_ind).def_factors();
181 if (def_factors.size() == 0) // Ignore unused tensor
182 return;
183 const auto &def_factor = def_factors.getOnlyElement();
184 const auto backend = def_factor.backend();
185 auto &partial_graph = *context_data_map[backend].graph;
186 auto &operand_layouts = context_data_map[backend].operand_layouts;
187 assert(operand_layouts.find(operand_ind) == operand_layouts.end());
188 operand_layouts[operand_ind] = def_factor.layout();
189
190 // Copy the operand and insert it to the partial graph
191 auto new_operand = std::make_unique<ir::Operand>(operand);
192 new_operand->clearDefUse();
193 operand.releaseData(); // Deref data of LoweredGraph
194 auto new_operand_ind = partial_graph.addOperand(operand_ind, std::move(new_operand));
195 UNUSED_RELEASE(new_operand_ind);
196 assert(new_operand_ind == operand_ind);
197 });
198 // Separate operations into partial graphs
199 whole_graph.operations().iterate(
200 [&](const ir::OperationIndex &op_ind, const ir::IOperation &operation) {
201 auto &op_li = lgraph.lower_info().operation;
202 auto backend = op_li.at(op_ind).backend();
203 auto &partial_graph = *context_data_map[backend].graph;
204 auto &external_operands = context_data_map[backend].external_operands;
205 auto &operand_layouts = context_data_map[backend].operand_layouts;
206
207 {
208 // Add missing operands (externals)
209 auto io_list = (operation.getInputs() + operation.getOutputs()) | ir::Remove::DUPLICATED |
210 ir::Remove::UNDEFINED;
211 for (auto &&operand_ind : io_list)
212 {
213 if (partial_graph.operands().exist(operand_ind))
214 continue;
215
216 // Copy the operand and insert it to the partial graph
217 const auto &operand = whole_graph.operands().at(operand_ind);
218 auto new_operand = std::make_unique<ir::Operand>(operand);
219 new_operand->clearDefUse();
220 auto new_operand_ind = partial_graph.addOperand(operand_ind, std::move(new_operand));
221 UNUSED_RELEASE(new_operand_ind);
222 assert(new_operand_ind == operand_ind);
223
224 auto layout =
225 lgraph.lower_info().operand.at(operand_ind).def_factors().getOnlyElement().layout();
226 assert(operand_layouts.find(operand_ind) == operand_layouts.end());
227 operand_layouts[operand_ind] = layout;
228 external_operands.add(operand_ind);
229 }
230
231 auto new_op_ind = partial_graph.addOperation(op_ind, clone(operation));
232 UNUSED_RELEASE(new_op_ind);
233 assert(new_op_ind == op_ind);
234 }
235 });
236
237 // Create contexts
238 auto whole_op_order = lgraph.graph().topolSortOperations();
239 for (auto &&pair : context_data_map)
240 {
241 auto backend = pair.first;
242 auto &data = pair.second;
243 // Handle graph input/outputs or external tensors
244 data.graph->operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &operand) {
245 if (whole_graph.getInputs().contains(ind) || whole_graph.getOutputs().contains(ind))
246 data.external_operands.add(ind);
247 // Inputs are either "graph input" or "no def op and non-constant"
248 if (whole_graph.getInputs().contains(ind) ||
249 (!operand.getDef().valid() && !operand.isConstant()))
250 // Outputs are either "graph output" or "no uses"
251 data.graph->addInput(ind);
252 if (whole_graph.getOutputs().contains(ind) || operand.getUses().size() == 0)
253 data.graph->addOutput(ind);
254 });
255 VERBOSE(ExecutorFactory) << "createBackendContexts: partial graph for backend="
256 << backend->config()->id() << std::endl;
257 dumper::text::dumpGraph(*data.graph);
258
259 std::copy_if(whole_op_order.begin(), whole_op_order.end(), std::back_inserter(data.op_order),
260 [&](const auto &ind) { return data.graph->operations().exist(ind); });
261 data.is_linear_executor = linear_executor;
262 data.custom_kernel_builder = custom_kernel_builder;
263 contexts.emplace(backend, backend->newContext(std::move(data)));
264 }
265 return contexts;
266}

268template <typename Context>
269std::deque<std::pair<const backend::Backend *, Context *>> orderBackendContext(
270 const std::unordered_map<const backend::Backend *, std::unique_ptr<Context>> &tbackend_contexts)
271{
272 std::deque<std::pair<const backend::Backend *, Context *>> ordered_contexts;
273
274 for (auto &&pair : tbackend_contexts)
275 {
276 // NOTE builtin backend must be processed lastly.
277 // This is because of Permute layer's specialty which is the only operation that could have
278 // different ITensor objects for the input and the output. And it requires all other backends'
279 // tensors are ready to use.
280 if (pair.first->config()->id() == "builtin")
281 ordered_contexts.emplace_back(pair.first, pair.second.get());
282 else
283 ordered_contexts.emplace_front(pair.first, pair.second.get());
284 }
285
286 return ordered_contexts;
287}

320void ExecutorFactory::prepareMigrantTensors(compiler::ILoweredGraph &lowered_graph,
321 const backend::BackendContexts &backend_contexts)
322{
323 TensorRegistries tensor_regs{backend_contexts, true};
324
325 lowered_graph.graph().operations().iterate(
326 [&](const ir::OperationIndex &op_ind, const ir::IOperation &op) {
327 auto lower_info = lowered_graph.lower_info().operation.getRawPtr(op_ind);
328 auto &backend_ctx = backend_contexts.at(lower_info->backend());
329 for (auto &&ind :
330 (op.getInputs() + op.getOutputs()) | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED)
331 {
332 // If an Operation's input/output tensor does not have an own tensor object,
333 // it must be using migrant tensors, so find the tensor from other tensor registries and
334 // register it to the current tensor registry if it is portable
335 if (!backend_ctx->tensor_registry->getITensor(ind))
336 {
337 auto tensor = tensor_regs.getITensor(ind);
338 assert(tensor); // The tensor must have been registered
339 auto ptensor = dynamic_cast<backend::IPortableTensor *>(tensor);
340 if (ptensor)
341 backend_ctx->tensor_registry->setMigrantTensor(ind, ptensor);
342 }
343 }
344 });
345}

347void ExecutorFactory::prepareBuiltinBackend(const TensorRegistries &tensor_regs,
348 const std::shared_ptr<exec::IExecutors> &executors,
349 const backend::BackendContexts &backend_contexts,
350 const ir::ModelIndex &index)
351{
352 for (auto &&pair : backend_contexts)
353 {
354 auto builtin_context = dynamic_cast<backend::builtin::BackendContext *>(pair.second.get());
355 if (builtin_context != nullptr)
356 {
357 auto builtin_kernel_gen = builtin_context->kernel_gen;
358 builtin_kernel_gen->setTensorRegistries(tensor_regs);
359 builtin_kernel_gen->setExecutors(executors);
360 builtin_kernel_gen->setModelIndex(index);
361 }
362 }

383exec::IExecutor *
384ExecutorFactory::createLinearExecutor(std::unique_ptr<compiler::LoweredGraph> lowered_graph,
385 const std::shared_ptr<exec::IExecutors> &executors,
386 const ExecutorFactoryArgs &args)
387{
388 const auto options = args.options;
389 const auto &model_index = args.model_index;
390 const auto tracing_ctx = args.tracing_ctx;
391 auto custom_kernel_builder = args.custom_kernel_builder;
392 auto &graph = lowered_graph->graph();
393
394 backend::BackendContexts backend_contexts =
395 createBackendContexts(*lowered_graph, options->executor == "Linear", custom_kernel_builder);
396
397 TensorRegistries tensor_regs{backend_contexts, true};
398
399 initializeSubgraphIOTensors(
400 *lowered_graph, backend_contexts,
401 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
402 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
403
404 // linearize
405 auto order = Linear::linearize(*lowered_graph);
406 Linear::dump(*lowered_graph, order);
407
408 for (auto &&pair : backend_contexts)
409 {
410 pair.second->genTensors();
411 }
412
413 prepareMigrantTensors(*lowered_graph, backend_contexts);
414
415 // Give some runtime objects to builtin KernelGenerator
416 prepareBuiltinBackend(tensor_regs, executors, backend_contexts, model_index);
417
418 ExecutionBuilder builder;
419
420 // Adjust the order of backends for the upcoming iteration
421 auto ordered_contexts = orderBackendContext(backend_contexts);
422
423 // Simulate the execution for deallocation of tensors
424 std::unordered_map<ir::OperationIndex, DeallocList> dealloc_list_map;
425 {
426 ir::OperandIndexMap<uint32_t> uses_map;
427 ir::OperandIndexSequence constants;
428
429 auto model_io =
430 (graph.getInputs() + graph.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
431
432 // Prepare scanning
433 graph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
434 uses_map[ind] = obj.getUses().size();
435
436 if (obj.isConstant())
437 constants.append(ind);
438 });
439
440 // A trick to consider constants as an execption
441 for (const auto &ind : constants)
442 {
443 uses_map[ind]++;
444 }
445
446 for (const auto &op_ind : order)
447 {
448 const auto &op = graph.operations().at(op_ind);
449 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
450 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
451
452 for (const auto &ind : op_inputs)
453 {
454 const auto &operand = graph.operands().at(ind);
455 assert(uses_map.find(ind) != uses_map.end());
456 assert(uses_map[ind] > 0);
457 uses_map[ind]--;
458 if (uses_map[ind] == 0 && !operand.info().isVariable() && !model_io.contains(ind))
459 {
460 dealloc_list_map[op_ind].emplace_back(tensor_regs.getITensor(ind));
461 }
462 }
463 }
464
465 // Dispose and validate
466 for (const auto &ind : constants)
467 {
468 --uses_map[ind];
469 }
470
471 assert(
472 std::all_of(uses_map.begin(), uses_map.end(),
473 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
474 }
475
476 // Generate kernels
477 for (auto &&pair : ordered_contexts)
478 {
479 auto codes = pair.second->genKernels();
480 for (auto &&pair : codes)
481 {
482 auto &op_ind = pair.first;
483 auto &fn_seq = pair.second;
484 auto &op = lowered_graph->graph().operations().at(op_ind);
485 auto lower_info = lowered_graph->lower_info().operation.getRawPtr(op_ind);
486 if (options->he_profiling_mode)
487 fn_seq->wrap<SyncFunction>(lower_info->backend()->config());
488 if (!dealloc_list_map[op_ind].empty())
489 fn_seq->append(std::make_unique<DeallocFunction>(dealloc_list_map[op_ind]));
490 builder.append(op_ind, {op_ind, &op, lower_info, std::move(fn_seq)});
491 }
492 }
493
494 auto code_map = builder.releaseCodeMap();
495
496 auto exec = new exec::LinearExecutor{std::move(lowered_graph),
497 std::move(backend_contexts),
498 tensor_regs,
499 std::move(code_map),
500 order,
501 tracing_ctx};
502
503 if (!options->trace_filepath.empty())
504 {
505 std::unique_ptr<exec::IExecutionObserver> ctp =
506 std::make_unique<exec::TracingObserver>(options->trace_filepath, exec->graph(), tracing_ctx);
507 exec->addObserver(std::move(ctp));
508 }
509#ifdef MINMAX_H5DUMPER
510 if (!options->minmax_filepath.empty())
511 exec->addObserver(std::make_unique<exec::MinMaxRecorder>(
512 options->minmax_filepath, exec->graph(), exec->getBackendContexts()));
513#endif
514
515 return exec;
516}

518exec::IExecutor *
519ExecutorFactory::createDataflowExecutor(std::unique_ptr<compiler::LoweredGraph> lowered_graph,
520 const std::shared_ptr<exec::IExecutors> &executors,
521 const ExecutorFactoryArgs &args, bool parallel)
522{
523 const auto options = args.options;
524 const auto &model_index = args.model_index;
525 const auto tracing_ctx = args.tracing_ctx;
526 auto custom_kernel_builder = args.custom_kernel_builder;
527
528 backend::BackendContexts backend_contexts =
529 createBackendContexts(*lowered_graph, options->executor == "Linear", custom_kernel_builder);
530
531 TensorRegistries tensor_regs{backend_contexts, true};
532
533 initializeSubgraphIOTensors(
534 *lowered_graph, backend_contexts,
535 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
536 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
537
538 for (auto &&pair : backend_contexts)
539 {
540 pair.second->genTensors();
541 }
542
543 prepareMigrantTensors(*lowered_graph, backend_contexts);
544
545 // Give some runtime objects to builtin KernelGenerator
546 prepareBuiltinBackend(tensor_regs, executors, backend_contexts, model_index);
547
548 ExecutionBuilder builder;
549
550 // Adjust the order of backends for the upcoming iteration
551 auto ordered_contexts = orderBackendContext(backend_contexts);
552
553 // Generate kernels
554 for (auto &&pair : ordered_contexts)
555 {
556 auto codes = pair.second->genKernels();
557 for (auto &&pair : codes)
558 {
559 auto &op_ind = pair.first;
560 auto &fn_seq = pair.second;
561 auto &op = lowered_graph->graph().operations().at(op_ind);
562 auto lower_info = lowered_graph->lower_info().operation.getRawPtr(op_ind);
563 if (options->he_profiling_mode)
564 fn_seq->wrap<SyncFunction>(lower_info->backend()->config());
565 builder.append(op_ind, {op_ind, &op, lower_info, std::move(fn_seq)});
566 }
567 }
568
569 auto code_map = builder.releaseCodeMap();
570
571 exec::ExecutorBase *exec = nullptr;
572 if (parallel)
573 {
574 exec = new exec::ParallelExecutor{std::move(lowered_graph), std::move(backend_contexts),
575 tensor_regs, std::move(code_map), tracing_ctx};
576 }
577 else
578 {
579 auto dataflow_exec =
580 new exec::DataflowExecutor{std::move(lowered_graph), std::move(backend_contexts), tensor_regs,
581 std::move(code_map), tracing_ctx};
582 if (options->he_profiling_mode)
583 {
584 std::vector<const backend::Backend *> backends;
585 for (const auto &pair : backend_contexts)
586 {
587 backends.push_back(pair.first);
588 }
589 auto et = std::make_shared<exec::ExecTime>(backends);
590 std::unique_ptr<exec::IExecutionObserver> obs =
591 std::make_unique<exec::ProfileObserver>(et, dataflow_exec->graph());
592 dataflow_exec->addObserver(std::move(obs));
593 }
594 exec = dataflow_exec;
595 }
596
597 if (!options->trace_filepath.empty())
598 {
599 std::unique_ptr<exec::IExecutionObserver> ctp =
600 std::make_unique<exec::TracingObserver>(options->trace_filepath, exec->graph(), tracing_ctx);
601 exec->addObserver(std::move(ctp));
602 }
603
604 return exec;
605}

656 auto custom_kernel_builder = args.custom_kernel_builder;
657
658 auto &graph = lowered_graph->graph();
659
660 lowered_graph->trainable_graph().operations().iterate([](const onert::ir::OperationIndex &,
661 const onert::ir::IOperation &op) {
662 try
663 {
664 UNUSED_RELEASE(dynamic_cast<const ir::train::ITrainableOperation &>(op));
665 }
666 catch (std::bad_cast &)
667 {
668 throw std::runtime_error("ExecutorFactory: " + op.name() + " is not trainable operation yet");
669 }
670 });
671
672 // TODO Create context only once instead of replacing
673 backend::train::TrainableBackendContexts tbackend_contexts;
674 backend::BackendContexts base_backend_contexts =
675 createBackendContexts(*lowered_graph, true, custom_kernel_builder);

677 // Replace BackendContext with TrainbleBackendContext
678 for (auto &&pair : base_backend_contexts)
679 {
680 auto ctx = pair.second.get();
681 const auto &data = ctx->data();
682
683 // Create partial and trainable graphs
684 auto tgraph = std::make_unique<ir::train::TrainableGraph>(*data.graph);
685 data.graph->operations().iterate(
686 [&](const onert::ir::OperationIndex &op_index, const onert::ir::IOperation &) {
687 const auto &orig_tgraph = lowered_graph->trainable_graph();
688 const auto &trainable_op = orig_tgraph.operation(op_index);
689 auto gen_index = tgraph->replaceOperation(op_index, trainable_op.clone());
690 UNUSED_RELEASE(gen_index);
691 assert(gen_index == op_index);
692 });
693 data.graph->operands().iterate([&](const ir::OperandIndex &index, const ir::Operand &) {
694 const auto &orig_tgraph = lowered_graph->trainable_graph();
695 if (orig_tgraph.backward_operands().exist(index))
696 {
697 const auto &bwd_operand = orig_tgraph.backward_operands().at(index);
698 auto new_bwd_operand = std::make_unique<ir::Operand>(bwd_operand);
699 auto gen_index = tgraph->addBackwardOperand(index, std::move(new_bwd_operand));
700 UNUSED_RELEASE(gen_index);
701 assert(gen_index == index);
702 }
703 });
704
705 // Remove outputs of whole graph from external_operands
706 auto external_operands = data.external_operands;
707 for (const auto &index : lowered_graph->trainable_graph().getOutputs())
708 {
709 if (external_operands.contains(index))
710 external_operands.remove(index);
711 }
712
713 // Set trainable context data
714 backend::train::TrainableContextData tdata;
715 tdata.tgraph = std::move(tgraph);
716 tdata.op_order = std::move(data.op_order);
717 tdata.external_operands = std::move(external_operands);
718 tdata.operand_layouts = std::move(data.operand_layouts);
719 tdata.custom_kernel_builder = std::move(data.custom_kernel_builder);
720 tdata.is_linear_executor = data.is_linear_executor;
721 tdata.optim_info = training_info.optimizerInfo();
722
723 // TODO Remove dynamic_cast
724 try
725 {
726 const auto backend = pair.first;
727 const auto tbackend = dynamic_cast<const backend::train::ITrainableBackend *>(backend);
728 tbackend_contexts.emplace(backend, tbackend->newContext(std::move(tdata)));
729 }
730 catch (const std::bad_cast &)
731 {
732 throw std::runtime_error("ExecutorFactory: Invalid backend - TrainableExecutor does not "
733 "support non-trainble backends");
734 }
735 }
736 base_backend_contexts.clear();
737
738 train::TensorRegistries tensor_regs{tbackend_contexts, true};
739
740 initializeSubgraphIOTensors(
741 *lowered_graph, tbackend_contexts,
742 (lowered_graph->graph().getInputs() + lowered_graph->graph().getOutputs()) |
743 ir::Remove::DUPLICATED | ir::Remove::UNDEFINED);
744
745 // linearize for forwarding
746 auto order = Linear::linearize(*lowered_graph);
747 VERBOSE(ExecutorFactory) << "Linearize for forwarding order" << std::endl;
748 Linear::dump(*lowered_graph, order);
749
750 // linearize for backwarding
751 auto backward_order = lowered_graph->trainable_graph().btopolSortOperations();
752 VERBOSE(ExecutorFactory) << "Linearize for backwarding order" << std::endl;
753 Linear::dump(*lowered_graph, backward_order);

755 for (auto &&pair : tbackend_contexts)
756 {
757 pair.second->genTensors();
758 }
759
760 for (auto &&pair : tbackend_contexts)
761 {
762 auto tctx = pair.second.get();
763 tctx->genTrainingTensors();
764 }
765
766 prepareMigrantTensors(*lowered_graph, tbackend_contexts);
767
768 // Give some runtime objects to builtin KernelGenerator
769 for (auto &&pair : tbackend_contexts)
770 {
771 auto builtin_context =
772 dynamic_cast<backend::builtin::train::BackendContext *>(pair.second.get());
773 if (builtin_context != nullptr)
774 {
775 auto builtin_kernel_gen = builtin_context->kernel_gen;
776 builtin_kernel_gen->setTensorRegistries(tensor_regs);
777 builtin_kernel_gen->setWholeGraphOutputs(lowered_graph->trainable_graph().getOutputs());
778 }
779 }
780
781 // Adjust the order of backends for the upcoming iteration
782 auto ordered_contexts =
783 onert::orderBackendContext<backend::train::TrainableBackendContext>(tbackend_contexts);
784
785 // TODO Remove this simulation
786 // Simulate the execution for deallocation of tensors
787 std::unordered_map<ir::OperationIndex, DeallocList> dealloc_list_map;
788 {
789 ir::OperandIndexMap<uint32_t> uses_map;
790 ir::OperandIndexSequence constants;
791
792 auto model_io =
793 (graph.getInputs() + graph.getOutputs()) | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED;
794
795 // Prepare scanning
796 graph.operands().iterate([&](const ir::OperandIndex &ind, const ir::Operand &obj) {
797 uses_map[ind] = obj.getUses().size();
798
799 if (obj.isConstant())
800 constants.append(ind);
801 });
802
803 // A trick to consider constants as an execption
804 for (const auto &ind : constants)
805 {
806 uses_map[ind]++;
807 }
808
809 for (const auto &op_ind : order)
810 {
811 const auto &op = graph.operations().at(op_ind);
812 auto op_inputs = op.getInputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
813 auto op_outputs = op.getOutputs() | ir::Remove::DUPLICATED | ir::Remove::UNDEFINED;
814
815 for (const auto &ind : op_inputs)
816 {
817 const auto &operand = graph.operands().at(ind);
818 assert(uses_map.find(ind) != uses_map.end());
819 assert(uses_map[ind] > 0);
820 uses_map[ind]--;
821 if (uses_map[ind] == 0 && !operand.info().isVariable() && !model_io.contains(ind))
822 {
823 dealloc_list_map[op_ind].emplace_back(tensor_regs.getITensor(ind));
824 }
825 }
826 }
827
828 // Dispose and validate
829 for (const auto &ind : constants)
830 {
831 --uses_map[ind];
832 }
833
834 assert(
835 std::all_of(uses_map.begin(), uses_map.end(),
836 [](std::pair<const ir::OperandIndex, uint32_t> it) { return it.second == 0; }));
837 }
838
839 // Check back propagation tensors
840 {
841 // TODO Support multiple subgraphs
842 // Check if the back propagation tensors corresponding to inputs of model are nullptr
843 // NOTE The back propagation tensors corresponding to inputs of model are for inputs of
844 // PermuteLayers
845 // and they are nullptr and because they are meaningless.
846 assert(std::all_of(
847 lowered_graph->trainable_graph().getInputs().begin(),
848 lowered_graph->trainable_graph().getInputs().end(),
849 [&](const auto &input_idx) { return tensor_regs.getBackPropITensor(input_idx) == nullptr; }));
850
851 // Check if the back propagation tensors corresponding to outputs of model exist
852 assert(std::all_of(lowered_graph->trainable_graph().getOutputs().begin(),
853 lowered_graph->trainable_graph().getOutputs().end(),
854 [&](const auto &output_idx) {
855 return tensor_regs.getBackPropITensor(output_idx) == nullptr;
856 }));
857 }
858
859 train::TrainableCodeMap code_map;
860 // Generate kernels
861 for (auto &&pair : ordered_contexts)
862 {
863 auto codes = pair.second->genKernels();
864 for (auto &&pair : codes)
865 {
866 auto &op_ind = pair.first;
867 auto &tn_seq = pair.second;
868 auto &op = lowered_graph->trainable_graph().operation(op_ind);
869 auto lower_info = lowered_graph->lower_info().operation.getRawPtr(op_ind);
870
871 assert(code_map.find(op_ind) == code_map.end());
872 code_map.insert(
873 {op_ind, train::TrainableCodeAndInfo{op_ind, &op, lower_info, std::move(tn_seq)}});
874 }
875 }
876
877 if (order.size() != code_map.size())
878 {
879 throw std::runtime_error("ExecutorFactory: Some kernels are not generated");
880 }
881
882 auto exec = new exec::train::TrainableExecutor{std::move(lowered_graph),
883 std::move(tbackend_contexts),
884 tensor_regs,
885 std::move(code_map),
886 order,
887 backward_order,
888 tracing_ctx,
889 training_info.lossInfo()};
890
891 if (!options->trace_filepath.empty())
892 {
893 std::unique_ptr<exec::IExecutionObserver> ctp =
894 std::make_unique<exec::TracingObserver>(options->trace_filepath, exec->graph(), tracing_ctx);
895 exec->addObserver(std::move(ctp));
896 }
897 // TODO Support MINMAX_H5DUMPER
898
899 return exec;
900}
