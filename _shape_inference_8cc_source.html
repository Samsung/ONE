41template <typename T, typename U>
42typename std::enable_if<std::is_integral<T>::value && std::is_integral<U>::value,
43 typename std::common_type<T, U>::type>::type
44ceil_div(T dividend, U divisor)
45{
46 assert(dividend > 0 && divisor > 0 && "this implementations is for positive numbers only");
47 return (dividend + divisor - 1) / divisor;
48}

50// Calculate the result of broadcast of two shapes
51ir::Shape broadcastShapes(const ir::Shape &lhs_shape, const ir::Shape &rhs_shape)
52{
53 ir::Shape out_shape;
54 auto max_rank = std::max(lhs_shape.rank(), rhs_shape.rank());
55
56 for (int idx = 0; idx < max_rank; ++idx)
57 {
58 // Go over operands dimensions from right to left
59 int lhs_idx = lhs_shape.rank() - idx - 1;
60 int rhs_idx = rhs_shape.rank() - idx - 1;
61
62 int32_t lhs_dim = lhs_idx >= 0 ? lhs_shape.dim(lhs_idx) : 1;
63 int32_t rhs_dim = rhs_idx >= 0 ? rhs_shape.dim(rhs_idx) : 1;
64
65 if (lhs_dim != 1 && rhs_dim != 1 && lhs_dim != rhs_dim)
66 throw std::runtime_error("Incompatible shapes for broadcast");
67
68 out_shape.prepend(std::max(lhs_dim, rhs_dim));
69 }
70
71 return out_shape;
72}

79{
80 int size = 0;
81 for (int idx = 0; idx < cluster_shape.dim(0); idx++)
82 {
83 size += cluster_buf[idx * 2 + 1];
84 }
85 return size;

93// Calculate output height and width of convolution-like operation
94std::pair<int, int> calcConvLikeHeightAndWidth(const int in_h, const int in_w, const int ker_h,
95 const int ker_w, const ir::Padding pad,
96 const ir::Stride stride,
97 const ir::Dilation dilation = {1, 1})
98{
99 int32_t out_h = 0, out_w = 0;
100 int32_t effective_filter_w_size = (ker_w - 1) * dilation.width_factor + 1;
101 int32_t effective_filter_h_size = (ker_h - 1) * dilation.height_factor + 1;
102 switch (pad.type)
103 {
104 case ir::PaddingType::SAME:
105 out_h = ceil_div(in_h, stride.vertical);
106 out_w = ceil_div(in_w, stride.horizontal);
107 break;
108 case ir::PaddingType::VALID:
109 out_h = ceil_div(in_h - effective_filter_h_size + 1, stride.vertical);
110 out_w = ceil_div(in_w - effective_filter_w_size + 1, stride.horizontal);
111 break;
112 case ir::PaddingType::EXPLICIT:
113 out_h =
114 (in_h + pad.param.top + pad.param.bottom - effective_filter_h_size) / stride.vertical + 1;
115 out_w =
116 (in_w + pad.param.left + pad.param.right - effective_filter_w_size) / stride.horizontal + 1;
117 break;
118 default:
119 assert(false);
120 }
121
122 return {out_h, out_w};
123}

150ir::Shape inferReduceShape(const ir::Shape &input_shape, const std::vector<int> &axes,
151 bool keep_dims)
152{
153 int num_axis = axes.size();
154 int input_num_dims = input_shape.rank();
155 if (input_num_dims == 0)
156 {
157 ir::Shape out_shape(0);
158 return out_shape;
159 }
160 if (keep_dims)
161 {
162 ir::Shape out_shape;
163 for (int idx = 0; idx < input_num_dims; ++idx)
164 {
165 bool is_axis = false;
166 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx)
167 {
168 if (axes[axis_idx] == idx || axes[axis_idx] + input_num_dims == idx)
169 {
170 is_axis = true;
171 break;
172 }
173 }
174 if (is_axis)
175 {
176 out_shape.append(1);
177 }
178 else
179 {
180 out_shape.append(input_shape.dim(idx));
181 }
182 }
183 return out_shape;
184 }
185 else
186 {
187 // Calculates size of reducing axis.
188 int num_reduce_axis = num_axis;
189 for (int i = 0; i < num_axis; ++i)
190 {
191 int current = axes[i];
192 if (!(-input_num_dims <= current && current < input_num_dims))
193 throw std::runtime_error{"Invalid dim value " + std::to_string(current)};
194 if (current < 0)
195 {
196 current += input_num_dims;
197 }
198 for (int j = 0; j < i; ++j)
199 {
200 int previous = axes[j];
201 if (previous < 0)
202 {
203 previous += input_num_dims;
204 }
205 if (current == previous)
206 {
207 --num_reduce_axis;
208 break;
209 }
210 }
211 }
212 // Determines output dimensions.
213 ir::Shape out_shape;
214 int num_skip_axis = 0;
215 for (int idx = 0; idx < input_num_dims; ++idx)
216 {
217 bool is_axis = false;
218 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx)
219 {
220 if (axes[axis_idx] == idx || axes[axis_idx] + input_num_dims == idx)
221 {
222 ++num_skip_axis;
223 is_axis = true;
224 break;
225 }
226 }
227 if (!is_axis)
228 {
229 out_shape.append(input_shape.dim(idx));
230 }
231 }
232 return out_shape;
233 }
234}

236ir::Shape inferBatchMatMulShape(const ir::Shape &lhs_shape, const ir::Shape &rhs_shape,
237 const ir::operation::BatchMatMul::Param &param)
238{
239 bool adj_x = param.adj_x;
240 bool adj_y = param.adj_y;
241 ir::Shape output_shape;
242
243 int output_rank = std::max(lhs_shape.rank(), rhs_shape.rank());
244
245 // Extend lhs and rhs shape
246 ir::Shape extended_lhs_shape(lhs_shape);
247 ir::Shape extended_rhs_shape(rhs_shape);
248 extended_lhs_shape.extendRank(output_rank);
249 extended_rhs_shape.extendRank(output_rank);
250
251 for (int i = 0; i < output_rank - 2; i++)
252 {
253 const int lhs_dim = extended_lhs_shape.dim(i);
254 const int rhs_dim = extended_rhs_shape.dim(i);
255 int broadcast_dim = lhs_dim;
256 if (lhs_dim != rhs_dim)
257 {
258 if (lhs_dim == 1)
259 {
260 broadcast_dim = rhs_dim;
261 }
262 else if (rhs_dim != 1)
263 {
264 throw std::runtime_error{"BatchMatMul shape inference: invalid brodcasting input shape"};
265 }
266 }
267
268 output_shape.append(broadcast_dim);
269 }
270
271 // Fill in the matmul dimensions.
272 int lhs_rows_index = adj_x ? output_rank - 1 : output_rank - 2;
273 int rhs_cols_index = adj_y ? output_rank - 2 : output_rank - 1;
274
275 output_shape.append(extended_lhs_shape.dim(lhs_rows_index));
276 output_shape.append(extended_rhs_shape.dim(rhs_cols_index));
277
278 return output_shape;
279}

304ir::Shape inferConcatShape(const Shapes &in_shapes, const ir::operation::Concat::Param &param)
305{
306 const int32_t concat_axis = param.axis >= 0 ? param.axis : in_shapes[0].rank() + param.axis;
307 const auto &first_in_shape = in_shapes[0];
308
309 // Check that all shapes are equal except for concat axis dimension
310 for (const auto &in_shape : in_shapes)
311 {
312 if (in_shape.rank() != first_in_shape.rank())
313 throw std::runtime_error("Rank in all input tensors should be same");
314
315 for (int64_t dim_idx = 0; dim_idx < in_shape.rank(); ++dim_idx)
316 if (!(dim_idx == concat_axis || in_shape.dim(dim_idx) == first_in_shape.dim(dim_idx)))
317 throw std::runtime_error("All tensor should have same dimension "
318 "except dimension on passed axis");
319 }
320
321 // Calculate output shape
322 ir::Shape out_shape(first_in_shape);
323 out_shape.dim(concat_axis) = 0;
324 for (const auto &in_shape : in_shapes)
325 out_shape.dim(concat_axis) += in_shape.dim(concat_axis);
326 return out_shape;
327}

329ir::Shape inferConv2DShape(const ir::Shape &in_shape, const ir::Shape &ker_shape,
330 const ir::operation::Conv2D::Param &param, ir::Layout layout)
331{
332 if (param.stride.horizontal == 0 || param.stride.vertical == 0)
333 throw std::runtime_error{"Conv2D: stride values must be positive"};
334
335 auto ifm_shape = in_shape.asFeature(layout);
336
337 // Kernel format is [depth_out, kernel_height, kernel_width, depth_in]
338 auto kf_shape = ker_shape.asFeature(layout);
339 assert(ifm_shape.C == kf_shape.C);
340
341 const auto out_h_w = calcConvLikeHeightAndWidth(ifm_shape.H, ifm_shape.W, kf_shape.H, kf_shape.W,
342 param.padding, param.stride, param.dilation);
343
344 return ir::Shape{ifm_shape.N, out_h_w.first, out_h_w.second, kf_shape.N};
345}

347ir::Shape inferDepthwiseConv2DShape(const ir::Shape &in_shape, const ir::Shape &ker_shape,
348 const ir::operation::DepthwiseConv2D::Param &param,
349 ir::Layout layout)
350{
351 if (param.stride.horizontal == 0 || param.stride.vertical == 0)
352 throw std::runtime_error{"DepthwiseConv2D: stride values must be positive"};
353
354 assert(layout == ir::Layout::NHWC);
355 auto ifm_shape = in_shape.asFeature(layout);
356
357 // Kernel format is [1, kernel_height, kernel_width, depth_out]
358 auto kf_shape = ker_shape.asFeature(layout);
359 assert(kf_shape.C == static_cast<int32_t>(ifm_shape.C * param.multiplier));
360 assert(kf_shape.N == 1);
361
362 const auto out_h_w = calcConvLikeHeightAndWidth(ifm_shape.H, ifm_shape.W, kf_shape.H, kf_shape.W,
363 param.padding, param.stride, param.dilation);
364
365 return ir::Shape{ifm_shape.N, out_h_w.first, out_h_w.second, kf_shape.C};
366}

368ir::Shape inferExpandDimsShape(const ir::Shape &in_shape, int32_t axis)
369{
370 ir::Shape out_shape(in_shape.rank() + 1);
371
372 axis = ((axis >= 0) ? axis : /* when axis < 0 */ (out_shape.rank() + axis));
373 if (!(0 <= axis && axis <= in_shape.rank()))
374 throw std::runtime_error("axis of dim is out of range");
375
376 for (int x = 0, out_x = 0; out_x < out_shape.rank(); ++out_x)
377 {
378 if (out_x == axis)
379 out_shape.dim(out_x) = 1;
380 else
381 out_shape.dim(out_x) = in_shape.dim(x++);
382 }
383
384 return out_shape;
385}

403ir::Shape inferFullyConnectedShape(const ir::Shape &in_shape, const ir::Shape &ker_shape)
404{
405 assert(in_shape.rank() >= 2);
406 assert(ker_shape.rank() == 2);
407
408 const auto input_size_with_batch = in_shape.num_elements();
409 const auto num_units = ker_shape.dim(0);
410 const auto input_size = ker_shape.dim(1);
411 const auto batch_size = input_size_with_batch / input_size;
412 assert(input_size_with_batch % input_size == 0);
413
414 return {ir::Shape({static_cast<int32_t>(batch_size), num_units})};
415}

490{
491 assert(depth >= 0);
492 const auto rank = input_shape.rank() + 1;
493 ir::Shape newShape(rank);
494

496
497 for (int i = 0; i < rank; ++i)
498 {
499 if (i < axis)
500 {
501 newShape.dim(i) = input_shape.dim(i);
502 }

504 {
505 newShape.dim(i) = depth;
506 }
507 else
508 {
509 newShape.dim(i) = input_shape.dim(i - 1);
510 }
511 }
512
513 return newShape;
514}

536ir::Shape inferPadShape(const ir::Shape &in_shape, const int32_t *pad_buf, const size_t num_pads)
537{
538 assert(num_pads % 2 == 0);
539 const int32_t rank = num_pads / 2;
540

542 for (int32_t i = 0; i < rank; ++i)

545 const auto after_padding = pad_buf[i * 2 + 1];
546

548 }
549
550 return ret;

592template <typename T> ir::Shape inferRangeShape(T start_val, T limit_val, T delta_val)
593{
594 ir::Shape out_shape(static_cast<int>(1));
595
596 out_shape.dim(0) =
597 (std::is_integral<T>::value
598 ? ((std::abs(start_val - limit_val) + std::abs(delta_val) - 1) / std::abs(delta_val))
599 : std::ceil(std::abs((start_val - limit_val) / delta_val)));
600 return out_shape;
601}

607ir::Shape inferReshapeShape(const int32_t *shape_buf, const int32_t shape_num_elements,
608 const size_t total_num_elements)
609{
610 ir::Shape ret(shape_num_elements);
611 int32_t flatten_dim = ir::Shape::kUnspecifiedDim;
612 for (int32_t i = 0; i < shape_num_elements; ++i)
613 {
614 if (shape_buf[i] < 0)
615 {
616 if (flatten_dim != ir::Shape::kUnspecifiedDim)
617 throw std::runtime_error("Reshape: 2nd param has special dim(for flatten) more than twice");
618 flatten_dim = i;
619 ret.dim(i) = 1;
620 }
621 else
622 {
623 ret.dim(i) = shape_buf[i];
624 }
625 }
626 if (flatten_dim != ir::Shape::kUnspecifiedDim)
627 ret.dim(flatten_dim) = total_num_elements / ret.num_elements();
628
629 // Check reshapable
630 if (total_num_elements != static_cast<size_t>(ret.num_elements()))
631 throw std::runtime_error("Reshape: 2nd param is not compatible with the shape of input");
632
633 return ret;
634}

636ir::Shape inferSelectShape(const ir::Shape &input_cond_shape, const ir::Shape &input_true_shape,
637 const ir::Shape &input_false_shape)
638{
639 auto haveSameShapes = [](const ir::Shape &input_cond_shape, const ir::Shape &input_true_shape,
640 const ir::Shape &input_false_shape) {
641 if ((input_cond_shape.rank() != input_true_shape.rank()) ||
642 input_cond_shape.rank() != input_false_shape.rank())
643 {
644 return false;
645 }
646
647 int rank = input_cond_shape.rank();
648 for (int i = 0; i < rank; ++i)
649 {
650 if (input_cond_shape.dim(i) != input_true_shape.dim(i) ||
651 input_cond_shape.dim(i) != input_false_shape.dim(i))
652 {
653 return false;
654 }
655 }
656
657 return true;
658 };
659
660 auto calculateShape = [](const ir::Shape &input_cond_shape, const ir::Shape &input_true_shape,
661 const ir::Shape &input_false_shape, ir::Shape &new_shape) {
662 ir::Shape cond_shape = input_cond_shape;
663 ir::Shape true_shape = input_true_shape;
664 ir::Shape false_shape = input_false_shape;
665 int most_rank =
666 (cond_shape.rank() >= true_shape.rank()) && (cond_shape.rank() >= false_shape.rank())
667 ? cond_shape.rank()
668 : (false_shape.rank() >= true_shape.rank() ? false_shape.rank() : true_shape.rank());
669
670 ir::Shape calculate_shape(most_rank);
671
672 cond_shape.extendRank(most_rank);
673 true_shape.extendRank(most_rank);
674 false_shape.extendRank(most_rank);
675
676 for (int i = 0; i < most_rank; ++i)
677 {
678 calculate_shape.dim(i) =
679 (cond_shape.dim(i) >= true_shape.dim(i)) && (cond_shape.dim(i) >= false_shape.dim(i))
680 ? cond_shape.dim(i)
681 : (false_shape.dim(i) >= true_shape.dim(i) ? false_shape.dim(i) : true_shape.dim(i));
682
683 if ((cond_shape.dim(i) != calculate_shape.dim(i) && cond_shape.dim(i) != 1) ||
684 (true_shape.dim(i) != calculate_shape.dim(i) && true_shape.dim(i) != 1) ||
685 (false_shape.dim(i) != calculate_shape.dim(i) && false_shape.dim(i) != 1))
686 {
687 return false;
688 }
689 }
690
691 new_shape = calculate_shape;
692
693 return true;
694 };
695
696 bool havesame = haveSameShapes(input_cond_shape, input_true_shape, input_false_shape);
697 if (havesame)
698 {

713template <typename T>
714ir::Shape inferSliceShape(const ir::Shape &input_shape, const T *begins_buf, const T *sizes_buf)
715{
716 const uint32_t rank = input_shape.rank();
717 ir::Shape out_shape(rank);
718
719 for (uint32_t idx = 0; idx < rank; ++idx)
720 {
721 const auto input_dim = input_shape.dim(idx);
722
723 // begin is zero-based
724 auto begin = begins_buf[idx];
725 if (begin < 0)
726 throw std::runtime_error("shape inference Slice: Invalid begin.");
727

730 if (size < -1)
731 throw std::runtime_error("shape inference Slice: Invalid size.");
732
733 if (size == -1)
734 {
735 size = input_dim - begin;
736 }
737 else
738 {
739 if (input_dim < static_cast<int32_t>(begin + size))
740 throw std::runtime_error("shape inference Slice: Invalid begin and size.");
741 }
742 out_shape.dim(idx) = static_cast<int32_t>(size);
743 }
744
745 return out_shape;

753ir::Shape inferSpaceToBatchNDShape(const ir::Shape &input_shape, const ir::Shape &block_shape_shape,
754 const ir::Shape &padding_shape, const int32_t *block_shape_buf,
755 const int32_t *padding_buf)
756{
757 const uint32_t rank = input_shape.rank();
758 ir::Shape out_shape(rank);
759
760 // Currently, only 4D NHWC input/output op_context are supported.
761 // The 4D array need to have exactly 2 spatial dimensions.
762 // TODO(nupurgarg): Support arbitrary dimension in SpaceToBatchND.
763 const int32_t kInputDimensionNum = 4;
764 const int32_t kBlockSizeDimensionNum = 1;
765 const int32_t kSpatialDimensionNum = 2;
766
767 UNUSED_RELEASE(kInputDimensionNum);
768 UNUSED_RELEASE(kBlockSizeDimensionNum);
769 UNUSED_RELEASE(block_shape_shape);
770 UNUSED_RELEASE(padding_shape);
771
772 assert(block_shape_shape.rank() == kBlockSizeDimensionNum);
773 assert(block_shape_shape.dim(0) == kSpatialDimensionNum);
774 assert(padding_shape.dim(0) == kSpatialDimensionNum);
775 assert(padding_shape.dim(1) == 2); // fixed, meaning left/right padding for each element
776 assert(padding_shape.rank() == 2); // fixed, meaning dimension(dim 0) and padding length(dim 1)
777
778 // Ensures the input height and width (with padding) is a multiple of block
779 // shape height and width.
780 for (int dim = 0; dim < kSpatialDimensionNum; ++dim)
781 {
782 int final_dim_size =
783 (input_shape.dim(dim + 1) + padding_buf[dim * 2] + padding_buf[dim * 2 + 1]);
784
785 assert(final_dim_size % block_shape_buf[dim] == 0);
786
787 out_shape.dim(dim + 1) = final_dim_size / block_shape_buf[dim];
788 }
789
790 const int output_batch_size = input_shape.dim(0) * block_shape_buf[0] * block_shape_buf[1];
791 const int output_channel_size = input_shape.dim(3);
792
793 out_shape.dim(0) = output_batch_size;
794 out_shape.dim(3) = output_channel_size;
795
796 return out_shape;
797}

815ir::Shape inferSqueezeShape(const ir::Shape &in_shape, const ir::operation::Squeeze::Param &param)
816{
817 const int ndims = param.ndim;
818 const int *squeeze_dims = param.dims;
819 bool should_squeeze[8] = {false};
820 int num_squeezed_dims = 0;
821 int shape_rank = in_shape.rank();
822 if (ndims == 0)
823 {
824 for (int idx = 0; idx < shape_rank; ++idx)
825 {
826 if (in_shape.dim(idx) == 1)
827 {
828 should_squeeze[idx] = true;
829 ++num_squeezed_dims;
830 }
831 }
832 }
833 else
834 {
835 for (int idx = 0; idx < ndims; ++idx)
836 {
837 int current = squeeze_dims[idx];
838 if (current < 0)
839 {
840 current += shape_rank;
841 }
842
843 if (!(current >= 0 && current < shape_rank && in_shape.dim(current) == 1))
844 {
845 throw std::runtime_error(
846 "The following conditions must be met: 0 <= dim < Shape rank, dim == 1");
847 }
848
849 if (!should_squeeze[current])
850 {
851 ++num_squeezed_dims;
852 }
853 should_squeeze[current] = true;
854 }
855 }
856
857 // Set output shape.
858 ir::Shape out_shape(shape_rank - num_squeezed_dims);
859 for (int in_idx = 0, out_idx = 0; in_idx < shape_rank; ++in_idx)
860 {
861 if (!should_squeeze[in_idx])
862 {
863 out_shape.dim(out_idx++) = in_shape.dim(in_idx);
864 }
865 }
866
867 return out_shape;
868}

917int StartForAxis(const StridedSliceParams &params, const ir::Shape &input_shape, int axis)
918{
919 const auto begin_mask = params.begin_mask;
920 const auto *start_indices = params.start_indices;
921 const auto *strides = params.strides;
922 // Begin with the specified index.
923 int start = start_indices[axis];
924
925 // begin_mask override
926 if (begin_mask & 1 << axis)
927 {
928 if (strides[axis] > 0)
929 {
930 // Forward iteration - use the first element. These values will get
931 // clamped below (Note: We could have set them to 0 and axis_size-1, but
932 // use lowest() and max() to maintain symmetry with StopForAxis())
933 start = std::numeric_limits<int>::lowest();
934 }
935 else
936 {
937 // Backward iteration - use the last element.
938 start = std::numeric_limits<int>::max();
939 }
940 }
941
942 // Handle negative indices
943 int axis_size = input_shape.dim(axis);
944 if (start < 0)
945 {
946 start += axis_size;
947 }
948
949 // Clamping
950 start = Clamp(start, 0, axis_size - 1);
951
952 return start;
953}
954
955// Return the "real" index for the end of iteration along that axis. This is an
956// "end" in the traditional C sense, in that it points to one past the last
957// element. ie. So if you were iterating through all elements of a 1D array of
958// size 4, this function would return 4 as the stop, because it is one past the
959// "real" indices of 0, 1, 2 & 3.
960int StopForAxis(const StridedSliceParams &params, const ir::Shape &input_shape, int axis,
961 int start_for_axis)
962{
963 const auto end_mask = params.end_mask;
964 const auto shrink_axis_mask = params.shrink_axis_mask;
965 const auto *stop_indices = params.stop_indices;
966 const auto *strides = params.strides;
967
968 // Begin with the specified index
969 const bool shrink_axis = shrink_axis_mask & (1 << axis);
970 int stop = stop_indices[axis];
971
972 // When shrinking an axis, the end position does not matter (and can be
973 // incorrect when negative indexing is used, see Issue #19260). Always use
974 // start_for_axis + 1 to generate a length 1 slice, since start_for_axis has
975 // already been adjusted for negative indices.
976 if (shrink_axis)
977 {
978 stop = start_for_axis + 1;
979 }
980
981 // end_mask override
982 if (end_mask & (1 << axis))
983 {
984 if (strides[axis] > 0)
985 {
986 // Forward iteration - use the last element. These values will get
987 // clamped below
988 stop = std::numeric_limits<int>::max();
989 }
990 else
991 {
992 // Backward iteration - use the first element.
993 stop = std::numeric_limits<int>::lowest();
994 }
995 }
996
997 // Handle negative indices
998
999 const int axis_size = input_shape.dim(axis);
1000 if (stop < 0)
1001 {
1002 stop += axis_size;
1003 }
1004
1005 // Clamping
1006 // Because the end index points one past the last element, we need slightly
1007 // different clamping ranges depending on the direction.
1008 if (strides[axis] > 0)
1009 {
1010 // Forward iteration
1011 stop = Clamp(stop, 0, axis_size);
1012 }
1013 else
1014 {
1015 // Backward iteration
1016 stop = Clamp(stop, -1, axis_size - 1);
1017 }
1018
1019 return stop;
1020}

1022ir::Shape inferStridedSliceShape(const ir::Shape &input_shape, const StridedSliceParams &op_params,
1023 uint32_t rank)
1024{
1025 ir::Shape out_shape;
1026
1027 for (uint32_t idx = 0; idx < rank; ++idx)
1028 {
1029 int32_t stride = op_params.strides[idx];
1030 int32_t begin = StartForAxis(op_params, input_shape, idx);
1031 int32_t end = StopForAxis(op_params, input_shape, idx, begin);
1032
1033 // When shrinking an axis, the end position does not matter (and can be
1034 // incorrect when negative indexing is used, see Issue #19260). Always use
1035 // begin + 1 to generate a length 1 slice, since begin has
1036 // already been adjusted for negative indices by StartForAxis.
1037 const bool shrink_axis = op_params.shrink_axis_mask & (1 << idx);
1038 if (shrink_axis)
1039 {
1040 end = begin + 1;
1041 }
1042
1043 int32_t dim_shape = std::ceil((end - begin) / static_cast<float>(stride));
1044 dim_shape = dim_shape < 0 ? 0 : dim_shape;
1045 if (!shrink_axis)
1046 {
1047 out_shape.append(dim_shape);
1048 }
1049 }
1050
1051 return out_shape;
1052}

1074 const int32_t perm_size)
1075{
1076 const auto rank = in_shape.rank();

1080 std::to_string(perm_size));

1084 std::vector<int32_t> regular_perm_vec;

1088 regular_perm_vec.resize(rank);
1089 std::iota(regular_perm_vec.begin(), regular_perm_vec.end(), 0);
1090 std::reverse(regular_perm_vec.begin(), regular_perm_vec.end());

1093 else
1094 {
1095 assert(rank == perm_size);
1096 }
1097

1099 std::vector<bool> visit_perms(rank, false);
1100 for (int idx = 0; idx < rank; idx++)
1101 {
1102 const auto perm_val = perm_data[idx];
1103 // Check invalid permutation value
1104 if (perm_val < 0 || perm_val >= rank)
1105 {
1106 throw std::runtime_error("inferTransposeShape failed, bad permutation value: " +
1107 std::to_string(perm_val));
1108 }
1109
1110 // Check duplicated permutation value
1111 if (visit_perms.at(perm_val))
1112 {
1113 throw std::runtime_error("inferTransposeShape failed, duplicated permutation value: " +
1114 std::to_string(perm_val));
1115 }
1116 visit_perms.at(perm_val) = true;
1117
1118 out_shape.dim(idx) = in_shape.dim(perm_val);
1119 }
