38std::vector<float> reciprocal(const std::vector<float> &val)
39{
40 std::vector<float> res(val.size());
41 for (uint32_t i = 0; i < res.size(); i++)
42 {
43 if (val[i] == 0.0)
44 throw std::runtime_error("Scale is zero. Divide-by-zero error.");
45
46 res[i] = 1.0 / val[i];
47 }
48 return res;
49}

51bool calculate_smooth_quant_scale(luci::CircleNode *node, EqualizePattern *p)
52{
53 if (p->scale.size() != 0)
54 {
55 throw std::runtime_error("scale should be empty at this moment.");
56 }
57
58 luci::CircleConst *weight = nullptr;
59 switch (node->opcode())
60 {
61 case luci::CircleOpcode::CONV_2D:
62 {
63 auto conv = loco::must_cast<luci::CircleConv2D *>(node);
64 weight = dynamic_cast<luci::CircleConst *>(conv->filter());
65 break;
66 }
67 case luci::CircleOpcode::DEPTHWISE_CONV_2D:
68 {
69 auto conv = loco::must_cast<luci::CircleDepthwiseConv2D *>(node);
70 weight = dynamic_cast<luci::CircleConst *>(conv->filter());
71 break;
72 }
73 case luci::CircleOpcode::FULLY_CONNECTED:
74 {
75 auto fc = loco::must_cast<luci::CircleFullyConnected *>(node);
76 weight = dynamic_cast<luci::CircleConst *>(fc->weights());
77 break;
78 }
79 case luci::CircleOpcode::TRANSPOSE_CONV:
80 {
81 auto conv = loco::must_cast<luci::CircleTransposeConv *>(node);
82 weight = dynamic_cast<luci::CircleConst *>(conv->filter());
83 break;
84 }
85 default:
86 {
87 throw std::runtime_error("(calculate_smooth_quant_scale) NYI operator: " + node->name());
88 }
89 }
90
91 if (not weight)
92 return false;
93 if (weight->dtype() != loco::DataType::FLOAT32)
94 return false;
95
96 auto act_scale = p->act_scale;
97 switch (node->opcode())
98 {
99 case luci::CircleOpcode::CONV_2D:
100 case luci::CircleOpcode::DEPTHWISE_CONV_2D:
101 case luci::CircleOpcode::TRANSPOSE_CONV:
102 {
103 auto weight_rank = weight->rank();
104 if (weight_rank != 4)
105 return false;
106
107 if (act_scale.size() != weight->dim(3).value())
108 {
109 throw std::runtime_error(
110 "Mismatch between 'act_scale' size and 'filter' input channel size" +
111 std::to_string(act_scale.size()) + " != " + std::to_string(weight->dim(3).value()));
112 }
113
114 // Find filter max along with In-channel dimension
115 auto weight_size = weight->size<loco::DataType::FLOAT32>();
116 const auto weight_O = weight->dim(0).value();
117 const auto weight_H = weight->dim(1).value();
118 const auto weight_W = weight->dim(2).value();
119 const auto weight_I = weight->dim(3).value();
120 const auto norm_dim = weight_O * weight_H * weight_W;
121 std::vector<float> weight_max(weight_I, std::numeric_limits<float>::min());
122 uint32_t cur = 0;
123 for (uint32_t i = 0; i < weight_I; i++)
124 {
125 cur = i;
126 for (uint32_t j = 0; j < norm_dim; j++)
127 {
128 weight_max.at(i) = std::max(weight_max.at(i), weight->at<loco::DataType::FLOAT32>(cur));
129 cur += weight_I;
130 }
131 }
132 // Check if it properly iterates the filter.
133 assert(cur - weight_I == weight_size - 1);
134
135 // TODO parameterize "alpha"
136 auto alpha = 0.5f;
137 assert(p->act_scale.size() == weight_max.size());
138 for (uint32_t i = 0; i < p->act_scale.size(); i++)
139 {
140 p->scale.push_back(std::max(
141 std::pow(p->act_scale.at(i), alpha) / std::pow(weight_max.at(i), (1 - alpha)), 1e-5f));
142 }
143 break;
144 }
145 case luci::CircleOpcode::FULLY_CONNECTED:
146 {
147 auto weight_rank = weight->rank();
148 if (weight_rank != 2)
149 return false;
150
151 if (act_scale.size() != weight->dim(1).value())
152 {
153 throw std::runtime_error(
154 "Mismatch between 'act_scale' size and 'filter' input channel size" +
155 std::to_string(act_scale.size()) + " != " + std::to_string(weight->dim(1).value()));
156 }
157
158 // Find filter max along with In-channel dimension
159 auto weight_size = weight->size<loco::DataType::FLOAT32>();
160 const auto weight_O = weight->dim(0).value();
161 const auto weight_I = weight->dim(1).value();
162 std::vector<float> weight_max(weight_I, std::numeric_limits<float>::min());
163 uint32_t cur = 0;
164 for (uint32_t i = 0; i < weight_I; i++)
165 {
166 cur = i;
167 for (uint32_t j = 0; j < weight_O; j++)
168 {
169 weight_max.at(i) = std::max(weight_max.at(i), weight->at<loco::DataType::FLOAT32>(cur));
170 cur += weight_I;
171 }
172 }
173 // Check if it properly iterates the filter.
174 assert(cur - weight_I == weight_size - 1);
175
176 // TODO parameterize "alpha"
177 auto alpha = 0.5f;
178 assert(p->act_scale.size() == weight_max.size());
179 for (uint32_t i = 0; i < p->act_scale.size(); i++)
180 {
181 p->scale.push_back(std::max(
182 std::pow(p->act_scale.at(i), alpha) / std::pow(weight_max.at(i), (1 - alpha)), 1e-5f));
183 }
184 break;
185 }
186 default:
187 {
188 throw std::runtime_error("(calculate_smooth_quant_scale) NYI operator: " + node->name());
189 }
190 }
191
192 return true;
193}

238 ss->inputs(1, param);

241 luci::add_origin(ss, luci::get_origin(node));
242 }
243 set_input(node, ss);

246 void insert_scale_after(luci::CircleNode *node, const std::vector<float> &scale)
247 {
248 if (not node)
249 {
250 throw std::runtime_error("(insert_scale_after) Invalid node.");
251 }
252
253 // Create const for scale
254 auto param = node->graph()->nodes()->create<luci::CircleConst>();
255 {
256 param->name(node->name() + "_scale_const");
257 param->dtype(loco::DataType::FLOAT32);
258 param->rank(1);
259 param->dim(0).set(scale.size());
260 param->size<loco::DataType::FLOAT32>(scale.size());
261 for (uint32_t i = 0; i < scale.size(); i++)
262 {
263 param->at<loco::DataType::FLOAT32>(i) = scale.at(i);
264 }
265 param->shape_status(luci::ShapeStatus::VALID);
266 luci::add_origin(param, luci::get_origin(node));
267 }
268
269 // NOTE. Get a node user before setting CircleCustom("scale")'s input.
270 auto uses = loco::succs(node);
271 if (uses.size() != 1)
272 {
273 throw std::runtime_error("Not support multiple output nodes.");
274 }
275
276 // Create Scale Op
277 auto ss = node->graph()->nodes()->create<luci::CircleCustom>(2, 1);
278 {
279 ss->name(node->name() + "_scale");
280 ss->dtype(loco::DataType::FLOAT32);
281 ss->rank(node->rank());
282 for (uint32_t i = 0; i < node->rank(); i++)
283 ss->dim(i).set(node->dim(i).value());
284 ss->custom_code("scale");
285
286 ss->inputs(0, node);
287 ss->inputs(1, param);
288
289 ss->shape_status(luci::ShapeStatus::VALID);
290 luci::add_origin(ss, luci::get_origin(node));
291 }
292
293 set_input(loco::must_cast<luci::CircleNode *>(*uses.begin()), ss);
294 }

296private:
297 void insert(luci::CircleNode *node)
298 {
299 assert(_pattern->scale.size() ==
300 0); // scale for smooth qaunt is not calculated yet at this moment.
301 auto valid = ::calculate_smooth_quant_scale(node, _pattern);
302 auto back_node = node;
303 // Find front node.
304 const auto support_depth = 2;
305 auto front_node = find_arg_with_name(node, _pattern->front, support_depth);
306 if (not front_node)
307 {
308 throw std::runtime_error("Cannot find front node: " + _pattern->front);
309 }
310 insert_scale_after(front_node, reciprocal(_pattern->scale));
311 insert_scale_before(back_node, _pattern->scale);
312 }

336void InsertScaleShift::run(loco::Graph *g)
337{
338 // Create a map for pattern matching
339 // { back(string) -> EqualizationPattern*}
340 // This assumes that each EqualizePattern has a unique 'back'
341 std::map<std::string, EqualizePattern *> pattern_by_back;
342 {
343 for (auto &pattern : _patterns)
344 {
345 auto back = pattern.back;
346 pattern_by_back[back] = &pattern;
347 }
348 }
349
350 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
351 {
352 auto cnode = loco::must_cast<luci::CircleNode *>(node);
353
354 if (pattern_by_back.find(cnode->name()) == pattern_by_back.end())
355 continue;
356
357 auto pattern = pattern_by_back.at(cnode->name());
358 InsertScaleShiftVisitor issv(pattern);
359 cnode->accept(&issv);
360 }
361}
