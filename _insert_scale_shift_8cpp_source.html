41luci::CircleCustom *create_scale_or_shift(luci::CircleNode *node, const std::vector<float> &val,
42 const std::string &code)
43{
44 assert(node); // FIX_CALLER_UNLESS
45
46 THROW_UNLESS(node->rank() == 4, "Node rank is not four");
47
48 // Assume NHWC (index of C: 3)
49 auto channel_size = node->dim(3).value();
50 THROW_UNLESS(val.size() == channel_size, "Channel and scale/shift size mismatch");
51
52 // Create parameter
53 auto param = node->graph()->nodes()->create<luci::CircleConst>();
54 {
55 param->name(node->name() + "_" + code + "_param");
56 param->dtype(loco::DataType::FLOAT32);
57 param->rank(1);
58 param->dim(0).set(channel_size);
59 param->size<loco::DataType::FLOAT32>(channel_size);
60 for (uint32_t i = 0; i < channel_size; i++)
61 {
62 param->at<loco::DataType::FLOAT32>(i) = val[i];
63 }
64 param->shape_status(luci::ShapeStatus::VALID);
65 luci::add_origin(param, luci::get_origin(node));
66 }
67
68 // Create Scale/Shift Op
69 auto ss = node->graph()->nodes()->create<luci::CircleCustom>(2, 1);
70 {
71 ss->name(node->name() + "_" + code);
72 ss->dtype(loco::DataType::FLOAT32);
73 ss->rank(node->rank());
74 for (uint32_t i = 0; i < node->rank(); i++)
75 ss->dim(i).set(node->dim(i).value());
76 ss->custom_code(code);
77
78 ss->inputs(0, node);
79 ss->inputs(1, param);
80
81 ss->shape_status(luci::ShapeStatus::VALID);
82 luci::add_origin(ss, luci::get_origin(node));
83 }
84
85 return ss;
86}

88std::vector<float> reciprocal(const std::vector<float> &val)
89{
90 std::vector<float> res(val.size());
91 for (uint32_t i = 0; i < res.size(); i++)
92 {
93 if (val[i] == 0.0)
94 throw std::runtime_error("Scale is zero. Divide-by-zero error.");
95
96 res[i] = 1.0 / val[i];
97 }
98 return res;
99}

101std::vector<float> minus(const std::vector<float> &val)
102{
103 std::vector<float> res(val.size());
104 for (uint32_t i = 0; i < res.size(); i++)
105 {
106 res[i] = -val[i];
107 }
108 return res;
109}

139struct InsertScaleShiftVisitor final : public luci::CircleNodeMutableVisitor<void>
140{
141 InsertScaleShiftVisitor(const EqualizePattern *p) : _pattern(p)
142 {
143 // DO NOTHING
144 }
145
146private:
147 const EqualizePattern *_pattern = nullptr;
148
149private:
150 // Generate scale/shift Ops and return the last one
151 // lnode: 'front' of EqualizePattern
152 // Never return nullptr
153 luci::CircleCustom *gen_scale_shift(loco::Node *lnode) const
154 {
155 auto node = loco::must_cast<luci::CircleNode *>(lnode);
156
157 assert(node->name() == _pattern->front); // FIX_CALLER_UNLESS
158
159 luci::CircleCustom *bottom = nullptr;
160
161 switch (_pattern->type)
162 {
163 case EqualizePattern::Type::ScaleOnly:
164 {
165 auto post_scale = create_post_scale(node, _pattern);
166 bottom = create_pre_scale(post_scale, _pattern);
167 break;
168 }
169 case EqualizePattern::Type::ShiftOnly:
170 {
171 auto post_shift = create_post_shift(node, _pattern);
172 bottom = create_pre_shift(post_shift, _pattern);
173 break;
174 }
175 case EqualizePattern::Type::ScaleShift:
176 {
177 auto post_scale = create_post_scale(node, _pattern);
178 auto post_shift = create_post_shift(post_scale, _pattern);
179 auto pre_shift = create_pre_shift(post_shift, _pattern);
180 bottom = create_pre_scale(pre_shift, _pattern);
181 break;
182 }
183 default:
184 throw std::runtime_error("Unsupported EqualizePattern type");
185 }
186
187 assert(bottom != nullptr); // FIX_ME_UNLESS
188
189 return bottom;
190 }

192 void visit(luci::CircleOutput *) {}
193
194 void visit(luci::CircleNode *node) { throw std::runtime_error("NYI operator: " + node->name()); }
195
196 void visit(luci::CircleConv2D *node)
197 {
198 auto bottom = gen_scale_shift(node->input());
199 node->input(bottom);
200 }
201
202 void visit(luci::CircleDepthwiseConv2D *node)
203 {
204 auto bottom = gen_scale_shift(node->input());
205 node->input(bottom);
206 }
207
208 void visit(luci::CircleTransposeConv *node)

210 auto bottom = gen_scale_shift(node->outBackprop());

214 void visit(luci::CircleInstanceNorm *node)
215 {
216 auto bottom = gen_scale_shift(node->input());
217 node->input(bottom);
218 }
219 void visit(luci::CirclePad *node)
220 {
221 auto bottom = gen_scale_shift(node->input());
222 node->input(bottom);
223 }
224 void visit(luci::CircleSlice *node)
225 {
226 auto bottom = gen_scale_shift(node->input());
227 node->input(bottom);
228 }

236void InsertScaleShift::run(loco::Graph *g)
237{
238 // Create a map for pattern matching
239 // { back(string) -> EqualizationPattern*}
240 // This assumes that each EqualizePattern has a unique 'back'
241 std::map<std::string, const EqualizePattern *> pattern_by_back;
242 {
243 for (const auto &pattern : _patterns)
244 {
245 auto back = pattern.back;
246 pattern_by_back[back] = &pattern;
247 }
248 }
249
250 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
251 {
252 auto cnode = loco::must_cast<luci::CircleNode *>(node);
253
254 if (pattern_by_back.find(cnode->name()) == pattern_by_back.end())
255 continue;
256
257 auto pattern = pattern_by_back.at(cnode->name());
258 InsertScaleShiftVisitor issv(pattern);
259 cnode->accept(&issv);
260 }
261}
