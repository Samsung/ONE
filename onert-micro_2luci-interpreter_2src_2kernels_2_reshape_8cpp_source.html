29static Shape extractShapeFromTensor(const Tensor *tensor)
30{
31 assert(tensor->element_type() == DataType::S32);

42static void resolveUnknownDimension(const Shape &input_shape, Shape *output_shape)
43{
44 const int32_t num_input_elements = input_shape.num_elements();
45 int32_t num_output_elements = 1;
46 int unknown_dim_index = -1;
47 for (int i = 0; i < output_shape->num_dims(); ++i)
48 {
49 const int32_t value = output_shape->dim(i);
50 if (value == -1)
51 {
52 assert(unknown_dim_index == -1);
53 unknown_dim_index = i;
54 }
55 else
56 {
57 num_output_elements *= value;
58 }
59 }
60 if (unknown_dim_index != -1)
61 {
62 output_shape->dim(unknown_dim_index) = num_input_elements / num_output_elements;
63 num_output_elements *= output_shape->dim(unknown_dim_index);
64 }
65 assert(num_output_elements == num_input_elements);
66}

86 input_tensor->set_data_buffer(nullptr);
