83bool find_loop(TransferSequence::const_iterator beg, TransferSequence::const_iterator end,
84 uint32_t *p_count, uint32_t *p_src_step, uint32_t *p_dst_step)
85{
86 assert(p_count != nullptr);
87 assert(p_src_step != nullptr);
88 assert(p_dst_step != nullptr);
89
90 const uint32_t size = end - beg;
91
92 for (uint32_t window = 1; window <= size; ++window)
93 {
94 if (size % window != 0)
95 {
96 continue;
97 }
98
99 auto src_step_at = [&beg, window](uint32_t n) {
100 return (beg + n)->from() - (beg + n - window)->from();
101 };
102
103 auto dst_step_at = [&beg, window](uint32_t n) {
104 return (beg + n)->into() - (beg + n - window)->into();
105 };
106
107 const uint32_t count = size / window;
108 const uint32_t src_step = src_step_at(window);
109 const uint32_t dst_step = dst_step_at(window);
110
111 bool consistent = true;
112
113 for (uint32_t n = window + 1; n < size; ++n)
114 {
115 if ((src_step_at(n) != src_step) || (dst_step_at(n) != dst_step))
116 {
117 consistent = false;
118 break;
119 }
120 }
121
122 if (consistent)
123 {
124 *p_count = count;
125 *p_src_step = src_step;
126 *p_dst_step = dst_step;
127 return true;
128 }
129 }
130
131 return false;
132}

181TransferNest as_nest(const TransferSequence &seq)
182{
183 TransferNest nest;
184
185 auto beg = seq.begin();
186 auto end = seq.end();
187
188 uint32_t window = end - beg;
189 uint32_t count = 0;
190 uint32_t src_step = 0;
191 uint32_t dst_step = 0;
192
193 while ((window > 1) && find_loop(beg, end, &count, &src_step, &dst_step))
194 {
195 assert(window % count == 0);
196
197 window /= count;
198 end = beg + window;
199
200 nest.emplace_back(count, src_step, dst_step);
201 }
202
203 return nest;
204};

206uint32_t loop_count(const TransferNest &nest)
207{
208 uint32_t count = 1;
209
210 for (const auto &loop : nest)

215 return count;

218class InstrPrinter : public coco::Instr::Visitor<pp::LinearDocument>
219{
220public:
221 InstrPrinter(const enco::MemoryContext &mem) : _mem(mem)

224 }

226private:
227 pp::LinearDocument visit(const coco::Shuffle *shuffle) override
228 {
229 auto from = shuffle->from();
230 auto into = shuffle->into();
231
232 //
233 // Analyze 'Shuffle' pattern, and convert it as nested loops
234 //
235 auto tseq = as_transfer_sequence(shuffle);
236 auto nest = as_nest(tseq);
237 assert(tseq.size() % loop_count(nest) == 0);
238 uint32_t window = tseq.size() / loop_count(nest);
239
240 //
241 // Generate loop body
242 //
243 pp::EnclosedDocument loop_body;
244
245 auto var_at = [](uint32_t lv) { return pp::fmt("_", lv); };
246
247 for (uint32_t lv = 0; lv < nest.size(); ++lv)
248 {
249 auto var = var_at(lv);
250
251 loop_body.front().append("for (uint32_t ", var, " = 0; ", var, " < ", nest.at(lv).count(),
252 "; ++", var, ") {");
253 loop_body.front().indent();
254
255 loop_body.back().append("}");
256 loop_body.back().indent();
257 }
258
259 std::string src_index = "0";
260 std::string dst_index = "0";
261
262 for (uint32_t lv = 0; lv < nest.size(); ++lv)
263 {
264 src_index += pp::fmt(" + ", nest.at(lv).step().src(), " * ", var_at(lv));
265 dst_index += pp::fmt(" + ", nest.at(lv).step().dst(), " * ", var_at(lv));
266 }
267
268 for (uint32_t n = 0; n < window; ++n)
269 {
270 const auto src_base = pp::fmt("reinterpret_cast<const float *>(", _mem.base(from), ")");
271 const auto dst_base = pp::fmt("reinterpret_cast<float *>(", _mem.base(into), ")");
272
273 loop_body.front().append(dst_base, "[", dst_index, " + ", tseq.at(n).into(), "] = ", src_base,
274 "[", src_index, " + ", tseq.at(n).from(), "];");
275 }
276
277 pp::LinearDocument res;
278 res.append(loop_body);
279 return res;
280 }
281
282private:
283 const enco::MemoryContext &_mem;
284};
