114 // "mutable_" prefix is deliberately introduced below to avoid resolution issue.
115 //
116 // Let's assume that two "deps" are overloaded in Bag as follows:
117 // class Bag
118 // {
119 // private:
120 // DepSet *deps(void); <-- 1
121 // public:
122 // const DepSet *deps(void) const; <-- 2
123 // };
124 //
125 // C++ compiler tries to invoke method 1 unless a bag itself is const. Thus, any "deps" calls
126 // over non-const bags except those calls from friend classes will introduce build error.
127
128 // WARN Only Dep is allowed to access this method
129 DepSet *mutable_deps(void) { return &_deps; }
130 // WARN Only Read is allowed to access this method
131 ReadSet *mutable_reads(void) { return &_reads; }
132 // WARN Only Update is allowed to access this method

135private:
