38void registerGraphInputTensors(loco::Graph *graph, luci::SubGraphContext &ctx)
39{
40 for (uint32_t n = 0; n < graph->inputs()->size(); ++n)
41 {
42 auto node = luci::input_node(graph, n);
43 assert(node != nullptr);
44 ctx._inputs.push_back(luci::get_tensor_index(node));
45 }
46}
47
48void registerGraphOutputTensors(loco::Graph *graph, luci::SubGraphContext &ctx)
49{
50 for (uint32_t n = 0; n < graph->outputs()->size(); ++n)
51 {
52 auto push = luci::output_node(graph, n);
53 assert(push != nullptr);
54 auto node = push->from();
55 assert(node != nullptr);
56
57 // Do not export CircleOutput when it's input is CircleOutputExclude
58 if (dynamic_cast<luci::CircleOutputExclude *>(push->from()) != nullptr)
59 {
60 continue;
61 }
62
63 ctx._outputs.push_back(luci::get_tensor_index(node));
64 }
65}

75Offset<Vector<Offset<OperatorCode>>>
76encodeOperatorCodes(FlatBufferBuilder &builder, std::unordered_map<luci::OpCode, uint32_t> &opcodes)
77{
78 std::vector<Offset<OperatorCode>> operator_codes_vec(opcodes.size());
79 for (auto it : opcodes)
80 {
81 uint32_t idx = it.second;
82 int8_t dep_code = 127; // BuiltinOperator_PLACEHOLDER_FOR_GREATER_OP_CODES
83 if (it.first.opcode < BuiltinOperator_PLACEHOLDER_FOR_GREATER_OP_CODES)
84 dep_code = static_cast<int8_t>(it.first.opcode);
85 if (it.first.opcode != BuiltinOperator_CUSTOM)
86 {
87 operator_codes_vec[idx] =
88 CreateOperatorCode(builder, dep_code, 0, it.first.version, it.first.opcode);
89 }
90 else
91 {
92 operator_codes_vec[idx] =
93 CreateOperatorCode(builder, dep_code, builder.CreateString(it.first.custom_code),
94 it.first.version, it.first.opcode);
95 }
96 }
97
98 return builder.CreateVector(operator_codes_vec);
99}

113CircleExporterImpl::exportSubgraph(SerializedGraphData &gd)
114{
115 auto tensors = _builder.CreateVector(gd._tensors);

118 auto operators = _builder.CreateVector(gd._operators);
119 auto name = _builder.CreateString(gd._name);

125void CircleExporterImpl::exportGraph(loco::Graph *graph)
126{
127 // do graph optimization
128 optimize(graph);
129
130 _builder.Clear();
131
132 SerializedModelData md;
133 SerializedGraphData gd;
134
135 // This version is taken from comment in fbs
136 constexpr uint32_t version = 0;
137
138 // set Subgraph name
139 gd._name = graph->name();
140
141 // TODO set this value properly
142 gd._data_format = circle::DataFormat::DataFormat_CHANNELS_LAST;
143
144 // prepare model data
145 prepareModelData(_builder, md);
146
147 // parse graph into SerializedModelData structure
148 exportOpDefinedTensors(graph, _builder, md, gd);
149
150 // NOTE Invoke these register functions only after each node is annotated with its tensor_index
151 registerGraphInputTensors(graph, gd);
152 registerGraphOutputTensors(graph, gd);
153
154 exportNodes(graph, _builder, md, gd);
155
156 // encode operator codes
157 auto operator_codes = encodeOperatorCodes(_builder, md._operator_codes);
158
159 // Subgraphs
160 Offset<SubGraph> subgraph = exportSubgraph(gd);
161 auto subgraphs = _builder.CreateVector(std::vector<Offset<SubGraph>>{subgraph});
162
163 // Description
164 std::string description_str = "nnpackage";
165 auto description = _builder.CreateString(description_str);
166
167 // Metadata
168 auto metadata_vec = createCircleMetadataVector(_builder, md);
169 auto metadata = _builder.CreateVector(std::vector<Offset<Metadata>>(metadata_vec));
170
171 // create array of buffers
172 auto buffers = _builder.CreateVector(md._buffers);
173
174 // Model
175 auto model_offset = CreateModel(_builder, version, operator_codes, subgraphs, description,
176 buffers, 0 /* metadata_buffer */, metadata);
177 FinishModelBuffer(_builder, model_offset);
178}

180void CircleExporterImpl::exportModule(Module *module)
181{
182 assert(module->size() > 0);
183 // do graph optimization
184
185 SerializedModelData md;
186
187 _builder.Clear();
188
189 // prepare model data
190 prepareModelData(_builder, md);
191
192 std::vector<flatbuffers::Offset<circle::SubGraph>> subgraph_vec;
193
194 for (size_t g = 0; g < module->size(); ++g)
195 {
196 auto graph = module->graph(g);
197
198 optimize(graph);
199
200 SerializedGraphData gd;
201
202 // set Subgraph name
203 gd._name = graph->name();
204
205 // TODO set this value properly
206 gd._data_format = circle::DataFormat::DataFormat_CHANNELS_LAST;
207
208 // parse graph into SerializedModelData structure
209 exportOpDefinedTensors(graph, _builder, md, gd);
210
211 // NOTE Invoke these register functions only after each node is annotated with its tensor_index
212 registerGraphInputTensors(graph, gd);
213 registerGraphOutputTensors(graph, gd);
214
215 exportNodes(graph, _builder, md, gd);
216
217 // Subgraphs
218 Offset<SubGraph> subgraph = exportSubgraph(gd);
219 subgraph_vec.push_back(subgraph);
220 }
221
222 auto subgraphs = _builder.CreateVector(std::vector<Offset<SubGraph>>{subgraph_vec});
223
224 // encode operator codes
225 auto operator_codes = encodeOperatorCodes(_builder, md._operator_codes);
226
227 // Description
228 std::string description_str = "nnpackage";
229 auto description = _builder.CreateString(description_str);
230
231 // Metadata
232 md._metadata.source_table(module->source_table());
233 auto metadata_vec = createCircleMetadataVector(_builder, md);
234 auto metadata = _builder.CreateVector(std::vector<Offset<Metadata>>(metadata_vec));
235
236 // create array of buffers
237 auto buffers = _builder.CreateVector(md._buffers);
238
239 // This version is taken from comment in fbs
240 constexpr uint32_t version = 0;
241
242 // Model
243 auto model_offset = CreateModel(_builder, version, operator_codes, subgraphs, description,
244 buffers, 0 /* metadata_buffer */, metadata);
245 FinishModelBuffer(_builder, model_offset);
246}
