29// Helper method that fills the left and right pads.
30template <typename T>
31inline void getPadding(const T *data, int offset, int64_t *left_pad, int64_t *right_pad)
32{
33 *left_pad = static_cast<int64_t>(*(data + offset * 2));
34 *right_pad = static_cast<int64_t>(*(data + offset * 2 + 1));
35}

37// Given dimension index and the left/right padding.
38// Returns the corresponding dimension in the input array.
39inline int getInputDimension(int padded_dimension, int left_pad, int right_pad, int input_dim_size,
40 int offset)
41{
42 if (padded_dimension < left_pad)
43 {
44 const int original_ind = left_pad + offset - 1;
45 return original_ind - (std::min(padded_dimension, original_ind - offset));
46 }
47 padded_dimension -= left_pad;
48 if (padded_dimension >= input_dim_size)
49 {
50 padded_dimension -= input_dim_size;
51 const int original_ind = input_dim_size - (1 + offset);
52 return original_ind - std::min(padded_dimension, original_ind);
53 }
54 return padded_dimension;
55}

57// Given and index in output array, returns the index of the value
58// in input array.
59int getFlatIndex(int index, int num_dims, const luci_interpreter::DataType padding_matrix_type,
60 const uint8_t *padding_matrix_data, const int32_t *input_dims,
61 int *output_dims_num_elements, int *input_dims_num_elements, const int offset)
62{
63 int flat_index = 0;
64 int64_t left_pad = 0, right_pad = 0, dimension_index, index_in_input;
65
66 for (int i = 0; i < num_dims; ++i)
67 {
68 switch (padding_matrix_type)
69 {
70 case luci_interpreter::DataType::S32:
71 getPadding(luci_interpreter::kernels::getTensorData<int32_t>(padding_matrix_data), i,
72 &left_pad, &right_pad);
73 break;
74 case luci_interpreter::DataType::S64:
75 getPadding(luci_interpreter::kernels::getTensorData<int64_t>(padding_matrix_data), i,
76 &left_pad, &right_pad);
77 break;
78 default:
79 break;
80 }
81 dimension_index = index / output_dims_num_elements[i];
82
83 index_in_input = getInputDimension(dimension_index, left_pad, right_pad, input_dims[i], offset);
84
85 flat_index += index_in_input * (input_dims_num_elements)[i];
86 index %= output_dims_num_elements[i];
87 }
88
89 return flat_index;
90}

94template <typename T>
95void MirrorPad(const luci_interpreter::DataType padding_matrix_type,
96 const uint8_t *padding_matrix_data, const int32_t *input_dims,
97 int *output_dims_num_elements, int *input_dims_num_elements, const T *input_data,
98 T *output_data, const int offset, const int num_dims, const int output_size)
99{
100 for (int i = 0; i < output_size; ++i)
101 {
102 output_data[i] =
103 input_data[getFlatIndex(i, num_dims, padding_matrix_type, padding_matrix_data, input_dims,
104 output_dims_num_elements, input_dims_num_elements, offset)];
105 }
106}
