59class TensorShapeExpander
60{
61public:
62 TensorShapeExpander(const loco::TensorShape &shape) : _shape{shape}
63 {
64 // DO NOTHING
65 }
66
67public:
68 loco::TensorShape to(uint32_t output_rank)
69 {
70 auto const &input_shape = _shape;
71 uint32_t const input_rank = input_shape.rank();
72
73 assert(input_rank <= output_rank && "Cannot shrink rank");
74 uint32_t const axis_shift = output_rank - input_rank;
75
76 loco::TensorShape output_shape;
77
78 output_shape.rank(output_rank);
79 for (uint32_t axis = 0; axis < output_rank; ++axis)
80 {
81 output_shape.dim(axis) = (axis < axis_shift) ? 1 : input_shape.dim(axis - axis_shift);
82 }
83
84 return output_shape;
85 }
86
87private:
88 const loco::TensorShape _shape;
89};

96class ExpandRankFunctor final
97{
98public:
99 ExpandRankFunctor(uint32_t rank) : _rank{rank}
100 {
101 // DO NOTHING
102 }
103
104public:
105 NodeWithTensorShape operator()(const NodeWithTensorShape &in) const
106 {
107 auto const input_node = in.node();
108 auto const input_shape = in.shape();
109 auto const input_rank = input_shape.rank();
110
111 uint32_t const expected_rank = _rank;
112
113 assert(input_rank <= expected_rank);
114 if (input_rank == expected_rank)
115 {
116 // Nothing to expand
117 return in;
118 }
119
120 auto g = input_node->graph();
121 assert(g != nullptr);
122
123 auto output_shape = expand(input_shape).to(expected_rank);
124 auto output_node = g->nodes()->create<loco::FixedReshape>();
125
126 output_node->input(input_node);
127 output_node->rank(expected_rank);
128 for (uint32_t axis = 0; axis < expected_rank; ++axis)
129 {
130 output_node->dim(axis) = output_shape.dim(axis);
131 }
132
133 return glue(output_node, output_shape);
134 }
135
136private:
137 uint32_t _rank;
138};

145class ExpandDimsFunctor final
146{
147public:
148 ExpandDimsFunctor(const loco::TensorShape &shape) : _shape{shape}
149 {
150 // DO NOTHING
151 }
152
153public:
154 NodeWithTensorShape operator()(const NodeWithTensorShape &in) const
155 {
156 auto const input_node = in.node();
157 auto const input_shape = in.shape();
158 const auto &output_shape = _shape;
159
160 assert(input_shape.rank() == output_shape.rank());
161
162 if (input_shape == output_shape)
163 {
164 // Nothing to expand
165 return in;
166 }
167
168 uint32_t const rank = output_shape.rank();
169
170 auto g = input_node->graph();
171 assert(g != nullptr);
172
173 auto output_node = g->nodes()->create<loco::TensorBroadcast>();
174
175 for (uint32_t axis = 0; axis < rank; ++axis)
176 {
177 auto input_dim = input_shape.dim(axis);
178 auto output_dim = output_shape.dim(axis);
179
180 assert(input_dim.known() and output_dim.known());
181
182 if (!(input_dim == output_dim))
183 {
184 assert(input_dim == 1);
185 output_node->mapping()->dim(axis) = output_dim;
186 }
187 }
188
189 output_node->input(input_node);
190
191 return glue(output_node, output_shape);
192 }
193
194private:
195 loco::TensorShape _shape;
196};
