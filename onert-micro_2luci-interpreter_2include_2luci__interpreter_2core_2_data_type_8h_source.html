150inline uint32_t size(DataType data_type)
151{
152 switch (data_type)
153 {
154 case DataType::S8:
155 return sizeof(DataTypeImpl<DataType::S8>::Type);
156 case DataType::U8:
157 return sizeof(DataTypeImpl<DataType::U8>::Type);
158 case DataType::S16:
159 return sizeof(DataTypeImpl<DataType::S16>::Type);
160 case DataType::U16:
161 return sizeof(DataTypeImpl<DataType::U16>::Type);
162 case DataType::S32:
163 return sizeof(DataTypeImpl<DataType::S32>::Type);
164 case DataType::U32:
165 return sizeof(DataTypeImpl<DataType::U32>::Type);
166 case DataType::S64:
167 return sizeof(DataTypeImpl<DataType::S64>::Type);
168 case DataType::U64:
169 return sizeof(DataTypeImpl<DataType::U64>::Type);
170 case DataType::FLOAT16:
171 return sizeof(DataTypeImpl<DataType::FLOAT16>::Type);
172 case DataType::FLOAT32:
173 return sizeof(DataTypeImpl<DataType::FLOAT32>::Type);
174 case DataType::FLOAT64:
175 return sizeof(DataTypeImpl<DataType::FLOAT64>::Type);
176 case DataType::BOOL:
177 return sizeof(DataTypeImpl<DataType::BOOL>::Type);
178 case DataType::STRING:
179 // STRING is variable length. Cannot decide size by type
180 assert(false && "Invalid size call with STRING type");
181 default:
182 // TODO Support remaining data types.
183 assert(false);
184 return UINT32_MAX; // Avoid compiler warning.
185 }
186}

188inline size_t getDataTypeSize(DataType data_type) { return size(data_type); }
