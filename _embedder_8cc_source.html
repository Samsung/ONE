35/* NOTE: getNthPercentile is copied from compiler/record-minmax/include/RecordFunction.h */
40float getNthPercentile(std::vector<float> &vector, float percentile)
41{
42 if (percentile < 0 || percentile > 100)
43 throw std::runtime_error("Percentile must be ranged from 0 to 100");
44
45 if (vector.empty())
46 throw std::runtime_error("Percentile must take a non-empty vector as an argument");
47
48 if (vector.size() == 1)
49 return vector[0];
50
51 std::vector<float> copy;
52 copy.assign(vector.begin(), vector.end());
53 std::sort(copy.begin(), copy.end());
54
55 if (percentile == 0.0)
56 return copy.front();
57
58 if (percentile == 100.0)
59 return copy.back();
60
61 int index = static_cast<int>(std::floor((copy.size() - 1) * percentile / 100.0));
62
63 float percent_i = static_cast<float>(index) / static_cast<float>(copy.size() - 1);
64 float fraction =
65 (percentile / 100.0 - percent_i) / ((index + 1.0) / (copy.size() - 1.0) - percent_i);
66 float res = copy[index] + fraction * (copy[index + 1] - copy[index]);
67 return res;
68}

77void Embedder::embed(luci::Module *module, const std::string &minmax_path,
78 const EmbedderOptions &opt)
79{
80 if (module == nullptr)
81 throw std::runtime_error{"Input module is nullptr"};
82
83 MinMaxReader mmr{minmax_path};
84
85 for (size_t idx = 0; idx < module->size(); ++idx)
86 {
87 auto graph = module->graph(idx);
88
89 /* read subgraph inputs */
90 const auto input_nodes = loco::input_nodes(graph);
91 const auto n_inputs = input_nodes.size();
92 for (size_t input_idx = 0; input_idx < n_inputs; ++input_idx)
93 {
94 const auto *circle_input = loco::must_cast<const luci::CircleInput *>(input_nodes[input_idx]);
95 if (circle_input->index() != input_idx)
96 throw std::runtime_error("Input order in minmax recording does not match to circle");
97
98 auto minmax = mmr.readInput(0, idx, input_idx);
99 auto min = getNthPercentile(minmax.min_vector, opt.min_percentile);
100 auto max = getNthPercentile(minmax.max_vector, opt.max_percentile);
101 auto quantparam = std::make_unique<luci::CircleQuantParam>();
102 quantparam->min.push_back(min);
103 quantparam->max.push_back(max);
104 const auto *circle_node = loco::must_cast<const luci::CircleNode *>(input_nodes[input_idx]);
105 auto mutable_node = const_cast<luci::CircleNode *>(circle_node);
106 mutable_node->quantparam(std::move(quantparam));
107 }
108
109 /* read op outputs */
110 uint32_t n_nodes = graph->nodes()->size();
111 for (uint32_t i = 0; i < n_nodes; ++i)
112 {
113 auto node = loco::must_cast<luci::CircleNode *>(graph->nodes()->at(i));
114 if (not luci::has_node_id(node)) // Skip non-op nodes (e.g. input/const/output)
115 continue;
116 auto op_idx = luci::get_node_id(node);
117 auto minmax = mmr.readOP(0, idx, op_idx);
118 auto min = getNthPercentile(minmax.min_vector, opt.min_percentile);
119 auto max = getNthPercentile(minmax.max_vector, opt.max_percentile);
120 auto quantparam = std::make_unique<luci::CircleQuantParam>();
121 quantparam->min.push_back(min);
122 quantparam->max.push_back(max);
123 auto mutable_node = const_cast<luci::CircleNode *>(node);
124 mutable_node->quantparam(std::move(quantparam));
125 }
126
127 if (!luci::validate(graph))
128 throw std::runtime_error{"Circle after embedding minmax is invalid"};
129 }
130}
