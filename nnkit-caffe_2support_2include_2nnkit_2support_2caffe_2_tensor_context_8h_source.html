46 static nncc::core::ADT::tensor::Shape shapeOf(const ::caffe::Blob<DType> &blob)
47 {
48 nncc::core::ADT::tensor::Shape shape;
49
50 const uint32_t rank = blob.shape().size();
51
52 shape.resize(rank);
53 for (uint32_t axis = 0; axis < rank; ++axis)
54 {
55 shape.dim(axis) = blob.shape(axis);
56 }
57
58 return shape;
59 }
