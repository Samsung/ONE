29class TrainInfoBuilder
30{
31public:
32 TrainInfoBuilder(const std::unique_ptr<ir::train::TrainingInfo> &training_info) : _builder(1024)
33 {
34 const auto &optimizerInfo = training_info->optimizerInfo();
35 const auto &lossInfo = training_info->lossInfo();
36
37 ::circle::Optimizer optimizer;
38 ::circle::OptimizerOptions optimizer_opt_type;
39 ::flatbuffers::Offset<void> optimizer_opt;
40 switch (optimizerInfo.optim_code)
41 {
42 case ir::train::OptimizerCode::SGD:
43 optimizer = ::circle::Optimizer_SGD;
44 optimizer_opt_type = ::circle::OptimizerOptions_SGDOptions;
45 optimizer_opt = ::circle::CreateSGDOptions(_builder, optimizerInfo.learning_rate).Union();
46 break;
47 case ir::train::OptimizerCode::Adam:
48 optimizer = ::circle::Optimizer_ADAM;
49 optimizer_opt_type = ::circle::OptimizerOptions_AdamOptions;
50 optimizer_opt = ::circle::CreateAdamOptions(_builder, optimizerInfo.learning_rate).Union();
51 break;
52 default:
53 throw std::runtime_error("Not supported optimizer code");
54 }
55
56 ::circle::LossFn lossfn;
57 ::circle::LossFnOptions lossfn_opt_type;
58 ::flatbuffers::Offset<void> lossfn_opt;
59 switch (lossInfo.loss_code)
60 {
61 case ir::train::LossCode::MeanSquaredError:
62 lossfn = ::circle::LossFn_MEAN_SQUARED_ERROR;
63 lossfn_opt_type = ::circle::LossFnOptions_MeanSquaredErrorOptions;
64 lossfn_opt = ::circle::CreateMeanSquaredErrorOptions(_builder).Union();
65 break;
66 case ir::train::LossCode::CategoricalCrossentropy:
67 lossfn = ::circle::LossFn_CATEGORICAL_CROSSENTROPY;
68 lossfn_opt_type = ::circle::LossFnOptions_CategoricalCrossentropyOptions;
69 lossfn_opt = ::circle::CreateCategoricalCrossentropyOptions(_builder).Union();
70 break;
71 default:
72 throw std::runtime_error("Not supported loss code");
73 }
74
75 ::circle::LossReductionType loss_reduction_type;
76 switch (lossInfo.reduction_type)
77 {
78 case ir::train::LossReductionType::SumOverBatchSize:
79 loss_reduction_type = ::circle::LossReductionType_SumOverBatchSize;
80 break;
81 case ir::train::LossReductionType::Sum:
82 loss_reduction_type = ::circle::LossReductionType_Sum;
83 break;
84 default:
85 throw std::runtime_error("Not supported loss reduction type");
86 }
87
88 std::vector<int32_t> trainable_ops;
89 for (const auto &op : training_info->getTrainableOps())
90 {
91 trainable_ops.push_back(op.value());
92 }
93
94 const auto end = ::circle::CreateModelTrainingDirect(
95 _builder, training_info->version(), optimizer, optimizer_opt_type, optimizer_opt, lossfn,
96 lossfn_opt_type, lossfn_opt, 0, training_info->batchSize(), loss_reduction_type,
97 &trainable_ops);
98 _builder.Finish(end, ::circle::ModelTrainingIdentifier());
99
100 ::flatbuffers::Verifier v(_builder.GetBufferPointer(), _builder.GetSize());
101 bool verified = ::circle::VerifyModelTrainingBuffer(v);
102 if (not verified)
103 throw std::runtime_error{"TrainingInfo buffer is not accessible"};
104 }
105
106 uint8_t *get() const { return _builder.GetBufferPointer(); }
107 uint32_t size() const { return _builder.GetSize(); }
108
109private:
110 ::flatbuffers::FlatBufferBuilder _builder;
111};
