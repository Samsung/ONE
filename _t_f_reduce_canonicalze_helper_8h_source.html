34template <typename TFNode> bool canonicalize_reduce_node(TFNode *node)
35{
36 LOGGER(l);
37
38 INFO(l) << "TFNodeCanonicalize ReduceNode begin";
39
40 auto graph = node->graph();
41
60 // TFSqueeze had to be inserted if keep_dims() was false
61 assert(node->keep_dims());
62
63 auto axes_node = node->reduction_indices();
64 assert(axes_node != nullptr);
65
66 auto node_tensor_shape = loco::shape_get(node).template as<loco::TensorShape>();
67
68 // Canonicalization into TensorReduce is valid when reduction indices is constant
69 // TODO Support general TensorReduce case
70 std::vector<int32_t> axes_values;
71 if (auto const_axes = dynamic_cast<moco::TFConst *>(axes_node))
72 {
73 // TODO Support S64 type

75
76 for (uint32_t i = 0; i < const_axes->size<loco::DataType::S32>(); ++i)
77 {
78 int32_t axis = const_axes->at<loco::DataType::S32>(i);
79 if (axis < 0)
80 axis += node_tensor_shape.rank();
81 axes_values.push_back(axis);
82 }
83 }
84 else if (auto const_axes = dynamic_cast<loco::ConstGen *>(axes_node))
85 {
86 // TODO Support S64 type
87 assert(const_axes->dtype() == loco::DataType::S32);
88
89 for (uint32_t i = 0; i < const_axes->size<loco::DataType::S32>(); ++i)
90 {
91 int32_t axis = const_axes->at<loco::DataType::S32>(i);
92 if (axis < 0)
93 axis += node_tensor_shape.rank();
94 axes_values.push_back(axis);
95 }
96 }
97 else
98 return false;
99
100 // Create loco node to replace
101 auto reduce = graph->nodes()->template create<loco::TensorReduce>();
102
103 // replace
104 reduce->func(reduceFunc<TFNode>());
105 reduce->input(node->input());
106 for (uint32_t i = 0; i < axes_values.size(); ++i)
107 reduce->axes()->insert(axes_values.at(i));
108
109 replace(node).with(reduce);
110
111 INFO(l) << "TFNodeCanonicalize ReduceNode done";
