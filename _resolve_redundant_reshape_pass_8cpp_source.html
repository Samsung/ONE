33bool are_same_tensor_shapes(const loco::NodeShape &lhs, const loco::NodeShape &rhs)
34{
35 assert(lhs.domain() == loco::Domain::Tensor);
36 assert(rhs.domain() == loco::Domain::Tensor);
37
38 auto lts = lhs.as<loco::TensorShape>();
39 auto rts = rhs.as<loco::TensorShape>();
40
41 if (lts.rank() != rts.rank())
42 return false;
43
44 for (uint32_t axis = 0; axis < lts.rank(); ++axis)
45 {
46 assert(lts.dim(axis).known());
47 assert(rts.dim(axis).known());
48 if (lts.dim(axis).value() != rts.dim(axis).value())
49 return false;
50 }
51 return true;
52}

55bool is_redundant_reshape(loco::FixedReshape *reshape)
56{
57 auto input_shape = loco::shape_get(reshape->input());
58 auto output_shape = loco::shape_get(reshape);
59
60 // Note that FixedReshape's input and output are always tensor
61 return are_same_tensor_shapes(input_shape, output_shape);
62}

82bool ResolveRedundantReshapePass::run(loco::Graph *graph)
83{
84 bool changed = false;
85 for (auto node : loco::postorder_traversal(loco::output_nodes(graph)))
86 {
87 if (auto reshape = dynamic_cast<loco::FixedReshape *>(node))
88 {
89 if (shape_inference_done(reshape))
90 {
91 if (is_redundant_reshape(reshape))
92 {
93 replace(reshape).with(reshape->input());
94 changed = true;
95 }
96 }
97 }
98 }
99
100 return changed;
101}
