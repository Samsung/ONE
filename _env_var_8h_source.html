78 bool asBool(bool def) const
79 {

81 return def;
82 static const std::array<std::string, 5> false_list{"0", "OFF", "FALSE", "N", "NO"};
83 auto false_found = std::find(false_list.begin(), false_list.end(), _value);

85 }

93 int asInt(int def) const
