37std::unique_ptr<exec::train::TrainableFnSequence> KernelGenerator::generate(ir::OperationIndex ind)
38{
39 auto ret = std::make_unique<exec::train::TrainableFnSequence>();
40 const auto &op = _tgraph.operation(ind);
41 op.accept(*this);
42 // _return_fn must have been generated
43 if (_return_fn == nullptr)
44 {
45 throw std::runtime_error(op.name() + " op does not supported trainable kernel yet");
46 }
47
48 ret->_functions.emplace_back(std::move(_return_fn));
49
50 return ret;
51}

53void KernelGenerator::visit(const ir::train::operation::Permute &node)
54{
55 const auto output_index{node.getOutputs().at(0)};
56 const auto input_index{node.getInputs().at(0)};
57
58 // Add PermuteLayer
59 std::vector<ITensor *> output_tensors{getTensor(output_index)};
60 std::vector<ITensor *> input_tensors{getTensor(input_index)};
61
62 std::vector<ITensor *> output_back_prop_tensors;
63 std::vector<ITensor *> input_back_prop_tensors;
64
65 auto input_back_prop_tensor = getBackPropTensor(input_index);
66 auto output_back_prop_tensor = getBackPropTensor(output_index);
67 output_back_prop_tensors.emplace_back(output_back_prop_tensor);
68 input_back_prop_tensors.emplace_back(input_back_prop_tensor);
69
70 // NOTE IOTensors of graph outputs for passing data to users must be ignored in training
71 // because the buffers of those IOTensors are unnecessary and nullptr
72 bool ignore_forward_in_training = _whole_graph_outputs.contains(output_index);
73 auto fn = std::make_unique<kernel::PermuteLayer>(
74 input_tensors, output_tensors, input_back_prop_tensors, output_back_prop_tensors,
75 ignore_forward_in_training, _external_context);
76
77 _return_fn = std::move(fn);
78}
