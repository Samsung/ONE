48bool inputs_shape_ready(const luci::CircleNode *node)
49{
50 for (uint32_t arity = 0; arity < node->arity(); ++arity)
51 {
52 auto node_input = loco::must_cast<luci::CircleNode *>(node->arg(arity));
53 if (node_input->shape_status() == luci::ShapeStatus::UNDEFINED)
54 return false;
55 }
56
57 return true;
