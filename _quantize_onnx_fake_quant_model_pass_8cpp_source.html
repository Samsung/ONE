60bool QuantizeOnnxFakeQuantModelPass::run(loco::Graph *g)
61{
62 LOGGER(l);
63 INFO(l) << "QuantizeOnnxFakeQuantModelPass Start" << std::endl;
64
65 // Quantize Onnx QuantizeLinear-DequantizeLinear pattern
66 {
67 QuantizeOnnxQDQPass pass;
68 pass.run(g);
69 }
70
71 // Quantize Onnx const-DequantizeLinear pattern
72 {
73 QuantizeOnnxDequantizeLinearPass pass;
74 pass.run(g);
75 }
76
77 // Quantize const input activation
78 for (auto node : loco::active_nodes(loco::output_nodes(g)))
79 {
80 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
81
82 QuantizeConstInputActivation qcia(_ctx->default_activation_dtype);
83 circle_node->accept(&qcia);
84 }
85
86 // Quantize nodes using their predecessors' qparams
87 {
88 QuantizeWithPredecessorPass pass;
89 pass.run(g);
90 }
91
92 // Update qparam of output of special Ops
93 for (auto node : loco::active_nodes(loco::output_nodes(g)))
94 {
95 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
96
97 if (is_quantized(circle_node))
98 {
99 QuantizeSpecialActivation qsa(circle_node->dtype());
100 circle_node->accept(&qsa);
101 }
102 }
103
104 // Update output dtype
105 auto graph_outputs = g->outputs();
106 for (auto node : loco::output_nodes(g))
107 {
108 auto circle_node = loco::must_cast<luci::CircleOutput *>(node);
109 auto from = loco::must_cast<luci::CircleNode *>(circle_node->from());
110 circle_node->dtype(from->dtype());
111
112 auto graph_output = graph_outputs->at(circle_node->index());
113 graph_output->dtype(circle_node->dtype());
114 }
115
116 INFO(l) << "QuantizeOnnxFakeQuantModelPass End" << std::endl;
117 return false; // one time run
118}
