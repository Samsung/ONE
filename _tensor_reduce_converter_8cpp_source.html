42bool convert_as_mean(loco::TensorReduce *origin)
43{
44 EXO_ASSERT(origin->func() == loco::ReduceFunc::Mean, "func should be Mean for this helper");
45 EXO_ASSERT(origin->input(), "TensorReduce has no input");
46
47 auto *graph = origin->graph();
48
49 // Make reduction indicies TFLConst node
50 auto reduction = graph->nodes()->create<locoex::TFLConst>();
51 {
52 auto input_rank = loco::shape_get(origin->input()).as<loco::TensorShape>().rank();
53
54 std::vector<int32_t> red_vec;
55 for (uint32_t axis = 0; axis < input_rank; ++axis)
56 if (origin->axes()->defined(axis))
57 red_vec.push_back(static_cast<int32_t>(axis));
58
59 const loco::DataType S32 = loco::DataType::S32;
60
61 reduction->dtype(S32);
62 reduction->rank(1);
63 reduction->dim(0) = red_vec.size();
64 reduction->size<S32>(red_vec.size());
65 for (uint32_t i = 0; i < red_vec.size(); ++i)
66 reduction->at<S32>(i) = red_vec.at(i);
67 }
68
69 // Make TFLMean node to replace
70 auto mean = graph->nodes()->create<locoex::TFLMean>();
71 mean->input(origin->input());
72 mean->reduction_indices(reduction);
73 mean->keep_dims(true); // Canonical TensorReduce always keep dimensions
74
75 // replace canonical node
76 loco::replace(origin).with(mean);
77 origin->input(nullptr);
78
79 return true;
80}
