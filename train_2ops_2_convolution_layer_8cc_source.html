33template <typename Tensor>
34std::unique_ptr<Tensor> createTransposedWeights(const backend::IPortableTensor *origin_weights)
35{
36 const auto &origin_shape = origin_weights->getShape();
37 assert(origin_shape.rank() == 4);
38
39 auto transposed_info = origin_weights->get_info();
40 // OHWI to HWIO
41 auto transposed_shape =
42 ir::Shape{origin_shape.dim(1), origin_shape.dim(2), origin_shape.dim(3), origin_shape.dim(0)};
43 transposed_info.shape(transposed_shape);
44
45 return std::make_unique<Tensor>(transposed_info, origin_weights->layout());
46}

68void ConvolutionLayer::configure(
69 const IPortableTensor *input, const IPortableTensor *weights, const IPortableTensor *bias,
70 IPortableTensor *output, IPortableTensor *back_prop_input, IPortableTensor *grad_weights,
71 IPortableTensor *grad_bias, const IPortableTensor *back_prop_output, ir::PaddingType paddingType,
72 const uint32_t paddingLeft, const uint32_t paddingRight, const uint32_t paddingTop,
73 const uint32_t paddingBottom, const uint32_t strideWidth, const uint32_t strideHeight,
74 const uint32_t dilationWidthFactor, const uint32_t dilationHeightFactor,
75 const ir::Activation activation)
76{
77 const bool is_cacheable_weights = false;
78 cpu::ops::ConvolutionLayer::configure(input, weights, bias, paddingType, paddingLeft,
79 paddingRight, paddingTop, paddingBottom, strideWidth,
80 strideHeight, dilationWidthFactor, dilationHeightFactor,
81 activation, output, is_cacheable_weights);
82
83 _back_prop_input = back_prop_input;
84 _grad_weights = grad_weights;
85 _grad_bias = grad_bias;
86 _back_prop_output = back_prop_output;
87
88 if (_dilationHeightFactor != 1 || _dilationWidthFactor != 1)
89 throw std::runtime_error("train ConvolutionLayer: Unsupported dilation yet");
90
91 // TODO Optimize transposed tensors
92 _transposed_weights = createTransposedWeights<Tensor>(weights);
93 _transposed_weights->setBuffer(
94 std::make_shared<basic::Allocator>(_transposed_weights->total_size()));
95
96 _conv_back_prop_output =
97 std::make_unique<BackPropTensor>(back_prop_output->get_info(), back_prop_output->layout());
98 _conv_back_prop_output->setBuffer(
99 std::make_shared<basic::Allocator>(_conv_back_prop_output->total_size()));
100
101 _transposed_grad_weights = createTransposedWeights<GradientTensor>(weights);
102 _transposed_grad_weights->setBuffer(
103 std::make_shared<basic::Allocator>(_transposed_grad_weights->total_size()));
104
105 if (activation != ir::Activation::NONE)
106 {
107 _act_back_prop_output =
108 std::make_unique<BackPropTensor>(_back_prop_output->get_info(), _back_prop_output->layout());
109 _act_back_prop_output->setBuffer(
110 std::make_shared<basic::Allocator>(_act_back_prop_output->total_size()));
111 }

115void ConvolutionLayer::backward()
116{
117 const auto data_type = _back_prop_output->data_type();
118 assert(data_type == _input->data_type());
119 switch (data_type)
120 {
121 case OperandType::FLOAT32:
122 {
123 assert(data_type == _grad_bias->data_type());
124 backwardFloat32();
125 break;
126 }
127 default:
128 throw std::runtime_error{"train ConvolutionLayer: unsupported data type"};
129 }

132void ConvolutionLayer::backwardFloat32()
133{
134 // Calculate gradient for activation
135 const IPortableTensor *backprop_act;
136 try
137 {
138 backprop_act =
139 backpropActivation(_activation, _output, _back_prop_output, _act_back_prop_output.get());
140 }
141 catch (const std::exception &e)
142 {
143 throw std::runtime_error{"train ConvolutionLayer: " + std::string(e.what())};
144 }
145 assert(backprop_act != nullptr);
146
147 // Initialize conv params for training kernels
148 nnfw::cker::ConvParams conv_train_params;
149 conv_train_params.padding_type = getPaddingType(_paddingType);
150 conv_train_params.padding_values.width = _paddingLeft;
151 conv_train_params.padding_values.height = _paddingTop;
152 conv_train_params.stride_width = _strideWidth;
153 conv_train_params.stride_height = _strideHeight;
154 conv_train_params.dilation_width_factor = _dilationWidthFactor;
155 conv_train_params.dilation_height_factor = _dilationHeightFactor;
156
157 // Transpose weights from OHWI to HWIO
158 auto transposed_weights = _transposed_weights.get();
159 assert(transposed_weights->getShape().rank() == 4);
160 nnfw::cker::TransposeParams transpose_param;
161 transpose_param.perm_count = transposed_weights->getShape().rank();
162 transpose_param.perm[0] = 1;
163 transpose_param.perm[1] = 2;
164 transpose_param.perm[2] = 3;
165 transpose_param.perm[3] = 0;
166 nnfw::cker::Transpose(transpose_param, getShape(_kernel), getBuffer<float>(_kernel),
167 getShape(transposed_weights), getBuffer<float>(transposed_weights));
168
169 // Calculate gradient for input
170 nnfw::cker::train::ConvInputGrad(
171 conv_train_params, getShape(backprop_act), getBuffer<float>(backprop_act),
172 getShape(transposed_weights), getBuffer<float>(transposed_weights), _paddingBottom,
173 _paddingRight, getShape(_back_prop_input), getBuffer<float>(_back_prop_input));
174
175 // Calculate gradient for weights
176 auto transposed_grad_weights = _transposed_grad_weights.get();
177 assert(_grad_weights->getShape().rank() == 4);
178 assert(transposed_grad_weights->getShape().rank() == 4);
179 nnfw::cker::train::ConvFilterGrad(
180 conv_train_params, getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_input),
181 getBuffer<float>(_input), _paddingBottom, _paddingRight, getShape(transposed_grad_weights),
182 getBuffer<float>(transposed_grad_weights));
183
184 // Transpose weights'gradient from HWIO to OHWI
185 nnfw::cker::TransposeParams transpose_grad_param;
186 transpose_grad_param.perm_count = transposed_grad_weights->getShape().rank();
187 transpose_grad_param.perm[0] = 3;
188 transpose_grad_param.perm[1] = 0;
189 transpose_grad_param.perm[2] = 1;
190 transpose_grad_param.perm[3] = 2;
191 nnfw::cker::Transpose(transpose_grad_param, getShape(transposed_grad_weights),
192 getBuffer<float>(transposed_grad_weights), getShape(_grad_weights),
193 getBuffer<float>(_grad_weights));
194
195 // Calculate gradient for bias
196 if (_bias)
197 {
198 assert(_grad_bias);
199 biasGrad(backprop_act, _grad_bias);
200 }
201}
