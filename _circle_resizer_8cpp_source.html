48int entry(const int argc, char **argv)
49{
50 arser::Arser arser("circle-resizer provides capabilities to change inputs of the models");
51
52 arser.add_argument("--input_path")
53 .nargs(1)
54 .type(arser::DataType::STR)
55 .required(true)
56 .help("Path to the input model (.circle)");
57
58 arser.add_argument("--output_path")
59 .nargs(1)
60 .type(arser::DataType::STR)
61 .required(true)
62 .help("Path to the resized model (.circle)");
63
64 arser.add_argument("--input_shapes")
65 .nargs(1)
66 .type(arser::DataType::STR)
67 .required(true)
68 .help("New inputs shapes in the comma separated format. An example for 2 inputs: [1,2],[3,4].");
69
70 arser.add_argument("--version")
71 .nargs(0)
72 .required(false)
73 .default_value(false)
74 .help("Show version information and exit")
75 .exit_with(print_version);
76
77 try
78 {
79 arser.parse(argc, argv);
80
81 const auto input_path = arser.get<std::string>("--input_path");
82
83 auto circle_model = std::make_shared<CircleModel>(input_path);
84 ModelEditor resizer(circle_model);
85
86 std::cout << "Shapes of inputs before resizing:" << std::endl;
87 list_shapes(circle_model->input_shapes());
88
89 std::cout << "Shapes of outputs before resizing:" << std::endl;
90 list_shapes(circle_model->output_shapes());
91
92 const auto output_path = arser.get<std::string>("--output_path");
93 const auto new_input_shapes_str = arser.get<std::string>("--input_shapes");
94 resizer.resize_inputs(parse_shapes(new_input_shapes_str));
95
96 std::cout << "Shapes of inputs after resizing:" << std::endl;
97 list_shapes(circle_model->input_shapes());
98
99 std::cout << "Shapes of outputs after resizing:" << std::endl;
100 list_shapes(circle_model->output_shapes());
101
102 circle_model->save(output_path);
103 std::cout << "Resizing complete, the model saved to: " << output_path << std::endl;
104 }
105 catch (const std::runtime_error &err)
106 {
107 std::cerr << "Exception during resizing: " << err.what() << std::endl;
108 return EXIT_FAILURE;
109 }
110 return EXIT_SUCCESS;
111}
