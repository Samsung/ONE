22void StaticMemoryManager::base_allocate_memory(luci_interpreter::Tensor &tensor,
23 uint8_t *buffer_ptr)
24{
25 if (buffer_ptr == nullptr)
26 assert("Buffer should be allocated\n");
27
28 if (!tensor.is_allocatable())
29 {
30 return;
31 }
32
33 const auto offset = tensor.get_offset();
34 assert(offset >= 0);
35 auto tensor_ptr = buffer_ptr + offset;
36 tensor.set_data_buffer(tensor_ptr);
37}
