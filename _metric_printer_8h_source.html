30// Class to print metrics
31// How to use?
32//
33// MetricPrinter metric;
34// metric.init(first_module, second_module); // optional initialization
35//
36// for (..) // Evaluate data one by one
37// {
38// ..
39// metric.accumulate(first_result, second_result); // accumulate results
40// }
41//
42// std::cout << &metric << std::endl; // print result
43class MetricPrinter
44{
45public:
46 virtual ~MetricPrinter() = default;
47
48 // Child class can implement this function if necessary
49 // NOTE init can be skipped
50 virtual void init(const luci::Module *, const luci::Module *) {}
51
52 // Accumulate results of comparing the first and the second model's outputs
53 virtual void accumulate(const std::vector<std::shared_ptr<Tensor>> &first,
54 const std::vector<std::shared_ptr<Tensor>> &second) = 0;
55
56 // Dump the final result of the corresponding metric
57 virtual void dump(std::ostream &os) const = 0;
58};
59
60static inline std::ostream &operator<<(std::ostream &os, const MetricPrinter *m)
61{
62 m->dump(os);
63 return os;
64}

160// Ratio of matched indices between top-k results of two models (a, b).
161//
162// top-k match = intersection(top_k_idx(a), top_k_idx(b)) / k
163// mean top-k match = sum(top-k match) / num_data
164//
165// For example,
166// num_data = 2
167// first model output = [1, 2, 3], [2, 3, 1]
168// second model output = [2, 4, 6], [3, 2, 1]
169//
170// if k = 1,
171// first model top-1 index = ([2], [1])
172// second model top-1 index = ([2], [0])
173// mean top-1 accuracy = (1 + 0) / 2 = 0.5
174//
175// if k = 2,
176// first model output = [1, 2, 3], [2, 3, 1]
177// second model output = [2, 4, 6], [3, 2, 1]
178// first model top-2 index = ([2, 1], [1, 0])
179// second model top-2 index = ([2, 1], [0, 1])
180// mean top-2 accuracy = (2 + 2) / 4 = 1
181//
182// NOTE Order of elements is ignored when comparing two top-k sets.
183// NOTE If two elements have the same value and only one can be included in top-k,
184// the one with an earlier index will be included.
185class TopKMatchPrinter : public MetricPrinter
186{
187public:
188 TopKMatchPrinter(uint32_t k) : _k(k) {}
189
190public:
191 void init(const luci::Module *first, const luci::Module *second);
192
193 void accumulate(const std::vector<std::shared_ptr<Tensor>> &first,
194 const std::vector<std::shared_ptr<Tensor>> &second);
195
196 void dump(std::ostream &os) const;
197
198private:
199 void accum_topk_accuracy(uint32_t index, const std::shared_ptr<Tensor> &a,
200 const std::shared_ptr<Tensor> &b);
201
202 // Return true if the output is in the skip list (_skip_output)
203 bool in_skip_list(uint32_t output_index) const;
204
205private:
206 const uint32_t _k = 0;
207 // Store accumulated accuracy
208 std::vector<float> _intermediate;
209 std::vector<std::string> _output_names;
210 uint32_t _num_data = 0;
211 // Save index of output whose num_elements is less than k
212 std::vector<uint32_t> _skip_output;
213};
