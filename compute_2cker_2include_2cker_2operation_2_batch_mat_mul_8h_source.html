46 void prepare(const Shape &lhs_shape, const Shape &rhs_shape, bool adj_x, bool adj_y)
47 {
48 if (adj_x)
49 {
50 int32_t rank = lhs_shape.DimensionsCount();
51 _temp_lhs_shape.Resize(rank);
52
53 for (int32_t i = 0; i < rank - 2; i++)
54 {
55 _temp_lhs_shape.SetDim(i, lhs_shape.Dims(i));
56 }
57 _temp_lhs_shape.SetDim(rank - 2, lhs_shape.Dims(rank - 1));
58 _temp_lhs_shape.SetDim(rank - 1, lhs_shape.Dims(rank - 2));
59
60 _temp_lhs.resize(_temp_lhs_shape.FlatSize());
61 }
62
63 if (!adj_y)
64 {
65 int32_t rank = rhs_shape.DimensionsCount();
66 _temp_rhs_shape.Resize(rank);
67
68 for (int32_t i = 0; i < rank - 2; i++)
69 {
70 _temp_rhs_shape.SetDim(i, rhs_shape.Dims(i));
71 }
72 _temp_rhs_shape.SetDim(rank - 2, rhs_shape.Dims(rank - 1));
73 _temp_rhs_shape.SetDim(rank - 1, rhs_shape.Dims(rank - 2));
74
75 _temp_rhs.resize(_temp_rhs_shape.FlatSize());
76 }
77 }

79 void operator()(const Shape &lhs_shape, const float *lhs_data, const Shape &rhs_shape,
80 const float *rhs_data, bool adj_x, bool adj_y, const Shape &output_shape,
81 float *output_data)
82 {
83 // Assume lhs and rhs is not constant
84 // TODO Handle constant input
85
86 if (!adj_y)
87 {
88 transposeRowsCols(rhs_shape, rhs_data, _temp_rhs_shape, _temp_rhs.data());
89 }
90
91 if (adj_x)
92 {
93 transposeRowsCols(lhs_shape, lhs_data, _temp_lhs_shape, _temp_lhs.data());
94 }
95
96 Shape new_lhs_shape = adj_x ? lhs_shape : swapRowColDims(lhs_shape);
97 Shape new_rhs_shape = adj_y ? rhs_shape : swapRowColDims(rhs_shape);
98 const float *new_lhs_data = adj_x ? _temp_lhs.data() : lhs_data;
99 const float *new_rhs_data = adj_y ? rhs_data : _temp_rhs.data();
100
101 // Note we pass RHS args first, LHS args second
102 // Check accumulative dimensions of lhs and rhs of are equal
103 assert(Shape::ExtendedShape(5, new_rhs_shape).Dims(4) ==
104 Shape::ExtendedShape(5, new_lhs_shape).Dims(3));
105 reference::BatchMatMul(new_rhs_shape, new_rhs_data, new_lhs_shape, new_lhs_data, output_shape,
106 output_data);
107 }
108
109private:
110 Shape swapRowColDims(const Shape &shape)
111 {
112 Shape swapped_shape(shape);
113 const uint32_t dims = shape.DimensionsCount();
114 swapped_shape.SetDim(dims - 2, shape.Dims(dims - 1));
115 swapped_shape.SetDim(dims - 1, shape.Dims(dims - 2));
116
117 return swapped_shape;
118 }

120 void transposeRowsCols(const Shape &input_shape, const float *input_data,
121 const Shape &output_shape, float *output_data)
122 {
123 TransposeParams params;
124 int rank = input_shape.DimensionsCount();
125 params.perm_count = rank;
126 for (int i = 0; i < 2; i++)
127 {
128 params.perm[i] = i;
129 }
130 params.perm[rank - 2] = rank - 1;
131 params.perm[rank - 1] = rank - 2;
132
133 Transpose<float>(params, input_shape, input_data, output_shape, output_data);
134 }
135
136private:
137 std::vector<float> _temp_lhs;
138 Shape _temp_lhs_shape;
139 std::vector<float> _temp_rhs;
140 Shape _temp_rhs_shape;
