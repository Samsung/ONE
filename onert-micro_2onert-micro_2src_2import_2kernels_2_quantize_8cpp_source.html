38 onert_micro::execute::OMRuntimeKernel runtime_kernel;
39
40 OMStatus status = runtime_kernel.readKernel(op_index, runtime_context);
41 if (status != Ok)
42 return status;
43
44 const circle::Tensor *input = runtime_kernel.inputs[inputTensorIdx];
45 const circle::Tensor *output = runtime_kernel.outputs[outputTensorIdx];
46
47 assert(input != nullptr);
48 assert(output != nullptr);
49
50 OMRuntimeShape input_shape(input);
51 OMRuntimeShape output_shape(output);

53 // Check shapes

55 if (status != Ok)

57

60 if (status != Ok)

62

68 output_quantization->scale()->size() == 1);
69 if (status != Ok)
70 return status;
71
72 return status;
