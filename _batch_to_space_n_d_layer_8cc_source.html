29void KernelGenerator::visit(const ir::operation::BatchToSpaceND &node)
30{
31 const auto output_index{node.getOutputs().at(0)};
32 const auto input_index{node.getInputs().at(ir::operation::BatchToSpaceND::INPUT)};
33 const auto block_size_index{node.getInputs().at(ir::operation::BatchToSpaceND::BLOCK_SIZE)};
34
35 auto output_alloc = _tensor_reg->getPortableTensor(output_index);
36 auto input_alloc = _tensor_reg->getPortableTensor(input_index);
37 auto block_size_alloc = _tensor_reg->getPortableTensor(block_size_index);
38
39 auto fn = std::make_unique<ops::BatchToSpaceNDLayer>();
40
41 IPortableTensor *crops_alloc = nullptr;
42 const auto NNApiInputs = 2;
43
44 if (node.getInputs().size() != NNApiInputs)
45 {
46 const auto crops_data_index{node.getInputs().at(ir::operation::BatchToSpaceND::CROPS_DATA)};
47 crops_alloc = _tensor_reg->getPortableTensor(crops_data_index);
48 }
49
50 fn->configure(input_alloc, output_alloc, block_size_alloc, crops_alloc);
51
52 _return_fn = std::move(fn);
53}

66template <typename T> void BatchToSpaceNDLayer::batchToSpaceNDGeneric()
67{
68 const int32_t NNapiCrops[]{0, 0, 0, 0};
69 const int32_t *_crops_buffer;
70
71 if (_crops == nullptr)
72 {
73 _crops_buffer = NNapiCrops;
74 }
75 else
76 {
77 _crops_buffer = getBuffer<int32_t>(_crops);
78 }
79 nnfw::cker::BatchToSpaceND<T>(getShape(_input), getBuffer<T>(_input),
80 getBuffer<int32_t>(_block_shape), _crops_buffer, getShape(_output),
81 getBuffer<T>(_output));
82}

93void BatchToSpaceNDLayer::run()
94{
95 if (_output->data_type() == OperandType::FLOAT32)
96 {
97 batchToSpaceNDGeneric<float>();
98 }
99 else if (_output->data_type() == OperandType::QUANT_UINT8_ASYMM)
100 {
101 batchToSpaceNDGeneric<uint8_t>();
102 }
103 else
104 {
105 throw std::runtime_error{"NYI"};
106 }
107}
