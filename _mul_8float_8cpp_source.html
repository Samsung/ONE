27template <FusedActivationFunctionType Ac>
28void Mul(const float *input1_data, const Dims<4> &input1_dims, const float *input2_data,
29 const Dims<4> &input2_dims, float *output_data, const Dims<4> &output_dims)
30{
31 MatchingArraySize(input1_dims, 3, input2_dims, 3, output_dims, 3);
32 MatchingArraySize(input1_dims, 2, input2_dims, 2, output_dims, 2);
33 MatchingArraySize(input1_dims, 1, input2_dims, 1, output_dims, 1);
34 MatchingArraySize(input1_dims, 0, input2_dims, 0, output_dims, 0);
35 DCHECK(IsPackedWithoutStrides(input1_dims));
36 DCHECK(IsPackedWithoutStrides(input2_dims));
37 DCHECK(IsPackedWithoutStrides(output_dims));
38
39 int i = 0;
40 const int size = input1_dims.sizes[3] * input1_dims.strides[3];
41
42 for (; i < size; i++)
43 {
44 auto x = input1_data[i] * input2_data[i];
45 output_data[i] = ActivationFunction<Ac>(x);
46 }
47}

49// From optimized_ops.h in TensorFlow Lite
50//
51// TODO: We can implement BroadcastMul on buffers of arbitrary
52// dimensionality if the runtime code does a single loop over one dimension
53// that handles broadcasting as the base case. The code generator would then
54// generate max(D1, D2) nested for loops.
55// TODO: BroadcastMul is intentionally duplicated from
56// reference_ops.h. Once an optimized version is implemented and NdArrayDesc<T>
57// is no longer referenced in this file, move NdArrayDesc<T> from types.h to
58// reference_ops.h.
59template <FusedActivationFunctionType Ac>
60void BroadcastMul(const float *input1_data, const Dims<4> &input1_dims, const float *input2_data,
61 const Dims<4> &input2_dims, float *output_data, const Dims<4> &output_dims)
62{
63 NdArrayDesc<4> desc1;
64 NdArrayDesc<4> desc2;
65 NdArrayDescsForElementwiseBroadcast(input1_dims, input2_dims, &desc1, &desc2);
66
67 // In Tensorflow, the dimensions are canonically named (batch_number, row,
68 // col, channel), with extents (batches, height, width, depth), with the
69 // trailing dimension changing most rapidly (channels has the smallest stride,
70 // typically 1 element).
71 //
72 // In generated C code, we store arrays with the dimensions reversed. The
73 // first dimension has smallest stride.
74 //
75 // We name our variables by their Tensorflow convention, but generate C code
76 // nesting loops such that the innermost loop has the smallest stride for the
77 // best cache behavior.
78 for (int b = 0; b < ArraySize(output_dims, 3); ++b)
79 {
80 for (int y = 0; y < ArraySize(output_dims, 2); ++y)
81 {
82 for (int x = 0; x < ArraySize(output_dims, 1); ++x)
83 {
84 for (int c = 0; c < ArraySize(output_dims, 0); ++c)
85 {
86 output_data[Offset(output_dims, c, x, y, b)] =
87 ActivationFunction<Ac>(input1_data[SubscriptToIndex(desc1, c, x, y, b)] *
88 input2_data[SubscriptToIndex(desc2, c, x, y, b)]);
89 }
90 }
91 }
92 }
93}
