33void AclConstantInitializer::copyInputInitialize(const ir::Operation &node, uint32_t index)
34{
35 assert(node.getInputs().size() > index);
36
37 const auto &input_index = node.getInputs().at(index);
38 if (input_index.valid())
39 {
40 const auto &input_obj = _operands.at(input_index);

54void AclConstantInitializer::visit(const ir::operation::BatchToSpaceND &node)
55{
56 const auto &block_size_index = node.getInputs().at(ir::operation::BatchToSpaceND::BLOCK_SIZE);
57 const auto &block_size_obj = _operands.at(block_size_index);
58
59 if (block_size_obj.isConstant())
60 {
61 _init_map[block_size_index] = [](const ir::Operand &model_obj, backend::ITensor &obj) {
62 assert(model_obj.data());
63 const auto &shape = model_obj.shape();
64 const auto base = reinterpret_cast<const int32_t *>(model_obj.data()->base());
65 assert(model_obj.shape().rank() == 1);
66 obj.access([&](ITensor &tensor) {
67 for (size_t i = 0; i < shape.num_elements(); ++i)
68 {
69 const int32_t value = base[shape.num_elements() - i - 1];
70 int32_t *into = reinterpret_cast<int32_t *>(tensor.buffer() +
71 tensor.calcOffset({static_cast<int32_t>(i)}));
72 *into = value;
73 }
74 });
75 };
76 }
77}

132void AclConstantInitializer::registerCopyInitializer(const ir::OperandIndex &index,
133 const ir::Operand &obj)
134{
135 // For only CONSTANTS
136 // TODO Add to check if tensor has been allocated
137 if (!obj.isConstant())
138 return;
139
140 const auto type = obj.typeInfo().type();
141 using ir::DataType;
142
143 switch (type)
144 {
145 case DataType::FLOAT32:
146 _init_map[index] = copyInit<float>;
147 break;
148 case DataType::INT32:
149 _init_map[index] = copyInit<int32_t>;
150 break;
151 case DataType::UINT32:
152 _init_map[index] = copyInit<uint32_t>;
153 break;
154 case DataType::BOOL8:
155 case DataType::QUANT_UINT8_ASYMM:
156 _init_map[index] = copyInit<uint8_t>;
157 break;
158 case DataType::QUANT_INT8_SYMM:
159 case DataType::QUANT_INT8_ASYMM:
160 _init_map[index] = copyInit<int8_t>;
161 break;
162 case DataType::FLOAT16:
163 _init_map[index] = copyInit<float16>;
164 break;
165 case DataType::INT64:
166 _init_map[index] = copyInit<int64_t>;
167 break;
168 default:
169 throw std::runtime_error("Not supported, yet");
170 break;
171 }
172}

174void AclConstantInitializer::registerPermuteInitializer(const ir::OperandIndex &index,
175 const ir::Operand &obj)
176{
177 // For only CONSTANTS
178 // TODO Add to check if tensor has been allocated
179 if (!obj.isConstant())
180 return;
181
182 const auto type = obj.typeInfo().type();
183 using ir::DataType;
184 using namespace std::placeholders;
185
186 switch (type)
187 {
188 case DataType::FLOAT32:
189 _init_map[index] = std::bind(permuteInit<float>, _1, _2, _current_layout);
190 break;
191 case DataType::INT32:
192 _init_map[index] = std::bind(permuteInit<int32_t>, _1, _2, _current_layout);
193 break;
194 case DataType::UINT32:
195 _init_map[index] = std::bind(permuteInit<uint32_t>, _1, _2, _current_layout);
196 break;
197 case DataType::BOOL8:
198 case DataType::QUANT_UINT8_ASYMM:
199 _init_map[index] = std::bind(permuteInit<uint8_t>, _1, _2, _current_layout);
200 break;
201 case DataType::QUANT_INT8_SYMM:
202 case DataType::QUANT_INT8_ASYMM:
203 _init_map[index] = std::bind(permuteInit<int8_t>, _1, _2, _current_layout);
204 break;
205 case DataType::FLOAT16:
206 _init_map[index] = std::bind(permuteInit<float16>, _1, _2, _current_layout);
207 break;
208 case DataType::INT64:
209 _init_map[index] = std::bind(permuteInit<int64_t>, _1, _2, _current_layout);
210 break;
211 default:
212 throw std::runtime_error("Not supported, yet");
213 break;
214 }
