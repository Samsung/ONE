51bool BulkPipelineModel::prepare()
52{
53 if (_prepared.load())
54 {
55 return true;
56 }
57
58 try
59 {
60 if (_ownership == BufferOwnership::OWNER)
61 {
62 openDevice();
63 allocateBuffers();
64 fillBuffers();
65 markBufferReady();
66 registerModel();
67 }
68
69 _prepared = true;
70 return true;
71 }
72 catch (const std::exception &e)
73 {
74 std::cerr << "Failed to prepare model " << _model_path << ": " << e.what() << std::endl;
75 release();
76 return false;
77 }
78}

80void BulkPipelineModel::release()
81{
82 if (!_prepared.load())
83 {
84 return;
85 }
86
87 // Cancel a asynchronous job
88 if (_async_fill_future.valid())
89 {
90 _async_fill_future.wait();
91 }
92
93 if (_ownership == BufferOwnership::OWNER)
94 {
95 unregisterModel();
96 closeDevice();
97 }
98
99 if (_fp)
100 {
101 fclose(_fp);
102 _fp = nullptr;
103 }
104
105 _program_buffer.reset();
106 _weight_buffer.reset();
107 _meta.reset();
108 _meta_size = 0;
109 _model_id = 0;
110
111 _prepared = false;
112}

114void BulkPipelineModel::run(const std::vector<const IPortableTensor *> &inputs,
115 std::vector<IPortableTensor *> &outputs)
116{
117 if (!_prepared.load())
118 {
119 throw std::runtime_error("Model is not prepared: " + _model_path);
120 }
121
122 if (!_meta)
123 {
124 throw std::runtime_error("Model metadata is not loaded: " + _model_path);
125 }
126
127 // Prepare input buffers
128 input_buffers input;
129 input.num_buffers = _meta->input_seg_num;
130 for (uint32_t i = 0; i < input.num_buffers; i++)
131 {
132 uint32_t idx = _meta->input_seg_idx[i];
133 input.bufs[i].addr = inputs[i]->buffer();
134 input.bufs[i].type = BUFFER_MAPPED;
135 input.bufs[i].size = _meta->segment_size[idx];
136 }
137
138 // Prepare output buffers
139 output_buffers output;
140 output.num_buffers = _meta->output_seg_num;
141 for (uint32_t i = 0; i < output.num_buffers; i++)
142 {
143 uint32_t idx = _meta->output_seg_idx[i];
144 output.bufs[i].addr = outputs[i]->buffer();
145 output.bufs[i].type = BUFFER_MAPPED;
146 output.bufs[i].size = _meta->segment_size[idx];
147 }
148
149 // Execute the model
150 int ret = runNPU_model(_dev, _model_id, NPU_INFER_BLOCKING, &input, &output, nullptr, nullptr);
151 if (ret < 0)
152 {
153 throw std::runtime_error("runNPU_model() failed for " + _model_path +
154 ", ret: " + std::to_string(ret));
155 }
156}

158void BulkPipelineModel::shareBuffersFrom(const BulkPipelineModel &owner)
159{
160 if (_ownership == BufferOwnership::OWNER)
161 {
162 throw std::runtime_error("Cannot share buffers with owner model: " + _model_path);
163 }
164
165 if (!owner.isPrepared())
166 {
167 throw std::runtime_error("Owner model is not prepared: " + owner.modelPath());
168 }
169
170 // Sharing the buffers
171 _program_buffer = owner._program_buffer;
172 _weight_buffer = owner._weight_buffer;
173
174 // Sharing the device and model id
175 _dev = owner.device();
176 _model_id = owner.modelId();
177}

184void BulkPipelineModel::waitForBufferReady()
185{
186 std::unique_lock<std::mutex> lock(_buffer_mutex);
187 _buffer_cv.wait(lock, [this] { return _buffer_ready.load() || _buffer_error.load(); });
188
189 if (_buffer_error.load())
190 {
191 throw std::runtime_error("Buffer fill failed");
192 }
193}

195void BulkPipelineModel::markBufferReady()
196{
197 {
198 std::lock_guard<std::mutex> lock(_buffer_mutex);
199 _buffer_ready = true;
200 }
201 _buffer_cv.notify_all();
202}

204void BulkPipelineModel::startAsyncBufferFill()
205{
206 _buffer_ready = false;
207 _buffer_error = false;
208 _async_fill_future = std::async(std::launch::async, [this] {
209 try
210 {
211 fillBuffers();
212 markBufferReady();
213 }
214 catch (const std::exception &e)
215 {
216 {
217 std::lock_guard<std::mutex> lock(_buffer_mutex);
218 _buffer_ready = false;
219 _buffer_error = true;
220 }
221 _buffer_cv.notify_all();
222 std::cerr << "Failed to fill buffers asynchronously: " << e.what() << std::endl;
223 }
224 });
225}

227bool BulkPipelineModel::loadMetadata()
228{
229 _fp = fopen(_model_path.c_str(), "rb");
230 if (!_fp)
231 {
232 throw std::runtime_error("Failed to open model file: " + _model_path);
233 }
234
235 _meta = std::make_unique<npubin_meta>();
236 if (fread(_meta.get(), NPUBIN_META_SIZE, 1, _fp) != 1)
237 {
238 throw std::runtime_error("Failed to read metadata from: " + _model_path);
239 }
240
241 _meta_size = _meta->extended_metasize ? sizeof(npubin_meta) + _meta->extended_metasize
242 : NPUBIN_META_TOTAL_SIZE(_meta->magiccode);
243
244 return true;
245}

247void BulkPipelineModel::allocateBuffers()
248{
249 if (_ownership != BufferOwnership::OWNER)
250 {
251 throw std::runtime_error("Not allowed to allocate buffers for non-owner model: " + _model_path);
252 }
253
254 if (!_meta)
255 {
256 throw std::runtime_error("Metadata not loaded for: " + _model_path);
257 }
258
259 _program_buffer =
260 std::make_shared<BulkPipelineBuffer>(BulkPipelineBuffer::BufferType::DMABUF_CONT,
261 static_cast<size_t>(_meta->program_size), _device_id);
262
263 _weight_buffer =
264 std::make_shared<BulkPipelineBuffer>(BulkPipelineBuffer::BufferType::DMABUF_IOMMU,
265 static_cast<size_t>(_meta->weight_size), _device_id);
266
267 _program_buffer->allocate();
268 if (_meta->weight_size > 0)
269 {
270 _weight_buffer->allocate();
271 }
272}
273
274void BulkPipelineModel::fillBuffers()
275{
276 if (!_fp || !_program_buffer || !_weight_buffer)
277 {
278 throw std::runtime_error("Buffers not properly initialized for: " + _model_path);
279 }
280
281 // Fill program buffer
282 _program_buffer->fillFromFile(_fp, _meta_size);
283
284 // Fill weight buffer
285 if (_weight_buffer->size() > 0)
286 {
287 _weight_buffer->fillFromFile(_fp, _meta_size + _meta->program_size);
288 }
289}

291void BulkPipelineModel::registerModel()
292{
293 if (!_dev || !_program_buffer || !_weight_buffer)
294 {
295 throw std::runtime_error("Device or buffers not ready for: " + _model_path);
296 }
297
298 generic_buffer modelfile;
299 modelfile.type = BUFFER_FILE;
300 modelfile.filepath = _model_path.c_str();
301 modelfile.size = _meta->size;
302
303 int ret = registerNPUmodel_ext(_dev, &modelfile, _program_buffer->getGenericBuffer(),
304 _weight_buffer->getGenericBuffer(), &_model_id);
305 if (ret < 0)
306 {
307 throw std::runtime_error("Failed to register model: " + _model_path +
308 ", ret: " + std::to_string(ret));
309 }
310}
311
312void BulkPipelineModel::unregisterModel()
313{
314 if (_dev && _model_id > 0)
315 {
316 int ret = unregisterNPUmodel(_dev, _model_id);
317 if (ret < 0)
318 {
319 std::cerr << "Failed to unregister model: " << _model_path << ", ret: " << ret << std::endl;
320 }
321 _model_id = 0;
322 }
