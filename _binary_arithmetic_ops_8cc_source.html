40void prepare(ExecEnv *env, const ir::Operation &node)
41{
42 const auto &arithmetic_node =
43 nnfw::misc::polymorphic_downcast<const ir::operation::BinaryArithmetic &>(node);
44
45 const auto lhs_index = node.getInputs().at(arithmetic_node.LHS);
46 const auto rhs_index = node.getInputs().at(arithmetic_node.RHS);
47 const auto out_index = node.getOutputs().at(0);
48
49 const auto lhs_tensor = env->tensorAt(lhs_index);
50 const auto rhs_tensor = env->tensorAt(rhs_index);
51
52 // Check shape and type lhs is same with rhs
53 // TODO Util function to compare TensorInfo
54 if (lhs_tensor->data_type() != rhs_tensor->data_type())
55 {
56 throw std::runtime_error{"Interp(" + arithmetic_node.name() + "): Different input types"};
57 }
58
59 bool try_broadcast = (lhs_tensor->tensorInfo().shape() != rhs_tensor->tensorInfo().shape());
60 if (try_broadcast)
61 {
62 bool success = true;
63 auto out_shape = calcBroadcastShape(lhs_tensor->tensorInfo().shape(),
64 rhs_tensor->tensorInfo().shape(), success);
65 if (!success)
66 {
67 throw std::runtime_error{"Interp(" + arithmetic_node.name() + "): Fail to brodcasting"};
68 }
69
70 auto output_info =
71 ir::OperandInfo::createStaticInfo(out_shape, lhs_tensor->tensorInfo().typeInfo());
72 // We can handle already allocated (ex. model output)
73 env->allocateIfNeeded(out_index, output_info);
74 }
75 else
76 {
77 // Output's shape and type is same with input
78 auto output_info = lhs_tensor->tensorInfo();
79 // We can handle already allocated (ex. model output)
80 env->allocateIfNeeded(out_index, output_info);
81 }
82
83 auto out_tensor = env->tensorAt(out_index);
84 // Check shape and type lhs is same with output
85 // TODO Util function to compare TensorInfo
86 if (lhs_tensor->data_type() != out_tensor->data_type())
87 {
88 throw std::runtime_error{"Interp(" + arithmetic_node.name() + "): Invalid output type"};
89 }
90}

105template <typename raw_type, OpType op_type>
106void invoke(const ITensor *lhs_tensor, const ITensor *rhs_tensor, const ITensor *out_tensor,
107 const ir::operation::BinaryArithmetic::Param &param)
108{
109 const auto lhs_buffer = lhs_tensor->bufferRO();
110 const auto rhs_buffer = rhs_tensor->bufferRO();
111 auto out_buffer = out_tensor->buffer();
112
113 nnfw::cker::BinaryArithmeticOpParam cker_param;
114 raw_type activation_min, activation_max;
115 calculateActivationRange(param.activation, &activation_min, &activation_max);
116 setActivationParams(activation_min, activation_max, &cker_param);
117 const raw_type *lhs_ptr = reinterpret_cast<const raw_type *>(lhs_buffer);
118 const raw_type *rhs_ptr = reinterpret_cast<const raw_type *>(rhs_buffer);
119 raw_type *out_ptr = reinterpret_cast<raw_type *>(out_buffer);
120
121 const auto cker_op_type =
122 (op_type == OpType::ADD) ? nnfw::cker::BinaryArithmeticOpType::ADD
123 : ((op_type == OpType::SUB) ? nnfw::cker::BinaryArithmeticOpType::SUB
124 : nnfw::cker::BinaryArithmeticOpType::MUL);
125
126 const bool need_broadcast =
127 nnfw::cker::ProcessBroadcastShapes(convertShape(lhs_tensor->tensorInfo().shape()),
128 convertShape(rhs_tensor->tensorInfo().shape()), &cker_param);
129
130 if (need_broadcast)
131 {
132 const auto lhs_shape = convertShape(lhs_tensor->tensorInfo().shape());
133 const auto rhs_shape = convertShape(rhs_tensor->tensorInfo().shape());
134 const auto out_shape = convertShape(out_tensor->tensorInfo().shape());
135 nnfw::cker::BroadcastBinaryArithmeticOp<cker_op_type>(cker_param, lhs_shape, lhs_ptr, rhs_shape,
136 rhs_ptr, out_shape, out_ptr);
137 return;
138 }
139
140 const auto lhs_shape = convertShape(lhs_tensor->tensorInfo().shape());
141 const auto rhs_shape = convertShape(rhs_tensor->tensorInfo().shape());
142 const auto out_shape = convertShape(out_tensor->tensorInfo().shape());
143 nnfw::cker::BinaryArithmeticOp<cker_op_type>(cker_param, lhs_shape, lhs_ptr, rhs_shape, rhs_ptr,
144 out_shape, out_ptr);
145}

147template <OpType op_type>
148void invokeBinaryArithmetic(const ExecEnv *env, const ir::operation::BinaryArithmetic &node)
149{
150 const auto lhs_index = node.getInputs().at(node.LHS);
151 const auto rhs_index = node.getInputs().at(node.RHS);
152 const auto out_index = node.getOutputs().at(0);
153 const auto lhs_tensor = env->tensorAt(lhs_index);
154 const auto rhs_tensor = env->tensorAt(rhs_index);
155 const auto out_tensor = env->tensorAt(out_index);
156 const auto data_type = lhs_tensor->data_type();
157
158 if (data_type == ir::DataType::INT32)
159 {
160 invoke<int32_t, op_type>(lhs_tensor, rhs_tensor, out_tensor, node.param());
161 }
162 else if (data_type == ir::DataType::FLOAT32)
163 {
164 invoke<float, op_type>(lhs_tensor, rhs_tensor, out_tensor, node.param());
165 }
166 else
167 {
168 throw std::runtime_error{"NYI: Unsupported data type"};
169 }
170}

172void invokeBinaryArithmeticOps(const ExecEnv *env, const ir::Operation &node)
173{
174 const auto &arithmetic_node =
175 nnfw::misc::polymorphic_downcast<const ir::operation::BinaryArithmetic &>(node);
176
177 switch (arithmetic_node.param().arithmetic_type)
178 {
179 case ir::operation::BinaryArithmetic::ArithmeticType::ADD:
180 invokeBinaryArithmetic<OpType::ADD>(env, arithmetic_node);
181 break;
182 case ir::operation::BinaryArithmetic::ArithmeticType::SUB:
183 invokeBinaryArithmetic<OpType::SUB>(env, arithmetic_node);
184 break;
185 case ir::operation::BinaryArithmetic::ArithmeticType::MUL:
186 invokeBinaryArithmetic<OpType::MUL>(env, arithmetic_node);
187 break;
188 default:
189 throw std::runtime_error{"Interp(BinaryArithmetic): NYI unsupported operation " +
190 arithmetic_node.name()};
191 break;
192 }
193}
