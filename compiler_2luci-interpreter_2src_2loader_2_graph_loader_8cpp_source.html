29template <typename NodeT> Shape getNodeShape(const NodeT *node)
30{
31 Shape shape(node->rank());
32 for (uint32_t i = 0; i < node->rank(); ++i)
33 {
34 shape.dim(i) = node->dim(i).value();
35 }
36 return shape;
37}

39template <DataType DT> const void *getNodeDataImpl(const luci::CircleConst *node, size_t *data_size)
40{
41 const size_t element_size = getDataTypeSize(DT);
42 const int32_t num_elements = node->size<DT>();
43
44 *data_size = num_elements * element_size;
45 if (*data_size > 0)
46 {
47 // FIXME There is no good way to get the pointer to the data currently.
48 return &node->at<DT>(0);
49 }
50 return nullptr;
51}

53const void *getNodeData(const luci::CircleConst *node, size_t *data_size)
54{
55 switch (node->dtype())
56 {
57 case DataType::U8:
58 return getNodeDataImpl<DataType::U8>(node, data_size);
59 case DataType::FLOAT32:
60 return getNodeDataImpl<DataType::FLOAT32>(node, data_size);
61 case DataType::S8:
62 return getNodeDataImpl<DataType::S8>(node, data_size);
63 case DataType::S16:
64 return getNodeDataImpl<DataType::S16>(node, data_size);
65 case DataType::S32:
66 return getNodeDataImpl<DataType::S32>(node, data_size);
67 case DataType::S64:
68 return getNodeDataImpl<DataType::S64>(node, data_size);
69 case DataType::BOOL:
70 return getNodeDataImpl<DataType::BOOL>(node, data_size);
71 default:
72 throw std::runtime_error("Unsupported type.");
73 }
74}

76const void *getNodeData(const luci::CircleCustom *node, size_t *data_size)
77{
78 if (node->custom_code() != "CircleReferencingConst")
79 return nullptr;
80
81 // helper struct which describes data loaded to custom_options of CircleReferencingConst node
82 // TODO move this struct to header
83 struct ConstDataReference
84 {
85 const uint8_t *data = nullptr;
86 uint32_t size = 0;
87 };
88
89 const auto &custom_options = node->custom_options();
90 const auto &const_data_ref = *reinterpret_cast<const ConstDataReference *>(custom_options.data());
91
92 *data_size = const_data_ref.size;
93 return const_data_ref.data;
94}

96bool isExecutableNode(const luci::CircleNode *node)
97{
98 switch (node->opcode())
99 {
100 // These nodes denote inputs / outputs of a graph.
101 case luci::CircleOpcode::CIRCLECONST:
102 case luci::CircleOpcode::CIRCLEINPUT:
103 case luci::CircleOpcode::CIRCLEOUTPUT:
104 case luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE:
105 // The following nodes denote outputs of multiple-output nodes.
106 case luci::CircleOpcode::CIRCLEBIDIRECTIONAL_SEQUENCE_LSTM_OUT:
107 case luci::CircleOpcode::CIRCLECUSTOMOUT:
108 case luci::CircleOpcode::CIRCLEIFOUT:
109 case luci::CircleOpcode::CIRCLENONMAXSUPPRESSIONV4OUT:
110 case luci::CircleOpcode::CIRCLENONMAXSUPPRESSIONV5OUT:
111 case luci::CircleOpcode::CIRCLESPLITOUT:
112 case luci::CircleOpcode::CIRCLESPLITVOUT:
113 case luci::CircleOpcode::CIRCLETOPKV2OUT:
114 case luci::CircleOpcode::CIRCLEUNIQUEOUT:
115 case luci::CircleOpcode::CIRCLEUNPACKOUT:
116 case luci::CircleOpcode::CIRCLEVARIABLE:
117 case luci::CircleOpcode::CIRCLEWHILEOUT:
118 return false;
119 // Custom nodes may be executable and non-executable
120 case luci::CircleOpcode::CUSTOM:
121 {
122 auto const custom_node = loco::must_cast<const luci::CircleCustom *>(node);
123
124 // TODO handle more non-executable Custom ops here
125 if (custom_node->custom_code() == "CircleReferencingConst")
126 return false;
127
128 return true;
129 }
130 default:
131 return true;
132 }
133}

135bool isTensorProducingNode(const luci::CircleNode *node)
136{
137 switch (node->opcode())
138 {
139 // Output nodes do not produce tensors.
140 case luci::CircleOpcode::CIRCLEOUTPUT:
141 // The following nodes are multiple-output nodes. They do not produce tensors, the tensors
142 // are produced by the corresponding *Out nodes instead.
143 case luci::CircleOpcode::BIDIRECTIONAL_SEQUENCE_LSTM:
144 case luci::CircleOpcode::CUSTOM:
145 case luci::CircleOpcode::IF:
146 case luci::CircleOpcode::NON_MAX_SUPPRESSION_V4:
147 case luci::CircleOpcode::NON_MAX_SUPPRESSION_V5:
148 case luci::CircleOpcode::SPLIT:
149 case luci::CircleOpcode::SPLIT_V:
150 case luci::CircleOpcode::TOPK_V2:
151 case luci::CircleOpcode::UNIQUE:
152 case luci::CircleOpcode::UNPACK:
153 case luci::CircleOpcode::WHILE:
154 return false;
155 default:
156 return true;
157 }
158}

183void GraphLoader::loadTensors()
184{
185 for (uint32_t i = 0; i < _graph->nodes()->size(); ++i)
186 {
187 const auto *node = loco::must_cast<const luci::CircleNode *>(_graph->nodes()->at(i));
188
189 if (node->opcode() == luci::CircleOpcode::CUSTOM && !isSupportedCustomNode(node))
190 {
191 const auto *cnode = loco::must_cast<const luci::CircleCustom *>(node);
192 throw std::runtime_error("Unsupported Custom operator. " + cnode->custom_code() + " in " +
193 node->name());
194 }
195
196 if (!isTensorProducingNode(node))
197 continue;
198
199 // Only Input, Const, Custom and Variable nodes have shapes. Shapes of intermediate tensors will
200 // be inferred.
201 Shape shape{};
202 switch (node->opcode())
203 {
204 case luci::CircleOpcode::CIRCLECONST:
205 case luci::CircleOpcode::CIRCLECUSTOMOUT:
206 case luci::CircleOpcode::CIRCLEINPUT:
207 case luci::CircleOpcode::CIRCLEVARIABLE:
208 shape = getNodeShape(node);
209 break;
210 default:
211 break;
212 }
213
214 AffineQuantization quantization;
215 if (node->quantparam() != nullptr)
216 {
217 const luci::CircleQuantParam *params = node->quantparam();
218 assert(params->scale.size() == params->zerop.size());
219 quantization.scale.assign(params->scale.cbegin(), params->scale.cend());
220 quantization.zero_point.assign(params->zerop.cbegin(), params->zerop.cend());
221 quantization.quantized_dimension = params->quantized_dimension;
222 }
223
224 auto tensor = std::make_unique<Tensor>(node->dtype(), std::move(shape), std::move(quantization),
225 node->name());
226
227 // If node has execution plan then read memory offsets for nodes
228 // from the beginning of shared memory buffer. Used in Static Memory Manager.
229 if (luci::has_execution_plan(node))
230 {
231 auto execution_plan = luci::get_execution_plan(node);
232 assert(!execution_plan.offsets().empty());
233 tensor->set_offset(execution_plan.offsets().front());
234 }
235
236 if (const auto *const_node = dynamic_cast<const luci::CircleConst *>(node))
237 {
238 size_t data_size{};
239 const void *const_data = getNodeData(const_node, &data_size);
240 if (const_data != nullptr)
241 {
242 _memory_manager->allocate_memory(*tensor);
243 tensor->writeData(const_data, data_size);
244 }
245 }
246 else if (const auto *custom_out_node = dynamic_cast<const luci::CircleCustomOut *>(node))
247 {
248 const auto *custom_node =
249 loco::must_cast<const luci::CircleCustom *>(custom_out_node->input());
250
251 if (custom_node->custom_code() == "CircleReferencingConst")
252 {
253 size_t data_size{};
254 const void *const_data = getNodeData(custom_node, &data_size);
255 if (const_data != nullptr)
256 {
257 _memory_manager->allocate_memory(*tensor);
258 tensor->writeData(const_data, data_size);
259 }
260 }
261 }
262
263 _node_to_tensor.emplace(node, tensor.get());
264 _runtime_to_ir.tensor_to_node.emplace(tensor.get(), node);
265
266 _runtime_graph->addTensor(std::move(tensor));
267 }
268}

270void GraphLoader::initInputOutputTensors() const
271{
272 auto input_nodes = loco::input_nodes(_graph);
273 std::vector<Tensor *> input_tensors(input_nodes.size());
274 for (size_t i = 0; i < input_nodes.size(); ++i)
275 {
276 input_tensors[i] = _node_to_tensor.at(input_nodes[i]);
277 _memory_manager->allocate_memory(*input_tensors[i]);
278 }
279 _runtime_graph->setInputTensors(input_tensors);
280
281 auto output_nodes = loco::output_nodes(const_cast<loco::Graph *>(_graph));
282 std::vector<Tensor *> output_tensors(output_nodes.size());
283 for (size_t i = 0; i < output_nodes.size(); ++i)
284 {
285 const auto *node = loco::must_cast<const luci::CircleOutput *>(output_nodes[i]);
286 output_tensors[i] = _node_to_tensor.at(node->from());
287 }
288 _runtime_graph->setOutputTensors(output_tensors);
289}

291void GraphLoader::loadOperators()
292{
293 KernelBuilder kernel_builder(_graph_to_runtime_graph, _node_to_tensor);
294
295 // Create kernels for executable nodes. This has to be done in execution order.
296 auto graph = const_cast<loco::Graph *>(_graph);
297
298 auto const graph_nodes = loco::all_nodes(graph);
299
300 // Checking for execution plan in node annotations.
301 bool has_execution_annotation = true;
302 auto const checking_exec_plan = [&has_execution_annotation](auto const node) {
303 const auto *circle_node = loco::must_cast<const luci::CircleNode *>(node);
304 if (!luci::has_execution_plan(circle_node))
305 has_execution_annotation = false;
306 };
307 std::for_each(begin(graph_nodes), end(graph_nodes), checking_exec_plan);
308
309 if (has_execution_annotation)
310 {
311 // Build ordered_nodes vector that stores the order of execution of graph nodes.
312 std::vector<const luci::CircleNode *> ordered_nodes(graph_nodes.size());
313
314 auto const filler = [&ordered_nodes](auto const node) {
315 const auto *circle_node = loco::must_cast<const luci::CircleNode *>(node);
316 auto const position = luci::get_execution_plan(circle_node).order_in_plan();
317 ordered_nodes.at(position) = circle_node;
318 };
319 std::for_each(begin(graph_nodes), end(graph_nodes), filler);
320
321 for (auto node : ordered_nodes)
322 {
323 if (isExecutableNode(node))
324 {
325 std::unique_ptr<Kernel> kernel = kernel_builder.build(node);
326 _runtime_to_ir.kernel_to_node.emplace(kernel.get(), node);
327 _runtime_graph->addKernel(std::move(kernel));
328 }
329 }
330 }
331 else
332 {
333 // If it is impossible to build the execution order plan,
334 // then we use the default postorder_traversal approach.
335 for (const loco::Node *loco_node : loco::postorder_traversal(loco::output_nodes(graph)))
336 {
337 const auto *node = loco::must_cast<const luci::CircleNode *>(loco_node);
338 if (isExecutableNode(node))
339 {
340 std::unique_ptr<Kernel> kernel = kernel_builder.build(node);
341 _runtime_to_ir.kernel_to_node.emplace(kernel.get(), node);
342 _runtime_graph->addKernel(std::move(kernel));
343 }
344 }
345 }
346}
