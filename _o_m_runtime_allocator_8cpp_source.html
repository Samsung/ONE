26OMStatus OMRuntimeAllocator::clearAllTensorsData(OMRuntimeContext *context,
27 OMRuntimeStorage *storage)
28{
29 auto &tensor_index_to_data = storage->getTensorIndexToData();
30
31 for (auto &cur_tensor_index_data : tensor_index_to_data)
32 {
33 uint8_t *allocated_data = cur_tensor_index_data.second;
34#ifdef OM_MEMORY_ESTIMATE
35 auto tensor_index = cur_tensor_index_data.first;
36
37 auto tensor = context->getTensorByIndex(tensor_index);
38 auto num_elements = OMRuntimeShape(tensor).flatSize();
39
40#ifndef DIS_DYN_SHAPES
41 int32_t dynamic_tensor_size = storage->getDynamicRuntimeShape(tensor_index).flatSize();
42 if (dynamic_tensor_size != 0)
43 num_elements = dynamic_tensor_size;
44#endif // DIS_DYN_SHAPES
45
46 auto tensor_size = num_elements * sizeof(OMDataType(tensor->type()));
47
48 OMMemoryManager::deallocateMemory(tensor_size, allocated_data);
49#else
50 OMMemoryManager::deallocateMemory(allocated_data);
51#endif // OM_MEMORY_ESTIMATE
52 }
53
54 return Ok;
55}

57OMStatus OMRuntimeAllocator::allocate(size_t kernel_index, OMRuntimeContext *context,
58 OMRuntimeStorage *storage)
59{
60 assert(kernel_index < _alloc_plan.size() && "Wrong kernel index");
61 if (kernel_index >= _alloc_plan.size())
62 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
63
64 const std::vector<uint16_t> &current_allocate_plan = _alloc_plan[kernel_index];
65
66 for (const uint16_t tensor_index : current_allocate_plan)
67 {
68 const circle::Tensor *tensor = context->getTensorByIndex(tensor_index);
69 const OMRuntimeShape tensor_shape(tensor);
70
71 int32_t num_elements = tensor_shape.flatSize();
72
73#ifndef DIS_DYN_SHAPES
74 int32_t dynamic_tensor_size = storage->getDynamicRuntimeShape(tensor_index).flatSize();
75 if (dynamic_tensor_size != 0)
76 num_elements = dynamic_tensor_size;
77#endif // DIS_DYN_SHAPES
78
79 assert(num_elements >= 0 && "Num elements should be greater zero");
80 if (num_elements < 0)
81 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
82 const auto casted_num_elements = static_cast<uint32_t>(num_elements);
83 const auto type_size =
84 static_cast<uint32_t>(getOMDataTypeSize(onertMicroDatatype(tensor->type())));
85 if (casted_num_elements > std::numeric_limits<uint32_t>::max() / type_size)
86 {
87 return FailedCheckCondition;
88 }
89 // allocate data
90 uint8_t *allocated_data = nullptr;
91 assert(storage->getDataByTensorIndex(&allocated_data, tensor_index) == Ok &&
92 allocated_data == nullptr && "Double allocate, memory leak");
93 OMStatus status =
94 OMMemoryManager::allocateMemory(casted_num_elements * type_size, &allocated_data);
95 if (status != Ok)
96 return status;
97
98 storage->saveDataToTensorIndex(allocated_data, tensor_index);
99 }
100
101 return Ok;
102}

105OMStatus OMRuntimeAllocator::deallocate(size_t kernel_index, OMRuntimeStorage *storage,
106 OMRuntimeContext *context)
107{
108 assert(kernel_index < _alloc_plan.size() && "Wrong kernel index");
109 if (kernel_index >= _alloc_plan.size())
110 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
111
112 const std::vector<uint16_t> &current_deallocate_plan = _dealloc_plan[kernel_index];
113
114 for (const uint16_t tensor_index : current_deallocate_plan)
115 {
116 uint8_t *allocated_data = nullptr;
117 OMStatus status = storage->getDataByTensorIndex(&allocated_data, tensor_index);
118 // To continue deallocate due to current tensor is not saved in storage
119 if (allocated_data == nullptr)
120 continue;
121 if (status != Ok)
122 return status;
123
124 auto tensor = context->getTensorByIndex(tensor_index);
125 auto num_elements = OMRuntimeShape(tensor).flatSize();
126
127#ifndef DIS_DYN_SHAPES
128 int32_t dynamic_tensor_size = storage->getDynamicRuntimeShape(tensor_index).flatSize();
129 if (dynamic_tensor_size != 0)
130 num_elements = dynamic_tensor_size;
131#endif // DIS_DYN_SHAPES
132
133 auto tensor_size = num_elements * sizeof(OMDataType(tensor->type()));
134 status = OMMemoryManager::deallocateMemory(tensor_size, allocated_data);
135 if (status != Ok)
136 return status;
137
138 status = storage->removeTensorFromTensorIndexToData(tensor_index);
139 if (status != Ok)
140 return status;
141 }
142
143 return Ok;
144}

146#endif // OM_MEMORY_ESTIMATE

148OMStatus OMRuntimeAllocator::deallocate(size_t kernel_index, OMRuntimeStorage *storage)
149{
150 assert(kernel_index < _alloc_plan.size() && "Wrong kernel index");
151 if (kernel_index >= _alloc_plan.size())
152 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
153
154 const std::vector<uint16_t> &current_deallocate_plan = _dealloc_plan[kernel_index];
155
156 for (const uint16_t tensor_index : current_deallocate_plan)
157 {
158 uint8_t *allocated_data = nullptr;
159 OMStatus status = storage->getDataByTensorIndex(&allocated_data, tensor_index);
160 assert(status == Ok); // note that status always 0
161
162 // To continue deallocate due to current tensor is not saved in storage
163 if (allocated_data == nullptr)
164 continue;
165
166 status = OMMemoryManager::deallocateMemory(allocated_data);
167 assert(status == Ok); // note that status always 0
168
169 status = storage->removeTensorFromTensorIndexToData(tensor_index);
170 if (status != Ok)
171 return status;
172 }
173
174 return Ok;
175}

177OMStatus OMRuntimeAllocator::allocateGraphInputs(OMRuntimeContext *context,
178 OMRuntimeStorage *storage)
179{
180 OMStatus status = Ok;
181 const auto &graph_inputs = context->getCircleInputs();
182
183 for (uint32_t i = 0; i < graph_inputs->size(); ++i)
184 {
185 auto tensor_index = graph_inputs->operator[](i);
186 const circle::Tensor *tensor = context->getTensorByIndex(tensor_index);
187 const OMRuntimeShape tensor_shape(tensor);
188
189 int32_t num_elements = tensor_shape.flatSize();
190 assert(num_elements >= 0 && "Num elements should be positive");
191 if (num_elements < 0)
192 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
193 const auto casted_num_elements = static_cast<uint32_t>(num_elements);
194 const auto type_size =
195 static_cast<uint32_t>(getOMDataTypeSize(onertMicroDatatype(tensor->type())));
196
197 uint8_t *allocated_data = nullptr;
198 // First clear if already allocated
199 status = storage->getDataByTensorIndex(&allocated_data, tensor_index);
200
201#ifdef OM_MEMORY_ESTIMATE
202#ifndef DIS_DYN_SHAPES
203 int32_t dynamic_tensor_size = storage->getDynamicRuntimeShape(tensor_index).flatSize();
204 if (dynamic_tensor_size != 0)
205 num_elements = dynamic_tensor_size;
206#endif // DIS_DYN_SHAPES
207
208 auto tensor_size = num_elements * sizeof(OMDataType(tensor->type()));
209 OMMemoryManager::deallocateMemory(tensor_size, allocated_data);
210#else
211 OMMemoryManager::deallocateMemory(allocated_data);
212#endif // OM_MEMORY_ESTIMATE
213
214 // Then Allocate
215 status = OMMemoryManager::allocateMemory(casted_num_elements * type_size, &allocated_data);
216 if (status != Ok)
217 return status;
218
219 storage->saveDataToTensorIndex(allocated_data, tensor_index);
220 }
221
222 return status;
223}
