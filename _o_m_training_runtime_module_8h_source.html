31/*
32 * Class to handle all training process
33 */
34class OMTrainingRuntimeModule : public OMRuntimeModule
35{
36private:
37 // Store backwards graphs
38 std::vector<OMRuntimeGraph> _backward_graphs;
39 // Store training handler
40 train::OMTrainingHandler _training_handler;
41
42public:
43 OMTrainingRuntimeModule() = default;
44 OMTrainingRuntimeModule(const OMTrainingRuntimeModule &) = delete;
45 OMTrainingRuntimeModule(OMTrainingRuntimeModule &&) = delete;
46 OMTrainingRuntimeModule &operator=(const OMTrainingRuntimeModule &) = delete;
47 OMTrainingRuntimeModule &&operator=(const OMTrainingRuntimeModule &&) = delete;
48 ~OMTrainingRuntimeModule() = default;
49
50 // Import train model with current config settings
51 // Note: model ptr should be non-const to save result
52 OMStatus importTrainModel(char *model_ptr, const OMConfig &config);
53
54 // Train single step: run forward graph (with data which was set in SetInput) ->
55 // -> calculate error (with target data which was set in SetTarget) ->
56 // -> run backward graph -> update optimizer state -> after batch_size steps update weights
57 // Warning: before using trainSingleStep call: 1) importTrainModel; 2) setInput; 3) setTarget
58 OMStatus trainSingleStep(OMConfig &config);
59
60 // Reset all states and data saved into OMTrainingInterpreter (trained weights will not be reset)
61 OMStatus reset();
62
63 // Calculate and save metric into metric_val: run forward graph -> calculate metric
64 // Note: calculation will be done on test_size number of test samples
65 // Warning: before using evaluateMetric call: 1) importTrainModel; 2) setInput; 3) setTarget
66 // Note: number of the samples in data should be equal to the test_size
67 // Warning: 1) assume that all metric_val for all OMMetrics types actually are float values.
68 // 2) metric_val should be initialized with some value before calling this method due to
69 // after calculation for current batch_num (the sequence number of the current sample)
70 // this value is added to metric_val
71 OMStatus evaluateMetric(OMMetrics metric, void *metric_val, uint32_t test_size);
72
73 // Set input data for input with input_index
74 // Note: number of the samples in data should be equal to the batch_size in config structure
75 void setInputData(uint8_t *data, uint32_t input_index)
76 {
77 _training_handler.setInputData(data, input_index);
78 }
79 // Set target data for output with target_index
80 // Note: number of the samples in data should be equal to the batch_size in config structure
81 void setTargetData(uint8_t *data, uint32_t target_index)
82 {
83 _training_handler.setTargetData(data, target_index);
84 }
85
86 // Create and save checkpoint data into data_buffer vector
87 // To check checkpoint file format please see https://github.com/Samsung/ONE/discussions/13037
88 OMStatus createCheckpointFile(const OMConfig &config, std::vector<char> &data_buffer);
89
90 // Load checkpoints data and save it in model data and in config
91 // To check checkpoint file format please see https://github.com/Samsung/ONE/discussions/13037
92 OMStatus loadCheckpointData(OMConfig &config, const char *data);
93};
