143class ANNConv2DAppender final : public ANNOpAppender
144{
145public:
146 void session(const enco::SessionID &sess) { _sess = sess; }
147
148 void pad(const coco::Padding2D *pad) { _pad = *pad; }
149 void stride(const coco::Stride2D *stride) { _stride = *stride; }
150
151 void ifm(coco::FeatureObject *ifm) { _ifm = ifm; }
152 void ker(coco::KernelObject *ker) { _ker = ker; }
153 // Q: Should we take a bias as a feature object?
154 // NOTE This interface is subject to change
155 void bias(coco::FeatureObject *bias) { _bias = bias; }
156 void ofm(coco::FeatureObject *ofm) { _ofm = ofm; }
157
158public:
159 void append(ANNBinder *binder) const override
160 {
161 auto data = enco::data(_sess);
162
163 auto ifm = binder->addOperand<float>(_ifm);
164 auto ker = binder->addOperand<float>(_ker);
165
166 // Fill kernel data
167 {
168 auto ker_bag = _ker->bag();
169 auto ker_weight = data->f32()->weight(ker_bag);
170
171 assert(ker_weight.data() != nullptr);
172
173 binder->setOperand(ker, ker_weight.data(), ker_weight.data() + ker_weight.size());
174 }
175
176 // Conv2D in coco IR has no bias, but bias is mandatory in Android NN API
177 auto bias = binder->addOperand<float>(nncc::core::ADT::tensor::Shape{_ker->shape().count()});
178
179 // Fill bias data
180 if (_bias == nullptr)
181 {
182 // Use a fresh empty bias if "bias" is not specified
183 auto length = _ker->shape().count();
184
185 std::vector<float> values;
186 values.resize(length, 0.0f);
187
188 binder->setOperand(bias, values.begin(), values.end());
189 }
190 else
191 {
192 // Use specified "bias"
193 auto bias_bag = _bias->bag();
194 auto bias_weight = data->f32()->weight(bias_bag);
195
196 assert(bias_weight.data() != nullptr);
197 assert(bias_weight.size() == _ker->shape().count());
198
199 binder->setOperand(bias, bias_weight.data(), bias_weight.data() + bias_weight.size());
200 }
201
202 auto left = binder->addOperand<int32_t>();
203 binder->setOperand(left, _pad.left());
204 auto right = binder->addOperand<int32_t>();
205 binder->setOperand(right, _pad.right());
206 auto top = binder->addOperand<int32_t>();
207 binder->setOperand(top, _pad.top());
208 auto bottom = binder->addOperand<int32_t>();
209 binder->setOperand(bottom, _pad.bottom());
210 auto hstride = binder->addOperand<int32_t>();
211 binder->setOperand(hstride, _stride.horizontal());
212 auto vstride = binder->addOperand<int32_t>();
213 binder->setOperand(vstride, _stride.vertical());
214 auto fuse = binder->addOperand<int32_t>();
215 binder->setOperand(fuse, 0);
216
217 auto ofm = binder->addOperand<float>(_ofm);
218
219 binder->addOperation(ann::Operation::Code::CONV_2D,
220 {ifm, ker, bias, left, right, top, bottom, hstride, vstride, fuse}, {ofm});
221 }
222
223private:
224 enco::SessionID _sess;
225
226private:
227 coco::Padding2D _pad;
228 coco::Stride2D _stride;
229
230private:
231 coco::FeatureObject *_ifm = nullptr;
232 coco::KernelObject *_ker = nullptr;
233 coco::FeatureObject *_bias = nullptr;
234 coco::FeatureObject *_ofm = nullptr;
235};

237class ANNDepthwiseConv2DAppender final : public ANNOpAppender
238{
239public:
240 void session(const enco::SessionID &sess) { _sess = sess; }
241
242 void multiplier(const uint32_t &multiplier) { _multiplier = multiplier; }
243 void pad(const coco::Padding2D *pad) { _pad = *pad; }
244 void stride(const coco::Stride2D *stride) { _stride = *stride; }
245
246 void ifm(coco::FeatureObject *ifm) { _ifm = ifm; }
247 void ker(coco::KernelObject *ker) { _ker = ker; }
248 void ofm(coco::FeatureObject *ofm) { _ofm = ofm; }
249
250public:
251 void append(ANNBinder *binder) const override
252 {
253 using namespace nncc::core::ADT;
254
255 auto data = enco::data(_sess);
256
257 const uint32_t ker_N = _ker->shape().count();
258 const uint32_t ker_H = _ker->shape().height();
259 const uint32_t ker_W = _ker->shape().width();
260
261 assert(ker_N % _multiplier == 0);
262 const uint32_t group = ker_N / _multiplier;
263
264 auto ifm = binder->addOperand<float>(_ifm);
265 auto ker = binder->addOperand<float>(tensor::Shape{1, ker_H, ker_W, ker_N});
266
267 // Fill kernel data
268 {
269 auto obj = _ker;
270 auto shape = obj->shape();
271
272 auto ovl = data->f32()->read(obj);
273 assert(ovl != nullptr);
274
275 // Flatten?
276 std::vector<float> values;
277
288 for (uint32_t row = 0; row < shape.height(); ++row)
289 {
290 for (uint32_t col = 0; col < shape.width(); ++col)
291 {
292 for (uint32_t g = 0; g < group; ++g)
293 {
294 for (uint32_t m = 0; m < _multiplier; ++m)
295 {
296 const auto value = ovl->at(g * _multiplier + m, 0, row, col);
297 values.emplace_back(value);
298 }
299 }
300 }
301 }
302
303 assert(values.size() == nncc::core::ADT::kernel::num_elements(shape));
304 binder->setOperand(ker, values.begin(), values.end());
305 }
306
307 // Conv2D in coco IR has no bias, but bias is mandatory in Android NN API
308 auto bias = binder->addOperand<float>(nncc::core::ADT::tensor::Shape{_ker->shape().count()});
309
310 // Fill bias data
311 {
312 auto length = _ker->shape().count();
313
314 std::vector<float> values;
315 values.resize(length, 0.0f);
316
317 binder->setOperand(bias, values.begin(), values.end());
318 }
319
320 auto left = binder->addOperand<int32_t>();
321 binder->setOperand(left, _pad.left());
322 auto right = binder->addOperand<int32_t>();
323 binder->setOperand(right, _pad.right());
324 auto top = binder->addOperand<int32_t>();
325 binder->setOperand(top, _pad.top());
326 auto bottom = binder->addOperand<int32_t>();
327 binder->setOperand(bottom, _pad.bottom());
328 auto hstride = binder->addOperand<int32_t>();
329 binder->setOperand(hstride, _stride.horizontal());
330 auto vstride = binder->addOperand<int32_t>();
331 binder->setOperand(vstride, _stride.vertical());
332 auto multiplier = binder->addOperand<int32_t>();
333 binder->setOperand(multiplier, _multiplier);
334 auto fuse = binder->addOperand<int32_t>();
335 binder->setOperand(fuse, 0);
336
337 auto ofm = binder->addOperand<float>(_ofm);
338
339 binder->addOperation(
340 ann::Operation::Code::DEPTHWISE_CONV_2D,
341 {ifm, ker, bias, left, right, top, bottom, hstride, vstride, multiplier, fuse}, {ofm});
342 }
343
344private:
345 enco::SessionID _sess;
346
347private:
348 uint32_t _multiplier;
349 coco::Padding2D _pad;
350 coco::Stride2D _stride;
351
352private:
353 coco::FeatureObject *_ifm = nullptr;
354 coco::KernelObject *_ker = nullptr;
355 coco::FeatureObject *_ofm = nullptr;
356};

517{
518public:
519 void pad(const coco::Padding2D *pad) { _pad = *pad; }

524
525public:

527 {
528 using nncc::core::ADT::tensor::Shape;
529
530 auto ifm = binder->addOperand<float>(_ifm);
531 auto pad = binder->addOperand<int32_t>(Shape{4, 2});
532 {
533 std::vector<int32_t> values;
534 values.resize(8);
535 // For 'N'
536 values.at(0) = values.at(1) = 0;
537 // For 'H'
538 values.at(2) = _pad.top();
539 values.at(3) = _pad.bottom();
540 // For 'W'
541 values.at(4) = _pad.left();
542 values.at(5) = _pad.right();
543 // For 'C'
544 values.at(6) = values.at(7) = 0;
545
546 binder->setOperand(pad, values.begin(), values.end());
547 }
548
549 auto ofm = binder->addOperand<float>(_ofm);
550
551 binder->addOperation(ann::Operation::Code::PAD, {ifm, pad}, {ofm});
552 }
553
554private:
555 coco::Padding2D _pad;
556
557private:
558 coco::FeatureObject *_ifm = nullptr;
559 coco::FeatureObject *_ofm = nullptr;

629class ANNOpBuilder : public coco::Instr::Visitor<std::unique_ptr<ANNOpAppender>>
630{
631public:
632 std::unique_ptr<ANNOpAppender> visit(const coco::Eval *eval)
633 {
634 if (auto conv = eval->op()->asConv2D())
635 {
636 if (auto load = conv->arg()->asLoad())
637 {
638 auto sess = enco::session(eval->module());
639
640 auto ifm = load->object()->asFeature();
641 auto ker = conv->ker();
642 auto ofm = eval->out()->asFeature();
643
644 const auto group = conv->group();
645
646 if (group == 1)
647 {
648 auto app = make_unique<ANNConv2DAppender>();
649
650 app->session(sess);
651
652 app->pad(conv->pad());
653 app->stride(conv->stride());
654
655 app->ifm(ifm);
656 app->ofm(ofm);
657 app->ker(ker);
658
659 return app;
660 }
661 else
662 {
663 assert(ifm->shape().depth() == group);
664 assert(ker->shape().count() % group == 0);
665 assert(ker->shape().depth() == 1);
666
667 auto app = make_unique<ANNDepthwiseConv2DAppender>();
668
669 app->session(sess);
670
671 app->multiplier(ker->shape().count() / group);
672 app->pad(conv->pad());
673 app->stride(conv->stride());
674
675 app->ifm(ifm);
676 app->ofm(ofm);
677 app->ker(ker);
678
679 return app;
680 }
681 }
682 }
683 else if (auto op = eval->op()->asAdd())
684 {
685 auto left_load = op->left()->asLoad();
686 auto right_load = op->right()->asLoad();
687
688 if (left_load && right_load)
689 {
690 // Let's compile the following code fragment:
691 //
692 // %ofm = eval(Add(Load(%left), Load(%right)))
693 //
694 auto left = left_load->object()->asFeature();
695 auto right = right_load->object()->asFeature();
696 assert(left != nullptr && right != nullptr);
697
698 auto out = eval->out()->asFeature();
699 assert(out != nullptr);
700
701 auto app = make_unique<ANNAddAppender>();
702
703 app->left(left);
704 app->right(right);
705 app->out(out);
706
707 return app;
708 }
709 }
710 else if (auto op = eval->op()->asMul())
711 {
712 auto left_load = op->left()->asLoad();
713 auto right_load = op->right()->asLoad();
714
715 if (left_load && right_load)
716 {
717 // Let's compile the following code fragment:
718 //
719 // %ofm = eval(Mul(Load(%left), Load(%right)))
720 //
721 auto left = left_load->object()->asFeature();
722 auto right = right_load->object()->asFeature();
723 assert(left != nullptr && right != nullptr);
724
725 auto out = eval->out()->asFeature();
726 assert(out != nullptr);
727
728 auto app = make_unique<ANNMulAppender>();
729
730 app->left(left);
731 app->right(right);
732 app->out(out);
733
734 return app;
735 }
736 }
737 else if (auto op = eval->op()->asPadF())
738 {
739 if (auto load = op->arg()->asLoad())
740 {
741 // Let's compile the following code fragment:
742 //
743 // %ofm = eval(PadF(Load(%ifm))
744 //
745 auto ifm = load->object()->asFeature();
746 auto ofm = eval->out()->asFeature();
747
748 assert(ifm != nullptr && ofm != nullptr);
749
750 auto app = make_unique<ANNPadFAppender>();
751
752 app->pad(op->pad());
753
754 app->ifm(ifm);
755 app->ofm(ofm);
756
757 return app;
758 }
759 }

1029
1030void ANNGroupBuilder::build(enco::Code *code) const
1031{
1032 auto m = code->module();
1033
1034 // ANNGroupBuilder will construct a sequence of blocks from the original block sequence, and
1035 // a destination block (that dst_blk points to) is the tail of the generated sequence.
1036 coco::Block *dst_blk = nullptr;
1037
1038 auto append = [&](const Compatibility &t) {
1039 auto blk = m->entity()->block()->create();
1040
1041 if (dst_blk == nullptr)
1042 {
1043 m->block()->prepend(blk);
1044 }
1045 else
1046 {
1047 blk->insertAfter(dst_blk);
1048 }
1049
1050 dst_blk = blk;
1051
1052 if (COMPATIBLE == t)
1053 {
1054 _ctx->create(blk);
1055 }
1056 };
1057
1058 for (auto blk = m->block()->head(); blk;)
1059 {
1060 // Let's move instructions from a block of interest (referred to as source block) into
1061 // a destination block
1062 auto src_blk = blk;
1063 blk = src_blk->next();
1064 src_blk->detach();
1065
1066 for (auto ins = src_blk->instr()->head(); ins;)
1067 {
1068 auto cur_ins = ins;
1069 ins = cur_ins->next();
1070 cur_ins->detach();
1071
1072 auto cur_append = make_appender(cur_ins);
1073
1074 // Create a new compatible block and use it as a destination block if the current
1075 // destination block is absent or incompatible with the instruction of intereset.
1076 if ((dst_blk == nullptr) || (kind(cur_append) != kind(dst_blk)))
1077 {
1078 append(kind(cur_append));
1079 }
1080
1081 assert(dst_blk != nullptr);
1082 assert(kind(cur_append) == kind(dst_blk));
1083
1084 // Append ins to the dst_blk block
1085 dst_blk->instr()->append(cur_ins);
1086
1087 if (cur_append)
1088 {
1089 // Update Android NN IR if the current instruction is compatible
1090 auto binder = _ctx->find(dst_blk);
1091 assert(binder != nullptr);
1092 cur_append->append(binder);
1093 }
1094 }
1095
1096 // Destroy the source block
1097 assert(src_blk->instr()->empty());
1098 m->entity()->block()->destroy(src_blk);
1099 }
