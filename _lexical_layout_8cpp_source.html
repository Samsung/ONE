25static uint32_t lexical_offset(const Shape &shape, const Index &index)
26{
27 assert(shape.rank() > 0);
28 assert(shape.rank() == index.rank());
29
30 const uint32_t rank = shape.rank();
31
32 uint32_t res = index.at(0);
33
34 for (uint32_t axis = 1; axis < rank; ++axis)
35 {
36 res *= shape.dim(axis);
37 res += index.at(axis);
38 }
39
40 return res;
41}
