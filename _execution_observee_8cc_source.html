24ExecutionObservee::ExecutionObservee(const ExecObservers &observers)
25{
26 // TODO Use execution option
27 if (util::getConfigBool(util::config::MINMAX_DUMP))
28 {
29 auto observer = observers.get(ObserverType::MINMAX_DUMP);
30 if (!observer)
31 throw std::runtime_error{"MinMaxRecorder is only supported on LinearExecutor, single model"};
32
33 _observers.emplace_back(observer);
34 }
35
36 if (util::getConfigBool(util::config::TRACING_MODE))
37 {
38 auto observer = observers.get(ObserverType::TRACING);
39 if (!observer)
40 throw std::runtime_error{"Cannot find TracingObserver"};
41
42 _observers.emplace_back(observer);
43 }
44
45 if (util::getConfigBool(util::config::PROFILING_MODE))
46 {
47 auto observer = observers.get(ObserverType::PROFILE);
48 if (!observer)
49 throw std::runtime_error{
50 "Profiling is only supported on DataflowExecutor with heterogenous scheduler"};
51
52 _observers.emplace_back(observer);
53 }
54}
