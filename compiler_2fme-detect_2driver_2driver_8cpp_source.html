36// Return luci::Module from model_path
37std::unique_ptr<luci::Module> import(const std::string &model_path)
38{
39 // Load model from the file
40 foder::FileLoader loader{model_path};
41 std::vector<char> model_data = loader.load();
42
43 // Verify flatbuffers
44 flatbuffers::Verifier verifier{reinterpret_cast<const uint8_t *>(model_data.data()),
45 model_data.size()};
46 if (not circle::VerifyModelBuffer(verifier))
47 {
48 throw std::runtime_error("Failed to verify circle '" + model_path + "'");
49 }
50
51 auto circle_model = circle::GetModel(model_data.data());
52
53 if (not circle_model)
54 throw std::runtime_error("Failed to load '" + model_path + "'");
55
56 auto module = luci::Importer().importModule(circle_model);
57
58 if (not module)
59 throw std::runtime_error("Failed to load '" + model_path + "'");
60
61 return module;
62}

66int entry(int argc, char **argv)
67{
68 arser::Arser arser("fme-detect detects patterns to apply FM equalization in "
69 "the circle model");
70
71 arser::Helper::add_verbose(arser);
72
73 arser.add_argument("--input").required().help("Input circle model");
74
75 arser.add_argument("--output")
76 .required()
77 .help("Output json file that describes equalization patterns");
78
79 arser.add_argument("--allow_dup_op")
80 .nargs(0)
81 .default_value(false)
82 .help("Allow to create duplicate operations when a feature map matches "
83 "with multiple equalization patterns. This can increase the size of "
84 "the model. Default is false.");
85
86 try
87 {
88 arser.parse(argc, argv);
89 }
90 catch (const std::runtime_error &err)
91 {
92 std::cerr << err.what() << std::endl;
93 std::cout << arser;
94 return EXIT_FAILURE;
95 }
96
97 if (arser.get<bool>("--verbose"))
98 {
99 // The third parameter of setenv means REPLACE.
100 // If REPLACE is zero, it does not overwrite an existing value.
101 setenv("LUCI_LOG", "100", 0);
102 }
103
104 const std::string input_path = arser.get<std::string>("--input");
105 const std::string output_path = arser.get<std::string>("--output");
106 const bool allow_dup_op = arser.get<bool>("--allow_dup_op");
107
108 EqualizePatternFinder::Context ctx;
109 {
110 ctx._allow_dup_op = allow_dup_op;
111 }
112 EqualizePatternFinder finder(ctx);
113
114 auto module = import(input_path);
115 assert(module != nullptr); // FIX_ME_UNLESS
116
117 std::vector<EqualizePattern> patterns;
118 for (size_t idx = 0; idx < module->size(); ++idx)
119 {
120 auto graph = module->graph(idx);
121
122 auto matched = finder.find(graph);
123 patterns.insert(patterns.end(), matched.begin(), matched.end());
124 }
125
126 fme_detect::write(patterns, output_path);
127
128 return EXIT_SUCCESS;
129}
