26template <typename MemoryManager, typename TensorMap>
27void allocateMemory(MemoryManager *mgr, const TensorMap &tensors, const std::string tensor_type)
28{
29 mgr->allocate();
30
31 for (auto &&pair : tensors)
32 {
33 const auto &index = pair.first;
34 auto tensor = pair.second.get();
35 assert(!tensor->is_dynamic());
36
37 auto *buffer = mgr->getBuffer(index);
38 tensor->setBuffer(buffer);
39 VERBOSE(TensorManager) << tensor_type << index << " : " << static_cast<void *>(buffer)
40 << std::endl;
41 }
42}

44inline size_t alignedSize(const size_t size, const uint64_t align)
45{
46 return (((size) + ((align)-1)) & ~((align)-1));
47}

98void TensorManager::claimNonConstPlan(const ir::OperandIndex &index)
99{
100 auto tensor = _tensors->getNonConstTensor(index);
101 assert(tensor && !tensor->is_dynamic());
102
