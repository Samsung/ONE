80std::string group_from_partition(const luci::CircleNode *node,
81 const luci::PartitionTable &partition)
82{
83 LOGGER(l);
84
85 auto group = partition.default_group;
86
87 std::string opcodename; // opcodename or opname
88
89 switch (partition.comply)
90 {
91 case luci::PartitionTable::COMPLY::OPCODE:
92 {
93 opcodename = luci::opcode_name(node);
94 assert(!opcodename.empty());
95
96 auto it = partition.byopcodes.find(opcodename);
97 if (it != partition.byopcodes.end())
98 group = it->second;
99 break;
100 }
101 case luci::PartitionTable::COMPLY::OPNAME:
102 {
103 opcodename = node->name();
104 assert(!opcodename.empty());
105
106 auto it = partition.byopnames.find(opcodename);
107 if (it != partition.byopnames.end())
108 group = it->second;
109 break;
110 }
111
112 default:
113 throw std::runtime_error("Unsupported partition.comply");
114 }
115
116 INFO(l) << "Op: " << node->name() << ": " << opcodename << ", " << node << ", " << group
117 << std::endl;
118
119 return group;
120}

150void append(luci::CircleNode *node, luci::PGroups *pgroups, const std::string &group, uint32_t idx)
151{
152 auto pgroup = std::make_unique<luci::PGroup>();
153 pgroup->group = group;
154 pgroup->id = idx + 1;
155
156 auto pnode = std::make_unique<luci::PNode>();
157 pnode->node = node;
158 pnode->group = group;
159 pnode->pgroup = pgroup.get();
160
161 pgroup->pnodes.push_back(std::move(pnode));
162
163 IsVirtualInputNode queryvi;
164 // Set input of PGroup
165 for (uint32_t in = 0; in < node->arity(); ++in)
166 {
167 auto input = loco::must_cast<luci::CircleNode *>(node->arg(in));
168 if (input->accept(&queryvi))
169 {
170 auto pnode = std::make_unique<luci::PNode>();
171 pnode->node = input;
172 pnode->group = group;
173 pnode->pgroup = pgroup.get();
174
175 pgroup->pnodes.push_back(std::move(pnode));
176
177 pgroups->node2group[input] = group;
178 }
179 else
180 {
181 // this input maybe CircleInput in source graph
182 // --> not confident this is safe
183 pgroup->inputs.push_back(input);
184 }
185 }
186
187 IsMultiOutputNode query;
188 if (node->accept(&query))
189 {
190 // Include CircleXXXOut virtual nodes in this group
191 auto succs = loco::succs(node);
192 for (auto &succ_node : succs)
193 {
194 auto nodeout = loco::must_cast<luci::CircleNode *>(succ_node);
195
196 auto pnode = std::make_unique<luci::PNode>();
197 pnode->node = nodeout;
198 pnode->group = group;
199 pnode->pgroup = pgroup.get();
200
201 pgroup->pnodes.push_back(std::move(pnode));
202
203 pgroups->node2group[nodeout] = group;
204
205 pgroup->outputs.push_back(nodeout);
206 }
207 }
208 else
209 {
210 // Set output of PGroup: node itself
211 pgroup->outputs.push_back(node);
212 }
213
214 pgroups->node2group[node] = group;
215 pgroups->id2pgroup[pgroup->id] = pgroup.get();
216
217 pgroups->pgroups.push_back(std::move(pgroup));
218}

225std::unique_ptr<luci::PGroups> produce_pgroups(const luci::Module *source,
226 const luci::PartitionTable &partition)
227{
228 assert(source != nullptr);
229 // NOTE Only main graph (subgraph index 0) will be partitioned.
230 // Other subgraphs will follow the owner (IF/WHILE/...) group
231
232 LOGGER(l);
233
234 auto pgroups = std::make_unique<luci::PGroups>();
235
236 pgroups->default_group = partition.default_group;
237
238 // Create a PGroup per CircleNode: each PGroup will have one CircleNode
239 auto graph = source->graph();
240 auto nodes = graph->nodes();
241 for (uint32_t idx = 0; idx < nodes->size(); ++idx)
242 {
243 auto node = loco::must_cast<luci::CircleNode *>(nodes->at(idx));
244
245 // check if node is normal node that we are interested
246 if (check_allocate_partition(node))
247 {
248 auto group = group_from_partition(node, partition);
249
250 append(node, pgroups.get(), group, idx);
251 }
252 else
253 {
254 INFO(l) << "Skip Op: " << node->name() << std::endl;
255 // record as default group
256 pgroups->node2group[node] = partition.default_group;
257 }
258 }
259
260 return pgroups;
261}
