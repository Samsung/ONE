49bool fused_add_with_conv(luci::CircleAdd *add)
50{
51 // find the pattern of CircleAdd(CircleConv2D, CircleConst)
52 luci::CircleConst *shift = nullptr;
53 luci::CircleConv2D *conv2d = nullptr;
54 if (not luci::fill(&conv2d, &shift).with_commutative_args_of(add))
55 return false;
56
57 // check conditions for conv2d
58 if (conv2d->rank() != 4)
59 return false;
60 if (conv2d->fusedActivationFunction() != luci::FusedActFunc::NONE)
61 return false;
62
63 luci::CircleConst *filter = dynamic_cast<luci::CircleConst *>(conv2d->filter());
64 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(conv2d->bias());
65 luci::CircleOutputExclude *biasex = dynamic_cast<luci::CircleOutputExclude *>(conv2d->bias());
66
67 // filter should exist, bias should be const or none(output exclude)
68 if (filter == nullptr || (bias == nullptr && biasex == nullptr))
69 return false;
70 if (filter->rank() != 4)
71 return false;
72 if (filter->dtype() != shift->dtype())
73 return false;
74 // TODO support more data type
75 if (filter->dtype() != loco::DataType::FLOAT32)
76 return false;
77
78 // filter is OHWI
79 uint32_t out_channel = filter->dim(0).value();
80
81 // shape of shift should be [1, 1, 1, out_channel] or [out_channel]
82 if (shift->rank() == 4)
83 {
84 for (uint32_t i = 0; i < 3; ++i)
85 if (shift->dim(i).value() != 1)
86 return false;
87 if (shift->dim(3).value() != out_channel)
88 return false;
89 }
90 else if (shift->rank() == 1)
91 {
92 if (shift->dim(0).value() != out_channel)
93 return false;
94 }
95 else
96 return false;
97
98 auto conv2d_name = conv2d->name();
99 auto shift_name = shift->name();
100 assert(conv2d_name.length() > 0);
101 assert(shift_name.length() > 0);
102 auto bias_name = (bias ? bias->name() : conv2d_name) + ";" + shift_name;
103
104 luci::CircleConv2D *fused_conv2d = add->graph()->nodes()->create<luci::CircleConv2D>();
105 luci::CircleConst *fused_bias = add->graph()->nodes()->create<luci::CircleConst>();
106
107 fused_bias->dtype(conv2d->dtype());
108 fused_bias->rank(1);
109 fused_bias->dim(0).set(out_channel);
110 fused_bias->shape_status(luci::ShapeStatus::VALID);
111 fused_bias->name(bias_name);
112 fused_bias->size<loco::DataType::FLOAT32>(out_channel);
113 // fuse shift to bias
114 for (uint32_t b = 0; b < out_channel; ++b)
115 {
116 auto bias_val = shift->at<loco::DataType::FLOAT32>(b);
117 if (bias)
118 bias_val += bias->at<loco::DataType::FLOAT32>(b);
119 fused_bias->at<loco::DataType::FLOAT32>(b) = bias_val;
120 }
121
122 // Set attributes of fused_conv2d
123 fused_conv2d->input(conv2d->input());
124 fused_conv2d->filter(conv2d->filter());
125 fused_conv2d->bias(fused_bias);
126 fused_conv2d->fusedActivationFunction(add->fusedActivationFunction());
127 fused_conv2d->padding(conv2d->padding());
128 fused_conv2d->stride()->h(conv2d->stride()->h());
129 fused_conv2d->stride()->w(conv2d->stride()->w());
130 fused_conv2d->dilation()->h(conv2d->dilation()->h());
131 fused_conv2d->dilation()->w(conv2d->dilation()->w());
132 fused_conv2d->name(conv2d_name);
133 luci::add_origin(fused_conv2d,
134 luci::composite_origin({luci::get_origin(add), luci::get_origin(conv2d)}));
135
136 replace(add).with(fused_conv2d);
137
138 return true;
139}

146bool FuseAddWithConvPass::run(loco::Graph *g)
147{
148 bool changed = false;
149 for (auto node : loco::active_nodes(loco::output_nodes(g)))
150 {
151 if (auto add = dynamic_cast<luci::CircleAdd *>(node))
152 {
153 if (fused_add_with_conv(add))
154 changed = true;
155 }
156 }
157
158 return changed;
159}
