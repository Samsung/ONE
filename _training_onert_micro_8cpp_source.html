33float calculateMSEError(const float *predicted_values, const float *target_values,
34 const uint32_t output_size)
35{
36 const uint32_t output_number_values = output_size / sizeof(float);
37 float result = 0.0f;
38 for (int i = 0; i < output_number_values; ++i)
39 {
40 result += std::pow(predicted_values[i] - target_values[i], 2);
41 }
42
43 return result / output_number_values;
44}

46float calculateMAEError(const float *predicted_values, const float *target_values,
47 const uint32_t output_size)
48{
49 const uint32_t output_number_values = output_size / sizeof(float);
50 float result = 0.0f;
51 for (int i = 0; i < output_number_values; ++i)
52 {
53 result += std::abs(predicted_values[i] - target_values[i]);
54 }
55
56 return result / output_number_values;
57}

59Status calculateError(const uint8_t *predicted_value, const uint8_t *target_value, void *result,
60 const uint32_t output_size, MetricsTypeEnum error_type)
61{
62 switch (error_type)
63 {
64 case MSE:
65 {
66 float *result_float = reinterpret_cast<float *>(result);
67 *result_float +=
68 calculateMSEError(reinterpret_cast<const float *>(predicted_value),
69 reinterpret_cast<const float *>(target_value), output_size);
70 break;
71 }
72 case MAE:
73 {
74 float *result_float = reinterpret_cast<float *>(result);
75 *result_float +=
76 calculateMAEError(reinterpret_cast<const float *>(predicted_value),
77 reinterpret_cast<const float *>(target_value), output_size);
78 break;
79 }
80 default:
81 {
82 return Error;
83 }
84 }
85
86 return Ok;
87}

132Status TrainingOnertMicro::train(uint32_t number_of_train_samples, const uint8_t *train_data,
133 const uint8_t *label_train_data)
134{
135 if (_is_training_mode == false)
136 return EnableTrainModeError;
137
138 const uint32_t batch_size = _settings.batch_size;
139
140 const uint32_t num_inferences = number_of_train_samples / batch_size;
141
142 const uint32_t remains = number_of_train_samples % batch_size;
143
144 const uint32_t epochs = _settings.number_of_epochs;
145
146 const int32_t input_tensor_size = _interpreter->getInputDataSizeByIndex(0);
147 const int32_t output_tensor_size = _interpreter->getOutputDataSizeByIndex(0);
148
149 const uint8_t *cur_train_data = train_data;
150 const uint8_t *cur_label_train_data = label_train_data;
151
152 for (uint32_t epoch = 0; epoch < epochs; ++epoch)
153 {
154 for (uint32_t infer = 0; infer < num_inferences; ++infer)
155 {
156 for (uint32_t batch = 0; batch < batch_size; ++batch)
157 {
158 _interpreter->allocateAndWriteInputTensor(0, cur_train_data, input_tensor_size);
159
160 _interpreter->interpret();
161
162 _module.computeGradients(_settings, cur_label_train_data);
163 cur_train_data += input_tensor_size;
164 cur_label_train_data += output_tensor_size;
165 }
166
167 _module.updateWeights(_settings);
168 }
169 cur_train_data = train_data;
170 cur_label_train_data = label_train_data;
171 }
172
173 return Ok;
174}

176Status TrainingOnertMicro::test(uint32_t number_of_train_samples, const uint8_t *test_data,
177 const uint8_t *label_test_data, void *metric_value_result)
178{
179 const int32_t input_tensor_size = _interpreter->getInputDataSizeByIndex(0);
180 const int32_t output_tensor_size = _interpreter->getOutputDataSizeByIndex(0);
181
182 const uint8_t *cur_test_data = test_data;
183 const uint8_t *cur_label_test_data = label_test_data;
184
185 switch (_settings.metric)
186 {
187 case MSE:
188 case MAE:
189 {
190 float *result_float = reinterpret_cast<float *>(metric_value_result);
191 *result_float = 0.0f;
192 break;
193 }
194 default:
195 {
196 return Error;
197 }
198 }
199
200 for (uint32_t sample = 0; sample < number_of_train_samples; ++sample)
201 {
202 _interpreter->allocateAndWriteInputTensor(0, cur_test_data, input_tensor_size);
203
204 _interpreter->interpret();
205
206 const uint8_t *output_data = _interpreter->readOutputTensor(0);
207
208 Status status = calculateError(output_data, cur_label_test_data, metric_value_result,
209 output_tensor_size, _settings.metric);
210
211 if (status != Ok)
212 return status;
213
214 cur_test_data += input_tensor_size;
215 cur_label_test_data += output_tensor_size;
216 }
217
218 switch (_settings.metric)
219 {
220 case MSE:
221 case MAE:
222 {
223 float *result_float = reinterpret_cast<float *>(metric_value_result);
224 *result_float /= number_of_train_samples;
225 break;
226 }
227 default:
228 {
229 return Error;
230 }
231 }
232
233 return Ok;
234}
