34bool substitute_transpose_to_reshape(luci::CircleTranspose *node)
35{
36 auto perm_const = dynamic_cast<luci::CircleConst *>(node->perm());
37 if (perm_const == nullptr)
38 return false;
39
40 assert(perm_const->dtype() == loco::DataType::S32);
41
42 auto input_node = loco::must_cast<luci::CircleNode *>(node->a());
43 if (perm_const->dim(0).value() != input_node->rank())
44 return false;
45
46 // If input have more than 2 unknown dimension, transpose will not be changed.
47 int count = 0;
48 for (uint32_t i = 0; i < input_node->rank(); i++)
49 if (!input_node->dim(i).known())
50 count++;
51 if (count > 1)
52 return false;
53
54 uint32_t idx = 0;
55 auto size_items = perm_const->size<loco::DataType::S32>();
56 for (uint32_t i = 0; i < size_items; i++)
57 {
58 assert(perm_const->at<loco::DataType::S32>(i) >= 0 &&
59 perm_const->at<loco::DataType::S32>(i) < static_cast<int32_t>(input_node->rank()));
60 const auto perm_value = static_cast<uint32_t>(perm_const->at<loco::DataType::S32>(i));
61 if (input_node->dim(perm_value).known() && input_node->dim(perm_value).value() == 1)
62 continue;
63 // To check idx values are increasing
64 if (idx > perm_value)
65 return false;
66 idx = perm_value;
67 }
68
69 auto name = node->name();
70 assert(name.length() > 0);
71
72 auto new_const_node = node->graph()->nodes()->create<luci::CircleConst>();
73 new_const_node->dtype(loco::DataType::S32);
74 new_const_node->size<loco::DataType::S32>(size_items);
75 new_const_node->shape_status(luci::ShapeStatus::VALID);
76 new_const_node->rank(1);
77 new_const_node->dim(0).set(size_items);
78 for (uint32_t i = 0; i < size_items; i++)
79 {
80 if (input_node->dim(static_cast<uint32_t>(perm_const->at<loco::DataType::S32>(i))).known())
81 new_const_node->at<loco::DataType::S32>(i) = static_cast<int32_t>(
82 input_node->dim(static_cast<uint32_t>(perm_const->at<loco::DataType::S32>(i))).value());
83 else
84 new_const_node->at<loco::DataType::S32>(i) = -1;
85 }
86
87 auto new_reshape_node = node->graph()->nodes()->create<luci::CircleReshape>();
88 new_reshape_node->tensor(input_node);
89 new_reshape_node->shape(new_const_node);
90 new_reshape_node->name(name + "/Reshape");
91 luci::add_origin(new_reshape_node, luci::get_origin(node));
92 new_const_node->name(name + "/Reshape/shape");
93
94 replace(node).with(new_reshape_node);
95 return true;
96}

121bool SubstituteTransposeToReshapePass::run(loco::Graph *g)
122{
123 bool changed = false;
124 for (auto node : loco::active_nodes(loco::output_nodes(g)))
125 {
126 if (auto circle_node = dynamic_cast<luci::CircleTranspose *>(node))
127 {
128 if (substitute_transpose_to_reshape(circle_node))
129 {
130 changed = true;
131 }
132 }
133 }
134 return changed;
135}
