81NNFW_STATUS getTensorIndexImpl(const onert::ir::IGraph &graph, const char *tensorname,
82 uint32_t *index, bool is_input)
83{
84 if (!tensorname || !index)
85 return NNFW_STATUS_UNEXPECTED_NULL;
86
87 if (!null_terminating(tensorname, MAX_TENSOR_NAME_LENGTH))
88 {
89 std::cerr << "nnpackage path is too long" << std::endl;
90 return NNFW_STATUS_ERROR;
91 }
92
93 auto ind_found = is_input ? graph.getInputIndex(tensorname) : graph.getOutputIndex(tensorname);
94
95 if (ind_found.undefined())
96 {
97 // Not found
98 return NNFW_STATUS_ERROR;
99 }
100 else
101 {
102 *index = ind_found.value();
103 return NNFW_STATUS_NO_ERROR;
104 }
105}

107std::string trim(const std::string &value)
108{
109 std::string whitespace = " \t";
110 auto begin = value.find_first_not_of(whitespace);
111 if (begin == std::string::npos)
112 return ""; // no content
113
114 auto end = value.find_last_not_of(whitespace);
115 auto range = end - begin + 1;
116 return value.substr(begin, range);
117}

119bool loadConfigure(const std::string cfgfile, onert::util::CfgKeyValues &keyValues)
120{
121 std::ifstream ifs(cfgfile);
122 if (ifs.is_open())
123 {
124 std::string line;
125 while (std::getline(ifs, line))
126 {
127 auto cmtpos = line.find('#');
128 if (cmtpos != std::string::npos)
129 {
130 line = line.substr(0, cmtpos);
131 }
132 std::istringstream isline(line);
133 std::string key;
134 if (std::getline(isline, key, '='))
135 {
136 std::string value;
137 if (std::getline(isline, value))
138 {
139 key = trim(key);
140 keyValues[key] = trim(value);
141 }
142 }
143 }
144 ifs.close();
145 return true;
146 }
147 return false;
148}

150NNFW_TYPE datatype_to_nnfw_dtype(onert::ir::DataType dt)
151{
152 using onert::ir::DataType;
153 switch (dt)
154 {
155 case DataType::FLOAT32:
156 return NNFW_TYPE_TENSOR_FLOAT32;
157 case DataType::INT32:
158 return NNFW_TYPE_TENSOR_INT32;
159 case DataType::QUANT_UINT8_ASYMM:
160 return NNFW_TYPE_TENSOR_QUANT8_ASYMM;
161 case DataType::BOOL8:
162 return NNFW_TYPE_TENSOR_BOOL;
163 case DataType::UINT8:
164 return NNFW_TYPE_TENSOR_UINT8;
165 case DataType::INT64:
166 return NNFW_TYPE_TENSOR_INT64;
167 case DataType::QUANT_INT8_ASYMM:
168 return NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED;
169 case DataType::QUANT_INT16_SYMM:
170 return NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED;
171 case DataType::UINT32:
172 case DataType::QUANT_INT8_SYMM:
173 default:
174 throw std::runtime_error("Error: Model has type that runtime API does not support.");
175 }

221uint64_t getBufSize(const nnfw_tensorinfo *info)
222{
223 static int elmsize[] = {
224 sizeof(float), /* NNFW_TYPE_TENSOR_FLOAT32 = 0 */
225 sizeof(int), /* NNFW_TYPE_TENSOR_INT32 = 1 */
226 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM = 2 */
227 sizeof(bool), /* NNFW_TYPE_TENSOR_BOOL = 3 */
228 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_UINT8 = 4 */
229 sizeof(int64_t), /* NNFW_TYPE_TENSOR_INT64 = 5 */
230 sizeof(int8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED = 6 */
231 sizeof(int16_t), /* NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED = 7 */
232 };
233
234 uint64_t n = 1;
235 for (int32_t i = 0; i < info->rank; ++i)
236 {
237 assert(info->dims[i] >= 0);
238 n *= info->dims[i];
239 }
240 return elmsize[info->dtype] * n;
241}

338NNFW_STATUS nnfw_session::load_model_from_nnpackage(const char *package_dir)
339{
340 if (!isStateInitialized())
341 return NNFW_STATUS_INVALID_STATE;
342
343 if (!package_dir)
344 {
345 std::cerr << "package_dir is null." << std::endl;
346 return NNFW_STATUS_UNEXPECTED_NULL;
347 }
348
349 if (!null_terminating(package_dir, MAX_PATH_LENGTH))
350 {
351 std::cerr << "nnpackage path is too long" << std::endl;
352 return NNFW_STATUS_ERROR;
353 }
354
355 // TODO : add support for zipped package file load
356 DIR *dir;
357 if (!(dir = opendir(package_dir)))
358 {
359 std::cerr << "invalid nnpackge directory: " << package_dir << std::endl;
360 return NNFW_STATUS_ERROR;
361 }
362 closedir(dir);
363
364 try
365 {
366 std::string package_path(package_dir);
367 std::string manifest_file_name = package_path + "/metadata/MANIFEST";
368 std::ifstream mfs(manifest_file_name);
369
370 // extract the filename of the first(index 0) model
371 // e.g. In MANIFEST file, { "models" : [ "firstmodel.tflite", "2nd.tflite" ] }
372 Json::Value root;
373 mfs >> root;
374 const Json::Value &models = root["models"];
375 const Json::Value &model_types = root["model-types"];
376 const Json::Value &configs = root["configs"];
377
378 if (!configs.empty() && !configs[0].empty())
379 {
380 auto filepath = package_path + std::string("/metadata/") + configs[0].asString();
381
382 onert::util::CfgKeyValues keyValues;
383 if (loadConfigure(filepath, keyValues))
384 {
385 onert::util::setConfigKeyValues(keyValues);
386 }
387 }
388 _nnpkg = std::make_shared<onert::ir::NNPkg>();
389 auto num_models = models.size();
390 if (num_models == 0 || (num_models - 1) > onert::ir::ModelIndex::max())
391 {
392 std::cerr << "Invalid model size - " << std::to_string(num_models) << std::endl;
393 return NNFW_STATUS_ERROR;
394 }
395
396 // Not support backend mapping to operator index for multiple models yet
397 // TODO Support this
398 if (num_models > 1 && _coptions->manual_scheduler_options.index_to_backend.size() != 0)
399 {
400 std::cerr << "Cannot set backend to operator index for multiple models" << std::endl;
401 return NNFW_STATUS_ERROR;
402 }
403
404 for (uint16_t i = 0; i < num_models; ++i)
405 {
406 auto model_file_path = package_path + std::string("/") + models[i].asString();
407 auto model_type = model_types[i].asString();
408 auto model = loadModel(model_file_path, model_type);
409 if (model == nullptr)
410 return NNFW_STATUS_ERROR;
411 _model_path = std::string(model_file_path); // TODO Support multiple models
412 model->bindKernelBuilder(_kernel_registry->getBuilder());
413 _nnpkg->push(onert::ir::ModelIndex{i}, std::move(model));
414 }
415
416 _train_info = loadTrainingInfo(_nnpkg->primary_model());
417
418 auto toIODesc = [](std::string str) {
419 auto indices = nnfw::misc::split(str, ':');
420 if (indices.size() != 3)
421 {
422 std::cerr << "IODesc should be 3-tuple." << std::endl;
423 return onert::ir::IODesc{};
424 }
425 auto model_idx = static_cast<uint32_t>(std::stoi(indices.at(0)));
426 auto subgraph_idx = static_cast<uint32_t>(std::stoi(indices.at(1)));
427 auto operand_idx = static_cast<uint32_t>(std::stoi(indices.at(2)));
428 return onert::ir::IODesc{model_idx, subgraph_idx, operand_idx};
429 };
430 // read pkg-inputs and pkg-outputs
431 const Json::Value &pkg_inputs = root["pkg-inputs"];
432 for (uint32_t i = 0; i < pkg_inputs.size(); ++i)
433 _nnpkg->addInput(toIODesc(pkg_inputs[i].asString()));
434 const Json::Value &pkg_outputs = root["pkg-outputs"];
435 for (uint32_t i = 0; i < pkg_outputs.size(); ++i)
436 _nnpkg->addOutput(toIODesc(pkg_outputs[i].asString()));
437 // read model-connect
438 const Json::Value &fromtos = root["model-connect"];
439 for (uint32_t i = 0; i < fromtos.size(); ++i)
440 {
441 const Json::Value &tos = fromtos[i]["to"];
442 for (uint32_t j = 0; j < tos.size(); ++j)
443 _nnpkg->addEdge(toIODesc(fromtos[i]["from"].asString()), toIODesc(tos[j].asString()));
444 }
445
446 _nnpkg->verify();
447 _state = State::MODEL_LOADED;
448 }
449 catch (const std::exception &e)
450 {
451 std::cerr << "Error during model loading : " << e.what() << std::endl;
452 return NNFW_STATUS_ERROR;
453 }
454 return NNFW_STATUS_NO_ERROR;
455}

457NNFW_STATUS nnfw_session::prepare()
458{
459 // NOTE. If users want to run prepare() more than one time, this could be removed.
460 if (!isStateModelLoaded())
461 {
462 std::cerr << "Error during model prepare : ";
463 if (isStateInitialized())
464 {
465 std::cerr << "prepare should be run once";
466 }
467 else
468 {
469 std::cerr << "invalid state";
470 }
471 std::cerr << std::endl;
472 return NNFW_STATUS_INVALID_STATE;
473 }
474
475 try
476 {
477 auto compiler = onert::compiler::CompilerFactory::get().create(_nnpkg, _coptions.get());
478 _nnpkg.reset();
479 _compiler_artifact = compiler->compile();
480 _execution = std::make_unique<onert::exec::Execution>(_compiler_artifact->_executors);
481 }
482 catch (const std::exception &e)
483 {
484 std::cerr << "Error during model prepare : " << e.what() << std::endl;
485 return NNFW_STATUS_ERROR;
486 }
487
488 _state = State::PREPARED;
489 return NNFW_STATUS_NO_ERROR;
490}

999NNFW_STATUS nnfw_session::get_config(const char *key, char *value, size_t value_size)
1000{
1001 if (!isStateModelLoaded())
1002 return NNFW_STATUS_INVALID_STATE;
1003
1004 if (!key || !value)
1005 return NNFW_STATUS_UNEXPECTED_NULL;
1006
1007 auto check_boundary = [](size_t dest_size, std::string &src) {
1008 if (dest_size < src.length() + 1 /* for '\0' */)
1009 {
1010 std::cerr << "buffer is small to copy config value." << std::endl;
1011 return false;
1012 }
1013 return true;
1014 };
1015
1016 const std::string skey = key;
1017
1018 if (skey == onert::util::config::BACKENDS)
1019 {
1020 if (_coptions->backend_list.size() == 0)
1021 return NNFW_STATUS_NO_ERROR; // no setting backend is not an error of get_config_str()
1022
1023 auto str =
1024 nnfw::misc::join(_coptions->backend_list.begin(), _coptions->backend_list.end(), ";");
1025
1026 if (!check_boundary(value_size, str))
1027 return NNFW_STATUS_ERROR;
1028
1029 strncpy(value, str.c_str(), value_size);
1030 }
1031 else if (skey == onert::util::config::EXECUTOR)
1032 {
1033 if (!check_boundary(value_size, _coptions->executor))
1034 return NNFW_STATUS_ERROR;
1035
1036 strncpy(value, _coptions->executor.c_str(), _coptions->executor.length());
1037 }
1038 else
1039 {
1040 return NNFW_STATUS_ERROR;
1041 }
1042
1043 return NNFW_STATUS_NO_ERROR;
1044}

1159NNFW_STATUS nnfw_session::train_get_traininfo(nnfw_train_info *info)
1160{
1161 if (isStateInitialized())
1162 {
1163 // There is no _train_info in INITIALIZED, since _train_info is set when a model loaded
1164 std::cerr << "Error during nnfw_session::train_get_traininfo : invalid state";
1165 return NNFW_STATUS_INVALID_STATE;
1166 }
1167
1168 if (info == nullptr)
1169 {
1170 std::cerr << "Error during nnfw_session::train_get_traininfo : info is nullptr" << std::endl;
1171 return NNFW_STATUS_UNEXPECTED_NULL;
1172 }
1173
1174 // after model loaded, it ensures that _train_info is not nullptr
1175 assert(_train_info != nullptr);
1176
1177 auto convertLossCode = [](const onert::ir::train::LossCode &code) -> NNFW_TRAIN_LOSS {
1178 switch (code)
1179 {
1180 case onert::ir::train::LossCode::Undefined:
1181 return NNFW_TRAIN_LOSS_UNDEFINED;
1182 case onert::ir::train::LossCode::MeanSquaredError:
1183 return NNFW_TRAIN_LOSS_MEAN_SQUARED_ERROR;
1184 case onert::ir::train::LossCode::CategoricalCrossentropy:
1185 return NNFW_TRAIN_LOSS_CATEGORICAL_CROSSENTROPY;
1186 default:
1187 throw std::runtime_error{"fail to convert ir::train::LossCode"};
1188 }
1189 };
1190
1191 auto convertLossReduction =
1192 [](const onert::ir::train::LossReductionType &type) -> NNFW_TRAIN_LOSS_REDUCTION {
1193 switch (type)
1194 {
1195 case onert::ir::train::LossReductionType::Undefined:
1196 return NNFW_TRAIN_LOSS_REDUCTION_UNDEFINED;
1197 case onert::ir::train::LossReductionType::SumOverBatchSize:
1198 return NNFW_TRAIN_LOSS_REDUCTION_SUM_OVER_BATCH_SIZE;
1199 case onert::ir::train::LossReductionType::Sum:
1200 return NNFW_TRAIN_LOSS_REDUCTION_SUM;
1201 default:
1202 throw std::runtime_error{"fail to convert from ir::train::LossReductionType"};
1203 break;
1204 }
1205 };
1206
1207 auto convertOptimizerCode =
1208 [](const onert::ir::train::OptimizerCode &code) -> NNFW_TRAIN_OPTIMIZER {
1209 switch (code)
1210 {
1211 case onert::ir::train::OptimizerCode::Undefined:
1212 return NNFW_TRAIN_OPTIMIZER_UNDEFINED;
1213 case onert::ir::train::OptimizerCode::SGD:
1214 return NNFW_TRAIN_OPTIMIZER_SGD;
1215 case onert::ir::train::OptimizerCode::Adam:
1216 return NNFW_TRAIN_OPTIMIZER_ADAM;
1217 default:
1218 throw std::runtime_error{"fail to convert from ir::train::OptimizerCode"};
1219 }
1220 };
1221
1222 const auto &loss = _train_info->lossInfo();
1223 const auto &optim = _train_info->optimizerInfo();
1224
1225 try
1226 {
1227 info->learning_rate = optim.learning_rate;
1228 info->batch_size = _train_info->batchSize();
1229 info->loss_info.loss = convertLossCode(loss.loss_code);
1230 info->loss_info.reduction_type = convertLossReduction(loss.reduction_type);
1231 info->opt = convertOptimizerCode(optim.optim_code);
1232
1233 if (_train_info->getTrainableOps().size() > 0)
1234 {
1235 const uint32_t first_trainable_idx = _train_info->getTrainableOps().cbegin()->value();
1236 const uint32_t last_trainable_idx = _train_info->getTrainableOps().crbegin()->value();
1237 const uint32_t ops_size = primary_subgraph()->operations().size();
1238 const uint32_t trainable_indexes_range = last_trainable_idx - first_trainable_idx + 1;
1239
1240 // check if trainable ops set contains continuous indexes on the back of the set
1241 if (last_trainable_idx == ops_size - 1 &&
1242 trainable_indexes_range == _train_info->getTrainableOps().size())
1243 {
1244 // check if all ops are trainable
1245 if (0 == first_trainable_idx)
1246 {
1247 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_ALL;
1248 }
1249 else
1250 {
1251 info->num_of_trainable_ops = trainable_indexes_range;
1252 }
1253 }
1254 else
1255 {
1256 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_INCORRECT_STATE;
1257 std::cerr << "conversion from set of trainable ops to num_of_trainable_ops is impossible"
1258 << std::endl;
1259 return NNFW_STATUS_INVALID_STATE;
1260 }
1261 }
1262 else
1263 {
1264 // no layer will be trained
1265 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_NONE;
1266 }
1267 }
1268 catch (const std::exception &e)
1269 {
1270 std::cerr << "Error during nnfw_session::train_get_traininfo" << e.what() << std::endl;
1271 return NNFW_STATUS_ERROR;
1272 }
1273
1274 return NNFW_STATUS_NO_ERROR;
1275}

1277NNFW_STATUS nnfw_session::train_set_traininfo(const nnfw_train_info *info)
1278{
1279 if (not isStateModelLoaded())
1280 {
1281 std::cerr << "Error during nnfw_session::train_set_traininfo : invalid state" << std::endl;
1282 return NNFW_STATUS_INVALID_STATE;
1283 }
1284
1285 if (info == nullptr)
1286 {
1287 std::cerr << "nnfw_session::train_set_traininfo : info is nullptr" << std::endl;
1288 return NNFW_STATUS_UNEXPECTED_NULL;
1289 }
1290
1291 // after model loaded, it ensures that _train_info is not nullptr
1292 assert(_train_info != nullptr);
1293
1294 auto convertLossType = [](const int &type) {
1295 if (type == NNFW_TRAIN_LOSS_MEAN_SQUARED_ERROR)
1296 return onert::ir::train::LossCode::MeanSquaredError;
1297 else if (type == NNFW_TRAIN_LOSS_CATEGORICAL_CROSSENTROPY)
1298 return onert::ir::train::LossCode::CategoricalCrossentropy;
1299 else
1300 throw std::runtime_error("not supported loss type");
1301 };
1302
1303 auto convertLossReductionType = [](const int &type) {
1304 if (type == NNFW_TRAIN_LOSS_REDUCTION_SUM_OVER_BATCH_SIZE)
1305 return onert::ir::train::LossReductionType::SumOverBatchSize;
1306 else if (type == NNFW_TRAIN_LOSS_REDUCTION_SUM)
1307 return onert::ir::train::LossReductionType::Sum;
1308 else
1309 throw std::runtime_error("not supported loss reduction type");
1310 };
1311
1312 auto convertOptType = [](const int &type) {
1313 if (type == NNFW_TRAIN_OPTIMIZER_SGD)
1314 return onert::ir::train::OptimizerCode::SGD;
1315 else if (type == NNFW_TRAIN_OPTIMIZER_ADAM)
1316 return onert::ir::train::OptimizerCode::Adam;
1317 else
1318 throw std::runtime_error("not supported optimizer type");
1319 };
1320
1321 try
1322 {
1323 onert::ir::train::LossInfo loss_info;
1324 loss_info.loss_code = convertLossType(info->loss_info.loss);
1325 loss_info.reduction_type = convertLossReductionType(info->loss_info.reduction_type);
1326
1327 onert::ir::train::OptimizerInfo opt_info;
1328 opt_info.learning_rate = info->learning_rate;
1329 opt_info.optim_code = convertOptType(info->opt);
1330
1331 _train_info->setBatchSize(info->batch_size);
1332 _train_info->setLossInfo(loss_info);
1333 _train_info->setOptimizerInfo(opt_info);
1334
1335 if (info->num_of_trainable_ops < -1)
1336 {
1337 std::cerr << "Error during nnfw_session::train_set_traininfo: provided num_of_trainable_ops "
1338 "has incorrect value: "
1339 << info->num_of_trainable_ops << std::endl;
1340 return NNFW_STATUS_ERROR;
1341 }
1342
1343 const uint32_t ops_size = primary_subgraph()->operations().size();
1344 std::set<onert::ir::OperationIndex> trainable_ops;
1345
1346 if (NNFW_TRAIN_TRAINABLE_ALL == info->num_of_trainable_ops)
1347 {
1348 for (uint32_t idx = 0; idx < ops_size; ++idx)
1349 {
1350 trainable_ops.emplace(idx);
1351 }
1352 }
1353 else
1354 {
1355 if (static_cast<uint32_t>(info->num_of_trainable_ops) > ops_size)
1356 {
1357 std::cerr
1358 << "Error during nnfw_session::train_set_traininfo: provided num_of_trainable_ops="
1359 << info->num_of_trainable_ops << " is out of operators range equals: " << ops_size
1360 << std::endl;
1361 return NNFW_STATUS_ERROR;
1362 }
1363 for (uint32_t i = 1; i <= static_cast<uint32_t>(info->num_of_trainable_ops); ++i)
1364 {
1365 trainable_ops.emplace(ops_size - i);
1366 }
1367 }
1368 // Note that possible setting an empty trainable_ops set (for NNFW_TRAIN_TRAINABLE_NONE value)
1369 _train_info->setTrainableOps(trainable_ops);
1370 }
1371 catch (const std::exception &e)
1372 {
1373 std::cerr << "Error during nnfw_session::train_set_traininfo : " << e.what() << std::endl;
1374 return NNFW_STATUS_ERROR;
1375 }
1376
1377 return NNFW_STATUS_NO_ERROR;
1378}

1380NNFW_STATUS nnfw_session::train_prepare()
1381{
1382 // We may need different state to represent training model is loaded
1383 if (!isStateModelLoaded())
1384 {
1385 std::cerr << "Error during model prepare training: ";
1386 if (_state == State::PREPARED_TRAINING)
1387 std::cerr << "prepare should be run once";
1388 else
1389 std::cerr << "invalid state";
1390 std::cerr << std::endl;
1391 return NNFW_STATUS_INVALID_STATE;
1392 }
1393
1394 // after model loaded, it ensures that _train_info is not nullptr
1395 assert(_train_info != nullptr);
1396
1397 try
1398 {
1399 if (not _train_info->isValid())
1400 throw std::runtime_error{"training info is not valid"};
1401
1402 // initialize trainingStep count
1403 _train_info->trainingStep() = 0;

1899 {
1900 if (isStateInitialized() || isStateRunning())

1903 return NNFW_STATUS_INVALID_STATE;

1906 std::string target_str{target};
1907 if (target_str.empty() || target_str.substr(target_str.size() - 4) != "-gen")
1908 {

1910 return NNFW_STATUS_ERROR;

1913 onert::odc::CodegenPreference codegen_pref;
1914 switch (pref)
1915 {
1916 case NNFW_CODEGEN_PREF_DEFAULT:
1917 codegen_pref = onert::odc::CodegenPreference::CODEGEN_PREF_DEFAULT;
1918 break;
1919 case NNFW_CODEGEN_PREF_PERFORMANCE_FIRST:
1920 codegen_pref = onert::odc::CodegenPreference::CODEGEN_PREF_PERFORMANCE_FIRST;
1921 break;
1922 case NNFW_CODEGEN_PREF_MEMORY_FIRST:
1923 codegen_pref = onert::odc::CodegenPreference::CODEGEN_PREF_MEMORY_FIRST;
1924 break;
1925 case NNFW_CODEGEN_PREF_COMPILE_TIME_FIRST:
1926 codegen_pref = onert::odc::CodegenPreference::CODEGEN_PREF_COMPILE_TIME_FIRST;
1927 break;
1928 default:
1929 std::cerr << "Error during nnfw_session::codegen : Invalid preference" << std::endl;
1930 return NNFW_STATUS_ERROR;
1931 }
1932
1933 assert(_codegen_manager != nullptr);
1934 auto export_model_path = _codegen_manager->exportModelPath();
1935 // If the export_model_path is not set, it generates a compiled model path
1936 // automatically.
1937 if (export_model_path.empty())
1938 {
1939 // model path always has a dot. (valid extension)
1940 auto dotidx = _model_path.rfind('.');
1941 assert(dotidx != std::string::npos);
1942 auto genidx = target_str.rfind("-gen");
1943 assert(genidx != std::string::npos);
1944 // The compiled model path is the same directory of the original model/package with
1945 // target backend extension.
1946 export_model_path = _model_path.substr(0, dotidx + 1) + target_str.substr(0, genidx);
1947 _codegen_manager->exportModelPath(export_model_path);
1948 }
1949
1950 _codegen_manager->codegen(_model_path, target, codegen_pref);
1951
1952 // Replace model

1954 // TODO: Use std::filesystem::path when we can use c++17.
1955 auto dotidx = export_model_path.rfind('.');
1956 if (dotidx == std::string::npos)
1957 {
1958 std::cerr << "Error during nnfw_session::codegen : Invalid compiled model path. Please use a "
1959 "path that includes the extension."
1960 << std::endl;
1961 return NNFW_STATUS_ERROR;
1962 }
1963
1964 std::string model_type = export_model_path.substr(dotidx + 1); // + 1 to exclude dot
1965
1966 // Replace model
1967 // TODO Support buffer replace, not file reload
1968 return loadModelFile(export_model_path, model_type);
1969 }
