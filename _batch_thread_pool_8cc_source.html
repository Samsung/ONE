26BatchThreadPool::BatchThreadPool(size_t num_threads) : _num_threads(num_threads), _stop_all(false)
27{
28 _worker_threads.reserve(_num_threads);
29 for (uint32_t thread_num = 0; thread_num < _num_threads; ++thread_num)
30 {
31 _worker_threads.emplace_back([this, thread_num]() { this->worker(thread_num); });
32 }
33}

35void BatchThreadPool::worker(uint32_t thread_num)
36{
37 while (true)
38 {
39 std::unique_lock<std::mutex> lock(_m_job_queue);
40 _cv_job_queue.wait(lock, [this]() { return !this->_job_queue.empty() || _stop_all; });
41 if (_stop_all && this->_job_queue.empty())
42 {
43 return;
44 }
45
46 // Pop a job in front of queue
47 auto job = std::move(_job_queue.front());
48 _job_queue.pop();
49 lock.unlock();
50
51 // Run the job
52 job(thread_num);
53 }
54}

56BatchThreadPool::~BatchThreadPool()
57{
58 _stop_all = true;
59 _cv_job_queue.notify_all();
60
61 for (auto &t : _worker_threads)
62 {
63 t.join();
64 }
65}
