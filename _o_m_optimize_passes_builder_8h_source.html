61 OMStatus getOptimizeFunc(uint32_t idx, OptimizePassFunc **optimize_func) const
62 {
63 assert(idx < size_t(optimize::OMPassBuilderID::Size));
64 if (idx >= size_t(optimize::OMPassBuilderID::Size))
65 {
66 *optimize_func = nullptr;
67 return UnknownError;
68 }
69 *optimize_func = _optimize_passes[idx];
70 return Ok;
71 }

74
75private:

77 {
78 assert(size_t(id) < size_t(optimize::OMPassBuilderID::Size));
79 _optimize_passes[size_t(id)] = func;
80 }

82private:
