106bool GeluPattern1::matched()
107{
108 // check pattern
109 CHECK_OR_FALSE(luci::fill(&_mul, &_const_half).with_commutative_args_of(_mul_half));
110 CHECK_OR_FALSE(luci::fill(&_ifm, &_add_one).with_commutative_args_of(_mul));
111 CHECK_OR_FALSE(luci::fill(&_erf_out, &_const_one).with_commutative_args_of(_add_one));
112
113 if (auto erf = dynamic_cast<luci::CircleCustom *>(_erf_out->input()))
114 _erf = erf;
115
116 CHECK_OR_FALSE(_erf != nullptr);
117
118 // Check erf
119 CHECK_OR_FALSE(_erf->custom_code() == "Erf");
120 CHECK_OR_FALSE(_erf->numInputs() == 1);
121 CHECK_OR_FALSE(_erf->numOutputs() == 1);
122
123 if (auto mul_sqrt = dynamic_cast<luci::CircleMul *>(_erf->inputs(0)))
124 _mul_sqrt = mul_sqrt;
125
126 CHECK_OR_FALSE(_mul_sqrt != nullptr);
127
128 CHECK_OR_FALSE(luci::fill(&_ifm, &_const_sqrt).with_commutative_args_of(_mul_sqrt));
129
130 CHECK_OR_FALSE(_mul_sqrt->x() == _ifm);
131 CHECK_OR_FALSE(_mul->x() == _ifm);
132
133 // Check Activation to be NONE
134 CHECK_OR_FALSE(_mul_sqrt->fusedActivationFunction() == luci::FusedActFunc::NONE);
135 CHECK_OR_FALSE(_add_one->fusedActivationFunction() == luci::FusedActFunc::NONE);
136 CHECK_OR_FALSE(_mul->fusedActivationFunction() == luci::FusedActFunc::NONE);
137 CHECK_OR_FALSE(_mul_half->fusedActivationFunction() == luci::FusedActFunc::NONE);
138
139 // check _const_sqrt condition
140 CHECK_OR_FALSE(_const_sqrt->dtype() == loco::DataType::FLOAT32);
141 CHECK_OR_FALSE(_const_sqrt->size<loco::DataType::FLOAT32>() == 1);
142 CHECK_OR_FALSE(::same(_const_sqrt->at<loco::DataType::FLOAT32>(0), sqrtf(0.5f)));
143
144 // check if _const_half is 0.5 (fp32)
145 CHECK_OR_FALSE(_const_half->dtype() == loco::DataType::FLOAT32);
146 CHECK_OR_FALSE(_const_half->size<loco::DataType::FLOAT32>() == 1);
147 CHECK_OR_FALSE(_const_half->at<loco::DataType::FLOAT32>(0) == 0.5);
148
149 // check _const_one condition
150 CHECK_OR_FALSE(_const_one->dtype() == loco::DataType::FLOAT32);
151 CHECK_OR_FALSE(_const_one->size<loco::DataType::FLOAT32>() == 1);
152 CHECK_OR_FALSE(_const_one->at<loco::DataType::FLOAT32>(0) == 1);
153
154 return true;
155}

174luci::CircleGelu *FuseGelu::create_gelu(loco::Graph *graph)
175{
176 assert(graph);
177
178 auto gelu = graph->nodes()->create<luci::CircleGelu>();
179 gelu->features(_p->_ifm);
180 // TODO Support approximate = True pattern
181 gelu->approximate(false);
182 gelu->name(_p->_mul_half->name() + "_gelu");
183 return gelu;
184}

186void FuseGelu::apply()
187{
188 auto graph = _p->_mul_half->graph();
189
190 auto gelu = create_gelu(graph);
191
192 // set origin
193 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> origin_vec{
194 luci::get_origin(_p->_mul_sqrt), luci::get_origin(_p->_erf), luci::get_origin(_p->_add_one),
195 luci::get_origin(_p->_mul), luci::get_origin(_p->_mul_half)};
196
197 luci::add_origin(gelu, luci::composite_origin(origin_vec));
198
199 replace(_p->_mul_half).with(gelu);
200}

207bool fuse_gelu(luci::CircleMul *mul)
208{
209 assert(mul);
210
211 // check first pattern
212 GeluPattern1 pattern(mul);
213 if (pattern.matched())
214 {
215 FuseGelu fuse(&pattern);
216 fuse.apply();
217 return true;
218 }
219 return false;
220}

227bool FuseGeluPass::run(loco::Graph *g)
228{
229 bool changed = false;
230
231 for (auto node : loco::active_nodes(loco::output_nodes(g)))
232 {
233 auto mul = dynamic_cast<luci::CircleMul *>(node);
234 if (not mul)
235 continue;
236
237 if (fuse_gelu(mul))
238 changed = true;
239 }
240
241 return changed;
242}
