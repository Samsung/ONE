32struct OnnxQDQPattern final
33{
34public:
35 OnnxQDQPattern(luci::CircleCustomOut *candidate) { dq_out = candidate; }
36
37public:
38 bool matched()
39 {
40 if (not dq_out)
41 return false;
42
43 dq = loco::must_cast<luci::CircleCustom *>(dq_out->input());
44 if (not is_onnx_dequantize_linear(dq))
45 return false;
46
47 q_out = dynamic_cast<luci::CircleCustomOut *>(dq->inputs(0));
48 if (not q_out)
49 return false;
50
51 dq_scale = dynamic_cast<luci::CircleConst *>(dq->inputs(1));
52 if (not dq_scale)
53 return false;
54
55 dq_zerop = dynamic_cast<luci::CircleConst *>(dq->inputs(2));
56 if (not dq_zerop)
57 return false;
58
59 q = loco::must_cast<luci::CircleCustom *>(q_out->input());
60 if (not is_onnx_quantize_linear(q))
61 return false;
62
63 input = loco::must_cast<luci::CircleNode *>(q->inputs(0));
64 if (input->dtype() != loco::DataType::FLOAT32)
65 return false;
66
67 q_scale = dynamic_cast<luci::CircleConst *>(q->inputs(1));
68 if (not q_scale)
69 return false;
70
71 q_zerop = dynamic_cast<luci::CircleConst *>(q->inputs(2));
72 if (not q_zerop)
73 return false;
74
75 const auto q_dtype = q->dtype();
76 const auto q_scale_dtype = q_scale->dtype();
77 const auto q_zerop_dtype = q_zerop->dtype();
78 const auto dq_scale_dtype = dq_scale->dtype();
79 const auto dq_zerop_dtype = dq_zerop->dtype();
80
81 if (q_scale_dtype != loco::DataType::FLOAT32)
82 return false;
83
84 if (dq_scale_dtype != loco::DataType::FLOAT32)
85 return false;
86
87 // Invariant from onnx Quantize operator
88 if (q_dtype != q_zerop_dtype)
89 return false;
90
91 // Invariant from onnx Dequantize operator
92 if (q_dtype != dq_zerop_dtype)
93 return false;
94
95 // Check length of scale, zp = 1
96 if (q_scale->size<loco::DataType::FLOAT32>() != 1)
97 return false;
98
99 if (dq_scale->size<loco::DataType::FLOAT32>() != 1)
100 return false;
101
102 auto q_zerop_size = 0;
103 auto dq_zerop_size = 0;
104 switch (q_zerop_dtype)
105 {
106 case loco::DataType::S16:
107 q_zerop_size = q_zerop->size<loco::DataType::S16>();
108 dq_zerop_size = dq_zerop->size<loco::DataType::S16>();
109 break;
110 case loco::DataType::U8:
111 q_zerop_size = q_zerop->size<loco::DataType::U8>();
112 dq_zerop_size = dq_zerop->size<loco::DataType::U8>();
113 break;
114 default:
115 throw std::runtime_error("Unsupported zerop dtype in " + q_zerop->name());
116 }
117
118 if (q_zerop_size != 1)
119 return false;
120
121 if (dq_zerop_size != 1)
122 return false;
123
124 return true;
125 }
126
127public:
128 luci::CircleCustomOut *dq_out = nullptr;
129 luci::CircleCustom *dq = nullptr;
130 luci::CircleConst *dq_scale = nullptr;
131 luci::CircleConst *dq_zerop = nullptr;
132 luci::CircleCustomOut *q_out = nullptr;
133 luci::CircleCustom *q = nullptr;
134 luci::CircleConst *q_scale = nullptr;
135 luci::CircleConst *q_zerop = nullptr;
136 luci::CircleNode *input = nullptr;
137};

139class QuantizeOnnxQDQ final
140{
141public:
142 QuantizeOnnxQDQ(const OnnxQDQPattern &p) : _p(p) {}
143
144public:
145 void apply(void)
146 {
147 const auto quantized_dtype = _p.q->dtype();
148
149 // Get scale
150 assert(_p.q_scale->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
151 assert(_p.q_scale->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
152 const float q_scale = _p.q_scale->at<loco::DataType::FLOAT32>(0);
153
154 assert(_p.dq_scale->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
155 assert(_p.dq_scale->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
156 const float dq_scale = _p.dq_scale->at<loco::DataType::FLOAT32>(0);
157
158 if (q_scale != dq_scale)
159 throw std::runtime_error("Invalid scale value in " + _p.dq_scale->name());
160
161 // Get zerop
162 int64_t q_zerop = 0;
163 int64_t dq_zerop = 0;
164 switch (quantized_dtype)
165 {
166 case loco::DataType::S16:
167 assert(_p.q_zerop->size<loco::DataType::S16>() == 1); // FIX_CALLER_UNLESS
168 assert(_p.dq_zerop->size<loco::DataType::S16>() == 1); // FIX_CALLER_UNLESS
169 q_zerop = _p.q_zerop->at<loco::DataType::S16>(0);
170 dq_zerop = _p.dq_zerop->at<loco::DataType::S16>(0);
171 break;
172 case loco::DataType::U8:
173 assert(_p.q_zerop->size<loco::DataType::U8>() == 1); // FIX_CALLER_UNLESS
174 assert(_p.dq_zerop->size<loco::DataType::U8>() == 1); // FIX_CALLER_UNLESS
175 q_zerop = _p.q_zerop->at<loco::DataType::U8>(0);
176 dq_zerop = _p.dq_zerop->at<loco::DataType::U8>(0);
177 break;
178 default:
179 throw std::runtime_error("Unsupported zerop dtype in " + _p.q_zerop->name());
180 }
181
182 if (q_zerop != dq_zerop)
183 throw std::runtime_error("Invalid zerop value in " + _p.dq_zerop->name());
184
185 auto qparam = std::make_unique<luci::CircleQuantParam>();
186 {
187 qparam->scale.push_back(q_scale);
188 qparam->zerop.push_back(q_zerop);
189 qparam->quantized_dimension = 0;
190 }
191
192 if (auto const_input = dynamic_cast<luci::CircleConst *>(_p.input))
193 {
194 assert(const_input->dtype() == loco::DataType::FLOAT32); // FIX_ME_UNLESS
195 const auto num_elem = const_input->size<loco::DataType::FLOAT32>();
196
197 auto new_const = luci::clone(const_input);
198 new_const->name(new_const->name() + "_quant");
199 add_origin(new_const, luci::get_origin(const_input));
200
201 new_const->dtype(quantized_dtype);
202
203 // Quantize const
204 switch (quantized_dtype)
205 {
206 case loco::DataType::S16:
207 {
208 new_const->size<loco::DataType::S16>(num_elem);
209
210 const int64_t max_val = std::numeric_limits<int16_t>::max();
211 const int64_t min_val = -max_val;
212 for (uint32_t i = 0; i < num_elem; i++)
213 {
214 const float fp_val = const_input->at<loco::DataType::FLOAT32>(i);
215 const int64_t q_val = std::round(fp_val / q_scale) + q_zerop;
216 new_const->at<loco::DataType::S16>(i) = std::min(max_val, std::max(min_val, q_val));
217 }
218 break;
219 }
220 default:
221 throw std::runtime_error("Unsupported quantized_dtype");
222 }
223 new_const->quantparam(std::move(qparam));
224
225 replace(_p.dq_out).with(new_const);
226 }
227 else
228 {
229 // clang-format off
230 // NOTE We overwrite dtype and qparam to _p.input
231 // This can be problematic if a single tensor has
232 // multiple different qparams. Let's fix later.
233 _p.input->dtype(quantized_dtype);
234 _p.input->quantparam(std::move(qparam));
235
236 replace(_p.dq_out).with(_p.input);
237 // clang-format on
238 }
239 }
240
241private:
242 const OnnxQDQPattern &_p;
243};

270bool QuantizeOnnxQDQPass::run(loco::Graph *g)
271{
272 bool changed = false;
273
274 for (auto node : loco::active_nodes(loco::output_nodes(g)))
275 {
276 if (auto circle_custom_out = dynamic_cast<luci::CircleCustomOut *>(node))
277 {
278 OnnxQDQPattern p(circle_custom_out);
279 if (p.matched())
280 {
281 QuantizeOnnxQDQ quantize(p);
282 quantize.apply();
283 changed = true;
284 }
285 }
286 }
287
288 return changed;
289}
