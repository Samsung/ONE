31struct OnnxQDQPattern final
32{
33public:
34 OnnxQDQPattern(luci::CircleCustomOut *candidate) { dq_out = candidate; }
35
36public:
37 bool matched()
38 {
39 if (not dq_out)
40 return false;
41
42 dq = loco::must_cast<luci::CircleCustom *>(dq_out->input());
43 if (not is_onnx_dequantize_linear(dq))
44 return false;
45
46 q_out = dynamic_cast<luci::CircleCustomOut *>(dq->inputs(0));
47 if (not q_out)
48 return false;
49
50 dq_scale = dynamic_cast<luci::CircleConst *>(dq->inputs(1));
51 if (not dq_scale)
52 return false;
53
54 dq_zerop = dynamic_cast<luci::CircleConst *>(dq->inputs(2));
55 if (not dq_zerop)
56 return false;
57
58 q = loco::must_cast<luci::CircleCustom *>(q_out->input());
59 if (not is_onnx_quantize_linear(q))
60 return false;
61
62 input = loco::must_cast<luci::CircleNode *>(q->inputs(0));
63 if (input->dtype() != loco::DataType::FLOAT32)
64 return false;
65
66 q_scale = dynamic_cast<luci::CircleConst *>(q->inputs(1));
67 if (not q_scale)
68 return false;
69
70 q_zerop = dynamic_cast<luci::CircleConst *>(q->inputs(2));
71 if (not q_zerop)
72 return false;
73
74 const auto q_dtype = q->dtype();
75 const auto q_scale_dtype = q_scale->dtype();
76 const auto q_zerop_dtype = q_zerop->dtype();
77 const auto dq_scale_dtype = dq_scale->dtype();
78 const auto dq_zerop_dtype = dq_zerop->dtype();
79
80 if (q_scale_dtype != loco::DataType::FLOAT32)
81 return false;
82
83 if (dq_scale_dtype != loco::DataType::FLOAT32)
84 return false;
85
86 // Invariant from onnx Quantize operator
87 if (q_dtype != q_zerop_dtype)
88 return false;
89
90 // Invariant from onnx Dequantize operator
91 if (q_dtype != dq_zerop_dtype)
92 return false;
93
94 // Check length of scale, zp = 1
95 if (q_scale->size<loco::DataType::FLOAT32>() != 1)
96 return false;
97
98 if (dq_scale->size<loco::DataType::FLOAT32>() != 1)
99 return false;
100
101 auto q_zerop_size = 0;
102 auto dq_zerop_size = 0;
103 switch (q_zerop_dtype)
104 {
105 case loco::DataType::S16:
106 q_zerop_size = q_zerop->size<loco::DataType::S16>();
107 dq_zerop_size = dq_zerop->size<loco::DataType::S16>();
108 break;
109 case loco::DataType::U8:
110 q_zerop_size = q_zerop->size<loco::DataType::U8>();
111 dq_zerop_size = dq_zerop->size<loco::DataType::U8>();
112 break;
113 default:
114 throw std::runtime_error("Unsupported zerop dtype in " + q_zerop->name());
115 }
116
117 if (q_zerop_size != 1)
118 return false;
119
120 if (dq_zerop_size != 1)
121 return false;
122
123 return true;
124 }
125
126public:
127 luci::CircleCustomOut *dq_out = nullptr;
128 luci::CircleCustom *dq = nullptr;
129 luci::CircleConst *dq_scale = nullptr;
130 luci::CircleConst *dq_zerop = nullptr;
131 luci::CircleCustomOut *q_out = nullptr;
132 luci::CircleCustom *q = nullptr;
133 luci::CircleConst *q_scale = nullptr;
134 luci::CircleConst *q_zerop = nullptr;
135 luci::CircleNode *input = nullptr;
136};

138class QuantizeOnnxQDQ final
139{
140public:
141 QuantizeOnnxQDQ(const OnnxQDQPattern &p) : _p(p) {}
142
143public:
144 void apply(void)
145 {
146 const auto quantized_dtype = _p.q->dtype();
147
148 // Get scale
149 assert(_p.q_scale->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
150 assert(_p.q_scale->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
151 const float q_scale = _p.q_scale->at<loco::DataType::FLOAT32>(0);
152
153 assert(_p.dq_scale->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
154 assert(_p.dq_scale->size<loco::DataType::FLOAT32>() == 1); // FIX_CALLER_UNLESS
155 const float dq_scale = _p.dq_scale->at<loco::DataType::FLOAT32>(0);
156
157 if (q_scale != dq_scale)
158 throw std::runtime_error("Invalid scale value in " + _p.dq_scale->name());
159
160 // Get zerop
161 int64_t q_zerop = 0;
162 int64_t dq_zerop = 0;
163 switch (quantized_dtype)
164 {
165 case loco::DataType::S16:
166 assert(_p.q_zerop->size<loco::DataType::S16>() == 1); // FIX_CALLER_UNLESS
167 assert(_p.dq_zerop->size<loco::DataType::S16>() == 1); // FIX_CALLER_UNLESS
168 q_zerop = _p.q_zerop->at<loco::DataType::S16>(0);
169 dq_zerop = _p.dq_zerop->at<loco::DataType::S16>(0);
170 break;
171 case loco::DataType::U8:
172 assert(_p.q_zerop->size<loco::DataType::U8>() == 1); // FIX_CALLER_UNLESS
173 assert(_p.dq_zerop->size<loco::DataType::U8>() == 1); // FIX_CALLER_UNLESS
174 q_zerop = _p.q_zerop->at<loco::DataType::U8>(0);
175 dq_zerop = _p.dq_zerop->at<loco::DataType::U8>(0);
176 break;
177 default:
178 throw std::runtime_error("Unsupported zerop dtype in " + _p.q_zerop->name());
179 }
180
181 if (q_zerop != dq_zerop)
182 throw std::runtime_error("Invalid zerop value in " + _p.dq_zerop->name());
183
184 auto qparam = std::make_unique<luci::CircleQuantParam>();
185 {
186 qparam->scale.push_back(q_scale);
187 qparam->zerop.push_back(q_zerop);
188 qparam->quantized_dimension = 0;
189 }
190
191 if (auto const_input = dynamic_cast<luci::CircleConst *>(_p.input))
192 {
193 assert(const_input->dtype() == loco::DataType::FLOAT32); // FIX_ME_UNLESS
194 const auto num_elem = const_input->size<loco::DataType::FLOAT32>();
195
196 auto new_const = luci::clone(const_input);
197 new_const->name(new_const->name() + "_quant");
198 add_origin(new_const, luci::get_origin(const_input));
199
200 new_const->dtype(quantized_dtype);
201
202 // Quantize const
203 switch (quantized_dtype)
204 {
205 case loco::DataType::S16:
206 {
207 new_const->size<loco::DataType::S16>(num_elem);
208
209 const int64_t max_val = std::numeric_limits<int16_t>::max();
210 const int64_t min_val = -max_val;
211 for (uint32_t i = 0; i < num_elem; i++)
212 {
213 const float fp_val = const_input->at<loco::DataType::FLOAT32>(i);
214 const int64_t q_val = std::round(fp_val / q_scale) + q_zerop;
215 new_const->at<loco::DataType::S16>(i) = std::min(max_val, std::max(min_val, q_val));
216 }
217 break;
218 }
219 default:
220 throw std::runtime_error("Unsupported quantized_dtype");
221 }
222 new_const->quantparam(std::move(qparam));
223
224 replace(_p.dq_out).with(new_const);
225 }
226 else
227 {
228 // clang-format off
229 // NOTE We overwrite dtype and qparam to _p.input
230 // This can be problematic if a single tensor has
231 // multiple different qparams. Let's fix later.
232 _p.input->dtype(quantized_dtype);
233 _p.input->quantparam(std::move(qparam));
234
235 replace(_p.dq_out).with(_p.input);
236 // clang-format on
237 }
238 }
239
240private:
241 const OnnxQDQPattern &_p;
242};

269bool QuantizeOnnxQDQPass::run(loco::Graph *g)
270{
271 bool changed = false;
272
273 for (auto node : loco::active_nodes(loco::output_nodes(g)))
274 {
275 if (auto circle_custom_out = dynamic_cast<luci::CircleCustomOut *>(node))
276 {
277 OnnxQDQPattern p(circle_custom_out);
278 if (p.matched())
279 {
280 QuantizeOnnxQDQ quantize(p);
281 quantize.apply();
282 changed = true;
283 }
284 }
285 }
286
287 return changed;
288}
