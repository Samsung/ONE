38void iterate_per_channel(CircleConst *node, IterFunc func)
39{
40 loco::TensorShape dimension;
41 dimension.rank(4);
42 uint32_t indices[4] = {
43 0,
44 };
45 int32_t channel_dim_index{0};
46
47 if (!get_channel_dim_index(node, dimension, channel_dim_index))
48 {
49 assert(false);
50 return;
51 }
52
53 for (indices[0] = 0; indices[0] < dimension.dim(0).value(); indices[0]++)
54 {
55 for (indices[1] = 0; indices[1] < dimension.dim(1).value(); indices[1]++)
56 {
57 for (indices[2] = 0; indices[2] < dimension.dim(2).value(); indices[2]++)
58 {
59 for (indices[3] = 0; indices[3] < dimension.dim(3).value(); indices[3]++)
60 {
61 func(indices, dimension, channel_dim_index);
62 }
63 }
64 }
65 }
66}

76void cal_minmax_per_channel(CircleConst *node, std::vector<float> &min, std::vector<float> &max)
77{
78 loco::TensorShape dimension;
79 dimension.rank(4);
80 int32_t channel_dim_index{0};
81
82 if (!get_channel_dim_index(node, dimension, channel_dim_index))
83 {
84 assert(false);
85 return;
86 }
87 auto size = dimension.dim(channel_dim_index).value();
88
89 std::vector<bool> has_min_max_value(size, false);
90 min.resize(size);
91 max.resize(size);
92
93 auto cal_minmax = [&](uint32_t *indices, loco::TensorShape &dimension, int channel_dim_index) {
94 int channel_idx = indices[channel_dim_index];
95 auto data = node->at<loco::DataType::FLOAT32>(cal_offset(dimension, indices));
96 if (has_min_max_value[channel_idx])
97 {
98 min[channel_idx] = data < min[channel_idx] ? data : min[channel_idx];
99 max[channel_idx] = data > max[channel_idx] ? data : max[channel_idx];
100 }
101 else
102 {
103 min[channel_idx] = data;
104 max[channel_idx] = data;
105 has_min_max_value[channel_idx] = true;
106 }
107 };
108
109 iterate_per_channel(node, cal_minmax);
110}

112void sym_wquant_per_channel(CircleConst *node, std::vector<float> &min, std::vector<float> &max,
113 std::vector<float> &scaling_factor, std::vector<int64_t> &zp,
114 std::vector<float> &nudged_min, std::vector<float> &nudged_max)
115{
116 assert(node->dtype() == loco::DataType::FLOAT32);
117 const int32_t kMaxScale = std::numeric_limits<int16_t>::max();
118 const int32_t kMinScale = -kMaxScale;
119
120 uint32_t size = node->size<loco::DataType::FLOAT32>();
121 std::vector<int32_t> quantized_values(size);
122
123 for (size_t i = 0; i < min.size(); ++i)
124 {
125 compute_sym_scale_zp(min[i], max[i], scaling_factor[i], zp[i], nudged_min[i], nudged_max[i]);
126 }
127
128 auto quantize = [&](uint32_t *indices, loco::TensorShape &dimension, int channel_dim_index) {
129 int channel_idx = indices[channel_dim_index];
130 const float scaling_factor_inv = 1.0 / scaling_factor[channel_idx];
131 auto data = node->at<loco::DataType::FLOAT32>(cal_offset(dimension, indices));
132 data = data < nudged_min[channel_idx] ? nudged_min[channel_idx] : data;
133 data = data > nudged_max[channel_idx] ? nudged_max[channel_idx] : data;
134 quantized_values[cal_offset(dimension, indices)] =
135 static_cast<int32_t>(std::round(data * scaling_factor_inv));
136 };
137
138 iterate_per_channel(node, quantize);
139
140 node->dtype(loco::DataType::S16); // change the type of tensor
141 node->size<loco::DataType::S16>(size); // resize tensor
142 for (uint32_t i = 0; i < size; ++i)
143 {
144 node->at<loco::DataType::S16>(i) =
145 std::min(kMaxScale, std::max(kMinScale, quantized_values[i]));
146 }
147}

172void asymmetric_wquant_per_channel(CircleConst *node, std::vector<float> &min,
173 std::vector<float> &max, std::vector<float> &scaling_factor,
174 std::vector<int64_t> &zp, std::vector<float> &nudged_min,
175 std::vector<float> &nudged_max)
176{
177 assert(node->dtype() == loco::DataType::FLOAT32);
178
179 const int32_t kMinScale = 0;
180 const int32_t kMaxScale = 255;
181
182 uint32_t size = node->size<loco::DataType::FLOAT32>();
183 std::vector<int32_t> quantized_values(size);
184
185 for (size_t i = 0; i < min.size(); ++i)
186 {
187 compute_asym_scale_zp(min[i], max[i], scaling_factor[i], zp[i], nudged_min[i], nudged_max[i]);
188 }
189
190 auto quantize = [&](uint32_t *indices, loco::TensorShape &dimension, int channel_dim_index) {
191 int channel_idx = indices[channel_dim_index];
192 const float scaling_factor_inv = 1.0 / scaling_factor[channel_idx];
193 auto data = node->at<loco::DataType::FLOAT32>(cal_offset(dimension, indices));
194 data = data < nudged_min[channel_idx] ? nudged_min[channel_idx] : data;
195 data = data > nudged_max[channel_idx] ? nudged_max[channel_idx] : data;
196 quantized_values[cal_offset(dimension, indices)] =
197 static_cast<int32_t>(std::round((data - nudged_min[channel_idx]) * scaling_factor_inv));
198 };
199
200 iterate_per_channel(node, quantize);
201
202 node->dtype(loco::DataType::U8); // change the type of tensor
203 node->size<loco::DataType::U8>(size); // resize tensor
204 for (uint32_t i = 0; i < size; ++i)
205 {
206 node->at<loco::DataType::U8>(i) = std::min(kMaxScale, std::max(kMinScale, quantized_values[i]));
207 }

257struct QuantizeDequantizeWeights final : public luci::CircleNodeMutableVisitor<void>
258{
259 QuantizeDequantizeWeights(loco::DataType input, loco::DataType output,
260 QuantizationGranularity granularity)
261 : input_type(input), output_type(output), granularity(granularity)
262 {
263 }
264
265 loco::DataType input_type;
266 loco::DataType output_type;
267 QuantizationGranularity granularity;
268
269private:
270 // Fake quantize weights (Only u8 quantization is supported for LWQ)
271 void fake_quantize_lwq(luci::CircleConst *weights) const
272 {
273 assert(output_type == loco::DataType::U8); // FIX_CALLER_UNLESS
274
275 // Find min/max per layer
276 float min = std::numeric_limits<float>::max();
277 float max = std::numeric_limits<float>::lowest();
278 for (uint32_t i = 0; i < weights->size<loco::DataType::FLOAT32>(); i++)
279 {
280 auto data = weights->at<loco::DataType::FLOAT32>(i);
281 min = data < min ? data : min;
282 max = data > max ? data : max;
283 }
284 float scaling_factor{0};
285 int64_t zp{0};
286 float nudged_min{0};
287 float nudged_max{0};
288
289 asymmetric_wquant_with_minmax_per_layer(weights, min, max, scaling_factor, zp, nudged_min,
290 nudged_max);
291 asymmetric_wdequant_with_minmax_per_layer(weights, scaling_factor, nudged_min);
292 auto quantparam = std::make_unique<CircleQuantParam>();
293 quantparam->min.push_back(nudged_min);
294 quantparam->max.push_back(nudged_max);
295 quantparam->scale.push_back(scaling_factor);
296 quantparam->zerop.push_back(zp);
297 weights->quantparam(std::move(quantparam));
298 }
299
300private:
301 // Fake quantize weights (u8/s16 quantization are supported for CWQ)
302 void fake_quantize_cwq(luci::CircleConst *weights) const
303 {
304 assert(output_type == loco::DataType::U8 ||
305 output_type == loco::DataType::S16); // FIX_CALLER_UNLESS
306
307 // Find min/max per channel
308 std::vector<float> min;
309 std::vector<float> max;
310
311 cal_minmax_per_channel(weights, min, max);
312
313 std::vector<float> nudged_min(min.size());
314 std::vector<float> nudged_max(min.size());
315 std::vector<float> scaling_factor(min.size());
316 std::vector<int64_t> zp(min.size());
317
318 if (output_type == loco::DataType::U8)
319 {
320 asymmetric_wquant_per_channel(weights, min, max, scaling_factor, zp, nudged_min, nudged_max);
321 asymmetric_wdequant_per_channel(weights, scaling_factor, nudged_min);
322 }
323 else
324 {
325 sym_wquant_per_channel(weights, min, max, scaling_factor, zp, nudged_min, nudged_max);
326 sym_wdequant_per_channel(weights, scaling_factor);
327 }
328
329 auto quantparam = std::make_unique<CircleQuantParam>();
330 quantparam->min = nudged_min;
331 quantparam->max = nudged_max;
332 quantparam->scale = scaling_factor;
333 quantparam->zerop = zp;
334 weights->quantparam(std::move(quantparam));
335 }
336
337private:
338 void fake_quantize(luci::CircleConst *weights) const
339 {
340 switch (granularity)
341 {
342 case luci::QuantizationGranularity::ChannelWise:
343 fake_quantize_cwq(weights);
344 break;
345 case luci::QuantizationGranularity::LayerWise:
346 fake_quantize_lwq(weights);
347 break;
348 default:
349 throw std::invalid_argument("Unsupported granularity");
350 }
351 }

353private:
354 // Check if
355 // 1. node is const
356 // 2. node's dtype is float32
357 bool is_quantizable(loco::Node *node)
358 {
359 auto const_node = dynamic_cast<luci::CircleConst *>(node);
360 if (not const_node)
361 return false;
362
363 // Skip if this is not float32
364 if (const_node->dtype() != loco::DataType::FLOAT32)
365 return false;
366
367 return true;
368 }

432bool QuantizeDequantizeWeightsPass::run(loco::Graph *g)
433{
434 LOGGER(l);
435 INFO(l) << "QuantizeDequantizeWeightsPass Start" << std::endl;
436
437 auto info_by_name = layer_info_map(g, _ctx->layers_info);
438
439 auto quantize_dtype = [&](const luci::CircleNode *node) {
440 auto iter = info_by_name.find(node->name());
441
442 // Return designated quantization dtype
443 if (iter != info_by_name.end())
444 return iter->second.dtype;
445
446 // Return default quantization dtype
447 return _ctx->output_model_dtype;
448 };
449
450 auto quantize_granularity = [&](const luci::CircleNode *node) {
451 auto iter = info_by_name.find(node->name());
452
453 // Return designated quantization granularity
454 if (iter != info_by_name.end())
455 return iter->second.granularity;
456
457 // Return default quantization granularity
458 return _ctx->granularity;
459 };
460
461 // Quantize weights
462 for (auto node : loco::active_nodes(loco::output_nodes(g)))
463 {
464 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
465 QuantizeDequantizeWeights qw(_ctx->input_model_dtype, quantize_dtype(circle_node),
466 quantize_granularity(circle_node));
467 circle_node->accept(&qw);
468 }
469
470 INFO(l) << "QuantizeDequantizeWeightsPass End" << std::endl;
471 return false; // one time run
472}
