35int handleAutoAlgorithm(arser::Arser &arser, mpqsolver::bisection::BisectionSolver &solver)
36{
37 solver.algorithm(mpqsolver::bisection::BisectionSolver::Algorithm::Auto);
38 auto data_path = arser.get<std::string>("--visq_file");
39 if (data_path.empty())
40 {
41 std::cerr << "ERROR: please provide visq_file for auto mode" << std::endl;
42 return false;
43 }
44 solver.setVisqPath(data_path);
45 return true;
46}

48int entry(int argc, char **argv)
49{
50 const std::string bisection_str = "--bisection";
51 const std::string patterns_str = "--patterns";
52 const std::string layernorm_str = "--u8_layernorm_with_s16_variance";
53 const std::string save_intermediate_str = "--save_intermediate";
54
55 arser::Arser arser("circle-mpqsolver provides light-weight methods for finding a high-quality "
56 "mixed-precision model within a reasonable time.");
57
58 arser::Helper::add_version(arser, print_version);
59 arser::Helper::add_verbose(arser);
60
61 // if patterns are set we don't need data
62 arser.add_argument("--data").required(false).default_value("").help("Path to the test data");
63 arser.add_argument("--data_format").required(false).help("Test data format (default: h5)");
64
65 arser.add_argument("--qerror_ratio")
66 .type(arser::DataType::FLOAT)
67 .default_value(0.5f)
68 .help("quantization error ratio ([0, 1])");
69
70 arser.add_argument(bisection_str)
71 .nargs(1)
72 .required(false)
73 .type(arser::DataType::STR)
74 .help("Single optional argument for bisection method. "
75 "Whether input node should be quantized to Q16: 'auto', 'true', 'false'.");
76
77 arser.add_argument(patterns_str)
78 .nargs(0)
79 .required(false)
80 .help("Argument to define patterns applied (LayerNorm is the only supported) ");
81
82 arser.add_argument(layernorm_str)
83 .nargs(0)
84 .required(false)
85 .help("Use int16 for computing variance in uint8 layer normalization");
86
87 arser.add_argument("--input_model")
88 .required(true)
89 .help("Input float model with min max initialized");
90
91 arser.add_argument("--input_dtype")
92 .type(arser::DataType::STR)
93 .default_value("uint8")
94 .help("Data type of quantized model's inputs (default: uint8)");
95
96 arser.add_argument("--output_dtype")
97 .type(arser::DataType::STR)
98 .default_value("uint8")
99 .help("Data type of quantized model's outputs (default: uint8)");
100
101 arser.add_argument("--output_model").required(true).help("Output quantized model");
102
103 arser.add_argument("--visq_file")
104 .type(arser::DataType::STR)
105 .default_value("")
106 .required(false)
107 .help("*.visq.json file with quantization errors");
108
109 arser.add_argument(save_intermediate_str)
110 .type(arser::DataType::STR)
111 .required(false)
112 .help("path to save intermediate results");
113
114 try
115 {
116 arser.parse(argc, argv);
117 }
118 catch (const std::runtime_error &err)
119 {
120 std::cerr << err.what() << std::endl;
121 std::cout << arser;
122 return EXIT_FAILURE;
123 }
124
125 if (arser.get<bool>("--verbose"))
126 {
127 // The third parameter of setenv means REPLACE.
128 // If REPLACE is zero, it does not overwrite an existing value.
129 setenv("LUCI_LOG", "100", 0);
130 }
131
132 auto data_path = arser.get<std::string>("--data");
133 auto input_model_path = arser.get<std::string>("--input_model");
134 auto output_model_path = arser.get<std::string>("--output_model");
135 auto input_dtype = arser.get<std::string>("--input_dtype");
136 auto output_dtype = arser.get<std::string>("--output_dtype");
137
138 float qerror_ratio = arser.get<float>("--qerror_ratio");
139 if (qerror_ratio < 0.0 || qerror_ratio > 1.f)
140 {
141 std::cerr << "ERROR: quantization ratio must be in [0, 1]" << std::endl;
142 return EXIT_FAILURE;
143 }
144
145 if (arser[bisection_str] && arser[patterns_str])
146 {
147 // only one solver can be used for now
148 std::cerr << "ERROR: only one method is allowed to use" << std::endl;
149 return EXIT_FAILURE;
150 }
151
152 SolverOutput::get() << ">> Searching mixed precision configuration \n"
153 << "model:" << input_model_path << "\n"
154 << "dataset: " << data_path << "\n"
155 << "input dtype: " << input_dtype << "\n"
156 << "output dtype: " << output_dtype << "\n";
157
158 std::unique_ptr<mpqsolver::MPQSolver> solver;
159 if (arser[bisection_str])
160 {
161 // optimize
162 SolverOutput::get() << "using bisection\n";
163
164 using namespace mpqsolver::bisection;
165 auto bi_solver =
166 std::make_unique<BisectionSolver>(data_path, qerror_ratio, input_dtype, output_dtype);
167
168 {
169 auto value = arser.get<std::string>(bisection_str);
170 if (value == "auto")
171 {
172 SolverOutput::get() << "algorithm: bisection (auto)\n";
173 if (!handleAutoAlgorithm(arser, *bi_solver))
174 {
175 return EXIT_FAILURE;
176 }
177 }
178 else if (value == "true")
179 {
180 SolverOutput::get() << "algorithm: bisection (Q16AtFront)";
181 bi_solver->algorithm(BisectionSolver::Algorithm::ForceQ16Front);
182 }
183 else if (value == "false")
184 {
185 SolverOutput::get() << "algorithm: bisection (Q8AtFront)";
186 bi_solver->algorithm(BisectionSolver::Algorithm::ForceQ16Back);
187 }
188 else
189 {
190 std::cerr << "ERROR: Unrecognized option for bisection algortithm" << input_model_path
191 << std::endl;
192 return EXIT_FAILURE;
193 }
194 }
195
196 if (arser[save_intermediate_str])
197 {
198 auto data_path = arser.get<std::string>(save_intermediate_str);
199 if (!data_path.empty())
200 {
201 bi_solver->set_save_intermediate(data_path);
202 }
203 }
204
205 SolverOutput::get() << "qerror metric: MAE\n"
206 << "target qerror ratio: " << qerror_ratio << "\n";
207
208 solver.reset(bi_solver.release());
209 }
210 else if (arser[patterns_str])
211 {
212 SolverOutput::get() << "using patterns\n";
213
214 std::vector<mpqsolver::pattern::QuantizationPattern> patterns;
215 if (arser[layernorm_str])
216 {
217 patterns.push_back(mpqsolver::pattern::QuantizationPattern::Q8LayerNormWithQ16Variance);
218 }
219 solver =
220 std::make_unique<mpqsolver::pattern::PatternSolver>(input_dtype, output_dtype, patterns);
221 }
222 else
223 {
224 std::cerr << "ERROR: Unrecognized solver" << std::endl;
225 return EXIT_FAILURE;
226 }
227
228 auto optimized = solver->run(input_model_path);
229 if (optimized == nullptr)
230 {
231 std::cerr << "ERROR: Failed to build mixed precision model" << input_model_path << std::endl;
232 return EXIT_FAILURE;
233 }
234
235 // save optimized
236 {
237 SolverOutput::get() << "Saving output model to " << output_model_path << "\n";
238 luci::CircleExporter exporter;
239 luci::CircleFileExpContract contract(optimized.get(), output_model_path);
240 if (!exporter.invoke(&contract))
241 {
242 std::cerr << "ERROR: Failed to export mixed precision model" << input_model_path << std::endl;
243 return EXIT_FAILURE;
244 }
245 }
246
247 return EXIT_SUCCESS;
248}
