34template <> void unpack<float>(tensorflow::TensorProto *input_tensor)
35{
36 const auto &input_shape = input_tensor->tensor_shape();
37 assert(input_shape.dim_size() <= 6);
38 int input_flat_size = tfkit::tf::GetElementCount(input_shape);
39
40 // Adjust where shape is not set but actual value exist
41 if (input_tensor->tensor_content().size() > 0 && input_flat_size == -1)
42 {
43 input_flat_size = input_tensor->tensor_content().size() / sizeof(float);
44 }
45
46 if (input_tensor->tensor_content().size() == 0)
47 {
48 // Do nothing as there is no tensor content to unpack
49 }
50 else if (input_tensor->tensor_content().size() == input_flat_size * sizeof(float))
51 {
52 input_tensor->clear_float_val();
53
54 const float *tensor_content =
55 reinterpret_cast<const float *>(input_tensor->tensor_content().data());
56 for (int i = 0; i < input_flat_size; i++)
57 {
58 input_tensor->add_float_val(tensor_content[i]);
59 }
60 input_tensor->clear_tensor_content();
61 }
62 else
63 {
64 throw std::runtime_error{"Number of elements mismatch in unpack<float>."};
65 // TODO: support for these
66 }
67}

174void unpack(tensorflow::GraphDef &graph_def)
175{
176 auto nodes = graph_def.mutable_node();
177 for (int i = 0; i < nodes->size(); ++i)
178 {
179 tensorflow::NodeDef *n = nodes->Mutable(i);
180 // TODO: handle for other operators
181 if (n->op() == "Const")
182 {
183 const auto dtype = tfkit::tf::GetDataTypeAttr(*n, "dtype");
184 tensorflow::TensorProto *tensor = tfkit::tf::GetTensorAttr(*n, "value");
185
186 switch (dtype)
187 {
188 case tensorflow::DT_FLOAT:
189 unpack<float>(tensor);
190 break;
191 case tensorflow::DT_INT32:
192 unpack<int32_t>(tensor);
193 break;
194 case tensorflow::DT_INT8:
195 unpack<int8_t>(tensor);
196 break;
197 case tensorflow::DT_BOOL:
198 unpack<bool>(tensor);
199 break;
200 default:
201 throw std::runtime_error{"Unsupported dtype"};
202 }
203 }
204 }
205}

212int UnpackCommand::run(int argc, const char *const *argv) const
213{
214 tensorflow::GraphDef graph_def;
215
216 CmdArguments cmdargs(argc, argv);
217
218 auto ioconfig = make_ioconfig(cmdargs);
219
220 google::protobuf::io::IstreamInputStream is{ioconfig->in()};
221
222 if (!google::protobuf::TextFormat::Parse(&is, &graph_def))
223 {
224 std::cerr << "ERROR: Failed to parse prototxt" << std::endl;
225 return 255;
226 }
227
228 // convert tensor_content to float_val
229 unpack(graph_def);
230
231 google::protobuf::io::OstreamOutputStream os{ioconfig->out()};
232 google::protobuf::TextFormat::Print(graph_def, &os);
233
234 return 0;
235}
