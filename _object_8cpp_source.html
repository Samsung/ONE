35void Object::def(Def *d)
36{
37 // This assert enforces users to explicitly reset def before update.
38 //
39 // Let's consider an object o with def d0.
40 //
41 // The following code is allowed:
42 // o->def(nullptr);
43 // o->def(d1);
44 //
45 // However, the following code is not allowed:
46 // o->def(d1);
47 //
48 assert((_def == nullptr) || (d == nullptr));
49 _def = d;
50}
