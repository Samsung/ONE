28luci::CircleConst *create_expanded_constant(luci::CircleConst *node, luci::CircleNode *successor)
29{
30 LOGGER(l);
31
32 if (successor->rank() != node->rank())
33 return nullptr;
34
35 std::vector<uint32_t> broadcast_dims;
36 for (uint32_t dim = 0; dim < node->rank(); ++dim)
37 {
38 if (node->dim(dim) == successor->dim(dim))
39 continue;
40
41 if (node->dim(dim) == 1)
42 broadcast_dims.push_back(dim);
43 }
44
45 if (broadcast_dims.size() != 1)
46 {
47 WARN(l) << "NYI: Only single dimension broadcast is supported";
48 return nullptr;
49 }
50
51 if (successor->rank() < 2 || successor->rank() > 4)
52 {
53 WARN(l) << "NYI: Only 2D/3D/4D tensor broadcast removal is supported";
54 return nullptr;
55 }
56
57 auto constant = node->graph()->nodes()->create<luci::CircleConst>();
58 constant->name(node->name());
59 constant->dtype(node->dtype());
60 constant->rank(node->rank());
61 constant->shape_status(luci::ShapeStatus::VALID);
62
63 uint32_t constant_size = 1;
64 for (uint32_t i = 0; i < successor->rank(); ++i)
65 {
66 constant->dim(i).set(successor->dim(i).value());
67 constant_size *= constant->dim(i).value();
68 }
69 constant->size<loco::DataType::FLOAT32>(constant_size);
70
71 auto const node_data = &node->at<loco::DataType::FLOAT32>(0);
72 auto const constant_data = &constant->at<loco::DataType::FLOAT32>(0);
73
74 assert(successor->rank() >= 2 && successor->rank() <= 4);
75
76 // Virtually extend the constant node to 4D to support all cases
77 // (Only for index calculation)
78 // Example. (2, 4) -> (1, 1, 2, 4)
79 auto const D0 = (successor->rank() < 4) ? 1 : successor->dim(successor->rank() - 4).value();
80 auto const D1 = (successor->rank() < 3) ? 1 : successor->dim(successor->rank() - 3).value();
81 auto const D2 = (successor->rank() < 2) ? 1 : successor->dim(successor->rank() - 2).value();
82 auto const D3 = successor->dim(successor->rank() - 1).value();
83
84 auto idx = luci::Array4DIndex(D0, D1, D2, D3);
85
86 auto const D0_orig = (node->rank() < 4) ? 1 : node->dim(node->rank() - 4).value();
87 auto const D1_orig = (node->rank() < 3) ? 1 : node->dim(node->rank() - 3).value();
88 auto const D2_orig = (node->rank() < 2) ? 1 : node->dim(node->rank() - 2).value();
89 auto const D3_orig = node->dim(node->rank() - 1).value();
90
91 auto idx_orig = luci::Array4DIndex(D0_orig, D1_orig, D2_orig, D3_orig);
92
93 for (uint32_t d0 = 0; d0 < D0; ++d0)
94 for (uint32_t d1 = 0; d1 < D1; ++d1)
95 for (uint32_t d2 = 0; d2 < D2; ++d2)
96 for (uint32_t d3 = 0; d3 < D3; ++d3)
97 {
98 auto const d0_orig = (D0_orig == 1) ? 0 : d0;
99 auto const d1_orig = (D1_orig == 1) ? 0 : d1;
100 auto const d2_orig = (D2_orig == 1) ? 0 : d2;
101 auto const d3_orig = (D3_orig == 1) ? 0 : d3;
102 constant_data[idx(d0, d1, d2, d3)] =
103 node_data[idx_orig(d0_orig, d1_orig, d2_orig, d3_orig)];
104 }
105
106 return constant;
107}

109template <typename N> bool expand_node_input(luci::CircleConst *node, luci::CircleNode *successor)
110{
111 static_assert(std::is_base_of<luci::CircleNode, N>::value,
112 "Successor node should have CircleNode base");
113
114 auto const successor_node = loco::must_cast<N *>(successor);
115 auto const successor_x = loco::must_cast<luci::CircleNode *>(successor_node->x());
116 auto const successor_y = loco::must_cast<luci::CircleNode *>(successor_node->y());
117
118 luci::CircleConst *expanded_const;
119
120 if (node == successor_x)
121 {
122 expanded_const = create_expanded_constant(node, successor_y);
123
124 if (expanded_const == nullptr)
125 return false;
126
127 successor_node->x(expanded_const);
128 }
129 else if (node == successor_y)
130 {
131 expanded_const = create_expanded_constant(node, successor_x);
132
133 if (expanded_const == nullptr)
134 return false;
135
136 successor_node->y(expanded_const);
137 }
138
139 return true;
140}

157bool expand_broadcast_const(luci::CircleConst *node)
158{
159 if (node->dtype() != loco::DataType::FLOAT32)
160 return false; // Unsupported data type
161
162 bool changed = false;
163
164 for (auto successor : loco::succs(node))
165 {
166 auto const circle_successor = loco::must_cast<luci::CircleNode *>(successor);
167 switch (circle_successor->opcode())
168 {
169 case luci::CircleOpcode::ADD:
170 if (expand_node_input<luci::CircleAdd>(node, circle_successor))
171 changed = true;
172 break;
173 case luci::CircleOpcode::MUL:
174 if (expand_node_input<luci::CircleMul>(node, circle_successor))
175 changed = true;
176 break;
177 case luci::CircleOpcode::DIV:
178 if (expand_node_input<luci::CircleDiv>(node, circle_successor))
179 changed = true;
180 break;
181 default:
182 break; // Unsupported successor node
183 }
184 }
185
186 return changed;
187}

197bool ExpandBroadcastConstPass::run(loco::Graph *g)
198{
199 bool changed = false;
200 for (auto node : loco::active_nodes(loco::output_nodes(g)))
201 {
202 auto const_node = dynamic_cast<luci::CircleConst *>(node);
203 if (const_node == nullptr)
204 continue;
205
206 if (expand_broadcast_const(const_node))
207 changed = true;
208 }
209 return changed;
210}
