31// Enum to indicate the degree(rank) to which part of the operation we will train:
32// this is an indicator of how much data of the current operation we will train
33// (for example, the entire operation, only the bias, only the upper half, and so on)
34enum OpTrainableRank
35{
36 ALL = 0, // 0 - Train all weights in the operation
37 ONLY_BIAS = 1, // 1 - Train bias only in the operation
38 UP_1_2_PART = 2, // 2 - Train the upper 1/2 part of the operation
39 LOWER_1_2_PART = 3, // 3 - Train the lower 1/2 part of the operation
40 MAX_VALUE = 4,
41 // TODO add more
42};

44// Information for saving the data necessary for training.
45// metrics_to_check_best_config - the metric by which the best configuration will be selected.
46// acceptable_diff - acceptable difference in metric values in order to select the best result in
47// memory. memory_above_restriction - the upper limit of memory that cannot be exceeded
48struct TrainData
49{
50 const char *circle_model_path = nullptr;
51 const char *wof_file_path = nullptr;
52 const char *output_tool_file_path = nullptr;
53 const char *input_input_train_data_path = nullptr;
54 const char *input_target_train_data_path = nullptr;
55 const char *input_input_test_data_path = nullptr;
56 const char *input_target_test_data_path = nullptr;
57 int32_t num_train_data_samples = 0;
58 int32_t num_test_data_samples = 0;
59 onert_micro::OMMetrics metrics_to_check_best_config = {};
60 float acceptable_diff = 0.01;
61 size_t memory_above_restriction = 0;
62};
63
64// Struct to save data which will be saved in result file
65struct TrainConfigFileData
66{
67 std::unordered_map<uint16_t, OpTrainableRank> trainable_op_indexes_with_ranks;
68};

70// Information that is the result of training
71// best_metrics_results - obtained best metric result during training
72// peak_memory_footprint - peak memory footprint obtained during training
73struct TrainResult
74{
75 std::pair<onert_micro::OMMetrics, float> best_metrics_results = {};
76 size_t peak_memory_footprint = 0;
77
78 TrainResult() = default;
79 explicit TrainResult(TrainData train_data)
80 {
81 peak_memory_footprint = std::numeric_limits<size_t>::max();
82 switch (train_data.metrics_to_check_best_config)
83 {
84 case onert_micro::ACCURACY:
85 best_metrics_results = {onert_micro::ACCURACY, 0.f};
86 break;
87 case onert_micro::CROSS_ENTROPY_METRICS:
88 best_metrics_results = {onert_micro::CROSS_ENTROPY_METRICS,
89 std::numeric_limits<float>::max()};
90 break;
91 case onert_micro::MSE_METRICS:
92 best_metrics_results = {onert_micro::MSE_METRICS, std::numeric_limits<float>::max()};
93 break;
94 case onert_micro::MAE_METRICS:
95 best_metrics_results = {onert_micro::MAE_METRICS, std::numeric_limits<float>::max()};
96 break;
97 default:
98 assert(false && "Unsupported type");
99 break;
100 }
101 }
102};
