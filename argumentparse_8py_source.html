49 def add_argument(self, *args, **kwargs):
50 if not 'action' in kwargs:
51 raise RuntimeError('"action" keyword argument is required')
52 action = kwargs['action']
53 if not action in self._SUPPORTED_ACTIONS:
54 raise RuntimeError('Invalid action')
55 if not args:
56 raise RuntimeError('Invalid option name')
57 dtype = kwargs.get('dtype', str)
58 # use first option.
59 arg = args[0]
60 if dtype == bool and action != NormalOption:
61 raise RuntimeError('Only normal option can be boolean type')
62 if dtype == bool and not all(a.startswith('-') for a in args):
63 raise RuntimeError('Boolean type option should start with dash("-")')
64
65 if action == DriverName:
66 self.driver = arg
67 else:
68 self._actions.append((arg, kwargs['action'], dtype))
69
70 def make_cmd(self, cfg_args: SimpleNamespace) -> List:
71 assert self.target, "Target should be set before making commands"
72 assert self.driver, "Driver should be set before making commands"
73 # find driver path
74 driver_name = self.driver
75 driver_list = backends.get_list(driver_name)
76 if not driver_list:
77 driver_list = [shutil.which(driver_name)]
78 if not driver_list:
79 raise FileNotFoundError(f'{driver_name} not found')
80 # use first driver
81 driver_path = driver_list[0]
82 cmd: List = [driver_path]
83 # traverse the action in order and make commands
84 for action in self._actions:
85 arg, act, dtype = action
86 assert act in [NormalOption, TargetOption]
87 # positional input doesn't have dash(-) in the string
88 option_name = arg
89 # optional input
90 if arg.startswith('--'):
91 option_name = arg[len('--'):]
92 elif arg.startswith('-'):
93 option_name = arg[len('-'):]
94
95 if act == NormalOption:
96 if not oneutils.is_valid_attr(cfg_args, option_name):
97 # TODO raise error when invalid option is given in the cfg file.
98 continue
99 if dtype == bool and getattr(cfg_args, option_name).lower() == "false":
100 continue
101 if arg.startswith('-'):
102 cmd += [arg]
103 if act == TargetOption:
104 cmd += [self.target]
105 else:
106 assert act == NormalOption
107 if dtype == str:
108 cmd += [getattr(cfg_args, option_name)]
109 return cmd
