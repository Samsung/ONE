16"""
17This is for the command schema feature.
18
19_one-cmds_ has lots of tools such as one-import, one-optimize, etc.
20They have their own section in the configuration file and users can
21 give arguments with key-value pairs.
22
23But, backend tools such as one-codegen and one-profile hasn't the same
24 mechanism. Rather, they should pass all the arguments with `command` key
25 because _onecc_ can't know the backends' interface in advance.
26
27The command schema has been introduced for resolving these difficulties.
28If users provide _onecc_ with the command schema that describes the interface
29 of the backend, users can give arguments with key-value paris like other tools.
30
31NOTE. Command schema feature works only when target option is given.
32
33[AS-IS]
34
35# example.cfg
36[backend]
37target=my_target
38
39[one-codegen]
40backend=my_backend
41commnad=--output sample.tvn sample.circle
42
43[TO-BE]
44
45# /usr/share/one/backends/command/my_backend/codegen.py
46from onelib import argumentparse
47from onelib.argumentparse import DriverName, NormalOption, TargetOption
48
49
50def command_schema():
51 parser = argumentparse.ArgumentParser()
52 parser.add_argument("my_backend-compile", action=DriverName)
53 parser.add_argument("--output", action=NormalOption)
54 parser.add_argument("input", action=NormalOption)
55
56 return parser
57
58# /usr/share/one/target/my_target.ini
59TARGET=my_target
60BACKEND=my_backend
61
62# example.cfg
63[one-codegen]
64output=sample.tvn
65input=sample.circle
66
67
68---
69
70Command schema file should define `command_schema` function. And, you can add
71 arguments by calling `add_argument`. You should specify an action according to
72the option category.
73
74[Action List]
75- DriverName: the name of backend driver
76- TargetOption: the target option of the drvier.
77- NormalOption: the option of the driver. Starting with dash('-') implies the option
78 is optional rather than positional one.
79"""

109 def __init__(self):

114 def print_help(self):
115 backends_list = backends.get_list(self.driver)
116 driver_path = None
117 for cand in backends_list:
118 if ntpath.basename(cand) == self.driver:
119 driver_path = cand
120 if not driver_path:
121 driver_path = shutil.which(self.driver)
122
123 if not driver_path:
124 raise FileNotFoundError(self.driver + ' not found')
125
126 oneutils.run([driver_path, '-h'], err_prefix=self.driver)
127
128 def add_argument(self, *args, **kwargs):
129 if not 'action' in kwargs:
130 raise RuntimeError('"action" keyword argument is required')
131 action = kwargs['action']
132 if not action in self._SUPPORTED_ACTIONS:
133 raise RuntimeError('Invalid action')
134 if not args:
135 raise RuntimeError('Invalid option name')
136 dtype = kwargs.get('dtype', str)
137 # use first option.
138 arg = args[0]
139 if dtype == bool and action != NormalOption:
140 raise RuntimeError('Only normal option can be boolean type')
141 if dtype == bool and not all(a.startswith('-') for a in args):
142 raise RuntimeError('Boolean type option should start with dash("-")')
143
144 if action == DriverName:
145 self.driver = arg
146 else:
147 self._actions.append((arg, kwargs['action'], dtype))
148
149 def make_cmd(self, cfg_args: SimpleNamespace) -> List:
150 assert self.target, "Target should be set before making commands"
151 assert self.driver, "Driver should be set before making commands"
152 # find driver path
153 driver_name = self.driver
154 driver_list = backends.get_list(driver_name)
155 if not driver_list:
156 driver_list = [shutil.which(driver_name)]
157 if not driver_list:
158 raise FileNotFoundError(f'{driver_name} not found')
159 # use first driver
160 driver_path = driver_list[0]
161 cmd: List = [driver_path]
162 invalid_options = list(cfg_args.__dict__.keys())
163 # traverse the action in order and make commands
164 for action in self._actions:
165 arg, act, dtype = action
166 assert act in [NormalOption, TargetOption]
167 # positional input doesn't have dash(-) in the string
168 option_name = arg
169 # optional input
170 if arg.startswith('--'):
171 option_name = arg[len('--'):]
172 elif arg.startswith('-'):
173 option_name = arg[len('-'):]
174
175 if act == NormalOption:
176 if not oneutils.is_valid_attr(cfg_args, option_name):
177 continue
178 if dtype == bool and getattr(cfg_args, option_name).lower() == "false":
179 continue
180 if arg.startswith('-'):
181 cmd += [arg]
182 if act == TargetOption:
183 cmd += [self.target]
184 else:
185 assert act == NormalOption
186 if dtype == str:
187 cmd += [getattr(cfg_args, option_name)]
188 invalid_options.remove(option_name)
189 if len(invalid_options):
190 print(f'WARNING: there are invalid options {invalid_options}')
191 self.print_help()
192 return cmd
