43bool use_predefined_values(ActivationQType qtype)
44{
45 switch (qtype)
46 {
47 case ActivationQType::PreDefinedLogistic:
48 case ActivationQType::PreDefinedTanh:
49 case ActivationQType::PreDefinedSoftmax:
50 return true;
51 default:
52 // This ensures this switch-statement handles all ActivationQTypes
53 assert(qtype == ActivationQType::IntScale or qtype == ActivationQType::MinMax);
54 break;
55 }
56
57 return false;
58}

60// Create a Quantize Op whose
61// dtype is out_type
62// shape is the same with node
63// qparam is computed according to node's qtype
64luci::CircleQuantize *create_quantize_op(luci::CircleNode *node, loco::DataType out_type)
65{
66 auto quantize = node->graph()->nodes()->create<CircleQuantize>();
67 quantize->name(node->name() + "_Quantize");
68 quantize->dtype(out_type);
69 quantize->rank(node->rank());
70 for (uint32_t i = 0; i < node->rank(); i++)
71 quantize->dim(i).set(node->dim(i).value());
72
73 quantize->shape_status(luci::ShapeStatus::VALID);
74
75 auto qparam = node->quantparam();
76 assert(qparam); // FIX_CALLER_UNLESS
77
78 auto qtype = luci::activation_qtype(node);
79 if (use_predefined_values(qtype))
80 {
81 quantize->quantparam(luci::make_predefined_qparam(qtype, out_type));
82 return quantize;
83 }
84
85 assert(qtype == ActivationQType::MinMax or qtype == ActivationQType::IntScale);
86
87 assert(qparam->min.size() == 1); // FIX_CALLER_UNLESS
88 assert(qparam->max.size() == 1); // FIX_CALLER_UNLESS
89 auto min = qparam->min[0];
90 auto max = qparam->max[0];
91
92 float scaling_factor{0};
93 int64_t zp{0};
94 float nudged_min{0};
95 float nudged_max{0};
96
97 if (out_type == loco::DataType::U8)
98 {
99 compute_asym_scale_zp(min, max, scaling_factor, zp, nudged_min, nudged_max);
100 }
101 else
102 {
103 assert(out_type == loco::DataType::S16);
104 compute_sym_scale_zp(min, max, scaling_factor, zp, nudged_min, nudged_max);
105 }
106
107 auto quantparam = std::make_unique<CircleQuantParam>();
108 quantparam->scale.push_back(scaling_factor);
109 quantparam->zerop.push_back(zp);
110 // Save original min/max (not nudged_min/max). Nudged min/max
111 // is different from the real min/max values, causing wrong
112 // qparam when quantization dtype is changed.
113 quantparam->min.push_back(min);
114 quantparam->max.push_back(max);
115
116 quantize->quantparam(std::move(quantparam));
117
118 if (qtype == ActivationQType::IntScale)
119 set_int_scale(quantize);
120
121 return quantize;
122}

124// Create Dequantize Op whose shape is the same with node
125luci::CircleDequantize *create_dequantize(luci::CircleNode *node)
126{
127 auto dequantize = node->graph()->nodes()->create<luci::CircleDequantize>();
128 dequantize->name(node->name() + "_Dequantize");
129 dequantize->dtype(loco::DataType::FLOAT32);
130 dequantize->rank(node->rank());
131 for (uint32_t i = 0; i < node->rank(); i++)
132 dequantize->dim(i).set(node->dim(i).value());
133
134 dequantize->shape_status(luci::ShapeStatus::VALID);
135

137
138 return dequantize;
139}

163struct InsertQuantizeOp final : public luci::CircleNodeMutableVisitor<void>
164{
165 InsertQuantizeOp(loco::DataType default_dtype, loco::DataType op_dtype)
166 : _default_dtype(default_dtype), _op_dtype(op_dtype)
167 {
168 assert(default_dtype != op_dtype); // FIX_CALLER_UNLESS
169 }
170
171private:
172 loco::DataType _default_dtype;
173 loco::DataType _op_dtype;
174
175private:
176 luci::CircleQuantize *create_in_quantize(loco::Node *in, loco::Node *origin)
177 {
178 auto input = loco::must_cast<luci::CircleNode *>(in);
179 if (input->opcode() == luci::CircleOpcode::CIRCLECONST)
180 return nullptr;
181
182 // input is not quantizable (ex: index)
183 if (input->quantparam() == nullptr)
184 return nullptr;
185
186 auto input_quant = create_quantize_op(input, _op_dtype);
187 input_quant->input(input);
188 auto origin_node = loco::must_cast<luci::CircleNode *>(origin);
189 luci::add_origin(input_quant, luci::get_origin(origin_node));
190 return input_quant;
191 }
192
193 void insert_out_quantize(loco::Node *node)
194 {
195 auto output = loco::must_cast<luci::CircleNode *>(node);
196 assert(output->opcode() != luci::CircleOpcode::CIRCLECONST); // FIX_CALLER_UNLESS
197
198 // output is not quantizable (ex: index)
199 if (output->quantparam() == nullptr)
200 return;
201
202 auto output_quant = create_quantize_op(output, _default_dtype);
203
204 luci::add_origin(output_quant, luci::get_origin(output));
205 loco::replace(node).with(output_quant);
206 output_quant->input(node);
207 }

209// INPUT_NAME is the only activation of NODE
210#define INSERT_QUANTIZE_TO_UNARY_OP(NODE, INPUT_NAME) \
211 void visit(NODE *node) \
212 { \
213 if (auto input_quant = create_in_quantize(node->INPUT_NAME(), node)) \
214 node->INPUT_NAME(input_quant); \
215 \
216 insert_out_quantize(node); \
217 }
218
219// INPUT_NAME is the only activation of NODE
220#define INSERT_QUANTIZE_TO_UNARY_MULTI_OUTPUT_OP(NODE, INPUT_NAME, OUT_NAME) \
221 void visit(NODE *node) \
222 { \
223 if (auto input_quant = create_in_quantize(node->INPUT_NAME(), node)) \
224 node->INPUT_NAME(input_quant); \
225 \
226 auto out_nodes = loco::succs(node); \
227 for (auto out_node : out_nodes) \
228 { \
229 auto out_circle = loco::must_cast<OUT_NAME *>(out_node); \
230 insert_out_quantize(out_circle); \
231 } \
232 }
233
234// INPUT_NAME1 and INPUT_NAME2 are the only activations of NODE
235#define INSERT_QUANTIZE_TO_BINARY_OP(NODE, INPUT_NAME1, INPUT_NAME2) \
236 void visit(NODE *node) \
237 { \
238 if (auto input1_quant = create_in_quantize(node->INPUT_NAME1(), node)) \
239 node->INPUT_NAME1(input1_quant); \
240 \
241 if (auto input2_quant = create_in_quantize(node->INPUT_NAME2(), node)) \
242 node->INPUT_NAME2(input2_quant); \
243 \
244 insert_out_quantize(node); \
245 }

373void QuantizeWithMinMaxPass::set_input_type(loco::Graph *g) const
374{
375 auto inputs = g->inputs();
376 for (auto node : loco::input_nodes(g))
377 {
378 auto input = loco::must_cast<luci::CircleInput *>(node);
379 if (input->dtype() == _ctx->input_type)
380 continue;
381
382 // Bool type is not quantizable
383 if (input->dtype() == loco::DataType::BOOL)
384 continue;
385 if (input->dtype() == loco::DataType::S32)
386 continue;
387 if (input->dtype() == loco::DataType::S64)
388 continue;
389
390 // Insert Quantize Op
391 auto quant_op = create_quantize_op(input, input->dtype());
392 loco::replace(input).with(quant_op);
393 quant_op->input(input);
394
395 // TODO Set a proper origin (Quantize should have its own Origin)
396 {
397 auto succs = loco::succs(quant_op);
398 assert(succs.size() > 0);
399 auto succ = loco::must_cast<luci::CircleNode *>(*succs.begin());
400 luci::add_origin(quant_op, luci::get_origin(succ));
401 }
402
403 // Update qparam of input
404 // This step is skipped if input_type is float32
405 if (_ctx->input_type != loco::DataType::FLOAT32)
406 {
407 auto quantparam = input->quantparam();
408 assert(quantparam);
409 assert(quantparam->min.size() == 1); // only support layer-wise quant
410 assert(quantparam->max.size() == 1); // only support layer-wise quant
411 auto min = quantparam->min[0];
412 auto max = quantparam->max[0];
413
414 float scaling_factor{0};
415 int64_t zp{0};
416 float nudged_min{0};
417 float nudged_max{0};
418
419 if (_ctx->input_type == loco::DataType::U8)
420 {
421 compute_asym_scale_zp(min, max, scaling_factor, zp, nudged_min, nudged_max);
422 }
423 else
424 {
425 assert(_ctx->input_type == loco::DataType::S16);
426 compute_sym_scale_zp(min, max, scaling_factor, zp, nudged_min, nudged_max);
427 }
428 input->quantparam()->scale[0] = scaling_factor;
429 input->quantparam()->zerop[0] = zp;
430 }
431
432 // Update dtype of input
433 input->dtype(_ctx->input_type);
434
435 auto graph_input = inputs->at(input->index());
436 graph_input->dtype(_ctx->input_type);
437 }
438}

440void QuantizeWithMinMaxPass::set_output_type(loco::Graph *g) const
441{
442 auto outputs = g->outputs();
443 for (auto node : loco::output_nodes(g))
444 {
445 auto output = loco::must_cast<luci::CircleOutput *>(node);
446 if (output->dtype() == _ctx->output_type)
447 continue;
448
449 // Bool type is not quantizable
450 if (output->dtype() == loco::DataType::BOOL)
451 continue;
452
453 auto from = loco::must_cast<luci::CircleNode *>(output->from());
454
455 // The last Op is not quantizable (ex: ArgMax)
456 if (not from->quantparam())
457 continue;
458
459 // Insert Dequantize Op for float32 output_type
460 if (_ctx->output_type == loco::DataType::FLOAT32)
461 {
462 auto dequant_op = create_dequantize(from);
463 loco::replace(from).with(dequant_op);
464 dequant_op->input(from);
465 }
466 else
467 {
468 // Insert Quantize Op for non-float32 output_type
469 auto quant_op = create_quantize_op(from, _ctx->output_type);
470 loco::replace(from).with(quant_op);
471 quant_op->input(from);
472
473 // TODO Set a proper origin (Quantize should have its own Origin)
474 luci::add_origin(quant_op, luci::get_origin(from));
475 }
476
477 // Update dtype of output
478 output->dtype(_ctx->output_type);
479
480 auto graph_output = outputs->at(output->index());
481 graph_output->dtype(_ctx->output_type);
482 }

531bool QuantizeWithMinMaxPass::run(loco::Graph *g)
532{
533 LOGGER(l);
534 INFO(l) << "QuantizeWithMinMaxPass Start" << std::endl;
535
536 auto info_by_name = layer_info_map(g, _ctx->layers_info);
537
538 auto quantize_dtype = [&](const luci::CircleNode *node) {
539 auto iter = info_by_name.find(node->name());
540
541 // Return designated quantization dtype
542 if (iter != info_by_name.end())
543 return iter->second.dtype;
544
545 // Return default quantization dtype
546 return _ctx->output_model_dtype;
547 };
548
549 auto quantize_granularity = [&](const luci::CircleNode *node) {
550 auto iter = info_by_name.find(node->name());
551
552 // Return designated quantization granularity
553 if (iter != info_by_name.end())
554 return iter->second.granularity;
555
556 // Return default quantization granularity
557 return _ctx->granularity;
558 };
559
560 // Quantize activation
561 for (auto node : loco::active_nodes(loco::output_nodes(g)))
562 {
563 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
564 QuantizeActivation qa(_ctx->input_model_dtype, quantize_dtype(circle_node));
565 circle_node->accept(&qa);
566 }
567
568 // Insert Quantize Op
569 for (auto node : loco::active_nodes(loco::output_nodes(g)))
570 {
571 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
572 auto op_dtype = quantize_dtype(circle_node);
573 if (op_dtype != _ctx->output_model_dtype)
574 {
575 InsertQuantizeOp iqo(_ctx->output_model_dtype, op_dtype);
576 circle_node->accept(&iqo);
577 }
578 }
579
580 // Remove redundant Quantize Op
581 {
582 logo::Phase phase;
583
584 phase.emplace_back(std::make_unique<luci::RemoveRedundantQuantizePass>());
585
586 ProgressReporter prog(g, logo::PhaseStrategy::Saturate);
587 logo::PhaseRunner<logo::PhaseStrategy::Saturate> phase_runner{g};
588 phase_runner.attach(&prog);
589 phase_runner.run(phase);
590 }
591
592 // Backward propagation of activation qparam
593 {
594 PropagateQParamBackwardPass pqbp(_ctx->output_model_dtype);
595 pqbp.run(g);
596 }
597
598 // Quantize const input activation
599 for (auto node : loco::active_nodes(loco::output_nodes(g)))
600 {
601 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
602 QuantizeConstInputActivation qcia(quantize_dtype(circle_node));
603 circle_node->accept(&qcia);
604 }
605
606 // Update qparam of output of special Ops
607 for (auto node : loco::active_nodes(loco::output_nodes(g)))
608 {
609 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
610 QuantizeSpecialActivation qsa(_ctx->input_model_dtype, quantize_dtype(circle_node));
611 circle_node->accept(&qsa);
612 }
613
614 // Forward propagation of activation qparam
615 logo::Phase phase;
616
617 phase.emplace_back(std::make_unique<luci::PropagateQParamForwardPass>(_ctx->TF_style_maxpool));
618
619 ProgressReporter prog(g, logo::PhaseStrategy::Saturate);
620 logo::PhaseRunner<logo::PhaseStrategy::Saturate> phase_runner{g};
621 phase_runner.attach(&prog);
622 phase_runner.run(phase);
623
624 // Quantize weights
625 for (auto node : loco::active_nodes(loco::output_nodes(g)))
626 {
627 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
628 QuantizeWeights qw(_ctx->input_model_dtype, quantize_dtype(circle_node),
629 quantize_granularity(circle_node));
630 circle_node->accept(&qw);
631 }
632
633 // Quantize bias
634 for (auto node : loco::active_nodes(loco::output_nodes(g)))
635 {
636 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
637 QuantizeBias qb(_ctx->input_model_dtype, quantize_dtype(circle_node),
638 quantize_granularity(circle_node));
639 circle_node->accept(&qb);
640 }
641
642 // Update output dtype
643 auto graph_outputs = g->outputs();
644 for (auto node : loco::output_nodes(g))
645 {
646 auto circle_node = loco::must_cast<luci::CircleOutput *>(node);
647 if (static_cast<luci::CircleNode *>(circle_node->from())->dtype() == _ctx->output_model_dtype)
648 {
649 circle_node->dtype(_ctx->output_model_dtype);
650 auto graph_output = graph_outputs->at(circle_node->index());
651 graph_output->dtype(_ctx->output_model_dtype);
652 }
653 }
654
655 // Set input type
656 set_input_type(g);
657
658 // Set output type
659 set_output_type(g);
660
661 // Remove redundant Quantize Op
662 {
663 logo::Phase phase;
664
665 phase.emplace_back(std::make_unique<luci::RemoveRedundantQuantizePass>());
666
667 ProgressReporter prog(g, logo::PhaseStrategy::Saturate);
668 logo::PhaseRunner<logo::PhaseStrategy::Saturate> phase_runner{g};
669 phase_runner.attach(&prog);
670 phase_runner.run(phase);
671 }
672
673 // Remove min/max values
