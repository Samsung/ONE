47void DBus::on_bus_acquired(GDBusConnection *conn, const gchar *name, gpointer user_data)
48{
49 VERBOSE(DBus) << name << " on bus acquired" << std::endl;
50
51 GError *error = NULL;
52 NpudCore *core = npud_core_skeleton_new();
53 NpudCoreIface *iface = NPUD_CORE_GET_IFACE(core);
54
55 iface->handle_device_get_available_list = &on_handle_device_get_available_list;
56 iface->handle_context_create = &on_handle_context_create;
57 iface->handle_context_destroy = &on_handle_context_destroy;
58 iface->handle_buffers_create = &on_handle_buffers_create;
59 iface->handle_buffers_destroy = &on_handle_buffers_destroy;
60 iface->handle_network_create = &on_handle_network_create;
61 iface->handle_network_destroy = &on_handle_network_destroy;
62 iface->handle_request_create = &on_handle_request_create;
63 iface->handle_request_destroy = &on_handle_request_destroy;
64 iface->handle_request_set_data = &on_handle_request_set_data;
65 iface->handle_execute_run = &on_handle_execute_run;
66
67 if (!g_dbus_interface_skeleton_export(G_DBUS_INTERFACE_SKELETON(core), conn, "/org/tizen/npud",
68 &error))
69 {
70 VERBOSE(DBus) << "Failed to export skeleton, Server will stop." << std::endl;
71 Server::instance().stop();
72 }
73
74 _isReady.exchange(true);
75}

234gboolean DBus::on_handle_request_set_data(NpudCore *object, GDBusMethodInvocation *invocation,
235 guint64 arg_ctx, guint arg_rq_handle,
236 GVariant *arg_input_buffers, GVariant *arg_output_buffers)
237{
238 VERBOSE(DBus) << "on_handle_request_set_data with " << arg_ctx << ", " << arg_rq_handle
239 << std::endl;
240 GVariantIter *iter = NULL;
241 InputBuffers inBufs;
242 OutputBuffers outBufs;
243 gint32 type;
244 guint64 addr;
245 guint32 size;
246 int index = 0;
247
248 // inBufs
249 g_variant_get(arg_input_buffers, "a(itu)", &iter);
250 index = 0;
251 while (iter != NULL && g_variant_iter_loop(iter, "(itu)", &type, &addr, &size))
252 {
253 VERBOSE(DBus) << "in [" << index << "] Type: " << type << ", Addr: " << (void *)addr
254 << ", Size: " << size << std::endl;
255 if (type == 0) // NPU_BUFFER_MAPPED
256 {
257 inBufs.buffers[index].addr = reinterpret_cast<void *>(addr);
258 }
259 else if (type == 1) // NPU_BUFFER_DMABUF
260 {
261 // TODO Support dma buffer
262 VERBOSE(DBus) << "[NYI] NPU_BUFFER_DMABUF" << std::endl;
263 continue;
264 }
265 else
266 {
267 VERBOSE(DBus) << "Wrong buffer type. Ignored." << std::endl;
268 continue;
269 }
270 inBufs.buffers[index].size = size;
271 inBufs.buffers[index].type = static_cast<BufferTypes>(type);
272 index++;
273 }
274 inBufs.numBuffers = index;
275 g_variant_iter_free(iter);
276
277 // outBufs
278 g_variant_get(arg_output_buffers, "a(itu)", &iter);
279 index = 0;
280 while (iter != NULL && g_variant_iter_loop(iter, "(itu)", &type, &addr, &size))
281 {
282 VERBOSE(DBus) << "out [" << index << "] Type: " << type << ", Addr: " << (void *)addr
283 << ", Size: " << size << std::endl;
284 if (type == 0) // NPU_BUFFER_MAPPED
285 {
286 outBufs.buffers[index].addr = reinterpret_cast<void *>(addr);
287 }
288 else if (type == 1) // NPU_BUFFER_DMABUF
289 {
290 // TODO Support dma buffer
291 VERBOSE(DBus) << "[NYI] NPU_BUFFER_DMABUF" << std::endl;
292 continue;
293 }
294 else
295 {
296 VERBOSE(DBus) << "Wrong buffer type. Ignored." << std::endl;
297 continue;
298 }
299 outBufs.buffers[index].size = size;
300 outBufs.buffers[index].type = static_cast<BufferTypes>(type);
301 index++;
302 }
303 outBufs.numBuffers = index;
304 g_variant_iter_free(iter);
305
306 // TODO Invoke Core function.
307 int ret = -1;
308 npud_core_complete_request_set_data(object, invocation, ret);
309 return TRUE;
310}
