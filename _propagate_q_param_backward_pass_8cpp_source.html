31void quant_const_values(luci::CircleConst *const_node, float scaling_factor, float zerop,
32 loco::DataType quant_type)
33{
34 uint32_t size = const_node->size<loco::DataType::FLOAT32>();
35
36 const float scaling_factor_inv = 1.0 / scaling_factor;
37 std::vector<int32_t> quantized_values(size);
38 for (uint32_t i = 0; i < size; ++i)
39 {
40 auto data = static_cast<double>(const_node->at<loco::DataType::FLOAT32>(i));
41 double quantized_data = std::round(data * scaling_factor_inv) + zerop;
42 constexpr double int_max = static_cast<double>(std::numeric_limits<int32_t>::max());
43 constexpr double int_min = static_cast<double>(std::numeric_limits<int32_t>::min());
44 quantized_data = std::min(int_max, std::max(int_min, quantized_data));
45
46 quantized_values[i] = static_cast<int32_t>(quantized_data);
47 }
48
49 switch (quant_type)
50 {
51 case loco::DataType::U8:
52 const_node->dtype(loco::DataType::U8); // change the type of tensor
53 const_node->size<loco::DataType::U8>(size); // resize tensor
54 for (uint32_t i = 0; i < size; ++i)
55 const_node->at<loco::DataType::U8>(i) = std::min(255, std::max(0, quantized_values[i]));
56 break;
57 case loco::DataType::S16:
58 assert(zerop == 0);
59 const_node->dtype(loco::DataType::S16); // change the type of tensor
60 const_node->size<loco::DataType::S16>(size); // resize tensor
61 for (uint32_t i = 0; i < size; ++i)
62 const_node->at<loco::DataType::S16>(i) =
63 std::min(32767, std::max(-32767, quantized_values[i]));
64 break;
65 default:
66 throw std::runtime_error("Unsupported data type");
67 }
68}

70void overwrite_quantparam(const luci::CircleNode *source, luci::CircleNode *target)
71{
72 auto source_qparam = source->quantparam();
73 if (source_qparam == nullptr)
74 throw std::runtime_error("source quantparam is not found during overwrite");
75
76 auto target_qparam = target->quantparam();
77 if (target_qparam == nullptr)
78 {
79 auto quantparam = std::make_unique<luci::CircleQuantParam>();
80 target->quantparam(std::move(quantparam));
81 target_qparam = target->quantparam();
82
83 if (target_qparam == nullptr)
84 throw std::runtime_error("Creating new quant param failed");
85 }
86 target_qparam->min = source_qparam->min;
87 target_qparam->max = source_qparam->max;
88 target_qparam->scale = source_qparam->scale;
89 target_qparam->zerop = source_qparam->zerop;
90 target_qparam->quantized_dimension = source_qparam->quantized_dimension;
91}

97bool ignore_pad_v2_const_quantization(const luci::CirclePadV2 *pad)
98{
99 // This is a workaround to quantize pad generated from MaxPoolWithArgmax operation properly
100 // TODO use metadata hints to detect this case
101 auto const_value_node = dynamic_cast<const luci::CircleConst *>(pad->arg(2));
102 if (!const_value_node)
103 return false;
104 if (const_value_node->dtype() == loco::DataType::FLOAT32)
105 {
106 float const_value = const_value_node->at<loco::DataType::FLOAT32>(0);
107 if (const_value == std::numeric_limits<float>::lowest())
108 return true;
109 }
110 return false;
111}

135void propagate_pack_quantparam(luci::CirclePack *pack)
136{
137 assert(pack->quantparam() != nullptr);
138
139 const auto num_inputs = pack->values_count();
140
141 for (uint32_t i = 0; i < num_inputs; i++)
142 {
143 auto node = loco::must_cast<luci::CircleNode *>(pack->arg(i));
144
145 // Quantize constant values
146 if (node->opcode() == luci::CircleOpcode::CIRCLECONST)
147 {
148 luci::CircleConst *const_node = loco::must_cast<luci::CircleConst *>(node);
149 if (const_node->dtype() != loco::DataType::FLOAT32)
150 throw std::runtime_error("Unsupported data type for constant input of pack Op");
151
152 const auto pack_qparam = pack->quantparam();
153 if (pack_qparam == nullptr)
154 throw std::runtime_error("quantparam of pack is not found during propagation");
155
156 assert(pack_qparam->scale.size() == 1);
157 assert(pack_qparam->zerop.size() == 1);
158 const auto scaling_factor = pack_qparam->scale[0];
159 const auto zerop = pack_qparam->zerop[0];
160
161 auto new_const = luci::clone(const_node);
162 quant_const_values(new_const, scaling_factor, zerop, pack->dtype());
163 pack->values(i, new_const);
164 overwrite_quantparam(pack, new_const);
165 }
166 else
167 {
168 const auto succs = loco::succs(node);
169 if (succs.size() > 1)
170 continue;
171
172 // Non-const input must have been quantized
173 assert(node->quantparam() != nullptr);
174 overwrite_quantparam(pack, node);
175 }
176 }

205void propagate_one_hot_quantparam(luci::CircleOneHot *one_hot)
206{
207 assert(one_hot->quantparam() != nullptr);
208
209 // Propagate quantization parameters from output to inputs,
210 // to fit both input and counstant_value in one quant range.
211 auto quant_input = [one_hot](void (luci::CircleOneHot::*arg_setter)(loco::Node *),
212 loco::Node *(luci::CircleOneHot::*arg_getter)() const) {
213 auto node = loco::must_cast<luci::CircleNode *>((one_hot->*arg_getter)());
214
215 // Quantize constant values
216 if (node->opcode() == luci::CircleOpcode::CIRCLECONST)
217 {
218 luci::CircleConst *const_node = loco::must_cast<luci::CircleConst *>(node);
219 if (is_quantized(const_node))
220 return;
221
222 if (const_node->dtype() != loco::DataType::FLOAT32)
223 throw std::runtime_error("Unsupported data type for constant input of OneHot Op");
224
225 const auto qparam = one_hot->quantparam();
226 if (qparam == nullptr)
227 throw std::runtime_error("quantparam of OneHot is not found during propagation");
228
229 assert(qparam->scale.size() == 1);
230 const auto scaling_factor = qparam->scale.at(0);
231 const auto zerop = qparam->zerop.at(0);
232
233 auto new_const = luci::clone(const_node);
234 quant_const_values(new_const, scaling_factor, zerop, one_hot->dtype());
235 overwrite_quantparam(one_hot, new_const);
236 (one_hot->*arg_setter)(new_const);
237 }
238 else
239 {
240 const auto succs = loco::succs(node);
241 if (succs.size() > 1)
242 return;
243
244 // Non-const input must have been quantized
245 assert(node->quantparam() != nullptr);
246 overwrite_quantparam(one_hot, node);
247 }
248 };
249
250 quant_input(&luci::CircleOneHot::on_value, &luci::CircleOneHot::on_value);
251 quant_input(&luci::CircleOneHot::off_value, &luci::CircleOneHot::off_value);
252}

276void propagate_concat_quantparam(luci::CircleConcatenation *concat)
277{
278 assert(concat->quantparam() != nullptr);
279
280 const auto num_inputs = concat->numValues();
281
282 // Quantize const inputs using their values if concat has fused act function
283 if (concat->fusedActivationFunction() != luci::FusedActFunc::NONE)
284 {
285 for (uint32_t i = 0; i < num_inputs; i++)
286 {
287 auto node = concat->arg(i);
288 auto const_node = dynamic_cast<luci::CircleConst *>(node);
289 if (const_node != nullptr)
290 {
291 auto new_const = luci::clone(const_node);
292 quant_const(new_const, concat->dtype());
293 concat->values(i, new_const);
294 }
295 }
296 return;
297 }
298
299 for (uint32_t i = 0; i < num_inputs; i++)
300 {
301 auto node = loco::must_cast<luci::CircleNode *>(concat->arg(i));
302
303 // Quantize constant values
304 if (node->opcode() == luci::CircleOpcode::CIRCLECONST)
305 {
306 luci::CircleConst *const_node = loco::must_cast<luci::CircleConst *>(node);
307
308 const auto concat_qparam = concat->quantparam();
309 assert(concat_qparam->scale.size() == 1);
310 const auto scaling_factor = concat_qparam->scale[0];
311 const auto zerop = concat_qparam->zerop[0];
312
313 auto new_const = luci::clone(const_node);
314 quant_const_values(new_const, scaling_factor, zerop, concat->dtype());
315 concat->values(i, new_const);
316 overwrite_quantparam(concat, new_const);
317 }
318 else
319 {
320 const auto succs = loco::succs(node);
321 if (succs.size() > 1)
322 continue;
323
324 // Non-const input must have been quantized
325 assert(node->quantparam() != nullptr);
326 overwrite_quantparam(concat, node);
327 }
328 }

366void propagate_pad_v2_quantparam(luci::CirclePadV2 *pad_v2)
367{
368 if (ignore_pad_v2_const_quantization(pad_v2))
369 {
370 // propagate input quantization paramters from input to output and padding const value
371 auto pad_v2_input = loco::must_cast<luci::CircleNode *>(pad_v2->arg(0));
372 overwrite_quantparam(pad_v2_input, pad_v2);
373
374 auto const_value_node = loco::must_cast<luci::CircleConst *>(
375 pad_v2->arg(2)); // FIX ignore_pad_v2_const_quantization UNLESS
376 auto new_const = luci::clone(const_value_node);
377
378 const auto pad_v2_input_qparam = pad_v2_input->quantparam();
379 assert(pad_v2_input_qparam != nullptr);
380 assert(pad_v2_input_qparam->scale.size() == 1);
381 const auto scaling_factor = pad_v2_input_qparam->scale.at(0);
382 const auto zerop = pad_v2_input_qparam->zerop.at(0);
383
384 quant_const_values(new_const, scaling_factor, zerop, pad_v2->dtype());
385 overwrite_quantparam(pad_v2_input, new_const);
386 pad_v2->constant_values(new_const);
387 return;
388 }
389
390 // Propagate quantization paramters from output to inputs,
391 // to fit both input and counstant_value in one quant range.
392 auto quant_input = [pad_v2](void (CirclePadV2::*arg_setter)(loco::Node *), uint32_t arg) {
393 auto node = loco::must_cast<luci::CircleNode *>(pad_v2->arg(arg));
394
395 // Quantize constant values
396 if (node->opcode() == luci::CircleOpcode::CIRCLECONST)
397 {
398 luci::CircleConst *const_node = loco::must_cast<luci::CircleConst *>(node);
399 if (is_quantized(const_node))
400 return;
401
402 if (const_node->dtype() != loco::DataType::FLOAT32)
403 throw std::runtime_error("Unsupported data type for constant input of PadV2 Op");
404
405 const auto pad_v2_qparam = pad_v2->quantparam();
406 if (pad_v2_qparam == nullptr)
407 throw std::runtime_error("quantparam of PadV2 is not found during propagation");
408
409 assert(pad_v2_qparam->scale.size() == 1);
410 const auto scaling_factor = pad_v2_qparam->scale.at(0);
411 const auto zerop = pad_v2_qparam->zerop.at(0);
412
413 auto new_const = luci::clone(const_node);
414 quant_const_values(new_const, scaling_factor, zerop, pad_v2->dtype());
415 overwrite_quantparam(pad_v2, new_const);
416 (pad_v2->*arg_setter)(new_const);
417 }
418 else
419 {
420 const auto succs = loco::succs(node);
421 if (succs.size() > 1)
422 return;
423
424 // Non-const input must have been quantized
425 assert(node->quantparam() != nullptr);
426 overwrite_quantparam(pad_v2, node);
427 }
428 };
429
430 quant_input(&CirclePadV2::input, 0);
431 quant_input(&CirclePadV2::constant_values, 2);
432}

440struct PropagateQParamBackward final : public luci::CircleNodeMutableVisitor<void>
441{

443

458bool PropagateQParamBackwardPass::run(loco::Graph *g)
459{
460 LOGGER(l);
461
462 // We use reverse post-order traversal as qparam is propagated backward
463 auto nodes = loco::postorder_traversal(loco::output_nodes(g));
464 std::reverse(nodes.begin(), nodes.end());
465 for (auto node : nodes)
466 {
467 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
468 INFO(l) << "PropagateQParamBackwardPass visit node: " << circle_node->name() << std::endl;
469
470 // We can't propagate non-existent qparam
471 if (circle_node->quantparam() == nullptr)
472 continue;
473
474 PropagateQParamBackward pqb;
475 circle_node->accept(&pqb);
476 }
477
478 // This pass is only run once, so return false
479 // TODO Refactoring not to return meaningless value
480 return false;
481}
