29bool inputs_dtype_ready(const luci::CircleNode *node)
30{
31 for (uint32_t arity = 0; arity < node->arity(); ++arity)
32 {
33 auto input_node = loco::must_cast<luci::CircleNode *>(node->arg(arity));
34 if (input_node->dtype() == loco::DataType::Unknown)
35 return false;
36 }
37
38 return true;
39}

48bool Rule::infer(const luci::CircleNode *circle_node, loco::DataType &dtype) const
49{
50 LOGGER(l);
51 VERBOSE(l, 1) << "[CircleTypeInference] " << circle_node->name();

53

55 {
56 VERBOSE(l, 1) << " after: Some inputs are not ready for inference";
57 return false;
58 }
59
60 Algorithm alg;

62
63 VERBOSE(l, 1) << " after: " << static_cast<int>(dtype);
