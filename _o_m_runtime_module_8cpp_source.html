51OMStatus OMRuntimeModule::importModel(const char *model_ptr, const OMConfig &config)
52{
53 assert(model_ptr != nullptr && "Model ptr shouldn't be nullptr");
54 if (model_ptr == nullptr)
55 return UnknownError;
56
57 // First - parse reader
58 // Second - load default graph
59 // Third - optimize it until can
60 // 4 - AllocDeallocPlan creation
61 // 5 - KernelConfigure
62 // 6 - Allocate inputs
63
64 OMStatus status;
65 // First - parse reader
66 uint32_t num_subgraph = 0;
67 {
68 reader::OMCircleReader reader;
69 status = reader.parse(model_ptr);
70 if (status != Ok)
71 return status;
72 num_subgraph = reader.num_subgraph();
73 }
74
75 assert(num_subgraph >= 1);
76 if (num_subgraph == 0)
77 return UnknownError;
78
79 _graphs.resize(num_subgraph);
80
81 for (uint32_t i = 0; i < num_subgraph; ++i)
82 {
83 // Second - load default graph
84 OMRuntimeGraph &graph = _graphs.at(i);
85
86 OMRuntimeContext &runtime_context = graph.getRuntimeContext();
87 OMRuntimeStorage &runtime_storage = graph.getRuntimeStorage();
88 memory::OMRuntimeAllocator &runtime_allocator = graph.getRuntimeAllocator();
89
90 runtime_context.setModel(model_ptr, i);
91
92 // Third - optimize it until can
93 status = optimize::OMOptimizer::optimize(runtime_storage, runtime_context, config);
94 if (status != Ok)
95 return status;
96
97 // 4 - AllocDeallocPlan creation
98 status = import::OMExecutionPlanCreator::createExecutionPlan(runtime_storage, runtime_context,
99 runtime_allocator, config);
100 if (status != Ok)
101 return status;
102
103 // 5 - KernelConfigure
104 import::OMConfigureArgs configure_args = {runtime_storage, runtime_context, 0, config, *this};
105
106 status = import::OMKernelConfiguration::configureKernels(configure_args);
107 if (status != Ok)
108 return status;
109
110 // Done!
111 }
112
113 return Ok;
114}

124OMStatus OMRuntimeModule::run()
125{
126 OMStatus status = Ok;
127
128 if (_graphs.empty())
129 return ModelNotImport;
130
131 core::OMRuntimeGraph &main_graph = _graphs.at(0);
132
133 execute::OMExecuteArgs execute_args = {main_graph.getRuntimeStorage(),
134 main_graph.getRuntimeContext(), 0, *this};
135
136 status = execute::OMKernelExecute::executeKernel(execute_args, main_graph.getRuntimeAllocator());
137 if (status != Ok)
138 return status;
139
140 return status;
141}

143OMStatus OMRuntimeModule::reset()
144{
145 OMStatus status = Ok;
146
147 if (_graphs.empty())
148 return ModelNotImport;
149
150 for (auto &graph : _graphs)
151 {
152 graph.reset();
153 }
154
155 return status;
156}
