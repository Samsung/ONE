48// Convert axis in acl order
49inline ARMComputeAxis ToARMComputeAxis(uint32_t rank, uint32_t axis,
50 const ir::Layout org_layout = ir::Layout::UNKNOWN,
51 const ir::Layout acl_layout = ir::Layout::UNKNOWN)
52{
53 assert(rank > axis);
54
55 const ARMComputeAxis reversed{(rank - axis) - 1};
56
57 if (rank >= 4 && org_layout == ir::Layout::NHWC && acl_layout == ir::Layout::NCHW)
58 {
59 // NHWC -> WHCN
60 // DEPTH
61 if (0 == reversed.value())
62 {
63 return ARMComputeAxis{2};
64 }
65 // WIDTH
66 if (1 == reversed.value())
67 {
68 return ARMComputeAxis{0};
69 }
70 // HEIGHT
71 if (2 == reversed.value())
72 {
73 return ARMComputeAxis{1};
74 }
75 }
76 if (rank >= 4 && org_layout == ir::Layout::NCHW && acl_layout == ir::Layout::NHWC)
77 {
78 // NCHW -> CWHN
79 // WIDTH
80 if (0 == reversed.value())
81 {
82 return ARMComputeAxis{1};
83 }
84 // HEIGHT
85 if (1 == reversed.value())
86 {
87 return ARMComputeAxis{2};
88 }
89 // DEPTH
90 if (2 == reversed.value())
91 {
92 return ARMComputeAxis{0};
93 }
94 }
95
96 return reversed;
97}

99inline ::arm_compute::Coordinates
100getARMComputeAxises(uint32_t rank, const ir::Layout org_layout = ir::Layout::UNKNOWN,
101 const ir::Layout acl_layout = ir::Layout::UNKNOWN)
102{
103 ::arm_compute::Coordinates res{};
104
105 res.set_num_dimensions(rank);
106
107 for (uint32_t axis = 0; axis < rank; ++axis)
108 {
109 res.set(axis, ToARMComputeAxis(rank, axis, org_layout, acl_layout).value());
110 }
111
112 return res;
113}

115// Restructure runtime_permutationVector to ACL_permutationVector
116inline ::arm_compute::PermutationVector
117getARMComputePermutationVector(uint32_t rank, const std::vector<int32_t> runtime_pv,
118 const ir::Layout org_layout = ir::Layout::UNKNOWN,
119 const ir::Layout acl_layout = ir::Layout::UNKNOWN)
120{
121 // rank upto 4 is supported
122 assert(rank <= 4);
123 assert(runtime_pv.size() > 0);
124
125 int new_pv[4] = {0};
126 ::arm_compute::Coordinates axises = getARMComputeAxises(rank, org_layout, acl_layout);
127
128 for (uint32_t i = 0; i < rank; ++i)
129 {
130 new_pv[axises[i]] = ToARMComputeAxis(rank, runtime_pv[i], org_layout, acl_layout).value();
131 }
132
133 ::arm_compute::PermutationVector ACL_PV =
134 ::arm_compute::PermutationVector{new_pv[0], new_pv[1], new_pv[2], new_pv[3]};
135 ACL_PV.set_num_dimensions(rank);
136
137 return ACL_PV;
138}

140template <typename T>
141inline T ReorderBits(T in, size_t numOfBits, const ir::Layout org_layout = ir::Layout::UNKNOWN,
142 const ir::Layout acl_layout = ir::Layout::UNKNOWN)
143{
144 assert(numOfBits > 0);
145 T out = 0;
146 for (int32_t i = numOfBits - 1; i >= 0; --i)
147 {
148 const uint32_t toShift =
149 numOfBits - ToARMComputeAxis(numOfBits, i, org_layout, acl_layout).value() - 1;
150 out += ((in & 1) << toShift);
151 in >>= 1;
152 }
153 return out;
154}
