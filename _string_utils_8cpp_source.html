45const std::string toString(luci::FusedActFunc fused_act)
46{
47 switch (fused_act)
48 {
49 case (luci::FusedActFunc::UNDEFINED):
50 return std::string("undefined");
51 case (luci::FusedActFunc::NONE):
52 return std::string("none");
53 case (luci::FusedActFunc::RELU):
54 return std::string("relu");
55 case (luci::FusedActFunc::RELU_N1_TO_1):
56 return std::string("relu_n1_to_1");
57 case (luci::FusedActFunc::RELU6):
58 return std::string("relu6");
59 case (luci::FusedActFunc::TANH):
60 return std::string("tanh");
61 case (luci::FusedActFunc::SIGN_BIT):
62 return std::string("sign_bit");
63 default:
64 throw std::runtime_error("Unsupported activation function");
65 }
66}
