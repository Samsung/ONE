79def pytest_generate_tests(metafunc):

87 tests_default_tol = []

96 # eval(TEST_NAME)
97 tests_default_tol = [(arg, binary_path, eval_driver_path) for arg in test_args

99 # eval(TEST_NAME RTOL ATOL)
100 tests_with_tol = [(arg.split()[0], binary_path, eval_driver_path, arg.split()[1],
101 arg.split()[2]) for arg in test_args if len(arg.split()) == 3]
102
103 # copy circle file to binary
104 for test_item in tests_default_tol:
105 prepare_materials(test_item[0], tflrecipe_path, circlerecipe_path,
106 binary_path, artifacts_path)
107
108 for test_item in tests_with_tol:
109 prepare_materials(test_item[0], tflrecipe_path, circlerecipe_path,
110 binary_path, artifacts_path)
111
112 if 'default_test_name' in metafunc.fixturenames:
113 metafunc.parametrize('default_test_name,binary_path,eval_driver_path',
114 tests_default_tol)
115
116 if 'tol_test_name' in metafunc.fixturenames:
117 metafunc.parametrize('tol_test_name,binary_path,eval_driver_path,rtolf32,atolf32',
118 tests_with_tol)
