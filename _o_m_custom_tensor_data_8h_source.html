36template <typename T>
37class IOTensorData : public OMTensorData<T>
38{
39 using QuantData = OMQuantizationData<T>;
40 using QuantDataPtr = std::unique_ptr<QuantData>;
41
42private:
43 QuantDataPtr _quant = {};
44
45public:
46 IOTensorData(T *data, const circle::Tensor *tensor)
47 : OMTensorData<T>(data, tensor)
48 {
49 if (IsQuantized<T>)
50 {
51 _quant = std::make_unique<QuantData>(this->_data, tensor);
52 }
53 }
54
55public:
56 template <typename U = T>
57 EnableIfNotQuantized<U, U> ValueAt(size_t idx) const
58 {
59 return this->At(idx);
60 }
61
62 template <typename U = T>
63 EnableIfQuantized<U, float> ValueAt(size_t idx) const
64 {
65 this->CheckIndex(idx);
66 return _quant->DataAt(idx);
67 }
68
69public:
70 template <typename U = T>
71 EnableIfNotQuantized<U, void> SetValueAt(size_t idx, T value)
72 {
73 this->SetAt(idx, value);
74 }
75
76 template <typename U = T>
77 EnableIfQuantized<U, void> SetValueAt(size_t idx, float value)
78 {
79 this->CheckIndex(idx);
80 _quant->SetDataAt(idx, value);
81 }
82};

86template <typename IntType = uint32_t>
87class OMAxisData : public OMTensorData<int32_t, IntType>
88{
89 std::vector<IntType> _axes = {};
90
91public:
92 OMAxisData(int32_t *data, const circle::Tensor *tensor)
93 : OMTensorData<int32_t, IntType>(data, tensor)
94 {
95 /*
96 Handle negative index.
97 A positive index 'p_idx' can be represented as a negative index 'n_idx' as:
98 n_idx = p_idx - num_dims.
99 eg: For num_dims=3, [0, 1, 2] is the same as [-3, -2, -1].
100 */
101 std::transform(data, data + this->_size, std::back_inserter(_axes), [this](auto axis)
102 {
103 auto offset = this->_size * static_cast<size_t>(axis < 0);
104 return static_cast<IntType>(axis + offset);
105 });
106 }
107
108public:
109 IntType *Get() noexcept override
110 {
111 return _axes.data();
112 }
113
114 const IntType *Get() const noexcept override
115 {
116 return _axes.data();
117 }
118};
