32// Validates the type. The used dimensions can be underspecified.
33int validateOperandType(const ANeuralNetworksOperandType &type, const char *tag, bool allowPartial)
34{
35 if (!allowPartial)
36 {
37 for (uint32_t i = 0; i < type.dimensionCount; i++)
38 {
39 if (type.dimensions[i] == 0)
40 {
41 LOG(ERROR) << tag << " OperandType invalid dimensions[" << i
42 << "] = " << type.dimensions[i];
43 return ANEURALNETWORKS_BAD_DATA;
44 }
45 }
46 }
47 if (!validCode(kNumberOfDataTypes, type.type))
48 {
49 LOG(ERROR) << tag << " OperandType invalid type " << type.type;
50 return ANEURALNETWORKS_BAD_DATA;
51 }
52 if (type.type == ANEURALNETWORKS_TENSOR_QUANT8_ASYMM)
53 {
54 if (type.zeroPoint < 0 || type.zeroPoint > 255)
55 {
56 LOG(ERROR) << tag << " OperandType invalid zeroPoint " << type.zeroPoint;
57 return ANEURALNETWORKS_BAD_DATA;
58 }
59 if (type.scale < 0.f)
60 {
61 LOG(ERROR) << tag << " OperandType invalid scale " << type.scale;
62 return ANEURALNETWORKS_BAD_DATA;
63 }
64 }
65
66 // TODO-NNRT : add 'type.type == ANEURALNETWORKS_OEM_SCALAR' later.
67 // OEM operaters are not supported now.
68 if (type.type == ANEURALNETWORKS_FLOAT32 || type.type == ANEURALNETWORKS_INT32 ||
69 type.type == ANEURALNETWORKS_UINT32)
70 {
71 if (type.dimensionCount != 0 || type.dimensions != nullptr)
72 {
73 LOG(ERROR) << tag << " Invalid dimensions for scalar type";
74 return ANEURALNETWORKS_BAD_DATA;
75 }
76 }
77
78 return ANEURALNETWORKS_NO_ERROR;
79}

109static bool validOperands(const std::vector<Operand> &operands, const std::vector<uint8_t> &operandValues)
110{
111 for (auto &operand : operands)
112 {
113 if (!validCode(kNumberOfDataTypes, static_cast<uint32_t>(operand.type)))
114 {
115 LOG(ERROR) << "Invalid operand type ";
116 return false;
117 }
118 /* TODO validate dim with type
119 if (!validOperandIndexes(operand.dimensions, mDimensions)) {
120 return false;
121 }
122 */
123 switch (operand.lifetime)
124 {
125 case OperandLifeTime::CONSTANT_COPY:
126 if (operand.location.offset + operand.location.length > operandValues.size())
127 {
128 LOG(ERROR) << "OperandValue location out of range. Starts at " << operand.location.offset
129 << ", length " << operand.location.length << ", max " << operandValues.size();
130 return false;
131 }
132 break;
133 case OperandLifeTime::TEMPORARY_VARIABLE:
134 case OperandLifeTime::MODEL_INPUT:
135 case OperandLifeTime::MODEL_OUTPUT:
136 case OperandLifeTime::NO_VALUE:
137 if (operand.location.offset != 0 || operand.location.length != 0)
138 {
139 LOG(ERROR) << "Unexpected offset " << operand.location.offset << " or length "
140 << operand.location.length << " for runtime location.";
141 return false;
142 }
143 break;
144 case OperandLifeTime::CONSTANT_REFERENCE:
145#if 0
146 if (operand.location.poolIndex >= poolCount)
147 {
148 LOG(ERROR) << "Invalid poolIndex " << operand.location.poolIndex << "/" << poolCount;
149 return false;
150 }
151#endif
152 break;
153 // TODO: Validate that we are within the pool.
154 default:
155 LOG(ERROR) << "Invalid lifetime";
156 return false;
157 }
158 }
159 return true;
160}

162static bool validOperations(const std::vector<Operation> &operations, size_t operandCount)
163{
164 for (auto &op : operations)
165 {
166 if (!validCode(kNumberOfOperationTypes, static_cast<uint32_t>(op.type)))
167 {
168 LOG(ERROR) << "Invalid operation type ";
169 return false;
170 }
171 if (!validOperandIndexes(op.inputs, operandCount) ||
172 !validOperandIndexes(op.outputs, operandCount))
173 {
174 return false;
175 }
176 }
177 return true;
178}

190bool validRequestArguments(const std::vector<RequestArgument> &arguments,
191 const std::vector<uint32_t> &operandIndexes,
192 const std::vector<Operand> &operands, size_t poolCount, const char *type)
193{
194 const size_t argumentCount = arguments.size();
195 if (argumentCount != operandIndexes.size())
196 {
197 LOG(ERROR) << "Request specifies " << argumentCount << " " << type << "s but the model has "
198 << operandIndexes.size();
199 return false;
200 }
201 for (size_t argumentIndex = 0; argumentIndex < argumentCount; argumentIndex++)
202 {
203 const RequestArgument &argument = arguments[argumentIndex];
204 const uint32_t operandIndex = operandIndexes[argumentIndex];
205 const Operand &operand = operands[operandIndex];
206 if (argument.hasNoValue)
207 {
208 if (argument.location.poolIndex != 0 || argument.location.offset != 0 ||
209 argument.location.length != 0 || argument.dimensions.size() != 0)
210 {
211 LOG(ERROR) << "Request " << type << " " << argumentIndex
212 << " has no value yet has details.";
213 return false;
214 }
215 }
216 if (argument.location.poolIndex >= poolCount)
217 {
218 LOG(ERROR) << "Request " << type << " " << argumentIndex << " has an invalid poolIndex "
219 << argument.location.poolIndex << "/" << poolCount;
220 return false;
221 }
222 // TODO: Validate that we are within the pool.
223 uint32_t rank = argument.dimensions.size();
224 if (rank > 0)
225 {
226 if (rank != operand.dimensions.size())
227 {
228 LOG(ERROR) << "Request " << type << " " << argumentIndex << " has number of dimensions ("
229 << rank << ") different than the model's (" << operand.dimensions.size() << ")";
230 return false;
231 }
232 for (size_t i = 0; i < rank; i++)
233 {
234 if (argument.dimensions[i] != operand.dimensions[i] && operand.dimensions[i] != 0)
235 {
236 LOG(ERROR) << "Request " << type << " " << argumentIndex << " has dimension " << i
237 << " of " << operand.dimensions[i] << " different than the model's "
238 << operand.dimensions[i];
239 return false;
240 }
241 if (argument.dimensions[i] == 0)
242 {
243 LOG(ERROR) << "Request " << type << " " << argumentIndex << " has dimension " << i
244 << " of zero";
245 return false;
246 }
247 }
248 }
249 }
250 return true;
251}
