30void KernelGenerator::visit(const ir::operation::Split &node)
31{
32 const auto num_splits = node.param().num_splits;
33 assert(num_splits == static_cast<int>(node.getOutputs().size()));
34
35 const auto input_idx{node.getInputs().at(ir::operation::Split::Input::INPUT)};
36 const auto axis_idx{node.getInputs().at(ir::operation::Split::Input::AXIS)};
37
38 auto in_tensor = _tensor_reg->getPortableTensor(input_idx);
39 auto axis_tensor = _tensor_reg->getPortableTensor(axis_idx);
40
41 std::vector<IPortableTensor *> out_tensors;
42 for (const auto &output_idx : node.getOutputs())
43 out_tensors.emplace_back(_tensor_reg->getPortableTensor(output_idx));
44
45 auto fn = std::make_unique<ops::SplitLayer>();
46
47 fn->configure(in_tensor, axis_tensor, num_splits, out_tensors);
48
49 _return_fn = std::move(fn);
50}

62template <typename T> void SplitLayer::split(void)
63{
64 nnfw::cker::SplitParams op_params;
65 if (_axis->total_size() != sizeof(int32_t))
66 {
67 throw std::runtime_error("ArgMinMax: wrong shape of axis");
68 }
69 auto axis = *getBuffer<int32_t>(_axis);
70 if (axis < 0)
71 {
72 axis += _input->getShape().rank();
73 }
74 op_params.axis = axis;
75 op_params.num_split = _num_splits;
76
77 std::vector<T *> outputPtrs;
78
79 for (const auto output : _outputs)
80 {
81 assert(output->total_size() == sizeOfData(output->data_type(), output->getShape().dims()));
82 outputPtrs.emplace_back(getBuffer<T>(output));
83 }
84
85 assert(_input->total_size() == sizeOfData(_input->data_type(), _input->getShape().dims()));
86 nnfw::cker::Split<T>(op_params, getShape(_input), getBuffer<T>(_input), getShape(_outputs[0]),
87 outputPtrs.data());
88}

101void SplitLayer::run()
102{
103 if (_input->data_type() == OperandType::FLOAT32)
104 {
105 split<float>();
106 }
107 else if (_input->data_type() == OperandType::QUANT_UINT8_ASYMM)
108 {
109 split<uint8_t>();
110 }
111 else if (_input->data_type() == OperandType::INT32)
112 {
113 split<int32_t>();
114 }
115 else if (_input->data_type() == OperandType::INT64)
116 {
117 split<int64_t>();
118 }
119 else
120 {
121 throw std::runtime_error{"Split: unsupported input type"};
122 }
123}
