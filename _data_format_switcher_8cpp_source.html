36PassData DataFormatSwitcher::run(PassData data)
37{
38 _graph = static_cast<mir::Graph *>(data);
39 assert(_graph);
40
41 // Collect nodes which use DataFormat
42 for (auto *node : _graph->getNodes())
43 {
44 switch (node->getType())
45 { // nodes using DataFormat
46 case mir::Operation::Type::avgPool2D:
47 case mir::Operation::Type::conv2D:
48 case mir::Operation::Type::deConv2D:
49 case mir::Operation::Type::depthwiseConv:
50 case mir::Operation::Type::maxPool2D:
51 _candidates_for_switch.push_back(node);
52 break;
53 default:
54 break; // not use DataFormat
55 }
56 }
57 // Switch collected ops
58 for (auto *op : _candidates_for_switch)
59 {
60 switch (op->getType())
61 {
62 case mir::Operation::Type::avgPool2D:
63 switchAvgPool2D(dynamic_cast<mir::ops::AvgPool2DOp *>(op));
64 break;
65 case mir::Operation::Type::conv2D:
66 switchConv2D(dynamic_cast<mir::ops::Conv2DOp *>(op));
67 break;
68 case mir::Operation::Type::deConv2D:
69 switchDeConv2D(dynamic_cast<mir::ops::DeConv2DOp *>(op));
70 break;
71 case mir::Operation::Type::depthwiseConv:
72 switchDepthwiseConv2D(dynamic_cast<mir::ops::DepthwiseConv2DOp *>(op));
73 break;
74 case mir::Operation::Type::maxPool2D:
75 switchMaxPool2D(dynamic_cast<mir::ops::MaxPool2DOp *>(op));
76 break;
77 default:
78 assert(false && "Can't switch DataFormat for this operation!");
79 }
80 }
81
82 return _graph;
83}

87mir::Operation::Output *DataFormatSwitcher::insertTransposeBefore(mir::Operation::Output *out)
88{
89 mir::Operation::Output *new_out;
90 if (_target_format == mir::DataFormat::NHWC)
91 new_out = _graph->create<mir::ops::TransposeOp>(out, std::vector<std::size_t>{0, 2, 3, 1})
92 ->getOutput(0); // NCHW -> NHWC
93 else
94 new_out = _graph->create<mir::ops::TransposeOp>(out, std::vector<std::size_t>{0, 3, 1, 2})
95 ->getOutput(0); // NHWC -> NCHW
96 if (out->getType().isQuantized())
97 new_out->setQuantization(out->getType().getQuantization());
98 return new_out;
99}

115void DataFormatSwitcher::switchAvgPool2D(mir::ops::AvgPool2DOp *op)
116{
117 if (op->getDataFormat() == _target_format)
118 return;
119
120 auto *input = op->getInput(0);
121
122 mir::AvgPool2DOpAttributes attributes(op->getAttributes());
123 attributes.data_format = _target_format;
124
125 auto *trans_in = insertTransposeBefore(input);
126
127 auto new_pool = _graph->create<mir::ops::AvgPool2DOp>(trans_in, attributes);
128
129 auto *trans_out = insertTransposeAfter(new_pool->getOutput(0));
130
131 _graph->replaceNode(op, trans_out->getNode());
132}

134void DataFormatSwitcher::switchConv2D(mir::ops::Conv2DOp *op)
135{
136 if (op->getDataFormat() == _target_format)
137 return;
138
139 assert(op->getNumInputs() >= 2);
140 auto *input = op->getInput(0);
141 auto *kernel = op->getInput(1);
142
143 mir::Conv2DOpAttributes attributes(op->getAttributes());
144 attributes.data_format = _target_format;
145
146 auto *trans_in = insertTransposeBefore(input);
147
148 mir::Operation *new_conv;
149 if (op->getNumInputs() == 2)
150 new_conv = _graph->create<mir::ops::Conv2DOp>(trans_in, kernel, attributes);
151 else
152 {
153 auto bias = op->getInput(2);
154 new_conv = _graph->create<mir::ops::Conv2DOp>(trans_in, kernel, bias, attributes);
155 }
156
157 if (op->getOutput(0)->getType().isQuantized())
158 new_conv->getOutput(0)->setQuantization(op->getOutput(0)->getType().getQuantization());
159
160 auto *trans_out = insertTransposeAfter(new_conv->getOutput(0));
161
162 _graph->replaceNode(op, trans_out->getNode());
163}

165void DataFormatSwitcher::switchDeConv2D(mir::ops::DeConv2DOp *op)
166{
167 if (op->getDataFormat() == _target_format)
168 return;
169
170 assert(op->getNumInputs() == 2);
171 auto *input = op->getInput(0);
172 auto *kernel = op->getInput(1);
173
174 auto *trans_in = insertTransposeBefore(input);
175
176 mir::Operation *new_deconv;
177 mir::Deconv2DOpAttributes attributes(op->getAttributes());
178 attributes.data_format = _target_format;
179 if (attributes.padding_type == mir::ops::PaddingType::Explicit)
180 {
181 new_deconv = _graph->create<mir::ops::DeConv2DOp>(trans_in, kernel, attributes);
182 }
183 else
184 {
185 mir::Shape output_shape = op->getOutputShape(0);
186 if (_target_format == mir::DataFormat::NHWC)
187 output_shape = mir::transposeShape<0, 2, 3, 1>(output_shape);
188 else
189 output_shape = mir::transposeShape<0, 3, 1, 2>(output_shape);
190 new_deconv = _graph->create<mir::ops::DeConv2DOp>(trans_in, kernel, attributes, output_shape);
191 }
192
193 auto *trans_out = insertTransposeAfter(new_deconv->getOutput(0));
194
195 _graph->replaceNode(op, trans_out->getNode());
196}
