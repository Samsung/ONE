59 .default_value(true)

64 .type(arser::DataType::BOOL)

191 return EXIT_FAILURE;

198 // Do main job
199 circle_planner::ExecutionPlanner execution_planner(module->graph(), {platform_type, use_dsp},
200 runtime_type, allocating_mode);
201 execution_planner.change_planning_mode(is_allocate_const, is_allocate_input, true);
202 execution_planner.make_execution_plan();
203
204 if (is_save_allocations)
205 execution_planner.create_json_allocation_file(json_path);
206
207 // Export to output Circle file
208 luci::CircleExporter exporter;
209 luci::CircleFileExpContract contract(module.get(), output_path);
210
211 if (!exporter.invoke(&contract))
212 {
213 std::cerr << "ERROR: Failed to export '" << output_path << "'" << std::endl;
214 return 255;
215 }
216
217 return 0;
