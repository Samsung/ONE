30// Create new Transpose Op including perm
31// Return nullptr if failed
32CircleTranspose *create_cloned_transpose(CircleTranspose *transpose)
33{
34 assert(transpose != nullptr); // FIX_CALLER_UNLESS
35
36 auto perm = dynamic_cast<CircleConst *>(transpose->perm());
37 if (not perm)
38 return nullptr;
39
40 CircleConst *cloned_perm = clone(perm);
41 if (cloned_perm == nullptr)
42 return nullptr;
43
44 cloned_perm->name(perm->name() + "_C");
45 luci::add_origin(cloned_perm, luci::get_origin(perm));
46
47 auto cloned_node = clone_node(transpose, transpose->graph());
48 if (cloned_node == nullptr)
49 return nullptr;
50
51 auto new_transpose = loco::must_cast<luci::CircleTranspose *>(cloned_node);
52 new_transpose->perm(cloned_perm);
53 new_transpose->name(transpose->name() + "_C");
54 luci::add_origin(new_transpose, luci::get_origin(transpose));
55
56 return new_transpose;
57}

59uint32_t cal_offset(const std::vector<uint32_t> &shape, const std::vector<uint32_t> &indices)
60{
61 assert(shape.size() == indices.size()); // FIX_CALLER_UNLESS
62
63 uint32_t offset = 0;
64 for (uint32_t i = 0; i < indices.size(); i++)
65 {
66 uint32_t index = indices[i];
67 for (uint32_t j = shape.size() - 1; j > i; j--)
68 {
69 index *= shape[j];
70 }
71 offset += index;
72 }
73 return offset;
74}

76// Return reverse-transpose of 'node'
77// i.e., Transpose(return value) = node
78CircleConst *reverse_transposed(CircleConst *node, std::vector<uint32_t> &t)
79{
80 assert(node->rank() == t.size()); // FIX_CALLER_UNLESS
81 assert(node->rank() == 4); // FIX_CALLER_UNLESS
82
83 std::vector<uint32_t> orig_shape(node->rank());
84 std::vector<uint32_t> new_shape(node->rank());
85
86 for (uint32_t i = 0; i < node->rank(); i++)
87 {
88 assert(t[i] < node->rank()); // FIX_CALLER_UNLESS
89
90 orig_shape[i] = node->dim(i).value();
91 new_shape[t[i]] = node->dim(i).value();
92 }
93
94 auto clone_const = clone(node);
95 for (uint32_t i = 0; i < node->rank(); i++)
96 clone_const->dim(i).set(new_shape[i]);
97
98 clone_const->name(clone_const->name() + "_r_transposed");
99 add_origin(clone_const, luci::get_origin(node));
100
101 for (uint32_t n = 0; n < clone_const->dim(0).value(); n++)
102 {
103 for (uint32_t h = 0; h < clone_const->dim(1).value(); h++)
104 {
105 for (uint32_t w = 0; w < clone_const->dim(2).value(); w++)
106 {
107 for (uint32_t c = 0; c < clone_const->dim(3).value(); c++)
108 {
109 std::vector<uint32_t> new_indices{n, h, w, c};
110 std::vector<uint32_t> orig_indices{new_indices[t[0]], new_indices[t[1]],
111 new_indices[t[2]], new_indices[t[3]]};
112
113 const auto data = node->at<loco::DataType::FLOAT32>(cal_offset(orig_shape, orig_indices));
114 clone_const->at<loco::DataType::FLOAT32>(cal_offset(new_shape, new_indices)) = data;
115 }
116 }
117 }
118 }
119
120 return clone_const;
121}

125// Return true if below conditions are met
126// 1. t->perm() is CircleConst
127// 2. t->perm() is S32
128bool check_perm(const CircleTranspose *t)
129{
130 auto perm = dynamic_cast<CircleConst *>(t->perm());
131 if (not perm)
132 return false;
133
134 switch (perm->dtype())
135 {
136 case loco::DataType::S32:
137 for (uint32_t i = 0; i < perm->size<loco::DataType::S32>(); i++)
138 {
139 auto data = perm->at<loco::DataType::S32>(i);
140 // TODO Support not normalized index
141 if (data < 0 or data >= static_cast<int32_t>(t->rank()))
142 return false;
143 }
144 break;
145 // TODO Support S64 data type
146 default:
147 return false;
148 }
149
150 return true;
151}

157// Elementwise Binary Operator with const
158class EBOWithConstPattern final : public CircleNodeMutableVisitor<bool>
159{
160private:
161 template <typename CIRCLE_OP_PTR> bool has_pattern(CIRCLE_OP_PTR node)
162 {
163 if (auto x = dynamic_cast<luci::CircleConst *>(node->x()))
164 {
165 if (auto y = dynamic_cast<luci::CircleTranspose *>(node->y()))
166 {
167 RETURN_FALSE_UNLESS(check_rank_four(x));
168 RETURN_FALSE_UNLESS(check_perm(y));
169
170 auto new_const = gen_new_const(y, x);
171 assert(new_const); // FIX_ME_UNLESS
172
173 auto new_transpose = create_cloned_transpose(y);
174 assert(new_transpose); // FIX_ME_UNLESS
175
176 // Reconnect network
177 node->x(new_const);
178 node->y(y->a());
179 loco::replace(node).with(new_transpose);
180 new_transpose->a(node);
181
182 // Do shape inference for this node again.
183 node->shape_status(luci::ShapeStatus::UNDEFINED);
184
185 return true;
186 }
187 }
188
189 if (auto y = dynamic_cast<luci::CircleConst *>(node->y()))
190 {
191 if (auto x = dynamic_cast<luci::CircleTranspose *>(node->x()))
192 {
193 RETURN_FALSE_UNLESS(check_rank_four(y));
194 RETURN_FALSE_UNLESS(check_perm(x));
195
196 auto new_const = gen_new_const(x, y);
197 assert(new_const); // FIX_ME_UNLESS
198
199 auto new_transpose = create_cloned_transpose(x);
200 assert(new_transpose); // FIX_ME_UNLESS
201
202 // Reconnect network
203 node->y(new_const);
204 node->x(x->a());
205 loco::replace(node).with(new_transpose);
206 new_transpose->a(node);
207
208 // Do shape inference for this node again.
209 node->shape_status(luci::ShapeStatus::UNDEFINED);
210
211 return true;
212 }
213 }
214
215 return false;
216 }
217
218public:
219 // Default
220 bool visit(luci::CircleNode *) { return false; }
221
222 bool visit(luci::CircleAdd *node) { return has_pattern(node); }
223
224 bool visit(luci::CircleMul *node) { return has_pattern(node); }
225
226private:
227 // Return a new const node after Tranpose Op is forwarded
228 // Return nullptr if unsupported cases
229 CircleConst *gen_new_const(CircleTranspose *t, CircleConst *c)
230 {
231 const auto perm = dynamic_cast<CircleConst *>(t->perm());
232
233 // Only support constant perm
234 if (not perm)
235 return nullptr;
236
237 std::vector<uint32_t> perm_data;
238 switch (perm->dtype())
239 {
240 case loco::DataType::S32:
241 for (uint32_t i = 0; i < perm->size<loco::DataType::S32>(); i++)
242 {
243 auto data = perm->at<loco::DataType::S32>(i);
244 assert(data >= 0 and data < static_cast<int32_t>(t->rank()));
245 perm_data.emplace_back(static_cast<uint32_t>(data));
246 }
247 break;
248 // TODO Support S64 data type
249 default:
250 return nullptr;
251 }
252
253 assert(perm_data.size() == t->rank()); // FIX_CALLER_UNLESS
254
255 return reverse_transposed(c, perm_data);
256 }
257};

291bool ForwardTransposeOpPass::run(loco::Graph *g)
292{
293 bool changed = false;
294 EBOWithConstPattern eboc;
295 for (auto node : loco::active_nodes(loco::output_nodes(g)))
296 {
297 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
298 if (circle_node->accept(&eboc))
299 changed = true;
300 }
301 return changed;
302}
