35void readDataFromFile(const std::string &filename, char *data, size_t data_size,
36 size_t start_position = 0)
37{
38 std::streampos start = start_position;
39
40 std::ifstream fs(filename, std::ifstream::binary);
41 if (fs.fail())
42 throw std::runtime_error("Cannot open file \"" + filename + "\".\n");
43
44 fs.seekg(start);
45
46 if (fs.read(data, data_size).fail())
47 throw std::runtime_error("Failed to read data from file \"" + filename + "\".\n");
48 fs.close();
49}

73DataBuffer readFile(const char *path)
74{
75 std::ifstream file(path, std::ios::binary | std::ios::in);
76 if (!file.good())
77 {
78 std::string errmsg = "Failed to open file";
79 throw std::runtime_error(errmsg.c_str());
80 }
81
82 file.seekg(0, std::ios::end);
83 auto fileSize = file.tellg();
84 file.seekg(0, std::ios::beg);
85
86 // reserve capacity
87 DataBuffer model_data(fileSize);
88
89 // read the data
90 file.read(model_data.data(), fileSize);
91 if (file.fail())
92 {
93 std::string errmsg = "Failed to read file";
94 throw std::runtime_error(errmsg.c_str());
95 }
96
97 return model_data;
98}

102/*
103 * @brief EvalDriver main
104 *
105 * Driver for testing training onert micro
106 * Current example for testing classification task
107 *
108 */
109int entry(int argc, char **argv)
110{
111 if (argc != 11 and argc != 10)
112 {
113 std::cerr << "Two variant of usage with and without wof file: " << argv[0]
114 << " <path/to/circle/model> "
115 " optional(<path/to/wof/file>) <path/to/save/trained/model> "
116 "<path/to/save/checkpoint> <path/to/input/train_data> "
117 "<path/to/input/target_train_data> "
118 "<path/to/input/test_data> <path/to/input/target_test_data> num_of_train_smpl "
119 "num_of_test_smpl\n";
120 return EXIT_FAILURE;
121 }
122
123 const char *circle_model_path = nullptr;
124 const char *wof_file_path = nullptr;
125 const char *output_trained_file_path = nullptr;
126 const char *input_input_train_data_path = nullptr;
127 const char *input_target_train_data_path = nullptr;
128 const char *input_input_test_data_path = nullptr;
129 const char *input_target_test_data_path = nullptr;
130 const char *checkpoints_path = nullptr;
131 int32_t num_train_data_samples = 0;
132 int32_t num_test_data_samples = 0;
133
134 if (argc == 11)
135 {
136 circle_model_path = argv[1];
137 wof_file_path = argv[2];
138 output_trained_file_path = argv[3];
139 checkpoints_path = argv[4];
140 input_input_train_data_path = argv[5];
141 input_target_train_data_path = argv[6];
142 input_input_test_data_path = argv[7];
143 input_target_test_data_path = argv[8];
144 num_train_data_samples = atoi(argv[9]);
145 num_test_data_samples = atoi(argv[10]);
146 }
147 else if (argc == 10)
148 {
149 circle_model_path = argv[1];
150 output_trained_file_path = argv[2];
151 checkpoints_path = argv[3];
152 input_input_train_data_path = argv[4];
153 input_target_train_data_path = argv[5];
154 input_input_test_data_path = argv[6];
155 input_target_test_data_path = argv[7];
156 num_train_data_samples = atoi(argv[8]);
157 num_test_data_samples = atoi(argv[9]);
158 }
159 else
160 {
161 throw std::runtime_error("Unknown commands number\n");
162 }
163
164 DataBuffer circle_model = readFile(circle_model_path);
165 DataBuffer wof_data;
166 // If defined wof file
167 if (wof_file_path != nullptr)
168 wof_data = readFile(wof_file_path);
169
170 // Configure training mode
171 onert_micro::OMConfig config;
172
173 // Save model size and model ptr in config
174 config.model_size = circle_model.size();
175 config.model_ptr = circle_model.data();
176
177 // If defined wof file
178 if (wof_file_path != nullptr)
179 config.wof_ptr = nullptr;
180
181 // Set user defined training settings
182 const uint32_t training_epochs = 30;
183 const float lambda = 0.001f;
184 const uint32_t BATCH_SIZE = 32;
185 const uint32_t INPUT_SIZE = 180;
186 const uint32_t OUTPUT_SIZE = 4;
187 const uint32_t num_train_layers = 10;
188 const onert_micro::OMLoss loss = onert_micro::CROSS_ENTROPY;
189 const onert_micro::OMTrainOptimizer train_optim = onert_micro::ADAM;
190 const float beta = 0.9;
191 const float beta_squares = 0.999;
192 const float epsilon = 1e-07;
193
194 config.train_mode = true;
195 {
196 onert_micro::OMTrainingContext train_context;
197 train_context.batch_size = BATCH_SIZE;
198 train_context.num_of_train_layers = num_train_layers;
199 train_context.learning_rate = lambda;
200 train_context.loss = loss;
201 train_context.optimizer = train_optim;
202 train_context.beta = beta;
203 train_context.beta_squares = beta_squares;
204 train_context.epsilon = epsilon;
205 train_context.epochs = training_epochs;
206
207 config.training_context = train_context;
208 }
209
210 // Create training interpreter and import models
211 onert_micro::OMTrainingInterpreter train_interpreter;
212 train_interpreter.importTrainModel(circle_model.data(), config);
213
214 // Temporary buffer to read input data from file using BATCH_SIZE
215 float training_input[BATCH_SIZE * INPUT_SIZE];
216 float training_target[BATCH_SIZE * OUTPUT_SIZE];
217 // Note: here test size used with BATCH_SIZE = 1
218 float test_input[INPUT_SIZE];
219 float test_target[OUTPUT_SIZE];
220
221 std::vector<float> accuracy_v;
222 std::vector<float> cross_entropy_v;
223 std::vector<float> mse_v;
224 std::vector<float> mae_v;
225
226 float max_accuracy = std::numeric_limits<float>::min();
227 float min_mae = std::numeric_limits<float>::max();
228
229 for (uint32_t e = 0; e < training_epochs; ++e)
230 {
231 // Run train for current epoch
232 std::cout << "Run training for epoch: " << e + 1 << "/" << training_epochs << "\n";
233 config.training_context.num_epoch = e + 1;
234 uint32_t num_steps = num_train_data_samples / BATCH_SIZE;
235 for (int i = 0; i < num_steps; ++i)
236 {
237 uint32_t cur_batch_size = std::min(BATCH_SIZE, num_train_data_samples - BATCH_SIZE * i - 1);
238 cur_batch_size = std::max(1u, cur_batch_size);
239
240 config.training_context.batch_size = cur_batch_size;
241
242 // Read current input and target data
243 readDataFromFile(input_input_train_data_path, reinterpret_cast<char *>(training_input),
244 sizeof(float) * INPUT_SIZE * cur_batch_size,
245 i * sizeof(MODEL_TYPE) * INPUT_SIZE * BATCH_SIZE);
246
247 readDataFromFile(input_target_train_data_path, reinterpret_cast<char *>(training_target),
248 sizeof(float) * OUTPUT_SIZE * cur_batch_size,
249 i * sizeof(MODEL_TYPE) * OUTPUT_SIZE * BATCH_SIZE);
250
251 // Set input and target
252 train_interpreter.setInput(reinterpret_cast<uint8_t *>(training_input), 0);
253 train_interpreter.setTarget(reinterpret_cast<uint8_t *>(training_target), 0);
254
255 // Train with current batch size
256 train_interpreter.trainSingleStep(config);
257
258 float mse = 0.f;
259 float mae = 0.f;
260 float cross_entropy_metric = 0.f;
261 float accuracy = 0.f;
262
263 if (CLASSIFICATION_TASK)
264 {
265 // Evaluate cross_entropy and accuracy metrics
266 train_interpreter.evaluateMetric(onert_micro::CROSS_ENTROPY_METRICS,
267 reinterpret_cast<void *>(&cross_entropy_metric),
268 cur_batch_size);
269 train_interpreter.evaluateMetric(onert_micro::ACCURACY, reinterpret_cast<void *>(&accuracy),
270 cur_batch_size);
271
272 // Save them into vectors
273 accuracy_v.push_back(accuracy);
274 cross_entropy_v.push_back(cross_entropy_metric);
275 }
276 else
277 {
278 // Evaluate mse and mae metrics
279 train_interpreter.evaluateMetric(onert_micro::MSE_METRICS, reinterpret_cast<void *>(&mse),
280 cur_batch_size);
281 train_interpreter.evaluateMetric(onert_micro::MAE_METRICS, reinterpret_cast<void *>(&mae),
282 cur_batch_size);
283
284 // Save them into vectors
285 mse_v.push_back(mse);
286 mae_v.push_back(mae);
287 }
288 }
289
290 // Reset num step value
291 config.training_context.num_step = 0;
292
293 // Calculate and print average values
294 if (CLASSIFICATION_TASK)
295 {
296 float sum_acc = std::accumulate(accuracy_v.begin(), accuracy_v.end(), 0.f);
297 float sum_ent = std::accumulate(cross_entropy_v.begin(), cross_entropy_v.end(), 0.f);
298 std::cout << "Train Average ACCURACY = " << sum_acc / accuracy_v.size() << "\n";
299 std::cout << "Train Average CROSS ENTROPY = " << sum_ent / cross_entropy_v.size() << "\n";
300 }
301 else
302 {
303 float sum_mse = std::accumulate(mse_v.begin(), mse_v.end(), 0.f);
304 float sum_mae = std::accumulate(mae_v.begin(), mae_v.end(), 0.f);
305 std::cout << "Train Average MSE = " << sum_mse / mse_v.size() << "\n";
306 std::cout << "Train Average MAE = " << sum_mae / mae_v.size() << "\n";
307 }
308
309 // Run test for current epoch
310 std::cout << "Run test for epoch: " << e + 1 << "/" << training_epochs << "\n";
311 num_steps = num_test_data_samples;
312
313 accuracy_v.clear();
314 cross_entropy_v.clear();
315
316 for (int i = 0; i < num_steps; ++i)
317 {
318 uint32_t cur_batch_size = 1;
319 readDataFromFile(input_input_test_data_path, reinterpret_cast<char *>(test_input),
320 sizeof(float) * INPUT_SIZE * cur_batch_size,
321 i * sizeof(MODEL_TYPE) * INPUT_SIZE);
322
323 readDataFromFile(input_target_test_data_path, reinterpret_cast<char *>(test_target),
324 sizeof(float) * OUTPUT_SIZE * cur_batch_size,
325 i * sizeof(MODEL_TYPE) * OUTPUT_SIZE);
326
327 train_interpreter.setInput(reinterpret_cast<uint8_t *>(test_input), 0);
328 train_interpreter.setTarget(reinterpret_cast<uint8_t *>(test_target), 0);
329
330 float mse = 0.f;
331 float mae = 0.f;
332 float cross_entropy_metric = 0.f;
333 float accuracy = 0.f;
334
335 if (CLASSIFICATION_TASK)
336 {
337 // Evaluate cross_entropy and accuracy metrics
338 train_interpreter.evaluateMetric(onert_micro::CROSS_ENTROPY_METRICS,
339 reinterpret_cast<void *>(&cross_entropy_metric),
340 cur_batch_size);
341 train_interpreter.evaluateMetric(onert_micro::ACCURACY, reinterpret_cast<void *>(&accuracy),
342 cur_batch_size);
343
344 // Save them into vectors
345 accuracy_v.push_back(accuracy);
346 cross_entropy_v.push_back(cross_entropy_metric);
347 }
348 else
349 {
350 // Evaluate mse and mae metrics
351 train_interpreter.evaluateMetric(onert_micro::MSE_METRICS, reinterpret_cast<void *>(&mse),
352 cur_batch_size);
353 train_interpreter.evaluateMetric(onert_micro::MAE_METRICS, reinterpret_cast<void *>(&mae),
354 cur_batch_size);
355
356 // Save them into vectors
357 mse_v.push_back(mse);
358 mae_v.push_back(mae);
359 }
360 }
361 // Calculate and print average values
362 if (CLASSIFICATION_TASK)
363 {
364 float sum_acc = std::accumulate(accuracy_v.begin(), accuracy_v.end(), 0.f);
365 float sum_ent = std::accumulate(cross_entropy_v.begin(), cross_entropy_v.end(), 0.f);
366 std::cout << "Test Average ACCURACY = " << sum_acc / accuracy_v.size() << "\n";
367 std::cout << "Test Average CROSS ENTROPY = " << sum_ent / cross_entropy_v.size() << "\n";
368
369 // Best checkpoint part
370 float acc = sum_acc / accuracy_v.size();
371 if (acc > max_accuracy)
372 {
373 // Save best checkpoint
374 train_interpreter.saveCheckpoint(config, checkpoints_path);
375 max_accuracy = acc;
376 std::cout << "Found new max Test ACCURACY = " << max_accuracy << " in epoch = " << e + 1
377 << " / " << training_epochs << "\n";
378 }
379 }
380 else
381 {
382 float sum_mse = std::accumulate(mse_v.begin(), mse_v.end(), 0.f);
383 float sum_mae = std::accumulate(mae_v.begin(), mae_v.end(), 0.f);
384 std::cout << "Test Average MSE = " << sum_mse / mse_v.size() << "\n";
385 std::cout << "Test Average MAE = " << sum_mae / mae_v.size() << "\n";
386
387 // Best checkpoint part
388 float acc = sum_mae / mae_v.size();
389 if (acc < min_mae)
390 {
391 // Save best checkpoint
392 train_interpreter.saveCheckpoint(config, checkpoints_path);
393 min_mae = acc;
394 std::cout << "Found new min Test MAE = " << min_mae << " in epoch = " << e + 1 << " / "
395 << training_epochs << "\n";
396 }
397 }
398 }
399
400 // Load best model
401 train_interpreter.loadCheckpoint(config, checkpoints_path);
402
403 // Save training best result
404 train_interpreter.saveModel(config, output_trained_file_path);
405
406 return EXIT_SUCCESS;
407}
