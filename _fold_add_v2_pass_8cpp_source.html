26bool same_shape(const luci::CircleConst *x, const luci::CircleConst *y)
27{
28 if (x->rank() != y->rank())
29 return false;
30
31 for (uint32_t i = 0; i < x->rank(); i++)
32 {
33 if (!(x->dim(i) == y->dim(i)))
34 return false;
35 }
36
37 return true;
38}

43template <loco::DataType T> bool fold_add_v2(luci::CircleCustom *add_v2)
44{
45 // This should hold for AddV2
46 if (add_v2->numInputs() != 2)
47 return false;
48
49 // Check first input is const
50 auto x = dynamic_cast<luci::CircleConst *>(add_v2->inputs(0));
51 if (not x)
52 return false;
53
54 // Check second input is const
55 auto y = dynamic_cast<luci::CircleConst *>(add_v2->inputs(1));
56 if (not y)
57 return false;
58
59 if (x->dtype() != y->dtype())
60 return false;
61
62 if (!same_shape(x, y))
63 return false;
64
65 auto name_x = x->name();
66 auto name_y = y->name();
67 assert(name_x.length() > 0);
68 assert(name_y.length() > 0);
69 auto constant = add_v2->graph()->nodes()->create<luci::CircleConst>();
70 constant->dtype(x->dtype());
71 constant->rank(x->rank());
72 for (uint32_t i = 0; i < x->rank(); i++)
73 constant->dim(i).set(x->dim(i).value());
74
75 const auto size = x->size<T>();
76 constant->size<T>(size);
77 for (uint32_t i = 0; i < size; i++)
78 constant->at<T>(i) = x->at<T>(i) + y->at<T>(i);
79
80 constant->shape_status(luci::ShapeStatus::VALID);
81 constant->name(name_x + ";" + name_y);
82
83 for (auto succ : loco::succs(add_v2))
84 {
85 auto custom_out = loco::must_cast<luci::CircleCustomOut *>(succ);
86 loco::replace(custom_out).with(constant);
87 }
88
89 return true;
90}

100bool FoldAddV2Pass::run(loco::Graph *g)
101{
102 bool changed = false;
103 for (auto node : loco::active_nodes(loco::output_nodes(g)))
104 {
105 if (auto custom = dynamic_cast<luci::CircleCustom *>(node))
106 {
107 if (custom->custom_code() == "AddV2")
108 {
109 // TODO: Support more data types
110 if (custom->dtype() == loco::DataType::S64)
111 {
112 if (fold_add_v2<loco::DataType::S64>(custom))
113 changed = true;
114 }
115 }
116 }
117 }
118
119 return changed;
120}
