65template <typename InputIt> std::string join(InputIt first, InputIt last, const std::string &concat)
66{
67 std::string ret;
68 if (first == last)
69 return ret;
70
71 ret += *first;
72 for (++first; first != last; ++first)
73 {
74 ret += concat;
75 ret += *first;
76 }
77 return ret;
78}
