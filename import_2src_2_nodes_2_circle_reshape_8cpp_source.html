25bool CircleReshapeGraphBuilder::validate(const ValidateArgs &args) const
26{
27 if (args.op.inputs.size() != 1 && args.op.inputs.size() != 2)
28 return false;
29
30 if (args.op.outputs.size() != 1)
31 return false;
32
33 // for two inputs, check if type is S32
34 if (args.op.inputs.size() == 2)
35 {
36 const auto &inputs = args.op.inputs;
37 const auto tensors = args.reader.tensors();
38 const auto tensor_in = tensors.at(inputs.at(1));
39 assert(tensor_in != nullptr);
40
41 // NOTE fix this if there is any other case
42 // TensorFlow lite and circle only supports S32
43 if (tensor_in->type() != circle::TensorType::TensorType_INT32)
44 return false;
45 }
46
47 return true;
48}

59static CircleNode *create_shape_node(const std::vector<int32_t> &shape, loco::Graph *graph)
60{
61 auto *shape_node = graph->nodes()->create<luci::CircleConst>();
62 shape_node->dtype(loco::DataType::S32);
63 shape_node->rank(1);
64 shape_node->dim(0) = shape.size();
65 shape_node->size<loco::DataType::S32>(shape.size());
66 for (uint32_t i = 0; i < shape.size(); ++i)
67 {
68 shape_node->at<loco::DataType::S32>(i) = shape[i];
69 }
70 shape_node->name("Reshape/shape");
71 return shape_node;
72}

74CircleNode *CircleReshapeGraphBuilder::build_node(const circle::OperatorT &op,
75 const std::vector<CircleNode *> &inputs,
76 loco::Graph *graph) const
77{
78 // If the second input is not provided, generate it based on the value of the attribute.
79 // TODO Presence of the second input is the current requirement of the IR.
80 auto *shape_node = (inputs.size() == 2) ? inputs.at(1) : nullptr;
81 if (shape_node == nullptr)
82 {
83 const auto *options = op.builtin_options.AsReshapeOptions();
84 if (options != nullptr)
85 shape_node = create_shape_node(options->new_shape, graph);
86 else
87 {
88 shape_node = graph->nodes()->create<CircleOutputDummy>();
89 shape_node->dtype(loco::DataType::S32);
90 shape_node->rank(0);
91 shape_node->name("Reshape/dummy");
92 }
93 }
94
95 auto *node = graph->nodes()->create<CircleReshape>();
96 node->tensor(inputs.at(0));
97 node->shape(shape_node);
98
99 const auto *options = op.builtin_options.AsReshapeOptions();
100 if (options)
101 setup_shape_attribute(options->new_shape, node);
102
103 return node;
104}
