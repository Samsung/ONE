30int64_t value_from_circle_const(const luci::CircleConst *node, uint32_t idx)
31{
32 assert(node->rank() == 1 && node->dim(0).value() > idx);
33 assert(node->dtype() == loco::DataType::S64 || node->dtype() == loco::DataType::S32);

40bool remove_no_effect_slice(luci::CircleNode *node)
41{
42 auto target_node = dynamic_cast<luci::CircleSlice *>(node);
43 if (target_node == nullptr)
44 return false;
45
46 auto begin_const = dynamic_cast<luci::CircleConst *>(target_node->begin());
47 if (begin_const == nullptr)
48 return false;
49
50 auto size_const = dynamic_cast<luci::CircleConst *>(target_node->size());
51 if (size_const == nullptr)
52 return false;
53
54 // Check input output shape.
55 auto input_node = loco::must_cast<luci::CircleNode *>(target_node->input());
56 for (uint32_t i = 0; i < input_node->rank(); i++)
57 {
58 if (value_from_circle_const(begin_const, i) != 0)
59 return false;
60
61 int64_t size_value = value_from_circle_const(size_const, i);
62 if (size_value == -1)
63 continue;
64 if (size_value != static_cast<int64_t>(input_node->dim(i).value()))
65 return false;
66
67 if (!input_node->dim(i).known())
68 return false;
69 }
70 replace(target_node).with(input_node);
71 return true;
72}

97bool RemoveUnnecessarySlicePass::run(loco::Graph *g)
98{
