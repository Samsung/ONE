26bool canonicalize_pad(loco::Graph *graph, moco::TFPad *node)
27{
46 auto pad_node = graph->nodes()->create<loco::TensorConstantPad>();
47
48 auto constant_node = graph->nodes()->create<loco::ConstGen>();
49
50 auto input_node = node->input();
51 // TODO: support other dtype.
52 assert(loco::dtype_get(input_node) == loco::DataType::FLOAT32);
53 constant_node->dtype(loco::DataType::FLOAT32);
54 // TODO: constant node changes to scalar when it is implemented.
55 constant_node->shape({1});
56 constant_node->size<loco::DataType::FLOAT32>(1);
57 constant_node->at<loco::DataType::FLOAT32>(0) = 0.0f;
58
59 auto const_paddings_node = loco::must_cast<loco::ConstGen *>(node->paddings());
60 // TODO: support S64 type.
61 assert(const_paddings_node->dtype() == loco::DataType::S32);
62 assert(const_paddings_node->rank() == 2);
63 assert(const_paddings_node->dim(1).value() == 2);
64
65 auto padding = pad_node->padding();
66 uint32_t padding_rank = const_paddings_node->dim(0).value();
67 padding->rank(padding_rank);
68
69 for (uint32_t i = 0; i < padding_rank; i++)
70 {
71 padding->front(i) = const_paddings_node->at<loco::DataType::S32>(i << 1);
72 padding->back(i) = const_paddings_node->at<loco::DataType::S32>((i << 1) + 1);
73 }
74
75 // update connections
76 pad_node->input(input_node);
77 pad_node->constant(constant_node);
78
79 // replace node
80 replace(node).with(pad_node);
81
82 return true;
83}
