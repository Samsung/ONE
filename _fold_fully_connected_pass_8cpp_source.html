37bool set_params(const luci::CircleFullyConnected *node, compute::FullyConnected &cfc)
38{
39 assert(node);
40
41 LOGGER(l);
42
43 // NOTE only support default for now
44 if (node->weights_format() != luci::CircleFullyConnected::WeightsFormat::DEFAULT)
45 {
46 WARN(l) << "FoldFullyConnectedPass unsupported weights_format: "
47 << uint32_t(node->weights_format());
48 return false;
49 }
50 cfc.params().weights_format = compute::FullyConnectedWeightsFormat::kDefault;
51
52 compute::FusedActFunc fac;
53 if (!to_compute(node->fusedActivationFunction(), fac))
54 {
55 WARN(l) << "FoldFullyConnectedPass unsupported activation: "
56 << uint32_t(node->fusedActivationFunction());
57 return false;
58 }
59 cfc.fused_act_func(fac);
60
61 return true;
62}

81bool fold_fully_connected(luci::CircleFullyConnected *node)
82{
83 RETURN_FALSE_UNLESS(node != nullptr);
84
85 auto const input = dynamic_cast<luci::CircleConst *>(node->input());
86 auto const weights = dynamic_cast<luci::CircleConst *>(node->weights());
87 auto const bias = dynamic_cast<luci::CircleConst *>(node->bias());
88 auto const no_bias = dynamic_cast<luci::CircleOutputExclude *>(node->bias());
89
90 RETURN_FALSE_UNLESS(input != nullptr);
91 RETURN_FALSE_UNLESS(weights != nullptr);
92 RETURN_FALSE_UNLESS(bias != nullptr or no_bias != nullptr);
93
94 RETURN_FALSE_UNLESS(input->dtype() == loco::DataType::FLOAT32);
95 RETURN_FALSE_UNLESS(weights->dtype() == loco::DataType::FLOAT32);
96
97 auto const input_data = &input->at<loco::DataType::FLOAT32>(0);
98 auto const weights_data = &weights->at<loco::DataType::FLOAT32>(0);
99 float *bias_data = nullptr;
100 if (bias)
101 {
102 RETURN_FALSE_UNLESS(bias->dtype() == loco::DataType::FLOAT32);
103 bias_data = &bias->at<loco::DataType::FLOAT32>(0);
104 }
105
106 auto static_shape = [](luci::CircleNode *node) {
107 loco::TensorShape shape;
108 if (not node)
109 return shape;
110 shape.rank(node->rank());
111 for (uint32_t i = 0; i < node->rank(); ++i)
112 shape.dim(i) = node->dim(i);
113 return shape;
114 };
115
116 compute::FullyConnected comp_fc{};
117 if (!set_params(node, comp_fc))
118 return false;
119 comp_fc.input(static_shape(input), input_data);
120 comp_fc.weights(static_shape(weights), weights_data);
121 comp_fc.bias(static_shape(bias), bias_data);
122
123 comp_fc.keep_num_dims(node->keep_num_dims());
124
125 if (!comp_fc.prepare())
126 return false;
127
128 const auto &output_shape = comp_fc.output_shape();
129 assert(is_same_shape(node, output_shape));
130 auto output_size = loco::element_count(&output_shape);
131
132 auto constant = node->graph()->nodes()->create<luci::CircleConst>();
133 {
134 constant->dtype(node->dtype());
135 constant->rank(node->rank());
136 for (uint32_t i = 0; i < node->rank(); ++i)
137 constant->dim(i).set(node->dim(i).value());
138 constant->shape_status(luci::ShapeStatus::VALID);
139 constant->size<loco::DataType::FLOAT32>(output_size);
140 constant->name(node->name());
141 }
142 auto constant_data = &constant->at<loco::DataType::FLOAT32>(0);
143 comp_fc.output(constant_data);
144 comp_fc.compute();
145
146 loco::replace(node).with(constant);
147
148 return true;
149}

156bool FoldFullyConnectedPass::run(loco::Graph *g)
157{
158 bool changed = false;
159 for (auto node : loco::active_nodes(loco::output_nodes(g)))
160 {
161 auto fc = dynamic_cast<CircleFullyConnected *>(node);
162
163 if (fold_fully_connected(fc))
164 changed = true;
165 }
166
167 return changed;
168}
