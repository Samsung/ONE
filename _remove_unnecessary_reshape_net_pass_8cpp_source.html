24bool acceptable_intermediate_op(const loco::Node *node)
25{
26 if (not node)
27 return false;
28
29 const auto opcode = loco::must_cast<const luci::CircleNode *>(node)->opcode();
30
31 switch (opcode)
32 {
33 case luci::CircleOpcode::ADD:
34 case luci::CircleOpcode::MUL:
35 case luci::CircleOpcode::TANH:
36 case luci::CircleOpcode::LOGISTIC:
37 break;
38
39 default:
40 return false;
41 }
42
43 return true;
44}

46bool same_shape(const loco::Node *a, const loco::Node *b)
47{
48 auto a_cnode = loco::must_cast<const luci::CircleNode *>(a);
49 auto b_cnode = loco::must_cast<const luci::CircleNode *>(b);
50
51 if (a_cnode->rank() != b_cnode->rank())
52 return false;
53
54 for (uint32_t i = 0; i < a_cnode->rank(); i++)
55 {
56 if (not(a_cnode->dim(i) == b_cnode->dim(i)))
57 return false;
58 }
59 return true;
60}

62class PreReshapeFinder

64public:
65 PreReshapeFinder(const luci::CircleReshape *post_reshape) : _post_reshape(post_reshape)

67 assert(post_reshape != nullptr); // FIX_CALLER_UNLESS

70public:
71 // Return true if pre_reshapes are found
72 bool collect_pre_reshapes(loco::Node *node)
73 {
74 // TODO Support diamond case
75 if (loco::succs(node).size() != 1)
76 return false;
77
78 if (auto pre_reshape = dynamic_cast<luci::CircleReshape *>(node))
79 {
80 // Check ifm of pre-reshape and ofm of post_reshape
81 if (not same_shape(pre_reshape->tensor(), _post_reshape))
82 return false;
83
84 // Check ofm of pre-reshape and ifm of post_reshape
85 if (not same_shape(pre_reshape, _post_reshape->tensor()))
86 return false;
87
88 _pre_reshapes.emplace_back(pre_reshape);
89 return true;
90 }
91
92 if (not acceptable_intermediate_op(node))
93 return false;
94
95 for (uint32_t i = 0; i < node->arity(); i++)
96 {
97 if (not collect_pre_reshapes(node->arg(i)))
98 return false;
99 }
100
101 return true;
102 }
103
104public:
105 std::vector<luci::CircleReshape *> pre_reshapes(void) const { return _pre_reshapes; }
106
107private:
108 const luci::CircleReshape *_post_reshape = nullptr;

110};

112bool remove_unnecessary_reshape_net(luci::CircleReshape *reshape)
113{
114 PreReshapeFinder finder(reshape);
115 if (not finder.collect_pre_reshapes(reshape->tensor()))
116 return false;
117
118 // Remove pre_reshapes
119 for (auto pre_reshape : finder.pre_reshapes())
120 {
121 loco::replace(pre_reshape).with(pre_reshape->tensor());
122 }
123
124 // Remove post_reshape
125 loco::replace(reshape).with(reshape->tensor());
126
127 return true;
128}

158bool RemoveUnnecessaryReshapeNetPass::run(loco::Graph *g)
159{
