27void evalFloat(const circle::Tensor *input, const circle::Tensor *weights,
28 const circle::Tensor *bias, const circle::Tensor *output,
29 const circle::FullyConnectedOptions *options, BaseRuntimeGraph *runtime_graph)
30{
31 float activation_min{};
32 float activation_max{};
33 kernels::calculateActivationRange(luci_actfunc(options->fused_activation_function()),
34 &activation_min, &activation_max);
35
36 luci_interpreter_pal::FullyConnectedParams params{};
37 params.float_activation_min = activation_min;
38 params.float_activation_max = activation_max;
39
40 auto *input_data = runtime_graph->getDataByTensor(input);
41 auto *output_data = runtime_graph->getDataByTensor(output);
42
43 auto *weights_data = runtime_graph->getConstDataByTensor(weights);
44 auto *bias_data = runtime_graph->getConstDataByTensor(bias);
45
46 assert(input_data != nullptr);
47 assert(weights_data != nullptr);
48 assert(output_data != nullptr);
49
50 int32_t input_shape[kMaxSmallSize];
51 kernels::getTensorDims(input, runtime_graph, input_shape);
52
53 int32_t weight_shape[kMaxSmallSize];
54 kernels::getTensorDims(weights, runtime_graph, weight_shape);
55
56 int32_t output_shape[kMaxSmallSize];
57 kernels::getTensorDims(output, runtime_graph, output_shape);
58
59 luci_interpreter_pal::FullyConnected(
60 params, input_shape, kernels::getTensorData<float>(input_data), weight_shape,
61 kernels::getTensorData<float>(weights_data), kernels::getTensorData<float>(bias_data),
62 output_shape, kernels::getTensorData<float>(output_data));
63}

66void evalQuantized(const circle::Tensor *input, const circle::Tensor *weights,
67 const circle::Tensor *bias, const circle::Tensor *output,
68 const circle::FullyConnectedOptions *options, BaseRuntimeGraph *runtime_graph)
69{
70 double real_multiplier = 0.0;
71 int output_shift;
72 int32_t output_activation_min;
73 int32_t output_activation_max;
74 int32_t output_multiplier;
75 real_multiplier = kernels::getQuantizedConvolutionMultipler(
76 Tensor::scale(input), Tensor::scale(weights), Tensor::scale(output));
77 kernels::quantizeMultiplier(real_multiplier, &output_multiplier, &output_shift);
78 kernels::calculateActivationRangeQuantized(luci_actfunc(options->fused_activation_function()),
79 output, &output_activation_min,
80 &output_activation_max);
81
82 int32_t input_offset = -Tensor::zero_point(input);
83 int32_t filter_offset = -Tensor::zero_point(weights);
84 int32_t output_offset = Tensor::zero_point(output);
85
86 luci_interpreter_pal::FullyConnectedParams op_params{};
87 op_params.input_offset = input_offset;
88 op_params.weights_offset = filter_offset;
89 op_params.output_offset = output_offset;
90 op_params.output_multiplier = output_multiplier;
91 op_params.output_shift = output_shift;
92 op_params.quantized_activation_min = output_activation_min;
93 op_params.quantized_activation_max = output_activation_max;
94 op_params.lhs_cacheable = false;
95 op_params.rhs_cacheable = false;
96
97 auto *input_data = runtime_graph->getDataByTensor(input);
98 auto *output_data = runtime_graph->getDataByTensor(output);
99
100 auto *weights_data = runtime_graph->getConstDataByTensor(weights);

125void configure_kernel_CircleFullyConnected(const circle::Operator *cur_op,
126 BaseRuntimeGraph *runtime_graph)
127{
128 const auto input_index = cur_op->inputs()->operator[](0);
129 const auto weight_index = cur_op->inputs()->operator[](1);
130 const auto bias_index = cur_op->inputs()->operator[](2);
131 const auto output_index = cur_op->outputs()->operator[](0);
132
133 assert(input_index != -1);
134 assert(weight_index != -1);
135 assert(output_index != -1);
136
137 const auto input = runtime_graph->getCircleTensorByIndex(input_index);
138 const auto weights = runtime_graph->getCircleTensorByIndex(weight_index);
139 const auto bias = runtime_graph->getCircleTensorByIndex(bias_index);
140 const auto output = runtime_graph->getCircleTensorByIndex(output_index);
141
142 assert(input != nullptr);
143 assert(weights != nullptr);
144 assert(output != nullptr);
145
146#ifndef DIS_FLOAT
147 if (Tensor::element_type(weights) == DataType::FLOAT32)
148 {
149 LUCI_INTERPRETER_CHECK(Tensor::element_type(input) == DataType::FLOAT32);
150 LUCI_INTERPRETER_CHECK(Tensor::element_type(output) == DataType::FLOAT32);
151 LUCI_INTERPRETER_CHECK(!bias || Tensor::element_type(bias) == DataType::FLOAT32)
152 }
153#endif // DIS_FLOAT
154#ifndef DIS_QUANT
155 else if (Tensor::element_type(weights) == DataType::U8)
156 {
157 LUCI_INTERPRETER_CHECK(Tensor::element_type(input) == DataType::U8);
158 LUCI_INTERPRETER_CHECK(Tensor::element_type(output) == DataType::U8);
159 LUCI_INTERPRETER_CHECK(!bias || Tensor::element_type(bias) == DataType::S32)
160 }
161 else if (Tensor::element_type(weights) == DataType::S8)
162 {
163 LUCI_INTERPRETER_CHECK(Tensor::element_type(input) == DataType::S8);
164 LUCI_INTERPRETER_CHECK(Tensor::element_type(output) == DataType::S8);
165 LUCI_INTERPRETER_CHECK(!bias || Tensor::element_type(bias) == DataType::S32)
166 }
167#endif // DIS_QUANT
168 else
169 {
170 assert(false && "Unsupported type.");
171 }
172
173 LUCI_INTERPRETER_CHECK(Tensor::num_dims(weights) == 2);
174 LUCI_INTERPRETER_CHECK(bias == nullptr || Tensor::num_elements(bias) == Tensor::dim(weights, 0));
175
176#ifdef DIS_DYN_SHAPES
177 int32_t input_num_elements = Tensor::num_elements(input);
178 LUCI_INTERPRETER_CHECK(input_num_elements % Tensor::dim(weights, 1) == 0);
179#endif // DIS_DYN_SHAPES
180
181 if (bias)
182 LUCI_INTERPRETER_CHECK(Tensor::num_elements(bias) == Tensor::dim(weights, 0));
183
184 const auto *options = cur_op->builtin_options_as_FullyConnectedOptions();
185
186 // TODO: handle with it
187 assert(options->keep_num_dims() == false);
188}

191void execute_kernel_CircleFullyConnected(const circle::Operator *cur_op,
192 BaseRuntimeGraph *runtime_graph)
193{
194 const auto input_index = cur_op->inputs()->operator[](0);
195 const auto weight_index = cur_op->inputs()->operator[](1);
196 const auto bias_index = cur_op->inputs()->operator[](2);
197 const auto output_index = cur_op->outputs()->operator[](0);
198
199 assert(input_index != -1);
200 assert(weight_index != -1);
201 assert(output_index != -1);
202
203 const auto input = runtime_graph->getCircleTensorByIndex(input_index);
204 const auto weights = runtime_graph->getCircleTensorByIndex(weight_index);
205 const auto bias = runtime_graph->getCircleTensorByIndex(bias_index);
206 const auto output = runtime_graph->getCircleTensorByIndex(output_index);
207
208 assert(input != nullptr);
209 assert(weights != nullptr);
210 assert(output != nullptr);
211
212 const auto *options = cur_op->builtin_options_as_FullyConnectedOptions();
213
214 switch (Tensor::element_type(input))
215 {
216#ifndef DIS_QUANT
217 case DataType::U8:
218 evalQuantized(input, weights, bias, output, options, runtime_graph);
219 break;
220#endif // DIS_QUANT
221#ifndef DIS_FLOAT
222 case DataType::FLOAT32:
223 evalFloat(input, weights, bias, output, options, runtime_graph);
224 break;
225#endif // DIS_FLOAT
226 default:
227 assert(false && "Unsupported type.");
228 }
229}
