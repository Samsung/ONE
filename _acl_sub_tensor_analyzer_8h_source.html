36{
37public:
42 AclSubTensorAnalyzer(const ir::Graph &graph) : _graph{graph}
43 {
44 // DO NOTHING
45 }

47public:
48 void setLayout(ir::Layout layout) { _current_op_layout = layout; }
49
50 void setUsePadding() { usePadding = true; }
51
52 void visit(const ir::operation::Concat &node) override
53 {
54 // If operator is concat, fill subsumption info
55 int32_t axis_raw = node.param().axis;
56
57 const auto &output_index = node.getOutputs().at(0);
58 const auto &inputs = node.getInputs();
59
60 int32_t axis_point = 0;
61 const auto rank = _graph.operands().at(output_index).shape().rank();
62 int32_t axis = axis_raw < 0 ? (axis_raw + rank) : axis_raw;
63 assert(rank > axis);
64
65 // Concat elimination when axis is last dimension is not supported
66 // https://github.com/Samsung/ONE/issues/4407
67 // TODO Enable if backend don't use padding
68 if ((axis == rank - 1) && usePadding)
69 return;
70
71 for (const auto &ind : inputs)
72 {
79 if (_graph.operands().at(ind).isConstant() || _graph.getInputs().contains(ind) ||
80 _parent_map.find(ind) != _parent_map.end())
81 {
82 return;
83 }
84 }
85
86 for (const auto &input_index : inputs)
87 {
88 auto input_shape = _graph.operands().at(input_index).shape();
89 assert(rank == input_shape.rank());
90
91 ir::Coordinates coordinate_info{};
92 for (int i = 0; i < rank; i++)
93 {
94 coordinate_info.set(i, 0);
95 }
96 coordinate_info.set(axis, axis_point);
97
98 _parent_map.emplace(input_index,
99 cl_common::ParentInfo{output_index, _current_op_layout, coordinate_info});
100
101 axis_point += input_shape.dim(axis);
102 }
103 }
104
105 std::unordered_map<ir::OperandIndex, cl_common::ParentInfo> &&releaseParentMap()
106 {
107 return std::move(_parent_map);
108 }
109
110private:
111 const ir::Graph &_graph;
112 std::unordered_map<ir::OperandIndex, cl_common::ParentInfo> _parent_map;
113 ir::Layout _current_op_layout{ir::Layout::UNKNOWN};
114 bool usePadding{false};
115};
