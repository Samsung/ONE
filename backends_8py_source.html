40def get_list(cmdname):

42 backend_set = set()
43
44 # bin folder
45 files = [f for f in glob.glob(dir_path + '/../' + cmdname)]
46 # backends folder
47 files += [
48 f for f in glob.glob(dir_path + '/../../backends/**/' + cmdname, recursive=True)

52 backends_list = []

54 base = ntpath.basename(cand)
55 if (not base in backend_set) and os.path.isfile(cand) and os.access(
56 cand, os.X_OK):
57 backend_set.add(base)
58 backends_list.append(cand)
