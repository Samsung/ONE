29static void resolveAxes(const int32_t *axes_data, int num_axes, tflite::MeanParams *params)
30{
31 params->axis_count = num_axes;
32 for (int i = 0; i < num_axes; ++i)
33 {
34 params->axis[i] = static_cast<int16>(axes_data[i]);
35 }
36 for (int i = num_axes; i < 4; ++i)
37 {
38 params->axis[i] = 1;
39 }
40}

42// Returns the number of axes that will be reduced. Removes duplicates.
43static int getAxisReductionCount(const int32_t *axes_data, int num_axes, int input_num_dims)
44{
45 int reduction_count = num_axes;
46 for (int i = 0; i < num_axes; ++i)
47 {
48 int current = axes_data[i] >= 0 ? axes_data[i] : axes_data[i] + input_num_dims;
49 assert(current >= 0 && current < input_num_dims);
50 for (int j = 0; j < i; j++)
51 {
52 int previous = axes_data[j] >= 0 ? axes_data[j] : axes_data[j] + input_num_dims;
53 // This checks for duplicate axis
54 if (current == previous)
55 {
56 --reduction_count;
57 break;
58 }
59 }
60 }
61 return reduction_count;
62}

64static Shape getOutputShape(const Shape &input_shape, const int32_t *axes_data, int num_axes,
65 bool keep_dims)
66{
67 int input_num_dims = input_shape.num_dims();
68 if (input_num_dims == 0)
69 {
70 return Shape(0);
71 }
72
73 if (keep_dims)
74 {
75 Shape output_shape(input_num_dims);
76 for (int idx = 0; idx < input_num_dims; ++idx)
77 {
78 bool is_axis = false;
79 for (int axis_idx = 0; axis_idx < num_axes; ++axis_idx)
80 {
81 if (axes_data[axis_idx] == idx || axes_data[axis_idx] + input_num_dims == idx)
82 {
83 is_axis = true;
84 break;
85 }
86 }
87 if (is_axis)
88 {
89 output_shape.dim(idx) = 1;
90 }
91 else
92 {
93 output_shape.dim(idx) = input_shape.dim(idx);
94 }
95 }
96 return output_shape;
97 }
98 else
99 {
100 int num_reduce_axes = getAxisReductionCount(axes_data, num_axes, input_num_dims);
101 Shape output_shape(input_num_dims - num_reduce_axes);
102 int num_skip_axes = 0;
103 for (int idx = 0; idx < input_num_dims; ++idx)
104 {
105 bool is_axis = false;
106 for (int axis_idx = 0; axis_idx < num_axes; ++axis_idx)
107 {
108 if (axes_data[axis_idx] == idx || axes_data[axis_idx] + input_num_dims == idx)
109 {
110 ++num_skip_axes;
111 is_axis = true;
112 break;
113 }
114 }
115 if (!is_axis)
116 {
117 output_shape.dim(idx - num_skip_axes) = input_shape.dim(idx);
118 }
119 }
120 return output_shape;
121 }

131void Mean::configure()
132{
133 LUCI_INTERPRETER_CHECK(input()->element_type() == output()->element_type());
134 LUCI_INTERPRETER_CHECK(axes()->element_type() == DataType::S32);
135 if (input()->element_type() == DataType::S16)
136 {
137 LUCI_INTERPRETER_CHECK(input()->zero_point() == 0 && output()->zero_point() == 0);
138 }
139
140 const Shape &input_shape = input()->shape();
141 int input_num_dims = input_shape.num_dims();
142
143 const auto *axes_data = getTensorData<int32_t>(axes());
144 int num_axes = axes()->shape().num_elements();
145 assert(num_axes <= 4);
146 // TODO: enable it only if kernel with dynamic shapes
147 Shape output_shape = getOutputShape(input_shape, axes_data, num_axes, _params.keep_dims);
148 output()->resize(output_shape);
149
150 tflite::MeanParams params{};
151 resolveAxes(axes_data, num_axes, &params);
152 _need_temporaries = !(
153 _params.keep_dims && input_num_dims == 4 && params.axis_count == 2 &&
154 ((params.axis[0] == 1 && params.axis[1] == 2) || (params.axis[0] == 2 && params.axis[1] == 1)));
155 if (_need_temporaries)
156 {
157 auto temp_index = getOutputTensors()[1];
158 auto resolved_axes = getOutputTensors()[2];
159 auto temp_sum = getOutputTensors()[3];
160
161 temp_index->resize(Shape(input_num_dims));
162 resolved_axes->resize(Shape(num_axes));
163 temp_sum->resize(output()->shape());
164 }
165 else
166 {
167 auto temp_index = getOutputTensors()[1];
168 auto resolved_axes = getOutputTensors()[2];
169 auto temp_sum = getOutputTensors()[3];
170
171 temp_index->set_allocatable(false);
172 resolved_axes->set_allocatable(false);
173 temp_sum->set_allocatable(false);
174 }
175}

177void Mean::execute() const
178{
179 switch (input()->element_type())
180 {
181 case DataType::FLOAT32:
182 evalFloat();
183 break;
184 case DataType::U8:
185 evalQuantized();
186 break;
187 case DataType::S16:
188 evalQuantizedS16();
189 break;
190 default:
191 assert(false && "Unsupported type.");
192 }
193}

195void Mean::evalFloat() const
196{
197 const Shape &input_shape = input()->shape();
198 int input_num_dims = input_shape.num_dims();
199 const auto *axes_data = getTensorData<int32_t>(axes());
200 int num_axes = axes()->shape().num_elements();
201
202 tflite::MeanParams params{};
203 resolveAxes(axes_data, num_axes, &params);
204
205 auto temp_index = getOutputTensors()[1];
206 auto resolved_axes = getOutputTensors()[2];
207 auto temp_sum = getOutputTensors()[3];
208
209 // Defer to specialized implementation for 4D Mean across axes 1 & 2.
210 if (_params.keep_dims && input_num_dims == 4 && params.axis_count == 2 &&
211 ((params.axis[0] == 1 && params.axis[1] == 2) ||
212 (params.axis[0] == 2 && params.axis[1] == 1)))
213 {
214 tflite::reference_ops::Mean(params, getTensorShape(input()), getTensorData<float>(input()),
215 getTensorShape(output()), getTensorData<float>(output()));
216 }
217 else
218 {
219 tflite::reference_ops::Mean(getTensorData<float>(input()), getTensorShape(input()).DimsData(),
220 input()->shape().num_dims(), getTensorData<float>(output()),
221 getTensorShape(output()).DimsData(), output()->shape().num_dims(),
222 axes_data, num_axes, _params.keep_dims,
223 getTensorData<int>(temp_index), getTensorData<int>(resolved_axes),
224 getTensorData<float>(temp_sum));
225 }

228void Mean::evalQuantized() const
229{
230 const Shape &input_shape = input()->shape();
231 int input_num_dims = input_shape.num_dims();
232 const auto *axes_data = getTensorData<int32_t>(axes());
233 int num_axes = axes()->shape().num_elements();
234
235 tflite::MeanParams params{};
236 resolveAxes(axes_data, num_axes, &params);
237
238 auto temp_index = getOutputTensors()[1];
239 auto resolved_axes = getOutputTensors()[2];
240 auto temp_sum = getOutputTensors()[3];
241
242 // Defer to specialized implementation for 4D Mean across axes 1 & 2.
243 if (_params.keep_dims && input_num_dims == 4 && params.axis_count == 2 &&
244 ((params.axis[0] == 1 && params.axis[1] == 2) ||
245 (params.axis[0] == 2 && params.axis[1] == 1)))
246 {
247 tflite::reference_ops::Mean(params, getTensorShape(input()), getTensorData<uint8_t>(input()),
248 input()->zero_point(), input()->scale(), getTensorShape(output()),
249 getTensorData<uint8_t>(output()), output()->zero_point(),
250 output()->scale());
251 }
252 else if (input()->zero_point() == output()->zero_point() && input()->scale() == output()->scale())
253 {
254 tflite::reference_ops::Mean(getTensorData<uint8_t>(input()), getTensorShape(input()).DimsData(),
255 input()->shape().num_dims(), getTensorData<uint8_t>(output()),
256 getTensorShape(output()).DimsData(), output()->shape().num_dims(),
257 axes_data, num_axes, _params.keep_dims,
258 getTensorData<int>(temp_index), getTensorData<int>(resolved_axes),
259 getTensorData<int>(temp_sum));
260 }
261 else
262 {
263 tflite::reference_ops::QuantizedMeanOrSum<>(
264 getTensorData<uint8_t>(input()), input()->zero_point(), input()->scale(),
265 getTensorShape(input()).DimsData(), input()->shape().num_dims(),
266 getTensorData<uint8_t>(output()), output()->zero_point(), output()->scale(),
267 getTensorShape(output()).DimsData(), output()->shape().num_dims(), axes_data, num_axes,
268 _params.keep_dims, getTensorData<int>(temp_index), getTensorData<int>(resolved_axes),
269 getTensorData<int>(temp_sum),
270 /*compute_sum=*/false);
271 }

274void Mean::evalQuantizedS16() const
275{
276 const auto *input_data = getTensorData<int16_t>(input());
277 auto *output_data = getTensorData<int16_t>(output());
278
279 const Shape &input_shape = input()->shape();
280 const Shape &output_shape = output()->shape();
281
282 const auto *axes_data = getTensorData<int32_t>(axes());
283 const int num_axes = axes()->shape().num_elements();
284
285 constexpr int32_t output_min = -std::numeric_limits<int16_t>::max();
286 constexpr int32_t output_max = std::numeric_limits<int16_t>::max();
287
288 // Defer to specialized implementation for 4D Mean across axes 1 & 2.
289 if (_params.keep_dims && input_shape.num_dims() == 4 && num_axes == 2 &&
290 ((axes_data[0] == 1 && axes_data[1] == 2) || (axes_data[0] == 2 && axes_data[1] == 1)))
291 {
292 const int32_t batches = input_shape.dim(0);
293 const int32_t input_height = input_shape.dim(1);
294 const int32_t input_width = input_shape.dim(2);
295 const int32_t depth = input_shape.dim(3);
296 assert(output_shape.num_dims() == 4);
297 assert(output_shape.dim(0) == batches);
298 assert(output_shape.dim(1) == 1);
299 assert(output_shape.dim(2) == 1);
300 assert(output_shape.dim(3) == depth);
301
302 const double real_multiplier =
303 static_cast<double>(input()->scale()) / static_cast<double>(output()->scale());
304
305 int32_t output_multiplier{};
306 int output_shift{};
307 quantizeMultiplier(real_multiplier, &output_multiplier, &output_shift);
308
309 const int32_t num_elements_in_axes = input_height * input_width;
310
311 for (int32_t batch = 0; batch < batches; ++batch)
312 {
313 for (int32_t c = 0; c < depth; ++c)
314 {
315 int32_t acc = 0;
316 for (int32_t in_y = 0; in_y < input_height; ++in_y)
317 {
318 for (int32_t in_x = 0; in_x < input_width; ++in_x)
319 {
320 acc += input_data[calcOffset(input_shape, batch, in_y, in_x, c)];
321 }
322 }
323 int32_t scaled_acc =
324 tflite::MultiplyByQuantizedMultiplier(acc, output_multiplier, output_shift);
325 // Divide by the number of elements rounding to the nearest integer.
326 scaled_acc = scaled_acc > 0
327 ? (scaled_acc + num_elements_in_axes / 2) / num_elements_in_axes
328 : (scaled_acc - num_elements_in_axes / 2) / num_elements_in_axes;
329
330 scaled_acc = std::max(scaled_acc, output_min);
331 scaled_acc = std::min(scaled_acc, output_max);
332
333 output_data[calcOffset(output_shape, batch, 0, 0, c)] = scaled_acc;
334 }
335 }
336 }
337 else
338 {
339 assert(false && "Unsupported configuration.");
340 }
341}
