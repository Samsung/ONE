43void ConsoleReporter::notify(const hermes::Message *m)
44{
45 const char *env_color_p = std::getenv("ONE_HERMES_COLOR");
46 if (env_color_p)
47 {
48 auto env_color_str = std::string(env_color_p);
49 if ((env_color_str == "1") or (env_color_str == "ON"))
50 _is_colored = true;
51 }
52
53 if (_is_colored)
54 {
55 switch (m->get_severity())
56 {
57 case FATAL:
58 std::cout << kTermColorRedTextCode << kTermBoldTextCode << kTermUnderlineTextCode;
59 break;
60 case ERROR:
61 std::cout << kTermColorRedTextCode;
62 break;
63 case WARN:
64 std::cout << kTermColorOrangeTextCode;
65 break;
66 case INFO:
67 std::cout << kTermColorGreenTextCode;
68 break;
69 case VERBOSE:
70 std::cout << kTermColorResetAllCode;
71 break;
72 };
73 }
74 for (uint32_t n = 0; n < m->text()->lines(); ++n)
75 {
76 std::cout << m->text()->line(n) << std::endl;
77 }
78 if (_is_colored)
79 {
80 std::cout << kTermColorResetAllCode;
81 }
82}
