47class ParsedTensor final
48{
49public:
50 enum class Kind
51 {
52 Input,
53 Output
54 };
55
56 ParsedTensor() = delete;
57
58 ParsedTensor(const Kind kind, const std::string &name, const DataType &dtype,
59 const std::vector<int32_t> &shape)
60 : _kind(kind), _dtype(dtype)
61 {
62 _tensor_name.assign(name);
63
64 _shape.resize(shape.size());
65 for (int rank = 0; rank < shape.size(); rank++)
66 _shape.dim(rank) = shape.at(rank);
67 }
68
69 ~ParsedTensor()
70 { /* empty */
71 }
72
73public:
74 Kind kind() const { return _kind; }
75
76 const std::string &name() { return _tensor_name; }
77
78 const nncc::core::ADT::tensor::Shape &shape() const { return _shape; }
79 // TODO This method is a bridge between testinfo and testinfo-v2. When testinfo-v2 is introduced,
80 // this method will be removed.
81 nncc::core::ADT::tensor::Shape &mutable_shape() { return _shape; }
82
83 const DataType &dtype() const { return _dtype; }
84
89 std::string nodeName() const { return _tensor_name.substr(0, _tensor_name.find(":")); }
90
95 int tensorIndex() const
96 {
97 int separator = _tensor_name.find(":");
98
99 // sanity check
100 if (separator == std::string::npos)
101 throw oops::UserExn("Tensor name in wrong format", "name", _tensor_name);
102
103 return std::stoi(_tensor_name.substr(separator + 1, _tensor_name.length() - separator - 1));
104 }
105
106public:
107 bool isFloatTensor() const { return _dtype == DataType::FLOAT32; }
108 bool hasShape() const { return _has_shape; }
109
110private:
111 Kind _kind;
112 std::string _tensor_name;
113 nncc::core::ADT::tensor::Shape _shape;
114 DataType _dtype;
115 // Now, every info file has a shape.
116 bool _has_shape = true;
117};
