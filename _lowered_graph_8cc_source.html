47void LoweredGraph::lowerGraph(const CompilerOptions &options)
48{
49 // Build backend contexts
50 auto &backend_manager = BackendManager::get();
51 // Create contexts for other backends
52 for (auto backend_str : options.backend_list)
53 {
54 backend_manager.loadBackend(backend_str);
55 auto backend = backend_manager.get(backend_str);
56
57 // TODO As the default value of backend list contains "cpu", "acl_cl" and "acl_neon", and some
58 // are not available on x64 or some other platforms. So this may be a workaround for x64 and
59 // we should change it back(throw if backend is not loaded) later.
60 if (!backend)
61 {
62 VERBOSE(LoweredGraph) << "Cannot load backend - " << backend_str << std::endl;
63 continue;
64 }
65 }
66 if (backend_manager.num_backends() == 0)
67 throw std::runtime_error{"No available backends loaded."};
68
69 // TODO Move "schedule" phase out of here
70 // Schedule
71 std::unique_ptr<BackendResolver> backend_resolver;
72 auto all_backends = backend_manager.getAll();
73 if (options.he_scheduler)
74 {
75 auto scheduler = HEScheduler(all_backends, options);
76 backend_resolver = scheduler.schedule(_graph);
77 _indexed_ranks = scheduler.getIndexedRanks();
78 }
79 else
80 {
81 auto scheduler = ManualScheduler(all_backends, options);
82 backend_resolver = scheduler.schedule(_graph);
83 }
84
85 makeLowerInfo(*backend_resolver);
86 VERBOSE(LoweredGraph) << "dump before mandatory passes" << std::endl;
87 dumper::text::dumpLoweredGraph(*this);
88
89 // Mandatory passes - kind of legalization(?)
90 pass::PassRunner{}
91 .append(std::make_unique<pass::ConstantInsertionPass>(*this))
92 .append(std::make_unique<pass::ConstantLoweringPass>(*this))
93 .append(std::make_unique<pass::PermutationOperationPass>(*this))
94 .append(std::make_unique<pass::PermutationInsertionPass>(*this))
95 .run();
96
97 dumpLowerInfo();
98
99 // Optimization passes (optional)
100 pass::PassRunner{}.append(std::make_unique<pass::PermutationEliminationPass>(*this)).run();
101
102 VERBOSE(LoweredGraph) << "Dump after all the passes" << std::endl;
103 for (auto operand : _graph.getInputs())
104 VERBOSE(LoweredGraph) << "Graph Input : " << operand << std::endl;
105 for (auto operand : _graph.getOutputs())
106 VERBOSE(LoweredGraph) << "Graph Output : " << operand << std::endl;
107 dumper::text::dumpLoweredGraph(*this);
108
109 // Graph verifications
110 {
111 assert(ir::verifier::InputOutputChecker().verify(_graph));
112 assert(ir::verifier::DAGChecker().verify(_graph));
113 assert(ir::verifier::EdgeChecker().verify(_graph));
114 }
115}

117void LoweredGraph::makeLowerInfo(const compiler::BackendResolver &backend_resolver)
118{
119 _graph.operands().iterate([&](const ir::OperandIndex &index, const ir::Operand &) {
120 lower_info().operand.set(index, std::make_unique<OperandLowerInfo>());
121 });
122
123 // Set operand lower info using assigned backends to operations
124 _graph.operations().iterate([&](const ir::OperationIndex &op_ind, const ir::Operation &) {
125 const ir::Operation &op = _graph.operations().at(op_ind);
126 auto backend = backend_resolver.getBackend(op_ind);
127 if (!backend)
128 {
129 throw std::runtime_error{"Fail to find backend for " + op.name() + " operation"};
130 }
131
132 auto frontend_layout = _graph.layout();
133
134 // The layout of each backend should be set at another place
135 // TODO Change setting layout of each backend at another place
136 auto backend_layout = backend->config()->supportLayout(op, frontend_layout);
137
138 for (auto ind : op.getInputs() | ir::Remove::UNDEFINED)
139 {
140 auto &operand_li = lower_info().operand.at(ind);
141 operand_li.addUsePermuteFactor(PermuteFactor{backend, backend_layout});
142 }
143 for (auto ind : op.getOutputs() | ir::Remove::UNDEFINED)
144 {
145 auto &operand_li = lower_info().operand.at(ind);
146 operand_li.addDefPermuteFactor(PermuteFactor{backend, backend_layout});
147 }
148 lower_info().operation.set(
149 op_ind, std::make_unique<compiler::OperationLowerInfo>(backend, backend_layout));
150 });
151
152 // Handle graph inputs and outputs
153 const auto builtin_backend = BackendManager::get().getBuiltin();
154 auto factor = PermuteFactor{builtin_backend, _graph.layout()};
155 for (auto index : _graph.getInputs() | ir::Remove::UNDEFINED)
156 {
157 auto &operand_li = lower_info().operand.at(index);
158 assert(operand_li.def_factors().empty());
159 operand_li.addDefPermuteFactor(factor);
160 }
161 for (auto index : _graph.getOutputs() | ir::Remove::UNDEFINED)
162 {
163 auto &operand_li = lower_info().operand.at(index);
164 operand_li.addUsePermuteFactor(factor);
165 }
166
167 // Handle variable tensors
168 _graph.operands().iterate([&](const ir::OperandIndex &index, ir::Operand &operand) {
169 // Some inputs of an operation could be non-constant, but not existed in graph inputs/outputs
170 // and not undefined operand - these are variable tensors. For example,
171 // UnidirectionalSequenceLSTM has such inputs.
172 if (operand.info().isVariable())
173 {
174 // The variable operand with buffer is not supported yet
175 assert(operand.data() == nullptr);
176 assert(operand.getUses().size() == 1 && !operand.getDef().valid());
177 auto operand_li = lower_info().operand.at(index);
178 assert(operand_li.def_factors().empty());
179 operand_li.addDefPermuteFactor(operand_li.use_factors().getOnlyElement());
180 }
181 });
182}

184void LoweredGraph::dumpLowerInfo()
185{
186 if (::onert::util::logging::ctx.enabled() == false)
187 return;
188
189 std::map<uint32_t, std::string> dumps;
190
191 _graph.operands().iterate([&](const ir::OperandIndex &index, ir::Operand &object) {
192 const auto operand_lower_info = lower_info().operand.getRawPtr(index);
193 assert(operand_lower_info);
194 if (!operand_lower_info->def_factors().empty() || !operand_lower_info->use_factors().empty())
195 {
196 auto shape_to_string = [](const ir::Shape &shape) {
197 std::stringstream sstream;
198 sstream << "{ ";
199 for (auto i = 0; i < shape.rank(); ++i)
200 sstream << (shape.dim(i)) << " ";
201 sstream << "}";
202 return sstream.str();
203 };
204
205 auto factors_to_string = [](const PermuteFactorSet &factors) {
206 std::string str;
207 for (auto factor : factors)
208 {
209 str += factor.backend()->config()->id();
210 str += "(" + to_string(factor.layout()) + ")";
211 str += " ";
212 }
213 return "{ " + str + "}";
214 };
215
216 auto operation_index_set_to_string = [](const ir::OperationIndexSet &operations) {
217 std::stringstream sstream;
218 sstream << "{ ";
219 for (auto op : operations)
220 sstream << op << " ";
221 sstream << "}";
222 return sstream.str();
223 };
224
225 auto data_to_str = [](const ir::Data *data) {
226 return (data ? (std::to_string(data->size()) + " bytes") : "N/A");
227 };
228
229 std::string shape_str = shape_to_string(object.shape());
230 std::string def_op = operation_index_set_to_string({object.getDef()});
231 std::string use_ops = operation_index_set_to_string(object.getUses());
232 std::string def_factors = factors_to_string(operand_lower_info->def_factors());
233 std::string use_factors = factors_to_string(operand_lower_info->use_factors());
234 std::stringstream sstream;
235 sstream << "Operand " << index << " Info" << std::endl;
236 sstream << " - Shape : " << shape_str << std::endl;
237 sstream << " - Def/Uses : Def " << def_op << " Uses " << use_ops << std::endl;
238 sstream << " - Data : " << data_to_str(object.data()) << std::endl;
239 sstream << " - LowerInfo : Def " << def_factors << " Uses " << use_factors << std::endl;
240 dumps.emplace(index.value(), sstream.str());
241 }
242 });
243
244 for (const auto &e : dumps)
245 {
246 if (!e.second.empty())
247 {
248 std::istringstream iss(e.second);
249 std::string line;
250 while (std::getline(iss, line))
251 VERBOSE(Lower) << line << std::endl;
252 }
253 }
254}
