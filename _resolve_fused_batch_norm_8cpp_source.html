33bool is_same_shape(moco::TFConst *lc, moco::TFConst *rc)
34{
35 if (lc->rank() != rc->rank())
36 return false;
37
38 for (auto r = 0; r < lc->rank(); ++r)
39 {
40 if (lc->dim(r).value() != rc->dim(r).value())
41 return false;
42 }
43 return true;
44}

46void copy_shape(const moco::TFConst *src, moco::TFConst *dst)
47{
48 assert(src != nullptr);
49 assert(dst != nullptr);
50
51 uint32_t rank = src->rank();
52 dst->rank(rank);
53 for (uint32_t index = 0; index < rank; ++index)
54 {
55 if (src->dim(index).known())
56 dst->dim(index) = src->dim(index).value();
57 else
58 dst->dim(index).unset();
59 }
60}

91bool resolve_to_muladd(loco::Graph *graph, moco::TFFusedBatchNorm *node)
92{
93 // LOGGER(lfbn);
94
95 auto tffbn_x = node->x();
96 if (tffbn_x == nullptr)
97 {
98 // This node is already converted
99 return false;
100 }
101
102 auto tffbn_scale = dynamic_cast<moco::TFConst *>(node->scale());
103 auto tffbn_offset = dynamic_cast<moco::TFConst *>(node->offset());
104 auto tffbn_mean = dynamic_cast<moco::TFConst *>(node->mean());
105 auto tffbn_variance = dynamic_cast<moco::TFConst *>(node->variance());
106
107 // all should be const
108 if (tffbn_scale == nullptr || tffbn_offset == nullptr || tffbn_mean == nullptr ||
109 tffbn_variance == nullptr)
110 {
111 // INFO(lfbn) << "TFFBN resolve_to_muladd: One of constant input node is not a constant"
112 // << std::endl;
113 return false;
114 }
115 assert(tffbn_scale->dtype() == loco::DataType::FLOAT32);
116 assert(tffbn_offset->dtype() == loco::DataType::FLOAT32);
117 assert(tffbn_mean->dtype() == loco::DataType::FLOAT32);
118 assert(tffbn_variance->dtype() == loco::DataType::FLOAT32);
119
120 // check all const shape are the same
121 if (!is_same_shape(tffbn_scale, tffbn_offset) || !is_same_shape(tffbn_scale, tffbn_mean) ||
122 !is_same_shape(tffbn_scale, tffbn_variance))
123 {
124 // INFO(lfbn) << "TFFBN resolve_to_muladd: Shape of constant are not same" << std::endl;
125 return false;
126 }
127
128 auto tffbn_epsilon = node->epsilon();
129 // INFO(lfbn) << "TFFBN tffbn_epsilon = " << tffbn_epsilon << std::endl;
130 auto const_num_elements = tffbn_scale->size<loco::DataType::FLOAT32>();
131 // INFO(lfbn) << "TFFBN const_num_elements = " << const_num_elements << std::endl;
132
133 // fbn_epsilon = %4:variance + fbn_epsilon_array
134 std::unique_ptr<float[]> fbn_epsilon{new float[const_num_elements]};
135 for (int32_t i = 0; i < const_num_elements; i++)
136 {
137 auto variance = tffbn_variance->at<loco::DataType::FLOAT32>(i);
138 fbn_epsilon.get()[i] = variance + tffbn_epsilon;
139 }
140
141 // fbn_rsqrt = 1.0 / math::sqrt(fbn_epsilon)
142 std::unique_ptr<float[]> fbn_rsqrt{new float[const_num_elements]};
143 for (int32_t i = 0; i < const_num_elements; i++)
144 {
145 fbn_rsqrt.get()[i] = 1.0 / sqrt(fbn_epsilon.get()[i]);
146 }
147
148 // fbn_mean = %3:mean : TODO remove this block and use %3:mean
149 std::unique_ptr<float[]> fbn_mean{new float[const_num_elements]};
150 for (int32_t i = 0; i < const_num_elements; i++)
151 {
152 fbn_mean.get()[i] = tffbn_mean->at<loco::DataType::FLOAT32>(i);
153 }
154
155 // fbn_mul = fbn_rsqrt * %1:gamma
156 std::unique_ptr<float[]> fbn_mul{new float[const_num_elements]};
157 for (int32_t i = 0; i < const_num_elements; i++)
158 {
159 fbn_mul.get()[i] = fbn_rsqrt.get()[i] * tffbn_scale->at<loco::DataType::FLOAT32>(i);
160 }
161
162 // fbn_offset = %2:beta : TODO remove this block and use %2:beta
163 std::unique_ptr<float[]> fbn_offset{new float[const_num_elements]};
164 for (int32_t i = 0; i < const_num_elements; i++)
165 {
166 fbn_offset.get()[i] = tffbn_offset->at<loco::DataType::FLOAT32>(i);
167 }
168
169 // fbn_mul_0_param = fbn_mul : remove this and use fbn_mul
170 std::unique_ptr<float[]> fbn_mul_0_param{new float[const_num_elements]};
171 for (int32_t i = 0; i < const_num_elements; i++)
172 {
173 fbn_mul_0_param.get()[i] = fbn_mul.get()[i];
174 }
175
176 // fbn_add_param = fbn_offset - fbn_mean * fbn_mul
177 std::unique_ptr<float[]> fbn_add_param{new float[const_num_elements]};
178 for (int32_t i = 0; i < const_num_elements; i++)
179 {
180 fbn_add_param.get()[i] = fbn_offset.get()[i] - fbn_mean.get()[i] * fbn_mul.get()[i];
181 }
182
183 // INFO(lfbn) << "TFFBN create ConstGen" << std::endl;
184
185 /*
186 * %11:fbn_mul_0_param = ConstGen(fbn_mul_0_param)
187 * %21:fbn_add_param = ConstGen(fbn_add_param)
188 */
189 auto const_fbn_mul_0_param = graph->nodes()->create<moco::TFConst>();
190 const_fbn_mul_0_param->dtype(loco::DataType::FLOAT32);
191 copy_shape(tffbn_scale, const_fbn_mul_0_param);
192 const_fbn_mul_0_param->size<loco::DataType::FLOAT32>(const_num_elements);
193 for (int32_t i = 0; i < const_num_elements; i++)
194 {
195 const_fbn_mul_0_param->at<loco::DataType::FLOAT32>(i) = fbn_mul_0_param.get()[i];
196 }
197 auto const_fbn_add_param = graph->nodes()->create<moco::TFConst>();
198 const_fbn_add_param->dtype(loco::DataType::FLOAT32);
199 copy_shape(tffbn_scale, const_fbn_add_param);
200 const_fbn_add_param->size<loco::DataType::FLOAT32>(const_num_elements);
201 for (int32_t i = 0; i < const_num_elements; i++)
202 {
203 const_fbn_add_param->at<loco::DataType::FLOAT32>(i) = fbn_add_param.get()[i];
204 }
205
206 // INFO(lfbn) << "TFFBN create TFMul, TFAdd" << std::endl;
207 /*
208 * %12:fbn_mul_0 = TFMul(%0:input, %11:fbn_mul_0_param)
209 * %22:fbn = TFAdd(%12:fbn_mul_0,%21:fbn_add_param)
210 */
211 auto fbn_mul_0 = graph->nodes()->create<moco::TFMul>();
212 fbn_mul_0->x(tffbn_x);
213 fbn_mul_0->y(const_fbn_mul_0_param);
214
215 auto fbn = graph->nodes()->create<moco::TFAdd>();
216 fbn->x(fbn_mul_0);
217 fbn->y(const_fbn_add_param);
218
219 // replace old node with new fbn
220 replace(node).with(fbn);
221 // unlink from graph
222 node->x(nullptr);
223 node->scale(nullptr);
224 node->offset(nullptr);
225 node->mean(nullptr);
226 node->variance(nullptr);
227
228 return true;
229}

236bool ResolveFusedBatchNorm::run(loco::Graph *graph)
237{
238 for (auto node : loco::active_nodes(loco::output_nodes(graph)))
239 {
240 if (as<moco::TFFusedBatchNorm>(node))
241 {
242 if (resolve_to_muladd(graph, as<moco::TFFusedBatchNorm>(node)))
243 {
244 // tree has been changed. let's return so that we don't need to
245 // considier about following node is correct or not.
246 return true;
247 }
248 }
249 }
250
251 return false;
252}
