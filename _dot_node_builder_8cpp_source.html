25template <typename T> static std::string toString(const std::vector<T> &v)
26{
27 std::stringstream ss;
28 ss << "[";
29 for (std::size_t i = 0; i < v.size(); ++i)
30 {
31 if (i != 0)
32 ss << ", ";
33 ss << v[i];
34 }
35 return ss.str();
36}

38DotNodeBuilder::DotNodeBuilder(const Operation &op)
39{
40 _type_name = getTypeName(op.getType());
41 _id = op.getId();
42
43 for (std::size_t i = 0; i < op.getNumInputs(); ++i)
44 {
45 _in_shapes.push_back(toString(op.getInputShape(i)));
46 }
47
48 for (std::size_t i = 0; i < op.getNumOutputs(); ++i)
49 {
50 _out_shapes.push_back(toString(op.getOutputShape(i)));
51 }
52
53 // Get attributes.
54 const_cast<Operation &>(op).accept(this);
55}

169std::string DotNodeBuilder::getLabel() const
170{
171 std::stringstream ss;
172
173 ss << "{" << _type_name << " | {{";
174
175 for (std::size_t i = 0; i < _in_shapes.size(); ++i)
176 {
177 if (i != 0)
178 ss << " | ";
179 ss << "in" << i << ": " << _in_shapes[i];
180 }
181
182 ss << " | ";
183
184 for (std::size_t i = 0; i < _out_shapes.size(); ++i)
185 {
186 if (i != 0)
187 ss << " | ";
188 ss << "out" << i << ": " << _out_shapes[i];
189 }
190
191 ss << "} | {";
192
193 for (std::size_t i = 0; i < _attributes.size(); ++i)
194 {
195 if (i != 0)
196 ss << " | ";
197 ss << _attributes[i].first << ": " << _attributes[i].second;
198 }
199
200 ss << "}}}";
201
202 return ss.str();
203}
