30bool is_foldable_const(luci::CircleConst *node)
31{
32 if (node->sparsityparam() == nullptr)
33 return false;
34
35 if (node->dtype() == loco::DataType::FLOAT32)
36 return true;
37 if (node->dtype() == loco::DataType::FLOAT16)
38 return true;
39
40 return false;
41}

43luci::CircleConst *densified_const_node(luci::CircleConst *const_node)
44{
45 assert(const_node->sparsityparam());
46
47 auto name = const_node->name();
48 assert(name.length() > 0);
49 auto g = const_node->graph();
50 auto new_const_node = g->nodes()->create<luci::CircleConst>();
51
52 new_const_node->dtype(const_node->dtype());
53 new_const_node->rank(const_node->rank());
54
55 uint32_t dim_size = 1;
56 std::vector<int> dense_shape;
57 for (uint32_t i = 0; i < new_const_node->rank(); ++i)
58 {
59 assert(const_node->dim(i).known());
60 new_const_node->dim(i) = const_node->dim(i);
61
62 uint32_t value = const_node->dim(i).value();
63 dim_size *= value;
64 dense_shape.emplace_back(static_cast<int32_t>(value));
65 }
66
67 if (const_node->dtype() == loco::DataType::FLOAT32)
68 new_const_node->size<loco::DataType::FLOAT32>(dim_size);
69 else
70 {
71 assert(const_node->dtype() == loco::DataType::FLOAT16);
72 new_const_node->size<loco::DataType::FLOAT16>(dim_size);
73 }
74
75 new_const_node->shape_status(luci::ShapeStatus::VALID);
76 new_const_node->name(name + "_DS");
77
78 if (const_node->dtype() == loco::DataType::FLOAT32)
79 {
80 auto const_items = const_node->size<loco::DataType::FLOAT32>();
81 auto f_data = std::make_unique<float[]>(const_items);
82 for (size_t i = 0; i < const_items; ++i)
83 f_data[i] = const_node->at<loco::DataType::FLOAT32>(i);
84
85 sparsity::TfLiteSparsity sp = to_tflite_sparsity(const_node->sparsityparam());
86 sparsity::FormatConverter<float> converter(dense_shape, sp);
87 converter.SparseToDense(f_data.get());
88 const auto &data_dense = converter.GetData();
89 assert(data_dense.size() == dim_size);
90
91 for (uint32_t i = 0; i < dim_size; ++i)
92 new_const_node->at<loco::DataType::FLOAT32>(i) = data_dense[i];
93
94 luci::freeTfLiteSparsity(sp);
95 }
96 else
97 {
98 assert(const_node->dtype() == loco::DataType::FLOAT16);
99
100 auto const_items = const_node->size<loco::DataType::FLOAT16>();
101 auto f_data = std::make_unique<uint16_t[]>(const_items);
102 for (size_t i = 0; i < const_items; ++i)
103 f_data[i] = const_node->at<loco::DataType::FLOAT16>(i);
104
105 // Primitive type for FLOAT16 is UINT16
106 sparsity::TfLiteSparsity sp = to_tflite_sparsity(const_node->sparsityparam());
107 sparsity::FormatConverter<uint16_t> converter(dense_shape, sp);
108 converter.SparseToDense(f_data.get());
109 const auto &data_dense = converter.GetData();
110 assert(data_dense.size() == dim_size);
111 for (uint32_t i = 0; i < dim_size; ++i)
112 new_const_node->at<loco::DataType::FLOAT16>(i) = data_dense[i];
113
114 luci::freeTfLiteSparsity(sp);
115 }
116
117 return new_const_node;
118}

123bool fold_densify(luci::CircleDensify *densify)
124{
125 auto const_input = dynamic_cast<luci::CircleConst *>(densify->input());
126 if (not const_input)
127 return false;
128
129 if (not is_foldable_const(const_input))
130 return false;
131
132 auto dense_const = densified_const_node(const_input);
133 assert(dense_const);
134
135 loco::replace(densify).with(dense_const);
136 luci::add_origin(dense_const, luci::composite_origin(
137 {luci::get_origin(densify), luci::get_origin(const_input)}));
138
139 return true;
140}

164bool FoldDensifyPass::run(loco::Graph *g)
165{
166 bool changed = false;
167
168 for (auto node : loco::active_nodes(loco::output_nodes(g)))
169 {
170 if (auto densify = dynamic_cast<luci::CircleDensify *>(node))
171 {
172 if (fold_densify(densify))
173 changed = true;
174 }
175 }
176
177 return changed;
178}
