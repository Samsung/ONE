87// This method parses the input 'axis' to remove duplicates and handle negative
88// values, and returns a valid 'out_axis'
89inline bool resolveAxis(const int num_dims, const int *axis, const int64_t num_axis, int *out_axis,
90 int *out_num_axis)
91{
92 *out_num_axis = 0; // Just in case.
93 // Short-circuit axis resolution for scalars; the axis will go unused.
94 if (num_dims == 0)
95 {
96 return true;
97 }
98 // o(n^2) is fine since out_num_axis should be really small, mostly <= 4
99 for (int64_t idx = 0; idx < num_axis; ++idx)
100 {
101 // Handle negative index. A positive index 'p_idx' can be represented as a
102 // negative index 'n_idx' as: n_idx = p_idx-num_dims
103 // eg: For num_dims=3, [0, 1, 2] is the same as [-3, -2, -1] */
104 int current = axis[idx] < 0 ? (axis[idx] + num_dims) : axis[idx];
105 if (current < 0 || current >= num_dims)
106 {
107 return false;
108 }
109 bool is_dup = false;
110 for (int j = 0; j < *out_num_axis; ++j)
111 {
112 if (out_axis[j] == current)
113 {
114 is_dup = true;
115 break;
116 }
117 }
118 if (!is_dup)
119 {
120 if (*out_num_axis > 1)
121 {
122 return false;
123 }
124 out_axis[*out_num_axis] = current;
125 *out_num_axis += 1;
126 }
127 }
128 return true;
129}

133// Old version (used in Sum and ReduceProd), to be replaced with the new one (see below).
134// Computes the generic value (i.e., sum/max/min/prod) of elements across
135// dimensions given in axis. It needs to pass in init_value and reducer.
136template <typename T>
137inline bool ReduceGeneric(const T *input_data, const int *input_dims, const int input_num_dims,
138 T *output_data, const int *axis, const int64_t num_axis_dimensions,
139 T init_value, const int output_flat_size, T reducer(const T, const T))
140{
141 // Return early when input shape has zero dim.
142 for (int i = 0; i < input_num_dims; ++i)
143 {
144 if (input_dims[i] == 0)
145 return false;
146 }
147
148 for (size_t idx = 0; idx < output_flat_size; ++idx)
149 {
150 output_data[idx] = init_value;
151 }
152
153 // Resolve axis.
154 int num_resolved_axis = 0;
155 int resolved_axis[2];
156
157 if (!resolveAxis(input_num_dims, axis, num_axis_dimensions, resolved_axis, &num_resolved_axis))
158 {
159 return false;
160 }
161
162 int temp_index[5];
163 // Reset input iterator.
164 for (int idx = 0; idx < input_num_dims; ++idx)
165 {
166 temp_index[idx] = 0;
167 }
168 // Iterate through input_data.
169 do
170 {
171 size_t input_offset = reducedOutputOffset(input_num_dims, input_dims, temp_index, 0, nullptr);
172 size_t output_offset =
173 reducedOutputOffset(input_num_dims, input_dims, temp_index, num_resolved_axis, axis);
174 output_data[output_offset] = reducer(output_data[output_offset], input_data[input_offset]);
175 } while (nextIndex(input_num_dims, input_dims, temp_index));
176
177 return true;
178}

181template <typename T>
182inline bool reduceSumImpl(const T *input_data, const int *input_dims, const int input_num_dims,
183 T *output_data, const int *axis, const int num_axis,
184 const int num_outputs)
185{
186 return ReduceGeneric<T>(input_data, input_dims, input_num_dims, output_data, axis, num_axis,
187 static_cast<T>(0), num_outputs,
188 [](const T current, const T in) -> T { return in + current; });
189}

193// New version (used in Mean).
194template <typename T, template <typename> class ReduceFn>
195bool ReduceGeneric(OMReduceDataContext<T> &ctx)
196{
197 auto &input = ctx.Input();
198 auto &input_data = input.Data();
199 auto input_dims = input.Dims();
200 auto input_num_dims = input.DimsCount();
201
202 auto &output = ctx.Output();
203 auto &output_data = output.Data();
204 auto output_flat_size = output.ShapeFlatSize();
205
206 auto &axis_ctx = ctx.Axis();
207 auto &axis = axis_ctx.Data();
208 auto num_axis_dimensions = axis_ctx.DimsCount();
209
210 // Return early when input shape has zero dim.
211 for (size_t i = 0; i < input_num_dims; ++i)
212 {
213 if (input_dims[i] == 0)
214 return false;
215 }
216
217 for (size_t idx = 0; idx < output_flat_size; ++idx)
218 {
219 output_data.SetValueAt(idx, ReduceFn<T>::InitValue);
220 }
221
222 // Resolve axis.
223 int num_resolved_axis = 0;
224 int resolved_axis[2];
225
226 if (!resolveAxis(input_num_dims, axis.Get(), num_axis_dimensions, resolved_axis,
227 &num_resolved_axis))
228 {
229 return false;
230 }
231
232 int temp_index[5];
233 // Reset input iterator.
234 for (size_t idx = 0; idx < input_num_dims; ++idx)
235 {
236 temp_index[idx] = 0;
237 }
238
239 // Iterate through input_data.
240 do
241 {
242 size_t input_offset = reducedOutputOffset(input_num_dims, input_dims, temp_index, 0, nullptr);
243 size_t output_offset =
244 reducedOutputOffset(input_num_dims, input_dims, temp_index, num_resolved_axis, axis.Get());
245
246 ReduceFn<T> reducer;
247 auto value = reducer(output_data.ValueAt(output_offset), input_data.ValueAt(input_offset));
248 output_data.SetValueAt(output_offset, value);
249
250 } while (nextIndex(input_num_dims, input_dims, temp_index));
251
252 return true;
253}

257template <typename T, template <typename> class ReduceFn>
258bool Reduce(OMReduceDataContext<T> &ctx, bool mean = false)
259{
260 if (!ReduceGeneric<T, ReduceFn>(ctx))
261 {
262 return false;
263 }
264
265 auto &input = ctx.Input();
266 auto input_dims = input.Dims();
267 auto input_num_dims = input.DimsCount();
268
269 auto &output = ctx.Output();
270 auto &output_data = output.Data();
271 auto num_outputs = output.ShapeFlatSize();
272
273 auto &axis = ctx.Axis().Data();
274 auto num_axis_dimensions = ctx.Axis().DimsCount();
275
276 // Resolve axis again for computing mean
277 int num_resolved_axis = 0;
278 int resolved_axis[2];
279
280 if (!resolveAxis(input_num_dims, axis.Get(), num_axis_dimensions, resolved_axis,
281 &num_resolved_axis))
282 {
283 return false;
284 }
285
286 // clang-format off
287
288 auto fnReduceOutput = [&](size_t divide_by = 1)
289 {
290 for (size_t idx = 0; idx < num_outputs; ++idx)
291 {
292 auto value = output_data.ValueAt(idx);
293 value /= static_cast<T>(divide_by);
294 output_data.SetAt(idx, value);
295 }
296 };
297
298 // clang-format on
299
300 if (!mean)
301 {
302 fnReduceOutput();
303 return true;
304 }
305
306 // Calculate mean by dividing output_data by num of aggregated element.
307 size_t num_elements_in_axis = 1;
308 for (int idx = 0; idx < num_resolved_axis; ++idx)
309 {
310 size_t current = static_cast<size_t>(input_dims[resolved_axis[idx]]);
311 // Overflow prevention.
312 if (current > (std::numeric_limits<size_t>::max() / num_elements_in_axis))
313 {
314 return false;
315 }
316 num_elements_in_axis *= current;
317 }
318
319 if (num_elements_in_axis > 0)
320 {
321 fnReduceOutput(num_elements_in_axis);
322 }
323
324 return true;
325}
