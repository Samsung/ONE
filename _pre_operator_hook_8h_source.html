36// Invoke a user-written Python hook before an operator is executed
37class PreOperatorHook final : public luci::CircleNodeVisitor<void>
38{
39
40// This macro creates two variables used for pre-operator hooks.
41// 1. hook: Python function to be invoked (type: py::object)
42// 2. inputs: input data (type: std::vector of numpy array)
43#define PRE_OPERATOR_HOOK_PROLOGUE(OP_NAME) \
44 if (!py::hasattr(_analysis, #OP_NAME "Pre")) \
45 { \
46 visit(loco::must_cast<const luci::CircleNode *>(node)); \
47 return; \
48 } \
49 py::object hook = _analysis.attr(#OP_NAME "Pre"); \
50 auto inputs = inputsPyArray(node, _interpreter);
51
52private:
53 py::object _analysis;
54 luci_interpreter::Interpreter *_interpreter{nullptr};
55
56public:
57 explicit PreOperatorHook(py::object analysis, luci_interpreter::Interpreter *interpreter)
58 : _analysis(analysis), _interpreter(interpreter)
59 {
60 // Do nothing
61 }
62
63 // default
64 void visit(const luci::CircleNode *node)
65 {
66 if (not py::hasattr(_analysis, "DefaultOpPre"))
67 return;
68
69 py::object hook = _analysis.attr("DefaultOpPre");
70 auto inputs = inputsPyArray(node, _interpreter);
71
72 py::list input_list;
73 for (uint32_t i = 0; i < inputs.size(); i++)
74 {
75 input_list.append(inputs[i]);
76 }
77
78 pySafeCall(hook,
79 node->name(), // name
80 toString(node->opcode()), // opcode
81 input_list // list of inputs
82 );
83 }
84
85 void visit(const luci::CircleConv2D *node)
86 {
87 PRE_OPERATOR_HOOK_PROLOGUE(Conv2D)
88
89 auto padding = node->padding();
90 auto stride = node->stride();
91 auto dilation = node->dilation();
92
93 auto py_stride = py::dict("w"_a = stride->w(), "h"_a = stride->h());
94 auto py_dilation = py::dict("w"_a = dilation->w(), "h"_a = dilation->h());
95
96 auto fused_act = node->fusedActivationFunction();
97
98 pySafeCall(hook,
99 node->name(), // name
100 inputs[0], // input
101 inputs[1], // filter
102 inputs[2], // bias
103 padding == luci::Padding::SAME ? "SAME" : "VALID", // padding
104 py_stride, // stride
105 py_dilation, // dilation
106 toString(fused_act) // fused activation
107 );
108 }

215 );
