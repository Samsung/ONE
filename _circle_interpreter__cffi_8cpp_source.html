35template <typename NodeT> size_t getTensorSize(const NodeT *node)
36{
37 uint32_t tensor_size = luci::size(node->dtype());
38 for (uint32_t i = 0; i < node->rank(); ++i)
39 tensor_size *= node->dim(i).value();
40 return tensor_size;
41}

67template <typename Func, typename... Args>
68auto exception_wrapper(Func func, Args... args) -> typename std::result_of<Func(Args...)>::type
69{
70 using ReturnType = typename std::result_of<Func(Args...)>::type;
71
72 try
73 {
74 return func(std::forward<Args>(args)...);
75 }
76 catch (const std::exception &e)
77 {
78 last_error_message = e.what();
79 if constexpr (not std::is_void<ReturnType>::value)
80 {
81 return ReturnType{};
82 }
83 }
84 catch (...)
85 {
86 last_error_message = "Unknown error";
87 if constexpr (not std::is_void<ReturnType>::value)
88 {
89 return ReturnType{};
90 }
91 }
92}

96/*
97 * Q) Why do we need this wrapper class?
98 *
99 * A) This class is designed to address specific constraints introduced by
100 * the use of CFFI for Python bindings. The original class lacked the
101 * ability to maintain internal state like luci::Module because it is defined
102 * out of the class. But, in Python, there's no way to keep luci::Module in memory.
103 * To overcome this limitation, a wrapper class is implemented to manage and
104 * keep states efficiently.
105 *
106 * Moreover, the original interface relied on passing C++ classes as arguments,
107 * which posed compatibility challenges when exposed to Python via CFFI. To
108 * simplify the integration process, the wrapper class redesigns the interface
109 * to accept more generic inputs, such as primitive types or standard containers,
110 * ensuring seamless interaction between C++ and Python.
111 *
112 * Overall, the redesigned class or the wrapper class preserves the original
113 * functionality while introducing state management and a more flexible inteface,
114 * making it highly suitable for Python-C++ interoperability through CFFI.
115 */
116class InterpreterWrapper
117{
118public:
119 explicit InterpreterWrapper(const uint8_t *data, const size_t data_size)
120 {
121 luci::Importer importer;
122 _module = importer.importModule(data, data_size);
123 if (_module == nullptr)
124 {
125 throw std::runtime_error{"Cannot import module."};
126 }
127 _intp = new luci_interpreter::Interpreter(_module.get());
128 }
129
130 ~InterpreterWrapper() { delete _intp; }
131
132 void interpret(void) { _intp->interpret(); }
133
134 void writeInputTensor(const int input_idx, const void *data, size_t input_size)
135 {
136 const auto input_nodes = loco::input_nodes(_module->graph());
137 const auto input_node = loco::must_cast<const luci::CircleInput *>(input_nodes.at(input_idx));
138 // Input size from model binary
139 const auto fb_input_size = ::getTensorSize(input_node);
140 if (fb_input_size != input_size)
141 {
142 const auto msg = "Invalid input size: " + std::to_string(fb_input_size) +
143 " != " + std::to_string(input_size);
144 throw std::runtime_error(msg);
145 }
146 _intp->writeInputTensor(input_node, data, fb_input_size);
147 }
148
149 void readOutputTensor(const int output_idx, void *output, size_t output_size)
150 {
151 const auto output_nodes = loco::output_nodes(_module->graph());
152 const auto output_node =
153 loco::must_cast<const luci::CircleOutput *>(output_nodes.at(output_idx));
154 const auto fb_output_size = ::getTensorSize(output_node);
155 if (fb_output_size != output_size)
156 {
157 const auto msg = "Invalid output size: " + std::to_string(fb_output_size) +
158 " != " + std::to_string(output_size);
159 throw std::runtime_error(msg);
160 }
161 _intp->readOutputTensor(output_node, output, fb_output_size);
162 }
163
164private:
165 luci_interpreter::Interpreter *_intp = nullptr;
166 std::unique_ptr<luci::Module> _module;
167};
