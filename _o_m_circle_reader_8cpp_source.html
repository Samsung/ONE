25circle::BuiltinOperator builtin_code_neutral(const ::circle::OperatorCode *opcode)
26{
27 assert(opcode != nullptr);

30 assert(opcode->builtin_code() >= 127);

40circle::BuiltinOperator OMCircleReader::builtin_code(const circle::Operator *op) const
41{
42 assert(op != nullptr);
43
44 const auto op_codes = opcodes();
45 uint32_t index = op->opcode_index();
46
47 assert(index < op_codes->size());
48
49 const auto opcode = op_codes->operator[](index);
50 assert(opcode != nullptr);
51
52 return builtin_code_neutral(opcode);

69OMStatus OMCircleReader::select_subgraph(uint32_t sgindex)
70{
71 if (num_subgraph() <= sgindex)
72 return UnknownError;
73
74 auto subgraphs = _model->subgraphs();
75 if (subgraphs == nullptr)
76 return UnknownError;
77
78 _current_subgraph = subgraphs->Get(sgindex);
79 if (_current_subgraph == nullptr)
80 return UnknownError;
81
82 _current_subgraph_index = sgindex;
83
84 return Ok;
85}

87bool OMCircleReader::isConstTensor(uint32_t tensor_index)
