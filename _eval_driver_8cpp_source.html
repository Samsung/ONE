29void readDataFromFile(const std::string &filename, char *data, size_t data_size)
30{
31 std::ifstream fs(filename, std::ifstream::binary);
32 if (fs.fail())
33 throw std::runtime_error("Cannot open file \"" + filename + "\".\n");
34 if (fs.read(data, data_size).fail())
35 throw std::runtime_error("Failed to read data from file \"" + filename + "\".\n");
36}

49template <typename NodeT> size_t getTensorSize(const NodeT *node)
50{
51 uint32_t tensor_size = loco::size(node->dtype());
52 for (uint32_t i = 0; i < node->rank(); ++i)
53 tensor_size *= node->dim(i).value();
54 return tensor_size;
55}

65int entry(int argc, char **argv)
66{
67 if (argc != 5)
68 {
69 std::cerr
70 << "Usage: " << argv[0]
71 << " <path/to/circle/model> <num_inputs> <path/to/input/prefix> <path/to/output/file>\n";
72 return EXIT_FAILURE;
73 }
74
75 const char *filename = argv[1];
76 const int32_t num_inputs = atoi(argv[2]);
77 const char *input_prefix = argv[3];
78 const char *output_file = argv[4];
79
80 // Load model from the file
81 luci::ImporterEx importer;
82 std::unique_ptr<luci::Module> module = importer.importVerifyModule(filename);
83 if (module == nullptr)
84 {
85 std::cerr << "ERROR: Failed to load '" << filename << "'" << std::endl;
86 return EXIT_FAILURE;
87 }
88
89 // Create interpreter.
90 luci_interpreter::Interpreter interpreter(module.get());
91
92 // Set input.
93 // Data for n'th input is read from ${input_prefix}n
94 // (ex: Add.circle.input0, Add.circle.input1 ..)
95 const auto input_nodes = loco::input_nodes(module->graph());
96 assert(num_inputs == input_nodes.size());
97 for (int32_t i = 0; i < num_inputs; i++)
98 {
99 const auto *input_node = loco::must_cast<const luci::CircleInput *>(input_nodes[i]);
100 std::vector<char> input_data(getTensorSize(input_node));
101 readDataFromFile(std::string(input_prefix) + std::to_string(i), input_data.data(),
102 input_data.size());
103 interpreter.writeInputTensor(input_node, input_data.data(), input_data.size());
104 }
105
106 // Do inference.
107 interpreter.interpret();
108
109 // Get output.
110 const auto output_nodes = loco::output_nodes(module->graph());
111 for (int i = 0; i < module->graph()->outputs()->size(); i++)
112 {
113 const auto *output_node = loco::must_cast<const luci::CircleOutput *>(output_nodes[i]);
114 std::vector<char> output_data(getTensorSize(output_node));
115 interpreter.readOutputTensor(output_node, output_data.data(), output_data.size());
116
117 // Output data is written in ${output_file}
118 // (ex: Add.circle.output0)
119 // Output shape is written in ${output_file}.shape
120 // (ex: Add.circle.output0.shape)
121 writeDataToFile(std::string(output_file) + std::to_string(i), output_data.data(),
122 output_data.size());
123 // In case of Tensor output is Scalar value.
124 // The output tensor with rank 0 is treated as a scalar with shape (1)
125 if (output_node->rank() == 0)
126 {
127 writeDataToFile(std::string(output_file) + std::to_string(i) + ".shape", "1", 1);
128 }
129 else
130 {
131 auto shape_str = std::to_string(output_node->dim(0).value());
132 for (int j = 1; j < output_node->rank(); j++)
133 {
134 shape_str += ",";
135 shape_str += std::to_string(output_node->dim(j).value());
136 }
137 writeDataToFile(std::string(output_file) + std::to_string(i) + ".shape", shape_str.c_str(),
138 shape_str.size());
139 }
140 }
141 return EXIT_SUCCESS;
142}
