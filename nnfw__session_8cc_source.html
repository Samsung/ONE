78NNFW_STATUS getTensorIndexImpl(const onert::ir::IGraph &graph, const char *tensorname,
79 uint32_t *index, bool is_input)
80{
81 if (!tensorname || !index)
82 return NNFW_STATUS_UNEXPECTED_NULL;
83
84 if (!null_terminating(tensorname, MAX_TENSOR_NAME_LENGTH))
85 {
86 std::cerr << "nnpackage path is too long" << std::endl;
87 return NNFW_STATUS_ERROR;
88 }
89
90 auto ind_found = is_input ? graph.getInputIndex(tensorname) : graph.getOutputIndex(tensorname);
91
92 if (ind_found.undefined())
93 {
94 // Not found
95 return NNFW_STATUS_ERROR;
96 }
97 else
98 {
99 *index = ind_found.value();
100 return NNFW_STATUS_NO_ERROR;
101 }
102}

104std::string trim(std::string_view value)
105{
106 constexpr std::string_view whitespace = " \t";
107
108 auto begin = value.find_first_not_of(whitespace);
109 if (begin == std::string_view::npos)
110 return ""; // no content
111
112 auto end = value.find_last_not_of(whitespace);
113 return std::string(value.substr(begin, end - begin + 1));
114}

116bool loadConfigure(const std::string cfgfile, onert::util::CfgKeyValues &keyValues)
117{
118 std::ifstream ifs(cfgfile);
119 if (ifs.is_open())
120 {
121 std::string line;
122 while (std::getline(ifs, line))
123 {
124 auto cmtpos = line.find('#');
125 if (cmtpos != std::string::npos)
126 {
127 line = line.substr(0, cmtpos);
128 }
129 std::istringstream isline(line);
130 std::string key;
131 if (std::getline(isline, key, '='))
132 {
133 std::string value;
134 if (std::getline(isline, value))
135 {
136 key = trim(key);
137 keyValues[key] = trim(value);
138 }
139 }
140 }
141 ifs.close();
142 return true;
143 }
144 return false;
145}

147NNFW_TYPE datatype_to_nnfw_dtype(onert::ir::DataType dt)
148{
149 using onert::ir::DataType;
150 switch (dt)
151 {
152 case DataType::FLOAT32:
153 return NNFW_TYPE_TENSOR_FLOAT32;
154 case DataType::INT32:
155 return NNFW_TYPE_TENSOR_INT32;
156 case DataType::QUANT_UINT8_ASYMM:
157 return NNFW_TYPE_TENSOR_QUANT8_ASYMM;
158 case DataType::BOOL8:
159 return NNFW_TYPE_TENSOR_BOOL;
160 case DataType::UINT8:
161 return NNFW_TYPE_TENSOR_UINT8;
162 case DataType::INT64:
163 return NNFW_TYPE_TENSOR_INT64;
164 case DataType::QUANT_INT8_ASYMM:
165 return NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED;
166 case DataType::QUANT_INT16_SYMM:
167 return NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED;
168 case DataType::UINT32:
169 case DataType::QUANT_INT8_SYMM:
170 default:
171 throw std::runtime_error("Error: Model has type that runtime API does not support.");
172 }
173}

218uint64_t getBufSize(const nnfw_tensorinfo *info)
219{
220 static int elmsize[] = {
221 sizeof(float), /* NNFW_TYPE_TENSOR_FLOAT32 = 0 */
222 sizeof(int), /* NNFW_TYPE_TENSOR_INT32 = 1 */
223 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM = 2 */
224 sizeof(bool), /* NNFW_TYPE_TENSOR_BOOL = 3 */
225 sizeof(uint8_t), /* NNFW_TYPE_TENSOR_UINT8 = 4 */
226 sizeof(int64_t), /* NNFW_TYPE_TENSOR_INT64 = 5 */
227 sizeof(int8_t), /* NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED = 6 */
228 sizeof(int16_t), /* NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED = 7 */
229 };
230
231 uint64_t n = 1;
232 for (int32_t i = 0; i < info->rank; ++i)
233 {
234 assert(info->dims[i] >= 0);
235 n *= info->dims[i];
236 }
237 return elmsize[info->dtype] * n;
238}

304NNFW_STATUS nnfw_session::load_model_from_path(const char *path)
305{
306 if (!isStateInitialized())
307 return NNFW_STATUS_INVALID_STATE;
308
309 if (!path)
310 {
311 std::cerr << "Path is null." << std::endl;
312 return NNFW_STATUS_UNEXPECTED_NULL;
313 }
314
315 if (!null_terminating(path, MAX_PATH_LENGTH))
316 {
317 std::cerr << "Path is too long" << std::endl;
318 return NNFW_STATUS_ERROR;
319 }
320
321 try
322 {
323 std::filesystem::path filename{path};
324 if (!std::filesystem::is_directory(filename) && filename.has_extension())
325 {
326 std::string model_type = filename.extension().string().substr(1); // + 1 to exclude dot
327 return loadModelFile(filename, model_type);
328 }
329
330 const auto &package_dir = filename;
331
332 // TODO : add support for zipped package file load
333 if (!std::filesystem::is_directory(package_dir))
334 {
335 std::cerr << "invalid path: " << package_dir << std::endl;
336 return NNFW_STATUS_ERROR;
337 }
338
339 const auto manifest_file_name = package_dir / "metadata/MANIFEST";
340 std::ifstream mfs(manifest_file_name);
341
342 // extract the filename of the first(index 0) model
343 // e.g. In MANIFEST file, { "models" : [ "firstmodel.tflite", "2nd.tflite" ] }
344 Json::Value root;
345 mfs >> root;
346 const Json::Value &models = root["models"];
347 const Json::Value &model_types = root["model-types"];
348 const Json::Value &configs = root["configs"];
349
350 if (!configs.empty() && !configs[0].empty())
351 {
352 const auto filepath = package_dir / "metadata" / configs[0].asString();
353
354 onert::util::CfgKeyValues keyValues;
355 if (loadConfigure(filepath.string(), keyValues))
356 {
357 onert::util::setConfigKeyValues(keyValues);
358 }
359 }
360 _nnpkg = std::make_shared<onert::ir::NNPkg>();
361 auto num_models = models.size();
362 if (num_models == 0 || (num_models - 1) > onert::ir::ModelIndex::max())
363 {
364 std::cerr << "Invalid model size - " << std::to_string(num_models) << std::endl;
365 return NNFW_STATUS_ERROR;
366 }
367
368 // Not support backend mapping to operator index for multiple models yet
369 // TODO Support this
370 if (num_models > 1 && _coptions->manual_scheduler_options.index_to_backend.size() != 0)
371 {
372 std::cerr << "Cannot set backend to operator index for multiple models" << std::endl;
373 return NNFW_STATUS_ERROR;
374 }
375
376 for (uint16_t i = 0; i < num_models; ++i)
377 {
378 const auto model_file_path = package_dir / models[i].asString();
379 const auto model_type = model_types[i].asString();
380 auto model = loadModel(model_file_path.string(), model_type);
381 if (model == nullptr)
382 return NNFW_STATUS_ERROR;
383 _model_path = model_file_path; // TODO Support multiple models
384 model->bindKernelBuilder(_kernel_registry->getBuilder());
385 _nnpkg->push(onert::ir::ModelIndex{i}, std::move(model));
386 }
387
388 _train_info = loadTrainingInfo(_nnpkg->primary_model());
389
390 auto toIODesc = [](std::string str) {
391 auto indices = nnfw::misc::split(str, ':');
392 if (indices.size() != 3)
393 {
394 std::cerr << "IODesc should be 3-tuple." << std::endl;
395 return onert::ir::IODesc{};
396 }
397 auto model_idx = static_cast<uint32_t>(std::stoi(indices.at(0)));
398 auto subgraph_idx = static_cast<uint32_t>(std::stoi(indices.at(1)));
399 auto operand_idx = static_cast<uint32_t>(std::stoi(indices.at(2)));
400 return onert::ir::IODesc{model_idx, subgraph_idx, operand_idx};
401 };
402 // read pkg-inputs and pkg-outputs
403 const Json::Value &pkg_inputs = root["pkg-inputs"];
404 for (uint32_t i = 0; i < pkg_inputs.size(); ++i)
405 _nnpkg->addInput(toIODesc(pkg_inputs[i].asString()));
406 const Json::Value &pkg_outputs = root["pkg-outputs"];
407 for (uint32_t i = 0; i < pkg_outputs.size(); ++i)
408 _nnpkg->addOutput(toIODesc(pkg_outputs[i].asString()));
409 // read model-connect
410 const Json::Value &fromtos = root["model-connect"];
411 for (uint32_t i = 0; i < fromtos.size(); ++i)
412 {
413 const Json::Value &tos = fromtos[i]["to"];
414 for (uint32_t j = 0; j < tos.size(); ++j)
415 _nnpkg->addEdge(toIODesc(fromtos[i]["from"].asString()), toIODesc(tos[j].asString()));
416 }
417
418 _nnpkg->verify();
419 _state = State::MODEL_LOADED;
420 }
421 catch (const std::exception &e)
422 {
423 std::cerr << "Error during model loading : " << e.what() << std::endl;
424 return NNFW_STATUS_ERROR;
425 }
426 return NNFW_STATUS_NO_ERROR;
427}

429NNFW_STATUS nnfw_session::prepare()
430{
431 // NOTE. If users want to run prepare() more than one time, this could be removed.
432 if (!isStateModelLoaded())
433 {
434 std::cerr << "Error during model prepare : ";
435 if (isStateInitialized())
436 {
437 std::cerr << "prepare should be run once";
438 }
439 else
440 {
441 std::cerr << "invalid state";
442 }
443 std::cerr << std::endl;
444 return NNFW_STATUS_INVALID_STATE;
445 }
446
447 try
448 {
449 auto compiler = onert::compiler::CompilerFactory::get().create(_nnpkg, _coptions.get());
450 _nnpkg.reset();
451 _compiler_artifact = compiler->compile();
452 _execution = std::make_unique<onert::exec::Execution>(_compiler_artifact->_executors);
453 }
454 catch (const std::exception &e)
455 {
456 std::cerr << "Error during model prepare : " << e.what() << std::endl;
457 return NNFW_STATUS_ERROR;
458 }
459
460 _state = State::PREPARED;
461 return NNFW_STATUS_NO_ERROR;
462}

1111 if (dest_size < src.length() + 1 /* for '\0' */)
1112 {
1113 std::cerr << "buffer is small to copy config value." << std::endl;
1114 return false;

1125
1126 auto str =
1127 nnfw::misc::join(_coptions->backend_list.begin(), _coptions->backend_list.end(), ";");
1128
1129 if (!check_boundary(value_size, str))
1130 return NNFW_STATUS_ERROR;
1131
1132 strncpy(value, str.c_str(), value_size);

1134 else if (skey == onert::util::config::EXECUTOR)

1137 return NNFW_STATUS_ERROR;

1139 strncpy(value, _coptions->executor.c_str(), _coptions->executor.length());

1144 }

1262NNFW_STATUS nnfw_session::train_get_traininfo(nnfw_train_info *info)
1263{
1264 if (isStateInitialized())
1265 {
1266 // There is no _train_info in INITIALIZED, since _train_info is set when a model loaded
1267 std::cerr << "Error during nnfw_session::train_get_traininfo : invalid state";
1268 return NNFW_STATUS_INVALID_STATE;
1269 }
1270
1271 if (info == nullptr)
1272 {
1273 std::cerr << "Error during nnfw_session::train_get_traininfo : info is nullptr" << std::endl;
1274 return NNFW_STATUS_UNEXPECTED_NULL;
1275 }
1276
1277 // after model loaded, it ensures that _train_info is not nullptr
1278 assert(_train_info != nullptr);
1279
1280 auto convertLossCode = [](const onert::ir::train::LossCode &code) -> NNFW_TRAIN_LOSS {
1281 switch (code)
1282 {
1283 case onert::ir::train::LossCode::Undefined:
1284 return NNFW_TRAIN_LOSS_UNDEFINED;
1285 case onert::ir::train::LossCode::MeanSquaredError:
1286 return NNFW_TRAIN_LOSS_MEAN_SQUARED_ERROR;
1287 case onert::ir::train::LossCode::CategoricalCrossentropy:
1288 return NNFW_TRAIN_LOSS_CATEGORICAL_CROSSENTROPY;
1289 default:
1290 throw std::runtime_error{"fail to convert ir::train::LossCode"};
1291 }
1292 };
1293
1294 auto convertLossReduction =
1295 [](const onert::ir::train::LossReductionType &type) -> NNFW_TRAIN_LOSS_REDUCTION {
1296 switch (type)
1297 {
1298 case onert::ir::train::LossReductionType::Undefined:
1299 return NNFW_TRAIN_LOSS_REDUCTION_UNDEFINED;
1300 case onert::ir::train::LossReductionType::SumOverBatchSize:
1301 return NNFW_TRAIN_LOSS_REDUCTION_SUM_OVER_BATCH_SIZE;
1302 case onert::ir::train::LossReductionType::Sum:
1303 return NNFW_TRAIN_LOSS_REDUCTION_SUM;
1304 default:
1305 throw std::runtime_error{"fail to convert from ir::train::LossReductionType"};
1306 break;
1307 }
1308 };
1309
1310 auto convertOptimizerCode =
1311 [](const onert::ir::train::OptimizerCode &code) -> NNFW_TRAIN_OPTIMIZER {
1312 switch (code)
1313 {
1314 case onert::ir::train::OptimizerCode::Undefined:
1315 return NNFW_TRAIN_OPTIMIZER_UNDEFINED;
1316 case onert::ir::train::OptimizerCode::SGD:
1317 return NNFW_TRAIN_OPTIMIZER_SGD;
1318 case onert::ir::train::OptimizerCode::Adam:
1319 return NNFW_TRAIN_OPTIMIZER_ADAM;
1320 default:
1321 throw std::runtime_error{"fail to convert from ir::train::OptimizerCode"};
1322 }
1323 };
1324
1325 const auto &loss = _train_info->lossInfo();
1326 const auto &optim = _train_info->optimizerInfo();
1327
1328 try
1329 {
1330 info->learning_rate = optim.learning_rate;
1331 info->batch_size = _train_info->batchSize();
1332 info->loss_info.loss = convertLossCode(loss.loss_code);
1333 info->loss_info.reduction_type = convertLossReduction(loss.reduction_type);
1334 info->opt = convertOptimizerCode(optim.optim_code);
1335
1336 if (_train_info->getTrainableOps().size() > 0)
1337 {
1338 const uint32_t first_trainable_idx = _train_info->getTrainableOps().cbegin()->value();
1339 const uint32_t last_trainable_idx = _train_info->getTrainableOps().crbegin()->value();
1340 const uint32_t ops_size = primary_subgraph()->operations().size();
1341 const uint32_t trainable_indexes_range = last_trainable_idx - first_trainable_idx + 1;
1342
1343 // check if trainable ops set contains continuous indexes on the back of the set
1344 if (last_trainable_idx == ops_size - 1 &&
1345 trainable_indexes_range == _train_info->getTrainableOps().size())
1346 {
1347 // check if all ops are trainable
1348 if (0 == first_trainable_idx)
1349 {
1350 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_ALL;
1351 }
1352 else
1353 {
1354 info->num_of_trainable_ops = trainable_indexes_range;
1355 }
1356 }
1357 else
1358 {
1359 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_INCORRECT_STATE;
1360 std::cerr << "conversion from set of trainable ops to num_of_trainable_ops is impossible"
1361 << std::endl;
1362 return NNFW_STATUS_INVALID_STATE;
1363 }
1364 }
1365 else
1366 {
1367 // no layer will be trained
1368 info->num_of_trainable_ops = NNFW_TRAIN_TRAINABLE_NONE;
1369 }
1370 }
1371 catch (const std::exception &e)
1372 {
1373 std::cerr << "Error during nnfw_session::train_get_traininfo" << e.what() << std::endl;
1374 return NNFW_STATUS_ERROR;
1375 }
1376
1377 return NNFW_STATUS_NO_ERROR;
1378}
1379
1380NNFW_STATUS nnfw_session::train_set_traininfo(const nnfw_train_info *info)
1381{
1382 if (not isStateModelLoaded())
1383 {
1384 std::cerr << "Error during nnfw_session::train_set_traininfo : invalid state" << std::endl;
1385 return NNFW_STATUS_INVALID_STATE;
1386 }
1387
1388 if (info == nullptr)
1389 {
1390 std::cerr << "nnfw_session::train_set_traininfo : info is nullptr" << std::endl;
1391 return NNFW_STATUS_UNEXPECTED_NULL;
1392 }
1393
1394 // after model loaded, it ensures that _train_info is not nullptr
1395 assert(_train_info != nullptr);
1396
1397 auto convertLossType = [](const int &type) {
1398 if (type == NNFW_TRAIN_LOSS_MEAN_SQUARED_ERROR)
1399 return onert::ir::train::LossCode::MeanSquaredError;
1400 else if (type == NNFW_TRAIN_LOSS_CATEGORICAL_CROSSENTROPY)
1401 return onert::ir::train::LossCode::CategoricalCrossentropy;
1402 else
1403 throw std::runtime_error("not supported loss type");
1404 };
1405
1406 auto convertLossReductionType = [](const int &type) {
1407 if (type == NNFW_TRAIN_LOSS_REDUCTION_SUM_OVER_BATCH_SIZE)
1408 return onert::ir::train::LossReductionType::SumOverBatchSize;
1409 else if (type == NNFW_TRAIN_LOSS_REDUCTION_SUM)
1410 return onert::ir::train::LossReductionType::Sum;
1411 else
1412 throw std::runtime_error("not supported loss reduction type");
1413 };
1414
1415 auto convertOptType = [](const int &type) {
1416 if (type == NNFW_TRAIN_OPTIMIZER_SGD)
1417 return onert::ir::train::OptimizerCode::SGD;
1418 else if (type == NNFW_TRAIN_OPTIMIZER_ADAM)
1419 return onert::ir::train::OptimizerCode::Adam;
1420 else
1421 throw std::runtime_error("not supported optimizer type");
1422 };
1423
1424 try
1425 {
1426 onert::ir::train::LossInfo loss_info;
1427 loss_info.loss_code = convertLossType(info->loss_info.loss);
1428 loss_info.reduction_type = convertLossReductionType(info->loss_info.reduction_type);
1429
1430 onert::ir::train::OptimizerInfo opt_info;
1431 opt_info.learning_rate = info->learning_rate;
1432 opt_info.optim_code = convertOptType(info->opt);
1433
1434 _train_info->setBatchSize(info->batch_size);
1435 _train_info->setLossInfo(loss_info);
1436 _train_info->setOptimizerInfo(opt_info);
1437
1438 if (info->num_of_trainable_ops < -1)
1439 {
1440 std::cerr << "Error during nnfw_session::train_set_traininfo: provided num_of_trainable_ops "
1441 "has incorrect value: "
1442 << info->num_of_trainable_ops << std::endl;
1443 return NNFW_STATUS_ERROR;
1444 }
1445
1446 const uint32_t ops_size = primary_subgraph()->operations().size();
1447 std::set<onert::ir::OperationIndex> trainable_ops;
1448
1449 if (NNFW_TRAIN_TRAINABLE_ALL == info->num_of_trainable_ops)
1450 {
1451 for (uint32_t idx = 0; idx < ops_size; ++idx)
1452 {
1453 trainable_ops.emplace(idx);
1454 }
1455 }
1456 else
1457 {
1458 if (static_cast<uint32_t>(info->num_of_trainable_ops) > ops_size)
1459 {
1460 std::cerr
1461 << "Error during nnfw_session::train_set_traininfo: provided num_of_trainable_ops="
1462 << info->num_of_trainable_ops << " is out of operators range equals: " << ops_size
1463 << std::endl;
1464 return NNFW_STATUS_ERROR;
1465 }
1466 for (uint32_t i = 1; i <= static_cast<uint32_t>(info->num_of_trainable_ops); ++i)
1467 {
1468 trainable_ops.emplace(ops_size - i);
1469 }
1470 }
1471 // Note that possible setting an empty trainable_ops set (for NNFW_TRAIN_TRAINABLE_NONE value)
1472 _train_info->setTrainableOps(trainable_ops);
1473 }
1474 catch (const std::exception &e)
1475 {
1476 std::cerr << "Error during nnfw_session::train_set_traininfo : " << e.what() << std::endl;
1477 return NNFW_STATUS_ERROR;
1478 }
1479
1480 return NNFW_STATUS_NO_ERROR;
1481}
