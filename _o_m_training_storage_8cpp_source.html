25OMStatus OMTrainingStorage::setOptimizer(const OMConfig &config)
26{
27 switch (config.training_context.optimizer)
28 {
29 case SGD:
30 {
31 if (_sgd_optimizer != nullptr)
32 {
33 return UnknownError;
34 }
35 _sgd_optimizer = std::make_unique<optimizers::SGD>();
36 break;
37 }
38 case ADAM:
39 {
40 if (_adam_optimizer != nullptr)
41 {
42 return UnknownError;
43 }
44 _adam_optimizer = std::make_unique<optimizers::Adam>();
45 break;
46 }
47 default:
48 assert(false && "Unsupported Optimizer type");
49 }
50 return Ok;
51}
