41bool fuse_add_with_fc(luci::CircleFullyConnected *fc)
42{
43 if (not fc)
44 return false;
45
46 if (fc->dtype() != loco::DataType::FLOAT32)
47 return false;
48
49 if (fc->fusedActivationFunction() != luci::FusedActFunc::NONE)
50 return false;
51
52 auto weights = dynamic_cast<luci::CircleConst *>(fc->weights());
53 if (not weights)
54 return false;
55
56 // Get add node
57 auto fc_output = loco::succs(fc);
58 if (fc_output.size() != 1)
59 return false;
60
61 auto add = dynamic_cast<luci::CircleAdd *>(*fc_output.begin());
62 if (not add)
63 return false;
64 if (add->dtype() != loco::DataType::FLOAT32)
65 return false;
66
67 // Get addition
68 auto addition = add->x() == fc ? dynamic_cast<luci::CircleConst *>(add->y())
69 : dynamic_cast<luci::CircleConst *>(add->x());
70
71 // Non-const addition
72 if (not addition)
73 return false;
74
75 auto rank = addition->rank();
76 // TODO Support scalar addition
77 if (rank == 0)
78 return false;
79
80 for (uint32_t i = 0; i < rank - 1; i++)
81 {
82 if (addition->dim(i).value() != 1)
83 return false;
84 }
85 // Check the last dimesion of addition is the same with the number of neurons of FC
86 if (not(addition->dim(rank - 1) == weights->dim(0)))
87 return false;
88
89 auto bias = loco::must_cast<luci::CircleNode *>(fc->bias());
90
91 // We only support (1) constant bias (2) no bias
92 // If bias is neither (1) nor (2), it would be a feature map
93 if (bias->opcode() != luci::CircleOpcode::CIRCLECONST and
94 bias->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
95 return false;
96
97 auto fused_bias = luci::clone(addition);
98
99 // Add existing bias values
100 if (auto const_bias = dynamic_cast<luci::CircleConst *>(fc->bias()))
101 {
102 assert(const_bias->dtype() == loco::DataType::FLOAT32);
103
104 auto bias_size = fused_bias->size<loco::DataType::FLOAT32>();
105 assert(bias_size == const_bias->size<loco::DataType::FLOAT32>());
106 for (uint32_t i = 0; i < bias_size; i++)
107 fused_bias->at<loco::DataType::FLOAT32>(i) += const_bias->at<loco::DataType::FLOAT32>(i);
108 }
109
110 // At this point, it is guarateed that fused_bias's shape is [1, 1, ..., N] or [N]
111 // where N is weights->dim(0).
112 // The shape is normalized to [N] to become the bias of FC
113 fused_bias->rank(1);
114 fused_bias->dim(0) = weights->dim(0);
115
116 fc->bias(fused_bias);
117 fc->fusedActivationFunction(add->fusedActivationFunction());
118
119 // set origin
120 luci::add_origin(fc, luci::get_origin(add));
121
122 replace(add).with(fc);
123
124 return true;
125}

132bool FuseAddWithFullyConnectedPass::run(loco::Graph *g)
133{
134 bool changed = false;
135 for (auto node : loco::active_nodes(loco::output_nodes(g)))
136 {
137 auto fc = dynamic_cast<luci::CircleFullyConnected *>(node);
138 if (not fc)
139 continue;
140
141 if (fuse_add_with_fc(fc))
142 changed = true;
143 }
144
145 return changed;
146}
