21void ensure_status(NNFW_STATUS status)
22{
23 switch (status)
24 {
25 case NNFW_STATUS::NNFW_STATUS_NO_ERROR:
26 break;
27 case NNFW_STATUS::NNFW_STATUS_ERROR:
28 std::cout << "[ERROR]\tNNFW_STATUS_ERROR\n";
29 exit(1);
30 case NNFW_STATUS::NNFW_STATUS_UNEXPECTED_NULL:
31 std::cout << "[ERROR]\tNNFW_STATUS_UNEXPECTED_NULL\n";
32 exit(1);
33 case NNFW_STATUS::NNFW_STATUS_INVALID_STATE:
34 std::cout << "[ERROR]\tNNFW_STATUS_INVALID_STATE\n";
35 exit(1);
36 case NNFW_STATUS::NNFW_STATUS_OUT_OF_MEMORY:
37 std::cout << "[ERROR]\tNNFW_STATUS_OUT_OF_MEMORY\n";
38 exit(1);
39 case NNFW_STATUS::NNFW_STATUS_INSUFFICIENT_OUTPUT_SIZE:
40 std::cout << "[ERROR]\tNNFW_STATUS_INSUFFICIENT_OUTPUT_SIZE\n";
41 exit(1);
42 }
43}

45NNFW_LAYOUT getLayout(const char *layout)
46{
47 if (!strcmp(layout, "NCHW"))
48 {
49 return NNFW_LAYOUT::NNFW_LAYOUT_CHANNELS_FIRST;

57 return NNFW_LAYOUT::NNFW_LAYOUT_NONE;

61 std::cout << "[ERROR]\tLAYOUT_TYPE\n";

66NNFW_TYPE getType(const char *type)
67{
68 if (!strcmp(type, "float32"))
69 {
70 return NNFW_TYPE::NNFW_TYPE_TENSOR_FLOAT32;
71 }
72 else if (!strcmp(type, "int32"))
73 {
74 return NNFW_TYPE::NNFW_TYPE_TENSOR_INT32;
75 }
76 else if (!strcmp(type, "uint8"))
77 {
78 return NNFW_TYPE::NNFW_TYPE_TENSOR_UINT8;
79 // return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT8_ASYMM;
80 }
81 else if (!strcmp(type, "bool"))
82 {
83 return NNFW_TYPE::NNFW_TYPE_TENSOR_BOOL;
84 }
85 else if (!strcmp(type, "int64"))
86 {
87 return NNFW_TYPE::NNFW_TYPE_TENSOR_INT64;
88 }
89 else if (!strcmp(type, "int8"))
90 {
91 return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED;
92 }
93 else if (!strcmp(type, "int16"))
94 {
95 return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED;
96 }
97 else
98 {
99 std::cout << "[ERROR] String to NNFW_TYPE Failure\n";
100 exit(1);
101 }
102}

129uint64_t num_elems(const nnfw_tensorinfo *tensor_info)
130{
131 uint64_t n = 1;
132 for (uint32_t i = 0; i < tensor_info->rank; ++i)
133 {
134 n *= tensor_info->dims[i];
135 }
136 return n;
137}

179void NNFW_SESSION::set_input_tensorinfo(uint32_t index, const tensorinfo *tensor_info)
180{
181 nnfw_tensorinfo ti;
182 ti.dtype = getType(tensor_info->dtype);
183 ti.rank = tensor_info->rank;
184 for (int i = 0; i < NNFW_MAX_RANK; i++)
185 {
186 ti.dims[i] = tensor_info->dims[i];
187 }
188 ensure_status(nnfw_set_input_tensorinfo(session, index, &ti));
189}
