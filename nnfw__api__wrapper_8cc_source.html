30void ensure_status(NNFW_STATUS status)
31{
32 switch (status)
33 {
34 case NNFW_STATUS::NNFW_STATUS_NO_ERROR:
35 break;
36 case NNFW_STATUS::NNFW_STATUS_ERROR:
37 std::cout << "[ERROR]\tNNFW_STATUS_ERROR\n";
38 exit(1);
39 case NNFW_STATUS::NNFW_STATUS_UNEXPECTED_NULL:
40 std::cout << "[ERROR]\tNNFW_STATUS_UNEXPECTED_NULL\n";
41 exit(1);
42 case NNFW_STATUS::NNFW_STATUS_INVALID_STATE:
43 std::cout << "[ERROR]\tNNFW_STATUS_INVALID_STATE\n";
44 exit(1);
45 case NNFW_STATUS::NNFW_STATUS_OUT_OF_MEMORY:
46 std::cout << "[ERROR]\tNNFW_STATUS_OUT_OF_MEMORY\n";
47 exit(1);
48 case NNFW_STATUS::NNFW_STATUS_INSUFFICIENT_OUTPUT_SIZE:
49 std::cout << "[ERROR]\tNNFW_STATUS_INSUFFICIENT_OUTPUT_SIZE\n";
50 exit(1);
51 case NNFW_STATUS::NNFW_STATUS_DEPRECATED_API:
52 std::cout << "[ERROR]\tNNFW_STATUS_DEPRECATED_API\n";
53 exit(1);
54 }
55}

57NNFW_LAYOUT getLayout(const char *layout)
58{
59 if (!strcmp(layout, "NCHW"))
60 {
61 return NNFW_LAYOUT::NNFW_LAYOUT_CHANNELS_FIRST;
62 }
63 else if (!strcmp(layout, "NHWC"))
64 {
65 return NNFW_LAYOUT::NNFW_LAYOUT_CHANNELS_LAST;
66 }
67 else if (!strcmp(layout, "NONE"))
68 {
69 return NNFW_LAYOUT::NNFW_LAYOUT_NONE;
70 }
71 else
72 {
73 std::cout << "[ERROR]\tLAYOUT_TYPE\n";
74 exit(1);
75 }
76}

78NNFW_TYPE getType(const char *type)
79{
80 if (!strcmp(type, "float32"))
81 {
82 return NNFW_TYPE::NNFW_TYPE_TENSOR_FLOAT32;
83 }
84 else if (!strcmp(type, "int32"))
85 {
86 return NNFW_TYPE::NNFW_TYPE_TENSOR_INT32;
87 }
88 else if (!strcmp(type, "uint8"))
89 {
90 return NNFW_TYPE::NNFW_TYPE_TENSOR_UINT8;
91 // return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT8_ASYMM;
92 }
93 else if (!strcmp(type, "bool"))
94 {
95 return NNFW_TYPE::NNFW_TYPE_TENSOR_BOOL;
96 }
97 else if (!strcmp(type, "int64"))
98 {
99 return NNFW_TYPE::NNFW_TYPE_TENSOR_INT64;
100 }
101 else if (!strcmp(type, "int8"))
102 {
103 return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT8_ASYMM_SIGNED;
104 }
105 else if (!strcmp(type, "int16"))
106 {
107 return NNFW_TYPE::NNFW_TYPE_TENSOR_QUANT16_SYMM_SIGNED;
108 }
109 else
110 {
111 std::cout << "[ERROR] String to NNFW_TYPE Failure\n";
112 exit(1);
113 }
114}

141uint64_t num_elems(const nnfw_tensorinfo *tensor_info)
142{
143 uint64_t n = 1;
144 for (int32_t i = 0; i < tensor_info->rank; ++i)
145 {
146 n *= tensor_info->dims[i];
147 }
148 return n;
149}

151py::list get_dims(const tensorinfo &tensor_info)
152{
153 py::list dims_list;
154 for (int32_t i = 0; i < tensor_info.rank; ++i)
155 {
156 dims_list.append(tensor_info.dims[i]);

161void set_dims(tensorinfo &tensor_info, const py::list &array)
162{

165 {
166 tensor_info.dims[i] = py::cast<int32_t>(array[i]);

190void NNFW_SESSION::set_input_tensorinfo(uint32_t index, const tensorinfo *tensor_info)
191{
192 nnfw_tensorinfo ti;

194 ti.rank = tensor_info->rank;

198 }
