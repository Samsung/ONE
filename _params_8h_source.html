140// For Add, Sub, Mul ops.
141struct ArithmeticParams
142{
143 // Shape dependent / common to data / op types.
144 BroadcastableOpCategory broadcast_category;
145 // uint8_t inference params.
146 int32_t input1_offset;
147 int32_t input2_offset;
148 int32_t output_offset;
149 int32_t output_multiplier;
150 int output_shift;
151 // Add / Sub, not Mul, uint8_t inference params.
152 int left_shift;
153 int32_t input1_multiplier;
154 int input1_shift;
155 int32_t input2_multiplier;
156 int input2_shift;
157 // uint8_t, etc, activation params.
158 int32_t quantized_activation_min;
159 int32_t quantized_activation_max;
160 // float activation params.
161 float float_activation_min;
162 float float_activation_max;
163 // int64_t activation params.
164 int64_t int64_activation_min;
165 int64_t int64_activation_max;
166
167 // Processed output dimensions.
168 // Let input "a" be the one that broadcasts in the faster-changing dimension.
169 // Then, after coalescing, for shapes {a0, a1, a2, a3, a4} and
170 // {b0, b1, b2, b3, b4},
171 // broadcast_shape[4] = b0 = a0.
172 // broadcast_shape[3] = b1; a1 = 1.
173 // broadcast_shape[2] = b2 = a2.
174 // broadcast_shape[1] = a3; b3 = 1.
175 // broadcast_shape[0] = b4 = a4.
176 int broadcast_shape[5];
177};
