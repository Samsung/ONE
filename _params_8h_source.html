168// For Add, Sub, Mul ops.
169struct ArithmeticParams
170{
171 // Shape dependent / common to data / op types.
172 BroadcastableOpCategory broadcast_category;
173 // uint8_t inference params.
174 int32_t input1_offset;
175 int32_t input2_offset;
176 int32_t output_offset;
177 int32_t output_multiplier;
178 int output_shift;
179 // Add / Sub, not Mul, uint8_t inference params.
180 int left_shift;
181 int32_t input1_multiplier;
182 int input1_shift;
183 int32_t input2_multiplier;
184 int input2_shift;
185 // uint8_t, etc, activation params.
186 int32_t quantized_activation_min;
187 int32_t quantized_activation_max;
188 // float activation params.
189 float float_activation_min;
190 float float_activation_max;
191 // int64_t activation params.
192 int64_t int64_activation_min;
193 int64_t int64_activation_max;
194
195 // Processed output dimensions.
196 // Let input "a" be the one that broadcasts in the faster-changing dimension.
197 // Then, after coalescing, for shapes {a0, a1, a2, a3, a4} and
198 // {b0, b1, b2, b3, b4},
199 // broadcast_shape[4] = b0 = a0.
200 // broadcast_shape[3] = b1; a1 = 1.
201 // broadcast_shape[2] = b2 = a2.
202 // broadcast_shape[1] = a3; b3 = 1.
203 // broadcast_shape[0] = b4 = a4.
204 int broadcast_shape[5];
205};
