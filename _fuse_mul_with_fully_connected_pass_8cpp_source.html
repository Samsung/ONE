32inline bool is_effectively_scalar_shape(const luci::CircleConst *node)
33{
34 if (node->size<loco::DataType::FLOAT32>() != 1)
35 return false;
36 auto const rank = node->rank();
37 for (uint32_t i = 0; i < rank; i++)
38 if (node->dim(i).value() != 1)
39 return false;
40 return true;
41}

43inline bool is_single_element(const luci::CircleConst *node)
44{
45 if (is_effectively_scalar_shape(node))
46 return true;
47 return ((node->rank() == 1 || node->rank() == 0) && node->size<loco::DataType::FLOAT32>() == 1);
48}

50inline void update_with_single_element(luci::CircleConst *fused_node,
51 const luci::CircleConst *multiplication)
52{
53 for (uint32_t i = 0; i < fused_node->size<loco::DataType::FLOAT32>(); i++)
54 {
55 fused_node->at<loco::DataType::FLOAT32>(i) *= multiplication->at<loco::DataType::FLOAT32>(0);
56 }
57}

59luci::CircleConst *gen_fused_weights(luci::CircleConst *weights,
60 const luci::CircleConst *multiplication)
61{
62 auto fused_weights = luci::clone(weights);
63 // Single element multiplication:
64 if (is_single_element(multiplication))
65 {
66 update_with_single_element(fused_weights, multiplication);
67 }
68 // N-size multiplication:
69 else
70 {
71 // Go along channels, multiplication size is ensured to be compatible with channels.
72 auto count = fused_weights->dim(0).value();
73 auto size = fused_weights->dim(fused_weights->rank() - 1).value();
74 float val;
75 for (uint32_t c = 0; c < count; c++)
76 {
77 val = multiplication->at<loco::DataType::FLOAT32>(c);
78 for (uint32_t i = 0; i < size; i++)
79 {
80 fused_weights->at<loco::DataType::FLOAT32>(c * size + i) *= val;
81 }
82 }
83 }
84 return fused_weights;
85}

87luci::CircleConst *gen_fused_bias(luci::CircleConst *bias, const luci::CircleConst *multiplication)
88{
89 auto fused_bias = luci::clone(bias);
90 // Single element multiplication:
91 if (is_single_element(multiplication))
92 {
93 update_with_single_element(fused_bias, multiplication);
94 }
95 // N-size multiplication:
96 else
97 {
98 // Go along channels, multiplication size is ensured to be compatible with channels.
99 for (uint32_t i = 0; i < fused_bias->size<loco::DataType::FLOAT32>(); i++)
100 {
101 fused_bias->at<loco::DataType::FLOAT32>(i) *= multiplication->at<loco::DataType::FLOAT32>(i);
102 }
103 }
104 return fused_bias;

123bool fuse_mul_with_fc(luci::CircleMul *mul)
124{
125 // Sanity check:
126 RETURN_FALSE_UNLESS(mul);
127 // Allow Mul node only with FLOAT32 data type:
128 RETURN_FALSE_UNLESS(mul->dtype() == loco::DataType::FLOAT32);
129 // Check if any FC node connects to Mul.
130 // Find the pattern of Mul(FC, CircleConst):
131 luci::CircleFullyConnected *fc = nullptr;
132 luci::CircleConst *multiplication = nullptr;
133 RETURN_FALSE_UNLESS(luci::fill(&fc, &multiplication).with_commutative_args_of(mul));
170 RETURN_FALSE_UNLESS(loco::succs(fc).size() == 1);
171 // Allow only FLOAT32 data type:
172 RETURN_FALSE_UNLESS(fc->dtype() == loco::DataType::FLOAT32);
173 // Allow only without activation functions as values are going to
174 // be multiplied before activation function.
175 RETURN_FALSE_UNLESS(fc->fusedActivationFunction() == luci::FusedActFunc::NONE);
176 // Check for weights being Constant:
177 auto weights = dynamic_cast<luci::CircleConst *>(fc->weights());
178 RETURN_FALSE_UNLESS(weights);
179 // Get rank of multiplication:
180 auto rank = multiplication->rank();
181 // Check that all dimensions are ones, checks broadcast capabilites.
182 // Last dimesion of multiplication must be compatible with FC.
183 // N-D case (N>1):
184 if (multiplication->rank() > 1)
185 {
186 // Check channel-wise broadcasting:
187 for (uint32_t i = 0; i < rank - 1; i++)
188 RETURN_FALSE_UNLESS(multiplication->dim(i).value() == 1);
189 // Check the last dimesion of Mul is the same with the first dimension of FullyConnected
190 RETURN_FALSE_UNLESS(multiplication->dim(rank - 1) == weights->dim(0) ||
191 is_effectively_scalar_shape(multiplication));
192 }
193 // 1-D or scalar case:
194 else if (multiplication->rank() == 1)
195 {
196 RETURN_FALSE_UNLESS(multiplication->size<loco::DataType::FLOAT32>() == 1 ||
197 multiplication->size<loco::DataType::FLOAT32>() == weights->dim(0));
198 }
199 else if (multiplication->rank() == 0)
200 {
201 RETURN_FALSE_UNLESS(multiplication->size<loco::DataType::FLOAT32>() == 1);
202 }
203
204 // Only supports:
205 // (1) constant bias
206 // (2) no bias
207 auto bias = loco::must_cast<luci::CircleNode *>(fc->bias());
208 if (bias->opcode() == luci::CircleOpcode::CIRCLECONST)
209 {
210 // Create new bias to be updated with values:
211 auto const_bias = dynamic_cast<luci::CircleConst *>(fc->bias());
212 RETURN_FALSE_UNLESS(const_bias)
213 RETURN_FALSE_UNLESS(const_bias->dtype() == loco::DataType::FLOAT32);
214 // Create new bias with updated values and replace:
215 auto fused_bias = gen_fused_bias(const_bias, multiplication);
216 fc->bias(fused_bias);
217 }
218 else if (bias->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
219 {
220 return false;
221 }
222
223 // Create new weights with updated values and replace:
224 auto fused_weights = gen_fused_weights(weights, multiplication);
225 fc->weights(fused_weights);
226
227 // Set origin and copy Activation Function if exisitng:
228 fc->fusedActivationFunction(mul->fusedActivationFunction());
229 luci::add_origin(fc, luci::get_origin(mul));
230
231 replace(mul).with(fc);
232
233 return true;
234}

241bool FuseMulWithFullyConnectedPass::run(loco::Graph *g)
242{
243 bool changed = false;
244 for (auto node : loco::active_nodes(loco::output_nodes(g)))
245 {
246 if (auto mul = dynamic_cast<luci::CircleMul *>(node))
247 {
248 if (fuse_mul_with_fc(mul))
249 changed = true;
250 }
251 }
252
253 return changed;
254}
