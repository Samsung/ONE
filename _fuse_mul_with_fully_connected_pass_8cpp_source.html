32inline bool is_single_element(const luci::CircleConst *node)
33{
34 return ((node->rank() == 1 || node->rank() == 0) && node->size<loco::DataType::FLOAT32>() == 1);
35}

37inline void update_with_single_element(luci::CircleConst *fused_node,
38 const luci::CircleConst *multiplication)
39{
40 for (uint32_t i = 0; i < fused_node->size<loco::DataType::FLOAT32>(); i++)
41 {
42 fused_node->at<loco::DataType::FLOAT32>(i) *= multiplication->at<loco::DataType::FLOAT32>(0);
43 }
44}

46luci::CircleConst *gen_fused_weights(luci::CircleConst *weights,
47 const luci::CircleConst *multiplication)
48{
49 auto fused_weights = luci::clone(weights);
50 // Single element multiplication:
51 if (is_single_element(multiplication))
52 {
53 update_with_single_element(fused_weights, multiplication);
54 }
55 // N-size multiplication:
56 else
57 {
58 // Go along channels, multiplication size is ensured to be compatible with channels.
59 auto count = fused_weights->dim(0).value();
60 auto size = fused_weights->dim(fused_weights->rank() - 1).value();
61 float val;
62 for (uint32_t c = 0; c < count; c++)
63 {
64 val = multiplication->at<loco::DataType::FLOAT32>(c);
65 for (uint32_t i = 0; i < size; i++)
66 {
67 fused_weights->at<loco::DataType::FLOAT32>(c * size + i) *= val;
68 }
69 }
70 }
71 return fused_weights;
72}

110bool fuse_mul_with_fc(luci::CircleMul *mul)
111{
112 // Sanity check:
113 RETURN_FALSE_UNLESS(mul);
114 // Allow Mul node only with FLOAT32 data type:
115 RETURN_FALSE_UNLESS(mul->dtype() == loco::DataType::FLOAT32);
116 // Check if any FC node connects to Mul.
117 // Find the pattern of Mul(FC, CircleConst):
118 luci::CircleFullyConnected *fc = nullptr;
119 luci::CircleConst *multiplication = nullptr;
120 RETURN_FALSE_UNLESS(luci::fill(&fc, &multiplication).with_commutative_args_of(mul));
157 RETURN_FALSE_UNLESS(loco::succs(fc).size() == 1);
158 // Allow only FLOAT32 data type:
159 RETURN_FALSE_UNLESS(fc->dtype() == loco::DataType::FLOAT32);
160 // Allow only without activation functions as values are going to
161 // be multiplied before activation function.
162 RETURN_FALSE_UNLESS(fc->fusedActivationFunction() == luci::FusedActFunc::NONE);
163 // Check for weights being Constant:
164 auto weights = dynamic_cast<luci::CircleConst *>(fc->weights());
165 RETURN_FALSE_UNLESS(weights);
166 // Get rank of multiplication:
167 auto rank = multiplication->rank();
168 // Check that all dimensions are ones, checks broadcast capabilites.
169 // Last dimesion of multiplication must be compatible with FC.
170 // N-D case (N>1):
171 if (multiplication->rank() > 1)
172 {
173 // Check channel-wise broadcasting:
174 for (uint32_t i = 0; i < rank - 1; i++)
175 RETURN_FALSE_UNLESS(multiplication->dim(i).value() == 1);
176 // Check the last dimesion of Mul is the same with the first dimension of FullyConnected
177 RETURN_FALSE_UNLESS(multiplication->dim(rank - 1) == weights->dim(0));
178 }
179 // 1-D or scalar case:
180 else if (multiplication->rank() == 1)
181 {
182 RETURN_FALSE_UNLESS(multiplication->size<loco::DataType::FLOAT32>() == 1 ||
183 multiplication->size<loco::DataType::FLOAT32>() == weights->dim(0));
184 }
185 else if (multiplication->rank() == 0)
186 {
187 RETURN_FALSE_UNLESS(multiplication->size<loco::DataType::FLOAT32>() == 1);
188 }
189
190 // Only supports:
191 // (1) constant bias
192 // (2) no bias
193 auto bias = loco::must_cast<luci::CircleNode *>(fc->bias());
194 if (bias->opcode() == luci::CircleOpcode::CIRCLECONST)
195 {
196 // Create new bias to be updated with values:
197 auto const_bias = dynamic_cast<luci::CircleConst *>(fc->bias());
198 RETURN_FALSE_UNLESS(const_bias)
199 RETURN_FALSE_UNLESS(const_bias->dtype() == loco::DataType::FLOAT32);
200 // Create new bias with updated values and replace:
201 auto fused_bias = gen_fused_bias(const_bias, multiplication);
202 fc->bias(fused_bias);
203 }
204 else if (bias->opcode() != luci::CircleOpcode::CIRCLEOUTPUTEXCLUDE)
205 {
206 return false;
207 }
208
209 // Create new weights with updated values and replace:
210 auto fused_weights = gen_fused_weights(weights, multiplication);
211 fc->weights(fused_weights);
212
213 // Set origin and copy Activation Function if exisitng:
214 fc->fusedActivationFunction(mul->fusedActivationFunction());
215 luci::add_origin(fc, luci::get_origin(mul));
216
217 replace(mul).with(fc);
218
219 return true;
220}
