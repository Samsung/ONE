101 Paddings(luci::CircleConst *paddings)
102 {
103 assert(paddings->dtype() == loco::DataType::S32);
104 assert(paddings->rank() == 2);
105 assert(paddings->dim(1).value() == 2);
106 assert(paddings->size<loco::DataType::S32>() == paddings->rank() * 4);
107
108 for (uint32_t i = 0; i < paddings->dim(0).value(); i++)
109 {
110 Pad pad{.front = paddings->at<loco::DataType::S32>(i * 2),
111 .end = paddings->at<loco::DataType::S32>(i * 2 + 1)};
112 _padding_pos.emplace_back(pad);
113 }
114
115 assert(_padding_pos.size() == 4);
116 }
117
124 bool smaller_than(int32_t filter_h, int32_t filter_w)
125 {
126 auto &pad_H = _padding_pos.at(1);
127 auto &pad_W = _padding_pos.at(2);
128
129 return (pad_H.front < filter_h) && (pad_H.end < filter_h) && (pad_W.front < filter_w) &&
130 (pad_W.end < filter_w);
131 }

154 void apply(luci::CircleTranspose *transpose)
155 {
156 assert(transpose);
157 luci::CircleConst *perm = loco::must_cast<luci::CircleConst *>(transpose->perm());
158
159 std::vector<Pad> transposed_pos;
160 transposed_pos.resize(4);
161
162 for (uint32_t to = 0; to < 4; to++)
163 {
164 int32_t from = perm->at<loco::DataType::S32>(to);
165 transposed_pos.at(to) = _padding_pos.at(from);
166 }
167
168 _padding_pos = transposed_pos;
169 }

229template <loco::DataType DT> bool has_all_positive_values(luci::CircleConst *node)
230{
231 // Only numeric datatype is allowed
232 static_assert(DT != loco::DataType::Unknown);
233 static_assert(DT != loco::DataType::STRING);
234
235 assert(node);
236
237 auto size = node->size<DT>();
238 for (decltype(size) t = 0; t < size; t++)
239 {
240 typename loco::DataTypeImpl<DT>::Type val = node->at<DT>(t);
241 if (val <= 0)
242 return false;

245 return true;

260bool used_by_maxpool_only(luci::CircleNode *node, Paddings &paddings)
261{
262 auto successor = get_only_succ(node);
263
264 // when successor is not only-succ
265 if (successor == nullptr)
266 return false;
267
268 if (auto maxpool = dynamic_cast<luci::CircleMaxPool2D *>(successor))
269 {
270 // Let's check condition C4)
271 return paddings.smaller_than(maxpool->filter()->h(), maxpool->filter()->w());
272 }
273
274 // Let's check condition C6)
275 if (auto transpose = dynamic_cast<luci::CircleTranspose *>(successor))
276 {
277 auto appropriate = [](luci::CircleTranspose *transpose) {
278 luci::CircleConst *perm = loco::must_cast<luci::CircleConst *>(transpose->perm());
279
280 // For Transpose to be an input for MaxPool2D
281 return (transpose->rank() == 4) && (perm && perm->dtype() == loco::DataType::S32) &&
282 (perm->size<loco::DataType::S32>() == 4);
283 };
284
285 if (not appropriate(transpose))
286 return false;
287
288 paddings.apply(transpose);
289 return used_by_maxpool_only(transpose, paddings);
290 }
291 // Support more 'reshaping op' later
292
293 return false;
294}

296// Check condition C3), C4) and C6)
297bool used_by_maxpool_only(luci::CirclePadV2 *pad_v2)

299 // For PadV2 to be an input for MaxPool2D
300 if (pad_v2->rank() != 4)

308loco::Node *build_pad_from(luci::CirclePadV2 *pad_v2)
309{
310 auto copy_shape = [](const luci::CircleNode *src, luci::CircleNode *dest) {
311 auto rank = src->rank();
312 dest->rank(rank);
313
314 for (decltype(rank) axis = 0; axis < rank; axis++)
315 dest->dim(axis) = src->dim(axis);
316 };
317
318 auto g = pad_v2->graph();
319
320 auto pad = g->nodes()->create<luci::CirclePad>();
321 {
322 pad->name(pad_v2->name() + "/pad");
323 luci::add_origin(pad, luci::get_origin(pad_v2));
324
325 pad->dtype(pad_v2->dtype());
326 copy_shape(pad_v2, pad);
327
328 pad->input(pad_v2->input());
329 pad->paddings(pad_v2->paddings());
330 }
331
332 return pad;
333}

346bool substitute_padv2_to_pad(luci::CircleMaxPool2D *maxp)
347{
348 // precondition
349 assert(maxp);
350 assert(maxp->value());
351
352 auto pad_v2 = get_padv2(maxp->value());
353
354 if (pad_v2 == nullptr)
355 return false;
356
357 assert(pad_v2->input());
358
359 auto paddings = loco::must_cast<luci::CircleConst *>(pad_v2->paddings());
360 auto constant_values = loco::must_cast<luci::CircleConst *>(pad_v2->constant_values());
361
362 (void)paddings;
363 assert(paddings);
364 assert(paddings->dtype() == loco::DataType::S32);
365 assert(constant_values);
366 assert(constant_values->dtype() == pad_v2->dtype());
367
368 if (not positive_or_zero(pad_v2->input()))
369 return false;
370
371 if (has_all_positive_values(constant_values))
372 return false;
373
374 if (not used_by_maxpool_only(pad_v2))
375 return false;
376
377 auto pad = build_pad_from(pad_v2);
378
379 replace(pad_v2).with(pad);
380
381 return true;
382}

453bool SubstitutePadV2ToPadPass::run(loco::Graph *g)
454{
455 bool changed = false;
456 for (auto node : loco::active_nodes(loco::output_nodes(g)))
457 {
458 if (auto circle_node = dynamic_cast<luci::CircleMaxPool2D *>(node))
459 {
460 if (substitute_padv2_to_pad(circle_node))
461 {
462 changed = true;
463 }
464 }
465 }
466 return changed;
