62Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output,
63 const ITensorInfo *scale_factor)
64{
65 ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
66 ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
67 ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 2);
68 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F16, DataType::F32);
69 ARM_COMPUTE_RETURN_ERROR_ON(output->tensor_shape().total_size() == 0);
70 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::QASYMM8_SIGNED);
71 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
72 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(scale_factor, 1, DataType::F16,
73 DataType::F32);
74 ARM_COMPUTE_RETURN_ERROR_ON(scale_factor->tensor_shape().total_size() == 0);
75 ARM_COMPUTE_RETURN_ERROR_ON(scale_factor->num_dimensions() > 1);
76 ARM_COMPUTE_RETURN_ERROR_ON(scale_factor->dimension(0) != input->dimension(1));
77
78 return Status{};
79}

96inline float32x4_t round(const float32x4_t &fv)
97{
98 const float32x4_t point5_f32x4 = vdupq_n_f32(0.5f);
99 const float32x4_t zero_f32x4 = vdupq_n_f32(0.0f);
100 // If value < 0, mask = -1, else mask = 0
101 int32x4_t mask_less_zero_ui32x4 = reinterpret_cast<int32x4_t>(vcltq_f32(fv, zero_f32x4));
102 return vaddq_f32(fv, vaddq_f32(vcvtq_f32_s32(mask_less_zero_ui32x4), point5_f32x4));
103}

105inline int8x16_t vquantizeSymm(const float32x4x4_t &fv, float scale_factor_inv, int32_t max_scale)
106{
107 const float32x4_t vinvscale = vdupq_n_f32(scale_factor_inv);
108 const int32x4_t vposend = vdupq_n_s32(max_scale);
109 const int32x4_t vnagend = vdupq_n_s32(-max_scale);
110
111 const int32x4x4_t rf = {{
112#ifdef __aarch64__
113 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtnq_s32_f32(round(vmulq_f32(fv.val[0], vinvscale))))),
114 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtnq_s32_f32(round(vmulq_f32(fv.val[1], vinvscale))))),
115 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtnq_s32_f32(round(vmulq_f32(fv.val[2], vinvscale))))),
116 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtnq_s32_f32(round(vmulq_f32(fv.val[3], vinvscale))))),
117#else //__aarch64__
118 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtq_s32_f32(round(vmulq_f32(fv.val[0], vinvscale))))),
119 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtq_s32_f32(round(vmulq_f32(fv.val[1], vinvscale))))),
120 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtq_s32_f32(round(vmulq_f32(fv.val[2], vinvscale))))),
121 vminq_s32(vposend, vmaxq_s32(vnagend, vcvtq_s32_f32(round(vmulq_f32(fv.val[3], vinvscale))))),
122#endif //__aarch64__
123 }};
124 const int8x8_t pa = vqmovn_s16(vcombine_s16(vqmovn_s32(rf.val[0]), vqmovn_s32(rf.val[1])));
125 const int8x8_t pb = vqmovn_s16(vcombine_s16(vqmovn_s32(rf.val[2]), vqmovn_s32(rf.val[3])));
126 return vcombine_s8(pa, pb);
127}

164template <typename T> void NEQuantizationSymmetricKernel::quantize(const Window &window)
165{
166 constexpr auto window_step = 16;
167 const auto window_start_x = static_cast<int>(window.x().start());
168 const auto window_end_x = static_cast<int>(window.x().end());
169
170#ifdef __aarch64__
171 constexpr RoundingPolicy rounding_policy = RoundingPolicy::TO_NEAREST_EVEN;
172#else //__aarch64__
173 constexpr RoundingPolicy rounding_policy = RoundingPolicy::TO_NEAREST_UP;
174#endif //__aarch64__
175
176 // Collapse window and reset first dimension to handle tail calculations manually
177 // Support Only 2D input
178 Window win_collapsed = window;
179 Iterator input(_input, win_collapsed);
180 Iterator output(_output, win_collapsed);
181 const auto dim_x = _input->info()->dimension(0);
182 win_collapsed.set(Window::DimX, Window::Dimension(0, 1, 1));
183 execute_window_loop(
184 win_collapsed,
185 [&](const Coordinates &id) {
186 const auto start = reinterpret_cast<const T *>(input.ptr());
187 const auto min_max = std::minmax_element(start, start + dim_x);
188 const auto int8_scale = 127;
189 auto range = std::max(std::abs(*min_max.first), std::abs(*min_max.second));
190 if (range == 0)
191 {
192 *reinterpret_cast<T *>(_scale_factor->ptr_to_element({id.y()})) = 1;
193 range = 1;
194 }
195 else
196 {
197 *reinterpret_cast<T *>(_scale_factor->ptr_to_element({id.y()})) = range / int8_scale;
198 }
199 const auto scale_factor_inv = int8_scale / range;
200
201 auto input_ptr = reinterpret_cast<const T *>(input.ptr());
202 auto output_ptr = reinterpret_cast<int8_t *>(output.ptr());
203 int x = window_start_x;
204 for (; x <= (window_end_x - window_step); x += window_step)
205 {
206 wrapper::vstore(&output_ptr[x],
207 vquantizeSymm(load_value(&input_ptr[x]), scale_factor_inv, int8_scale));
208 }
209 // Compute left-over elements
210 for (; x < window_end_x; ++x)
211 {
212 int quantized = arm_compute::round(input_ptr[x] * scale_factor_inv, rounding_policy);
213 quantized = std::min(int8_scale, std::max(quantized, -int8_scale));
214 output_ptr[x] = static_cast<int8_t>(quantized);
215 }
216 },
217 input, output);
218}

220void NEQuantizationSymmetricKernel::run(const Window &window, const ThreadInfo &info)
221{
222 ARM_COMPUTE_UNUSED(info);
223 ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
224 ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INEKernel::window(), window);
225
226 switch (_input->info()->data_type())
227 {
228 case DataType::F32:
229 NEQuantizationSymmetricKernel::quantize<float>(window);
230 break;
231#ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
232 case DataType::F16:
233 NEQuantizationSymmetricKernel::quantize<float16_t>(window);
234 break;
235#endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
236 default:
237 ARM_COMPUTE_ERROR("Unsupported data type.");
238 }
