84void TensorBuilder::allocate(void)
85{
86 auto lifetime_map = cl_common::createLifetimeMap(_lifetime_seq, _parent_map);
87
88 for (auto &entry : lifetime_map)
89 {
90 auto &use = entry.second;
91 auto use_type = use.first;
92 auto use_index = use.second;
93 assert(use_index.valid());
94 if (use_type == UsesType::FIRST)
95 _tensor_mgr->startLifetime(use_index);
96 else
97 _tensor_mgr->finishLifetime(use_index);
98 }
99
100 _tensor_mgr->allocateConsts();
101
102 // TODO Since `_parent_map` is filled for all Concat nodes even if the node this backend uses
103 // After refactoring BackendContext we can uncomment this
104 // assert(_tensor_info_map.size() ==
105 // _tensor_mgr->nonconstTensors().size() + num of constants of _tensor_info_map +
106 // _parent_map.size());
107 _tensor_mgr->allocateNonconsts();
108}

112void TensorBuilder::buildTensors(void)
113{
114 assert(_tensor_mgr->constTensors().size() == 0);
115 assert(_tensor_mgr->nonconstTensors().size() == 0);
116 // Normal tensors
117 for (auto &entry : _tensor_info_map)
118 {
119 auto ind = entry.first;
120 if (_parent_map.count(ind) > 0)
121 continue;
122 auto type = _tensor_type_map.at(ind);
123 const auto &info = entry.second;
124 _tensor_mgr->buildTensor(ind, info, type);
125 }
126}
