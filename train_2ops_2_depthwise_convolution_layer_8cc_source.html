43void DepthwiseConvolutionLayer::configure(
44 const IPortableTensor *input, const IPortableTensor *kernel, const IPortableTensor *bias,
45 IPortableTensor *output, IPortableTensor *back_prop_input, IPortableTensor *grad_weights,
46 IPortableTensor *grad_bias, const IPortableTensor *back_prop_output, const uint32_t paddingLeft,
47 const uint32_t paddingRight, const uint32_t paddingTop, const uint32_t paddingBottom,
48 const uint32_t strideWidth, const uint32_t strideHeight, const uint32_t multiplier,
49 const uint32_t dilationWidth, const uint32_t dilationHeight, const ir::Activation activation,
50 const std::shared_ptr<ExternalContext> &external_context)
51{
52 cpu::ops::DepthwiseConvolutionLayer::configure(
53 input, kernel, bias, paddingLeft, paddingRight, paddingTop, paddingBottom, strideWidth,
54 strideHeight, multiplier, dilationWidth, dilationHeight, activation, output, external_context);
55 _back_prop_input = back_prop_input;
56 _back_prop_output = back_prop_output;
57 _grad_weights = grad_weights;
58 _grad_bias = grad_bias;
59
60 if (activation != ir::Activation::NONE)
61 {
62 _act_back_prop_output =
63 std::make_unique<BackPropTensor>(_back_prop_output->get_info(), _back_prop_output->layout());
64 _act_back_prop_output->setBuffer(
65 std::make_shared<basic::Allocator>(_act_back_prop_output->total_size()));
66 }
67
68 const int64_t k_packet_size = [&]() {
69 const auto data_type = _back_prop_output->data_type();
70 switch (data_type)
71 {
72 case OperandType::FLOAT32:
73 {
74 return _dconv_kernel->kPacketSize<float>();
75 }
76 default:
77 throw std::runtime_error("train DepthwiseConvolutionLayer: unsupported data type");
78 }
79 }();
80
81 const auto incoming_shape = getShape(_back_prop_output);
82 const auto filter_shape = getShape(_kernel);
83 const int batch = incoming_shape.Dims(0);
84 const int out_depth = incoming_shape.Dims(3);
85 const int filter_rows = filter_shape.Dims(1);
86 const int filter_cols = filter_shape.Dims(2);
87
88 const int filter_spatial_size = filter_rows * filter_cols;
89 const int padded_filter_inner_dim_size =
90 ((out_depth + k_packet_size - 1) / k_packet_size) * k_packet_size;
91
92 _use_padded_filter = (out_depth % k_packet_size) == 0 ? false : true;
93
94 // prepare padded_filter buffer for cker
95 auto padded_filter_info = ir::OperandInfo(_kernel->get_info());
96 padded_filter_info.shape({batch, filter_spatial_size, padded_filter_inner_dim_size});
97 _padded_filter = std::make_unique<Tensor>(padded_filter_info, _kernel->layout());
98 _padded_filter->setBuffer(std::make_shared<basic::Allocator>(_padded_filter->total_size()));
99
100 // prepare out_bprop and in_bprop buffer for cker
101 const int thread_count = _dconv_kernel->getThreadCount();
102
103 auto filter_buffers_info = ir::OperandInfo(_kernel->get_info());
104 filter_buffers_info.shape({thread_count, filter_spatial_size, padded_filter_inner_dim_size});
105 _filter_buffers = std::make_unique<Tensor>(filter_buffers_info, _kernel->layout());
106 _filter_buffers->setBuffer(std::make_shared<basic::Allocator>(_filter_buffers->total_size()));
107
108 auto filter_dim_buffers_info = ir::OperandInfo(_back_prop_input->get_info());
109 filter_dim_buffers_info.shape({thread_count, padded_filter_inner_dim_size});
110 _filter_dim_buffers =
111 std::make_unique<Tensor>(filter_dim_buffers_info, _back_prop_input->layout());
112 _filter_dim_buffers->setBuffer(
113 std::make_shared<basic::Allocator>(_filter_dim_buffers->total_size()));
114}

118void DepthwiseConvolutionLayer::backward()
119{
120 const auto data_type = _back_prop_output->data_type();
121 assert(data_type == _input->data_type());
122 switch (data_type)
123 {
124 case OperandType::FLOAT32:
125 {
126 assert(data_type == _grad_bias->data_type());
127 backwardFloat32();
128 break;
129 }
130 default:
131 throw std::runtime_error{"train DepthwiseConvolutionLayer: unsupported data type"};
132 }
133}

135void DepthwiseConvolutionLayer::backwardFloat32()
136{
137 // Calculate gradient for activation
138 const IPortableTensor *backprop_act;
139 try
140 {
141 backprop_act =
142 backpropActivation(_activation, _output, _back_prop_output, _act_back_prop_output.get());
143 }
144 catch (const std::exception &e)
145 {
146 throw std::runtime_error{"train DepthwiseConvolutionLayer: " + std::string(e.what())};
147 }
148 assert(backprop_act != nullptr);
149
150 nnfw::cker::DepthwiseConvParams dconv_params;
151 dconv_params.stride_width = _strideWidth;
152 dconv_params.stride_height = _strideHeight;
153 dconv_params.padding_values.width = _paddingLeft;
154 dconv_params.padding_values.height = _paddingTop;
155 dconv_params.depth_multiplier = _multiplier;
156
157 // Calculate gradient for input
158 _dconv_kernel->backpropInput(
159 dconv_params, getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_kernel),
160 getBuffer<float>(_kernel), getBuffer<float>(_padded_filter.get()), getShape(_back_prop_input),
161 getBuffer<float>(_back_prop_input), _use_padded_filter, getBuffer<float>(_filter_buffers.get()),
162 getBuffer<float>(_filter_dim_buffers.get()));
163
164 // Calculate gradient for weights
165 _dconv_kernel->backpropFilter(
166 dconv_params, getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_input),
167 getBuffer<float>(_input), getShape(_grad_weights), getBuffer<float>(_grad_weights),
168 getBuffer<float>(_padded_filter.get()), getBuffer<float>(_filter_buffers.get()));
169
170 // Calculate gradient for bias
171 if (_bias)
172 {
173 // TODO Use optimized kernel
174 assert(_grad_bias);
175 std::vector<int32_t> axes{0, 1, 2};
176 nnfw::cker::Reduce reduce_kernel;
177 reduce_kernel.prepare(backprop_act->getShape().rank(), axes.size());
178 bool result = reduce_kernel.ReduceGeneric<float>(
179 getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_grad_bias),
180 getBuffer<float>(_grad_bias), axes, false /* keep_dims */, 0.f,
181 [](const float current, const float in) -> float { return in + current; });
182 if (!result)
183 throw std::runtime_error{"train DepthwiseConvolutionLayer: Fail to calculate bias gradient"};
184 }
185}
