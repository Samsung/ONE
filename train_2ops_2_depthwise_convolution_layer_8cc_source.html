42void DepthwiseConvolutionLayer::configureBackward(IPortableTensor *back_prop_input,
43 IPortableTensor *grad_weights,
44 IPortableTensor *grad_bias,
45 const IPortableTensor *back_prop_output,
46 const ir::Activation activation)
47{
48 _back_prop_input = back_prop_input;
49 _back_prop_output = back_prop_output;
50 _grad_weights = grad_weights;
51 _grad_bias = grad_bias;
52
53 if (activation != ir::Activation::NONE)
54 {
55 _act_back_prop_output =
56 std::make_unique<BackPropTensor>(_back_prop_output->get_info(), _back_prop_output->layout());
57 _act_back_prop_output->setBuffer(
58 std::make_shared<basic::Allocator>(_act_back_prop_output->total_size()));
59 }
60
61 const int64_t k_packet_size = [&]() {
62 const auto data_type = _back_prop_output->data_type();
63 switch (data_type)
64 {
65 case OperandType::FLOAT32:
66 {
67 return _dconv_kernel->kPacketSize<float>();
68 }
69 default:
70 throw std::runtime_error("train DepthwiseConvolutionLayer: unsupported data type");
71 }
72 }();
73
74 const auto incoming_shape = getShape(_back_prop_output);
75 const auto filter_shape = getShape(_kernel);
76 const int batch = incoming_shape.Dims(0);
77 const int out_depth = incoming_shape.Dims(3);
78 const int filter_rows = filter_shape.Dims(1);
79 const int filter_cols = filter_shape.Dims(2);
80
81 const int filter_spatial_size = filter_rows * filter_cols;
82 const int padded_filter_inner_dim_size =
83 ((out_depth + k_packet_size - 1) / k_packet_size) * k_packet_size;
84
85 _use_padded_filter = (out_depth % k_packet_size) == 0 ? false : true;
86
87 // prepare padded_filter buffer for cker
88 auto padded_filter_info = ir::OperandInfo(_kernel->get_info());
89 padded_filter_info.shape({batch, filter_spatial_size, padded_filter_inner_dim_size});
90 _padded_filter = std::make_unique<Tensor>(padded_filter_info, _kernel->layout());
91 _padded_filter->setBuffer(std::make_shared<basic::Allocator>(_padded_filter->total_size()));
92
93 // prepare out_bprop and in_bprop buffer for cker
94 const int thread_count = _dconv_kernel->getThreadCount();
95
96 auto filter_buffers_info = ir::OperandInfo(_kernel->get_info());
97 filter_buffers_info.shape({thread_count, filter_spatial_size, padded_filter_inner_dim_size});
98 _filter_buffers = std::make_unique<Tensor>(filter_buffers_info, _kernel->layout());
99 _filter_buffers->setBuffer(std::make_shared<basic::Allocator>(_filter_buffers->total_size()));
100
101 auto filter_dim_buffers_info = ir::OperandInfo(_back_prop_input->get_info());
102 filter_dim_buffers_info.shape({thread_count, padded_filter_inner_dim_size});
103 _filter_dim_buffers =
104 std::make_unique<Tensor>(filter_dim_buffers_info, _back_prop_input->layout());
105 _filter_dim_buffers->setBuffer(
106 std::make_shared<basic::Allocator>(_filter_dim_buffers->total_size()));
107}

111void DepthwiseConvolutionLayer::backward()
112{
113 const auto data_type = _back_prop_output->data_type();
114 assert(data_type == _input->data_type());
115 switch (data_type)
116 {
117 case OperandType::FLOAT32:
118 {
119 assert(data_type == _grad_bias->data_type());
120 backwardFloat32();
121 break;
122 }
123 default:
124 throw std::runtime_error{"train DepthwiseConvolutionLayer: unsupported data type"};
125 }
126}

128void DepthwiseConvolutionLayer::backwardFloat32()
129{
130 // Calculate gradient for activation
131 const IPortableTensor *backprop_act;
132 try
133 {
134 backprop_act =
135 backpropActivation(_activation, _output, _back_prop_output, _act_back_prop_output.get());
136 }
137 catch (const std::exception &e)
138 {
139 throw std::runtime_error{"train DepthwiseConvolutionLayer: " + std::string(e.what())};
140 }
141 assert(backprop_act != nullptr);
142
143 nnfw::cker::DepthwiseConvParams dconv_params;
144 dconv_params.stride_width = _strideWidth;
145 dconv_params.stride_height = _strideHeight;
146 dconv_params.padding_values.width = _paddingLeft;
147 dconv_params.padding_values.height = _paddingTop;
148 dconv_params.depth_multiplier = _multiplier;
149
150 // Calculate gradient for input
151 _dconv_kernel->backpropInput(
152 dconv_params, getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_kernel),
153 getBuffer<float>(_kernel), getBuffer<float>(_padded_filter.get()), getShape(_back_prop_input),
154 getBuffer<float>(_back_prop_input), _use_padded_filter, getBuffer<float>(_filter_buffers.get()),
155 getBuffer<float>(_filter_dim_buffers.get()));
156
157 // Calculate gradient for weights
158 _dconv_kernel->backpropFilter(
159 dconv_params, getShape(backprop_act), getBuffer<float>(backprop_act), getShape(_input),
160 getBuffer<float>(_input), getShape(_grad_weights), getBuffer<float>(_grad_weights),
161 getBuffer<float>(_padded_filter.get()), getBuffer<float>(_filter_buffers.get()));
162
163 // Calculate gradient for bias
164 if (_bias)
165 {
166 assert(_grad_bias);
167 biasGrad(backprop_act, _grad_bias);
168 }
169}
