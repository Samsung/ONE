77int entry(int argc, char **argv)
78{
79 LOGGER(l);
80
81 arser::Arser arser("circle-partitioner provides circle model partitioning");
82
83 build_arser(arser);
84
85 try
86 {
87 arser.parse(argc, argv);
88 }
89 catch (const std::runtime_error &err)
90 {
91 std::cerr << err.what() << std::endl;
92 std::cerr << arser;
93 return EXIT_FAILURE;
94 }
95
96 std::string partition_file = arser.get<std::string>(opt_part_file);
97 std::string input_file = arser.get<std::string>(opt_input_file);
98 std::string work_folder = ".";
99
100 if (arser[opt_work_path])
101 {
102 work_folder = arser.get<std::string>(opt_work_path);
103 }
104
105 std::string partition_path = work_folder + "/" + partition_file;
106 std::string input_path = work_folder + "/" + input_file;
107
108 auto module = load_model(input_path);
109 if (module.get() == nullptr)
110 {
111 return EXIT_FAILURE;
112 }
113 // Run default shape/dtype inference before validation
114 // NOTE CircleWhileOut default shape is INVALID as it needs initial shape
115 // inference. This is cause of WHILE may have dynamic shape.
116 luci::CircleOptimizer optimizer;
117 (void)optimizer.options(); // need to call this to make internal member
118 for (size_t g = 0; g < module->size(); ++g)
119 {
120 auto graph = module->graph(g);
121 optimizer.optimize(graph);
122 }
123 if (!luci::validate(module.get()))
124 {
125 return EXIT_FAILURE;
126 }
127
128 // Read partition information
129 INFO(l) << "--- Read PartitionConfig-----------------------" << std::endl;
130 auto partition = partee::read(partition_path);
131 INFO(l) << partition << std::endl;
132
133 // override with command line arguments
134 {
135 if (arser[opt_bks])
136 {
137 auto backend_backends = arser.get<std::string>(opt_bks);
138 partition.groups = pepper::csv_to_vector<std::string>(backend_backends);
139 }
140 if (arser[opt_def])
141 {
142 partition.default_group = arser.get<std::string>(opt_def);
143 }
144 }
145 if (!luci::validate(partition))
146 {
147 // NOTE error reason/message is put to std::cerr inside validate()
148 return EXIT_FAILURE;
149 }
150
151 INFO(l) << "--- PartitionConfig final----------------------" << std::endl;
152 INFO(l) << partition << std::endl;
153
154 // apply partition to module
155 auto pms = luci::apply(module.get(), partition);
156
157 // validate partitioned modules
158 for (auto &pmodule : pms.pmodules)
159 {
160 for (size_t g = 0; g < pmodule.module->size(); ++g)
161 {
162 auto graph = pmodule.module->graph(g);
163 if (graph == nullptr)
164 {
165 std::cerr << "ERROR: Failed to create partition model" << std::endl;
166 return EXIT_FAILURE;
167 }
168 if (!luci::validate(graph))
169 {
170 std::cerr << "ERROR: Failed to create partition model" << std::endl;
171 return EXIT_FAILURE;
172 }
173 }
174 }
175
176 INFO(l) << "--- Partition Export---------------------------" << std::endl;
177 uint32_t idx = 1;
178 for (auto &pmodule : pms.pmodules)
179 {
180 // Export to output circle file
181 luci::CircleExporter exporter;
182
183 auto output_path = partee::make_path(work_folder, input_path, idx, pmodule.group);
184 pmodule.name = partee::get_filename_ext(output_path);
185 INFO(l) << "--- " << output_path << ": " << pmodule.name << std::endl;
186
187 luci::CircleFileExpContract contract(pmodule.module.get(), output_path);
188 if (!exporter.invoke(&contract))
189 {
190 std::cerr << "ERROR: Failed to export '" << output_path << "'" << std::endl;
191 return EXIT_FAILURE;
192 }
193 idx++;
194 }
195
196 INFO(l) << "--- Partition connection information-----------" << std::endl;
197 if (!partee::export_part_conn_json(work_folder, input_file, module.get(), pms))
198 {
199 return EXIT_FAILURE;
200 }
201 if (!partee::export_part_conn_ini(work_folder, input_file, module.get(), pms))
202 {
203 return EXIT_FAILURE;
204 }
205
206 INFO(l) << "--- Partition done-----------------------------" << std::endl << std::endl;
207
208 return EXIT_SUCCESS;
209}
