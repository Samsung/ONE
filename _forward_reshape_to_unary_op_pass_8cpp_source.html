59luci::CircleReshape *create_cloned_reshape(luci::CircleReshape *reshape)
60{
61 assert(reshape != nullptr); // FIX_CALLER_UNLESS
62
63 luci::CircleConst *cloned_shape = clone_shape(reshape);
64 if (cloned_shape == nullptr)
65 return nullptr;
66
67 auto cloned_node = luci::clone_node(reshape, reshape->graph());
68 if (cloned_node == nullptr)
69 return nullptr;
70
71 auto new_reshape = loco::must_cast<luci::CircleReshape *>(cloned_node);
72 new_reshape->shape(cloned_shape);
73 new_reshape->name(reshape->name() + "_C");
74 luci::add_origin(new_reshape, luci::get_origin(reshape));
75
76 return new_reshape;
77}

79bool forward_reshape(luci::CircleReshape *reshape, luci::CircleNeg *neg)
80{
81 assert(reshape != nullptr);
82 assert(neg != nullptr);
83
84 luci::CircleConst *cloned_shape = clone_shape(reshape);
85 if (cloned_shape == nullptr)
86 return false;
87
88 auto name = reshape->name();
89 assert(name.length() > 0);
90 loco::Graph *graph = neg->graph();
91 // create reshape placed after neg
92 luci::CircleReshape *new_reshape = graph->nodes()->create<luci::CircleReshape>();
93 copy_shape(reshape, new_reshape);
94 new_reshape->shape(cloned_shape);
95 new_reshape->name(name + "_C");
96 luci::add_origin(new_reshape, luci::get_origin(reshape));
97
98 // reconnect network
99 loco::replace(neg).with(new_reshape);
100 neg->x(reshape->tensor());
101 new_reshape->tensor(neg);
102
103 // Do shape inference for this node again.
104 neg->shape_status(luci::ShapeStatus::UNDEFINED);
105
106 return true;
107}

109bool forward_reshape(luci::CircleReshape *reshape, luci::CircleLogistic *logit)
110{
111 assert(reshape != nullptr); // FIX_CALLER_UNLESS
112 assert(logit != nullptr); // FIX_CALLER_UNLESS
113
114 auto new_reshape = create_cloned_reshape(reshape);
115 if (not new_reshape)
116 return false;
117
118 // reconnect network
119 loco::replace(logit).with(new_reshape);
120 logit->x(reshape->tensor());
121 new_reshape->tensor(logit);
122
123 // Do shape inference for this node again.
124 logit->shape_status(luci::ShapeStatus::UNDEFINED);
125
126 return true;
127}

129class ForwardReshape final : public luci::CircleNodeMutableVisitor<bool>

190bool ForwardReshapeToUnaryOpPass::run(loco::Graph *g)
191{
192 bool changed = false;
193 ForwardReshape forward;
194 for (auto node : loco::active_nodes(loco::output_nodes(g)))
195 {
196 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
197 if (circle_node->accept(&forward))
198 changed = true;
199 }
200 return changed;
201}
