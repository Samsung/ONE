61luci::CircleReshape *create_cloned_reshape(luci::CircleReshape *reshape)
62{
63 assert(reshape != nullptr); // FIX_CALLER_UNLESS
64
65 luci::CircleConst *cloned_shape = clone_shape(reshape);
66 if (cloned_shape == nullptr)
67 return nullptr;
68
69 auto cloned_node = luci::clone_node(reshape, reshape->graph());
70 if (cloned_node == nullptr)
71 return nullptr;
72
73 auto new_reshape = loco::must_cast<luci::CircleReshape *>(cloned_node);
74 new_reshape->shape(cloned_shape);
75 new_reshape->name(reshape->name() + "_C");
76 luci::add_origin(new_reshape, luci::get_origin(reshape));
77
78 return new_reshape;

81bool forward_reshape(luci::CircleReshape *reshape, luci::CircleMean *mean, uint32_t axis)
82{
83 assert(reshape != nullptr); // FIX_CALLER_UNLESS
84 assert(mean != nullptr); // FIX_CALLER_UNLESS
85
86 auto new_reshape = create_cloned_reshape(reshape);
87 if (not new_reshape)
88 return false;
89
90 // reconnect network
91 loco::replace(mean).with(new_reshape);
92 mean->input(reshape->tensor());
93 new_reshape->tensor(mean);
94
95 // Change const shape axis value
96 auto *shape_reshape = loco::must_cast<luci::CircleConst *>(new_reshape->shape());
97 assert(shape_reshape->dtype() == loco::DataType::S32); // FIX_CALLER_UNLESS
98 assert(axis < shape_reshape->size<loco::DataType::S32>()); // FIX_CALLER_UNLESS
99 // Mean reduction will make value to '1'
100 shape_reshape->at<loco::DataType::S32>(axis) = 1;
101
102 // Do shape inference for this node again.
103 mean->shape_status(luci::ShapeStatus::UNDEFINED);
104 reshape->shape_status(luci::ShapeStatus::UNDEFINED);
105
106 return true;
107}

129bool forward_reshape(luci::CircleReshape *reshape, luci::CircleNeg *neg)
130{
131 assert(reshape != nullptr);
132 assert(neg != nullptr);
133
134 luci::CircleConst *cloned_shape = clone_shape(reshape);
135 if (cloned_shape == nullptr)
136 return false;
137
138 auto name = reshape->name();
139 assert(name.length() > 0);
140 loco::Graph *graph = neg->graph();
141 // create reshape placed after neg
142 luci::CircleReshape *new_reshape = graph->nodes()->create<luci::CircleReshape>();
143 copy_shape(reshape, new_reshape);
144 new_reshape->shape(cloned_shape);
145 new_reshape->name(name + "_C");
146 luci::add_origin(new_reshape, luci::get_origin(reshape));
147
148 // reconnect network
149 loco::replace(neg).with(new_reshape);
150 neg->x(reshape->tensor());
151 new_reshape->tensor(neg);
152
153 // Do shape inference for this node again.
154 neg->shape_status(luci::ShapeStatus::UNDEFINED);
155
156 return true;
157}

179bool forward_reshape(luci::CircleReshape *reshape, luci::CircleMul *div,
180 luci::CircleConst *const_value)
181{
182 assert(reshape != nullptr); // FIX_CALLER_UNLESS
183 assert(div != nullptr); // FIX_CALLER_UNLESS
184
185 auto new_reshape = create_cloned_reshape(reshape);
186 if (not new_reshape)
187 return false;
188
189 const auto prev = loco::must_cast<luci::CircleNode *>(reshape->tensor());
190
191 // Reshape can change rank of tensor, so we need to update constant value accordingly.
192 assert(const_value->size<loco::DataType::FLOAT32>() == 1);
193 auto cloned_const = clone(const_value);
194 cloned_const->rank(prev->rank());
195 for (uint32_t i = 0; i < prev->rank(); ++i)
196 {
197 cloned_const->dim(i).set(1);
198 }
199
200 // reconnect network
201 loco::replace(div).with(new_reshape);
202 if (div->x() == const_value)
203 {
204 div->x(cloned_const);
205 div->y(reshape->tensor());
206 }
207 else
208 {
209 assert(div->y() == const_value);
210 div->x(reshape->tensor());
211 div->y(cloned_const);
212 }
213 new_reshape->tensor(div);
214
215 // Do shape inference for this node again.
216 div->shape_status(luci::ShapeStatus::UNDEFINED);
217
218 return true;
219}

250class ForwardReshape final : public luci::CircleNodeMutableVisitor<bool>
251{
252protected:
253 bool visit(luci::CircleNode *node)
254 {
255 LOGGER(l);
256 INFO(l) << "ForwardReshape: Unsupported operator: " << node->name() << std::endl;
257 return false;
258 }
259
288 bool visit(luci::CircleMean *node)
289 {
290 luci::CircleReshape *reshape = nullptr;
291 luci::CircleConst *axis = nullptr;
292
293 reshape = dynamic_cast<luci::CircleReshape *>(node->input());
294 axis = dynamic_cast<luci::CircleConst *>(node->reduction_indices());
295
296 if (reshape == nullptr or axis == nullptr)
297 return false;
298
299 if (axis->dtype() != loco::DataType::S32)
300 return false;
301
302 // Should be scalar
303 if (axis->size<loco::DataType::S32>() != 1)
304 return false;
305
306 // axis value
307 auto axis_value = axis->at<loco::DataType::S32>(0);
308
309 if (axis_value < 0)
310 axis_value += static_cast<int32_t>(reshape->rank());
311
312 assert(axis_value >= 0);
313
314 if (node->keep_dims() != true)
315 return false;
316
317 auto reshape_input = loco::must_cast<luci::CircleNode *>(reshape->tensor());
318
319 // reshape shouldn't change rank
320 if (reshape_input->rank() != reshape->rank())
321 return false;
322
323 assert(reshape_input->rank() > static_cast<uint32_t>(axis_value));
324
325 for (int32_t i = 0; i <= axis_value; ++i)
326 {
327 if (not reshape_input->dim(i).known() or
328 reshape_input->dim(i).value() != reshape->dim(i).value())
329 return false;
330 }
331
332 return forward_reshape(reshape, node, axis_value);
333 }
334
335 bool visit(luci::CircleAbs *node)
336 {
337 auto reshape = as_reshape(node->x());
338 if (reshape == nullptr)
339 return false;
340 return forward_reshape(reshape, node);
341 }
342
343 bool visit(luci::CircleNeg *node)
344 {
345 auto reshape = as_reshape(node->x());
346 if (reshape == nullptr)
347 return false;
348 return forward_reshape(reshape, node);
349 }
350
351 bool visit(luci::CircleLogistic *node)
352 {
353 auto reshape = as_reshape(node->x());
354 if (reshape == nullptr)
355 return false;
356
357 return forward_reshape(reshape, node);
358 }
359
360 bool visit(luci::CircleDiv *node)
361 {
362 luci::CircleReshape *reshape = nullptr;
363 luci::CircleConst *const_value = nullptr;
364
365 if (not luci::fill(&reshape, &const_value).with_commutative_args_of(node))
366 return false;
367
368 if (const_value->dtype() != loco::DataType::FLOAT32)
369 return false;
370
371 // Should be scalar
372 if (const_value->size<loco::DataType::FLOAT32>() != 1)
373 return false;
374
375 return forward_reshape(reshape, node, const_value);
376 }
377
378 bool visit(luci::CircleMul *node)
379 {
380 luci::CircleReshape *reshape = nullptr;
381 luci::CircleConst *const_value = nullptr;
382
383 if (not luci::fill(&reshape, &const_value).with_commutative_args_of(node))
384 return false;
385
386 if (const_value->dtype() != loco::DataType::FLOAT32)
387 return false;
388
389 // Should be scalar
390 if (const_value->size<loco::DataType::FLOAT32>() != 1)
391 return false;
392
393 return forward_reshape(reshape, node, const_value);
394 }

436bool ForwardReshapeToUnaryOpPass::run(loco::Graph *g)
437{
438 bool changed = false;
439 ForwardReshape forward;
440 for (auto node : loco::active_nodes(loco::output_nodes(g)))
441 {
442 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
443 if (circle_node->accept(&forward))
444 changed = true;
445 }
446 return changed;
447}
