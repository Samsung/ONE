29size_t UserTensor::calcOffset(const ir::Coordinates &coords) const
30{
31 size_t rank = getShape().rank();
32 size_t offset = 0;
33 for (size_t i = 0; i < rank; ++i)
34 {
35 offset = offset * getShape().dim(i) + coords[i];
36 }
37 offset *= sizeOfDataType(data_type());
38 return offset;
39}

41bool UserTensor::applyShape(const ir::Shape &new_shape)
42{
43 // User tensors cannot be reallocated.
44 auto new_size = new_shape.num_elements() * ir::sizeOfDataType(data_type());
45 if (total_size() < new_size)
46 throw InsufficientBufferSizeException{"User given buffer size is too small."};
47 setShape(new_shape);
48 return true;
49}
