48template <loco::DataType DT>
49void copy_data(const VectorWrapper<uint8_t> &raw_data, uint32_t num_elements,
50 CircleConst *const_node)
51{
52 using T = typename loco::DataTypeImpl<DT>::Type;
53
54 // TODO calculate the exact buffer size of sparse tensor
55 if (const_node->sparsityparam())
56 {
57 num_elements = raw_data.size() / sizeof(T);
58 }
59
60 assert(raw_data.size() == num_elements * sizeof(T));
61 const auto *data = reinterpret_cast<const T *>(raw_data.data());
62
63 const_node->size<DT>(num_elements);
64 for (uint32_t i = 0; i < num_elements; ++i)
65 {
66 const_node->at<DT>(i) = data[i];
67 }
68}

70template <>
71void copy_data<loco::DataType::STRING>(const VectorWrapper<uint8_t> &raw_data,
72 uint32_t num_elements, CircleConst *const_node)
73{
74 assert(const_node->sparsityparam() == nullptr);
75
76 const auto *data = reinterpret_cast<const char *>(raw_data.data());
77 const auto *i32d = reinterpret_cast<const int32_t *>(raw_data.data());
78
79 // de-serialize string data
80 // int32_t count
81 // int32_t offsets[count + 1]
82 // string values[count]
83 assert(static_cast<uint32_t>(*i32d) == num_elements);
84 i32d++; // skip count
85
86 std::vector<int32_t> offsets;
87 offsets.push_back(*i32d++);
88 for (uint32_t i = 0; i < num_elements; ++i)
89 {
90 offsets.push_back(*i32d++);
91 }
92 assert(offsets.size() == num_elements + 1);
93
94 const_node->size<loco::DataType::STRING>(num_elements);
95 for (uint32_t i = 0; i < num_elements; ++i)
96 {
97 int32_t start = offsets[i];
98 int32_t next = offsets[i + 1];
99
100 std::string value(data + start, next - start);
101 const_node->at<loco::DataType::STRING>(i) = value;
102 }
103}

110CircleNode *CircleConstNodeBuilder::build(TensorIndex tensor_index,
111 GraphBuilderContext *context) const
112{
113 assert(tensor_index >= 0);
114 LOGGER(l);
115
116 auto graph = context->graph();
117 auto reader = context->reader();
118 const auto tensors = reader->tensors();
119 const auto const_tensor = tensors[tensor_index];
120 assert(const_tensor != nullptr);
121 if (const_tensor->is_variable())
122 {
123 // Create CircleVariable for variable
124 return nullptr;
125 }
126
127 assert(reader->buffers()[const_tensor->buffer()] != nullptr);
128 const auto buffer = wrap(reader->buffers()[const_tensor->buffer()]->data());
129 const auto const_dims = wrap(const_tensor->shape()); // in NHWC
130 if (const_dims.size() == 0 && buffer.empty())
131 {
132 // unknown shape tensor and scalar tensor
133 return nullptr;
134 }
135
136 // if tensor_index is used as output to some other operator, this is not a constant
137 auto tensoroutputs = context->tensoroutputs();
138 if (tensoroutputs->find(tensor_index))
139 {
140 // other operator output tensor
141 return nullptr;
142 }
143
144 uint32_t num_elements = 1;
145 for (uint32_t r = 0; r < const_dims.size(); ++r)
146 {
147 num_elements = num_elements * const_dims[r];
148 }
149
150 if (buffer.empty() && num_elements > 0)
151 {
152 // normal empty tensor
153 return nullptr;
154 }
155
156 auto const_node = graph->nodes()->create<CircleConst>();
157 copy_tensor_attributes(const_tensor, const_node);
158 const_node->shape_status(luci::ShapeStatus::VALID);
159 INFO(l) << "[luci] NodeFinder const_node(" << tensor_index << ") -> " << const_node << " "
160 << const_dims << std::endl;
161 if (num_elements > 0)
162 {
163 switch (luci_datatype(const_tensor->type()))
164 {
165 case loco::DataType::FLOAT32:
166 copy_data<loco::DataType::FLOAT32>(buffer, num_elements, const_node);
167 break;
168
169 case loco::DataType::FLOAT16:
170 copy_data<loco::DataType::FLOAT16>(buffer, num_elements, const_node);
171 break;
172
173 case loco::DataType::U8:
174 copy_data<loco::DataType::U8>(buffer, num_elements, const_node);
175 break;
176
177 case loco::DataType::S8:
178 copy_data<loco::DataType::S8>(buffer, num_elements, const_node);
179 break;
180
181 case loco::DataType::S16:
182 copy_data<loco::DataType::S16>(buffer, num_elements, const_node);
183 break;
184
185 case loco::DataType::S32:
186 copy_data<loco::DataType::S32>(buffer, num_elements, const_node);
187 break;
188
189 case loco::DataType::S64:
190 copy_data<loco::DataType::S64>(buffer, num_elements, const_node);
191 break;
192
193 case loco::DataType::BOOL:
194 copy_data<loco::DataType::BOOL>(buffer, num_elements, const_node);
195 break;
196
197 case loco::DataType::STRING:
198 copy_data<loco::DataType::STRING>(buffer, num_elements, const_node);
199 break;
200
201 default:
202 throw oops::UserExn("Unsupported tensor type",
203 circle::EnumNameTensorType(const_tensor->type()));
204 }
205 }
206
207 return const_node;
208}
