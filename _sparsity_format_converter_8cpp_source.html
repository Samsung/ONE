35uint64_t GetFlattenedIndex(const std::vector<int> &indices, const std::vector<int> &shape)
36{
37 uint64_t index = 0;
38 int sub_elements = 1;
39 for (int i = shape.size() - 1; i >= 0; i--)
40 {
41 index += indices[i] * sub_elements;
42 sub_elements *= shape[i];
43 }
44 return index;
45}

48{

66template <typename T>
67FormatConverter<T>::FormatConverter(const std::vector<int> &shape, const TfLiteSparsity &sparsity)
68{
69 auto traversal_order = TfLiteIntArrayToVector(sparsity.traversal_order);
70 auto block_map = TfLiteIntArrayToVector(sparsity.block_map);
71
72 std::vector<TfLiteDimensionType> format(sparsity.dim_metadata_size);
73 std::vector<int> dense_size(sparsity.dim_metadata_size);
74 std::vector<std::vector<int>> segments(sparsity.dim_metadata_size);
75 std::vector<std::vector<int>> indices(sparsity.dim_metadata_size);
76 for (int i = 0; i < sparsity.dim_metadata_size; i++)
77 {
78 format[i] = sparsity.dim_metadata[i].format;
79 dense_size[i] = sparsity.dim_metadata[i].dense_size;
80 segments[i] = TfLiteIntArrayToVector(sparsity.dim_metadata[i].array_segments);
81 indices[i] = TfLiteIntArrayToVector(sparsity.dim_metadata[i].array_indices);
82 }
83
84 InitSparseToDenseConverter(shape, std::move(traversal_order), std::move(format),
85 std::move(dense_size), std::move(segments), std::move(indices),
86 std::move(block_map));
87}

89template <typename T>
90void FormatConverter<T>::InitSparseToDenseConverter(
91 std::vector<int> shape, std::vector<int> traversal_order, std::vector<TfLiteDimensionType> format,
92 std::vector<int> dense_size, std::vector<std::vector<int>> segments,
93 std::vector<std::vector<int>> indices, std::vector<int> block_map)
94{
95 dense_shape_ = std::move(shape);
96 traversal_order_ = std::move(traversal_order);
97 block_map_ = std::move(block_map);
98 format_ = std::move(format);
99
100 dense_size_ = 1;
101 for (size_t i = 0; i < dense_shape_.size(); i++)
102 {
103 dense_size_ *= dense_shape_[i];
104 }
105
106 dim_metadata_.resize(2 * format_.size());
107 for (size_t i = 0; i < format_.size(); i++)
108 {
109 if (format_[i] == kTfLiteDimDense)
110 {
111 dim_metadata_[2 * i] = {dense_size[i]};
112 }
113 else
114 {
115 dim_metadata_[2 * i] = std::move(segments[i]);
116 dim_metadata_[2 * i + 1] = std::move(indices[i]);
117 }
118 }
119
120 int original_rank = dense_shape_.size();
121 int block_dim = 0;
122
123 blocked_shape_.resize(original_rank);
124 block_size_.resize(block_map_.size());
125 for (int i = 0; i < original_rank; i++)
126 {
127 if (block_dim < (int)block_map_.size() && block_map_[block_dim] == i)
128 {
129 if (original_rank + block_dim < (int)traversal_order_.size())
130 {
131 int orig_dim = traversal_order_[original_rank + block_dim];
132 block_size_[block_dim] = dense_size[orig_dim];
133 blocked_shape_[i] = dense_shape_[i] / dense_size[orig_dim];
134 block_dim++;
135 }
136 }
137 else
138 {
139 blocked_shape_[i] = dense_shape_[i];
140 }
141 }
142}

144template <typename T>
145void FormatConverter<T>::Populate(const T *src_data, std::vector<int> indices, int level,
146 int prev_idx, int *src_data_ptr, T *dest_data)
147{
148 if (static_cast<size_t>(level) == indices.size())
149 {
150 int orig_rank = dense_shape_.size();
151 std::vector<int> orig_idx;
152 orig_idx.resize(orig_rank);
153 int i = 0;
154 for (; static_cast<size_t>(i) < orig_idx.size(); i++)
155 {
156 int orig_dim = traversal_order_[i];
157 orig_idx[orig_dim] = indices[i];
158 }
159
160 for (; static_cast<size_t>(i) < indices.size(); i++)
161 {
162 const int block_idx = traversal_order_[i] - orig_rank;
163 const int orig_dim = block_map_[block_idx];
164 orig_idx[orig_dim] = orig_idx[orig_dim] * block_size_[block_idx] + indices[i];
165 }
166
167 dest_data[GetFlattenedIndex(orig_idx, dense_shape_)] = src_data[*src_data_ptr];
168
169 *src_data_ptr = *src_data_ptr + 1;
170 return;
171 }
172
173 const int metadata_idx = 2 * level;
174 const int shape_of_level = dim_metadata_[metadata_idx][0];
175 if (format_[level] == kTfLiteDimDense)
176 {
177 for (int i = 0; i < shape_of_level; i++)
178 {
179 indices[level] = i;
180 Populate(src_data, indices, level + 1, prev_idx * shape_of_level + i, src_data_ptr,
181 dest_data);
182 }
183 }
184 else if (static_cast<size_t>(prev_idx + 1) < dim_metadata_[metadata_idx].size())
185 {
186 const auto &array_segments = dim_metadata_[metadata_idx];
187 const auto &array_indices = dim_metadata_[metadata_idx + 1];
188 for (int i = array_segments[prev_idx]; i < array_segments[prev_idx + 1]; i++)
189 {
190 if (static_cast<size_t>(i) < array_indices.size() &&
191 static_cast<size_t>(level) < indices.size())
192 {
193 indices[level] = array_indices[i];
194 Populate(src_data, indices, level + 1, i, src_data_ptr, dest_data);
195 }
196 }
197 }
198}

200template <typename T> bool FormatConverter<T>::SparseToDense(const T *src_data)
201{
202 data_.resize(dense_size_);
203 std::fill(data_.begin(), data_.end(), T(0));
204
205 int total_rank = traversal_order_.size();
206 int src_data_ptr = 0;
207 std::vector<int> indices(total_rank);
208 Populate(src_data, indices, 0, 0, &src_data_ptr, data_.data());
209
210 return true;
211}

223sparsity::TfLiteDimensionType to_tflite_sparsity(luci::DimensionType dt)
224{
225 switch (dt)
226 {
227 case luci::DimensionType::DENSE:
228 return sparsity::TfLiteDimensionType::kTfLiteDimDense;
229 case luci::DimensionType::SPARSE_CSR:
230 return sparsity::TfLiteDimensionType::kTfLiteDimSparseCSR;
231 }
232 return sparsity::TfLiteDimensionType::kTfLiteDimDense;
233}

235sparsity::TfLiteIntArray *to_tflite_sparsity(const luci::SparseIndexVector &data)
236{
237 auto type = data.type();
238 switch (type)
239 {
240 case luci::SparseIndexVectorType::NONE:
241 {
242 std::vector<int32_t> empty;
243 return makeTfLiteArray(empty);
244 }
245 case luci::SparseIndexVectorType::I32:
246 return makeTfLiteArray<int32_t>(*data.as_int32_vector());
247 case luci::SparseIndexVectorType::U16:
248 return makeTfLiteArray<uint16_t>(*data.as_uint16_vector());
249 case luci::SparseIndexVectorType::U8:
250 return makeTfLiteArray<uint8_t>(*data.as_uint8_vector());
251 default:
252 INTERNAL_EXN_V("unsupported SparseIndexVectorType", oops::to_uint32(type));
253 }
254}

256sparsity::TfLiteSparsity to_tflite_sparsity(const luci::SparsityParam *sp)
257{
258 sparsity::TfLiteSparsity tflsp;
259 tflsp.traversal_order = makeTfLiteArray(sp->traversal_order);
260 tflsp.block_map = makeTfLiteArray(sp->block_map);
261 tflsp.dim_metadata = makeTfLiteDimensionMetadata(sp->dim_metadata);
262 tflsp.dim_metadata_size = sp->dim_metadata.size();
263 return tflsp;
264}

266template <typename T> sparsity::TfLiteIntArray *makeTfLiteArray(const std::vector<T> &data)
267{
268 size_t cn = data.size();
269 size_t sz = 1 + data.size();
270 sparsity::TfLiteIntArray *sp = (sparsity::TfLiteIntArray *)(new int[sz]);
271 sp->size = cn;
272 for (size_t i = 0; i < cn; ++i)
273 {
274 sp->data[i] = data[i];
275 }
276 return sp;
277}

279sparsity::TfLiteDimensionMetadata *
280makeTfLiteDimensionMetadata(const std::vector<luci::DimMetaData> &data)
281{
282 size_t cn = data.size();
283 sparsity::TfLiteDimensionMetadata *tfldm = new sparsity::TfLiteDimensionMetadata[cn];
284
285 for (size_t i = 0; i < cn; ++i)
286 {
287 tfldm[i].format = to_tflite_sparsity(data[i].format());
288 tfldm[i].dense_size = data[i].dense_size();
289 tfldm[i].array_segments = to_tflite_sparsity(data[i].array_segments());
290 tfldm[i].array_indices = to_tflite_sparsity(data[i].array_indices());
291 }
292
293 return tfldm;
294}
