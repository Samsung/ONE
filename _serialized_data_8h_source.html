33struct OpCode
34{
35 circle::BuiltinOperator opcode;
36 std::string custom_code{""};
37 int32_t version = 1;
38
39 bool operator==(const OpCode &rhs) const
40 {
41 if (opcode == circle::BuiltinOperator_CUSTOM)
42 {
43 return custom_code == rhs.custom_code;
44 }
45 else
46 {
47 return opcode == rhs.opcode;
48 }
49 }
50};

52class CircleExportMetadata
53{
54public:
55 void source_table(const std::map<uint32_t, std::string> &table) { _source_table = table; }
56
57 void add_op_table(uint32_t node_id, uint32_t source_id)
58 {
59 // Model with multiple subgraph may have duplicated node id.
60 // For now, as we do not consider about multiple subgraph in profiling,
61 // just ignore those cases and support them in the future.
62 if (_op_table.find(node_id) == _op_table.end())
63 _op_table.emplace(node_id, std::set<uint32_t>());
64 _op_table.at(node_id).emplace(source_id);
65 }
66
67 void add_execution_plan_table(uint32_t node_id,
68 const std::vector<uint32_t> &execution_plan_inform)
69 {
70 _execution_plan_table[node_id] = execution_plan_inform;
71 }
72
73public:
74 const std::vector<uint8_t> encoded_source_table(void);
75 const std::vector<uint8_t> encoded_op_table(void);
76 const std::vector<uint8_t> encoded_execution_plan_table(void);
77
78private:
79 std::map<uint32_t, std::string> _source_table;
80 std::map<uint32_t, std::set<uint32_t>> _op_table;
81 // _exec_plan_table stores for node with node_id order of execution, and memory offsets:
82 // first go execution order, then memory offsets for node output tensors.
83 luci::ExecutionPlanTable _execution_plan_table;
84};
