65template <typename T>
66void instance_normalization_nchw(ITensor *input, ITensor *output, ITensor *gamma, ITensor *beta,
67 float epsilon, const Window &window)
68{
70 using ExactTagType =
71 typename wrapper::traits::neon_bitvector_tag_t<T, wrapper::traits::BitWidth::W128>;
72
73 // Clear X/Y dimensions on execution window as we handle the planes manually
74 Window win = window;
75 win.set(Window::DimX, Window::Dimension(0, 1, 1));
76 win.set(Window::DimY, Window::Dimension(0, 1, 1));
77
78 constexpr int window_step_x = 16 / sizeof(T);
79 const unsigned int elements_plane = input->info()->dimension(0) * output->info()->dimension(1);
80 const auto channel_idx =
81 get_data_layout_dimension_index(input->info()->data_layout(), DataLayoutDimension::CHANNEL);
82
83 Iterator input_it(input, win);
84 execute_window_loop(
85 win,
86 [&](const Coordinates &id) {
87 Window win_plane = window;
88 win_plane.set(Window::DimX, Window::Dimension(0, 1, 1));
89 win_plane.set(Window::DimZ, Window::Dimension(id[2], id[2] + 1, 1));
90 win_plane.set(3, Window::Dimension(id[3], id[3] + 1, 1));
91
92 Iterator input_plane_it(input, win_plane);
93 Iterator output_plane_it(output, win_plane);
94
95 auto sum_h_w = static_cast<T>(0.f);
96 auto sum_squares_h_w = static_cast<T>(0.f);
97
98 execute_window_loop(
99 win_plane,
100 [&](const Coordinates &) {
101 const auto input_ptr = reinterpret_cast<const T *>(input_plane_it.ptr());
102
103 auto vec_sum_h_w = wrapper::vdup_n(static_cast<T>(0.f), ExactTagType{});
104 auto vec_sum_squares_h_w = wrapper::vdup_n(static_cast<T>(0.f), ExactTagType{});
105
106 // Compute S elements per iteration
107 int x = window.x().start();
108 for (; x <= (window.x().end() - window_step_x); x += window_step_x)
109 {
110 auto vec_input_val = wrapper::vloadq(input_ptr + x);
111 vec_sum_h_w = wrapper::vadd(vec_sum_h_w, vec_input_val);
112 vec_sum_squares_h_w =
113 wrapper::vadd(vec_sum_squares_h_w, wrapper::vmul(vec_input_val, vec_input_val));
114 }
115
116 auto vec2_sum_h_w =
117 wrapper::vpadd(wrapper::vgethigh(vec_sum_h_w), wrapper::vgetlow(vec_sum_h_w));
118 auto vec2_sum_squares_h_w = wrapper::vpadd(wrapper::vgethigh(vec_sum_squares_h_w),
119 wrapper::vgetlow(vec_sum_squares_h_w));
120 for (int i = 0; i < window_step_x / 4; ++i)
121 {
122 vec2_sum_h_w = wrapper::vpadd(vec2_sum_h_w, vec2_sum_h_w);
123 vec2_sum_squares_h_w = wrapper::vpadd(vec2_sum_squares_h_w, vec2_sum_squares_h_w);
124 }
125 sum_h_w += wrapper::vgetlane(vec2_sum_h_w, 0);
126 sum_squares_h_w += wrapper::vgetlane(vec2_sum_squares_h_w, 0);
127
128 // Compute left-over elements
129 for (; x < window.x().end(); ++x)
130 {
131 const auto value = *(input_ptr + x);
132 sum_h_w += value;
133 sum_squares_h_w += value * value;
134 }
135 },
136 input_plane_it, output_plane_it);
137
138 const auto mean_h_w = sum_h_w / elements_plane;
139 const auto var_h_w = sum_squares_h_w / elements_plane - mean_h_w * mean_h_w;
140
141 auto gamma_val = 1.0f;
142 if (gamma != nullptr)
143 {
144 gamma_val = *reinterpret_cast<T *>(gamma->ptr_to_element({id[channel_idx]}));
145 }
146 const auto multip_h_w = gamma_val / std::sqrt(var_h_w + epsilon);
147 const auto vec_mean_h_w = wrapper::vdup_n(static_cast<T>(mean_h_w), ExactTagType{});
148 const auto vec_multip_h_w = wrapper::vdup_n(static_cast<T>(multip_h_w), ExactTagType{});
149 auto beta_val = 0.0f;
150 if (beta != nullptr)
151 {
152 beta_val = *reinterpret_cast<T *>(beta->ptr_to_element({id[channel_idx]}));
153 }
154 const auto vec_beta = wrapper::vdup_n(static_cast<T>(beta_val), ExactTagType{});
155
156 execute_window_loop(
157 win_plane,
158 [&](const Coordinates &) {
159 auto input_ptr = reinterpret_cast<T *>(input_plane_it.ptr());
160 auto output_ptr = reinterpret_cast<T *>(output_plane_it.ptr());
161
162 // Compute S elements per iteration
163 int x = window.x().start();
164 auto vec_val = wrapper::vdup_n(static_cast<T>(0.0f), ExactTagType{});
165 for (; x <= (window.x().end() - window_step_x); x += window_step_x)
166 {
167 vec_val = wrapper::vloadq(input_ptr + x);
168 vec_val = wrapper::vadd(
169 wrapper::vmul(wrapper::vsub(vec_val, vec_mean_h_w), vec_multip_h_w), vec_beta);
170 wrapper::vstore(output_ptr + x, vec_val);
171 }
172
173 // Compute left-over elements
174 for (; x < window.x().end(); ++x)
175 {
176 *(output_ptr + x) = ((*(input_ptr + x)) - mean_h_w) * multip_h_w + beta_val;
177 }
178 },
179 input_plane_it, output_plane_it);
180 },
181 input_it);
182}

184Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output,
185 const ITensorInfo *gamma, const ITensorInfo *beta, float epsilon)
186{
187 ARM_COMPUTE_RETURN_ERROR_ON_CPU_F16_UNSUPPORTED(input);
188 ARM_COMPUTE_RETURN_ERROR_ON_MSG(epsilon == 0.f, "Epsilon must be different than 0");
189
190 ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_NOT_IN(input, DataType::F16, DataType::F32);
191 ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->data_layout() == DataLayout::NHWC,
192 "NHWC data layout is not supported by the kernel directly");
193
194 if (output != nullptr && output->total_size() != 0)
195 {
196 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
197 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
198 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
199 ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->num_channels() != output->num_channels(),
200 "Input and output have different number of channels");
201 }
202
203 if (gamma != nullptr)
204 {
205 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, gamma);
206 ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->dimension(get_data_layout_dimension_index(
207 input->data_layout(), DataLayoutDimension::CHANNEL)) !=
208 gamma->dimension(0),
209 "Gamma's size must be the same as size of input's channel");
210 }
211
212 if (beta != nullptr)
213 {
214 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, beta);
215 ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->dimension(get_data_layout_dimension_index(
216 input->data_layout(), DataLayoutDimension::CHANNEL)) !=
217 beta->dimension(0),
218 "Beta's size must be the same as size of input's channel");
219 }
220
221 return Status{};
222}

224std::tuple<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output)
225{
226 // We handle the planes manually
227 Window win = calculate_max_window(*input, Steps(1));
228
229 // Output auto initialization if not yet initialized
230 auto_init_if_empty(*output, input->tensor_shape(), 1, input->data_type());
231
232 // NEInstanceNormalizationLayerKernelEx doesn't need padding so update_window_and_padding() can be
233 // skipped
234 Coordinates coord;
235 coord.set_num_dimensions(output->num_dimensions());
236 output->set_valid_region(ValidRegion(coord, output->tensor_shape()));
237 return std::make_pair(Status{}, win);
238}

247void NEInstanceNormalizationLayerKernelEx::configure(ITensor *input, ITensor *output,
248 ITensor *gamma, ITensor *beta, float epsilon)
249{
250 ARM_COMPUTE_ERROR_ON_NULLPTR(input);
251
252 _input = input;
253 _output = output == nullptr ? input : output;
254 _gamma = gamma;
255 _beta = beta;
256 _epsilon = epsilon;
257
258 ARM_COMPUTE_ERROR_THROW_ON(
259 validate_arguments(_input->info(), _output->info(), gamma->info(), beta->info(), epsilon));
260
261 if (_input->info()->data_type() == DataType::F32)
262 {
263 _func = &instance_normalization_nchw<float>;
264 }
265#ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
266 else if (_input->info()->data_type() == DataType::F16)
267 {
268 _func = &instance_normalization_nchw<float16_t>;
269 }
270#endif // __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
271 else
272 {
273 ARM_COMPUTE_ERROR("Unsupported data type");
274 }
275
276 // Configure kernel window
277 auto win_config = validate_and_configure_window(_input->info(), _output->info());
278 ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
279
280 INEKernel::configure(std::get<1>(win_config));
281}

286 const ITensorInfo *beta, float epsilon)

288 ARM_COMPUTE_RETURN_ON_ERROR(validate_arguments(input, output, gamma, beta, epsilon));
