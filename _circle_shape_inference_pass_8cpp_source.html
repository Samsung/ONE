28bool is_same_shape(luci::CircleNode *node, loco::TensorShape shape)
29{
30 if (node->shape_status() != luci::ShapeStatus::VALID)
31 return false;
32
33 if (node->rank() != shape.rank())
34 return false;
35
36 for (uint32_t i = 0; i < node->rank(); ++i)
37 {
38 if (node->dim(i).known() != shape.dim(i).known())
39 return false;
40
41 if (node->dim(i).value() != shape.dim(i).value())
42 return false;
43 }
44
45 return true;
46}

66bool CircleShapeInferencePass::run(loco::Graph *g)
67{
68 luci::sinf::Rule shape_infer_rule;
69 bool changed = false;
70
71 for (auto node : inference_candidates(g))
72 {
73 loco::TensorShape shape;
74 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
75
76 if (shape_infer_rule.infer(circle_node, shape) && !is_same_shape(circle_node, shape))
77 {
78 circle_node->rank(shape.rank());
79 for (uint32_t i = 0; i < shape.rank(); ++i)
80 circle_node->dim(i) = shape.dim(i);
81
82 circle_node->shape_status(luci::ShapeStatus::VALID);
83
84 changed = true;
85 }
86 }
87
88 return changed;
89}
