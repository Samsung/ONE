30// Return true if all values of node are within value_range
31// value_range: [min, max]
32template <loco::DataType DT>
33bool value_range(const luci::CircleConst *node, const std::pair<int64_t, int64_t> &value_range)
34{
35 const auto min = value_range.first;
36 const auto max = value_range.second;
37
38 auto size = node->size<DT>();
39 for (uint32_t i = 0; i < size; i++)
40 {
41 const auto val = static_cast<int64_t>(node->at<DT>(i));
42 if (val < min or val > max)
43 return false;
44 }
45
46 return true;
47}

87class OnnxDequantizeLinearPattern final
88{
89public:
90 OnnxDequantizeLinearPattern(luci::CircleCustomOut *candidate) { custom_out = candidate; }
91
92public:
93 bool matched()
94 {
95 if (not custom_out)
96 return false;
97
98 dequantize = loco::must_cast<luci::CircleCustom *>(custom_out->input());
99 if (not is_onnx_dequantize_linear(dequantize))
100 return false;
101
102 input = dynamic_cast<luci::CircleConst *>(dequantize->inputs(0));
103 if (not input)
104 return false;
105
106 scale = dynamic_cast<luci::CircleConst *>(dequantize->inputs(1));
107 if (not scale)
108 return false;
109
110 zerop = dynamic_cast<luci::CircleConst *>(dequantize->inputs(2));
111 if (not zerop)
112 return false;
113
114 const auto input_dtype = input->dtype();
115 const auto scale_dtype = scale->dtype();
116 const auto zerop_dtype = zerop->dtype();
117
118 if (scale_dtype != loco::DataType::FLOAT32)
119 return false;
120
121 // Invariant from onnx DequantizeLinear operator
122 if (input_dtype != zerop_dtype)
123 return false;
124
125 return true;
126 }
127
128public:
129 luci::CircleCustomOut *custom_out = nullptr;
130 luci::CircleCustom *dequantize = nullptr;
131 luci::CircleConst *input = nullptr;
132 luci::CircleConst *scale = nullptr;
133 luci::CircleConst *zerop = nullptr;
134};

136class QuantizeOnnxDequantizeLinear final
137{
138public:
139 QuantizeOnnxDequantizeLinear(const OnnxDequantizeLinearPattern &p) : _p(p) {}
140
141public:
142 void apply(void)
143 {
144 // The final const's dtype is the same with input_dtype by default
145 auto const_dtype = _p.input->dtype();
146 if (const_dtype == loco::DataType::U8)
147 {
148 // Onnx does not support int4/uint4 as of writing. We assume uint8
149 // tensor is quantized in int4/uint4 if values are within [0,15]
150 if (value_range<loco::DataType::U8>(_p.input, {0, 15}))
151 {
152 if (value_range<loco::DataType::U8>(_p.zerop, {8, 8}))
153 {
154 const_dtype = loco::DataType::S4;
155 }
156 else if (value_range<loco::DataType::U8>(_p.zerop, {0, 15}))
157 {
158 const_dtype = loco::DataType::U4;
159 }
160 }
161 }
162
163 luci::CircleConst *quant_const = nullptr;
164 switch (const_dtype)
165 {
166 case loco::DataType::S4:
167 quant_const = gen_s4_quant();
168 break;
169 case loco::DataType::U4:
170 quant_const = gen_u4_quant();
171 break;
172 case loco::DataType::U8:
173 quant_const = gen_u8_quant();
174 break;
175 case loco::DataType::S16:
176 quant_const = gen_s16_quant();
177 break;
178 default:
179 throw std::runtime_error("Unsupported quantized dtype");
180 }
181
182 assert(quant_const); // FIX_ME_UNLESS
183
184 // set origin
185 std::vector<std::shared_ptr<luci::CircleNodeOrigin>> origin_vec{
186 luci::get_origin(_p.dequantize), luci::get_origin(_p.input), luci::get_origin(_p.scale),
187 luci::get_origin(_p.zerop)};
188
189 luci::add_origin(quant_const, luci::composite_origin(origin_vec));
190
191 replace(_p.custom_out).with(quant_const);
192 }
193
194private:
195 luci::CircleConst *gen_s4_quant(void)
196 {
197 assert(_p.input->dtype() == loco::DataType::U8); // FIX_CALLER_UNLESS
198 assert(_p.scale->dtype() == loco::DataType::FLOAT32); // FIX_CALLER_UNLESS
199 assert(_p.zerop->dtype() == loco::DataType::U8); // FIX_CALLER_UNLESS
200
201 auto quantized_node = _p.dequantize->graph()->nodes()->create<luci::CircleConst>();
202 quantized_node->dtype(loco::DataType::S4);
203 quantized_node->rank(_p.input->rank());
204 for (uint32_t i = 0; i < _p.input->rank(); ++i)
205 {
206 quantized_node->dim(i) = _p.input->dim(i);
207 }
208 quantized_node->shape_status(luci::ShapeStatus::VALID);
209
210 // Create S4 CircleConst
211 // NOTE S4 is saved as S8 in luci::CircleConst
212 const auto num_elems = _p.input->size<loco::DataType::U8>();
213 quantized_node->size<loco::DataType::S4>(num_elems);
214 for (uint32_t i = 0; i < num_elems; i++)
215 {
216 const uint8_t u8_val = _p.input->at<loco::DataType::U8>(i);
217 assert(u8_val <= 15); // FIX_CALLER_UNLESS
218 quantized_node->at<loco::DataType::S4>(i) = static_cast<int8_t>(u8_val) - 8;
219 }
220
221 auto qparam = std::make_unique<luci::CircleQuantParam>();
222 {
223 const std::vector<float> scale_vector = get_scales(_p.scale);
224 const std::vector<int64_t> zerop_vector = get_zerops<loco::DataType::U8>(_p.zerop);
225
226 if (scale_vector.size() != zerop_vector.size())
227 throw std::runtime_error("Scale/Zerop size mismatches in " + _p.dequantize->name());
228
229 const int32_t qdim = get_axis(_p.dequantize);
230
231 qparam->scale = scale_vector;
232 qparam->zerop = zerop_vector;
233 qparam->quantized_dimension = qdim;
234 }
235
236 quantized_node->quantparam(std::move(qparam));
237
238 quantized_node->name(_p.input->name());
239
240 return quantized_node;
241 }

410bool QuantizeOnnxDequantizeLinearPass::run(loco::Graph *g)
411{
412 bool changed = false;
413
414 for (auto node : loco::active_nodes(loco::output_nodes(g)))
415 {
416 if (auto circle_custom_out = dynamic_cast<luci::CircleCustomOut *>(node))
417 {
418 OnnxDequantizeLinearPattern p(circle_custom_out);
419 if (p.matched())
420 {
421 QuantizeOnnxDequantizeLinear quantize(p);
422 quantize.apply();
423 changed = true;
424 }
425 }
426 }
427
428 return changed;
429}
