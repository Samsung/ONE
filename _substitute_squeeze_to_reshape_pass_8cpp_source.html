81luci::CircleConst *create_shape_const(loco::Graph *graph, const std::vector<uint32_t> &new_shape)
82{
83 // NOTE dim_size can be 0
84 uint32_t dim_size = static_cast<uint32_t>(new_shape.size());
85
86 auto shape_const = graph->nodes()->create<luci::CircleConst>();
87
88 // const shape/dtype
89 shape_const->dtype(loco::DataType::S32);
90 if (dim_size > 0)
91 {
92 shape_const->rank(1);
93 shape_const->dim(0).set(dim_size);
94 }
95 else
96 shape_const->rank(0);
97 shape_const->shape_status(luci::ShapeStatus::VALID);
98
99 // constant values
100 shape_const->size<loco::DataType::S32>(dim_size);
101 for (uint32_t i = 0; i < dim_size; ++i)
102 shape_const->at<loco::DataType::S32>(i) = new_shape.at(i);
103
104 return shape_const;
105}

107bool substitute_squeeze_to_reshape(luci::CircleSqueeze *squeeze)
108{
109 assert(squeeze != nullptr);
110
111 auto input = loco::must_cast<luci::CircleNode *>(squeeze->input());
112 // we need input node shape and all dim should be known
113 if (input->shape_status() != luci::ShapeStatus::VALID)
114 return false;
115 if (not can_squeeze_shape(input))
116 return false;
117
118 // we will use squeeze shape for new shape
119 if (squeeze->shape_status() != luci::ShapeStatus::VALID)
120 return false;
121
122 auto squeeze_dims = squeeze->squeeze_dims();
123 if (not is_valid_input(input, squeeze_dims))
124 throw std::runtime_error("Invalid values in squeeze_dims: " + squeeze->name());
125
126 auto name = squeeze->name();
127 assert(name.length() > 0);
128
129 auto reshape_shape = node_shape(squeeze);
130 auto graph = squeeze->graph();
131 auto reshape = graph->nodes()->create<luci::CircleReshape>();
132 auto shape_const = create_shape_const(graph, reshape_shape);
133 copy_quantparam(squeeze, reshape);
134 reshape->name(name + "/Reshape");
135 luci::add_origin(reshape, luci::get_origin(squeeze));
136 shape_const->name(name + "/Reshape/shape");
137
138 // graph connection
139 reshape->tensor(input);
140 reshape->shape(shape_const);
141 replace(squeeze).with(reshape);
142
143 return true;
144}

170bool SubstituteSqueezeToReshapePass::run(loco::Graph *g)
171{
172 bool changed = false;
173 for (auto node : loco::active_nodes(loco::output_nodes(g)))
174 {
175 if (auto squeeze = dynamic_cast<luci::CircleSqueeze *>(node))
176 {
177 if (substitute_squeeze_to_reshape(squeeze))
178 changed = true;
179 }
180 }
181 return changed;
182}
