53 void allocate(void)
54 {
55 std::unique_ptr<tflite::gpu::TensorObjectConverterBuilder> converter_builder =
56 NewConverterBuilder(_environment.get());
57 for (const auto &tensor_entry : _tensors)
58 {
59 auto tensor = tensor_entry.second;
60 auto type = tensor->get_type();
61
62 if (type == TensorType::TENSOR_TYPE_DELETE)
63 {
64 continue;
65 }
66
67 const auto &shape = tensor->get_info()._shape;
68 const auto &descriptor = tensor->get_info()._desc;
69
70 if (!CreateTensor(*_context, shape, descriptor, tensor->handle()).ok())
71 {
72 std::runtime_error("Failed to CreateTensor");
73 }
74 switch (type)
75 {
76 case TensorType::TENSOR_TYPE_INPUT:
77 tensor->writeConvertInit(converter_builder.get(), _environment);
78 break;
79 case TensorType::TENSOR_TYPE_OUTPUT:
80 tensor->readConvertInit(converter_builder.get(), _environment);
81 break;
82 default:
83 break;
84 }
85 }
86 }

100 void buildTensor(const ir::OperandIndex &ind, const ir::OperandInfo &info, TensorType type)
101 {
102 auto data_type = DeduceDataTypeFromPrecision(_create_info.precision);
103
104 tflite::gpu::BHWC BHWC_shape = ToBHWC(info.shape());
105
106 tflite::gpu::TensorStorageType storage_type = _create_info.storage_type;
107 tflite::gpu::Layout layout =
108 BHWC_shape.b == 1 ? tflite::gpu::Layout::HWC : tflite::gpu::Layout::BHWC;
109
110 if (!SelectBestStorageType(_environment->device().GetInfo(), BHWC_shape, storage_type,
111 data_type, layout, &storage_type)
112 .ok())
113 {
114 throw std::runtime_error("Failed to SelectBestStorageType");
115 }
116 auto tensor = std::make_shared<operand::CLTensor>(
117 info.shape().rank(), type, BHWC_shape,
118 tflite::gpu::TensorDescriptor{data_type, storage_type, layout});
119 _tensors[ind] = tensor;
120 }

122 ir::OperandIndex addTensor(const ir::Shape &shape)
123 {
124 auto data_type = DeduceDataTypeFromPrecision(_create_info.precision);
125
126 tflite::gpu::BHWC BHWC_shape = ToBHWC(shape);
127
128 tflite::gpu::TensorStorageType storage_type = _create_info.storage_type;
129 tflite::gpu::Layout layout =
130 BHWC_shape.b == 1 ? tflite::gpu::Layout::HWC : tflite::gpu::Layout::BHWC;
131
132 if (!SelectBestStorageType(_environment->device().GetInfo(), BHWC_shape, storage_type,
133 data_type, layout, &storage_type)
134 .ok())
135 {
136 throw std::runtime_error("Failed to SelectBestStorageType");
137 }
138 auto ind = ir::OperandIndex(_new_id--);
139 auto tensor = std::make_shared<operand::CLTensor>(
140 shape.rank(), TensorType::TENSOR_TYPE_VALID, BHWC_shape,
141 tflite::gpu::TensorDescriptor{data_type, storage_type, layout});
142 _tensors[ind] = tensor;
143 return ind;
144 }
145
146 ir::OperandIndexMap<std::shared_ptr<operand::CLTensor>> &tensors(void) { return _tensors; }
147
148private:
149 ir::OperandIndexMap<std::shared_ptr<operand::CLTensor>> _tensors;
150 tflite::gpu::cl::CLContext *_context;
151 tflite::gpu::CreateGpuModelInfo _create_info;
152 std::shared_ptr<tflite::gpu::cl::Environment> _environment;
153 uint32_t _new_id = UINT32_MAX;
154};
