33struct FusePostScale final : public luci::CircleNodeMutableVisitor<bool>
34{
35 bool visit(luci::CircleNode *) { return false; }
36
37 bool visit(luci::CircleDepthwiseConv2D *node)
38 {
39 if (node->fusedActivationFunction() != luci::FusedActFunc::NONE and
40 node->fusedActivationFunction() != luci::FusedActFunc::RELU)
41 return false;
42
43 bool changed = false;
44 for (auto succ : loco::succs(node))
45 {
46 auto post_scale = to_post_scale(succ);
47 if (not post_scale)
48 continue;
49
50 auto param =
51 loco::must_cast<luci::CircleConst *>(post_scale->inputs(1)); // FIX_PostScale_UNLESS
52 auto filter = loco::must_cast<luci::CircleConst *>(node->filter());
53 auto bias = loco::must_cast<luci::CircleConst *>(node->bias());
54
55 uint32_t filter_i = filter->dim(0).value();
56 uint32_t filter_h = filter->dim(1).value();
57 uint32_t filter_w = filter->dim(2).value();
58 uint32_t filter_o = filter->dim(3).value();
59
60 if (filter_o != param->size<loco::DataType::FLOAT32>())
61 {
62 assert(false); // FIX_PostScale_Unless
63 return false;
64 }
65
66 auto cloned_dconv = luci::clone_node(node, node->graph());
67 assert(cloned_dconv != nullptr); // FIX_CALLER_UNLESS
68 auto fused_dconv = loco::must_cast<luci::CircleDepthwiseConv2D *>(cloned_dconv);
69 auto fused_filter = luci::clone(filter);
70 auto fused_bias = luci::clone(bias);
71
72 // Add random string to make unique name
73 fused_dconv->name(node->name() + "_postscale_" + random_str());
74 fused_filter->name(filter->name() + "_postscale_" + random_str());
75 fused_bias->name(bias->name() + "_postscale_" + random_str());
76
77 add_origin(fused_dconv, luci::get_origin(node));
78 add_origin(fused_filter, luci::get_origin(filter));
79 add_origin(fused_bias, luci::get_origin(bias));
80
81 // Multiply param to weights
82 for (uint32_t b = 0; b < filter_i; b++)
83 {
84 for (uint32_t h = 0; h < filter_h; h++)
85 {
86 for (uint32_t w = 0; w < filter_w; w++)
87 {
88 for (uint32_t c = 0; c < filter_o; c++)
89 {
90 uint32_t offset =
91 b * filter_h * filter_w * filter_o + h * filter_w * filter_o + w * filter_o + c;
92 float scale = param->at<loco::DataType::FLOAT32>(c);
93 assert(scale > 0.0); // Defensive guard
94
95 fused_filter->at<loco::DataType::FLOAT32>(offset) =
96 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
97 }
98 }
99 }
100 }
101
102 // Multiply param to bias
103 for (uint32_t c = 0; c < filter_o; ++c)
104 {
105 float scale = param->at<loco::DataType::FLOAT32>(c);
106 fused_bias->at<loco::DataType::FLOAT32>(c) =
107 fused_bias->at<loco::DataType::FLOAT32>(c) * scale;
108 }
109
110 fused_dconv->input(node->input());
111 fused_dconv->filter(fused_filter);
112 fused_dconv->bias(fused_bias);
113
114 loco::replace(post_scale).with(fused_dconv);
115 changed = true;
116 }
117
118 return changed;
119 }

206 bool visit(luci::CircleTransposeConv *node)
207 {
208 bool changed = false;
209
210 for (auto succ : loco::succs(node))
211 {
212 auto post_scale = to_post_scale(succ);
213 if (not post_scale)
214 continue;
215
216 auto param =
217 loco::must_cast<luci::CircleConst *>(post_scale->inputs(1)); // FIX_PostScale_UNLESS
218 auto filter = loco::must_cast<luci::CircleConst *>(node->filter());
219 auto bias = dynamic_cast<luci::CircleConst *>(node->bias());
220 // Non-const bias is not supported
221 if (not bias)
222 {
223 if (nullptr == dynamic_cast<luci::CircleOutputExclude *>(node->bias()))
224 continue;
225 }
226
227 uint32_t filter_o = filter->dim(0).value();
228 uint32_t filter_h = filter->dim(1).value();
229 uint32_t filter_w = filter->dim(2).value();
230 uint32_t filter_i = filter->dim(3).value();
231
232 if (filter_o != param->size<loco::DataType::FLOAT32>())
233 {
234 assert(false); // FIX_PostScale_Unless
235 return false;
236 }
237
238 auto cloned_tconv = luci::clone_node(node, node->graph());
239 assert(cloned_tconv != nullptr); // FIX_CALLER_UNLESS
240 auto fused_tconv = loco::must_cast<luci::CircleTransposeConv *>(cloned_tconv);
241 auto fused_filter = luci::clone(filter);
242
243 fused_tconv->name(node->name() + "_postscale_" + random_str());
244 fused_filter->name(filter->name() + "_postscale_" + random_str());
245
246 add_origin(fused_tconv, luci::get_origin(node));
247 add_origin(fused_filter, luci::get_origin(filter));
248
249 // Multiply param to weights
250 for (uint32_t c = 0; c < filter_o; c++)
251 {
252 float scale = param->at<loco::DataType::FLOAT32>(c);
253 assert(scale > 0.0); // Defensive guard
254
255 for (uint32_t h = 0; h < filter_h; h++)
256 {
257 for (uint32_t w = 0; w < filter_w; w++)
258 {
259 for (uint32_t b = 0; b < filter_i; b++)
260 {
261 uint32_t offset =
262 c * filter_h * filter_w * filter_i + h * filter_w * filter_i + w * filter_i + b;
263
264 fused_filter->at<loco::DataType::FLOAT32>(offset) =
265 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
266 }
267 }
268 }
269 }
270
271 if (bias)
272 {
273 auto fused_bias = luci::clone(bias);
274 fused_bias->name(bias->name() + "_postscale_" + random_str());
275 add_origin(fused_bias, luci::get_origin(bias));
276
277 // Multiply param to bias
278 for (uint32_t c = 0; c < filter_o; ++c)
279 {
280 float scale = param->at<loco::DataType::FLOAT32>(c);
281 fused_bias->at<loco::DataType::FLOAT32>(c) =
282 fused_bias->at<loco::DataType::FLOAT32>(c) * scale;
283 }
284 fused_tconv->bias(fused_bias);
285 }
286
287 fused_tconv->outBackprop(node->outBackprop());
288 fused_tconv->filter(fused_filter);
289 fused_tconv->inputSizes(node->inputSizes());
290
291 loco::replace(post_scale).with(fused_tconv);
292 changed = true;
293 }
294
295 return changed;
296 }

304bool FusePostScalePass::run(loco::Graph *g)
305{
306 bool changed = false;
307 for (auto node : loco::active_nodes(loco::output_nodes(g)))
308 {
309 FusePostScale fps;
310 auto cnode = loco::must_cast<luci::CircleNode *>(node);
311 if (cnode->accept(&fps))
312 changed = true;
313 }
314 return changed;
315}
