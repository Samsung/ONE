37 bool visit(luci::CircleConv2D *node)
38 {
39 if (node->fusedActivationFunction() != luci::FusedActFunc::NONE and
40 node->fusedActivationFunction() != luci::FusedActFunc::RELU)
41 return false;
42
43 luci::CircleCustom *post_scale = nullptr;
44 bool multi_custom = false;
45 auto succs = loco::succs(node);
46 for (const auto succ : succs)
47 {
48 post_scale = to_scale(succ);
49 if (multi_custom && post_scale)
50 {
51 throw std::runtime_error("Do not allow multiple scales.");
52 }
53 if (post_scale)
54 {
55 multi_custom = true;
56 }
57 }
58 if (not post_scale)
59 return false;
60
61 auto param =
62 loco::must_cast<luci::CircleConst *>(post_scale->inputs(1)); // FIX_PostScale_UNLESS
63 auto filter = loco::must_cast<luci::CircleConst *>(node->filter());
64 auto bias = loco::must_cast<luci::CircleConst *>(node->bias());
65
66 uint32_t filter_o = filter->dim(0).value();
67 uint32_t filter_h = filter->dim(1).value();
68 uint32_t filter_w = filter->dim(2).value();
69 uint32_t filter_i = filter->dim(3).value();
70
71 if (filter_o != param->size<loco::DataType::FLOAT32>())
72 {
73 throw std::runtime_error(
74 "Mismatch between scale size and filter output channel size: " + std::to_string(filter_o) +
75 " != " + std::to_string(param->size<loco::DataType::FLOAT32>()));
76 }
77
78 auto cloned_conv = luci::clone_node(node, node->graph());
79 assert(cloned_conv != nullptr); // FIX_CALLER_UNLESS
80 auto fused_conv = loco::must_cast<luci::CircleConv2D *>(cloned_conv);
81 auto fused_filter = luci::clone(filter);
82 auto fused_bias = luci::clone(bias);
83
84 fused_conv->name(node->name() + "_fused_" + random_str());
85 fused_filter->name(filter->name() + "_fused_" + random_str());
86 fused_bias->name(bias->name() + "_fused_" + random_str());
87
88 add_origin(fused_conv, luci::get_origin(node));
89 add_origin(fused_filter, luci::get_origin(filter));
90 add_origin(fused_bias, luci::get_origin(bias));
91
92 // Multiply param to weights
93 for (uint32_t c = 0; c < filter_o; c++)
94 {
95 float scale = param->at<loco::DataType::FLOAT32>(c);
96 assert(scale > 0.0); // Defensive guard
97
98 for (uint32_t h = 0; h < filter_h; h++)
99 {
100 for (uint32_t w = 0; w < filter_w; w++)
101 {
102 for (uint32_t b = 0; b < filter_i; b++)
103 {
104 uint32_t offset =
105 c * filter_h * filter_w * filter_i + h * filter_w * filter_i + w * filter_i + b;
106
107 fused_filter->at<loco::DataType::FLOAT32>(offset) =
108 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
109 }
110 }
111 }
112 }
113
114 // Multiply param to bias
115 for (uint32_t c = 0; c < filter_o; ++c)
116 {
117 float scale = param->at<loco::DataType::FLOAT32>(c);
118 fused_bias->at<loco::DataType::FLOAT32>(c) =
119 fused_bias->at<loco::DataType::FLOAT32>(c) * scale;
120 }
121
122 fused_conv->input(node->input());
123 fused_conv->filter(fused_filter);
124 fused_conv->bias(fused_bias);
125
126 loco::replace(post_scale).with(fused_conv);
127
128 return true;
129 }

234 {

248 auto param =
249 loco::must_cast<luci::CircleConst *>(post_scale->inputs(1)); // FIX_PostScale_UNLESS
250 auto filter = loco::must_cast<luci::CircleConst *>(node->weights());
251 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(node->bias());
252
253 uint32_t filter_o = filter->dim(0).value();
254 uint32_t filter_i = filter->dim(1).value();
255
256 const auto param_size = param->size<loco::DataType::FLOAT32>();
257 if (filter_o != param_size)
258 {
259 throw std::runtime_error("Mismatch between scale size and filter output channel size: " +
260 std::to_string(filter_o) + " != " + std::to_string(param_size));
261 }
262 if (bias)
263 {
264 const auto bias_size = bias->size<loco::DataType::FLOAT32>();
265 if (bias_size != param_size)
266 {
267 throw std::runtime_error("Mismatch between scale size and bias size: " +
268 std::to_string(bias_size) + " != " + std::to_string(param_size));
269 }
270 }
271
272 auto cloned_fc = luci::clone_node(node, node->graph());
273 assert(cloned_fc != nullptr); // FIX_CALLER_UNLESS
274 auto fused_fc = loco::must_cast<luci::CircleFullyConnected *>(cloned_fc);
275 auto fused_filter = luci::clone(filter);
276
277 fused_fc->name(node->name() + "_fused_" + random_str());
278 fused_filter->name(filter->name() + "_fused_" + random_str());
279
280 add_origin(fused_fc, luci::get_origin(node));
281 add_origin(fused_filter, luci::get_origin(filter));
282
283 luci::CircleConst *fused_bias = nullptr;
284 if (bias)
285 {
286 fused_bias = luci::clone(bias);
287 fused_bias->name(bias->name() + "_fused_" + random_str());
288 add_origin(fused_bias, luci::get_origin(bias));
289 }
290
291 // Multiply param to weights
292 for (uint32_t o = 0; o < filter_o; o++)
293 {
294 float scale = param->at<loco::DataType::FLOAT32>(o);
295 assert(scale > 0.0); // Defensive guard
296 for (uint32_t i = 0; i < filter_i; i++)
297 {
298 uint32_t offset = o * filter_i + i;
299
300 fused_filter->at<loco::DataType::FLOAT32>(offset) =
301 fused_filter->at<loco::DataType::FLOAT32>(offset) * scale;
302 }
303 }
304
305 fused_fc->input(node->input());
306 fused_fc->weights(fused_filter);
307 fused_fc->bias(node->bias());
308
309 if (bias)
310 {
311 // Multiply param to bias
312 for (uint32_t c = 0; c < filter_o; ++c)
313 {
314 float scale = param->at<loco::DataType::FLOAT32>(c);
315 fused_bias->at<loco::DataType::FLOAT32>(c) =
316 fused_bias->at<loco::DataType::FLOAT32>(c) * scale;
317 }
318 fused_fc->bias(fused_bias);
319 }
320
321 loco::replace(post_scale).with(fused_fc);
322
323 return true;
324 }

353 auto beta = loco::must_cast<luci::CircleConst *>(node->beta());
354

356 const auto beta_size = beta->size<loco::DataType::FLOAT32>();
357 assert(gamma_size == beta_size);
358 const auto param_size = param->size<loco::DataType::FLOAT32>();
359 if (gamma_size != param_size)
360 {
361 throw std::runtime_error("Mismatch between scale size and gamma size: " +
362 std::to_string(gamma_size) + " != " + std::to_string(param_size));
363 }
364
365 auto cloned_instnorm = luci::clone_node(node, node->graph());
366 assert(cloned_instnorm != nullptr); // FIX_CALLER_UNLESS
367 auto fused_instnorm = loco::must_cast<luci::CircleInstanceNorm *>(cloned_instnorm);
368 auto fused_gamma = luci::clone(gamma);
369 auto fused_beta = luci::clone(beta);
370
371 fused_instnorm->name(node->name() + "_fused_" + random_str());
372 fused_gamma->name(gamma->name() + "_fused_" + random_str());
373 fused_beta->name(beta->name() + "_fused_" + random_str());
374
375 add_origin(fused_instnorm, luci::get_origin(node));
376 add_origin(fused_gamma, luci::get_origin(gamma));
377 add_origin(fused_beta, luci::get_origin(beta));
378
379 // Multiply param to gamma and beta
380 for (uint32_t idx = 0; idx < gamma_size; idx++)
381 {
382 float scale = param->at<loco::DataType::FLOAT32>(idx);
383 assert(scale > 0.0); // Defensive guard
384
385 fused_gamma->at<loco::DataType::FLOAT32>(idx) =
386 fused_gamma->at<loco::DataType::FLOAT32>(idx) * scale;
387 fused_beta->at<loco::DataType::FLOAT32>(idx) =
388 fused_beta->at<loco::DataType::FLOAT32>(idx) * scale;
389 }
390
391 fused_instnorm->input(node->input());
392 fused_instnorm->gamma(fused_gamma);
393 fused_instnorm->beta(fused_beta);
394
395 loco::replace(post_scale).with(fused_instnorm);
396
397 return true;

507bool FusePostScalePass::run(loco::Graph *g)
508{
509 bool changed = false;
510 for (auto node : loco::active_nodes(loco::output_nodes(g)))
511 {
512 FusePostScale fps;
513 auto cnode = loco::must_cast<luci::CircleNode *>(node);
514 if (cnode->accept(&fps))
515 changed = true;
516 }
517 return changed;
