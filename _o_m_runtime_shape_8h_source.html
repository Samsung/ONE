30class OMRuntimeShape
31{
32 static const size_t kMaxDimsCount = 6;
33
34private:
35 size_t _size = 0;
36 std::array<int32_t, kMaxDimsCount> _dims = {0};
37 bool _is_scalar = false;
38
39public:
40 OMRuntimeShape() = default;
41
42 // clang-format off
43
44 OMRuntimeShape(const OMRuntimeShape &other)
45 : _size(other._size)
46 , _dims(other._dims)
47 {}
48
49 explicit OMRuntimeShape(size_t dimensions_count)
50 {
51 resize(dimensions_count);
52 }
53
54 template <size_t DimsCount>
55 explicit OMRuntimeShape(const std::array<int32_t, DimsCount> &source_dims)
56 {
57 resize(DimsCount);
58 std::copy(source_dims.cbegin(), source_dims.cend(), _dims.begin());
59 }

61 // clang-format on
62
63 OMRuntimeShape(const circle::Tensor *tensor)
64 {
65 if (tensor == nullptr)
66 return;
67
68 auto shape = tensor->shape();
69
70 if (shape == nullptr || shape->size() == 0)
71 {
72 _is_scalar = true;
73 _size = 1;
74 _dims[0] = 1;
75
76 return;
77 }
78
79 _size = shape->size();
80 std::copy(shape->cbegin(), shape->cend(), _dims.begin());
81 }
82
83 OMRuntimeShape(size_t new_shape_size, const OMRuntimeShape &shape, int32_t pad_value)
84 {
85 assert(new_shape_size >= shape._size);
86 resize(new_shape_size);
87
88 const size_t size_increase = new_shape_size - shape._size;
89
90 for (auto i = 0u; i < size_increase; ++i)
91 {
92 setDim(i, pad_value);
93 }
94
95 auto from = shape._dims.cbegin();
96 auto to = _dims.begin() + size_increase;
97
98 std::copy(from, from + shape._size, to);
99 }
100
101 OMRuntimeShape(size_t shape_size, int32_t value)
102 {
103 resize(shape_size);
104
105 for (auto i = 0u; i < shape_size; ++i)
106 {
107 setDim(i, value);
108 }
109 }
110
111 static OMRuntimeShape extendedShape(size_t new_shape_size, const OMRuntimeShape &shape)
112 {
113 return OMRuntimeShape(new_shape_size, shape, 1);
114 }
115
116 bool operator==(const OMRuntimeShape &other) const
117 {
118 return _size == other._size && _dims == other._dims;
119 }
120
121 size_t flatSize() const
122 {
123 if (_size == 0)
124 return 0;
125
126 auto it = _dims.cbegin();
127
128 return std::accumulate(it, it + _size, 1u, std::multiplies<size_t>());
129 }
130
131 // clang-format off
132
133 bool isScalar() const noexcept

148 size_t dimensionsCount() const noexcept
149 {
150 return _size;
151 }
152
153 int32_t dims(size_t i) const
154 {
155 assert(i <= _size);
156 return _dims[i];
157 }
158
159 void setDim(size_t i, int32_t val)
160 {
161 assert(i <= _size);
162 _dims[i] = val;
163 }
164
165 void resize(size_t dimensions_count)
166 {
167 assert(dimensions_count <= kMaxDimsCount);
168 _size = dimensions_count;
169 }
170};
