31static constexpr int maxTensorShapeSize = 5;
32
33class OMRuntimeShape
34{
35private:
36 int32_t _size = 0;
37 int32_t _dims[maxTensorShapeSize] = {0};
38
39public:
40 OMRuntimeShape(const OMRuntimeShape &other) : _size(other.dimensionsCount())
41 {
42 std::memcpy(dimsData(), other.dimsData(), sizeof(int32_t) * _size);
43 }
44
45 OMRuntimeShape(const circle::Tensor *tensor)
46 {
47 if (tensor == nullptr)
48 return;
49
50 _size = tensor->shape()->size();
51 std::memcpy(_dims, tensor->shape()->data(), sizeof(int32_t) * _size);
52 }
53
54 // Returns the total count of elements, that is the size when flattened into a
55 // vector.
56 inline int flatSize() const
57 {
58 int buffer_size = 1;
59 const int *dims_data = reinterpret_cast<const int *>(dimsData());
60 for (int i = 0; i < _size; i++)
61 {
62 buffer_size *= dims_data[i];
63 }
64 return buffer_size;
65 }
66
67 inline int32_t *dimsData() { return _dims; }
68 inline const int32_t *dimsData() const { return _dims; }
69
70 OMRuntimeShape() : _size(0) {}
71
72 explicit OMRuntimeShape(int dimensions_count) : _size(dimensions_count)
73 {
74 assert(dimensions_count <= maxTensorShapeSize);
75 assert(dimensions_count >= 0);
76 }
77
78 OMRuntimeShape(int dimensions_count, const int32_t *dims_data) : _size(0)
79 {
80 resize(dimensions_count);
81 int32_t *dst_dims = dimsData();
82 std::memcpy(dst_dims, dims_data, dimensions_count * sizeof(int32_t));
83 }
84
85 OMRuntimeShape(int new_shape_size, const OMRuntimeShape &shape, int pad_value) : _size(0)
86 {
87 resize(new_shape_size);
88 const int size_increase = new_shape_size - shape.dimensionsCount();
89 for (int i = 0; i < size_increase; ++i)
90 {
91 setDim(i, pad_value);
92 }
93 std::memcpy(dimsData() + size_increase, shape.dimsData(),
94 sizeof(int32_t) * shape.dimensionsCount());
95 }
96
97 OMRuntimeShape(int shape_size, int32_t value) : _size(0)
98 {
99 resize(shape_size);
100 for (int i = 0; i < shape_size; ++i)
101 {
102 setDim(i, value);
103 }
104 }
105
106 inline static OMRuntimeShape extendedShape(int new_shape_size, const OMRuntimeShape &shape)
107 {
108 return OMRuntimeShape(new_shape_size, shape, 1);
109 }
110
111 bool operator==(const OMRuntimeShape &comp) const
112 {
113 return this->_size == comp._size &&
114 std::memcmp(dimsData(), comp.dimsData(), _size * sizeof(int32_t)) == 0;
115 }
116
117 inline int32_t dimensionsCount() const { return _size; }
118
119 inline int32_t dims(int i) const
120 {
121 assert(i <= _size);
122 assert(i >= 0);
123 return _dims[i];
124 }
125
126 inline void setDim(int i, int32_t val)
127 {
128 assert(i <= _size);
129 assert(i >= 0);
130 _dims[i] = val;
131 }
132
133 inline void resize(int dimensions_count)
134 {
135 assert(dimensions_count <= maxTensorShapeSize);
136 assert(dimensions_count >= 0);
137 _size = dimensions_count;
138 }
139};
