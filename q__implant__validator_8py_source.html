17import h5py as h5
18import numpy as np
19import json
20
21
22def validate(h5_path, qparam_dir, qparam_json):
23 valid = True
24 with open(qparam_json, "r") as qparams:
25 json_load = json.load(qparams)
26 with h5.File(h5_path, "r") as model:
27 for node_name in model.keys():
28 # not quantized node exists (reshape, pad...)
29 if not json_load.get(node_name):
30 continue
31
32 for tensor_name in json_load[node_name]:
33 np_path = f"{qparam_dir}/{json_load[node_name][tensor_name]}"
34 if tensor_name == "value":
35 expected_weights = np.load(np_path)
36 h5_weights = model[node_name]["weights"][:]
37 if np.allclose(
38 h5_weights, expected_weights, rtol=1.e-5,
39 atol=1.e-5) == False:
40 print("Implanted weights of " + node_name + "." + tensor_name +
41 " (" + str(h5_weights) +
42 ") do not match with expected value (" +
43 str(expected_weights) + ").")
44 valid = False
45
46 if tensor_name == "scale":
47 expected_scale = np.load(np_path)
48 h5_scale = model[node_name]["scale"][:]
49 if np.allclose(
50 h5_scale, expected_scale, rtol=1.e-5, atol=1.e-5) == False:
51 print("Implanted scale of " + node_name + "." + tensor_name +
52 " (" + str(h5_scale) +
53 ") do not match with expected value (" +
54 str(expected_scale) + ").")
55 valid = False
56
57 if tensor_name == "zerop":
58 expected_zerop = np.load(np_path)
59 input_zerop = model[node_name]["zero_point"][:]
60 if np.allclose(input_zerop, expected_zerop, rtol=0, atol=1) == False:
61 print("Implanted zero point of " + tensor_name + " (" +
62 str(input_zerop) + ") do not match with expected value (" +
63 str(expected_zerop) + ").")
64 valid = False
65
66 return valid
