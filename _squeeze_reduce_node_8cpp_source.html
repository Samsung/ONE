50template <class TFNode> bool squeeze_reduce_node(loco::Graph *graph, TFNode *reduce_node)
51{
52 // Don't need to squeeze reduce node
53 if (reduce_node->keep_dims())
54 return false;
55
56 // Reduction indices are not yet constant
57 auto const_reduction_indices = dynamic_cast<moco::TFConst *>(reduce_node->reduction_indices());
58 if (const_reduction_indices == nullptr)
59 return false;
60
61 auto squeeze_node = graph->nodes()->create<moco::TFSqueeze>();
62 auto new_reduce_node = graph->nodes()->create<TFNode>();
63
64 new_reduce_node->input(reduce_node->input());
65 new_reduce_node->reduction_indices(reduce_node->reduction_indices());
66 new_reduce_node->keep_dims(true);
67
68 // Insert squeeze dims
69 // TODO Support S64 type
70 assert(const_reduction_indices->dtype() == loco::DataType::S32);
71
72 std::vector<int64_t> reduction_values;
73 for (uint32_t i = 0; i < const_reduction_indices->size<loco::DataType::S32>(); ++i)
74 reduction_values.push_back(const_reduction_indices->at<loco::DataType::S32>(i));
75 squeeze_node->squeeze_dims(reduction_values);
76
77 // replace
78 loco::replace(reduce_node).with(squeeze_node);
79 squeeze_node->input(new_reduce_node);
80
81 return true;
82}
