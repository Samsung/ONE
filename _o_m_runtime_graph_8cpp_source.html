34void *OMRuntimeGraph::getInputDataAt(uint32_t position)
35{
36 const auto input_index = _context.getGraphInputTensorIndex(position);
37
38 uint8_t *data;
39 _storage.getDataByTensorIndex(&data, input_index);
40
41 if (data == nullptr)
42 {
43 _allocator.allocateGraphInputs(&_context, &_storage);
44 _storage.getDataByTensorIndex(&data, input_index);
45 }
46
47 return reinterpret_cast<void *>(data);
48}
