40template <typename T> struct scalar_logistic_op
41{
42 EIGEN_EMPTY_STRUCT_CTOR(scalar_logistic_op)
43 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE T operator()(const T &x) const
44 {
45 const T one = T(1);
46 return one / (one + Eigen::numext::exp(-x));
47 }
48
49 template <typename Packet>
50 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Packet packetOp(const Packet &x) const
51 {
52 const Packet one = Eigen::internal::pset1<Packet>(T(1));
53 return pdiv(one, padd(one, pexp(pnegate(x))));
54 }
55};
