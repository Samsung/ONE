33void StaticDerivativeShapeInferer::infer()
34{
35 // It is not determined to iterate in reverse order.
36 auto sorted_ops = _lowered_subg->graph().topolSortOperations();
37 for (auto it = sorted_ops.rbegin(); it != sorted_ops.rend(); ++it)
38 {
39 const auto op_idx = *it;
40 const auto &op = _lowered_subg->trainable_graph().operation(op_idx);
41 if (checkDynamicInput(op))
42 {
43 std::stringstream msg;
44 msg << "StaticDerivativeShapeInferer does not support dynamic shape yet, ";
45 msg << op.name() << "(op index: " << op_idx << ") has dynamic shape.";
46 throw std::runtime_error(msg.str());
47 }
48 else
49 op.accept(*this);
50 }
51}

58bool StaticDerivativeShapeInferer::checkDynamicInput(const ir::IOperation &op)
59{
60 const auto &operands = _lowered_subg->graph().operands();
61 for (auto input_idx : op.getInputs() | ir::Remove::UNDEFINED | ir::Remove::DUPLICATED)
62 {
63 if (operands.at(input_idx).info().isDynamic())
64 {
65 return true;
66 }
67 }
68
69 return false;
70}

72void StaticDerivativeShapeInferer::setShape(const ir::OperandIndex &index, const ir::Shape &shape)
73{
74 auto &tgraph = _lowered_subg->trainable_graph();
75
76 if (tgraph.derivatives().exist(index))
77 tgraph.changeDerivativeShape(index, shape);
78 else
79 {
80 // NOTE This code assumes the types are always the same, but I'm not sure.
81 const auto &type = tgraph.operands().at(index).typeInfo();
82 tgraph.addDerivative(index, std::make_unique<ir::Operand>(shape, type));
83 }
84}
