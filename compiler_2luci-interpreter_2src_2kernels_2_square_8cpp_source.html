53void Square::evalFloat() const
54{
55 auto in = getTensorData<float>(input());
56 auto out = getTensorData<float>(output());
57 auto size = getTensorShape(input()).FlatSize();
58 for (auto i = in; i != in + size; ++i)
59 {
60 *out = (*i) * (*i);

62 }
