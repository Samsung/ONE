33void expand_rank(loco::TensorShape &x, loco::TensorShape &y)
34{
35 auto x_rank = x.rank();
36 auto y_rank = y.rank();
37
38 if (x_rank == y_rank)
39 return;
40
41 TensorShapeExpander x_exp(x);
42 TensorShapeExpander y_exp(y);
43
44 auto xy_rank = std::max(x_rank, y_rank);
45
46 x = x_rank > y_rank ? x : x_exp.to(xy_rank);
47 y = y_rank > x_rank ? y : y_exp.to(xy_rank);
48}

53loco::TensorShape expand_dimension(const loco::TensorShape &x, const loco::TensorShape &y)
54{
55 assert(x.rank() == y.rank()); // FIX_CALLER_UNLESS
56
57 auto rank = x.rank();
58
59 loco::TensorShape output_shape;
60
61 output_shape.rank(rank);
62
63 // Shape inference rule (commutative)
64 // Dimension values of x, y, and Result are categorized as 1, N, ?
65 // where N is a positive integer greater than 1 and ? is unknown.
66 // x | y | Result
67 // c1. 1 | N | N
68 // c2. 1 | ? | ?
69 // c3. N | N | N
70 // c4. N | ? | N
71 // c5. ? | ? | ?
72 // Throw exception if none of the below conditions are met
73 // e1. x == y
74 // e2. x != y and x == 1
75 // e3. x != y and y == 1
76 // e4. x is unknown
77 // e5. y is unknown
78 for (uint32_t axis = 0; axis < rank; ++axis)
79 {
80 const bool x_is_known = x.dim(axis).known();
81 const bool y_is_known = y.dim(axis).known();
82
83 if (x_is_known and y_is_known)
84 {
85 const auto x_dim = x.dim(axis).value();
86 const auto y_dim = y.dim(axis).value();
87
88 // each dimension of x and y should be same or one must be 1 if different
89 if (!((x_dim == y_dim) || (x_dim == 1 || y_dim == 1)))
90 INTERNAL_EXN("Cannot produce expand_dimension of two shapes");
91
92 // c1, c3
93 output_shape.dim(axis).set(std::max(x_dim, y_dim));
94 }
95 else if (not x_is_known and not y_is_known)
96 {
97 // c5
98 output_shape.dim(axis).unset();
99 }
100 else
101 {
102 const uint32_t known_dim = x_is_known ? x.dim(axis).value() : y.dim(axis).value();
103 if (known_dim == 1)
104 {
105 // c2
106 output_shape.dim(axis).unset();
107 }
108 else
109 {
110 // c4
111 output_shape.dim(axis).set(known_dim);
112 }
113 }
114 }
115
116 return output_shape;
117}

143loco::TensorShape circle_shape(const luci::CircleNode *node)
144{
145 loco::TensorShape shape;
146 shape.rank(node->rank());
147 for (uint32_t r = 0; r < node->rank(); ++r)
148 shape.dim(r) = node->dim(r);
149 return shape;
150}

152loco::TensorShape broadcast_shape(const loco::TensorShape &x, const loco::TensorShape &y)
153{
154 auto x_match = x;
155 auto y_match = y;
156
157 expand_rank(x_match, y_match);
158
159 auto output_shape = expand_dimension(x_match, y_match);
160
161 return output_shape;

164loco::TensorShape pad_shape(const loco::TensorShape &input_shape, const luci::CircleNode *paddings)
165{
166 const loco::DataType S32 = loco::DataType::S32;
167 const loco::DataType S64 = loco::DataType::S64;
168
169 // TODO support other data type
170 LUCI_ASSERT(paddings->dtype() == S32 || paddings->dtype() == S64, "Support int 32/64 for now");
171 if (paddings->rank() != 2)
172 INTERNAL_EXN("paddings should be rank 2");
173
174 int32_t n = paddings->dim(0).value();
175 int32_t v = paddings->dim(1).value();
176
177 if (v != 2)
178 INTERNAL_EXN("paddings should be [n, 2]");
179
180 if (n != int32_t(input_shape.rank()))
181 INTERNAL_EXN("paddings [n, 2] should have same value of input rank");
182
183 loco::TensorShape output_shape;
184
185 output_shape.rank(input_shape.rank());
186
187 auto const_padding = dynamic_cast<const luci::CircleConst *>(paddings);
188 if (const_padding == nullptr)
189 return output_shape;
190
191 for (int32_t ni = 0; ni < n; ++ni)
192 {
193 if (not input_shape.dim(ni).known())
194 {
195 output_shape.dim(ni).unset();
196 continue;
197 }
198 int32_t idx = ni * 2;
199 int value = input_shape.dim(ni).value();
200 if (const_padding->dtype() == S32)
201 {
202 value += const_padding->at<S32>(idx + 0); // left
203 value += const_padding->at<S32>(idx + 1); // right
204 }
205 else
206 {
207 auto pl = const_padding->at<S64>(idx + 0);
208 auto pr = const_padding->at<S64>(idx + 1);
209 auto max = static_cast<int64_t>(std::numeric_limits<int32_t>::max());
210 auto low = static_cast<int64_t>(std::numeric_limits<int32_t>::lowest());
211 LUCI_ASSERT(pl <= max, "paddings is over 32 bit limit");
212 LUCI_ASSERT(pl >= low, "paddings is over 32 bit limit");
213 LUCI_ASSERT(pr <= max, "paddings is over 32 bit limit");
214 LUCI_ASSERT(pr >= low, "paddings is over 32 bit limit");
215 value += static_cast<int32_t>(pl); // left
216 value += static_cast<int32_t>(pr); // right
217 }
218 output_shape.dim(ni) = value;
219 }
220
221 return output_shape;
222}
