22void SimpleMemoryManager::allocate_memory(luci_interpreter::Tensor &tensor)
23{
24 if (!tensor.is_allocatable())
25 {
26 return;
27 }
28 if (tensor.is_data_allocated())
29 {
30 release_memory(tensor);
31 }
32 const auto element_size = getDataTypeSize(tensor.element_type());
33 const auto num_elements = tensor.shape().num_elements();
34
35 auto *data = new uint8_t[num_elements * element_size];
36 tensor.set_data_buffer(data);
37}

39void SimpleMemoryManager::release_memory(luci_interpreter::Tensor &tensor)
40{
41 if (!tensor.is_data_allocated())
42 {
43 tensor.set_data_buffer(nullptr);
44 return;
45 }
46 auto data = tensor.data<uint8_t>();
47 delete[] data;
48 tensor.set_data_buffer(nullptr);
49}
