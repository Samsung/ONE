38template <typename T>
39static void Init(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj, const bool copy,
40 const onert::ir::Layout frontend_layout = onert::ir::Layout::UNKNOWN)
41{
42 const auto shape = model_obj.shape();
43 assert(model_obj.data());
44 obj.access([&](::onert::backend::ITensor &tensor) {
45 switch (shape.rank())
46 {
47 case 0:
48 case 1:
49 case 2:
50 case 3:
51 case 4:
52 if (copy)
53 {
54 tensor.enqueueWriteBuffer(model_obj.data()->base(), true);
55 }
56 else
57 {
58 // NYI
59 (void)frontend_layout;
60 throw std::runtime_error{"Not yet supported"};
61 }
62 break;
63 default:
64 throw std::runtime_error{"Not yet supported"};
65 }
66 });
67}

75template <typename T>
76void permuteInit(const onert::ir::Operand &model_obj, onert::backend::ITensor &obj,
77 const onert::ir::Layout frontend_layout)
78{
79 const bool copy = frontend_layout == obj.layout();

83class ClConstantInitializer : public ir::OperationVisitor
84{
85public:
86 void run()
87 {
88 assert(_tensor_reg);
89 for (const auto &it : _init_map)
90 {
91 const auto &ind = it.first;
92 const auto &fn = it.second;
93
94 const auto &model_obj = _operands.at(ind);
95 auto tensor_obj = _tensor_reg->getNativeITensor(ind);
96 assert(tensor_obj != nullptr);
97 fn(model_obj, *tensor_obj);
98 VERBOSE(FillOperandData) << "Fill data for operand " << ind << std::endl;
99 }
100 _init_map.clear();
101 }
102
103public:
104 ClConstantInitializer(const ir::Operands &operands,
105 const std::shared_ptr<ITensorRegistry> &tensor_reg);

107public:
108 using Initializer = std::function<void(const ir::Operand &, backend::ITensor &)>;
109
110public:
111 void registerDefaultInitializer(const ir::OperandIndex &index, const ir::Operand &obj)
112 {
113 registerPermuteInitializer(index, obj);
114 }

116 void registerPermuteInitializer(const ir::OperandIndex &index, const ir::Operand &obj);
117
118public:

129 std::shared_ptr<ITensorRegistry> _tensor_reg;

131 ir::Layout _current_layout;
132};
