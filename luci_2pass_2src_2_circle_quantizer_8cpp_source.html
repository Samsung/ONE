52template <typename T> T lexical_cast(const std::string &str)
53{
54 std::istringstream ss;
55 ss.str(str);
56 T data;
57 ss >> data;
58 return data;
59}
60
61template <typename T> std::vector<T> lexical_cast(std::vector<std::string> &sv)
62{
63 std::vector<T> result;
64 std::transform(sv.begin(), sv.end(), std::back_inserter(result),
65 [](std::string str) -> T { return lexical_cast<T>(str); });
66 return result;
67}

69class QuantizeOptionsImpl final : public luci::CircleQuantizer::Options

170void CircleQuantizer::quantize(loco::Graph *g) const
171{
172 // Fake quantization of weights
173 if (_options->query(Options::Algorithm::QuantizeDequantizeWeights))
174 {
175 static const std::vector<std::string> fakeq_supported_input_model_dtype{"float32"};
176 static const std::vector<std::string> fakeq_supported_output_model_dtype{"uint8", "int16"};
177 static const std::vector<std::string> fakeq_supported_granularity{"layer", "channel"};
178
179 auto input_model_dtype =
180 _options->param(Options::AlgorithmParameters::Quantize_input_model_dtype);
181 auto output_model_dtype =
182 _options->param(Options::AlgorithmParameters::Quantize_output_model_dtype);
183 auto granularity = _options->param(Options::AlgorithmParameters::Quantize_granularity);
184 auto layer_params = _options->layer_params(Options::AlgorithmParameters::Quantize_layer_params);
185
186 if (!in_array(to_lower_case(input_model_dtype), fakeq_supported_input_model_dtype))
187 throw std::runtime_error("Unsupported input type. List of supported input type: " +
188 to_string(fakeq_supported_input_model_dtype));
189
190 if (!in_array(to_lower_case(output_model_dtype), fakeq_supported_output_model_dtype))
191 throw std::runtime_error("Unsupported output type. List of supported output type: " +
192 to_string(fakeq_supported_output_model_dtype));
193
194 if (!in_array(to_lower_case(granularity), fakeq_supported_granularity))
195 throw std::runtime_error("Unsupported granularity. List of supported granularity: " +
196 to_string(fakeq_supported_granularity));
197
198 if (str_to_granularity(granularity) == QuantizationGranularity::LayerWise &&
199 str_to_dtype(output_model_dtype) != loco::DataType::U8)
200 throw std::runtime_error("Layer-wise quantization only supports uint8 dtype.");
201
202 // Check dtype/granularity of layer params
203 for (auto layer_param : layer_params)
204 {
205 auto name = layer_param->name;
206 if (!in_array(to_lower_case(layer_param->dtype), fakeq_supported_output_model_dtype))
207 {
208 throw std::runtime_error("Unsupported dtype in " + name + ". List of supported dtype: " +
209 to_string(fakeq_supported_output_model_dtype));
210 }
211 if (!in_array(to_lower_case(layer_param->granularity), fakeq_supported_granularity))
212 {
213 throw std::runtime_error(
214 "Unsupported granularity in " + name +
215 ". List of supported granularity: " + to_string(fakeq_supported_granularity));
216 }
217 }
218
219 // Clear existing quantparams before doing fake quantization
220 for (auto node : loco::active_nodes(loco::output_nodes(g)))
221 {
222 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
223 if (circle_node->quantparam() != nullptr)
224 circle_node->quantparam(nullptr);
225 }
226
227 auto ctx = std::make_unique<luci::QuantizeDequantizeWeightsPass::Context>();
228 {
229 ctx->input_model_dtype = str_to_dtype(input_model_dtype);
230 ctx->output_model_dtype = str_to_dtype(output_model_dtype);
231 ctx->granularity = str_to_granularity(granularity);
232
233 for (auto layer_param : layer_params)
234 {
235 LayerInfo info;
236 {
237 info.name = layer_param->name;
238 info.dtype = str_to_dtype(layer_param->dtype);
239 info.granularity = str_to_granularity(layer_param->granularity);
240 }
241 ctx->layers_info.emplace_back(info);
242 }
243 }
244
245 luci::QuantizeDequantizeWeightsPass fake_quantizer(std::move(ctx));
246
247 fake_quantizer.run(g);
248 }
249
250 // Actual quantization of weights, bias, and activation
251 if (_options->query(Options::Algorithm::QuantizeWithMinMax))
252 {
253 static const std::vector<std::string> qwmm_supported_input_model_dtype{"float32"};
254 static const std::vector<std::string> qwmm_supported_output_model_dtype{"uint8", "int16"};
255 static const std::vector<std::string> qwmm_supported_granularity{"layer", "channel"};
256 static const std::vector<std::string> qwmm_supported_input_type{"uint8", "int16", "float32"};
257 static const std::vector<std::string> qwmm_supported_output_type{"uint8", "int16", "float32"};
258
259 auto input_model_dtype =
260 _options->param(Options::AlgorithmParameters::Quantize_input_model_dtype);
261 auto output_model_dtype =
262 _options->param(Options::AlgorithmParameters::Quantize_output_model_dtype);
263 auto granularity = _options->param(Options::AlgorithmParameters::Quantize_granularity);
264 auto input_type = _options->param(Options::AlgorithmParameters::Quantize_input_type);
265 if (input_type.empty())
266 input_type = output_model_dtype;
267 auto output_type = _options->param(Options::AlgorithmParameters::Quantize_output_type);
268 if (output_type.empty())
269 output_type = output_model_dtype;
270
271 bool TF_style_maxpool =
272 _options->param(Options::AlgorithmParameters::Quantize_TF_style_maxpool) == "True";
273
274 auto layer_params = _options->layer_params(Options::AlgorithmParameters::Quantize_layer_params);
275
276 if (!in_array(to_lower_case(input_model_dtype), qwmm_supported_input_model_dtype))
277 throw std::runtime_error("Unsupported input type. List of supported input types: " +
278 to_string(qwmm_supported_input_model_dtype));
279
280 if (!in_array(to_lower_case(output_model_dtype), qwmm_supported_output_model_dtype))
281 throw std::runtime_error("Unsupported output type. List of supported output types: " +
282 to_string(qwmm_supported_output_model_dtype));
283
284 if (!in_array(to_lower_case(granularity), qwmm_supported_granularity))
285 throw std::runtime_error("Unsupported granularity. List of supported granularity: " +
286 to_string(qwmm_supported_granularity));
287
288 if (!in_array(to_lower_case(input_type), qwmm_supported_input_type))
289 throw std::runtime_error("Unsupported input type. List of supported input types: " +
290 to_string(qwmm_supported_input_type));
291
292 if (!in_array(to_lower_case(output_type), qwmm_supported_output_type))
293 throw std::runtime_error("Unsupported output type. List of supported output types: " +
294 to_string(qwmm_supported_output_type));
295
296 if (str_to_granularity(granularity) == QuantizationGranularity::LayerWise &&
297 str_to_dtype(output_model_dtype) != loco::DataType::U8)
298 throw std::runtime_error("Layer-wise quantization only supports uint8 dtype.");
299
300 // Check dtype/granularity of layer params
301 for (auto layer_param : layer_params)
302 {
303 auto name = layer_param->name;
304 if (!in_array(to_lower_case(layer_param->dtype), qwmm_supported_output_model_dtype))
305 {
306 throw std::runtime_error("Unsupported dtype in " + name + ". List of supported dtype: " +
307 to_string(qwmm_supported_output_model_dtype));
308 }
309 if (!in_array(to_lower_case(layer_param->granularity), qwmm_supported_granularity))
310 {
311 throw std::runtime_error(
312 "Unsupported granularity in " + name +
313 ". List of supported granularity: " + to_string(qwmm_supported_granularity));
314 }
315 }
316
317 // Input model checker for quantization
318 luci::QuantizePreCheckerPass input_model_checker{};
319 input_model_checker.run(g);
320
321 auto ctx = std::make_unique<luci::QuantizeWithMinMaxPass::Context>();
322 {
323 ctx->input_model_dtype = str_to_dtype(input_model_dtype);
324 ctx->output_model_dtype = str_to_dtype(output_model_dtype);
325 ctx->granularity = str_to_granularity(granularity);
326 ctx->input_type = str_to_dtype(input_type);
327 ctx->output_type = str_to_dtype(output_type);
328 ctx->TF_style_maxpool = TF_style_maxpool;
329
330 for (auto layer_param : layer_params)
331 {
332 LayerInfo info;
333 {
334 info.name = layer_param->name;
335 info.dtype = str_to_dtype(layer_param->dtype);
336 info.granularity = str_to_granularity(layer_param->granularity);
337 }
338 ctx->layers_info.emplace_back(info);
339 }
340 }
341
342 luci::QuantizeWithMinMaxPass quantizer(std::move(ctx));
343
344 quantizer.run(g);
345
346 auto verify_ctx = std::make_unique<luci::QuantizedModelVerifier::Context>();
347 {
348 verify_ctx->output_model_dtype = str_to_dtype(output_model_dtype);
349 verify_ctx->granularity = str_to_granularity(granularity);
350 verify_ctx->input_type = str_to_dtype(input_type);
351 verify_ctx->output_type = str_to_dtype(output_type);
352 verify_ctx->TF_style_maxpool = TF_style_maxpool;
353
354 for (auto layer_param : layer_params)
355 {
356 LayerInfo info;
357 {
358 info.name = layer_param->name;
359 info.dtype = str_to_dtype(layer_param->dtype);
360 info.granularity = str_to_granularity(layer_param->granularity);
361 }
362 verify_ctx->layers_info.emplace_back(info);
363 }
364 }
365
366 // Verify the type/granularity of the quantized model
367 luci::QuantizedModelVerifier verifier(std::move(verify_ctx));
368
369 verifier.verify(g);
370 }
371
372 // Requantize
373 if (_options->query(Options::Algorithm::Requantize))
374 {
375 static const std::vector<std::string> rq_supported_input_model_dtype{"int8"};
376 static const std::vector<std::string> rq_supported_output_model_dtype{"uint8"};
377
378 auto input_model_dtype =
379 _options->param(Options::AlgorithmParameters::Quantize_input_model_dtype);
380 auto output_model_dtype =
381 _options->param(Options::AlgorithmParameters::Quantize_output_model_dtype);
382
383 if (!in_array(to_lower_case(input_model_dtype), rq_supported_input_model_dtype))
384 throw std::runtime_error("Unsupported input type. List of supported input types: " +
385 to_string(rq_supported_input_model_dtype));
386
387 if (!in_array(to_lower_case(output_model_dtype), rq_supported_output_model_dtype))
388 throw std::runtime_error("Unsupported output type. List of supported output types: " +
389 to_string(rq_supported_output_model_dtype));
390
391 luci::RequantizePass requantizer(str_to_dtype(input_model_dtype),
392 str_to_dtype(output_model_dtype));
393 requantizer.run(g);
394 }
395
396 // Force to write quantparam to specified tensors
397 // NOTE Only per-tensor (not per-channel) qparam can be written
398 if (_options->query(Options::Algorithm::ForceQuantParam))
399 {
400 ForceQuantParamPass::TensorVector tensors =
401 _options->params(Options::AlgorithmParameters::Quantize_tensor_names);
402 auto str_scales = _options->params(Options::AlgorithmParameters::Quantize_scales);
403 auto str_zero_points = _options->params(Options::AlgorithmParameters::Quantize_zero_points);
404
405 // Cast scales/zero_points to proper types
406 ForceQuantParamPass::ScaleVector scales = lexical_cast<float>(str_scales);
407 ForceQuantParamPass::ZPVector zero_points = lexical_cast<int64_t>(str_zero_points);
408
409 ForceQuantParamPass fq(tensors, scales, zero_points);
410 fq.run(g);
411 }
412
413 // Copy quantparam of a tensor to another tensor
414 if (_options->query(Options::Algorithm::CopyQuantParam))
415 {
416 CopyQuantParamPass::TensorVector src_tensors =
417 _options->params(Options::AlgorithmParameters::Quantize_src_tensor_names);
418 CopyQuantParamPass::TensorVector dst_tensors =
419 _options->params(Options::AlgorithmParameters::Quantize_dst_tensor_names);
420
421 CopyQuantParamPass cq(src_tensors, dst_tensors);
422 cq.run(g);
423 }
424
425 // Convert quantized model to fake-quantized model
426 if (_options->query(Options::Algorithm::ConvertToFakeQuantizedModel))
427 {
428 luci::ConvertToFakeQuantizedModelPass fake_quantizer;
429 fake_quantizer.run(g);
430
431 logo::Phase phase;
432
433 // Default passes
434 phase.emplace_back(std::make_unique<logo::RemoveDeadNodeWithQueryPass>());
435 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
436 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
437
438 // Remove redundant Dequantize Ops generated during fake quantization
439 phase.emplace_back(std::make_unique<luci::RemoveRedundantDequantizePass>());
440 // Fold Dequantize Ops generated during fake quantization
441 phase.emplace_back(std::make_unique<luci::FoldDequantizePass>());
442
443 ProgressReporter prog(g, logo::PhaseStrategy::Restart);
444 logo::PhaseRunner<logo::PhaseStrategy::Restart> phase_runner{g};
445 phase_runner.attach(&prog);
446 phase_runner.run(phase);
447 }
448
449 logo::Phase phase;
450
451 // Do Shape/Type inference
452 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
453 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
454
455 ProgressReporter prog(g, logo::PhaseStrategy::Saturate);
456 logo::PhaseRunner<logo::PhaseStrategy::Saturate> phase_runner{g};
457 phase_runner.attach(&prog);
458 phase_runner.run(phase);
459}
