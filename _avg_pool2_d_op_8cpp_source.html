24void AvgPool2DOp::inferOutputTypes()
25{
26 const auto &input_shape = getInputShape(0);
27 const int batch_dim_index = getDataBatchDimIndex(_attributes.data_format);
28 const int channel_dim_index = getDataChannelDimIndex(_attributes.data_format);
29
30 constexpr int num_spatial_dims = 2;
31
32 assert(input_shape.rank() == 4);
33 assert(_attributes.window.size() == num_spatial_dims);
34 assert(_attributes.strides.size() == num_spatial_dims);
35 assert(_attributes.padding_before.size() == num_spatial_dims);
36 assert(_attributes.padding_after.size() == num_spatial_dims);
37
38 Shape output_shape(4);
39
40 output_shape.dim(batch_dim_index) = input_shape.dim(batch_dim_index);
41 output_shape.dim(channel_dim_index) = input_shape.dim(channel_dim_index);
42
43 for (int i = 0; i < num_spatial_dims; i++)
44 {
45 const int spatial_dim_index = getDataSpatialDimIndex(_attributes.data_format, i);
46 const std::int32_t padded_input = input_shape.dim(spatial_dim_index) +
47 _attributes.padding_before.at(i) +
48 _attributes.padding_after.at(i);
49 // out_size = ceil((in_size - window_size + 1) / stride) =
50 // (in_size - window_size + 1 + stride - 1) / stride =
51 // (in_size - window_size) / stride + 1
52 output_shape.dim(spatial_dim_index) =
53 (padded_input - _attributes.window[i]) / _attributes.strides[i] + 1;
54 }
55
56 setOutputType(0, {getInput(0)->getElementType(), output_shape});
57}
