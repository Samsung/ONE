35/*
36 * Class to store training specific information
37 */
38class OMTrainingStorage
39{
40 // Store mapping between input tensor (its input_index) and its current input data.
41 // The input data must have a number of samples equal to batch_size
42 std::unordered_map<uint32_t, uint8_t *> _input_index_to_input_data;
43 // Store mapping between output tensor (its output_index) and its current target data.
44 // The target data must have a number of samples equal to batch_size
45 std::unordered_map<uint32_t, uint8_t *> _target_index_to_target_data;
46
47 // Store SGD optimizer with its own internal states
48 // Note: initial its null
49 std::unique_ptr<onert_micro::train::optimizers::SGD> _sgd_optimizer = nullptr;
50 // Store Adam optimizer with its own internal states
51 // Note: initial its null
52 std::unique_ptr<onert_micro::train::optimizers::Adam> _adam_optimizer = nullptr;
53
54public:
55 OMTrainingStorage() = default;
56 OMTrainingStorage(const OMTrainingStorage &) = delete;
57 OMTrainingStorage(OMTrainingStorage &&) = delete;
58 OMTrainingStorage &operator=(const OMTrainingStorage &) = delete;
59 OMTrainingStorage &&operator=(const OMTrainingStorage &&) = delete;
60 ~OMTrainingStorage() = default;
61
62 // Set input data for current input tensor
63 void setInputData(uint8_t *data, uint32_t input_index)
64 {
65 _input_index_to_input_data[input_index] = data;
66 }
67 // Set target data for current output tensor
68 void setTargetData(uint8_t *data, uint32_t target_index)
69 {
70 _target_index_to_target_data[target_index] = data;
71 }
72
73 // Choose and set optimizer defined in config
74 OMStatus setOptimizer(const OMConfig &config);
75
76 // Get pointer to SGD optimizer
77 // Note: can return nullptr
78 onert_micro::train::optimizers::SGD *getSGD() { return _sgd_optimizer.get(); }
79 // Get pointer to Adam optimizer
80 // Note: can return nullptr
81 onert_micro::train::optimizers::Adam *getAdam() { return _adam_optimizer.get(); }
82
83 // Get pointer to saved input data for current input tensor
84 uint8_t *getInputData(uint32_t input_index) { return _input_index_to_input_data[input_index]; }
85 // Get pointer to saved target data for current output tensor
86 uint8_t *getTargetData(uint32_t target_index)
87 {
88 return _target_index_to_target_data[target_index];
89 }
90
91 // Reset and deallocate all states
92 void reset();
93};
