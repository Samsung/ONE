35/*
36 * Class to store training specific information
37 */
38class OMTrainingStorage
39{
40 // Store mapping between input tensor (its input_index) and its current input data.
41 // The input data must have a number of samples equal to batch_size
42 std::unordered_map<uint32_t, uint8_t *> _input_index_to_input_data;
43 // Store mapping between output tensor (its output_index) and its current target data.
44 // The target data must have a number of samples equal to batch_size
45 std::unordered_map<uint32_t, uint8_t *> _target_index_to_target_data;
46
47 // Store SGD optimizer with its own internal states
48 // Note: initial its null
49 std::unique_ptr<onert_micro::train::optimizers::SGD> _sgd_optimizer = nullptr;
50 // Store Adam optimizer with its own internal states
51 // Note: initial its null
52 std::unique_ptr<onert_micro::train::optimizers::Adam> _adam_optimizer = nullptr;
53
54 // Store rank types
55 std::unordered_map<uint16_t, core::OpTrainableRankType> _tensor_index_to_train_rank;
56
57public:
58 OMTrainingStorage() = default;
59 OMTrainingStorage(const OMTrainingStorage &) = delete;
60 OMTrainingStorage(OMTrainingStorage &&) = delete;
61 OMTrainingStorage &operator=(const OMTrainingStorage &) = delete;
62 OMTrainingStorage &&operator=(const OMTrainingStorage &&) = delete;
63 ~OMTrainingStorage() = default;
64
65 // Set input data for current input tensor
66 void setInputData(uint8_t *data, uint32_t input_index)
67 {
68 _input_index_to_input_data[input_index] = data;
69 }
70 // Set target data for current output tensor
71 void setTargetData(uint8_t *data, uint32_t target_index)
72 {
73 _target_index_to_target_data[target_index] = data;
74 }
75
76 void addTrainRank(uint16_t tensor_index, core::OpTrainableRankType train_rank)
77 {
78 _tensor_index_to_train_rank[tensor_index] = train_rank;
79 }
80
81 std::unordered_map<uint16_t, core::OpTrainableRankType> &getTensorIndexToRankTypeTable()
82 {
83 return _tensor_index_to_train_rank;
84 }
85
86 // Choose and set optimizer defined in config
87 OMStatus setOptimizer(const OMConfig &config);
88
89 // Get pointer to SGD optimizer
90 // Note: can return nullptr
91 onert_micro::train::optimizers::SGD *getSGD() { return _sgd_optimizer.get(); }
92 // Get pointer to Adam optimizer
93 // Note: can return nullptr
94 onert_micro::train::optimizers::Adam *getAdam() { return _adam_optimizer.get(); }
95
96 // Get pointer to saved input data for current input tensor
97 uint8_t *getInputData(uint32_t input_index) { return _input_index_to_input_data[input_index]; }
98 // Get pointer to saved target data for current output tensor
99 uint8_t *getTargetData(uint32_t target_index)
100 {
101 return _target_index_to_target_data[target_index];
102 }
103
104#ifdef OM_MEMORY_ESTIMATE
105
106 // Reset and deallocate all states
107 void reset(core::OMRuntimeContext &context, core::OMRuntimeStorage &storage);
108
109#endif // OM_MEMORY_ESTIMATE
110
111 // Reset and deallocate all states
112 void reset();
113};
