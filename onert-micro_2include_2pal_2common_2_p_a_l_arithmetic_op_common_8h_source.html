50template <typename T, typename Fn>
51OMStatus ArithmeticOp(const core::BinaryArithmeticBroadcastParams &params, const int flat_size,
52 const T *input1_data, const T *input2_data, T *output_data)
53{
54 T activation_min, activation_max;
55 getActivationParams(params, &activation_min, &activation_max);
56
57 Fn func;
58 for (int i = 0; i < flat_size; ++i)
59 output_data[i] =
60 std::min(std::max(func(input1_data[i], input2_data[i]), activation_min), activation_max);
61
62 return Ok;
63}

78template <typename T, typename Fn>
79OMStatus BroadcastArithmeticOp4DSlow(const core::BinaryArithmeticBroadcastParams &params,
80 const core::OMRuntimeShape &input1_shape, const T *input1_data,
81 const core::OMRuntimeShape &input2_shape, const T *input2_data,
82 const core::OMRuntimeShape &output_shape, T *output_data)
83{
84 NdArrayDesc<4> desc1;
85 NdArrayDesc<4> desc2;
86 NdArrayDescsForElementwiseBroadcast(input1_shape, input2_shape, &desc1, &desc2);
87 const core::OMRuntimeShape extended_output_shape =
88 core::OMRuntimeShape::extendedShape(4, output_shape);
89
90 T activation_min, activation_max;
91 getActivationParams(params, &activation_min, &activation_max);
92
93 // In Tensorflow, the dimensions are canonically named (batch_number, row,
94 // col, channel), with extents (batches, height, width, depth), with the
95 // trailing dimension changing most rapidly (channels has the smallest stride,
96 // typically 1 element).
97 //
98 // In generated C code, we store arrays with the dimensions reversed. The
99 // first dimension has smallest stride.
100 //
101 // We name our variables by their Tensorflow convention, but generate C code
102 // nesting loops such that the innermost loop has the smallest stride for the
103 // best cache behavior.
104 Fn func;
105 for (int b = 0; b < extended_output_shape.dims(0); ++b)
106 {
107 for (int y = 0; y < extended_output_shape.dims(1); ++y)
108 {
109 for (int x = 0; x < extended_output_shape.dims(2); ++x)
110 {
111 for (int c = 0; c < extended_output_shape.dims(3); ++c)
112 {
113 const int output_data_offset =
114 ((b * extended_output_shape.dims(1) + y) * extended_output_shape.dims(2) + x) *
115 extended_output_shape.dims(3) +
116 c;
117
118 output_data[output_data_offset] =
119 std::min(std::max(func(input1_data[subscriptToIndex(desc1, b, y, x, c)],
120 input2_data[subscriptToIndex(desc2, b, y, x, c)]),
121 activation_min),
122 activation_max);
123 }
124 }
125 }
126 }
127 return Ok;
128}
