31bool canonicalize_avgpool2d(loco::Graph *graph, moco::TFAvgPool *node)
32{
33 LOGGER(l);
34
53 auto data_layout = plier::tf::as_data_layout(node->data_layout());
54
55 auto feature_enc = graph->nodes()->create<loco::FeatureEncode>();
56 auto avgPool2d_node = graph->nodes()->create<loco::AvgPool2D>();
57 auto feature_dec = graph->nodes()->create<loco::FeatureDecode>();
58
59 set_feature_enc(feature_enc, data_layout);
60 set_feature_dec(feature_dec, data_layout);
61
62 avgPool2d_node->convention(loco::AvgPool2D::Convention::Valid);
63
64 auto value_shape = moco::node_shape(node->value());
65 assert(value_shape.domain() != loco::Domain::Unknown);
66
67 auto node_stride = moco::stride_of(node->strides(), node->data_layout());
68 auto node_window = moco::window_of(node->ksize(), node->data_layout());
69
70 moco::Padding2DInference infer_padding2d;
71
72 infer_padding2d.padding(node->padding());
73 infer_padding2d.stride(node_stride);
74 infer_padding2d.window(node_window);
75
76 auto input_feature_shape = moco::as_feature_shape(value_shape, node->data_layout());
77 auto input_plane_shape = moco::make_plane_shape(input_feature_shape);
78
79 *avgPool2d_node->pad() = infer_padding2d(input_plane_shape);
80 *avgPool2d_node->stride() = node_stride;
81 *avgPool2d_node->window() = node_window;
82
83 INFO(l) << "Canonicalize TFAvgPool pad = T " << avgPool2d_node->pad()->top() << ", L "
84 << avgPool2d_node->pad()->left() << ", B " << avgPool2d_node->pad()->bottom() << ", R "
85 << avgPool2d_node->pad()->right() << std::endl;
86
87 // update graph
88 auto node_A = node->value();
89
90 // update connections
91 feature_enc->input(node_A);
92 avgPool2d_node->ifm(feature_enc);
93 feature_dec->input(avgPool2d_node);
94
95 // replace node
96 replace(node).with(feature_dec);
97
98 return true;
99}
