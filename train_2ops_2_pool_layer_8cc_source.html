48class MaxPool2D final : public TrainingKernelRegistry
49{
50private:
51 const ir::Activation _activation;
52 const IPortableTensor *_output;
53 nnfw::cker::PoolParams _op_params;
54
55 std::unique_ptr<Tensor> _act_back_prop_output;
56 std::unique_ptr<Tensor> _arg_max_index;
57
58public:
59 MaxPool2D(const uint32_t paddingLeft, const uint32_t, const uint32_t, const uint32_t paddingTop,
60 const uint32_t strideWidth, const uint32_t strideHeight, const uint32_t kernelWidth,
61 const uint32_t kernelHeight, const ir::Activation activation,
62 const IPortableTensor *output)
63 : _activation(activation), _output(output)
64 {
65 {
66 _op_params.stride_height = strideHeight;
67 _op_params.stride_width = strideWidth;
68 _op_params.filter_height = kernelHeight;
69 _op_params.filter_width = kernelWidth;
70 _op_params.padding_values.height = (int8_t)paddingTop;
71 _op_params.padding_values.width = (int8_t)paddingLeft;
72 CalculateActivationRange<float>(activation, &_op_params.float_activation_min,
73 &_op_params.float_activation_max);
74 }
75
76 _arg_max_index = std::make_unique<Tensor>(_output->get_info(), _output->layout());
77 _arg_max_index->setBuffer(std::make_shared<basic::Allocator>(_output->total_size()));
78
79 if (activation != ir::Activation::NONE)
80 {
81 _act_back_prop_output = std::make_unique<Tensor>(_output->get_info(), _output->layout());
82 _act_back_prop_output->setBuffer(std::make_shared<basic::Allocator>(_output->total_size()));
83 }
84 };
85
86 ~MaxPool2D() {}
87
88public:
89 void forward(const IPortableTensor *in, IPortableTensor *out)
90 {
91 assert(in->layout() == ir::Layout::NHWC);
92
93 auto out_shape = getShape(out);
94 auto out_data = getBuffer<float>(out);
95 auto arg_max_index = _arg_max_index.get();
96
97 // maxpool forward
98 nnfw::cker::train::MaxPool2D(_op_params, getShape(in), getBuffer<float>(in), out_shape,
99 out_data, getBuffer<int>(arg_max_index));
100 }
101
102 void backward(const IPortableTensor *back_prop_out, IPortableTensor *back_prop_in)
103 {
104 assert(back_prop_out->layout() == ir::Layout::NHWC);
105
106 // activation backward
107 try
108 {
109 back_prop_out =
110 backpropActivation(_activation, _output, back_prop_out, _act_back_prop_output.get());
111 }
112 catch (const std::exception &e)
113 {
114 throw std::runtime_error{"train PoolLayer: " + std::string(e.what())};
115 }
116 assert(back_prop_out != nullptr);
117
118 // maxpool baackward
119 auto arg_max_index = _arg_max_index.get();
120 nnfw::cker::train::MaxPool2DGrad(getShape(back_prop_out), getBuffer<float>(back_prop_out),
121 getBuffer<int>(arg_max_index), getShape(back_prop_in),
122 getBuffer<float>(back_prop_in));
123 }
124};

134void PoolLayer::configure(const IPortableTensor *input, const uint32_t paddingLeft,
135 const uint32_t paddingRight, const uint32_t paddingTop,
136 const uint32_t paddingBottom, const uint32_t strideWidth,
137 const uint32_t strideHeight, const uint32_t kernelWidth,
138 const uint32_t kernelHeight, const ir::Activation activation,
139 IPortableTensor *output, const PoolType op_type,
140 IPortableTensor *back_prop_input, const IPortableTensor *back_prop_output)
141{
142 _input = input;
143 _output = output;
144
145 _back_prop_output = back_prop_output;
146 _back_prop_input = back_prop_input;
147
148 // ready inference kernel
149 cpu::ops::PoolLayer::configure(input, paddingLeft, paddingRight, paddingTop, paddingBottom,
150 strideWidth, strideHeight, kernelWidth, kernelHeight, activation,
151 output, convertToInfer(op_type));
152
153 if (output->data_type() != OperandType::FLOAT32)
154 {
155 throw std::runtime_error("PoolLayer : Unsupported data type for training");
156 }
157
158 // ready training kernel
159 switch (op_type)
160 {
161 case PoolType::kMax:
162 _kernel = std::make_unique<MaxPool2D>(paddingLeft, paddingRight, paddingTop, paddingBottom,
163 strideWidth, strideHeight, kernelWidth, kernelHeight,
164 activation, output);
165 break;
166 default:
167 throw std::runtime_error("PoolLayer: Unsupported pool type");
168 }
169}
