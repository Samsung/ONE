57luci::CircleConst *dequantized_const_node(luci::CircleConst *const_node)
58{
59 auto name = const_node->name();
60 assert(name.length() > 0);
61 auto g = const_node->graph();
62 auto new_const_node = g->nodes()->create<luci::CircleConst>();
63
64 new_const_node->dtype(loco::DataType::FLOAT32);
65 new_const_node->rank(const_node->rank());
66 uint32_t dim_size = 1;
67 for (uint32_t i = 0; i < new_const_node->rank(); ++i)
68 {
69 new_const_node->dim(i) = const_node->dim(i);
70 dim_size *= const_node->dim(i).value();
71 }
72 new_const_node->size<loco::DataType::FLOAT32>(dim_size);
73 new_const_node->shape_status(luci::ShapeStatus::VALID);
74 new_const_node->name(name + "_DQ");
75
76 if (const_node->dtype() == loco::DataType::FLOAT16)
77 {
78 for (uint32_t i = 0; i < new_const_node->size<loco::DataType::FLOAT32>(); ++i)
79 {
80 auto raw = const_node->at<loco::DataType::FLOAT16>(i);
81 new_const_node->at<loco::DataType::FLOAT32>(i) = fp16_ieee_to_fp32_value(raw);
82 }
83 return new_const_node;
84 }
85
86 if (const_node->quantparam() == nullptr)
87 {
88 throw std::runtime_error("Given constant node has no quantization parameter");
89 }
90
91 const int32_t q_dim = const_node->quantparam()->quantized_dimension;
92 // For scalar, q_dim_value is 1
93 // For non-scalar, q_dim_value is the size of quantized dimension
94 const int32_t q_dim_value = const_node->rank() == 0 ? 1 : const_node->dim(q_dim).value();
95
96 int32_t right_count = q_dim_value;
97 for (uint32_t i = q_dim + 1; i < const_node->rank(); ++i)
98 right_count *= const_node->dim(i).value();
99
100 for (uint32_t i = 0; i < new_const_node->size<loco::DataType::FLOAT32>(); ++i)
101 {
102 uint32_t qd = (i % right_count) / (right_count / q_dim_value);
103 if (qd >= const_node->quantparam()->zerop.size())
104 qd = 0;
105
106 switch (const_node->dtype())
107 {
108 case loco::DataType::S8:
109 new_const_node->at<loco::DataType::FLOAT32>(i) =
110 static_cast<float>(const_node->at<loco::DataType::S8>(i) -
111 const_node->quantparam()->zerop.at(qd)) *
112 const_node->quantparam()->scale.at(qd);
113 break;
114 case loco::DataType::S16:
115 new_const_node->at<loco::DataType::FLOAT32>(i) =
116 static_cast<float>(const_node->at<loco::DataType::S16>(i) -
117 const_node->quantparam()->zerop.at(qd)) *
118 const_node->quantparam()->scale.at(qd);
119 break;
120 case loco::DataType::S32:
121 new_const_node->at<loco::DataType::FLOAT32>(i) =
122 static_cast<float>(const_node->at<loco::DataType::S32>(i) -
123 const_node->quantparam()->zerop.at(qd)) *
124 const_node->quantparam()->scale.at(qd);
125 break;
126 case loco::DataType::S64:
127 new_const_node->at<loco::DataType::FLOAT32>(i) =
128 static_cast<float>(const_node->at<loco::DataType::S64>(i) -
129 const_node->quantparam()->zerop.at(qd)) *
130 const_node->quantparam()->scale.at(qd);
131 break;
132 case loco::DataType::U8:
133 new_const_node->at<loco::DataType::FLOAT32>(i) =
134 static_cast<float>(const_node->at<loco::DataType::U8>(i) -
135 const_node->quantparam()->zerop.at(qd)) *
136 const_node->quantparam()->scale.at(qd);
137 break;
138 default:
139 throw std::runtime_error("Not supported dtype for FoldDequantizePass");
140 }
141 }
142
143 return new_const_node;
144}

146bool replace_const_node(loco::Node *node, luci::CircleConst *const_node)

197bool FoldDequantizePass::run(loco::Graph *g)
198{
199 bool changed = false;
200
201 for (auto node : loco::active_nodes(loco::output_nodes(g)))
202 {
203 if (auto circle_dequant = dynamic_cast<luci::CircleDequantize *>(node))
204 {
205 if (auto const_input = dynamic_cast<luci::CircleConst *>(circle_dequant->input()))
206 {
207 // Pattern 1 - When input of Dequantize is foldable constant
208 if (is_foldable_const(const_input))
209 {
210 loco::replace(circle_dequant).with(dequantized_const_node(const_input));
211 changed = true;
212 }
213 }
214 }
215 else if (auto const_node = dynamic_cast<luci::CircleConst *>(node))
216 {
217 if (is_foldable_const(const_node))
218 {
219 for (auto const_node_user : loco::succs(const_node))
220 {
221 // If user is hybrid kernel supported operation, do not dequantize
222 if (is_hybrid_kernel_supported(const_node_user))
223 continue;
224
225 auto users = loco::succs(const_node_user);
226 if (users.size() > 1)
227 continue;
228
229 // Pattern 2 - When input of Dequantize uses quantized output value
230 if (auto dequant = dynamic_cast<luci::CircleDequantize *>(*users.begin()))
231 {
232 if (replace_const_node(const_node_user, const_node))
233 {
234 loco::replace(dequant).with(const_node_user);
235 luci::add_origin(loco::must_cast<luci::CircleNode *>(const_node_user),
236 luci::get_origin(dequant));
237 changed = true;
238 }
239 }
240 }
241 }
242 }
243 }
244
245 return changed;
246}
