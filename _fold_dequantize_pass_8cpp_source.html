35bool is_foldable_const(luci::CircleConst *node)
36{
37 if (node->dtype() == loco::DataType::FLOAT16)

61luci::CircleConst *dequantized_const_node(luci::CircleConst *const_node)
62{
63 auto name = const_node->name();
64 assert(name.length() > 0);
65 auto g = const_node->graph();
66 auto new_const_node = g->nodes()->create<luci::CircleConst>();
67
68 new_const_node->dtype(loco::DataType::FLOAT32);
69 new_const_node->rank(const_node->rank());
70 uint32_t dim_size = 1;
71 for (uint32_t i = 0; i < new_const_node->rank(); ++i)
72 {
73 new_const_node->dim(i) = const_node->dim(i);
74 dim_size *= const_node->dim(i).value();
75 }
76 new_const_node->size<loco::DataType::FLOAT32>(dim_size);
77 new_const_node->shape_status(luci::ShapeStatus::VALID);
78 new_const_node->name(name + "_DQ");
79
80 if (const_node->dtype() == loco::DataType::FLOAT16)
81 {
82 for (uint32_t i = 0; i < new_const_node->size<loco::DataType::FLOAT32>(); ++i)
83 {
84 auto raw = const_node->at<loco::DataType::FLOAT16>(i);
85 new_const_node->at<loco::DataType::FLOAT32>(i) = fp16_ieee_to_fp32_value(raw);
86 }
87 return new_const_node;
88 }
89
90 if (const_node->quantparam() == nullptr)
91 {
92 throw std::runtime_error("Given constant node has no quantization parameter");
93 }
94
95 const int32_t q_dim = const_node->quantparam()->quantized_dimension;
96 // For scalar, q_dim_value is 1
97 // For non-scalar, q_dim_value is the size of quantized dimension
98 const int32_t q_dim_value = const_node->rank() == 0 ? 1 : const_node->dim(q_dim).value();
99
100 int32_t right_count = q_dim_value;
101 for (uint32_t i = q_dim + 1; i < const_node->rank(); ++i)
102 right_count *= const_node->dim(i).value();
103
104 for (uint32_t i = 0; i < new_const_node->size<loco::DataType::FLOAT32>(); ++i)
105 {
106 uint32_t qd = (i % right_count) / (right_count / q_dim_value);
107 if (qd >= const_node->quantparam()->zerop.size())
108 qd = 0;
109
110 switch (const_node->dtype())
111 {
112 case loco::DataType::U4:
113 new_const_node->at<loco::DataType::FLOAT32>(i) =
114 static_cast<float>(const_node->at<loco::DataType::U4>(i) -
115 const_node->quantparam()->zerop.at(qd)) *
116 const_node->quantparam()->scale.at(qd);
117 break;
118 case loco::DataType::S4:
119 new_const_node->at<loco::DataType::FLOAT32>(i) =
120 static_cast<float>(const_node->at<loco::DataType::S4>(i) -
121 const_node->quantparam()->zerop.at(qd)) *
122 const_node->quantparam()->scale.at(qd);
123 break;
124 case loco::DataType::S8:
125 new_const_node->at<loco::DataType::FLOAT32>(i) =
126 static_cast<float>(const_node->at<loco::DataType::S8>(i) -
127 const_node->quantparam()->zerop.at(qd)) *
128 const_node->quantparam()->scale.at(qd);
129 break;
130 case loco::DataType::S16:
131 new_const_node->at<loco::DataType::FLOAT32>(i) =
132 static_cast<float>(const_node->at<loco::DataType::S16>(i) -
133 const_node->quantparam()->zerop.at(qd)) *
134 const_node->quantparam()->scale.at(qd);
135 break;
136 case loco::DataType::S32:
137 new_const_node->at<loco::DataType::FLOAT32>(i) =
138 static_cast<float>(const_node->at<loco::DataType::S32>(i) -
139 const_node->quantparam()->zerop.at(qd)) *
140 const_node->quantparam()->scale.at(qd);
141 break;
142 case loco::DataType::S64:
143 new_const_node->at<loco::DataType::FLOAT32>(i) =
144 static_cast<float>(const_node->at<loco::DataType::S64>(i) -
145 const_node->quantparam()->zerop.at(qd)) *
146 const_node->quantparam()->scale.at(qd);
147 break;
148 case loco::DataType::U8:
149 new_const_node->at<loco::DataType::FLOAT32>(i) =
150 static_cast<float>(const_node->at<loco::DataType::U8>(i) -
151 const_node->quantparam()->zerop.at(qd)) *
152 const_node->quantparam()->scale.at(qd);
153 break;
154 default:
155 throw std::runtime_error("Not supported dtype for FoldDequantizePass");
156 }
157 }
158
159 return new_const_node;
160}

162bool replace_const_node(loco::Node *node, luci::CircleConst *const_node)

213bool FoldDequantizePass::run(loco::Graph *g)
214{
215 bool changed = false;
216
217 for (auto node : loco::active_nodes(loco::output_nodes(g)))
218 {
219 if (auto circle_dequant = dynamic_cast<luci::CircleDequantize *>(node))
220 {
221 if (auto const_input = dynamic_cast<luci::CircleConst *>(circle_dequant->input()))
222 {
223 // Pattern 1 - When input of Dequantize is foldable constant
224 if (is_foldable_const(const_input))
225 {
226 loco::replace(circle_dequant).with(dequantized_const_node(const_input));
227 changed = true;
228 }
229 }
230 }
231 else if (auto const_node = dynamic_cast<luci::CircleConst *>(node))
232 {
233 if (is_foldable_const(const_node))
234 {
235 for (auto const_node_user : loco::succs(const_node))
236 {
237 // If user is hybrid kernel supported operation, do not dequantize
238 if (is_hybrid_kernel_supported(const_node_user))
239 continue;
240
241 auto users = loco::succs(const_node_user);
242 if (users.size() > 1)
243 continue;
244
245 // Pattern 2 - When input of Dequantize uses quantized output value
246 if (auto dequant = dynamic_cast<luci::CircleDequantize *>(*users.begin()))
247 {
248 if (replace_const_node(const_node_user, const_node))
249 {
250 loco::replace(dequant).with(const_node_user);
251 luci::add_origin(loco::must_cast<luci::CircleNode *>(const_node_user),
252 luci::get_origin(dequant));
253 changed = true;
254 }
255 }
256 }
257 }
258 }
259 }
260
261 return changed;
262}
