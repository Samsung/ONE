113template <typename T, class RuntimeKernel>
114OMInputContext<T> MakeInputContext(RuntimeKernel &rtk, size_t inputIdx = 0)
115{
116 auto data = MakeInputData<T>(rtk, inputIdx);
117 const circle::Tensor *tensor = rtk.inputs[inputIdx];
118 return OMInputContext<T>(std::move(data), tensor);
119}
120
121template <typename T, class RuntimeKernel>
122OMOutputContext<T> MakeOutputContext(RuntimeKernel &rtk, size_t outputIdx = 0)
123{
124 auto data = MakeOutputData<T>(rtk, outputIdx);
125 const circle::Tensor *tensor = rtk.outputs[outputIdx];
126 return OMOutputContext<T>(std::move(data), tensor);
127}
128
129template <class RuntimeKernel>
130OMAxisContext MakeAxisContext(RuntimeKernel &rtk, size_t axisInputIdx = 1)
131{
132 auto data = MakeAxisData<uint32_t>(rtk, axisInputIdx);
133 const circle::Tensor *tensor = rtk.inputs[axisInputIdx];
134 return OMAxisContext(std::move(data), tensor);
