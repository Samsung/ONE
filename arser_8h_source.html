328 Argument &add_argument(const std::string &arg_name)
329 {
330 if (arg_name.at(0) != '-') /* positional */
331 {
332 _positional_arg_vec.emplace_back(arg_name);
333 _arg_map[arg_name] = &_positional_arg_vec.back();
334 }
335 else /* optional */
336 {
337 // The length of optional argument name must be 2 or more.
338 // And it shouldn't be hard to recognize. e.g. '-', '--'
339 if (arg_name.size() < 2)
340 {
341 throw std::runtime_error("Too short name. The length of argument name must be 2 or more.");
342 }
343 if (arg_name == "--")
344 {
345 throw std::runtime_error(
346 "Too short name. Option name must contain at least one character other than dash.");
347 }
348 _optional_arg_vec.emplace_back(arg_name);
349 _optional_arg_vec.back()._short_name = arg_name;
350 _arg_map[arg_name] = &_optional_arg_vec.back();
351 }
352 return *_arg_map[arg_name];
353 }
354
355 Argument &add_argument(const std::vector<std::string> &arg_name_vec)
356 {
357 assert(arg_name_vec.size() >= 2);
358 std::string long_opt, short_opt;
359 // find long and short option
360 for (const auto &arg_name : arg_name_vec)
361 {
362 if (arg_name.at(0) != '-')
363 {
364 throw std::runtime_error("Invalid argument. "
365 "Positional argument cannot have short option.");
366 }
367 assert(arg_name.size() >= 2);
368 if (long_opt.empty() && arg_name.at(0) == '-' && arg_name.at(1) == '-')
369 {
370 long_opt = arg_name;
371 }
372 if (short_opt.empty() && arg_name.at(0) == '-' && arg_name.at(1) != '-')
373 {
374 short_opt = arg_name;
375 }
376 }
377 // If one of the two is empty, fill it with the non-empty one for pretty printing.
378 if (long_opt.empty())
379 {
380 assert(not short_opt.empty());
381 long_opt = short_opt;
382 }
383 if (short_opt.empty())
384 {
385 assert(not long_opt.empty());
386 short_opt = long_opt;
387 }
388
389 _optional_arg_vec.emplace_back(short_opt, long_opt, arg_name_vec);
390 for (const auto &arg_name : arg_name_vec)
391 {
392 _arg_map[arg_name] = &_optional_arg_vec.back();
393 }
394 return _optional_arg_vec.back();
395 }

397 template <typename... Ts> Argument &add_argument(const std::string &arg_name, Ts... arg_names)
398 {
399 if (sizeof...(arg_names) == 0)
400 {
401 return add_argument(arg_name);
402 }
403 // sizeof...(arg_names) > 0
404 else
405 {
406 return add_argument(std::vector<std::string>{arg_name, arg_names...});

410 void validate_arguments(void)
411 {
412 // positional argument is always required.
413 for (const auto &arg : _positional_arg_vec)
414 {
415 if (arg._is_required)
416 {
417 throw std::runtime_error("Invalid arguments. Positional argument must always be required.");
418 }
419 }
420 // TODO accumulated arguments shouldn't be enabled to positional arguments.
421 // TODO accumulated arguments shouldn't be enabled to optional arguments whose `narg` == 0.
422 }

424 void parse(int argc, char **argv)
425 {
426 validate_arguments();
427 _program_name = argv[0];
428 _program_name.erase(0, _program_name.find_last_of("/\\") + 1);
429 if (argc >= 2)
430 {
431 if (!std::strcmp(argv[1], "--help") || !std::strcmp(argv[1], "-h"))
432 {
433 std::cout << *this;
434 std::exit(0);
435 }
436 else
437 {
438 for (const auto &arg : _arg_map)
439 {
440 const auto &func = arg.second->_func;
441 if (func && !std::strcmp(argv[1], arg.first.c_str()))
442 {
443 func();
444 std::exit(0);
445 }
446 }
447 }
448 }
449 /*
450 ** ./program_name [optional argument] [positional argument]
451 */
452 // get the number of positioanl argument
453 size_t parg_num = _positional_arg_vec.size();
454 // get the number of "required" optional argument
455 size_t required_oarg_num = 0;
456 for (auto arg : _optional_arg_vec)
457 {
458 if (arg._is_required)
459 required_oarg_num++;
460 }
461 // parse argument
462 for (int c = 1; c < argc;)
463 {
464 std::string arg_name{argv[c++]};
465 auto arg = _arg_map.find(arg_name);
466 // check whether arg is positional or not
467 if (arg == _arg_map.end())
468 {
469 if (parg_num)
470 {
471 auto it = _positional_arg_vec.begin();
472 std::advance(it, _positional_arg_vec.size() - parg_num);
473 (*it)._values.clear();
474 (*it)._values.emplace_back(arg_name);
475 parg_num--;
476 }
477 else
478 throw std::runtime_error("Invalid argument. "
479 "You've given more positional argument than necessary.");
480 }
481 else // optional argument
482 {
483 // check whether arg is required or not
484 if (arg->second->_is_required)
485 required_oarg_num--;
486 arg->second->_values.clear();
487 for (uint32_t n = 0; n < arg->second->_nargs; n++)
488 {
489 if (c >= argc)
490 throw std::runtime_error("Invalid argument. "
491 "You must have missed some argument.");
492 arg->second->_values.emplace_back(argv[c++]);
493 }
494 // accumulate values
495 if (arg->second->_is_accumulated)
496 {
497 arg->second->_accum_values.emplace_back(arg->second->_values);
498 }
499 if (arg->second->_nargs == 0)
500 {
501 // TODO std::boolalpha for true or false
502 arg->second->_values.emplace_back("1");
503 }
504 }
505 }
506 if (parg_num || required_oarg_num)
507 throw std::runtime_error("Invalid argument. "
508 "You must have missed some argument.");
509 }
510
511 bool operator[](const std::string &arg_name)
512 {
513 auto arg = _arg_map.find(arg_name);
514 if (arg == _arg_map.end())
515 return false;
516
517 if (arg->second->_is_accumulated)
518 return arg->second->_accum_values.size() > 0 ? true : false;
519
520 return arg->second->_values.size() > 0 ? true : false;
521 }

532 friend std::ostream &operator<<(std::ostream &stream, const Arser &parser)
533 {
534 // print description
535 if (!parser._program_description.empty())
536 {
537 stream << "What " << parser._program_name << " does: " << parser._program_description
538 << "\n\n";
539 }
540 /*
541 ** print usage
542 */
543 auto print_usage_arg = [&](const arser::Argument &arg) {
544 stream << " ";
545 std::string arg_name = arser::internal::remove_dash(arg._long_name);
546 std::for_each(arg_name.begin(), arg_name.end(),
547 [&stream](const char &c) { stream << static_cast<char>(::toupper(c)); });
548 };
549 stream << "Usage: ./" << parser._program_name << " ";
550 // required optional argument
551 for (const auto &arg : parser._optional_arg_vec)
552 {
553 if (!arg._is_required)
554 continue;
555 stream << arg._short_name;
556 print_usage_arg(arg);
557 stream << " ";
558 }
559 // rest of the optional argument
560 for (const auto &arg : parser._optional_arg_vec)
561 {
562 if (arg._is_required)
563 continue;
564 stream << "[" << arg._short_name;
565 if (arg._nargs)
566 {
567 print_usage_arg(arg);
568 }
569 stream << "]"
570 << " ";
571 }
572 // positional arguement
573 for (const auto &arg : parser._positional_arg_vec)
574 {
575 stream << arg._long_name << " ";
576 }
577 stream << "\n\n";
578 /*
579 ** print argument list and its help message
580 */
581 // get the length of the longest argument
582 size_t length_of_longest_arg = 0;
583 for (const auto &arg : parser._positional_arg_vec)
584 {
585 length_of_longest_arg = std::max(length_of_longest_arg,
586 arser::internal::make_comma_concatenated(arg._names).size());
587 }
588 for (const auto &arg : parser._optional_arg_vec)
589 {
590 length_of_longest_arg = std::max(length_of_longest_arg,
591 arser::internal::make_comma_concatenated(arg._names).size());
592 }
593
594 const size_t message_width = 60;
595 auto print_help_args = [&](const std::list<Argument> &args, const std::string &title) {
596 if (!args.empty())
597 {
598 stream << title << std::endl;
599 for (const auto &arg : args)
600 {
601 stream.width(length_of_longest_arg);
602 stream << std::left << arser::internal::make_comma_concatenated(arg._names) << "\t";
603 for (size_t i = 0; i < arg._help_message.length(); i += message_width)
604 {
605 if (i)
606 stream << std::string(length_of_longest_arg, ' ') << "\t";
607 stream << arg._help_message.substr(i, message_width) << std::endl;
608 }
609 }
610 std::cout << std::endl;
611 }
612 };
613 // positional argument
614 print_help_args(parser._positional_arg_vec, "[Positional argument]");
615 // optional argument
616 print_help_args(parser._optional_arg_vec, "[Optional argument]");
617
618 return stream;
619 }

622 std::string _program_name;
623 std::string _program_description;
624 std::list<Argument> _positional_arg_vec;
625 std::list<Argument> _optional_arg_vec;
626 std::map<std::string, Argument *> _arg_map;

629template <typename T> T Arser::get_impl(const std::string &arg_name, T *)
630{
631 auto arg = _arg_map.find(arg_name);
632 if (arg == _arg_map.end())
633 throw std::runtime_error("Invalid argument. "
634 "There is no argument you are looking for: " +
635 arg_name);
636
637 if (arg->second->_is_accumulated)
638 throw std::runtime_error(
639 "Type mismatch. "
640 "You called get using a type different from the one you specified."
641 "Accumulated argument is returned as std::vector of the specified type");
642
643 if (arg->second->_type != TypeName<T>::Get())
644 throw std::runtime_error("Type mismatch. "
645 "You called get() method with a type different "
646 "from the one you specified. "
647 "Please check the type of what you specified in "
648 "add_argument() method.");
649
650 if (arg->second->_values.size() == 0)
651 throw std::runtime_error("Wrong access. "
652 "You must make sure that the argument is given before accessing it. "
653 "You can do it by calling arser[\"argument\"].");
654
655 return internal::lexical_cast<T>(arg->second->_values[0]);
656}

658template <typename T> std::vector<T> Arser::get_impl(const std::string &arg_name, std::vector<T> *)
659{
660 auto arg = _arg_map.find(arg_name);
661 if (arg == _arg_map.end())
662 throw std::runtime_error("Invalid argument. "
663 "There is no argument you are looking for: " +
664 arg_name);
665
666 // Accumulated arguments with scalar type (e.g., STR)
667 if (arg->second->_is_accumulated)
668 {
669 if (arg->second->_type != TypeName<T>::Get())
670 throw std::runtime_error("Type mismatch. "
671 "You called get using a type different from the one you specified.");
672
673 std::vector<T> data;
674 for (auto values : arg->second->_accum_values)
675 {
676 assert(values.size() == 1);
677 data.emplace_back(internal::lexical_cast<T>(values[0]));
678 }
679 return data;
680 }
681
682 if (arg->second->_type != TypeName<std::vector<T>>::Get())
683 throw std::runtime_error("Type mismatch. "
684 "You called get using a type different from the one you specified.");
685
686 std::vector<T> data;
687 std::transform(arg->second->_values.begin(), arg->second->_values.end(), std::back_inserter(data),
688 [](std::string str) -> T { return internal::lexical_cast<T>(str); });
689 return data;
690}

692// Accumulated arguments with vector type (e.g., STR_VEC)
693template <typename T>
694std::vector<std::vector<T>> Arser::get_impl(const std::string &arg_name,
695 std::vector<std::vector<T>> *)
696{
697 auto arg = _arg_map.find(arg_name);
698 if (arg == _arg_map.end())
699 throw std::runtime_error("Invalid argument. "
700 "There is no argument you are looking for: " +
701 arg_name);
702
703 if (not arg->second->_is_accumulated)
704 throw std::runtime_error("Type mismatch. "
705 "You called get using a type different from the one you specified.");
706
707 if (arg->second->_type != TypeName<std::vector<T>>::Get())
708 throw std::runtime_error(
709 "Type mismatch. "
710 "You called get using a type different from the one you specified."
711 "Accumulated argument is returned as std::vector of the specified type");
712
713 std::vector<std::vector<T>> result;
714 for (auto values : arg->second->_accum_values)
715 {
716 std::vector<T> data;
717 std::transform(values.begin(), values.end(), std::back_inserter(data),
718 [](std::string str) -> T { return internal::lexical_cast<T>(str); });
719 result.emplace_back(data);
720 }
721
722 return result;
723}
