38/*
39 * Class to registry kernels for execution of backward graph (calculation backpropagation)
40 * Kernels which will be register define in KernelsToTrain.lst current pal directory
41 */
42class KernelBuiltinTrainRegistry
43{
44public:
45 constexpr KernelBuiltinTrainRegistry() : _operator_train()
46 {
47#define REGISTER_TRAIN_KERNEL(builtin_operator, name) \
48 registerKernelTrain(core::OMBuilderID::BuiltinOperator_##builtin_operator, \
49 train_kernel_Circle##name);
50
51#include "KernelsToTrain.lst"
52
53#undef REGISTER_TRAIN_KERNEL
54 }
55
56public:
57 OMStatus getKernelTrainFunc(core::OMBuilderID builderID, KernelTrainFunc **train_func) const
58 {
59 const auto builder_id_opcode = size_t(builderID);
60 assert(builder_id_opcode < size_t(core::OMBuilderID::BuiltinOperatorsSize));
61 if (builder_id_opcode >= size_t(core::OMBuilderID::BuiltinOperatorsSize))
62 {
63 *train_func = nullptr;
64 return UnknownError;
65 }
66 *train_func = _operator_train[builder_id_opcode];
67 return Ok;
68 }
69
70private:
71 constexpr void registerKernelTrain(core::OMBuilderID id, KernelTrainFunc *func)
72 {
73 assert(size_t(id) < size_t(core::OMBuilderID::BuiltinOperatorsSize));
74 _operator_train[size_t(id)] = func;
75 }
76
77private:
78 KernelTrainFunc *_operator_train[size_t(core::OMBuilderID::BuiltinOperatorsSize)];
79};
