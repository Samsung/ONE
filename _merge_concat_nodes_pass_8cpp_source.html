27bool canMerge(locoex::TFLConcatenation *node1, locoex::TFLConcatenation *node2)
28{
29 if (node1->fusedActivationFunction() != node2->fusedActivationFunction())
30 return false;
31
32 if (node1->axis() != node2->axis())
33 return false;
34
35 switch (node1->fusedActivationFunction())
36 {
37 case locoex::FusedActFunc::NONE:
38 case locoex::FusedActFunc::RELU:
39 case locoex::FusedActFunc::RELU6:
40 return true;
41
42 // case locoex::FusedActFunc::TANH:
43 // return false;
44
45 default:
46 INTERNAL_EXN_V("Unknown FusedActFunc", oops::to_uint32(node1->fusedActivationFunction()));
47 }
48}

84std::vector<loco::Node *> dfs(locoex::TFLConcatenation *root)
85{
86 std::vector<loco::Node *> res;
87
88 for (uint32_t i = 0; i < root->numValues(); ++i)
89 {
90 auto input = dynamic_cast<locoex::TFLConcatenation *>(root->values(i));
91 if (input != nullptr && canMerge(input, root))
92 {
93 auto children = dfs(input);
94 for (auto child : children)
95 res.push_back(child);
96 }
97 else
98 {
99 res.push_back(root->values(i));
100 }
101 }
102
103 return res;
104}

147bool MergeConcatNodesPass::run(loco::Graph *graph)
148{
149 // Let's enumerate nodes required to compute output nodes
150 auto active_nodes = loco::active_nodes(loco::output_nodes(graph));
151
152 // Find TFLConcatenation nodes which have another TFLConcatenation nodes
153 // as inputs, with same axis and same fusedActivationFunction
154 std::vector<locoex::TFLConcatenation *> candidates;
155 for (auto node : active_nodes)
156 {
157 if (auto concat = dynamic_cast<locoex::TFLConcatenation *>(node))
158 {
159 for (uint32_t i = 0; i < concat->numValues(); ++i)
160 {
161 auto input = dynamic_cast<locoex::TFLConcatenation *>(concat->values(i));
162 if (input != nullptr && canMerge(input, concat))
163 {
164 candidates.push_back(concat);
165 break;
166 }
167 }
168 }
169 }
170
171 // Merge multiple TFLConcatenation nodes as one TFLConcatenation node
172 for (auto node : candidates)
173 {
174 auto inputs = dfs(node);
175
176 auto new_concat = graph->nodes()->create<locoex::TFLConcatenation>(inputs.size());
177 new_concat->axis(node->axis());
178 new_concat->fusedActivationFunction(node->fusedActivationFunction());
179
180 for (uint32_t i = 0; i < inputs.size(); ++i)
181 new_concat->values(i, inputs.at(i));
182
183 loco::replace(node).with(new_concat);
184 for (uint32_t i = 0; i < node->numValues(); ++i)
185 node->values(i, nullptr);
186 }
187
188 return candidates.size() > 0;
189}
