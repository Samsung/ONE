32PermuteLayer::PermuteLayer(const std::vector<ITensor *> &src_tensors,
33 const std::vector<ITensor *> &dst_tensors,
34 const std::shared_ptr<ExternalContext> &external_context)
35 : _external_context{external_context}, _tasks_map{}
36{
37 assert(src_tensors.size() == dst_tensors.size());
38 _src_tensors = src_tensors;
39 _dst_tensors = dst_tensors;
40 _src_tensors_offsets.resize(src_tensors.size());
41 _dst_tensors_offsets.resize(dst_tensors.size());
42}

44void PermuteLayer::optimize()
45{
46 // Remove copying of tensor as nullptr
47 auto src_it = _src_tensors.begin();
48 auto dst_it = _dst_tensors.begin();
49 auto src_offsets_it = _src_tensors_offsets.begin();
50 auto dst_offsets_it = _dst_tensors_offsets.begin();
51 while (src_it != _src_tensors.end())
52 {
53 if ((*src_it == *dst_it) || (*src_it == nullptr || *dst_it == nullptr))
54 {
55 src_it = _src_tensors.erase(src_it);
56 dst_it = _dst_tensors.erase(dst_it);
57 src_offsets_it = _src_tensors_offsets.erase(src_offsets_it);
58 dst_offsets_it = _dst_tensors_offsets.erase(dst_offsets_it);
59 }
60 else
61 {
62 auto src = *src_it;
63 auto dst = *dst_it;
64 src_offsets_it->resize(0);
65 dst_offsets_it->resize(0);
66 if (underlying_type(src->data_type()) != underlying_type(dst->data_type()))
67 continue;
68 const auto permute_type = [&]() -> PermuteType {
69 if (src->getShape().rank() == 4 && src->layout() == ir::Layout::NHWC &&
70 dst->layout() == ir::Layout::NCHW)
71 {
72 return PermuteType::NHWC_TO_NCHW;
73 }
74 else if (src->getShape().rank() == 4 && src->layout() == ir::Layout::NCHW &&
75 dst->layout() == ir::Layout::NHWC)
76 {
77 return PermuteType::NCHW_TO_NHWC;
78 }
79 else
80 {
81 return PermuteType::COPY;
82 }
83 }();
84
85 // TODO Support different types
86 auto fn = [&](backend::ITensor &src_tensor) {
87 dst->access([&](backend::ITensor &dst_tensor) {
88 // NOTE The buffer of both tensor can be nullptr in this step
89 const auto data_size = ir::sizeOfDataType(src_tensor.data_type());
90
91 if (permute_type == PermuteType::COPY)
92 {
93 if ((!src_tensor.has_padding() && !dst_tensor.has_padding()))
94 {
95 const auto num_elements = src_tensor.getShape().num_elements();
96 const int thread_count =
97 _external_context->ruy_context()->max_num_threads() < static_cast<int>(num_elements)
98 ? _external_context->ruy_context()->max_num_threads()
99 : num_elements;
100
101 std::vector<PermuteWorkerTask> tasks;
102 auto start = 0;
103 for (auto i = 0; i < thread_count; ++i)
104 {
105 int end = start + (num_elements - start) / (thread_count - i);
106 tasks.emplace_back(src_tensor.buffer(), dst_tensor.buffer(), start * data_size,
107 start * data_size, (end - start) * data_size);
108 start = end;
109 }
110 assert(tasks.size() >= 1);
111 _tasks_map[src] = std::move(tasks);
112 }
113 else
114 {
115 auto loop_shape = src_tensor.getShape();
116
117 auto copy_axis = loop_shape.rank() - 1;
118 copy_axis = copy_axis < 0 ? 1 : copy_axis;
119 const auto copy_len = loop_shape.dim(copy_axis) * data_size;
120 loop_shape.dim(copy_axis) = 1;
121
122 appendPermuteTasks(src, dst, loop_shape, copy_len);
123 }
124 }
125 else
126 {
127 assert(src_tensor.getShape().rank() == 4 &&
128 (permute_type == PermuteType::NHWC_TO_NCHW ||
129 permute_type == PermuteType::NCHW_TO_NHWC));
130 const auto loop_shape = src_tensor.getShape();
131 const auto copy_len = data_size;
132
133 appendPermuteTasks(src, dst, loop_shape, copy_len);
134 }
135 });
136 };
137 src->access(fn);
138 src_it++;
139 dst_it++;
140 src_offsets_it++;
141 dst_offsets_it++;
142 }
143 }
144}

146void PermuteLayer::appendPermuteTasks(const ITensor *src_tensor, ITensor *dst_tensor,
147 const ir::Shape &loop_shape, size_t size)
148{
149 size_t distributed_dim = 0;
150 auto src_shape = src_tensor->getShape();
151 if (src_tensor->layout() == dst_tensor->layout())
152 {
153 for (int i = 1; i < src_shape.rank() - 1; ++i)
154 {
155 distributed_dim = src_shape.dim(distributed_dim) < src_shape.dim(i) ? i : distributed_dim;
156 }
157 }
158 const auto distributed_dim_val = src_shape.dim(distributed_dim);
159 const int thread_count =
160 _external_context->ruy_context()->max_num_threads() < static_cast<int>(distributed_dim_val)
161 ? _external_context->ruy_context()->max_num_threads()
162 : distributed_dim_val;
163 // NOTE Do not remove this assertion. It would cause performance degradation by new threads to be
164 // created in the context's thread pool
165 assert(thread_count <= _external_context->ruy_context()->max_num_threads());
166
167 std::vector<PermuteWorkerTask> tasks;
168 int start = 0;
169 auto one_thread_loop_shape = loop_shape;
170 for (auto i = 0; i < thread_count; ++i)
171 {
172 ir::Coordinates start_coords(one_thread_loop_shape.rank());
173 start_coords.set(distributed_dim, start);
174 int end = start + (distributed_dim_val - start) / (thread_count - i);
175 one_thread_loop_shape.dim(distributed_dim) = end - start;
176 tasks.emplace_back(*src_tensor, *dst_tensor, start_coords, one_thread_loop_shape, size);
177 start = end;
178 }
179 assert(tasks.size() >= 1);
180 _tasks_map[src_tensor] = std::move(tasks);
181}

183void PermuteLayer::runPermuteTasks(backend::ITensor *src, uint8_t *dst_buffer)
184{
185 assert(src->getShape().num_elements() * ir::sizeOfDataType(src->data_type()) <=
186 src->total_size());
187 std::vector<PermuteWorkerTask> &tasks = _tasks_map.at(src);
188 for (size_t i = 0; i < tasks.size(); ++i)
189 {
190 tasks.at(i).setBuffers(src->buffer(), dst_buffer);
191 }
192 assert(tasks.size() >= 1);
193 _external_context->ruy_context()->mutable_thread_pool()->Execute(tasks.size(), tasks.data());
194}

196void PermuteLayer::run()
197{
198 assert(_src_tensors.size() == _dst_tensors.size());
199 // PermuteLayer infers dynamic shape inside itself whenever run is called for the following
200 // reasons:
201 // 1. PermuteLayer has to access dynamic tensor manager for input/output tensors of other backends
202 // 2. Other controlflow operation(If/While) uses this layout for copying tensors of other
203 // subgraphs(with other backends)
204 // 3. This infering code is placed here to avoid duplicated code that can be caused by above 2
205 // reasons
206
207 // check if output is not dynamic
208 for (size_t i = 0; i < _src_tensors.size(); ++i)
209 {
210 auto dst_tensor = _dst_tensors.at(i);
211 auto src_tensor = _src_tensors.at(i);
212 if (src_tensor->is_dynamic() || dst_tensor->is_dynamic())
213 {
214 // getting output shape
215 auto src_shape = src_tensor->getShape();
216
217 // set output shape and output buffer
218 ir::Shape new_shape =
219 exec::convertShape(src_shape, src_tensor->layout(), dst_tensor->layout());
220
221 try
222 {
223 if (!dst_tensor->applyShape(new_shape))
224 throw std::runtime_error{
225 "Error: PermuteLayer: output's TensorManager does not support dynamic tensor"};
226 assert(dst_tensor->buffer() != nullptr);
227 }
228 catch (const std::out_of_range &e)
229 {
230 std::cerr << "Error: out_of_range in PermuteLayer: output's TensorManager does not support "
231 "dynamic tensor"
232 << '\n';
233 throw;
234 }
235 }
236 assert(exec::convertShape(src_tensor->getShape(), src_tensor->layout(), dst_tensor->layout()) ==
237 dst_tensor->getShape());
238 }
239 assert(_src_tensors.size() == _dst_tensors.size());
240 assert(_src_tensors.size() == _src_tensors_offsets.size());
241 assert(_dst_tensors.size() == _dst_tensors_offsets.size());
242 auto src_it = _src_tensors.begin();
243 auto dst_it = _dst_tensors.begin();
244 auto src_offsets_it = _src_tensors_offsets.begin();
245 auto dst_offsets_it = _dst_tensors_offsets.begin();
246 while (src_it != _src_tensors.end())
247 {
248 auto src = *src_it;
249 auto dst = *dst_it;
250 auto &src_offsets = *src_offsets_it;
251 auto &dst_offsets = *dst_offsets_it;
252
253 if (src->total_size() == 0)
254 {
255 assert(dst->total_size() == 0);
256 }
257 else
258 {
259 if (src != dst)
260 {
261 // Conditions to run permutation with multithreading
262 // 1. The tasks for multithreathing was created
263 // 2. The tasks's size > 1
264 // 3. Both tensors are not dynamic
265 // 4. Data types of both tensors are different
266 if (_tasks_map.find(src) == _tasks_map.end() || _tasks_map.at(src).size() == 1 ||
267 src->is_dynamic() || dst->is_dynamic() ||
268 underlying_type(src->data_type()) != underlying_type(dst->data_type()))
269 {
270 permute(src, dst, src->getShape().rank(), src_offsets, dst_offsets);
271 }
272 // If dst is subtensor, we have to use clEnqueueMapBuffer instead of clEnqueueWirteBuffer
273 else if (dst->needMemoryMap() && !dst->is_subtensor())
274 {
275 if (!src->has_padding() && !dst->has_padding() && src->layout() == dst->layout())
276 {
277 // This is more effective than multi-threading
278 src->access([&](backend::ITensor &) { dst->enqueueWriteBuffer(src->buffer(), false); });
279 }
280 else
281 {
282 // TODO Optimize this block in case of that padding size of dst is big.
283 _buffers_map[dst].reserve(dst->total_size());
284 auto dst_buffer = _buffers_map[dst].data();
285
286 src->access([&](backend::ITensor &) { runPermuteTasks(src, dst_buffer); });
287 dst->enqueueWriteBuffer(dst_buffer, false);
288 }
289 }
290 else if (src->needMemoryMap() && !src->is_subtensor() && !src->has_padding() &&
291 !dst->has_padding() && src->layout() == dst->layout())
292 {
293 // This is more effective than multi-threading
294 assert(!dst->needMemoryMap());
295 dst->access([&](backend::ITensor &) { src->enqueueReadBuffer(dst->buffer(), true); });
296 }
297 else
298 {
299 auto fn = [&](backend::ITensor &) {
300 dst->access([&](backend::ITensor &) { runPermuteTasks(src, dst->buffer()); });
301 };
302 src->access(fn);
303 }
304 }
305 }
306 src_it++;
307 dst_it++;
308 src_offsets_it++;
309 dst_offsets_it++;
310 }
311}
