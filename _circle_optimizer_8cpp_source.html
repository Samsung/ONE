96class OptimizeOptionsImpl final : public luci::CircleOptimizer::Options
97{
98public:
99 void enable(Algorithm) final;
100 void param(AlgorithmParameters, const std::string &) final;
101 const std::string param(AlgorithmParameters) const final;
102 bool query(Algorithm) final;
103
104private:
105 std::vector<Algorithm> _algorithms;
106 std::map<AlgorithmParameters, const std::string> _algorithm_params;
107};
108
109void OptimizeOptionsImpl::enable(Algorithm algo) { _algorithms.push_back(algo); }
110
111void OptimizeOptionsImpl::param(AlgorithmParameters param, const std::string &str)
112{
113 _algorithm_params.insert(std::pair<AlgorithmParameters, const std::string>(param, str));
114}
115
116const std::string OptimizeOptionsImpl::param(AlgorithmParameters param) const
117{
118 auto param_str = _algorithm_params.find(param);
119 if (param_str != _algorithm_params.end())
120 {
121 return param_str->second;
122 }
123 else
124 {
125 return std::string();
126 }
127}
128
129bool OptimizeOptionsImpl::query(Algorithm algo)
130{
131 std::vector<Algorithm>::iterator it = std::find(_algorithms.begin(), _algorithms.end(), algo);
132 if (it == _algorithms.end())
133 return false;
134
135 return true;
136}

139void convert_nchw_to_nhwc(loco::Graph *g, bool preserve_input, bool preserve_output, bool fuse_fc)
140{
141 logo::Phase phase;
142
143 phase.emplace_back(std::make_unique<logo::RemoveDeadNodeWithQueryPass>());
144 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
145 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
146
147 // Resolve custom Ops
148 phase.emplace_back(std::make_unique<luci::ResolveCustomOpAddPass>());
149 phase.emplace_back(std::make_unique<luci::ResolveCustomOpBatchMatMulPass>());
150 phase.emplace_back(std::make_unique<luci::ResolveCustomOpMatMulPass>());
151 phase.emplace_back(std::make_unique<luci::ResolveCustomOpMaxPoolWithArgmaxPass>());
152 phase.emplace_back(std::make_unique<luci::ResolveCustomOpSplitVPass>());
153
154 // Fuse FullyConnected with Add
155 // Why we perform FuseAddWithFullyConnectedPass before ConvertNCHWToNHWCPass?
156 // FullyConnected Op's layout is not changed in ConvertNCHWToNHWCPass, while
157 // Add Op's layer is changed from NCHW to NHWC.
158 // This disables fusion of Add and FullyConnected after ConvertNCHWToNHWC.
159 if (fuse_fc)
160 phase.emplace_back(std::make_unique<luci::FuseAddWithFullyConnectedPass>());
161
162 phase.emplace_back(
163 std::make_unique<luci::ConvertNCHWToNHWCPass>(preserve_input, preserve_output));
164
165 ProgressReporter prog(g, logo::PhaseStrategy::Restart);
166 logo::PhaseRunner<logo::PhaseStrategy::Restart> phase_runner{g};
167 phase_runner.attach(&prog);
168 phase_runner.run(phase);
169}

186void CircleOptimizer::optimize(luci::Module *m) const
187{
188 luci::Phase phase;
189
190 // Following passes are needed everytime when other passes create new node or modify some nodes.
191 phase.emplace_back(std::make_unique<luci::CircleShapeInferencePass>());
192 phase.emplace_back(std::make_unique<luci::CircleTypeInferencePass>());
193
194 if (_options->query(Options::Algorithm::FuseBCQ))
195 {
196 phase.emplace_back(std::make_unique<FuseBCQPass>());
197 }
198
199 ModuleProgressReporter prog(m, logo::PhaseStrategy::Restart);
200 PhaseRunner<logo::PhaseStrategy::Restart> phase_runner{m};
201 phase_runner.attach(&prog);
202 phase_runner.run(phase);
203}

439void CircleOptimizer::sparsify(loco::Graph *g) const
440{
441 if (_options->query(Options::Algorithm::SparsifyTensorPass))
442 {
443 std::string tensor_name = _options->param(Options::AlgorithmParameters::Sparsify_tensor_name);
444 std::string str_tarversal_order =
445 _options->param(Options::AlgorithmParameters::Sparsify_traversal_order);
446 std::string str_format = _options->param(Options::AlgorithmParameters::Sparsify_format);
447 std::string str_block_size = _options->param(Options::AlgorithmParameters::Sparsify_block_size);
448 std::string str_block_map = _options->param(Options::AlgorithmParameters::Sparsify_block_map);
449
450 // traversal order
451 std::vector<int32_t> traversal_order = pepper::csv_to_vector<int32_t>(str_tarversal_order);
452 // format
453 std::vector<DimensionType> format;
454 std::istringstream is(str_format);
455 for (char c; is >> c;)
456 {
457 assert(c != ',');
458 if (c == 'd')
459 format.push_back(DimensionType::DENSE);
460 else if (c == 's')
461 format.push_back(DimensionType::SPARSE_CSR);
462 if (is.peek() == ',')
463 is.ignore();
464 }
465 // block size
466 std::vector<int32_t> block_size = pepper::csv_to_vector<int32_t>(str_block_size);
467 // block map
468 std::vector<int32_t> block_map = pepper::csv_to_vector<int32_t>(str_block_map);
469
470 luci::SparsifyTensorPass sparsifier{tensor_name, traversal_order, format, block_size,
471 block_map};
472 sparsifier.run(g);
473 }
474}
