49bool resolve_constant_shape(loco::Graph *graph, moco::TFShape *shape_node)
50{
51 auto input_shape = moco::node_shape(shape_node->input());
52
53 // Check condition
54 if (input_shape.domain() == loco::Domain::Unknown)
55 {
56 // Cannot resolve without known input_shape
57 return false;
58 }
59
60 auto input_tensor_shape = input_shape.as<loco::TensorShape>();
61
62 auto shape_rank = input_tensor_shape.rank();
63 for (uint32_t axis = 0; axis < shape_rank; ++axis)
64 {
65 if (!input_tensor_shape.dim(axis).known())
66 {
67 // Cannot resolve with unknown dimension
68 return false;
69 }
70 }
71
72 // Make TFConst to replace TFShape
73 auto const_node = graph->nodes()->create<moco::TFConst>();
74
75 // set dtype
76 auto dtype = shape_node->dtype();
77 const_node->dtype(dtype);
78
79 // set shape
80 const_node->rank(1);
81 const_node->dim(0) = shape_rank;
82
83 // set data
84 if (dtype == loco::DataType::S32)
85 {
86 // TODO Better to make template for this when support new dtype
87 const_node->size<loco::DataType::S32>(shape_rank);
88 for (uint32_t axis = 0; axis < shape_rank; ++axis)
89 {
90 int32_t dim = (int32_t)input_tensor_shape.dim(axis).value();
91 if (!(dim > 0))
92 {
93 throw oops::UserExn("Invalid input shape", shape_node->name());
94 }
95 const_node->at<loco::DataType::S32>(axis) = dim;
96 }
97 }
98 else
99 {
100 throw oops::UserExn("Unsupported data type", shape_node->name());
101 }
102
103 // replace
104 loco::replace(shape_node).with(const_node);
105
106 return true;
107}

114bool ResolveConstantShape::run(loco::Graph *graph)
115{
116 bool changed = false;
117 for (auto node : loco::active_nodes(loco::output_nodes(graph)))
118 {
119 if (auto shape_node = as<moco::TFShape>(node))
120 {
121 if (resolve_constant_shape(graph, shape_node))
122 changed = true;
123 }
124 }
125
126 return changed;
127}
