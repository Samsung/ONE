18import h5py as h5
19import numpy as np
20import argparse
21
22from circle.Model import Model
23from RandomDataGenerator import RandomDataGenerator
24
25#
26# This script generates a pack of random input data (.h5) expected by the input circle model
27#
28# Basic usage:
29# gen_h5_random_inputs.py --model <path/to/circle/model> --num_data <number/of/data> --output <path/to/output/data>
30# ex: gen_h5_random_inputs.py --model add.circle --num_data 3 --output add.circle.input.h5
31# (This will create add.circle.input.h5 composed of three random inputs in the same directory as the model)
32parser = argparse.ArgumentParser()
33parser.add_argument('--model', type=str, required=True)
34parser.add_argument('--num_data', type=int, required=True)
35parser.add_argument('--output', type=str, required=True)
36args = parser.parse_args()
37
38model = args.model
39num_data = args.num_data
40output_path = args.output
41
42with open(model, 'rb') as f:
43 buf = f.read()
44 circle_model = Model.GetRootAsModel(buf, 0)
45
46# Assume one subgraph
47assert (circle_model.SubgraphsLength() == 1)
48graph = circle_model.Subgraphs(0)
49inputs = graph.InputsAsNumpy()
50
51# Create h5 file
52h5_file = h5.File(output_path, 'w')
53group = h5_file.create_group("value")
54group.attrs['desc'] = "Input data for " + model
55
56# Generate random data
57for i in range(num_data):
58 sample = group.create_group(str(i))
59 for j in range(len(inputs)):
60 input_index = inputs[j]
61 tensor = graph.Tensors(input_index)
62 g = RandomDataGenerator(tensor.ShapeAsNumpy())
63 input_data = g.gen(tensor.Type())
64 sample.create_dataset(str(j), data=input_data)
65
66h5_file.close()
