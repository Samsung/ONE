33ExplicitPadding makeExplicitPadding(Operands &operands, const OperandIndex &left_index,
34 const OperandIndex &right_index, const OperandIndex &top_index,
35 const OperandIndex &bottom_index)
36{
37 auto left = operands.at(left_index).asScalar<int32_t>();
38 auto right = operands.at(right_index).asScalar<int32_t>();
39 auto top = operands.at(top_index).asScalar<int32_t>();
40 auto bottom = operands.at(bottom_index).asScalar<int32_t>();
41
42 if (left < 0 || right < 0 || top < 0 || bottom < 0)
43 {
44 throw std::runtime_error{"Cannot handle negative explicit padding value"};
45 }
46
47 ExplicitPadding param;
48 param.left = static_cast<uint32_t>(left);
49 param.right = static_cast<uint32_t>(right);
50 param.top = static_cast<uint32_t>(top);
51 param.bottom = static_cast<uint32_t>(bottom);
52
53 return param;
54}

56Stride makeStride(Operands &operands, const OperandIndex &horizontal_index,
57 const OperandIndex &vertical_index)
58{
59 auto horizontal = operands.at(horizontal_index).asScalar<int32_t>();
60 auto vertical = operands.at(vertical_index).asScalar<int32_t>();
61
62 if (vertical < 0 || horizontal < 0)
63 {
64 throw std::runtime_error{"Cannot handle negative stride value"};
65 }
66
67 Stride stride;
68 stride.horizontal = static_cast<uint32_t>(horizontal);
69 stride.vertical = static_cast<uint32_t>(vertical);
70
71 return stride;
72}

85Activation getActivation(Operands &operands, const OperandIndex index)
86{
87 switch (operands.at(index).asScalar<int32_t>())
88 {
89 case 0:
90 return Activation::NONE;
91 case 1:
92 return Activation::RELU;
93 case 2:
94 return Activation::RELU1;
95 case 3:
96 return Activation::RELU6;
97 case 4:
98 return Activation::TANH;
99 case 6:
100 return Activation::SIGMOID;
101 default:
102 throw std::runtime_error("Unsupported activation type");
103 }
104}

212OperationFactory::Generator
213getPool2DGenerator(const onert::ir::operation::Pool2D::PoolType pool_type)
214{
215 return [pool_type](const OperationFactory::Param &init_param, Operands &operands) {
216 assert(init_param.input_count == 7 || init_param.input_count == 10);
217 assert(init_param.output_count == 1);
218
219 // In common
220 // 0 -> IFM Tensor Index
221 OperandIndexSequence inputs{init_param.inputs[0]};
222 OperandIndexSequence outputs{init_param.outputs[0]};
223
224 operation::Pool2D::Param param;
225 param.op_type = pool_type;
226 if (init_param.input_count == 7) // support implicit padding
227 {
228 // Each input should be interpreted as follows:
229 //
230 // 1 -> Padding Code (ANEURALNETWORKS_PADDING_SAME or ANEURALNETWORKS_PADDING_VALID) Index
231 // 2 -> Horizontal (over width) Stride Index
232 // 3 -> Vertial (over height) Stride Index
233 // 4 -> Filter Width Index
234 // 5 -> Filter Height Index
235 // 6 -> FuseCode (activation) Index
236
237 const auto padding_index = OperandIndex{init_param.inputs[1]};
238 const auto hstride_index = OperandIndex{init_param.inputs[2]};
239 const auto vstride_index = OperandIndex{init_param.inputs[3]};
240 const auto kw_index = OperandIndex{init_param.inputs[4]};
241 const auto kh_index = OperandIndex{init_param.inputs[5]};
242 const auto activation_index = OperandIndex{init_param.inputs[6]};
243
244 param.padding.type =
245 NNAPIConvert::getPaddingType(operands.at(padding_index).asScalar<PaddingCode>());
246 param.stride = makeStride(operands, hstride_index, vstride_index);
247 param.kw = getUint32Scalar(operands, kw_index);
248 param.kh = operands.at(kh_index).asScalar<uint32_t>();
249 param.activation =
250 NNAPIConvert::getFusedActivation(operands.at(activation_index).asScalar<FuseCode>());
251 }
252 else // support explicit padding
253 {
254 // Each input should be interpreted as follows:
255 //
256 // 1 -> Padding_left index
257 // 2 -> Padding_right index
258 // 3 -> Padding_top index
259 // 4 -> Padding_bottom index
260 // 5 -> Horizontal (over width) Stride Index
261 // 6 -> Vertial (over height) Stride Index
262 // 7 -> Filter Width Index
263 // 8 -> Filter Height Index
264 // 9 -> FuseCode (activation) Index
265
266 const auto padding_left_index = OperandIndex{init_param.inputs[1]};
267 const auto padding_right_index = OperandIndex{init_param.inputs[2]};
268 const auto padding_top_index = OperandIndex{init_param.inputs[3]};
269 const auto padding_bottom_index = OperandIndex{init_param.inputs[4]};
270 const auto hstride_index = OperandIndex{init_param.inputs[5]};
271 const auto vstride_index = OperandIndex{init_param.inputs[6]};
272 const auto kw_index = OperandIndex{init_param.inputs[7]};
273 const auto kh_index = OperandIndex{init_param.inputs[8]};
274 const auto activation_index = OperandIndex{init_param.inputs[9]};
275
276 param.padding.type = PaddingType::EXPLICIT;
277 param.padding.param = makeExplicitPadding(operands, padding_left_index, padding_right_index,
278 padding_top_index, padding_bottom_index);
279 param.stride = makeStride(operands, hstride_index, vstride_index);
280 param.kw = getUint32Scalar(operands, kw_index);
281 param.kh = getUint32Scalar(operands, kh_index);
282 param.activation =
283 NNAPIConvert::getFusedActivation(operands.at(activation_index).asScalar<FuseCode>());
284 }
285 return new operation::Pool2D{inputs, outputs, param};
286 };

593 // 6 -> Padding_bottom index
594 // 7 -> Stride (width) Index
595 // 8 -> Stride (height) INdex
596 // 9 -> Activation Index

598 const auto padding_left_index = OperandIndex{init_param.inputs[3]};

616 }
617 else if (init_param.input_count == 13) // support dilation
618 {
619 // Each input should be interpreted as follows:
620 //
621 // 3 -> Padding_left Index
622 // 4 -> Padding_right Index
623 // 5 -> Padding_top Index
624 // 6 -> Padding_bottom Index
625 // 7 -> Stride (width) Index
626 // 8 -> Stride (height) Index
627 // 9 -> Activation Index

629 // 12 -> Dilation (height_factor) INdex
630
631 const auto padding_left_index = OperandIndex{init_param.inputs[3]};
632 const auto padding_right_index = OperandIndex{init_param.inputs[4]};
633 const auto padding_top_index = OperandIndex{init_param.inputs[5]};
634 const auto padding_bottom_index = OperandIndex{init_param.inputs[6]};
