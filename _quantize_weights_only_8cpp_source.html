33using IterFunc = std::function<void(uint32_t *, loco::TensorShape &, int32_t)>;

35void iterate_per_channel(CircleConst *node, int32_t &channel_dim_index, IterFunc func)
36{
37 loco::TensorShape dimension;
38 dimension.rank(4);
39 uint32_t indices[4] = {
40 0,
41 };
42
43 if (!get_channel_dim_index(node, dimension, channel_dim_index))
44 {
45 assert(false);
46 return;
47 }
48
49 for (indices[0] = 0; indices[0] < dimension.dim(0).value(); indices[0]++)
50 {
51 for (indices[1] = 0; indices[1] < dimension.dim(1).value(); indices[1]++)
52 {
53 for (indices[2] = 0; indices[2] < dimension.dim(2).value(); indices[2]++)
54 {
55 for (indices[3] = 0; indices[3] < dimension.dim(3).value(); indices[3]++)
56 {
57 func(indices, dimension, channel_dim_index);
58 }
59 }
60 }
61 }
62}

65template <loco::DataType out_type>
66void sym_wquant_per_channel(CircleConst *node, std::vector<float> &min, std::vector<float> &max,
67 std::vector<float> &scaling_factor, std::vector<float> &nudged_min,
68 std::vector<float> &nudged_max, int32_t &channel_dim_index)
69{
70 assert(node->dtype() == loco::DataType::FLOAT32);
71 assert(out_type == loco::DataType::S4 || out_type == loco::DataType::S8 ||
72 out_type == loco::DataType::S16);
73
74 const int32_t kMaxScale = max_for_sym_quant(out_type);
75 const int32_t kMinScale = -kMaxScale;
76
77 uint32_t size = node->size<loco::DataType::FLOAT32>();
78 std::vector<int32_t> quantized_values(size);
79
80 for (size_t i = 0; i < min.size(); ++i)
81 {
82 compute_sym_scale(min[i], max[i], scaling_factor[i], nudged_min[i], nudged_max[i], out_type);
83 }
84
85 auto quantize = [&](uint32_t *indices, loco::TensorShape &dimension, int channel_dim_index) {
86 int channel_idx = indices[channel_dim_index];
87 const float scaling_factor_inv = 1.0 / scaling_factor[channel_idx];
88 auto data = node->at<loco::DataType::FLOAT32>(cal_offset(dimension, indices));
89 data = data < nudged_min[channel_idx] ? nudged_min[channel_idx] : data;
90 data = data > nudged_max[channel_idx] ? nudged_max[channel_idx] : data;
91 quantized_values[cal_offset(dimension, indices)] =
92 static_cast<int32_t>(std::round(data * scaling_factor_inv));
93 };
94
95 iterate_per_channel(node, channel_dim_index, quantize);
96
97 node->dtype(out_type); // change the type of tensor
98 node->size<out_type>(size); // resize tensor
99 for (uint32_t i = 0; i < size; ++i)
100 {
101 node->at<out_type>(i) = std::min(kMaxScale, std::max(kMinScale, quantized_values[i]));
102 }
103}

105void cal_minmax_per_channel(CircleConst *node, std::vector<float> &min, std::vector<float> &max,
106 int32_t &channel_dim_index)
107{
108 loco::TensorShape dimension;
109 dimension.rank(4);
110
111 if (!get_channel_dim_index(node, dimension, channel_dim_index))
112 {
113 throw std::runtime_error("Failed to find channel index in " + node->name());
114 }
115 auto size = dimension.dim(channel_dim_index).value();
116
117 std::vector<bool> has_min_max_value(size, false);
118 min.resize(size);
119 max.resize(size);
120
121 auto cal_minmax = [&](uint32_t *indices, loco::TensorShape &dimension, int channel_dim_index) {
122 int channel_idx = indices[channel_dim_index];
123 auto data = node->at<loco::DataType::FLOAT32>(cal_offset(dimension, indices));
124 if (has_min_max_value[channel_idx])
125 {
126 min[channel_idx] = data < min[channel_idx] ? data : min[channel_idx];
127 max[channel_idx] = data > max[channel_idx] ? data : max[channel_idx];
128 }
129 else
130 {
131 min[channel_idx] = data;
132 max[channel_idx] = data;
133 has_min_max_value[channel_idx] = true;
134 }
135 };
136
137 iterate_per_channel(node, channel_dim_index, cal_minmax);
138}

145void QuantizeWeightsOnly::quantize_weights(luci::CircleConst *weights)
146{
147 // Find min/max per channel-wise
148 if (granularity == QuantizationGranularity::ChannelWise)
149 {
150 auto quantparam = weights->quantparam();
151 if (quantparam == nullptr)
152 {
153 // Find min/max on the fly
154 // NOTE This is for the case when QuantizeDequantizeWeights is skipped
155 // TODO Reduce duplicate codes
156 std::vector<float> min;
157 std::vector<float> max;
158 int32_t channel_dim_index = 0;
159
160 cal_minmax_per_channel(weights, min, max, channel_dim_index);
161
162 std::vector<float> nudged_min(min.size());
163 std::vector<float> nudged_max(min.size());
164 std::vector<float> scaling_factor(min.size());
165 std::vector<int64_t> zp(min.size());
166
167 if (output_type == loco::DataType::S4)
168 {
169 sym_wquant_per_channel<loco::DataType::S4>(weights, min, max, scaling_factor, nudged_min,
170 nudged_max, channel_dim_index);
171 }
172 else if (output_type == loco::DataType::S8)
173 {
174 sym_wquant_per_channel<loco::DataType::S8>(weights, min, max, scaling_factor, nudged_min,
175 nudged_max, channel_dim_index);
176 }
177 else if (output_type == loco::DataType::S16)
178 {
179 sym_wquant_per_channel<loco::DataType::S16>(weights, min, max, scaling_factor, nudged_min,
180 nudged_max, channel_dim_index);
181 }
182 else
183 {
184 throw std::runtime_error("Weights-only quantization supports s8 and s16");
185 }
186
187 auto quantparam = std::make_unique<CircleQuantParam>();
188 quantparam->scale = scaling_factor;
189 quantparam->zerop = zp;
190 quantparam->quantized_dimension = channel_dim_index;
191 weights->quantparam(std::move(quantparam));
192
193 return;
194 }
195 }
196 else
197 throw std::runtime_error("Weights-only quantization does not support layer-wise");
198}
