65 {
66 if (_tensors.type(n) != ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT)
67 {
68 throw std::runtime_error{"type mismatch"};

71 using nncc::core::ADT::tensor::LexicalLayout;
72 using nncc::core::ADT::tensor::make_overlay;
