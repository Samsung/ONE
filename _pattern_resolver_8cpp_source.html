70 assert(candidate);
71 mul_as_terminal = candidate;
72 }
73
74public:
75 bool matched();

77 std::vector<luci::CircleNode *> get_q16_nodes()
78 {

83
84public:
85 loco::Node *ifm = nullptr;
86 luci::CircleMean *mean_of_ifm = nullptr; // = |ifm|
87 luci::CircleSub *sub = nullptr; // = ifm - |ifm|
88 luci::CircleMul *sub_squared = nullptr; // = (ifm - |ifm|)^2
89 luci::CircleMean *mean_as_variance = nullptr; // = |(ifm - |ifm|)^2|
90 luci::CircleAdd *add_eps = nullptr; // = |(ifm - |ifm|)^2| + eps
91 luci::CircleRsqrt *rsqrt = nullptr; // = 1.0 / sqrt(|(ifm - |ifm|)^2| + eps)
92 luci::CircleMul *mul_as_terminal = nullptr; // = (ifm - |ifm|) / sqrt(|(ifm - |ifm|)^2| + eps)
93};

99bool LayerNormPattern::matched()
100{
101 sub = dynamic_cast<luci::CircleSub *>(mul_as_terminal->x());
102 rsqrt = dynamic_cast<luci::CircleRsqrt *>(mul_as_terminal->y());
103 if (!sub || !rsqrt)
104 {
105 sub = dynamic_cast<luci::CircleSub *>(mul_as_terminal->y());
106 rsqrt = dynamic_cast<luci::CircleRsqrt *>(mul_as_terminal->x());
107 }
108 CHECK_OR_FALSE(rsqrt != nullptr && sub != nullptr);
109
110 ifm = dynamic_cast<luci::CircleNode *>(sub->x());
111 mean_of_ifm = dynamic_cast<luci::CircleMean *>(sub->y());
112 CHECK_OR_FALSE(ifm != nullptr && mean_of_ifm != nullptr);
113
114 add_eps = dynamic_cast<luci::CircleAdd *>(rsqrt->x());
115 CHECK_OR_FALSE(add_eps != nullptr);
116
117 auto const *eps = dynamic_cast<luci::CircleConst *>(add_eps->x());
118 mean_as_variance = dynamic_cast<luci::CircleMean *>(add_eps->y());
119 if (!eps || !mean_as_variance)
120 {
121 eps = dynamic_cast<luci::CircleConst *>(add_eps->y());
122 mean_as_variance = dynamic_cast<luci::CircleMean *>(add_eps->x());
123 }
124
125 CHECK_OR_FALSE(eps != nullptr && mean_as_variance != nullptr);
126
127 // eps should be scalar value
128 CHECK_OR_FALSE(eps->size<loco::DataType::FLOAT32>() == 1);
129
130 sub_squared = dynamic_cast<luci::CircleMul *>(mean_as_variance->input());
131 CHECK_OR_FALSE(sub_squared != nullptr);
132
133 // check that sub_squared = sub * sub
134 CHECK_OR_FALSE(sub_squared->x() == sub_squared->y() && sub_squared->x() == sub);
135
136 auto const mean_as_variance_indices =
137 dynamic_cast<luci::CircleConst *>(mean_as_variance->reduction_indices());
138 auto const mean_of_ifm_indices =
139 dynamic_cast<luci::CircleConst *>(mean_of_ifm->reduction_indices());
140
141 // check validity of reduction indices
142 CHECK_OR_FALSE(mean_of_ifm_indices != nullptr && mean_as_variance_indices != nullptr);
143
144 // check dtype of reduction indices
145 CHECK_OR_FALSE(mean_of_ifm_indices->dtype() == loco::DataType::S32 &&
146 mean_as_variance_indices->dtype() == loco::DataType::S32);
147
148 // reduction indices of both mean operations must be the same
149 CHECK_OR_FALSE(mean_as_variance_indices->size<loco::DataType::S32>() ==
150 mean_of_ifm_indices->size<loco::DataType::S32>());
151
152 std::set<int32_t> set_of_mean_as_variance_indices;
153 std::set<int32_t> set_of_mean_of_ifm_indices;
154 for (uint32_t index = 0; index < mean_as_variance_indices->size<loco::DataType::S32>(); index++)
155 {
156 set_of_mean_as_variance_indices.insert(
157 mean_as_variance_indices->at<loco::DataType::S32>(index));
158 set_of_mean_of_ifm_indices.insert(mean_of_ifm_indices->at<loco::DataType::S32>(index));
159 }
160 // now make sure that reduction indices of mean_as_variance are the same as mean_of_ifm
161 CHECK_OR_FALSE(set_of_mean_as_variance_indices == set_of_mean_of_ifm_indices);
162
163 return true;
164}

192class SoftmaxPattern final
193{
194public:
195 SoftmaxPattern(luci::CircleDiv *candidate)
196 {
197 assert(candidate);
198 _div_as_terminal = candidate;
199 }
200
201public:
202 bool matched();
203
204 std::vector<luci::CircleNode *> get_q16_nodes() { return {_sub, _exp}; }
205
206 std::vector<luci::CircleNode *> get_q8_nodes() { return {_ifm, _max, _sum, _div_as_terminal}; }
207
208public:
209 luci::CircleNode *_ifm = nullptr; // input feature map
210 luci::CircleReduceMax *_max = nullptr; // = max(_ifm)
211 luci::CircleSub *_sub = nullptr; // = _ifm - max(_ifm)
212 luci::CircleExp *_exp = nullptr; // = exp(_ifm - max(_ifm))
213 luci::CircleSum *_sum = nullptr; // = sum(exp(_ifm - max(_ifm)))
214 luci::CircleDiv *_div_as_terminal = nullptr; // = exp(_ifm - max(_ifm))/sum(exp(_ifm - max(_ifm)))
215};

217bool SoftmaxPattern::matched()
218{
219 _exp = dynamic_cast<luci::CircleExp *>(_div_as_terminal->x());
220 _sum = dynamic_cast<luci::CircleSum *>(_div_as_terminal->y());
221 CHECK_OR_FALSE(_exp != nullptr && _sum != nullptr);
222
223 CHECK_OR_FALSE(_sum->input() == _exp);
224 CHECK_OR_FALSE(_sum->keep_dims() == true);
225
226 auto const sum_indices = dynamic_cast<luci::CircleConst *>(_sum->reduction_indices());
227 CHECK_OR_FALSE(sum_indices != nullptr);
228
229 _sub = dynamic_cast<luci::CircleSub *>(_exp->x());
230 CHECK_OR_FALSE(_sub != nullptr);
231
232 _ifm = loco::must_cast<luci::CircleNode *>(_sub->x());
233
234 _max = dynamic_cast<luci::CircleReduceMax *>(_sub->y());
235 CHECK_OR_FALSE(_max != nullptr);
236
237 CHECK_OR_FALSE(_max->input() == _ifm);
238 CHECK_OR_FALSE(_max->keep_dims() == true);
239
240 auto const max_indices = dynamic_cast<luci::CircleConst *>(_max->reduction_indices());
241 CHECK_OR_FALSE(max_indices != nullptr);
242
243 // check dtype of reduction indices
244 CHECK_OR_FALSE(max_indices->dtype() == loco::DataType::S32 &&
245 sum_indices->dtype() == loco::DataType::S32);
246
247 // reduction of max and sum must be done over the last (channel) dimension
248 {
249 CHECK_OR_FALSE(max_indices->size<loco::DataType::S32>() == 1 &&
250 sum_indices->size<loco::DataType::S32>() == 1);
251
252 auto const rank = _ifm->rank();
253 int32_t last_dim = (rank == 0) ? 0 : rank - 1;
254
255 CHECK_OR_FALSE(max_indices->at<loco::DataType::S32>(0) == -1 ||
256 max_indices->at<loco::DataType::S32>(0) == last_dim);
257
258 CHECK_OR_FALSE(sum_indices->at<loco::DataType::S32>(0) == -1 ||
259 sum_indices->at<loco::DataType::S32>(0) == last_dim);
260 }
261
262 return true;
263}

269std::map<luci::CircleNode *, LayerParam>
270Q8LayerNormWithQ16VarianceResolver::resolve(const luci::Module *module)
271{
272 if (!module)
273 {
274 throw std::runtime_error("No module for pattern resolving");
275 }
276
277 std::map<luci::CircleNode *, LayerParam> nodes_params;
278 for (size_t idx = 0; idx < module->size(); ++idx)
279 {
280 auto graph = module->graph(idx);
281
282 for (auto node : loco::active_nodes(loco::output_nodes(graph)))
283 {
284 auto const mul = dynamic_cast<luci::CircleMul *>(node);
285 if (!mul)
286 continue;
287
288 LayerNormPattern pattern(mul);
289 if (!pattern.matched())
290 continue;
291
292 // set quantization parameters of recognized pattern
293 for (auto q16_node : pattern.get_q16_nodes())
294 {
295 LayerParam param = {q16_node->name(), "int16", "channel"};
296 nodes_params[q16_node] = param;
297 }
298
299 for (auto q8_node : pattern.get_q8_nodes())
300 {
301 LayerParam param = {q8_node->name(), "uint8", "channel"};
302 nodes_params[q8_node] = param;
303 }
304 }
305 }
306
307 return nodes_params;
308}
