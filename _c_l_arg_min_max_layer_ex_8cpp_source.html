60Status CLArgMinMaxLayerEx::validate(const ITensorInfo *input, int axis, const ITensorInfo *output,
61 const ReductionOperation &op)
62{
63 ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
64 ARM_COMPUTE_RETURN_ERROR_ON_MSG(op != ReductionOperation::ARG_IDX_MAX &&
65 op != ReductionOperation::ARG_IDX_MIN,
66 "Invalid reduction operation");
67 ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis >= static_cast<int>(TensorShape::num_max_dimensions),
68 "Reduction axis greater than max number of dimensions");
69 ARM_COMPUTE_RETURN_ERROR_ON_MSG(axis > 3, "Unsupported reduction axis");
70 const unsigned int num_of_stages =
71 utils::calculate_number_of_stages_only_x_axis(input->dimension(0), axis);
72
73 DataType output_data_type = DataType::S32;
74 TensorInfo not_reshaped_output;
75 const auto input_num_channles = input->num_channels();
76 const auto input_qinfo = input->quantization_info();
77
78 if (output->total_size() != 0)
79 {
80 output_data_type = output->data_type();
81 const TensorInfo expected_output_shape =
82 output->clone()->set_tensor_shape(arm_compute::misc::shape_calculator::compute_reduced_shape(
83 input->tensor_shape(), axis, false));
84 ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(&expected_output_shape, output);
85 }
86
87 auto shape_before_reshape = input->tensor_shape();
88 shape_before_reshape.set(axis, 1);
89 auto initialize_tensorinfo = [](TensorInfo &ti, TensorShape shape, DataType data_type,
90 int num_channels, QuantizationInfo qinfo) {
91 ti.set_data_type(data_type)
92 .set_tensor_shape(shape)
93 .set_num_channels(num_channels)
94 .set_quantization_info(qinfo);
95 };
96
97 initialize_tensorinfo(not_reshaped_output, shape_before_reshape, output_data_type,
98 input_num_channles, input_qinfo);
99
100 if (num_of_stages == 1)
101 {
102 ARM_COMPUTE_RETURN_ON_ERROR(
103 CLArgMinMaxLayerKernelEx::validate(input, nullptr, &not_reshaped_output, axis, op));
104 }
105 else
106 {
107 // Create temporary tensor infos
108 std::vector<TensorInfo> sums_vector(num_of_stages - 1);
109
110 // Create intermediate tensor info
111 TensorShape shape{input->tensor_shape()};
112
113 for (unsigned int i = 0; i < num_of_stages - 1; i++)
114 {
115 shape.set(0, ceil(shape.x() / 128.f));
116 sums_vector[i].set_data_type(input->data_type());
117 sums_vector[i].set_tensor_shape(shape);
118 sums_vector[i].set_num_channels(input->num_channels());
119 }
120
121 // Validate ReductionOperation only on first kernel
122 ARM_COMPUTE_RETURN_ON_ERROR(
123 CLArgMinMaxLayerKernelEx::validate(input, nullptr, &sums_vector[0], axis, op));
124
125 // Validate ReductionOperation on intermediate stages
126 for (unsigned int i = 1; i < num_of_stages - 1; ++i)
127 {
128 ARM_COMPUTE_RETURN_ON_ERROR(
129 CLArgMinMaxLayerKernelEx::validate(input, &sums_vector[i - 1], &sums_vector[i], axis, op));
130 }
131
132 // Validate ReductionOperation on the last stage
133 const unsigned int last_stage = num_of_stages - 1;
134 ARM_COMPUTE_RETURN_ON_ERROR(CLArgMinMaxLayerKernelEx::validate(
135 input, &sums_vector[last_stage - 1], &not_reshaped_output, axis, op));
136 }
137 ARM_COMPUTE_RETURN_ON_ERROR(CLReshapeLayer::validate(&not_reshaped_output, output));
138 return Status{};
139}

141void CLArgMinMaxLayerEx::configure(const ICLTensor *input, int axis, ICLTensor *output,
142 const ReductionOperation &op)
143{
144 ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
145 _num_of_stages = utils::calculate_number_of_stages_only_x_axis(input->info()->dimension(0), axis);
146 _reduction_axis = axis;
147
148 const TensorShape output_shape = arm_compute::misc::shape_calculator::compute_reduced_shape(
149 input->info()->tensor_shape(), axis, false);
150 DataType output_data_type = (output->info()->data_type() == DataType::UNKNOWN)
151 ? DataType::S32
152 : output->info()->data_type();
153 auto_init_if_empty(*output->info(), input->info()
154 ->clone()
155 ->set_tensor_shape(output_shape)
156 .set_data_type(output_data_type)
157 .reset_padding()
158 .set_is_resizable(true));
159
160 // Configure reduction operation kernels
161 _reduction_kernels_vector.resize(_num_of_stages);
162
163 _memory_group.manage(&_not_reshaped_output);
164 // Create temporary tensors
165 if (_num_of_stages == 1)
166 {
167 // Force an early initialization for int64 output type
168 TensorShape output_shape{input->info()->tensor_shape()};
169 output_shape.set(axis, 1);
170 auto_init_if_empty(*_not_reshaped_output.info(), input->info()
171 ->clone()
172 ->set_tensor_shape(output_shape)
173 .set_data_type(output_data_type)
174 .reset_padding()
175 .set_is_resizable(true));
176 _not_reshaped_output.info()->set_tensor_shape(output_shape);
177 _reduction_kernels_vector[0].configure(input, nullptr, &_not_reshaped_output, axis, op);
178 }
179 else
180 {
181 _results_vector.resize(_num_of_stages - 1);
182 TensorShape shape{input->info()->tensor_shape()};
183 for (unsigned int i = 0; i < _num_of_stages - 1; i++)
184 {
185 shape.set(0, ceil(shape.x() / 128.f));
186 _results_vector[i].allocator()->init(
187 input->info()->clone()->set_tensor_shape(shape).set_data_type(output_data_type));
188 }
189
190 // Apply ReductionOperation only on first kernel
191 _memory_group.manage(&_results_vector[0]);
192 _reduction_kernels_vector[0].configure(input, nullptr, &_results_vector[0], axis, op);
193
194 // Apply ReductionOperation on intermediate stages
195 for (unsigned int i = 1; i < _num_of_stages - 1; ++i)
196 {
197 _memory_group.manage(&_results_vector[i]);
198 _reduction_kernels_vector[i].configure(input, &_results_vector[i - 1], &_results_vector[i],
199 axis, op);
200 _results_vector[i - 1].allocator()->allocate();
201 }
202
203 // Apply ReductionOperation on the last stage
204 const unsigned int last_stage = _num_of_stages - 1;
205 _reduction_kernels_vector[last_stage].configure(input, &_results_vector[last_stage - 1],
206 &_not_reshaped_output, axis, op);
207 _results_vector[last_stage - 1].allocator()->allocate();
208 }
209 _reshape_kernel.configure(CLKernelLibrary::get().get_compile_context(), &_not_reshaped_output,
210 output);
211 _not_reshaped_output.allocator()->allocate();
212}

214void CLArgMinMaxLayerEx::run()
215{
216 MemoryGroupResourceScope scope_mg(_memory_group);
217
218 for (unsigned int i = 0; i < _num_of_stages; ++i)
219 {
220 CLScheduler::get().enqueue(_reduction_kernels_vector[i], false);
221 }
222 _reshape_kernel.run();
223}
