23bool remove_unnecessary_split(luci::CircleNode *node)
24{
25 auto target_node = dynamic_cast<luci::CircleSplitOut *>(node);
26 if (target_node == nullptr)
27 return false;
28
29 auto split_node = dynamic_cast<luci::CircleSplit *>(target_node->input());
30 if (split_node == nullptr)
31 return false;
32
33 if (loco::succs(split_node).size() != 1)
34 return false;
35
36 if (split_node->num_split() == 1)
37 {
38 auto input_node = loco::must_cast<luci::CircleNode *>(split_node->input());
39 replace(target_node).with(input_node);
40 return true;
41 }
42 return false;
43}

50bool RemoveUnnecessarySplitPass::run(loco::Graph *g)
51{
52 bool changed = false;
53 for (auto node : loco::active_nodes(loco::output_nodes(g)))
54 {
55 auto circle_node = loco::must_cast<luci::CircleNode *>(node);
56 if (remove_unnecessary_split(circle_node))
57 {
58 changed = true;
59 }
60 }
61 return changed;
