50void Serializer::packData(const void *data, size_t size)
51{
52 auto p = static_cast<const char *>(data);
53 size_t old_size = _buffer.size();
54 _buffer.resize(old_size + size);
55 copy(p, p + size, _buffer.data() + old_size);
56}

71void Serializer::serializeShape(const Shape &s)
72{
73 int32_t rank = s.rank();
74 assert(rank <= MAX_DIMS);
75 serializeT<int32_t>(s.rank());
76 for (int32_t i = 0; i < rank; ++i)
77 {
78 int32_t dim = s.dim(i);
79 serializeT<int32_t>(dim);
80 }
81}

83void Serializer::serializeTensor(const TensorVariant &t)
84{
85 // serialize type
86 assert(etoi(t.getDataType()) < MAX_ENUM_VAL);
87 serializeT<int32_t>(etoi(t.getDataType()));
88 // seriazlie data size
89 size_t element_size = t.getElementSize();
90 assert(element_size <= MAX_DIMS);
91 serializeT<int32_t>(element_size);
92 // serialize shape
93 const Shape &shape = t.getShape();
94 serializeShape(shape);
95 // serialize actual data
96 size_t data_size = element_size * shape.numElements();
97
98 size_t old_serialized_data_size = _buffer.size();
99 _buffer.reserve(old_serialized_data_size + data_size);
100 for (const Index &idx : ShapeRange(shape))
101 {
102 packData(t.at(idx), element_size);
103 }
104}

115template <typename Op> void Serializer::serializePads(const Op &op, int32_t number_of_pads)
116{
117 assert(number_of_pads <= MAX_DIMS);
118 serializeT<int32_t>(number_of_pads);
119 for (int i = 0; i < static_cast<int>(number_of_pads); ++i)
120 {
121 auto pad = op.getPaddingBefore().at(i);
122 assert(pad <= MAX_DIM_SIZE);
123 assert(pad >= 0);
124 UNUSED(pad);
125 serializeT<int32_t>(op.getPaddingBefore().at(i));
126 }
127}

278void Serializer::visit(mir::ops::PadOp &op)
279{
280 _curOp->paramStartOffset = _buffer.size();
281
282 // serialize paddings
283 const int num_dims = op.getInputShape(0).rank();
284
285 // serialize output shape
286 serializeShape(op.getOutputShape(0));
287
288 // serialize num dimensions
289 serializeT<int32_t>(num_dims);
290
291 const auto &padding_before = op.getPaddingBefore();
292 const auto &padding_after = op.getPaddingAfter();
293 for (int i = 0; i < num_dims; i++)
294 {
295 serializeT<int32_t>(padding_before[num_dims - 1 - i]);
296 serializeT<int32_t>(padding_after[num_dims - 1 - i]);
297 }
298
299 // FIXME Make use of padding value.
300 assert(op.getPaddingValue() == 0.0f);
301}

356{

358 {
359 if (action->type != sir::Action::Type::callFunction)

362 _curOp->mirOp->accept(this);
