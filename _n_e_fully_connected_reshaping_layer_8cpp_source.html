27void NEFullyConnectedReshapingLayer::configure(const arm_compute::ITensor *input,
28 const arm_compute::ITensor *weights,
29 const arm_compute::ITensor *biases,
30 arm_compute::ITensor *output, bool needs_reshape,
31 const arm_compute::TensorShape &reshape,
32 KernelType kernel_type)
33{
34 _input = input;
35 _weights = weights;
36 _biases = biases;
37 _output = output;
38 _needs_reshape = needs_reshape;
39
40 const ITensor *input_to_use = input;
41 if (_needs_reshape)
42 {
43 // reshape
44 auto_init_if_empty(*_neon_buffer.info(), _input->info()->clone()->set_tensor_shape(reshape));
45 _neon_reshape.configure(_input, &_neon_buffer);
46 input_to_use = &_neon_buffer;
47 }
48
49 _neon_fc = [&]() {
50 if (kernel_type == KernelType::GENERAL)
51 {
52 auto fc = new arm_compute::NEFullyConnectedLayerEx{_memory_manager};
53 fc->configure(input_to_use, _weights, _biases, _output);
54 return std::unique_ptr<arm_compute::IFunction>(fc);
55 }
56 else
57 {
58 assert(kernel_type == KernelType::PREPROCESSED_WEIGHTS);
59
60 bool is_hybrid = input->info()->data_type() == DataType::F32 &&
61 (weights->info()->data_type() == DataType::QSYMM8 ||
62 weights->info()->data_type() == DataType::QASYMM8_SIGNED);
63
64 if (is_hybrid)
65 {
66 auto fc = new arm_compute::NEFullyConnectedHybridLayer{_memory_manager};
67 ITensorInfo *weights_info = const_cast<ITensorInfo *>(_weights->info());
68 const auto orgin_weights_data_type = weights_info->data_type();
69 weights_info->set_data_type(DataType::QASYMM8_SIGNED);
70 fc->configure(input_to_use, _weights, _biases, _output);
71 weights_info->set_data_type(orgin_weights_data_type);
72 return std::unique_ptr<arm_compute::IFunction>(fc);
73 }
74 else
75 {
76 auto fc = new arm_compute::NEFullyConnectedLayer{_memory_manager};
77 fc->configure(input_to_use, _weights, _biases, _output);
78 return std::unique_ptr<arm_compute::IFunction>(fc);
79 }
80 }
81 }();
82
83 // NOTE _neon_buffer is inaccessible from outside, and thus it is safe to invoke allocate here.
84 if (_needs_reshape)
85 {
86 _neon_buffer.allocator()->allocate();
87 }
88}

90void NEFullyConnectedReshapingLayer::run(void)
91{
92 if (_needs_reshape)
93 _neon_reshape.run();
94
95 _neon_fc->run();
96}
