33double Adam::getLearningRate(uint32_t training_step) const
34{
35 auto biasCorrection = [&](double f) { return 1.0f - std::pow(f, training_step + 1); };
36 return _learning_rate * (std::sqrt(biasCorrection(_props.beta2)) / biasCorrection(_props.beta1));
37}

39void Adam::applyGradient(const UpdateFactors &factors) const
40{
41 const auto training_step = std::get<size_t>(factors);
42 const auto &grad_tensor = std::get<const backend::IPortableTensor &>(factors);
43 auto &trainable_tensor = std::get<backend::train::ITrainableTensor &>(factors);
44 assert(trainable_tensor.data_type() == grad_tensor.data_type());
45 const auto opt_vars = trainable_tensor.optVars();
46 assert(opt_vars.size() == 2);
47 // Get the variable for exponential moving average of the gradient
48 auto m_tensor = nnfw::misc::polymorphic_downcast<IPortableTensor *>(opt_vars.at(0));
49 // Get the variable for exponential moving average of the squared_gradient
50 auto v_tensor = nnfw::misc::polymorphic_downcast<IPortableTensor *>(opt_vars.at(1));
51
52 const auto beta1_power = std::pow(_props.beta1, training_step + 1);
53 const auto beta2_power = std::pow(_props.beta2, training_step + 1);
54 // TODO Support nesterov
55 const bool use_nesterov = false;
56
57 if (trainable_tensor.getShape() != grad_tensor.getShape())
58 {
59 throw std::runtime_error("Adam: Invalid gradient tensor");
60 }
61
62 switch (grad_tensor.data_type())
63 {
64 case ir::DataType::FLOAT32:
65 nnfw::cker::train::Adam(
66 ops::getShape(&trainable_tensor), ops::getBuffer<float>(&trainable_tensor),
67 ops::getShape(&grad_tensor), ops::getBuffer<float>(&grad_tensor), ops::getShape(m_tensor),
68 ops::getBuffer<float>(m_tensor), ops::getShape(v_tensor), ops::getBuffer<float>(v_tensor),
69 beta1_power, beta2_power, _learning_rate, _props.beta1, _props.beta2, _props.epsilon,
70 use_nesterov);
71 break;
72 default:
73 throw std::runtime_error("Adam: Not supported data type");
74 }
75}
