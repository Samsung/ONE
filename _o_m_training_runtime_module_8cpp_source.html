31/*
32 * Import train model, validate it, prepare necessary structures and data for future calculations
33 */
34OMStatus OMTrainingRuntimeModule::importTrainModel(char *model_ptr, const OMConfig &config)
35{
36 // 1 - Import main model
37 // 2 - Import and create training entities
38
39 // 1 - Import main model
40 OMStatus status = importModel(model_ptr, config);
41 if (status != Ok)
42 return status;
43
44 // 2 - Import and create training entities
45 _backward_graphs.resize(_graphs.size());
46 for (uint32_t i = 0; i < _backward_graphs.size(); ++i)
47 {
48 // load default backward graph
49 OMRuntimeGraph &graph = _backward_graphs.at(i);
50
51 OMRuntimeContext &runtime_context = graph.getRuntimeContext();
52 OMRuntimeStorage &runtime_storage = graph.getRuntimeStorage();
53 memory::OMRuntimeAllocator &runtime_allocator = graph.getRuntimeAllocator();
54
55 runtime_context.setModel(model_ptr, i);
56
57 // Parse and validate WOF file if it is exist
58 // WARNING: setWofFile method of RuntimeContext should follow after setModel.
59 if (config.wof_ptr != nullptr)
60 runtime_context.setWofFile(config.wof_ptr);
61
62 // Parse and validate Train Config File if it is exists
63 // WARNING: setTrainConfigFile method of RuntimeContext should follow after setModel.
64 if (config.train_mode and config.training_context.training_config_info_data != nullptr)
65 runtime_context.setTrainConfigFile(config.training_context.training_config_info_data);
66
67 // AllocDeallocPlan backward graph creation
68 status = import::OMExecutionPlanCreator::createBackwardExecutionPlan(
69 runtime_storage, runtime_context, runtime_allocator, config);
70 if (status != Ok)
71 return status;
72
73 // Set tensor to train rank type
74 auto &train_storage = _training_handler.getTrainingStorage();
75 import::OMDynamicShapesHandler::importDynamicShapesFromTrainConfig(
76 runtime_storage, runtime_context, train_storage);
77 }
78
79 // Set current optimizer
80 _training_handler.setOptimizer(config);
81
82 return Ok;
83}

85/*
86 * Train single step: run forward graph (with data which was set in SetInput) ->
87 * -> calculate error (with target data which was set in SetTarget) ->
88 * -> run backward graph -> update optimizer state -> after batch_size steps update weights
89 * Warning: before using trainSingleStep call: 1) importTrainModel; 2) setInput; 3) setTarget
90 */
91OMStatus OMTrainingRuntimeModule::trainSingleStep(OMConfig &config)
92{
93 OMStatus status = Ok;
94 uint32_t batch_size = config.training_context.batch_size;
95 config.training_context.num_step += 1;
96
97 // A - Run and update optimizers state batch_size times
98 // a. Allocate forward inputs and memcopy current sample (in current batch) into forward inputs
99 // b. Run forward graph
100 // c. Handle (calculate) current error backpropagation function result
101 // d. Run backward graph (calculate backpropagation values)
102 // e. Update optimization state
103 // f. Reset runtime graphs (to deallocate forward outputs)
104 // B - Update weights according chosen optimization strategy (after batch_size times)
105
106 // A
107 for (uint32_t b = 0; b < batch_size; ++b)
108 {
109 // a. Allocate forward inputs and memcopy current sample (in current batch) into forward inputs
110 {
111 OMRuntimeGraph &forward_main_graph = _graphs.at(0);
112 forward_main_graph.allocateGraphInputs();
113 uint32_t input_nums = forward_main_graph.getNumberOfInputs();
114 for (uint32_t i = 0; i < input_nums; ++i)
115 {
116 uint8_t *allocated_input_data =
117 reinterpret_cast<uint8_t *>(forward_main_graph.getInputDataAt(i));
118 size_t type_size = forward_main_graph.getInputDataTypeSize(i);
119 size_t tensor_size = forward_main_graph.getInputSizeAt(i);
120 size_t offset = b * type_size * tensor_size;
121
122 uint8_t *input_data = _training_handler.getInputData(i);
123 std::memcpy(allocated_input_data, input_data + offset, tensor_size * type_size);
124 }
125 }
126
127 // b. Run forward graph
128 {
129 status = run(config);
130 assert(status == Ok);
131 if (status != Ok)
132 return status;
133 }
134
135 if (_backward_graphs.empty())
136 return ModelNotImport;
137
138 for (uint32_t i = 0; i < _graphs.size(); ++i)
139 {
140 // Get forward entities
141 OMRuntimeGraph &forward_graph = _graphs.at(i);
142 OMRuntimeContext &forward_context = forward_graph.getRuntimeContext();
143 OMRuntimeStorage &forward_storage = forward_graph.getRuntimeStorage();
144
145 // Get backward entities
146 OMRuntimeGraph &backward_graph = _backward_graphs.at(i);
147 OMRuntimeContext &backward_context = backward_graph.getRuntimeContext();
148 OMRuntimeStorage &backward_storage = backward_graph.getRuntimeStorage();
149
150 // c. Handle (calculate) current error backpropagation function result
151 {
152 status = _training_handler.handleError(config, forward_storage, backward_storage,
153 backward_context, b);
154 assert(status == Ok);
155 if (status != Ok)
156 return status;
157 }
158
159 // d. Run backward graph
160 {
161 onert_micro::train::OMBackpropExecuteArgs backprop_execute_args = {
162 forward_storage, backward_storage, backward_context, false, false, 0, ALL};
163
164 status = onert_micro::train::OMBackpropExecute::runBackward(
165 config, backprop_execute_args, backward_graph.getRuntimeAllocator());
166 assert(status == Ok);
167 if (status != Ok)
168 return status;
169 }
170
171 // e. Update optimization state
172 {
173 status = _training_handler.updateOptimizerState(config, backward_storage, backward_context);
174 assert(status == Ok);
175 if (status != Ok)
176 return status;
177 }
178
179 // f. Reset runtime graphs
180 backward_graph.reset();
181 forward_graph.reset();
182 }
183 }
184
185 // B - Update weights according chosen optimization strategy
186 for (uint32_t i = 0; i < _graphs.size(); ++i)
187 {
188 // Get backward context
189 OMRuntimeGraph &backward_graph = _backward_graphs.at(i);
190 OMRuntimeContext &backward_context = backward_graph.getRuntimeContext();
191 OMRuntimeStorage &backward_storage = backward_graph.getRuntimeStorage();
192 status = _training_handler.updateWeights(config, backward_context, backward_storage);
193 }
194
195 return status;
196}

198/*
199 * Evaluate metrics
200 *
201 * Warning: 1) assume that all metric_val for all OMMetrics types actually are float values.
202 * 2) metric_val should be initialized with some value before calling this method due to
203 * after calculation for current batch_num (the sequence number of the current sample)
204 * this value is added to metric_val
205 */
206OMStatus OMTrainingRuntimeModule::evaluateMetric(const OMConfig &config, OMMetrics metric,
207 void *metric_val, uint32_t test_size)
208{
209 OMStatus status = Ok;
210 OMRuntimeGraph &forward_main_graph = _graphs.at(0);
211
212 assert(test_size > 0);
213 if (test_size == 0)
214 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
215
216 uint32_t input_nums = forward_main_graph.getNumberOfInputs();
217 for (uint32_t b = 0; b < test_size; ++b)
218 {
219 // a. Allocate forward inputs and memcopy current sample (in current batch) into forward inputs
220 {
221 forward_main_graph.allocateGraphInputs();
222 for (uint32_t i = 0; i < input_nums; ++i)
223 {
224 uint8_t *allocated_input_data =
225 reinterpret_cast<uint8_t *>(forward_main_graph.getInputDataAt(i));
226 size_t type_size = forward_main_graph.getInputDataTypeSize(i);
227 size_t tensor_size = forward_main_graph.getInputSizeAt(i);
228 size_t offset = b * type_size * tensor_size;
229
230 uint8_t *input_data = _training_handler.getInputData(i);
231 std::memcpy(allocated_input_data, input_data + offset, tensor_size * type_size);
232 }
233 }
234
235 // b. Run forward graph
236 {
237 status = run(config);
238 assert(status == Ok);
239 if (status != Ok)
240 return status;
241 }
242
243 // c. Calculate loss on test_size data
244 _training_handler.evaluateMetric(metric, metric_val, forward_main_graph.getRuntimeStorage(),
245 forward_main_graph.getRuntimeContext(), b);
246
247 // Reset values
248 forward_main_graph.reset();
249 }
250
251 // Averaged result
252 {
253 float *f_metric_val = reinterpret_cast<float *>(metric_val);
254 *f_metric_val /= test_size;
255 }
256
257 return Ok;
258}

260/*
261 * Reset all forward and backward graphs
262 */
263OMStatus OMTrainingRuntimeModule::reset()
264{
265 OMStatus status = Ok;
266
267 if (_graphs.empty())
268 return ModelNotImport;
269
270 for (auto &graph : _graphs)
271 {
272 graph.reset();
273 }
274
275 for (auto &graph : _backward_graphs)
276 {
277 graph.reset();
278 }
279#ifdef OM_MEMORY_ESTIMATE
280 auto &context = _backward_graphs.begin()->getRuntimeContext();
281 auto &storage = _backward_graphs.begin()->getRuntimeStorage();
282 _training_handler.reset(context, storage);
283#else
284 _training_handler.reset();
285#endif // OM_MEMORY_ESTIMATE
286
287 return status;
288}

293OMStatus OMTrainingRuntimeModule::createCheckpointFile(const OMConfig &config,
294 std::vector<char> &data_buffer)
295{
296 // Model wasn't imported
297 if (_backward_graphs.size() == 0 or _graphs.size() == 0)
298 OM_LOG_AND_RETURN(UnknownError, "Unknown error encountered");
299
300 OMRuntimeContext &context = _backward_graphs.at(0).getRuntimeContext();
301 train::OMTrainingStorage &train_storage = _training_handler.getTrainingStorage();
302
303 OMStatus status =
304 train::OMCheckpointSaver::createCheckpointData(context, train_storage, data_buffer, config);
305
306 return status;
307}
