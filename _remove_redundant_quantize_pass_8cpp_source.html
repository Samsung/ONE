50bool remove_redundant_quantize(luci::CircleQuantize *node)
51{
52 auto pred_node = loco::must_cast<luci::CircleNode *>(node->input());
53
54 if (node->quantparam() == nullptr or pred_node->quantparam() == nullptr)
55 return false;
56
57 if (node->quantparam()->scale.size() != 1 or node->quantparam()->zerop.size() != 1 or
58 pred_node->quantparam()->scale.size() != 1 or pred_node->quantparam()->zerop.size() != 1)
59 {
60 return false;
61 }
62
63 if (node->dtype() != pred_node->dtype() or
64 pred_node->quantparam()->scale.at(0) != node->quantparam()->scale.at(0) or
65 pred_node->quantparam()->zerop.at(0) != node->quantparam()->zerop.at(0))
66 {
67 return false;
68 }
69
70 replace(node).with(pred_node);
71
72 return true;
73}

75bool remove_redundant_subsequent_quantize(luci::CircleQuantize *node)
76{
77 auto pred_node = dynamic_cast<luci::CircleQuantize *>(node->input());
78 if (pred_node == nullptr)
79 return remove_redundant_quantize(node);
80
81 node->input(pred_node->input());
82 return true;
83}

101 return changed;
