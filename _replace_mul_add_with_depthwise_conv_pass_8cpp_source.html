27luci::CircleConst *create_weights_from_gamma(luci::CircleConst *gamma)
28{
29 assert(gamma->rank() == 1 or gamma->rank() == 4);
30
31 uint32_t channel_idx = gamma->rank() - 1;
32 uint32_t channel_size = gamma->dim(channel_idx).value();
33
34 // Gamma should be broadcastable in the channel direction
35 for (uint32_t i = 0; i < gamma->rank(); i++)
36 {
37 if (i != channel_idx)
38 assert(gamma->dim(i).value() == 1); // FIX is_batchnorm_mul UNLESS
39 }
40
41 auto name = gamma->name();
42 assert(name.length() > 0);
43
44 // Channel-wise MUL is the same as DEPTHWISE_CONV2D with filter shape (1,1,1,channel_size)
45 auto weights = gamma->graph()->nodes()->create<luci::CircleConst>();
46 weights->dtype(loco::DataType::FLOAT32);
47 weights->rank(4);
48 weights->dim(0).set(1);
49 weights->dim(1).set(1);
50 weights->dim(2).set(1);
51 weights->dim(3).set(channel_size);
52 weights->shape_status(luci::ShapeStatus::VALID);
53 weights->size<loco::DataType::FLOAT32>(channel_size);
54 for (uint32_t i = 0; i < channel_size; i++)
55 {
56 weights->at<loco::DataType::FLOAT32>(i) = gamma->at<loco::DataType::FLOAT32>(i);
57 }
58 weights->name(name + "_weights");
59
60 return weights;
61}

113bool replace_mul_add_with_dwconv(luci::CircleAdd *add)
114{
115 luci::CircleNode *pred_node = nullptr;
116 luci::CircleMul *mul = nullptr;
117 luci::CircleConst *beta = nullptr;
118 luci::CircleConst *gamma = nullptr;
119
120 if (!is_batchnorm_add(add, mul, beta))
121 return false;
122
123 if (loco::succs(mul).size() != 1)
124 return false;
125
126 if (!is_batchnorm_mul(mul, pred_node, gamma))
127 return false;
128
129 if (pred_node->rank() != 4)
130 return false;
131
132 if (pred_node->dtype() != loco::DataType::FLOAT32 || beta->dtype() != loco::DataType::FLOAT32 ||
133 gamma->dtype() != loco::DataType::FLOAT32)
134 return false;
135
136 auto weights = create_weights_from_gamma(gamma);
137 auto bias = create_bias_from_beta(beta);
138
139 auto name = add->name();
140 assert(name.length() > 0);
141
142 auto dwconv = add->graph()->nodes()->create<luci::CircleDepthwiseConv2D>();
143 dwconv->input(pred_node);
144 dwconv->filter(weights);
145 dwconv->bias(bias);
146 dwconv->padding(luci::Padding::SAME);
147 dwconv->stride()->w(1);
148 dwconv->stride()->h(1);
149 dwconv->depthMultiplier(1);
150 dwconv->dilation()->w(1);
151 dwconv->dilation()->h(1);
152 dwconv->fusedActivationFunction(add->fusedActivationFunction());
153 dwconv->name(name + "/DepthwiseConv2D");
154 luci::add_origin(dwconv, luci::composite_origin({luci::get_origin(mul), luci::get_origin(add)}));
155
156 loco::replace(add).with(dwconv);
157 return true;
158}

165bool ReplaceMulAddWithDepthwiseConvPass::run(loco::Graph *g)
166{
167 bool changed = false;
168 for (auto node : loco::active_nodes(loco::output_nodes(g)))
169 {
170 if (auto add = dynamic_cast<luci::CircleAdd *>(node))
171 {
172 if (replace_mul_add_with_dwconv(add))
173 changed = true;
174 }
175 }
176
177 return changed;
178}
