48Shape Shape::from(const std::string &str)
49{
50 Shape shape(0);
51
52 bool pending = false;
53 int value = 0;
54
55 for (const char *cur = str.c_str(); true; ++cur)
56 {
57 if (*cur == ',' || *cur == '\0')
58 {
59 if (pending)
60 {
61 shape.append(value);
62 }
63
64 if (*cur == '\0')
65 {
66 break;
67 }
68
69 pending = false;
70 value = 0;
71 continue;
72 }
73
74 assert(*cur >= '0' && *cur <= '9');
75
76 pending = true;
77 value *= 10;
78 value += *cur - '0';
79 }
80
81 return shape;
82}
