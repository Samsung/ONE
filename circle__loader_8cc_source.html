88 bool allowOptionalInputTensor(BuiltinOperator op) override
89 {
90 switch (op)
91 {
92 case BuiltinOperator::BuiltinOperator_FULLY_CONNECTED:
93 case BuiltinOperator::BuiltinOperator_BCQ_FULLY_CONNECTED:
94 case BuiltinOperator::BuiltinOperator_UNIDIRECTIONAL_SEQUENCE_LSTM:
95 return true;

102 std::unique_ptr<ir::Graph> loadSubgraph(const circle::SubGraph *circle_subg) override
103 {
104 auto subg = std::make_unique<ir::Graph>();
105 // Load tensors
106 _tensor_to_operand.resize(circle_subg->tensors()->size());
107 for (flatbuffers::uoffset_t i = 0; i < circle_subg->tensors()->size(); ++i)
108 {
109 _tensor_to_operand[i] = loadOperand(circle_subg->tensors()->Get(i), *subg);
110 }
111 // Set inputs
112 for (const std::int32_t input_ind : *circle_subg->inputs())
113 {
114 subg->addInput(tensorIdxToOperandIdx(input_ind),
115 _tensor_names.at(_tensor_to_operand[input_ind]));
116 }
117 // Set outputs
118 for (const std::int32_t output_ind : *circle_subg->outputs())
119 {
120 subg->addOutput(tensorIdxToOperandIdx(output_ind),
121 _tensor_names.at(_tensor_to_operand[output_ind]));
122 }
123 // Create operations
124 for (const auto *op : *circle_subg->operators())
125 {
126 CircleLoader::loadOperation(op, *subg);
127 }
128
129 subg->setLayout(convertDataFormat(circle_subg->data_format()));
130
131 subg->verify();
132
133 return subg;
134 }

136 void loadOperation(const circle::Operator *op, ir::Graph &subg)
137 {
138 auto const builtin_op = getBuiltinOperator(op);
139
140 switch (builtin_op)
141 {
142 case circle::BuiltinOperator::BuiltinOperator_BATCH_MATMUL:
143 loadBatchMatMul(op, subg);
144 return;
145 case circle::BuiltinOperator::BuiltinOperator_INSTANCE_NORM:
146 loadInstanceNorm(op, subg);
147 return;
148 case circle::BuiltinOperator::BuiltinOperator_BCQ_FULLY_CONNECTED:
149 loadBCQFullyConnected(op, subg);
150 return;
151 case circle::BuiltinOperator::BuiltinOperator_BCQ_GATHER:
152 loadBCQGather(op, subg);
153 return;
154 default:
155 BaseLoader::loadOperation(op, subg);
156 return;
157 }
158 }
159};

161void CircleLoader::loadBatchMatMul(const Operator *op, ir::Graph &subg)
162{
163 ir::OperandIndexSequence inputs;
164 ir::OperandIndexSequence outputs;
165
166 loadOperationIO(op, inputs, outputs);
167
168 ir::operation::BatchMatMul::Param param;
169 const auto *options = op->builtin_options_as_BatchMatMulOptions();
170
171 param.adj_x = options->adjoint_lhs();
172 param.adj_y = options->adjoint_rhs();
173
174 std::unique_ptr<ir::Operation> new_op(new ir::operation::BatchMatMul(inputs, outputs, param));
175 subg.addOperation(std::move(new_op));
176}

178void CircleLoader::loadInstanceNorm(const Operator *op, ir::Graph &subg)
179{
180 ir::OperandIndexSequence inputs;
181 ir::OperandIndexSequence outputs;
182
183 loadOperationIO(op, inputs, outputs);
184
185 ir::operation::InstanceNorm::Param param;
186 const auto *options = op->builtin_options_as_InstanceNormOptions();
187
188 param.activation = convertActivation(options->fused_activation_function());
189 // Use default value 1e-5 if value of epsilon is zero
190 param.epsilon = options->epsilon() == 0.f ? 1e-5 : options->epsilon();
191
192 std::unique_ptr<ir::Operation> new_op(new ir::operation::InstanceNorm(inputs, outputs, param));
193 subg.addOperation(std::move(new_op));
194}
