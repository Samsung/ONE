103{
104 std::unique_ptr<nnkit::BackendPlugin> backend_plugin;
105 nnkit::VectorArguments backend_arguments;
106
107 uint32_t warmup_count = 3;
108 uint32_t benchmark_count = 1;
109
110 // Simple argument parser (based on map)
111 std::map<std::string, OptionHook> argparse;
112
113 argparse["--backend"] = [&backend_plugin](const std::string &tag) {
114 backend_plugin = std::move(nnkit::make_backend_plugin(tag));
115 };
116
117 argparse["--backend-arg"] = [&backend_arguments](const std::string &arg) {
118 backend_arguments.append(arg);
119 };
120
121 argparse["--benchmark-count"] = [&benchmark_count](const std::string &arg) {
122 benchmark_count = std::stoi(arg);
123 };
124
125 for (int n = 1; n < argc; n += 2)
126 {
127 const std::string tag{argv[n]};
128 const std::string arg{argv[n + 1]};
129
130 auto it = argparse.find(tag);
131
132 if (it == argparse.end())
133 {
134 std::cerr << "Option '" << tag << "' is not supported" << std::endl;
135 return 255;
136 }
137
138 it->second(arg);
139 }
140
141 // CHECK: Command-line arguments
142 if (backend_plugin == nullptr)
143 {
144 std::cerr << "ERROR: --backend is mssing" << std::endl;
145 return 255;
146 }
147
148 // Initialize
149 auto backend = backend_plugin->create(backend_arguments);
150
151 // Run warm-up iterations
152 info() << "Start warming-up iterations(" << warmup_count << ")";
153
154 for (uint32_t n = 0; n < warmup_count; ++n)
155 {
156 backend->prepare([](nnkit::TensorContext &ctx) {
157 // DO NOTHING
158 });
159
160 backend->run();
161
162 backend->teardown([](nnkit::TensorContext &ctx) {
163 // DO NOTHING
164 });
165 }
166
167 // Run benchmark iterations
168 info() << "Start benchmark iterations(" << benchmark_count << ")";
169
170 microseconds elapsed_min{0};
171 microseconds elapsed_max{0};
172 microseconds elapsed_total{0};
173
174 for (uint32_t n = 0; n < benchmark_count; ++n)
175 {
176 backend->prepare([](nnkit::TensorContext &ctx) {
177 // DO NOTHING
178 });
179
180 microseconds elapsed{0};
181
182 ::measure(elapsed) << [&](void) { backend->run(); };
183
184 info() << "#" << n + 1 << " takes " << duration_cast<milliseconds_f>(elapsed).count() << "ms";
185
186 elapsed_min = (n == 0) ? elapsed : std::min(elapsed_min, elapsed);
187 elapsed_max = (n == 0) ? elapsed : std::max(elapsed_max, elapsed);
188 elapsed_total += elapsed;
189
190 backend->teardown([](nnkit::TensorContext &ctx) {
191 // DO NOTHING
192 });
193 }
194
195 // Show summary
196 info() << "Show statistics";
197
198 auto min_ms = duration_cast<milliseconds_f>(elapsed_min).count();
199 auto max_ms = duration_cast<milliseconds_f>(elapsed_max).count();
200 auto avg_ms = duration_cast<milliseconds_f>(elapsed_total).count() / benchmark_count;
201
202 info() << "MIN: " << min_ms << "ms"
203 << ", MAX: " << max_ms << "ms, AVG: " << avg_ms << "ms";
204
205 return 0;
