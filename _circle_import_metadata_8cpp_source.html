24template <typename VECTORTYPE> uint32_t read_u32(const VECTORTYPE &buffer, uint32_t idx)
25{
26 static_assert(std::is_same<typename VECTORTYPE::value_type, uint8_t>::value, "Types mismatch!");
27
28 uint32_t val = 0;
29 val += (buffer.at(idx + 0) << 0 * 8);
30 val += (buffer.at(idx + 1) << 1 * 8);
31 val += (buffer.at(idx + 2) << 2 * 8);
32 val += (buffer.at(idx + 3) << 3 * 8);
33 return val;
34}

41// 'source_table' is decoded to std::map<uint32_t, std::string> format.
42template <typename VECTORTYPE>
43const std::map<uint32_t, std::string> decoded_source_table(const VECTORTYPE &source_table_data)
44{
45 static_assert(std::is_same<typename VECTORTYPE::value_type, uint8_t>::value, "Types mismatch!");
46
47 std::map<uint32_t, std::string> source_id_name_map;
48 uint32_t idx = 0;
49
50 if (source_table_data.size() < 4)
51 throw std::runtime_error("Source table decode error : invalid entry number");
52
53 uint32_t entry_number = read_u32(source_table_data, idx);
54 idx += sizeof(uint32_t);
55
56 while (idx < source_table_data.size())
57 {
58 if (idx + 2 * sizeof(uint32_t) > source_table_data.size())
59 throw std::runtime_error("Source table decode error : invalid entry item");
60
61 uint32_t id = read_u32(source_table_data, idx);
62 idx += sizeof(uint32_t);
63
64 uint32_t length = read_u32(source_table_data, idx);
65 idx += sizeof(uint32_t);
66
67 if (idx + sizeof(char) * length > source_table_data.size())
68 throw std::runtime_error("Source table decode error : invalid entry data");
69
70 // The last character of name is '\0'.
71 // However, as std::string do not use '\0' for finding the end of string,
72 // we ignore the character and do not include it in the string.
73 std::string origin_name;
74 for (uint32_t j = 0; j < length - 1; ++j)
75 origin_name += source_table_data.at(idx + j);
76 assert(source_table_data.at(idx + length - 1) == '\0');
77 idx += sizeof(char) * length;
78
79 if (source_id_name_map.insert({id, origin_name}).second == false)
80 throw std::runtime_error("Source table decode error : duplicated origin ID");
81 }
82
83 if (idx != source_table_data.size())
84 throw std::runtime_error("Source table decode error : data size invalid");
85
86 if (source_id_name_map.size() != entry_number)
87 throw std::runtime_error("Source table decode error : result size mismatch");
88
89 return source_id_name_map;
90}

92// 'op_table' is decoded to std::map<uint32_t, std::set<uint32_t>> format.
93template <typename VECTORTYPE>
94const std::map<uint32_t, std::set<uint32_t>> decoded_op_table(const VECTORTYPE &op_table_data)
95{
96 static_assert(std::is_same<typename VECTORTYPE::value_type, uint8_t>::value, "Types mismatch!");
97
98 std::map<uint32_t, std::set<uint32_t>> node_source_ids_map;
99 uint32_t idx = 0;
100
101 if (op_table_data.size() < 4)
102 throw std::runtime_error("Op table decode error : invalid entry number");
103
104 uint32_t entry_number = read_u32(op_table_data, idx);
105 idx += sizeof(uint32_t);
106
107 while (idx < op_table_data.size())
108 {
109 if (idx + 2 * sizeof(uint32_t) > op_table_data.size())
110 throw std::runtime_error("Op table decode error : invalid entry item");
111
112 uint32_t id = read_u32(op_table_data, idx);
113 idx += sizeof(uint32_t);
114
115 uint32_t node_num = read_u32(op_table_data, idx);
116 idx += sizeof(uint32_t);
117
118 if (idx + sizeof(uint32_t) * node_num > op_table_data.size())
119 throw std::runtime_error("Source table decode error : invalid entry data");
120
121 std::set<uint32_t> source_ids;
122 for (uint32_t j = 0; j < node_num; ++j)
123 {
124 uint32_t origin = read_u32(op_table_data, idx);
125 idx += sizeof(uint32_t);
126
127 source_ids.insert(origin);
128 }
129
130 if (node_source_ids_map.insert({id, source_ids}).second == false)
131 throw std::runtime_error("Op table decode error : duplicated origin ID");
132 }
133
134 if (idx != op_table_data.size())
135 throw std::runtime_error("Op table decode error : data size invalid");
136
137 if (node_source_ids_map.size() != entry_number)
138 throw std::runtime_error("Op table decode error : entry number invalid");
139
140 return node_source_ids_map;
141}

143// 'execution_plan_table' is decoded to std::map<uint32_t, std::vector<uint32_t>> format.
144template <typename VECTORTYPE>
145const luci::ExecutionPlanTable decoded_execution_plan(const VECTORTYPE &execution_plan_data)
146{
147 static_assert(std::is_same<typename VECTORTYPE::value_type, uint8_t>::value, "Types mismatch!");
148
149 luci::ExecutionPlanTable execution_plan_table;
150 uint32_t idx = 0;
151
152 if (execution_plan_data.size() < 4)
153 throw std::runtime_error("Op table decode error : invalid entry number");
154
155 uint32_t entry_number = read_u32(execution_plan_data, idx);
156 idx += sizeof(uint32_t);
157
158 while (idx < execution_plan_data.size())
159 {
160 if (idx + 2 * sizeof(uint32_t) > execution_plan_data.size())
161 throw std::runtime_error("Op table decode error : invalid entry item");
162
163 uint32_t id = read_u32(execution_plan_data, idx);
164 idx += sizeof(uint32_t);
165
166 uint32_t size = read_u32(execution_plan_data, idx);
167
168 if (size == 0)
169 throw std::runtime_error("Op table decode error : empty execution plan entry");
170
171 idx += sizeof(uint32_t);
172
173 if (idx + sizeof(uint32_t) * size > execution_plan_data.size())
174 throw std::runtime_error("Source table decode error : invalid entry data");
175
176 std::vector<uint32_t> execution_plan_vector;
177 for (uint32_t j = 0; j < size; ++j)
178 {
179 uint32_t execution_plan_inform = read_u32(execution_plan_data, idx);
180 idx += sizeof(uint32_t);
181
182 execution_plan_vector.push_back(execution_plan_inform);
183 }
184
185 if (execution_plan_table.insert({id, execution_plan_vector}).second == false)
186 throw std::runtime_error("Op table decode error : duplicated origin ID");
187 }
188
189 if (idx != execution_plan_data.size())
190 throw std::runtime_error("Op table decode error : data size invalid");
191
192 if (execution_plan_table.size() != entry_number)
193 throw std::runtime_error("Op table decode error : entry number invalid");
194
195 return execution_plan_table;
196}

203CircleImportMetadata::CircleImportMetadata(const luci::CircleReader &reader)
204{
205 const auto metadata = reader.metadata();
206 for (uint32_t i = 0; i < metadata.size(); ++i)
207 {
208 const auto *meta = metadata[i];
209 assert(meta != nullptr);
210
211 assert(meta->buffer() < reader.buffers().size());
212 assert(reader.buffers()[meta->buffer()] != nullptr);
213 const auto buffer = luci::wrap(reader.buffers()[meta->buffer()]->data());
214
215 assert(meta->name() != nullptr);
216 if (meta->name()->str().compare("ONE_op_table") == 0)
217 _op_table = decoded_op_table(buffer);
218 else if (meta->name()->str().compare("ONE_source_table") == 0)
219 _source_table = decoded_source_table(buffer);
220 else if (meta->name()->str().compare("ONE_execution_plan_table") == 0)
221 _execution_plan_table = decoded_execution_plan(buffer);
222 }
223}

225const OriginTable CircleImportMetadata::origin_table(void)
226{
227 OriginTable origin_table;
228
229 if (_op_table.size() > 0 && _source_table.size() > 0)
230 {
231 for (auto &kv : _op_table)
232 {
233 const auto node_id = kv.first;
234 const auto &source_ids = kv.second;
235
236 std::vector<std::shared_ptr<CircleNodeOrigin>> origins;
237 for (auto source_id : source_ids)
238 {
239 const auto source_name = _source_table.at(source_id);
240 origins.push_back(single_origin(source_id, source_name));
241 }
242
243 auto origin = composite_origin(origins);
244 origin_table.emplace(node_id, origin);
245 }
246 }
247
248 return origin_table;
249}
