72inline nnfw::cker::Shape getExtendedTensorShape(const IPortableTensor *tensor)
73{
74 assert(tensor);
75 const int32_t extended_rank = 4;
76 int32_t raw_shape[extended_rank];
77 auto shape = tensor->getShape();
78 uint32_t src = extended_rank - shape.rank();
79 for (uint32_t i = 0; i < extended_rank; ++i)
80 {
81 if (i < src)
82 {
83 raw_shape[i] = 1;
84 }
85 else
86 {
87 raw_shape[i] = shape.dim(i - src);
88 }
89 }
90
91 return nnfw::cker::Shape(extended_rank, raw_shape);
92}

94inline nnfw::cker::Shape getShape(const IPortableTensor *tensor)
95{
96 if (tensor == nullptr)
97 return nnfw::cker::Shape();
98
99 const ir::Shape &shape = tensor->get_info().shape();
100
101 assert(tensor->layout() == ir::Layout::NHWC);
102
103 auto rank = shape.rank();
104 nnfw::cker::Shape ret(rank);
105 auto data = ret.DimsData();
106 for (int i = 0; i < rank; ++i)
107 {
108 data[i] = shape.dim(i);
109 }
110 return ret;
111}

113inline nnfw::cker::FusedActivationFunctionType
114convertActivationType(const ir::Activation activation)
115{
116 switch (activation)
117 {
118 case ir::Activation::NONE:
119 return nnfw::cker::FusedActivationFunctionType::kNone;
120 case ir::Activation::RELU:
121 return nnfw::cker::FusedActivationFunctionType::kRelu;
122 case ir::Activation::RELU1:
123 return nnfw::cker::FusedActivationFunctionType::kRelu1;
124 case ir::Activation::RELU6:
125 return nnfw::cker::FusedActivationFunctionType::kRelu6;
126 case ir::Activation::TANH:
127 return nnfw::cker::FusedActivationFunctionType::kTanh;
128 case ir::Activation::SIGMOID:
129 return nnfw::cker::FusedActivationFunctionType::kSigmoid;
130 default:
131 throw std::runtime_error{"CPU backend: Cannot convert activation type"};
132 }
133}

135inline int32_t getAxis(uint32_t rank, int32_t axis, ir::Layout frontend_layout)
136{
137 auto ret = axis;
138
139 if (axis < 0)
140 {
141 ret += rank;
142 }
143
144 // NCHW -> NHWC
145 if (frontend_layout == ir::Layout::NCHW)
146 {
147 int32_t permutation[4] = {0, 3, 1, 2};
148 ret = permutation[ret];
149 }
150
151 return ret;
152}
