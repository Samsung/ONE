91bool TFTypeInferenceRule::infer(const loco::Node *node, loco::DataType &dtype) const
92{
93 assert(node->dialect() == TFDialect::get());
94
95 TypeForwardAlgorithm alg;
96
97// clang-format off
98#define TENSORFLOW_NODE(OPCODE,CLASS) \
99 if (dynamic_cast<const moco::CLASS *>(node)) \
100 { \
101 auto tfnode = loco::must_cast<const moco::CLASS *>(node); \
102 dtype = tfnode->accept(&alg); \
103 assert(dtype != loco::DataType::Unknown); \
104 return true; \
105 }
106#include "moco/IR/TFNodes.lst"
107#undef TENSORFLOW_NODE
108 // clang-format on
109
110 return false;
111}
