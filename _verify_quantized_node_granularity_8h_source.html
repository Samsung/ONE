53 {
54 auto circle_node = loco::must_cast<const luci::CircleNode *>(node);
55

433class VerifyQuantizedNodeChannelWiseGranularity final : public VerifyQuantizedNodeGranularity
434{
435private:
436 uint32_t rank(const loco::Node *node)
437 {
438 auto circle_node = loco::must_cast<const luci::CircleNode *>(node);
439 return circle_node->rank();
440 }
441
442 bool is_cwq_const(const loco::Node *node, uint32_t channel_dim)
443 {
444 auto circle_node = loco::must_cast<const luci::CircleConst *>(node);
445
446 assert(channel_dim < circle_node->rank()); // FIX_CALLER_UNLESS
447 auto channel_size = circle_node->dim(channel_dim).value();
448
449 if (circle_node->quantparam() == nullptr)
450 return false;
451
452 if (circle_node->quantparam()->quantized_dimension != static_cast<int32_t>(channel_dim))
453 return false;
454
455 if (circle_node->quantparam()->scale.size() != channel_size)
456 return false;
457
458 if (circle_node->quantparam()->zerop.size() != channel_size)
459 return false;
460
461 return true;
462 }
463
464private:
465 bool visit(const luci::CircleConv2D *node)
466 {
467 RETURN_FALSE_UNLESS(is_lwq(node))
468 RETURN_FALSE_UNLESS(is_lwq(node->input()))
469 RETURN_FALSE_UNLESS(is_cwq_const(node->filter(), 0))
470 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(node->bias());
471 if (bias != nullptr)
472 RETURN_FALSE_UNLESS(is_cwq_const(node->bias(), rank(node->bias()) - 1))
473 return true;
474 }
475
476 bool visit(const luci::CircleDepthwiseConv2D *node)
477 {
478 RETURN_FALSE_UNLESS(is_lwq(node))
479 RETURN_FALSE_UNLESS(is_lwq(node->input()))
480 RETURN_FALSE_UNLESS(is_cwq_const(node->filter(), 3))
481 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(node->bias());
482 if (bias != nullptr)
483 RETURN_FALSE_UNLESS(is_cwq_const(node->bias(), rank(node->bias()) - 1))
484 return true;
485 }
486
487 bool visit(const luci::CircleInstanceNorm *node)
488 {
489 RETURN_FALSE_UNLESS(is_lwq(node))
490 RETURN_FALSE_UNLESS(is_lwq(node->input()))
491 RETURN_FALSE_UNLESS(is_cwq_const(node->gamma(), rank(node->gamma()) - 1))
492 RETURN_FALSE_UNLESS(is_cwq_const(node->beta(), rank(node->beta()) - 1))
493 return true;
494 }
495
496 bool visit(const luci::CirclePRelu *node)
497 {
498 RETURN_FALSE_UNLESS(is_lwq(node))
499 RETURN_FALSE_UNLESS(is_lwq(node->input()))
500 RETURN_FALSE_UNLESS(is_cwq_const(node->alpha(), rank(node->alpha()) - 1))
501 return true;
502 }
503
504 bool visit(const luci::CircleTransposeConv *node)
505 {
506 RETURN_FALSE_UNLESS(is_lwq(node))
507 RETURN_FALSE_UNLESS(is_lwq(node->outBackprop()))
508 RETURN_FALSE_UNLESS(is_cwq_const(node->filter(), 0))
509 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(node->bias());
510 if (bias != nullptr)
511 RETURN_FALSE_UNLESS(is_cwq_const(node->bias(), rank(node->bias()) - 1))
512
513 return true;
514 }
515
516 bool visit(const luci::CircleFullyConnected *node)
517 {
518 RETURN_FALSE_UNLESS(is_lwq(node))
519 RETURN_FALSE_UNLESS(is_lwq(node->input()))
520 RETURN_FALSE_UNLESS(is_cwq_const(node->weights(), 0))
521 luci::CircleConst *bias = dynamic_cast<luci::CircleConst *>(node->bias());
522 // Bias is optional (it can be CircleOutputExclude)
523 if (bias != nullptr)
524 RETURN_FALSE_UNLESS(is_cwq_const(node->bias(), rank(node->bias()) - 1))
525 return true;
526 }
