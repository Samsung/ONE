40ggma::Tokenizer *SentencePieceTokenizer::create(const std::string &tokenizer_dir)
41{
42 auto tokenizer = std::make_unique<SentencePieceTokenizer>();
43 tokenizer->_processor = std::make_unique<::sentencepiece::SentencePieceProcessor>();
44
45 std::filesystem::path model_path = std::filesystem::path(tokenizer_dir) / "tokenizer.model";
46 auto status = tokenizer->_processor->Load(model_path.string());
47
48 return status.ok() ? tokenizer.release() : nullptr;
49}

51size_t SentencePieceTokenizer::tokenize(const char *text, size_t text_len, int32_t *tokens,
52 size_t max_tokens, size_t *n_tokens) const
53{
54 if (!text || !tokens || !n_tokens || max_tokens == 0 || !_processor)
55 {
56 if (n_tokens)
57 *n_tokens = 0;
58 return 0;
59 }
60
61 std::string input_text(text, text_len);
62 std::vector<int> piece_ids;
63
64 auto status = _processor->Encode(input_text, &piece_ids);
65 if (!status.ok())
66 {
67 if (n_tokens)
68 *n_tokens = 0;
69 return 0;
70 }
71
72 // Initialize tokens array to 0
73 std::fill(tokens, tokens + max_tokens, 0);
74
75 // Check BOS token ID and add it if it's 1
76 int bos_id = _processor->bos_id();
77 size_t bos_offset = 0;
78
79 if (bos_id >= 0 && max_tokens > 0)
80 {
81 tokens[0] = 1; // Add BOS token
82 bos_offset = 1; // Start actual tokens from index 1
83 }
84
85 size_t available_space = max_tokens - bos_offset;
86 size_t token_count = std::min(piece_ids.size(), available_space);
87
88 for (size_t i = 0; i < token_count; ++i)
89 tokens[bos_offset + i] = static_cast<int32_t>(piece_ids[i]);
90
91 *n_tokens = bos_offset + token_count;
92 return bos_offset + token_count;
93}

95size_t SentencePieceTokenizer::detokenize(const int32_t *tokens, size_t n_tokens, char *text,
96 size_t text_len) const
97{
98 if (!tokens || !text || n_tokens == 0 || text_len == 0 || !_processor)
99 return 0;
100
101 std::vector<int> piece_ids;
102 for (size_t i = 0; i < n_tokens; ++i)
103 piece_ids.push_back(tokens[i]);
104
105 std::string decoded_text;
106 auto status = _processor->Decode(piece_ids, &decoded_text);
107 if (!status.ok())
108 return 0;
109
110 size_t copy_len = std::min(decoded_text.length(), text_len - 1);
111 memcpy(text, decoded_text.c_str(), copy_len);
112 text[copy_len] = '\0';
113
114 return copy_len;
115}
