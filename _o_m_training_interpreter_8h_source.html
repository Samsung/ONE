41 // Import train model with current config settings
42 // Note: model ptr should be non-const to save result
43 OMStatus importTrainModel(char *model_ptr, const OMConfig &config);
44
45 // Set input data for input with input_index
46 // Note: number of the samples in data should be equal to the batch_size in config structure
47 void setInput(uint8_t *data, uint32_t input_index)
48 {
49 _training_runtime_module.setInputData(data, input_index);
50 }
51 // Set target data for output with target_index
52 // Note: number of the samples in data should be equal to the batch_size in config structure
53 void setTarget(uint8_t *data, uint32_t target_index)
54 {
55 _training_runtime_module.setTargetData(data, target_index);
56 }
57
58 // Train single step: run forward graph (with data which was set in SetInput) ->
59 // -> calculate error (with target data which was set in SetTarget) ->
60 // -> run backward graph -> update optimizer state -> after batch_size steps update weights
61 // Warning: before using trainSingleStep call: 1) importTrainModel; 2) setInput; 3) setTarget
62 OMStatus trainSingleStep(OMConfig &config);
63
64 // Reset all states and data saved into OMTrainingInterpreter (trained weights will not be reset)
65 OMStatus reset();
66
67 // Calculate and save metric into metric_val: run forward graph -> calculate metric
68 // Note: calculation will be done on test_size number of test samples
69 // Warning: before using evaluateMetric call: 1) importTrainModel; 2) setInput; 3) setTarget
70 // Note: number of the samples in data should be equal to the test_size
71 OMStatus evaluateMetric(OMMetrics metric, void *metric_val, uint32_t test_size)
72 {
73 return _training_runtime_module.evaluateMetric(metric, metric_val, test_size);
74 }
75
76 // To get input and output flat size
77 uint32_t getInputSizeAt(uint32_t position);
78 uint32_t getOutputSizeAt(uint32_t position);
79
80 // Save current model
81 OMStatus saveModel(const OMConfig &config, const char *save_path);
82
83 // Save current status as checkpoint
84 OMStatus saveCheckpoint(const OMConfig &config, const char *save_path);
85
86 // Load current status from checkpoint and save it in current model and in current config
87 OMStatus loadCheckpoint(OMConfig &config, const char *load_path);
88
89 OMStatus run() { return _training_runtime_module.run(); }
90 OMStatus allocateInputs() { return _training_runtime_module.allocateInputs(); }
91
92 void *getInputData(uint32_t position);
93 void *getInputDataAt(uint32_t position);
94 void *getOutputDataAt(uint32_t position);
95
96#ifdef OM_MEMORY_ESTIMATE

98 size_t getCurrentFootprintMemory() { return _training_runtime_module.getPeakFootprintMemory(); }
99#endif // OM_MEMORY_ESTIMATE
100};
