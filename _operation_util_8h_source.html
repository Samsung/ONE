32inline nnfw::cker::Shape convertShape(const ir::Shape &shape)
33{
34 auto dimensions = std::vector<uint32_t>(shape.dims().begin(), shape.dims().end());
35
36 std::vector<int32_t> raw_shape;
37 raw_shape.resize(dimensions.size());
38
39 for (uint32_t i = 0; i < dimensions.size(); ++i)
40 {
41 raw_shape[i] = dimensions[i];
42 }
43
44 return nnfw::cker::GetShape(raw_shape);
45}

47inline nnfw::cker::Shape convertExtendShape(const ir::Shape &shape)
48{
49 auto dimensions = std::vector<uint32_t>(shape.dims().begin(), shape.dims().end());
50
51 const int32_t extended_rank = 4;
52 int32_t raw_shape[extended_rank];
53 uint32_t start = extended_rank - dimensions.size();
54
55 for (uint32_t i = 0; i < extended_rank; ++i)
56 {
57 if (i < start)
58 {
59 raw_shape[i] = 1;
60 }
61 else
62 {
63 raw_shape[i] = dimensions[i - start];
64 }
65 }
66
67 return nnfw::cker::Shape(extended_rank, raw_shape);
68}

88template <typename T>
89void calculateActivationRange(ir::Activation activation, T *activation_min, T *activation_max)
90{
91 if (activation == ir::Activation::RELU)
92 {
93 *activation_min = 0;
94 *activation_max = std::numeric_limits<T>::max();
95 }
96 else if (activation == ir::Activation::RELU6)
97 {
98 *activation_min = 0;
99 *activation_max = 6;
100 }
101 else if (activation == ir::Activation::RELU1)
102 {
103 *activation_min = -1;
104 *activation_max = 1;
105 }
106 else if (activation == ir::Activation::NONE)
107 {
108 *activation_min = std::numeric_limits<T>::lowest();
109 *activation_max = std::numeric_limits<T>::max();
110 }
111 else
112 {
113 throw std::runtime_error{"Unsupported activation type"};
114 }
115}

117inline ir::Shape calcBroadcastShape(const ir::Shape &lhs, const ir::Shape &rhs, bool &success)
118{
119 int lhs_rank = lhs.rank();
120 int rhs_rank = rhs.rank();
121
122 int out_rank = (lhs_rank > rhs_rank ? lhs_rank : rhs_rank);
123 ir::Shape out_shape(out_rank);
124
125 int lhs_idim = lhs_rank - 1;
126 int rhs_idim = rhs_rank - 1;
127 success = true;
128 for (int out_idim = out_rank - 1; out_idim >= 0; out_idim--)
129 {
130 if (lhs_idim == -1 && rhs_idim == -1)
131 {
132 // invalid result
133 success = false;
134 break;
135 }
136
137 if (lhs_idim == -1)
138 {
139 out_shape.dim(out_idim) = rhs.dim(rhs_idim);
140 rhs_idim--;
141 }
142 else if (rhs_idim == -1)
143 {
144 out_shape.dim(out_idim) = lhs.dim(lhs_idim);
145 lhs_idim--;
146 }
147 else
148 {
149 if (lhs.dim(lhs_idim) == rhs.dim(rhs_idim))
150 {
151 out_shape.dim(out_idim) = lhs.dim(lhs_idim);
152 lhs_idim--;
153 rhs_idim--;
154 }
155 else if (lhs.dim(lhs_idim) == 1)
156 {
157 out_shape.dim(out_idim) = rhs.dim(rhs_idim);
158 lhs_idim--;
159 rhs_idim--;
160 }
161 else if (rhs.dim(rhs_idim) == 1)
162 {
163 out_shape.dim(out_idim) = lhs.dim(lhs_idim);
164 lhs_idim--;
165 rhs_idim--;
166 }
167 else
168 {
169 // invalid result
170 success = false;
171 break;
172 }
173 }
174 }
175
176 if (lhs_idim != -1 || rhs_idim != -1)
177 {
178 // invalid result
179 success = false;
180 }
181 return out_shape;
182}
