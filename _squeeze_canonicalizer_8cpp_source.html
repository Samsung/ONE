27bool canonicalize_squeeze_to_reshape(loco::Graph *graph, moco::TFSqueeze *node)
28{
29 LOGGER(l);
30
31 INFO(l) << "TFNodeCanonicalize TFSqueeze begin";
32
45 auto nodeshape = moco::node_shape(node);
46 // canonicalize into FixedReshape is valid when squeeze has shape info
47 // TODO Support general Squeeze case
48 assert(nodeshape.domain() != loco::Domain::Unknown);
49
50 auto squeeze_tensor_shape = nodeshape.as<loco::TensorShape>();
51
52 // Create loco node to replace
53 auto reshape = graph->nodes()->create<loco::FixedReshape>();
54
55 // Copy shape
56 reshape->rank(squeeze_tensor_shape.rank());
57 for (uint32_t axis = 0; axis < squeeze_tensor_shape.rank(); ++axis)
58 {
59 assert(squeeze_tensor_shape.dim(axis).known());
60 reshape->dim(axis) = squeeze_tensor_shape.dim(axis);
61 }
62
63 // replace
64 auto in = node->input();
65 reshape->input(in);
66 replace(node).with(reshape);
67
68 INFO(l) << "TFNodeCanonicalize TFSqueeze done";
69
70 return true;
71}
