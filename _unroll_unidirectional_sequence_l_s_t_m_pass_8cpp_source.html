180std::vector<luci::CircleUnpackOut *> UnrollLSTM::input_unpacks(luci::CircleNode *input)
181{
182 assert(input != nullptr);
183
184 // NOTE unpack input can be LSTM or Transpose
185 auto unpack = _nctx->create<luci::CircleUnpack>();
186 unpack->num(_timesteps);
187 unpack->axis(0);
188 unpack->value(input);
189 unpack->name(_name + "_unpack");
190 luci::add_origin(unpack, luci::get_origin(_lstm));
191
192 std::vector<luci::CircleUnpackOut *> outs;
193 for (uint32_t idx = 0; idx < _timesteps; ++idx)
194 {
195 auto unpackout = _nctx->create<luci::CircleUnpackOut>();
196 unpackout->input(unpack);
197 unpackout->index(idx);
198 unpackout->name(_name + "_unpackout_" + std::to_string(idx));
199 luci::add_origin(unpackout, luci::get_origin(_lstm));
200 outs.push_back(unpackout);
201 }
202
203 return outs;
204}

206luci::CircleConst *UnrollLSTM::merged_weights(luci::CircleConst *iw, luci::CircleConst *fw,
207 luci::CircleConst *cw, luci::CircleConst *ow)
208{
209 assert(iw != nullptr);
210 assert(fw != nullptr);
211 assert(cw != nullptr);
212 assert(ow != nullptr);
213
214 auto iw_rank = iw->rank();
215 assert(iw_rank == fw->rank());
216 assert(iw_rank == cw->rank());
217 assert(iw_rank == ow->rank());
218
219 uint32_t ne_w = 1;
220 for (uint32_t i = 0; i < iw_rank; i++)
221 ne_w *= iw->dim(i).value();
222
223 assert(iw->dtype() == loco::DataType::FLOAT32);
224 assert(fw->dtype() == loco::DataType::FLOAT32);
225 assert(cw->dtype() == loco::DataType::FLOAT32);
226 assert(ow->dtype() == loco::DataType::FLOAT32);
227
228 // merged weights
229 auto mw = _nctx->create<luci::CircleConst>();
230 mw->dtype(iw->dtype());
231 mw->rank(iw_rank);
232 mw->dim(0) = 4u * iw->dim(0).value();
233 for (uint32_t i = 1; i < iw_rank; i++)
234 mw->dim(i) = iw->dim(i);
235 mw->size<loco::DataType::FLOAT32>(4 * ne_w);
236 mw->shape_status(luci::ShapeStatus::VALID);
237 for (uint32_t i = 0; i < ne_w; ++i)
238 {
239 mw->at<loco::DataType::FLOAT32>(i + ne_w * 0) = iw->at<loco::DataType::FLOAT32>(i);
240 mw->at<loco::DataType::FLOAT32>(i + ne_w * 1) = fw->at<loco::DataType::FLOAT32>(i);
241 mw->at<loco::DataType::FLOAT32>(i + ne_w * 2) = cw->at<loco::DataType::FLOAT32>(i);
242 mw->at<loco::DataType::FLOAT32>(i + ne_w * 3) = ow->at<loco::DataType::FLOAT32>(i);
243 }
244 return mw;
245}

247luci::CircleFullyConnected *UnrollLSTM::create_input_matmul(luci::CircleNode *input)
248{
249 assert(input != nullptr);
250
251 // weights
252 auto iw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_input_weights());
253 auto fw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_forget_weights());
254 auto cw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_cell_weights());
255 auto ow = loco::must_cast<luci::CircleConst *>(_lstm->input_to_output_weights());
256
257 auto fcw = merged_weights(iw, fw, cw, ow);
258 fcw->name(_name + "_fc_w");
259 luci::add_origin(fcw, luci::get_origin(_lstm));
260
261 // bias
262 auto ib = loco::must_cast<luci::CircleConst *>(_lstm->input_gate_bias());
263 auto fb = loco::must_cast<luci::CircleConst *>(_lstm->forget_gate_bias());
264 auto cb = loco::must_cast<luci::CircleConst *>(_lstm->cell_gate_bias());
265 auto ob = loco::must_cast<luci::CircleConst *>(_lstm->output_gate_bias());
266
267 auto fcb = merged_weights(ib, fb, cb, ob);
268 fcb->name(_name + "_fc_b");
269 luci::add_origin(fcb, luci::get_origin(_lstm));
270
271 auto fc = _nctx->create<luci::CircleFullyConnected>();
272 fc->input(input);
273 fc->weights(fcw);
274 fc->bias(fcb);
275 fc->fusedActivationFunction(luci::FusedActFunc::NONE);
276 fc->name(_name + "_fc");
277 luci::add_origin(fc, luci::get_origin(_lstm));
278
279 return fc;
280}

282luci::CircleAdd *UnrollLSTM::create_input_matmul(luci::CircleNode *input, luci::CircleMul *mul,
283 uint32_t step)
284{
285 assert(input != nullptr);
286 assert(mul != nullptr);
287 assert(step < _timesteps);
288
289 auto base_name = _name + "_matmul" + std::to_string(step);
290
291 // input weights
292 auto iw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_input_weights());
293 auto fw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_forget_weights());
294 auto cw = loco::must_cast<luci::CircleConst *>(_lstm->input_to_cell_weights());
295 auto ow = loco::must_cast<luci::CircleConst *>(_lstm->input_to_output_weights());
296
297 auto fcw = merged_weights(iw, fw, cw, ow);
298 fcw->name(base_name + "_fc_w");
299 luci::add_origin(fcw, luci::get_origin(_lstm));
300
301 auto fcb = _nctx->create<luci::CircleOutputExclude>();
302
303 auto fc = _nctx->create<luci::CircleFullyConnected>();
304 fc->input(input);
305 fc->weights(fcw);
306 fc->bias(fcb);
307 fc->fusedActivationFunction(luci::FusedActFunc::NONE);
308 fc->name(base_name + "_fc");
309 luci::add_origin(fc, luci::get_origin(_lstm));
310
311 // recurrent weights
312 auto ri = loco::must_cast<luci::CircleConst *>(_lstm->recurrent_to_input_weights());
313 auto rf = loco::must_cast<luci::CircleConst *>(_lstm->recurrent_to_forget_weights());
314 auto rc = loco::must_cast<luci::CircleConst *>(_lstm->recurrent_to_cell_weights());
315 auto ro = loco::must_cast<luci::CircleConst *>(_lstm->recurrent_to_output_weights());
316
317 auto fcrw = merged_weights(ri, rf, rc, ro);
318 fcrw->name(base_name + "_fcr_w");
319 luci::add_origin(fcrw, luci::get_origin(_lstm));
320
321 auto fcrb = _nctx->create<luci::CircleOutputExclude>();
322
323 auto fcr = _nctx->create<luci::CircleFullyConnected>();
324 fcr->input(mul);
325 fcr->weights(fcrw);
326 fcr->bias(fcrb);
327 fcr->fusedActivationFunction(luci::FusedActFunc::NONE);
328 fcr->name(base_name + "_fcr");
329 luci::add_origin(fcr, luci::get_origin(_lstm));
330
331 auto add_fc = _nctx->create<luci::CircleAdd>();
332 add_fc->x(fcr);
333 add_fc->y(fc);
334 add_fc->fusedActivationFunction(luci::FusedActFunc::NONE);
335 add_fc->name(base_name + "_addfc");
336 luci::add_origin(add_fc, luci::get_origin(_lstm));
337
338 // bias
339 auto ib = loco::must_cast<luci::CircleConst *>(_lstm->input_gate_bias());
340 auto fb = loco::must_cast<luci::CircleConst *>(_lstm->forget_gate_bias());
341 auto cb = loco::must_cast<luci::CircleConst *>(_lstm->cell_gate_bias());
342 auto ob = loco::must_cast<luci::CircleConst *>(_lstm->output_gate_bias());
343
344 auto bias = merged_weights(ib, fb, cb, ob);
345 bias->name(base_name + "_bias");
346
347 auto add_bias = _nctx->create<luci::CircleAdd>();
348 add_bias->x(add_fc);
349 add_bias->y(bias);
350 add_bias->fusedActivationFunction(luci::FusedActFunc::NONE);
351 add_bias->name(base_name + "_addbias");
352 luci::add_origin(add_bias, luci::get_origin(_lstm));
353
354 return add_bias;
355}

357std::vector<luci::CircleSplitOut *> UnrollLSTM::matmul_splits(luci::CircleNode *input,
358 uint32_t step)
359{
360 assert(input != nullptr);
361 assert(step < _timesteps);
362
363 std::string split_name = _name + "_sp" + std::to_string(step);
364
365 auto split_dim = _nctx->create<luci::CircleConst>();
366 split_dim->dtype(loco::DataType::S32);
367 split_dim->rank(1);
368 split_dim->dim(0) = 1;
369 split_dim->size<loco::DataType::S32>(1);
370 split_dim->at<loco::DataType::S32>(0) = 1;
371 split_dim->shape_status(luci::ShapeStatus::VALID);
372 split_dim->name(split_name + "_dim");
373 luci::add_origin(split_dim, luci::get_origin(_lstm));
374
375 auto split = _nctx->create<luci::CircleSplit>();
376 split->num_split(4);
377 split->split_dim(split_dim);
378 split->input(input);
379 split->name(split_name);
380 luci::add_origin(split, luci::get_origin(_lstm));
381
382 auto split_o0 = _nctx->create<luci::CircleSplitOut>();
383 split_o0->input(split);
384 split_o0->index(0);
385 split_o0->name(split_name + "_spo0");
386 luci::add_origin(split_o0, luci::get_origin(_lstm));
387
388 auto split_o1 = _nctx->create<luci::CircleSplitOut>();
389 split_o1->input(split);
390 split_o1->index(1);
391 split_o1->name(split_name + "_spo1");
392 luci::add_origin(split_o1, luci::get_origin(_lstm));
393
394 auto split_o2 = _nctx->create<luci::CircleSplitOut>();
395 split_o2->input(split);
396 split_o2->index(2);
397 split_o2->name(split_name + "_spo2");
398 luci::add_origin(split_o2, luci::get_origin(_lstm));
399
400 auto split_o3 = _nctx->create<luci::CircleSplitOut>();
401 split_o3->input(split);
402 split_o3->index(3);
403 split_o3->name(split_name + "_spo3");
404 luci::add_origin(split_o3, luci::get_origin(_lstm));
405
406 std::vector<luci::CircleSplitOut *> outs;
407 outs.push_back(split_o0);
408 outs.push_back(split_o1);
409 outs.push_back(split_o2);
410 outs.push_back(split_o3);
411 return outs;
412}

414luci::CircleConst *UnrollLSTM::forget_zero(void)
415{
416 uint32_t amount = _batch * _units;
417
418 auto zero = _nctx->create<luci::CircleConst>();
419 zero->dtype(loco::DataType::FLOAT32);
420 zero->rank(2);
421 zero->dim(0) = _batch;
422 zero->dim(1) = _units;
423 zero->size<loco::DataType::FLOAT32>(amount);

426 zero->shape_status(luci::ShapeStatus::VALID);
427 zero->name(_name + "_zero");
428 luci::add_origin(zero, luci::get_origin(_lstm));
429 return zero;
430}

432luci::CircleMul *UnrollLSTM::forget_gate_cell(std::vector<luci::CircleSplitOut *> &splits,
433 luci::CircleNode *prev, uint32_t step,
434 luci::CircleNode **retadd)
435{
436 assert(splits.size() > 0);
437 assert(prev != nullptr);
438 assert(step < _timesteps);
439
440 std::string net_name = _name + "_net" + std::to_string(step);
441
442 auto split_0 = splits[0]; // input-input : Logistic - Mul(c) - Add - Tanh - Mul
443 auto split_1 = splits[1]; // input-forget : Logistic - Mul(p) - Add - Tanh - Mul
444 auto split_2 = splits[2]; // input-cell : Tanh - Mul(c) - Add - Tanh - Mul
445 auto split_3 = splits[3]; // input-output : Logistic - Mul
446
447 auto logis_0 = _nctx->create<luci::CircleLogistic>();
448 logis_0->x(split_0);
449 logis_0->name(net_name + "_log0");
450 luci::add_origin(logis_0, luci::get_origin(_lstm));
451
452 auto logis_1 = _nctx->create<luci::CircleLogistic>();
453 logis_1->x(split_1);
454 logis_1->name(net_name + "_log1");
455 luci::add_origin(logis_1, luci::get_origin(_lstm));
456
457 auto tanh_2 = _nctx->create<luci::CircleTanh>();
458 tanh_2->x(split_2);
459 tanh_2->name(net_name + "_tanh2");
460 luci::add_origin(tanh_2, luci::get_origin(_lstm));
461
462 auto logis_3 = _nctx->create<luci::CircleLogistic>();
463 logis_3->x(split_3);
464 logis_3->name(net_name + "_log3");
465 luci::add_origin(logis_3, luci::get_origin(_lstm));
466
467 auto mul_c = _nctx->create<luci::CircleMul>();
468 mul_c->x(logis_0);
469 mul_c->y(tanh_2);
470 mul_c->fusedActivationFunction(luci::FusedActFunc::NONE);
471 mul_c->name(net_name + "_mul1");
472 luci::add_origin(mul_c, luci::get_origin(_lstm));
473
474 auto mul_p = _nctx->create<luci::CircleMul>();
475 mul_p->x(logis_1);
476 mul_p->y(prev);
477 mul_p->fusedActivationFunction(luci::FusedActFunc::NONE);
478 mul_p->name(net_name + "_mul2");
479 luci::add_origin(mul_p, luci::get_origin(_lstm));
480
481 auto add_cp = _nctx->create<luci::CircleAdd>();
482 add_cp->x(mul_c);
483 add_cp->y(mul_p);
484 add_cp->fusedActivationFunction(luci::FusedActFunc::NONE);
485 add_cp->name(net_name + "_add1");
486 luci::add_origin(add_cp, luci::get_origin(_lstm));
487
488 if (retadd != nullptr)
489 *retadd = add_cp;
490
491 auto tanh_cp = _nctx->create<luci::CircleTanh>();
492 tanh_cp->x(add_cp);
493 tanh_cp->name(net_name + "_tanh3");
494 luci::add_origin(tanh_cp, luci::get_origin(_lstm));
495
496 auto mul_out = _nctx->create<luci::CircleMul>();
497 mul_out->x(logis_3);
498 mul_out->y(tanh_cp);
499 mul_out->fusedActivationFunction(luci::FusedActFunc::NONE);
500 mul_out->name(net_name + "_mul3");
501 luci::add_origin(mul_out, luci::get_origin(_lstm));
502
503 return mul_out;
504}

535luci::CircleTranspose *UnrollLSTM::last_transpose(std::vector<luci::CircleMul *> &output_muls)
536{
537 assert(output_muls.size() == _timesteps);
538
539 auto pack = _nctx->create<luci::CirclePack>(_timesteps);
540 pack->axis(0);
541 for (uint32_t idx = 0; idx < _timesteps; ++idx)
542 pack->values(idx, output_muls[idx]);
543 pack->name(_name + "_pack");
544 luci::add_origin(pack, luci::get_origin(_lstm));
545
546 auto perm = transpose_perm();
547 perm->name(_name + "_perm2");
548 luci::add_origin(perm, luci::get_origin(_lstm));
549
550 auto transpose = _nctx->create<luci::CircleTranspose>();
551 transpose->a(pack);
552 transpose->perm(perm);
553 transpose->name(_name + "_trans2");
554 luci::add_origin(transpose, luci::get_origin(_lstm));
555
556 return transpose;
557}

559bool unroll_lstm(luci::CircleUnidirectionalSequenceLSTM *lstm)
560{
561 // NOTE shape of input of lstm is interpreted as [batch, timesteps, feature]
562 // shape of output of lstm is interpreted as [batch, timesteps, units]
563 // TODO add more conditions to check LSTM
564 assert(lstm != nullptr);
565 assert(lstm->rank() == 3); // use assert to findout when this happens
566 if (lstm->rank() != 3)
567 return false;
568 if (!(lstm->dim(0).known() and lstm->dim(1).known() and lstm->dim(2).known()))
569 return false;
570
571 UnrollLSTM ulstm;
572 ulstm._lstm = lstm;
573 ulstm._nctx = lstm->graph()->nodes();
574 ulstm._name = lstm->name();
575 ulstm._batch = lstm->dim(0).value();
576 ulstm._timesteps = lstm->dim(1).value();
577 ulstm._units = lstm->dim(2).value(); // output space dim
578
579 luci::CircleNode *input = loco::must_cast<luci::CircleNode *>(lstm->input());
580 assert(input->rank() == 3); // use assert to findout when this happens
581 if (input->rank() != 3)
582 return false;
583 assert(input->dim(0).value() == ulstm._batch);
584 assert(input->dim(1).value() == ulstm._timesteps);
585
586 if (ulstm._timesteps > 1)
587 {
588 // Transpose to switch batch <-> timesteps
589 // NOTE TF uses Reshape when batch is 1 but as there is Transpose->Reshape
590 // Pass, we can just use Transpose for both cases
591 auto transpose = ulstm.first_transpose(input);
592 input = transpose;
593 }
594
595 auto unpacks = ulstm.input_unpacks(input);
596 assert(unpacks.size() == ulstm._timesteps);
597 uint32_t step = 0;
598 auto unpackout = unpacks[step];
599
600 // First FC
601 auto fc_1 = ulstm.create_input_matmul(unpackout);
602 assert(fc_1 != nullptr);
603 auto splits = ulstm.matmul_splits(fc_1, step);
604 assert(splits.size() == 4);
605
606 luci::CircleNode *prev = nullptr; // prev step CircleAdd
607 luci::CircleNode *this_add = nullptr;
608
609 prev = ulstm.forget_zero(); // provide all zero constant for first step
610
611 std::vector<luci::CircleMul *> output_muls;
612 auto mul_gc = ulstm.forget_gate_cell(splits, prev, step, &this_add);
613 assert(mul_gc != nullptr);
614 assert(this_add != nullptr);
615 // gather all Muls for last Pack
616 output_muls.push_back(mul_gc);
617
618 for (step = 1; step < ulstm._timesteps; ++step)
619 {
620 auto unpackout = unpacks[step];
621 auto add_n = ulstm.create_input_matmul(unpackout, mul_gc, step);
622
623 auto splits = ulstm.matmul_splits(add_n, step);
624 assert(splits.size() == 4);
625
626 prev = this_add;
627 mul_gc = ulstm.forget_gate_cell(splits, prev, step, &this_add);
628 assert(mul_gc != nullptr);
629 assert(this_add != nullptr);
630
631 output_muls.push_back(mul_gc);
632 }
633 assert(output_muls.size() == ulstm._timesteps);
634
635 if (ulstm._timesteps == 1)
636 {
637 // Reshape for single step
638 auto reshape = ulstm.last_reshape(mul_gc);
639 loco::replace(lstm).with(reshape);
640 }
641 else
