29void MaxPool(const float *input_data, const Dims<4> &input_dims, int stride_width,
30 int stride_height, int pad_width, int pad_height, int kwidth, int kheight,
31 float *output_data, const Dims<4> &output_dims)
32{
33 const int batches = MatchingArraySize(input_dims, 3, output_dims, 3);
34 const int input_height = ArraySize(input_dims, 2);
35 const int input_width = ArraySize(input_dims, 1);
36 const int output_height = ArraySize(output_dims, 2);
37 const int output_width = ArraySize(output_dims, 1);
38 const int depth = MatchingArraySize(input_dims, 0, output_dims, 0);
39
40 const auto in_mat = MapAsMatrixWithFirstDimAsRows(input_data, input_dims);
41 auto out_mat = MapAsMatrixWithFirstDimAsRows(output_data, output_dims);
42 // Prefill the output to minimum representable float value
43 out_mat.setConstant(std::numeric_limits<float>::lowest());
44 for (int b = 0; b < batches; ++b)
45 {
46 for (int h = 0; h < input_height; ++h)
47 {
48 for (int w = 0; w < input_width; ++w)
49 {
50 // (h_start, h_end) * (w_start, w_end) is the range that the input
51 // vector projects to.
52 int hpad = h + pad_height;
53 int wpad = w + pad_width;
54 int h_start = (hpad < kheight) ? 0 : (hpad - kheight) / stride_height + 1;
55 int h_end = std::min(hpad / stride_height + 1, output_height);
56 int w_start = (wpad < kwidth) ? 0 : (wpad - kwidth) / stride_width + 1;
57 int w_end = std::min(wpad / stride_width + 1, output_width);
58 // compute elementwise sum
59 for (int ph = h_start; ph < h_end; ++ph)
60 {
61 for (int pw = w_start; pw < w_end; ++pw)
62 {
63 int out_offset = NodeOffset(b, ph, pw, output_height, output_width);
64 out_mat.col(out_offset) =
65 out_mat.col(out_offset)
66 .cwiseMax(in_mat.col(NodeOffset(b, h, w, input_height, input_width)));
67 }
68 }
69 }
70 }
71 }
72
73 for (int b = 0; b < batches; ++b)
74 {
75 for (int y = 0; y < output_height; ++y)
76 {
77 for (int x = 0; x < output_width; ++x)
78 {
79 for (int c = 0; c < depth; ++c)
80 {
81 output_data[Offset(output_dims, c, x, y, b)] =
82 ActivationFunction<Ac>(output_data[Offset(output_dims, c, x, y, b)]);
83 }
84 }
85 }
86 }
87}
