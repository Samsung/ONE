86DataChefRegistry &data_chef_registry(const circlechef::TensorType &type)
87{
88 static DataChefRegistry s32;
89 static DataChefRegistry s64;
90 static DataChefRegistry fp32;
91 static DataChefRegistry u8;
92 static DataChefRegistry string;
93 static DataChefRegistry boolean;
94 static DataChefRegistry s16;
95
96 switch (type)
97 {
98 case circlechef::INT32:
99 return s32;
100 case circlechef::INT64:
101 return s64;
102 case circlechef::FLOAT32:
103 return fp32;
104 case circlechef::UINT8:
105 return u8;
106 case circlechef::STRING:
107 return string;
108 case circlechef::BOOL:
109 return boolean;
110 case circlechef::INT16:
111 return s16;
112 default:
113 break;
114 }
115
116 throw std::runtime_error{"Unknown tensor type"};

130std::map<circle::BuiltinOperator, int32_t>
131gather_builtincode_map(const ::circlechef::ModelRecipe &model_recipe)
132{
133 // Key and value of the map are BuiltinOperator and operator version
134 std::map<circle::BuiltinOperator, int32_t> builtin_map;
135
136 for (const auto &operation : model_recipe.operation())
137 {
138 auto op_chef = op_chef_registry().lookup(operation.type()).create(&operation);
139 if (op_chef->code() == circle::BuiltinOperator_CUSTOM)
140 continue;
141
142 // Various operation version is unified as the highest version among them
143 if (builtin_map.find(op_chef->code()) == builtin_map.end() ||
144 builtin_map[op_chef->code()] < operation.version())
145 builtin_map[op_chef->code()] = operation.version();
146 }
147
148 // Add ops used in Graphs(subgraphs)
149 for (int g = 0; g < model_recipe.graph_size(); ++g)
150 {
151 const auto &graph = model_recipe.graph(g);
152 for (const auto &operation : graph.operation())
153 {
154 auto op_chef = op_chef_registry().lookup(operation.type()).create(&operation);
155 if (op_chef->code() == circle::BuiltinOperator_CUSTOM)
156 continue;
157
158 // Various operation version is unified as the highest version among them
159 if (builtin_map.find(op_chef->code()) == builtin_map.end() ||
160 builtin_map[op_chef->code()] < operation.version())
161 builtin_map[op_chef->code()] = operation.version();
162 }
163 }
164
165 return builtin_map;
166}

169std::set<std::string> gather_customcode_set(const ::circlechef::ModelRecipe &model_recipe)
170{
171 std::set<std::string> customcode_set;
172 for (const auto &operation : model_recipe.operation())
173 {
174 auto op_chef = op_chef_registry().lookup(operation.type()).create(&operation);
175 if (op_chef->code() == circle::BuiltinOperator_CUSTOM)
176 customcode_set.insert(operation.type());
177 }
178
179 // Add ops used in Graphs(subgraphs)
180 for (int g = 0; g < model_recipe.graph_size(); ++g)
181 {
182 const auto &graph = model_recipe.graph(g);
183 for (const auto &operation : graph.operation())
184 {
185 auto op_chef = op_chef_registry().lookup(operation.type()).create(&operation);
186 if (op_chef->code() == circle::BuiltinOperator_CUSTOM)
187 customcode_set.insert(operation.type());
188 }
189 }
190
191 return customcode_set;
192}

199struct CookParams
200{
201 std::vector<flatbuffers::Offset<::circle::Buffer>> &buffer_vec;
202 std::vector<flatbuffers::Offset<::circle::OperatorCode>> &code_vec;
203 std::vector<flatbuffers::Offset<::circle::SubGraph>> &subgraph_vec;
204 std::unique_ptr<flatbuffers::FlatBufferBuilder> &flatbuffer_builder;
205 std::map<circle::BuiltinOperator, int32_t> &builtin_code_map;
206 std::string noname;
207};
208
209template <typename T> void cook_graph(const T &graph, CookParams &cp)
210{
211 LOGGER(l);
212
213 std::vector<flatbuffers::Offset<::circle::Buffer>> &buffer_vec = cp.buffer_vec;
214 std::vector<flatbuffers::Offset<::circle::OperatorCode>> &code_vec = cp.code_vec;
215 std::vector<flatbuffers::Offset<::circle::SubGraph>> &subgraph_vec = cp.subgraph_vec;
216 std::unique_ptr<flatbuffers::FlatBufferBuilder> &flatbuffer_builder = cp.flatbuffer_builder;
217 std::map<circle::BuiltinOperator, int32_t> &builtin_code_map = cp.builtin_code_map;
218
219 // Operand-related
220 std::vector<flatbuffers::Offset<::circle::Tensor>> tensor_vec;
221
222 // Operation-related
223 std::vector<flatbuffers::Offset<::circle::Operator>> operator_vec;
224
225 // default name for graph
226 std::string graph_name = cp.noname;
227 if (graph.has_name())
228 graph_name = graph.name();
229
230 // Tensor Name -> Tensor ID mapping (per Graph)
231 std::map<std::string, int32_t> symbol_table;
232
233 auto lookup = [&symbol_table, &graph_name](const std::string &name) {
234 if (symbol_table.find(name) != symbol_table.end())
235 return symbol_table.at(name);
236 else if (name == "")
237 return -1; // -1 in circle means that optional input tensor is empty.
238 else
239 {
240 std::string msg = "circlechef : input not found in " + graph_name + " graph";
241 throw std::runtime_error(msg.c_str());
242 }
243 };
244
245 int32_t buffer_start = buffer_vec.size();
246 int32_t buffer_index = 0;
247
248 // Create buffer(s) 1~n(I) for input(s)
249 const auto size_input = graph.input_size();
250 for (int ci = 0; ci < size_input; ++ci)
251 {
252 circle::BufferBuilder buffer_builder{*flatbuffer_builder};
253 buffer_vec.emplace_back(buffer_builder.Finish());
254 }
255 // Create buffer(s) n(I)+1~n(I)+n(O) for output(s)
256 const auto size_output = graph.output_size();
257 for (int co = 0; co < size_output; ++co)
258 {
259 circle::BufferBuilder buffer_builder{*flatbuffer_builder};
260 buffer_vec.emplace_back(buffer_builder.Finish());
261 }
262
263 auto input_names = as_dataset(graph.input()).vectorize();
264 auto output_names = as_dataset(graph.output()).vectorize();
265
266 for (const auto &operand : graph.operand())
267 {
268 assert(operand.has_name());
269
270 assert(operand.has_type());
271
272 flatbuffers::Offset<flatbuffers::Vector<int32_t>> shape;
273 std::vector<int32_t> dims;
274 if (operand.has_shape())
275 {
276 dims = as_dims(operand.shape());
277 shape = flatbuffer_builder->CreateVector(dims);
278 }
279
280 auto name = flatbuffer_builder->CreateString(operand.name());
281
282 buffer_index = 0;
283
284 // Create Buffer if filler is specified
285 if (operand.has_filler())
286 {
287 const auto &filler = operand.filler();
288
289 assert(filler.has_tag());
290
291 auto args = ranged_arguments(filler.arg().begin(), filler.arg().end());
292 auto chef = data_chef_registry(operand.type()).lookup(filler.tag()).create(args);
293
294 assert(chef != nullptr);
295
296 // Create Data
297 int32_t count = (element_count(dims) > 0) ? element_count(dims) : filler.arg_size();
298 auto data_vec = chef->generate(count);
299 auto data = flatbuffer_builder->CreateVector(data_vec);
300
301 // Create Buffer
302 circle::BufferBuilder buffer_builder{*flatbuffer_builder};
303 buffer_builder.add_data(data);
304 auto buffer = buffer_builder.Finish();
305
306 // Update Buffer Index & Vector
307 buffer_index = buffer_vec.size();
308 buffer_vec.emplace_back(buffer);
309 }
310 else
311 {
312 // if this is input or output, assign to that buffer_index
313 int idx = 0;
314 for (auto it = input_names.begin(); it != input_names.end(); ++it, ++idx)
315 {
316 if (*it == operand.name())
317 {
318 buffer_index = buffer_start + idx;
319 break;
320 }
321 }
322 if (buffer_index == 0)
323 {
324 idx = 0;
325 for (auto it = output_names.begin(); it != output_names.end(); ++it, ++idx)
326 {
327 if (*it == operand.name())
328 {
329 buffer_index = buffer_start + size_input + idx;
330 break;
331 }
332 }
333 }
334 if (buffer_index == 0)
335 {
336 // we couldn't find the buffer; create an empty buffer for this tensor
337 buffer_index = buffer_vec.size();
338
339 circle::BufferBuilder buffer_builder{*flatbuffer_builder};
340 buffer_vec.emplace_back(buffer_builder.Finish());
341 }
342 }
343 assert(buffer_index != 0);
344
345 flatbuffers::Offset<circle::QuantizationParameters> quant_index;
346
347 // Create QuantizationParameters if quant is specified
348 if (operand.has_quant())
349 {
350 const auto &quant = operand.quant();
351
352 // Create each parameters
353 // NOTE if some parameters are not given, those will be set to default value
354 std::vector<float> quant_max_vec(quant.max_size());
355 std::vector<float> quant_min_vec(quant.min_size());
356 std::vector<float> quant_scale_vec(quant.scale_size());
357 std::vector<int64_t> quant_zero_point_vec(quant.zero_point_size());
358
359 for (uint32_t i = 0; i < quant.max_size(); ++i)
360 quant_max_vec.at(i) = quant.max(i);
361 for (uint32_t i = 0; i < quant.min_size(); ++i)
362 quant_min_vec.at(i) = quant.min(i);
363 for (uint32_t i = 0; i < quant.scale_size(); ++i)
364 quant_scale_vec.at(i) = quant.scale(i);
365 for (uint32_t i = 0; i < quant.zero_point_size(); ++i)
366 quant_zero_point_vec.at(i) = quant.zero_point(i);
367
368 auto quant_max = flatbuffer_builder->CreateVector(quant_max_vec);
369 auto quant_min = flatbuffer_builder->CreateVector(quant_min_vec);
370 auto quant_scale = flatbuffer_builder->CreateVector(quant_scale_vec);
371 auto quant_zero_point = flatbuffer_builder->CreateVector(quant_zero_point_vec);
372
373 // Create QuantizationParameters
374 circle::QuantizationParametersBuilder quant_builder{*flatbuffer_builder};
375 quant_builder.add_max(quant_max);
376 quant_builder.add_min(quant_min);
377 quant_builder.add_scale(quant_scale);
378 quant_builder.add_zero_point(quant_zero_point);
379 quant_builder.add_quantized_dimension(quant.quantized_dimension());
380
381 // Update QuantizationParameters Index
382 quant_index = quant_builder.Finish();
383 }
384
385 flatbuffers::Offset<flatbuffers::Vector<int32_t>> shape_signature;
386 if (operand.has_shape_signature())
387 {
388 auto signature = as_dims(operand.shape_signature());
389 shape_signature = flatbuffer_builder->CreateVector(signature);
390 }
391
392 // Create Tensor
393 circle::TensorBuilder tensor_builder{*flatbuffer_builder};
394
395 tensor_builder.add_shape(shape);
396 tensor_builder.add_type(as_circle_tensortype(operand.type()));
397 tensor_builder.add_buffer(buffer_index);
398 tensor_builder.add_name(name);
399 if (operand.has_quant())
400 tensor_builder.add_quantization(quant_index);
401 if (operand.has_shape_signature())
402 tensor_builder.add_shape_signature(shape_signature);
403
404 // Append!
405 tensor_vec.emplace_back(tensor_builder.Finish());
406
407 // Update Tensor Name -> Tensor Index Map
408 int32_t tensor_index = symbol_table.size();
409 const auto &tensor_name = operand.name();
410
411 INFO(l) << "Symbol [" << tensor_name << "] = Tensor " << tensor_index << std::endl;
412
413 symbol_table[tensor_name] = tensor_index;
414 }
415
416 // Create Operator
417 for (const auto &operation : graph.operation())
418 {
419 assert(operation.has_type());
420
421 auto op_chef = op_chef_registry().lookup(operation.type()).create(&operation);
422
423 // Create 'inputs'
424 std::vector<int32_t> input_vec = as_dataset(operation.input()).map(lookup).vectorize();
425 auto inputs = flatbuffer_builder->CreateVector(input_vec);
426
427 // Create 'outputs'
428 std::vector<int32_t> output_vec = as_dataset(operation.output()).map(lookup).vectorize();
429 auto outputs = flatbuffer_builder->CreateVector(output_vec);
430
431 // Create Option
432 auto options = op_chef->value(*flatbuffer_builder);
433
434 // Create Custom option
435 auto circle_custom_options = op_chef->custom_value(*flatbuffer_builder);
436
437 // Create Operator
438 circle::OperatorBuilder op_builder{*flatbuffer_builder};
439
440 // Get operator code index from builtin_code_map with assumption, order of
441 // builtin_code_map is same as that of code_vec
442 auto op_it = builtin_code_map.find(op_chef->code());
443 assert(op_it != builtin_code_map.end());
444 uint32_t opcode_index = std::distance(builtin_code_map.begin(), op_it);
445
446 op_builder.add_opcode_index(opcode_index);
447 op_builder.add_inputs(inputs);
448 op_builder.add_outputs(outputs);
449 op_builder.add_builtin_options_type(op_chef->type());
450 op_builder.add_builtin_options(options);
451 op_builder.add_custom_options(circle_custom_options);
452 op_builder.add_custom_options_format(circle::CustomOptionsFormat_FLEXBUFFERS);
453 // Append Operator
454 operator_vec.emplace_back(op_builder.Finish());
455 }
456
457 // Create network input/output vector
458 std::vector<int32_t> input_vec = as_dataset(graph.input()).map(lookup).vectorize();
459 std::vector<int32_t> output_vec = as_dataset(graph.output()).map(lookup).vectorize();
460
461 // Create "SubGraph" arguments
462 auto tensors = flatbuffer_builder->CreateVector(tensor_vec);
463 auto inputs = flatbuffer_builder->CreateVector(input_vec);
464 auto outputs = flatbuffer_builder->CreateVector(output_vec);
465 auto operators = flatbuffer_builder->CreateVector(operator_vec);
466 auto name = flatbuffer_builder->CreateString(graph_name);
467
468 circle::SubGraphBuilder subgraph_builder{*flatbuffer_builder};
469
470 subgraph_builder.add_tensors(tensors);
471 subgraph_builder.add_inputs(inputs);
472 subgraph_builder.add_outputs(outputs);
473 subgraph_builder.add_operators(operators);
474 subgraph_builder.add_name(name);
475
476 subgraph_vec.emplace_back(subgraph_builder.Finish());
477}

487GeneratedModel cook(const ::circlechef::ModelRecipe &model_recipe)
488{
489// Initialize Op Chef Registry
490#define OP_CHEF(NAME, FACTORY_CLASS) \
491 op_chef_registry().add(#NAME, std::unique_ptr<FACTORY_CLASS>(new FACTORY_CLASS()));
492#include "OpChef.def"
493#undef OP_CHEF
494
495// Initialize Data Chef Registry
496#define DATA_CHEF(TYPE, NAME, FACTORY_CLASS) \
497 data_chef_registry(::circlechef::TYPE) \
498 .add(#NAME, std::unique_ptr<FACTORY_CLASS>(new FACTORY_CLASS()));
499#include "DataChef.def"
500#undef DATA_CHEF
501
502 //
503 // Create FlatBufferBuilder
504 //
505 auto flatbuffer_builder =
506 std::unique_ptr<flatbuffers::FlatBufferBuilder>(new flatbuffers::FlatBufferBuilder(1024));
507
508 // Operand-related
509 std::vector<flatbuffers::Offset<::circle::Buffer>> buffer_vec;
510
511 // Operation-related
512 std::vector<flatbuffers::Offset<::circle::OperatorCode>> code_vec;
513
514 // Graphs-related
515 std::vector<flatbuffers::Offset<::circle::SubGraph>> subgraph_vec;
516
517 // Create OperatorCode with Builtin Operator
518 std::map<circle::BuiltinOperator, int32_t> builtin_code_map =
519 gather_builtincode_map(model_recipe);
520 for (auto const &opcode : builtin_code_map)
521 {
522 circle::OperatorCodeBuilder code_builder{*flatbuffer_builder};
523 int8_t dep_code = 127; // BuiltinOperator_PLACEHOLDER_FOR_GREATER_OP_CODES
524 if (opcode.first < circle::BuiltinOperator_PLACEHOLDER_FOR_GREATER_OP_CODES)
525 dep_code = static_cast<int8_t>(opcode.first);
526 code_builder.add_deprecated_builtin_code(dep_code);
527 code_builder.add_builtin_code(opcode.first);
528 code_builder.add_version(opcode.second);
529 auto code = code_builder.Finish();
530 // Update OperatorCode vector
531 code_vec.emplace_back(code);
532 }
533
534 // Create OperatorCode with Custom Operator
535 std::set<std::string> custom_code_set = gather_customcode_set(model_recipe);
536 if (custom_code_set.size() &&
537 builtin_code_map.find(circle::BuiltinOperator_CUSTOM) == builtin_code_map.end())
538 builtin_code_map[circle::BuiltinOperator_CUSTOM] = 1;
539
540 for (auto opcode : custom_code_set)
541 {
542 auto custom_code = flatbuffer_builder->CreateString(opcode);
543 circle::OperatorCodeBuilder code_builder{*flatbuffer_builder};
544 code_builder.add_builtin_code(circle::BuiltinOperator_CUSTOM);
545 code_builder.add_custom_code(custom_code);
546 auto code = code_builder.Finish();
547 // Update OperatorCode vector
548 code_vec.emplace_back(code);
549 }
550
551 // Create an Empty Buffer
552 //
553 // Buffer 0 SHOULD be an empty buffer in TensorFlow Lite model file
554 // (Please refer to the comment for Tensor.buffer field in schema)
555 {
556 circle::BufferBuilder buffer_builder{*flatbuffer_builder};
557 buffer_vec.emplace_back(buffer_builder.Finish());
558 }
559
560 //
561 // Create Main graph
562 //
563 CookParams cp{buffer_vec, code_vec, subgraph_vec, flatbuffer_builder, builtin_code_map, "main"};
564
565 cook_graph<::circlechef::ModelRecipe>(model_recipe, cp);
566
567 //
568 // Create subgraphs if exist
569 //
570 for (int g = 0; g < model_recipe.graph_size(); ++g)
571 {
572 const auto &graph = model_recipe.graph(g);
573
574 std::ostringstream stringStream;
575 stringStream << "sub_" << (g + 1);
576
577 CookParams cp{buffer_vec, code_vec, subgraph_vec,
578 flatbuffer_builder, builtin_code_map, stringStream.str()};
579
580 cook_graph<::circlechef::Graph>(graph, cp);
581 }
582
583 // Create "Model" arguments
584 auto buffers = flatbuffer_builder->CreateVector(buffer_vec);
585 auto operator_codes = flatbuffer_builder->CreateVector(code_vec);
586 auto subgraphs = flatbuffer_builder->CreateVector(subgraph_vec);
587 auto description = flatbuffer_builder->CreateString("Generated by circlechef");
588
589 // Create "Model"
590 circle::ModelBuilder model_builder{*flatbuffer_builder};
591
592 model_builder.add_version(3);
593 model_builder.add_operator_codes(operator_codes);
594 model_builder.add_subgraphs(subgraphs);
595 model_builder.add_description(description);
596 model_builder.add_buffers(buffers);
597
598 auto model = model_builder.Finish();
599
600 // Finalize
601 ::circle::FinishModelBuffer(*flatbuffer_builder, model);
602
603 // Return "GenerateModel"
604 return GeneratedModel{
605 std::unique_ptr<GeneratedModelImpl>(new GeneratedModelImpl(std::move(flatbuffer_builder)))};
606}
