41void Minimum::execute() const
42{
43 switch (input1()->element_type())
44 {
45 case DataType::FLOAT32:
46 evalMinimum<float>();
47 break;
48 case DataType::U8:
49 evalMinimum<uint8_t>();
50 break;
51 default:
52 throw std::runtime_error("Unsupported type.");
53 }
54}

56template <typename T> inline void Minimum::evalMinimum() const
57{
58 BinaryOpBroadcastSlow(getTensorShape(input1()), getTensorData<T>(input1()),
59 getTensorShape(input2()), getTensorData<T>(input2()),
60 getTensorShape(output()), getTensorData<T>(output()),
61 [](T x, T y) { return std::min(x, y); });
62}
