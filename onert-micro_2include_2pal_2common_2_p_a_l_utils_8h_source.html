56inline std::int32_t saturatingRoundingDoublingHighMul(std::int32_t a, std::int32_t b)
57{
58 bool overflow = a == b && a == std::numeric_limits<std::int32_t>::min();
59 std::int64_t a_64(a);
60 std::int64_t b_64(b);
61 std::int64_t ab_64 = a_64 * b_64;
62 std::int32_t nudge = ab_64 >= 0 ? (1 << 30) : (1 - (1 << 30));
63 std::int32_t ab_x2_high32 = static_cast<std::int32_t>((ab_64 + nudge) / (1ll << 31));
64 return overflow ? std::numeric_limits<std::int32_t>::max() : ab_x2_high32;
65}

67// Correctly-rounded-to-nearest division by a power-of-two.
68// Also known as a rounding arithmetic right shift.
69inline int32_t roundingDivideByPOT(int32_t x, int32_t exponent)
70{
71 assert(exponent >= 0);
72 assert(exponent <= 31);
73 const int32_t mask = int32_t((1ll << exponent) - 1);
74 const int32_t zero = int32_t(0);
75 const int32_t one = int32_t(1);
76 const int32_t remainder = x & mask;
77 const int32_t threshold = (mask >> 1) + ((x < zero ? one : zero) & one);
78 return (x >> exponent) + ((remainder > threshold ? one : zero) & one);
79}

81inline int32_t multiplyByQuantizedMultiplier(int32_t x, int32_t quantized_multiplier, int shift)
82{
83 int left_shift = shift > 0 ? shift : 0;
84 int right_shift = shift > 0 ? 0 : -shift;
85 return roundingDivideByPOT(
86 saturatingRoundingDoublingHighMul(x * (1 << left_shift), quantized_multiplier), right_shift);
87}

89inline int32_t multiplyByQuantizedMultiplierSmallerThanOneExp(int32_t x,
90 int32_t quantized_multiplier,
91 int left_shift)
92{
93 return roundingDivideByPOT(saturatingRoundingDoublingHighMul(x, quantized_multiplier),

115// Gets offset of index if reducing on axis. When reducing, the flattened offset
116// will not change, if the input index changes on the given axis. For example,
117// if you have a 3D tensor and you are reducing to 2D by eliminating axis 0,
118// then index (0, 1, 2) and index (1, 1, 2) will map to the same flattened
119// offset.
120inline size_t reducedOutputOffset(const int32_t num_dims, const int32_t *dims, const int32_t *index,
121 const int32_t num_axis, const int32_t *axis)
122{
123 if (num_dims == 0)
124 {
125 return 0;
126 }
127 size_t offset = 0;
128 for (int idx = 0; idx < num_dims; ++idx)
129 {
130 // if we need to skip this axis
131 bool is_axis = false;
132 if (axis != nullptr)
133 {
134 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx)
135 {
136 if (idx == axis[axis_idx])
137 {
138 is_axis = true;
139 break;
140 }
141 }
142 }
143 if (!is_axis)
144 {
145 offset = offset * static_cast<size_t>(dims[idx]) + static_cast<size_t>(index[idx]);
146 }
147 }
148 return offset;
149}

151// Gets next index to iterate through a multidimensional array.
152inline bool nextIndex(const int32_t num_dims, const int32_t *dims, int32_t *current)
153{
154 if (num_dims == 0)
155 {
156 return false;
157 }
158 int carry = 1;
159 for (int idx = num_dims - 1; idx >= 0; --idx)
160 {
161 int current_val = current[idx] + carry;
162 if (dims[idx] == current_val)
163 {
164 current[idx] = 0;
165 }
166 else
167 {
168 current[idx] = current_val;
169 carry = 0;
170 break;
171 }
172 }
173 return (carry == 0);
174}

184// Data is required to be contiguous, and so many operators can use either the
185// full array flat size or the flat size with one dimension skipped (commonly
186// the depth).
187inline int flatSizeSkipDim(const int32_t *dims_data, int skip_dim, int num_dims)
188{
189 int flat_size = 1;
190 for (int i = 0; i < num_dims; ++i)
191 {
192 flat_size *= (i == skip_dim) ? 1 : dims_data[i];
193 }
194 return flat_size;
195}
