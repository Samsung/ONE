62void PermuteLayer::backward()
63{
64 for (uint32_t i = 0; i < _output_back_prop_tensors.size(); ++i)
65 {
66 auto src_back_prop = _output_back_prop_tensors.at(i);
67 auto dst_back_prop = _input_back_prop_tensors.at(i);
68
69 // NOTE The back propagation tensors corresponding to inputs/outputs of model are nullptr
70 // because permuting those tensors is meaningless
71 if (src_back_prop && dst_back_prop)
72 {
73 const auto rank = src_back_prop->getShape().rank();
74 auto output_offsets = _dst_tensors_offsets.at(i);
75 auto input_offsets = _src_tensors_offsets.at(i);
76
77 exec::IPermuteFunction::permute(src_back_prop, dst_back_prop, rank, output_offsets,
78 input_offsets);
79 }
80 }
