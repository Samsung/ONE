49void KernelGenerator::addClNode(const std::vector<ir::OperandIndex> &inputs,
50 const std::vector<ir::OperandIndex> &outputs,
51 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op)
52{
53 tflite::gpu::cl::CLNode cl_node;
54 cl_node.cl_operation.Init(std::move(gpu_op));
55 cl_node.inputs.resize(inputs.size());
56 for (size_t i = 0; i < inputs.size(); ++i)
57 {
58 cl_node.inputs[i] = inputs[i].value();
59 }
60 cl_node.outputs.resize(outputs.size());
61 for (size_t i = 0; i < outputs.size(); ++i)
62 {
63 cl_node.outputs[i] = outputs[i].value();
64 }
65 _nodes.push_back(std::move(cl_node));
66 _operation_indexes.push_back(_operation_index);
67 return;
68}

70void KernelGenerator::get_operation(FunctionMap &Functions)
71{
72 size_t size = _nodes.size();
73 size_t i = 0;
74 for (auto &it : Functions)
75 {
76 auto index = it.first;
77 auto node_index = _operation_indexes[i];
78 while (index == node_index)
79 {
80 auto &fn_seq = it.second;
81 auto &node = _nodes[i++];
82 for (size_t j = 0; j < node.inputs.size(); ++j)
83 {
84 uint32_t idx = node.inputs[j];
85 node.cl_operation.GetGpuOperation().SetSrc(
86 _tensor_reg->getClTensor(ir::OperandIndex{idx})->handle(), j);
87 }
88 for (size_t j = 0; j < node.outputs.size(); ++j)
89 {
90 uint32_t idx = node.outputs[j];
91 node.cl_operation.GetGpuOperation().SetDst(
92 _tensor_reg->getClTensor(ir::OperandIndex{idx})->handle(), j);
93 }
94 fn_seq->iterate([&](exec::IFunction &ifunc) {
95 static_cast<ClFunction &>(ifunc).add_operation(&node.cl_operation);
96 });
97 if (i == size)
98 {
99 break;
100 }
101 node_index = _operation_indexes[i];
102 }
103 if (i == size)
104 {
105 break;
106 }
107 }
108}

110absl::Status KernelGenerator::readConstTensor(const ir::OperandIndex &index,
111 tflite::gpu::TensorOrScalar *param)
112{
113 const auto shape = _ctx.at(index).shape();
114 if (shape.rank() == 0 && shape.num_elements() == 1)
115 {
116 tflite::gpu::Tensor<tflite::gpu::Scalar, tflite::gpu::DataType::FLOAT32> tensor;
117 tensor.shape.v = 1;
118 tensor.data.resize(1);
119 std::memcpy(&tensor.data[0], _ctx.at(index).data()->base(), _ctx.at(index).operandSize());
120 *param = tensor.data[0];
121 }
122 else
123 {
124 if (CheckIfLinearConvertible(&shape))
125 {
126 tflite::gpu::Tensor<tflite::gpu::Linear, tflite::gpu::DataType::FLOAT32> tensor;
127 tensor.shape.v = shape.dim(shape.rank() - 1);
128 tensor.data.resize(shape.num_elements());
129 std::memcpy(&tensor.data[0], _ctx.at(index).data()->base(), _ctx.at(index).operandSize());
130 *param = std::move(tensor);
131 }
132 else
133 {
134 tflite::gpu::Tensor<tflite::gpu::HWC, tflite::gpu::DataType::FLOAT32> tensor;
135 if (shape.rank() == 3)
136 {
137 tensor.shape.h = shape.dim(0);
138 tensor.shape.w = shape.dim(1);
139 tensor.shape.c = shape.dim(2);
140 }
141 else if (shape.rank() == 4)
142 {
143 if (shape.dim(0) != 1)
144 {
145 return absl::UnimplementedError("Batch size is not equal to 1.");
146 }
147 tensor.shape.h = shape.dim(1);
148 tensor.shape.w = shape.dim(2);
149 tensor.shape.c = shape.dim(3);
150 }
151 else
152 {
153 return absl::InvalidArgumentError(
154 "Expected a 3D tensor of shape HxWxC or a 4D tensor of shape 1xHxWxC.");
155 }
156 tensor.data.resize(shape.num_elements());
157 std::memcpy(&tensor.data[0], _ctx.at(index).data()->base(), _ctx.at(index).operandSize());
158 *param = std::move(tensor);
159 }
160 }
161 return absl::OkStatus();
162}

230void KernelGenerator::visit(const ir::operation::BinaryArithmetic &node)
231{
232 const auto ofm_index{node.getOutputs().at(0)};
233 const auto lhs_index{node.getInputs().at(ir::operation::BinaryArithmetic::Input::LHS)};
234 const auto rhs_index{node.getInputs().at(ir::operation::BinaryArithmetic::Input::RHS)};
235
236 tflite::gpu::OperationDef op_def;
237 op_def.precision = tflite::gpu::CalculationsPrecision::F32;
238
239 const bool lhs_const = _ctx.at(lhs_index).isConstant();
240 const bool rhs_const = _ctx.at(rhs_index).isConstant();
241
242 if (lhs_const && rhs_const)
243 {
244 throw std::runtime_error("No runtime input tensors for " + node.name());
245 }
246
247 auto fn = std::make_unique<ClFunction>(_creation_context);
248 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op;
249
250 tflite::gpu::OperationType op_type = convertArithmeticType(node.param().arithmetic_type);
251
252 if (!lhs_const && !rhs_const)
253 {
254 auto lhs_shape = _tensor_reg->getClTensor(lhs_index)->get_info()._shape;
255 auto rhs_shape = _tensor_reg->getClTensor(rhs_index)->get_info()._shape;
256
257 bool swap =
258 (op_type == tflite::gpu::OperationType::MUL) &&
259 (lhs_shape.h <= rhs_shape.h && lhs_shape.w <= rhs_shape.w && lhs_shape.c <= rhs_shape.c);
260
261 auto first_index = swap ? rhs_index : lhs_index;
262 auto second_index = swap ? lhs_index : rhs_index;
263
264 op_def.src_tensors.push_back(_tensor_reg->getClTensor(first_index)->get_info()._desc);
265 op_def.src_tensors.push_back(_tensor_reg->getClTensor(second_index)->get_info()._desc);
266 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(ofm_index)->get_info()._desc);
267
268 auto second_shape = _tensor_reg->getClTensor(second_index)->get_info()._shape;
269
270 tflite::gpu::GPUOperation operation = CreateElementwiseTwoInput(op_def, op_type, second_shape);
271 gpu_op = std::make_unique<tflite::gpu::GPUOperation>(std::move(operation));
272
273 addClNode({first_index, second_index}, {ofm_index}, std::move(gpu_op));
274 }
275 else
276 {
277 auto non_const_index = rhs_const ? lhs_index : rhs_index;
278 auto const_index = rhs_const ? rhs_index : lhs_index;
279
280 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(ofm_index)->get_info()._desc);
281 op_def.src_tensors.push_back(_tensor_reg->getClTensor(non_const_index)->get_info()._desc);
282
283 tflite::gpu::ElementwiseAttributes attr;
284
285 if (!readConstTensor(const_index, &attr.param).ok())
286 {
287 throw std::runtime_error("BinaryArithmetic unsupported constant tensor");
288 }
289
290 tflite::gpu::GPUOperation operation =
291 CreateElementwise(_creation_context->GetGpuInfo(), op_def, op_type, attr);
292 gpu_op = absl::make_unique<tflite::gpu::GPUOperation>(std::move(operation));
293
294 addClNode({non_const_index}, {ofm_index}, std::move(gpu_op));
295 }
296 _return_fn = std::move(fn);
297}

299void KernelGenerator::visit(const ir::operation::Conv2D &node)
300{
301 auto output{node.getOutputs().at(0)};
302
303 auto input{node.getInputs().at(ir::operation::Conv2D::INPUT)};
304 auto kernel{node.getInputs().at(ir::operation::Conv2D::KERNEL)};
305 auto bias{node.getInputs().at(ir::operation::Conv2D::BIAS)};
306
307 const auto param = node.param();
308
309 tflite::gpu::OperationDef op_def;
310 op_def.precision = tflite::gpu::CalculationsPrecision::F32;
311
312 op_def.src_tensors.push_back(_tensor_reg->getClTensor(input)->get_info()._desc);
313
314 auto input_shape = _tensor_reg->getClTensor(input)->get_info()._shape;
315 auto kernel_shape = _tensor_reg->getClTensor(kernel)->get_info()._shape;
316 auto output_shape = _tensor_reg->getClTensor(output)->get_info()._shape;
317 auto bias_shape = _tensor_reg->getClTensor(bias)->get_info()._shape;
318
319 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(output)->get_info()._desc);
320
321 tflite::gpu::ModelHints hints;
322 std::unique_ptr<tflite::gpu::GPUOperation>
323 gpu_op; // = InitSingleOpSubgraph(inputs, outputs, gpu_subgraph);
324
325 auto kernel_tensor = _tensor_reg->getClTensor(kernel);
326 auto bias_tensor = _tensor_reg->getClTensor(bias);
327
328 tflite::gpu::Convolution2DAttributes attr;
329 attr.strides = ToHW(param.stride.vertical, param.stride.horizontal);
330 attr.dilations =
331 tflite::gpu::HW(std::max(static_cast<u_int32_t>(1), param.dilation.height_factor),
332 std::max(static_cast<u_int32_t>(1), param.dilation.width_factor));
333
334 bool is_weight = (_ctx.at(kernel).isConstant() ? true : false);
335
336 if (is_weight)
337 {
338 attr.weights.id = kernel.value();
339 attr.weights.shape.o = kernel_shape.b;
340 attr.weights.shape.h = kernel_shape.h;
341 attr.weights.shape.w = kernel_shape.w;
342 attr.weights.shape.i = kernel_shape.c;
343 attr.weights.data.resize(kernel_shape.DimensionsProduct());
344 memcpy(attr.weights.data.data(), _ctx.at(kernel).data()->base(), kernel_tensor->total_size());
345 }
346
347 attr.bias.id = bias.value();
348 // TODO Modify
349 attr.bias.shape.v = bias_shape.b != 1 ? bias_shape.b : bias_shape.c;
350 attr.bias.data.resize(bias_shape.DimensionsProduct());
351 memcpy(attr.bias.data.data(), _ctx.at(bias).data()->base(), bias_tensor->total_size());
352
353 UpdatePadding(param.padding.type, input_shape, &attr);
354
355 gpu_op = SelectConvolution(attr, output_shape, _creation_context->GetGpuInfo(), op_def, hints);
356
357 tflite::gpu::cl::CLNode cl_node;
358 cl_node.inputs.resize(1);
359 cl_node.inputs[0] = input.value();
360 cl_node.outputs.resize(1);
361
362 auto fn = std::make_unique<ClFunction>(_creation_context);
363
364 const auto activation = node.param().activation;
365
366 switch (activation)
367 {
368 case ir::Activation::NONE:
369 {
370 addClNode({input}, {output}, std::move(gpu_op));
371 break;
372 }
373 case ir::Activation::RELU:
374 case ir::Activation::RELU6:
375 {
376 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op_1;
377 tflite::gpu::OperationDef op_def_1;
378 const auto shape = _ctx.at(output).shape();
379 auto new_ind = _tensor_reg->addNewClTensor(shape);
380
381 addClNode({input}, {new_ind}, std::move(gpu_op));
382
383 op_def_1.precision = tflite::gpu::CalculationsPrecision::F32;
384 op_def_1.src_tensors.push_back(_tensor_reg->getClTensor(output)->get_info()._desc);
385 op_def_1.dst_tensors.push_back(_tensor_reg->getClTensor(output)->get_info()._desc);
386
387 tflite::gpu::ReLUAttributes attr_1;
388 if (activation == ir::Activation::RELU6)
389 {
390 attr_1.clip = 6;
391 }
392 else
393 {
394 attr_1.clip = 0;
395 }
396 attr_1.alpha = 0;
397 gpu_op_1 = SelectReLU(attr_1, op_def_1);
398
399 addClNode({new_ind}, {output}, std::move(gpu_op_1));
400 break;
401 }
402 default:
403 {
404 throw std::runtime_error("gpu_cl KernelGenerator : Not supported Conv2D activiation");
405 }
406 }
407 _return_fn = std::move(fn);
408}

410void KernelGenerator::visit(const ir::operation::DepthwiseConv2D &node)
411{
412 using ir::operation::DepthwiseConv2D;
413
414 const auto ofm_index{node.getOutputs().at(0)};
415 const auto ifm_index{node.getInputs().at(DepthwiseConv2D::Input::INPUT)};
416 const auto ker_index{node.getInputs().at(DepthwiseConv2D::Input::KERNEL)};
417 const auto bias_index{node.getInputs().at(DepthwiseConv2D::Input::BIAS)};
418
419 const auto stride = node.param().stride;
420 const auto dilation = node.param().dilation;
421 const auto padding = node.param().padding;
422
423 const auto multiplier = node.param().multiplier;
424
425 bool is_weight = (_ctx.at(ker_index).isConstant() ? true : false);
426 tflite::gpu::OperationDef op_def;
427 op_def.precision = tflite::gpu::CalculationsPrecision::F32;
428
429 op_def.src_tensors.push_back(_tensor_reg->getClTensor(ifm_index)->get_info()._desc);
430 auto input_shape = _tensor_reg->getClTensor(ifm_index)->get_info()._shape;
431
432 auto ker_shape = _tensor_reg->getClTensor(ker_index)->get_info()._shape;
433
434 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(ofm_index)->get_info()._desc);
435 auto out_shape = _tensor_reg->getClTensor(ofm_index)->get_info()._shape;
436 auto bias_shape = _tensor_reg->getClTensor(bias_index)->get_info()._shape;
437
438 tflite::gpu::DepthwiseConvolution2DAttributes attr;
439 attr.strides = ToHW(stride.vertical, stride.horizontal);
440 attr.dilations = tflite::gpu::HW(std::max(static_cast<u_int32_t>(1), dilation.height_factor),
441 std::max(static_cast<u_int32_t>(1), dilation.width_factor));
442
443 if (is_weight)
444 {
445 attr.weights.id = ker_index.value();
446 attr.weights.shape.o = ker_shape.b;
447 attr.weights.shape.h = ker_shape.h;
448 attr.weights.shape.w = ker_shape.w;
449 attr.weights.shape.i = ker_shape.c;
450 attr.weights.data.resize(ker_shape.DimensionsProduct());
451 memcpy(attr.weights.data.data(), _ctx.at(ker_index).data()->base(),
452 _ctx.at(ker_index).operandSize());
453 }
454 attr.bias.id = bias_index.value();
455 attr.bias.shape.v = bias_shape.b != 1 ? bias_shape.b : bias_shape.c;
456 attr.bias.data.resize(bias_shape.DimensionsProduct());
457 memcpy(attr.bias.data.data(), _ctx.at(bias_index).data()->base(),
458 _ctx.at(bias_index).operandSize());
459 UpdatePadding(padding.type, input_shape, &attr);
460
461 if (multiplier != 1)
462 {
463 const int input_depth = input_shape.c;
464 const int filter_height = ker_shape.h;
465 const int filter_width = ker_shape.w;
466 const int output_depth = out_shape.c;
467
468 tflite::gpu::Tensor<tflite::gpu::OHWI, tflite::gpu::DataType::FLOAT32> weights;
469 weights.id = attr.weights.id;
470 weights.shape = tflite::gpu::OHWI(output_depth, filter_height, filter_width, input_depth);
471 weights.data.resize(weights.shape.DimensionsProduct());
472 float *dst = &weights.data[0];
473 for (int j = 0; j < output_depth; ++j)
474 {
475 const float *src = attr.weights.data.data() + j;
476 for (int i = 0; i < filter_height * filter_width; ++i)
477 {
478 *dst = *src;
479 dst++;
480 src += output_depth;
481 }
482 }
483 attr.weights = std::move(weights);
484 }
485
486 auto fn = std::make_unique<ClFunction>(_creation_context);
487 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op;
488
489 if (is_weight)
490 {
491 gpu_op = SelectDWConvolution(attr, _creation_context->GetGpuInfo(), op_def);
492 }
493 else
494 {
495 if (ker_shape.b != 1)
496 {
497 throw std::runtime_error(
498 "No support of depthwise runtime weights with channel multiplier != 1");
499 }
500 gpu_op = SelectDWConvolutionDynamicWeights(attr, _creation_context->GetGpuInfo(), op_def);
501 }
502
503 const auto activation = node.param().activation;
504
505 switch (activation)
506 {
507 case ir::Activation::NONE:
508 {
509 addClNode({ifm_index}, {ofm_index}, std::move(gpu_op));
510 break;
511 }
512 case ir::Activation::RELU:
513 case ir::Activation::RELU6:
514 {
515 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op_1;
516 tflite::gpu::OperationDef op_def_1;
517 const auto shape = _ctx.at(ofm_index).shape();
518 auto new_ind = _tensor_reg->addNewClTensor(shape);
519
520 addClNode({ifm_index}, {new_ind}, std::move(gpu_op));
521
522 op_def_1.precision = tflite::gpu::CalculationsPrecision::F32;
523
524 op_def_1.src_tensors.push_back(_tensor_reg->getClTensor(ofm_index)->get_info()._desc);
525 op_def_1.dst_tensors.push_back(_tensor_reg->getClTensor(ofm_index)->get_info()._desc);
526
527 tflite::gpu::ReLUAttributes attr_1;
528 if (activation == ir::Activation::RELU6)
529 {
530 attr_1.clip = 6;
531 }
532 else
533 {
534 attr_1.clip = 0;
535 }
536 attr_1.alpha = 0;
537 gpu_op_1 = SelectReLU(attr_1, op_def_1);
538
539 addClNode({new_ind}, {ofm_index}, std::move(gpu_op_1));
540 break;
541 }
542 default:
543 {
544 throw std::runtime_error("gpu_cl KernelGenerator : Not supported DepthwiseConv2D acvivation");
545 }
546 }
547
548 _return_fn = std::move(fn);
549}

551void KernelGenerator::visit(const ir::operation::ElementwiseActivation &node)
552{
553 const auto output_index{node.getOutputs().at(0)};
554 const auto input_index{node.getInputs().at(ir::operation::ElementwiseActivation::Input::INPUT)};
555
556 tflite::gpu::OperationDef op_def;
557 op_def.precision = tflite::gpu::CalculationsPrecision::F32;
558
559 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(output_index)->get_info()._desc);
560 op_def.src_tensors.push_back(_tensor_reg->getClTensor(input_index)->get_info()._desc);
561
562 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op;
563 auto fn = std::make_unique<ClFunction>(_creation_context);
564 switch (node.param().op_type)
565 {
566 case ir::operation::ElementwiseActivation::Type::LEAKY_RELU:
567 case ir::operation::ElementwiseActivation::Type::RELU:
568 {
569 tflite::gpu::ReLUAttributes attr;
570 if (ir::operation::ElementwiseActivation::Type::LEAKY_RELU == node.param().op_type)
571 {
572 attr.alpha = node.param().alpha;
573 attr.clip = 0;
574 }
575 else
576 {
577 attr.alpha = node.param().beta;
578 attr.clip = node.param().alpha;
579 }
580 gpu_op = SelectReLU(attr, op_def);
581 break;
582 }
583 case ir::operation::ElementwiseActivation::Type::LOGISTIC:
584 {
585 if (_ctx.at(input_index).typeInfo().type() != ir::DataType::FLOAT32)
586 {
587 throw std::runtime_error{"Unsupported data type of LOGISTIC"};
588 }
589 tflite::gpu::GPUOperation operation =
590 CreateElementwiseOneInput(_creation_context->GetGpuInfo(), op_def,
591 convertElementwiseActivationType(node.param().op_type));
592 gpu_op = std::make_unique<tflite::gpu::GPUOperation>(std::move(operation));
593 break;
594 }
595 case ir::operation::ElementwiseActivation::Type::TANH:
596 {
597 tflite::gpu::GPUOperation operation = CreateElementwiseOneInput(
598 _creation_context->GetGpuInfo(), op_def, tflite::gpu::OperationType::TANH);
599 gpu_op = std::make_unique<tflite::gpu::GPUOperation>(std::move(operation));
600 break;
601 }
602 default:
603 throw std::runtime_error(
604 "gpu_cl KernelGenerator : Not supported operation on ElementwiseActivation");
605 }

610void KernelGenerator::visit(const ir::operation::Pool2D &node)
611{
612 const auto output_index{node.getOutputs().at(0)};
613 const auto input_index{node.getInputs().at(ir::operation::Pool2D::Input::INPUT)};
614
615 tflite::gpu::OperationDef op_def;
616 op_def.precision = tflite::gpu::CalculationsPrecision::F32;
617
618 op_def.src_tensors.push_back(_tensor_reg->getClTensor(input_index)->get_info()._desc);
619 auto input_shape = _tensor_reg->getClTensor(input_index)->get_info()._shape;
620
621 op_def.dst_tensors.push_back(_tensor_reg->getClTensor(output_index)->get_info()._desc);
622
623 const auto kh = node.param().kh;
624 const auto kw = node.param().kw;
625 const auto stride = node.param().stride;
626 const auto op_type = convertPoolType(node.param().op_type);
627
628 tflite::gpu::Pooling2DAttributes attributes;
629 attributes.type = op_type;
630 attributes.kernel = tflite::gpu::HW(kh > 0 ? kh : 1, kw > 0 ? kw : 1);
631 attributes.strides = tflite::gpu::HW(stride.vertical > 0 ? stride.vertical : 1,
632 stride.horizontal > 0 ? stride.horizontal : 1);
633
634 if (node.param().padding.type == ir::PaddingType::SAME)
635 {
636 attributes.padding = CalculateSamePadding(input_shape, attributes);
637 }
638 else
639 {
640 attributes.padding.prepended = tflite::gpu::HW(0, 0);
641 attributes.padding.appended = tflite::gpu::HW(0, 0);
642 }
643
644 auto fn = std::make_unique<ClFunction>(_creation_context);
645 std::unique_ptr<tflite::gpu::GPUOperation> gpu_op;
646 gpu_op = SelectPooling(attributes, op_def);
647
648 addClNode({input_index}, {output_index}, std::move(gpu_op));
649 _return_fn = std::move(fn);
650}
