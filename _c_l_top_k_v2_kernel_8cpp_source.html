56void CLTopKV2Single::configure(ICLTensor *input, ICLTensor *topk_values, ICLTensor *topk_indices,
57 cl::Buffer *indices, cl::Buffer *temp_stack, int k, int n)
58{
59 ARM_COMPUTE_ERROR_ON(input == nullptr && indices == nullptr);
60 ARM_COMPUTE_ERROR_ON(topk_values == nullptr && topk_indices == nullptr);
61 ARM_COMPUTE_ERROR_ON(n == 0);
62
63 _input = input;
64 _topk_values = topk_values;
65 _topk_indices = topk_indices;
66
67 // Set kernel build options
68 std::set<std::string> build_opts;
69
70 // Create kernel
71 _kernel = static_cast<cl::Kernel>(
72 CLKernelLibraryEx::get().create_kernel("topkv2_quicksort", build_opts));
73
74 unsigned int idx = 3 * num_arguments_per_1D_tensor();
75 _kernel.setArg(idx++, *indices);
76 _kernel.setArg(idx++, *temp_stack);
77 _kernel.setArg<cl_int>(idx++, k);
78 _kernel.setArg<cl_int>(idx++, n);
79
80 // Configure kernel window
81 Window win;
82 win.set(0, Window::Dimension(0, 1, 1));
83 ICLKernel::configure_internal(win);
84}

144void CLRadixSortHistogram::configure(cl::Buffer *hist_buf, int bits, int n)
145{
146 ARM_COMPUTE_ERROR_ON(hist_buf == nullptr);
147
148 unsigned int radix = 1 << bits;
149 // Set kernel build options
150 std::set<std::string> build_opts;
151 build_opts.emplace("-D_BITS=" + support::cpp11::to_string(bits));
152 build_opts.emplace("-D_RADIX=" + support::cpp11::to_string(radix));
153 build_opts.emplace("-DPERMUT=1");
154
155 // Create kernel
156 _kernel = static_cast<cl::Kernel>(
157 CLKernelLibraryEx::get().create_kernel("radixsort_histogram", build_opts));
158
159 int loc_histo_size = radix * _ITEMS * sizeof(cl_int);
160
161 unsigned int idx = 1;
162 _kernel.setArg(idx++, *hist_buf);
163
164 idx = 3;
165 _kernel.setArg(idx++, loc_histo_size, nullptr);
166 _kernel.setArg<cl_int>(idx++, n);
167
168 // Configure kernel window
169 Window win;
170 win.set(0, Window::Dimension(0, _GROUPS * _ITEMS, 1));
171 ICLKernel::configure_internal(win);
172}

181
182 cl::NDRange lws = cl::NDRange(_ITEMS, 1);
