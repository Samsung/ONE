123mir::Operation *fillGraph(
124 mir::Graph &g,
125 const function<mir::Operation *(mir::Graph &g, vector<mir::Operation::Output *> &inputs)> &op_gen,
126 const vector<unique_ptr<mir::TensorVariant>> &input_ntensors)
127{
128 // Create operation inputs.
129 vector<mir::Operation::Output *> inputs;
130 for (const unique_ptr<mir::TensorVariant> &ntensor : input_ntensors)
131 {
132 auto input = g.create<mir::ops::ConstantOp>(*ntensor)->getOutput(0);
133 inputs.push_back(input);
134 }
135
136 return op_gen(g, inputs);
137}

142void fillNShape(mir::Shape &nshape, const vector<int> &raw_shape_data)
143{
144 int shape_rank = raw_shape_data.size();
145 nshape.resize(shape_rank);
146 for (int i = 0; i < shape_rank; ++i)
147 nshape.dim(i) = raw_shape_data[i];
148}
149
153void copyAShapeFromNShape(Shape &ashape, const mir::Shape &src)
154{
155 int shape_rank = src.rank();
156 ashape.setDims(shape_rank);
157 for (int i = 0; i < shape_rank; ++i)
158 ashape[i] = src.dim(i);
159}
160
164void fillShapes(mir::Shape &nshape, Shape &ashape, const vector<int> &raw_shape_data)
165{
166 fillNShape(nshape, raw_shape_data);
167 copyAShapeFromNShape(ashape, nshape);
168}

173void fillNTensor(mir::TensorVariant &dst, float start)
174{
175 float t = start;
176 mir::Tensor<float> wrapper(dst);
177 for (const mir::Index &idx : mir::ShapeRange(dst.getShape()))
178 {
179 wrapper.at(idx) = sin(t) * 2.0f;
180 t += 1.0f;
181 }
182}
183
187void copyATensorFromNTensor(Tensor &dst, mir::TensorVariant &src)
188{
189 mir::Tensor<float> wrapper(src);
190 Index art_idx;
191 int rank = src.getShape().rank();
192 art_idx.setDims(rank);
193 for (mir::Index idx : mir::ShapeRange(src.getShape()))
194 {
195 for (int i = 0; i < rank; ++i)
196 art_idx[i] = idx.at(i);
197 dst.at(art_idx) = wrapper.at(idx);
198 }
199}

204void fillTensors(unique_ptr<mir::TensorVariant> &ntensor, Tensor &atensor, const vector<int> &shape,
205 float start)
206{
207 Shape ashape;
208 mir::Shape nshape;
209 fillShapes(nshape, ashape, shape);
210 atensor.reshape(ashape);
211 ntensor.reset(new mir::TensorVariant(mir::DataType::FLOAT32, nshape));
212 fillNTensor(*ntensor, start);
213 copyATensorFromNTensor(atensor, *ntensor);
214}

247bool areFloatsNear(float a, float b, int32_t ulp, float eps)
248{
249 assert(ulp < (1 << 23) && "this algorithm is not applicable for such large diffs");
250 assert(eps >= 0 && "epsilon should be positive number");
251 if (fabs(a - b) <= eps)
252 return true;
253 // since this point need to dind difference between numbers
254 // in terms of ULP
255 int32_t ai;
256 int32_t bi;
257 memcpy(&ai, &a, sizeof(float));
258 memcpy(&bi, &b, sizeof(float));
259 // compare mantissa of numbers
260 if (ai > bi)
261 return ai - bi <= ulp;
262 return bi - ai <= ulp;
263}

270void compareResults(const mir::TensorVariant &ref_nnc_tensor, const Tensor &test_art_tensor)
271{
272 assert(ref_nnc_tensor.getElementSize() == 4L &&
273 ref_nnc_tensor.getDataType() == mir::DataType::FLOAT32);
274
275 const mir::Shape &nnc_shape = ref_nnc_tensor.getShape();
276 const Shape &art_shape = test_art_tensor.getShape();
277
278 // check that reference and test shapes are equal
279 ASSERT_EQ(nnc_shape.rank(), art_shape.getDims());
280
281 int rank = nnc_shape.rank();
282 for (int i = 0; i < rank; ++i)
283 ASSERT_EQ(nnc_shape.dim(i), art_shape[i]);
284
285 // check that reference and test tensor contents are equal
286 Index artifact_idx;
287 artifact_idx.setDims(rank);
288 for (mir::Index nnc_idx : mir::ShapeRange(nnc_shape))
289 {
290 for (int i = 0; i < rank; ++i)
291 artifact_idx[i] = nnc_idx.at(i);
292 // Input and output data lies in range of [-10, 10],
293 // chosen epsilon lies near the edge of float type computational precision
294 float ref_data = mir::Tensor<float>(ref_nnc_tensor).at(nnc_idx);
295 float test_data = test_art_tensor.at(artifact_idx);
296 ASSERT_TRUE(areFloatsNear(ref_data, test_data, 32, 1e-5))
297 << "Tensor element " << nnc_idx << " diverged, reference: " << ref_data
298 << " test result: " << test_data;
299 }

306template <typename TestFunc, typename... Args>
307void createAndRunTestGraph(
308 function<mir::Operation *(mir::Graph &, const std::vector<mir::Operation::Output *> &inputs)>
309 op_generator,
310 TestFunc artifactOperation, const vector<unique_ptr<mir::TensorVariant>> &input_ntensors,
311 Args &... input_atensors)
312{
313 mir::Graph g;
314 mir::Operation *actual_operation = fillGraph(g, op_generator, input_ntensors);
315
316 // serialize data for soft backend operation
317 vector<unique_ptr<sir::Action>> inference_sequence;
318 unique_ptr<sir::CallFunction> op_call(new sir::CallFunction);
319 op_call->mirOp = actual_operation;
320 inference_sequence.push_back(std::move(op_call));
321 Serializer serializer;
322 serializer.serialize(inference_sequence);
323 assert(static_cast<sir::CallFunction *>(inference_sequence.front().get())->paramStartOffset == 0);
324
325 mir::TensorVariant reference_output = getReferenceTensor(g, actual_operation);
326
327 Tensor test_output;
328 artifactOperation(test_output, serializer.getBuffer().data(), input_atensors...);
329
330 compareResults(reference_output, test_output);
331}
