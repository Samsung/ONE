32void dump_buffer(std::ostream &os, const uint8_t *buffer, size_t size, size_t amount)
33{
34 std::ios_base::fmtflags saveflags(os.flags());
35
36 bool second = false;
37 bool ellipsis = amount > 0 && size > 4;
38 size_t count = ellipsis ? std::min(size, amount) : size;
39
40 for (size_t i = 0; i < count; i++)
41 {
42 if (second)
43 {
44 os << " ";
45 }
46
47 os << std::showbase << std::setfill('0') << std::setw(2);
48 os << std::hex << (uint32_t)buffer[i];
49
50 second = true;
51 }
52 if (ellipsis)
53 {
54 os << " ...";
55 }
56
57 os.flags(saveflags);
58}

125void dump_sub_graph(std::ostream &os, mio::circle::Reader &reader)
126{
127 auto tensors = reader.tensors();
128 auto operators = reader.operators();
129 auto data_format = reader.data_format();
130
131 // dump data_format
132 os << "Data Format:" << std::endl;
133 if (data_format == circle::DataFormat::DataFormat_CHANNELS_LAST)
134 {
135 os << "CHANNEL_LAST (NHWC for 2d, NDHWC for 3d data)" << std::endl;
136 }
137 else if (data_format == circle::DataFormat::DataFormat_CHANNELS_FIRST)
138 {
139 os << "CHANNEL_FIRST (NCHW for 2d, NCDHW for 3d data)" << std::endl;
140 }
141 os << std::endl;
142
143 // dump operands(tensors)
144 os << "Operands: T(subgraph index : tensor index) TYPE (shape) (shape_signature) "
145 << "B(buffer index) (variable) OperandName" << std::endl;
146 for (uint32_t i = 0; i < tensors->Length(); ++i)
147 {
148 // TODO refactor to some better structure
149 auto tensor = tensors->Get(i);
150 std::vector<int32_t> dims = {-1};
151
152 if (tensor->shape())
153 dims = mio::circle::as_index_vector(tensor->shape());
154
155 os << "T(" << reader.subgraph_index() << ":" << i << ") " << mio::circle::tensor_type(tensor)
156 << " ";
157 os << "(" << dims << ") ";
158 if (tensor->shape_signature())
159 {
160 std::vector<int32_t> dims_sig = mio::circle::as_index_vector(tensor->shape_signature());
161 os << "(" << dims_sig << ") ";
162 }
163 os << "B(" << tensor->buffer() << ") ";
164 if (tensor->is_variable())
165 {
166 os << "(variable) ";
167 }
168 os << mio::circle::tensor_name(tensor) << std::endl;
169
170 if (auto q_params = tensor->quantization())
171 {
172 if ((q_params->min() && q_params->max()) || (q_params->scale() && q_params->zero_point()))
173 {
174 std::string strquantiz = " Quantization: ";
175 std::string strqindent(strquantiz.size(), ' ');
176 os << strquantiz;
177
178 if (q_params->min())
179 {
180 os << "min(" << q_params->min() << ") ";
181 if (q_params->min()->size() > 1)
182 os << std::endl << strqindent;
183 }
184 if (q_params->max())
185 {
186 os << "max(" << q_params->max() << ") ";
187 if (q_params->max()->size() > 1)
188 os << std::endl << strqindent;
189 }
190 if (q_params->scale())
191 {
192 os << "scale(" << q_params->scale() << ") ";
193 if (q_params->scale()->size() > 1)
194 os << std::endl << strqindent;
195 }
196 if (q_params->zero_point())
197 {
198 os << "zeropt(" << q_params->zero_point() << ") ";
199 if (q_params->zero_point()->size() > 1)
200 os << std::endl << strqindent;
201 }
202 os << "quantized_dimension(" << q_params->quantized_dimension() << ")";
203
204 os << std::endl;
205 }
206 }
207
208 if (const auto &s_params = tensor->sparsity())
209 {
210 std::string strsparsity = " Sparsity: ";
211 std::string strsindent(strsparsity.size(), ' ');
212 os << strsparsity;
213
214 if (s_params->traversal_order())
215 {
216 os << "traversal_order(" << s_params->traversal_order() << ") ";
217 os << std::endl << strsindent;
218 }
219 if (s_params->block_map())
220 {
221 os << "block_map(" << s_params->block_map() << ") ";
222 os << std::endl << strsindent;
223 }
224 if (const auto &dim_metadata = s_params->dim_metadata())
225 {
226 uint32_t idx = 0;
227 for (const auto &dm : *dim_metadata)
228 {
229 std::string strdm = "dim_metadata[" + std::to_string(idx++) + "]: ";
230 std::string strdm_indent = strsindent + std::string(strdm.size(), ' ');
231 os << strdm;
232
233 os << "format(" << circle::EnumNameDimensionType(dm->format()) << ") ";
234 os << std::endl << strdm_indent;
235
236 os << "dense_size(" << dm->dense_size() << ") ";
237 os << std::endl << strdm_indent;
238
239 os << "array_segments_type("
240 << circle::EnumNameSparseIndexVector(dm->array_segments_type()) << ") ";
241 os << std::endl << strdm_indent;
242
243 os << "array_segments(";
244 switch (dm->array_segments_type())
245 {
246 case circle::SparseIndexVector_NONE:
247 // DO NOTHING
248 break;
249 case circle::SparseIndexVector_Int32Vector:
250 os << dm->array_segments_as_Int32Vector()->values();
251 break;
252 case circle::SparseIndexVector_Uint16Vector:
253 os << dm->array_segments_as_Uint16Vector()->values();
254 break;
255 case circle::SparseIndexVector_Uint8Vector:
256 os << dm->array_segments_as_Uint8Vector()->values();
257 break;
258 default:
259 throw std::runtime_error("Invalid SparseIndexVector type of array_segments");
260 }
261 os << ")" << std::endl << strdm_indent;
262
263 os << "array_indices_type(" << circle::EnumNameSparseIndexVector(dm->array_indices_type())
264 << ") ";
265 os << std::endl << strdm_indent;
266
267 os << "array_indices(";
268 switch (dm->array_indices_type())
269 {
270 case circle::SparseIndexVector_NONE:
271 // DO NOTHING
272 break;
273 case circle::SparseIndexVector_Int32Vector:
274 os << dm->array_indices_as_Int32Vector()->values();
275 break;
276 case circle::SparseIndexVector_Uint16Vector:
277 os << dm->array_indices_as_Uint16Vector()->values();
278 break;
279 case circle::SparseIndexVector_Uint8Vector:
280 os << dm->array_indices_as_Uint8Vector()->values();
281 break;
282 default:
283 throw std::runtime_error("Invalid SparseIndexVector type of array_indices");
284 }
285 os << ")" << std::endl << strsindent;
286 }
287 }
288 }
289 os << std::endl;
290 }
291
292 // dump operators
293 os << "Operators: O(subgraph index : operator index) OpCodeName " << std::endl;
294 os << " Option(values) ... <-- depending on OpCode" << std::endl;
295 os << " I T(tensor index) OperandName <-- as input" << std::endl;
296 os << " O T(tensor index) OperandName <-- as output" << std::endl;
297 for (uint32_t i = 0; i < operators->Length(); ++i)
298 {
299 const auto op = operators->Get(i);
300 circle::BuiltinOperator builtincode = reader.builtin_code(op);
301
302 const std::vector<int32_t> &inputs = mio::circle::as_index_vector(op->inputs());
303 const std::vector<int32_t> &outputs = mio::circle::as_index_vector(op->outputs());
304 auto op_name = reader.opcode_name(op);
305
306 os << "O(" << reader.subgraph_index() << ":" << i << ") " << op_name << " ";
307 os << std::endl;
308
309 if (auto op_prn = OpPrinterRegistry::get().lookup(builtincode))
310 {
311 op_prn->options(op, os);
312 }
313
314 for (auto input : inputs)
315 {
316 os << " I T(" << reader.subgraph_index() << ":" << input << ") ";
317 if (input >= 0)
318 {
319 auto tensor = tensors->Get(input);
320 os << mio::circle::tensor_name(tensor);
321 }
322 os << std::endl;
323 }
324 for (auto output : outputs)
325 {
326 os << " O T(" << reader.subgraph_index() << ":" << output << ") ";
327 if (output >= 0)
328 {
329 auto tensor = tensors->Get(output);
330 os << mio::circle::tensor_name(tensor);
331 }
332 os << std::endl;
333 }
334 }
335 os << std::endl;
336
337 // dump network inputs/outputs
338 os << "Inputs/Outputs: I(input)/O(output) T(tensor index) OperandName" << std::endl;
339
340 for (const auto input : reader.inputs())
341 {
342 auto tensor = tensors->Get(input);
343 std::string name = mio::circle::tensor_name(tensor);
344 os << "I T(" << reader.subgraph_index() << ":" << input << ") " << name << std::endl;
345 }
346
347 for (const auto output : reader.outputs())
348 {
349 auto tensor = tensors->Get(output);
350 std::string name = mio::circle::tensor_name(tensor);
351 os << "O T(" << reader.subgraph_index() << ":" << output << ") " << name << std::endl;
352 }
353
354 os << std::endl;
355}

357void dump_model(std::ostream &os, const circle::Model *model)
358{
359 mio::circle::Reader reader(model);
360
361 uint32_t num_subgraph = reader.num_subgraph();
362
363 // dump model version
364 os << "===================================================================" << std::endl;
365 os << "Model version: " << reader.version() << std::endl;
366 os << " # sub graphs: " << num_subgraph << std::endl;
367 os << std::endl;
368
369 auto opcodes = reader.opcodes();
370 auto buffers = reader.buffers();
371 auto metadata = reader.metadata();
372 auto signaturedefs = reader.signature_defs();
373
374 // dump operator_codes
375 os << "Operator Codes: [order] OpCodeName (OpCode Enum)" << std::endl;
376 int32_t opcode_index = 0;
377 for (auto opcode : opcodes)
378 {
379 circle::BuiltinOperator op_code = opcode->builtin_code();
380 // cast to int32_t to print as number or int8_t will print as ascii code
381 int32_t dp_code = static_cast<int32_t>(opcode->deprecated_builtin_code());
382
383 auto op_name = mio::circle::opcode_name(opcode);
384 auto op_version = opcode->version();
385
386 os << "[" << opcode_index << "] " << op_name << " (code: " << op_code
387 << ", dep_code: " << dp_code << ", version: " << op_version << ")" << std::endl;
388
389 opcode_index++;
390 }
391 os << std::endl;
392
393 // dump buffer
394 os << "Buffers: B(index) (length) values, if any" << std::endl;
395 for (uint32_t i = 0; i < buffers->Length(); ++i)
396 {
397 const uint8_t *buff_data;
398 size_t size = reader.buffer_info(i, &buff_data);
399
400 os << "B(" << i << ") (" << size << ") ";
401 if (buff_data != nullptr)
402 {
403 dump_buffer(os, buff_data, size, 16);
404 }
405 os << std::endl;
406 }
407 os << std::endl;
408
409 // dump metadata
410 if (metadata != nullptr)
411 {
412 os << "metadata : B(index) name" << std::endl;
413 for (uint32_t i = 0; i < metadata->Length(); ++i)
414 {
415 const auto buff_id = metadata->Get(i)->buffer();
416 const auto metadata_name = metadata->Get(i)->name()->str();
417 os << "B(" << buff_id << ") " << metadata_name << std::endl;
418
419 const uint8_t *buff_data;
420 reader.buffer_info(buff_id, &buff_data);
421 if (auto meta_prn = MetadataPrinterRegistry::get().lookup(metadata_name))
422 {
423 meta_prn->print(buff_data, os);
424 }
425 }
426 os << std::endl;
427 }
428
429 // dump signaturedef
430 if (signaturedefs != nullptr)
431 {
432 os << "SignatureDef" << std::endl;
433 for (uint32_t i = 0; i < signaturedefs->Length(); ++i)
434 {
435 auto sign_i = signaturedefs->Get(i);
436 os << "S(" << i << ") signature_key(" << sign_i->signature_key()->c_str() << "), sub_graph("
437 << sign_i->subgraph_index() << ")" << std::endl;
438
439 auto inputs_i = sign_i->inputs();
440 for (uint32_t t = 0; t < inputs_i->Length(); ++t)
441 {
442 auto inputs_i_t = inputs_i->Get(t);
443 os << " I(" << t << ")"
444 << " T(" << sign_i->subgraph_index() << ":" << inputs_i_t->tensor_index() << ") "
445 << inputs_i_t->name()->c_str() << std::endl;
446 }
447
448 auto outputs_i = sign_i->outputs();
449 for (uint32_t t = 0; t < outputs_i->Length(); ++t)
450 {
451 auto outputs_i_t = outputs_i->Get(t);
452 os << " O(" << t << ")"
453 << " T(" << sign_i->subgraph_index() << ":" << outputs_i_t->tensor_index() << ") "
454 << outputs_i_t->name()->c_str() << std::endl;
455 }
456 }
457 os << std::endl;
458 }
459
460 for (uint32_t sg = 0; sg < num_subgraph; ++sg)
461 {
462 reader.select_subgraph(sg);
463
464 os << "-------------------------------------------------------------------" << std::endl;
465 os << "Sub-Graph: #" << sg << " " << reader.subgraph_name() << std::endl;
466 os << std::endl;
467
468 dump_sub_graph(os, reader);
469 }
470
471 os << "===================================================================" << std::endl;
472}
