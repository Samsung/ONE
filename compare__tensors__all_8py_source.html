107def compare_quantization(tensor, tensor_name, expect_dir):
108 test_result = True
109 with open(expect_dir + "/" + tensor_name + ".json", "r") as expect_file:
110 json_load = json.load(expect_file)
111 for key in json_load:
112 if key == "weights":
113 expected_weights = np.array(json_load["weights"])
114 input_weights = tensor["weights"][()]
115 abs_tolerance = 1
116 # We use higher tolerance for int64 data (bias of int16-quantized model)
117 if tensor["weights"].dtype == 'int64':
118 abs_tolerance = 5
119
120 if np.allclose(
121 input_weights, expected_weights, rtol=0, atol=abs_tolerance) == False:
122 logging.error(
123 "Quantized weights of " + tensor_name + " (" + str(input_weights) +
124 ") do not match with expected value (" + str(expected_weights) + ").")
125 test_result = False
126
127 if key == "scale":
128 expected_scale = np.array(json_load["scale"])
129 input_scale = tensor["scale"][:]
130 if np.allclose(input_scale, expected_scale, rtol=1.e-5, atol=1.e-5) == False:
131 logging.error("Quantized scale of " + tensor_name + " (" +
132 str(input_scale) + ") do not match with expected value (" +
133 str(expected_scale) + ").")
134 test_result = False
135
136 if key == "zero_point":
137 expected_zero_point = np.array(json_load["zero_point"])
138 input_zero_point = tensor["zero_point"][:]
139 if np.allclose(
140 input_zero_point, expected_zero_point, rtol=0, atol=1) == False:
141 logging.error("Quantized zero_point of " + tensor_name + " (" +
142 str(input_zero_point) +
143 ") do not match with expected value (" +
144 str(expected_zero_point) + ").")
145 test_result = False
146 return test_result
