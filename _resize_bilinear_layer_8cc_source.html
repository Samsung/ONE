31void KernelGenerator::visit(const ir::operation::ResizeBilinear &node)
32{
33 const auto output_index{node.getOutputs().at(0)};
34 const auto input_index{node.getInputs().at(ir::operation::ResizeBilinear::INPUT)};
35
36 auto align_corners = node.param().align_corners;
37 auto half_pixel_centers = node.param().half_pixel_centers;
38
39 auto output_tensor = _tensor_reg->getPortableTensor(output_index);
40 auto input_tensor = _tensor_reg->getPortableTensor(input_index);
41
42 auto fn = std::make_unique<ops::ResizeBilinearLayer>();
43
44 if (node.getInputs().size() == 1)
45 {
46 fn->configure(input_tensor, output_tensor, node.param().height_out, node.param().width_out,
47 align_corners, half_pixel_centers);
48 }
49 else
50 {
51 assert(node.getInputs().size() == 2);
52 const auto size_index{node.getInputs().at(ir::operation::ResizeBilinear::SIZE)};
53 auto size_tensor = _tensor_reg->getPortableTensor(size_index);
54 if (size_tensor->is_constant())
55 {
56 auto size_vec = _ctx.at(size_index).asVector<int32_t>();
57 const auto height_out = size_vec[0];
58 const auto width_out = size_vec[1];
59 fn->configure(input_tensor, output_tensor, height_out, width_out, align_corners,
60 half_pixel_centers);
61 }
62 else
63 {
64 fn->configure(input_tensor, output_tensor, size_tensor, align_corners, half_pixel_centers);
65 }
66 }
67
68 _return_fn = std::move(fn);
69}

118void ResizeBilinearLayer::run()
119{
120 nnfw::cker::ResizeBilinearParams params;
121 if (_size == nullptr)
122 {
123 params.output_height = _output_height;
124 params.output_width = _output_width;
125 }
126 else
127 {
128 const auto size_buf = getBuffer<int32_t>(_size);
129 params.output_height = size_buf[0];
130 params.output_width = size_buf[1];
131 }
132 params.align_corners = _align_corners;
133 params.half_pixel_centers = _half_pixel_centers;
134
135 switch (_input->data_type())
136 {
137 case OperandType::FLOAT32:
138 nnfw::cker::ResizeBilinear(params, getShape(_input), getBuffer<float>(_input),
139 getShape(_output), getBuffer<float>(_output));
140 break;
141
142 case OperandType::QUANT_UINT8_ASYMM:
143 nnfw::cker::ResizeBilinear(params, getShape(_input), getBuffer<uint8_t>(_input),
144 getShape(_output), getBuffer<uint8_t>(_output));
145 break;
146
147 case OperandType::QUANT_INT8_ASYMM:
148 nnfw::cker::ResizeBilinear(params, getShape(_input), getBuffer<int8_t>(_input),
149 getShape(_output), getBuffer<int8_t>(_output));
150 break;
151
152 case OperandType::UINT8:
153 case OperandType::BOOL8:
154 case OperandType::FLOAT16:
155 case OperandType::INT32:
156 case OperandType::INT64:
157 case OperandType::QUANT_INT8_SYMM:
158 std::runtime_error("ResizeBilinear NYI");
159 break;
160 default:
161 std::runtime_error("ResizeBilinear unsupported data type");
162 }
163}
