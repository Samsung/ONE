29template <typename T>
30void evalGeneric(const circle::Operator *cur_op, BaseRuntimeGraph *runtime_graph, bool)
31{
32 const auto output_index = cur_op->outputs()->operator[](0);
33
34 assert(output_index != -1);
35
36 auto output = runtime_graph->getCircleTensorByIndex(output_index);
37
38 const auto *options = cur_op->builtin_options_as_ConcatenationOptions();
39
40 int axis = options->axis();
41 if (axis < 0)
42 axis += Tensor::num_dims(output);
43
44 const auto input_sizes = cur_op->inputs()->size();
45
46 std::vector<const T *> all_input_data;
47 std::vector<tflite::RuntimeShape> all_shape;
48 std::vector<tflite::RuntimeShape *> all_shape_ptr;
49
50 for (int32_t i = 0; i < input_sizes; ++i)
51 {
52 auto input_index = cur_op->inputs()->operator[](i);
53 const auto *tensor = runtime_graph->getCircleTensorByIndex(input_index);
54
55 const auto *tensor_data = runtime_graph->getDataByTensor(tensor);
56 if (tensor_data == nullptr)
57 tensor_data = runtime_graph->getConstDataByTensor(tensor);
58
59 auto *data = reinterpret_cast<const T *>(tensor_data);
60
61#ifndef DIS_DYN_SHAPES
62 auto *dynamic_shape_vector = runtime_graph->getDynamicShapeTensor(tensor);
63 if (dynamic_shape_vector == nullptr)
64 {
65 all_input_data.push_back(data);
66 all_shape.push_back(kernels::getTensorShape(tensor));
67 }
68 else
69 {
70 if (data != nullptr)
71 {
72 tflite::RuntimeShape runtime_shape(dynamic_shape_vector->size());
73 for (int n = 0; n < dynamic_shape_vector->size(); ++n)
74 {
75 runtime_shape.SetDim(n, dynamic_shape_vector->at(n));
76 }
77 all_shape.push_back(runtime_shape);
78 all_input_data.push_back(data);
79 }
80 }
81#else // DIS_DYN_SHAPES
82 assert(tensor_data);
83 all_input_data.push_back(data);
84 all_shape.push_back(kernels::getTensorShape(tensor));
85#endif
86 }
87
88 for (tflite::RuntimeShape &shape : all_shape)
89 {
90 all_shape_ptr.push_back(&shape);
91 }
92
93 auto *output_data = reinterpret_cast<T *>(runtime_graph->getDataByTensor(output));
94
95 // kernels::VectorOfTensors<T, true> inputs(_inputs);
96 tflite::ConcatenationParams params{};
97 params.axis = axis;
98 params.inputs_count = all_shape.size();
99 tflite::reference_ops::Concatenation(params, all_shape_ptr.data(), all_input_data.data(),
100 kernels::getTensorShape(output), output_data);
101}

105void configure_kernel_CircleConcatenation(const circle::Operator *cur_op,
106 BaseRuntimeGraph *runtime_graph)
107{
108 const int num_inputs = cur_op->inputs()->size();
109 LUCI_INTERPRETER_CHECK(num_inputs > 0);
110
111 auto input_index = cur_op->inputs()->operator[](0);
112 auto output_index = cur_op->outputs()->operator[](0);
113
114 assert(input_index != -1);
115 assert(output_index != -1);
116
117 const auto *t0 = runtime_graph->getCircleTensorByIndex(input_index);
118 const auto *output = runtime_graph->getCircleTensorByIndex(output_index);
119
120 const auto *params = cur_op->builtin_options_as_ConcatenationOptions();
121
122 // TODO: Support concat with fused activation function
123 LUCI_INTERPRETER_CHECK(luci_actfunc(params->fused_activation_function()) == FusedActFunc::NONE);
124
125 int axis = params->axis();
126 if (axis < 0)
127 axis += Tensor::num_dims(t0);
128 LUCI_INTERPRETER_CHECK(axis >= 0 && axis < Tensor::num_dims(t0));
129
130 for (int i = 1; i < num_inputs; ++i)
131 {
132 input_index = cur_op->inputs()->operator[](i);
133 const auto *tensor = runtime_graph->getCircleTensorByIndex(input_index);
134 LUCI_INTERPRETER_CHECK(Tensor::element_type(tensor) == Tensor::element_type(t0));
135 LUCI_INTERPRETER_CHECK(Tensor::num_dims(tensor) == Tensor::num_dims(t0));
136 }
137
138#ifndef DIS_QUANT
139 // If input tensors are INT8 type then quantization parameters of all input tensors and the output
140 // should be the same
141 for (int i = 1; i < num_inputs; ++i)
142 {
143 input_index = cur_op->inputs()->operator[](i);
144 const auto *tensor = runtime_graph->getCircleTensorByIndex(input_index);
145 if (Tensor::element_type(tensor) == DataType::S8)
146 {
147 LUCI_INTERPRETER_CHECK(Tensor::quantized_dimension(tensor) ==
148 Tensor::quantized_dimension(output));
149
150 LUCI_INTERPRETER_CHECK(Tensor::zero_points(tensor).size() == Tensor::scales(tensor).size());
151 LUCI_INTERPRETER_CHECK(Tensor::zero_points(tensor) == Tensor::zero_points(output));
152 LUCI_INTERPRETER_CHECK(Tensor::scales(tensor) == Tensor::scales(output));
153 }
154 }
155#endif // DIS_QUANT
156}

158void execute_kernel_CircleConcatenation(const circle::Operator *cur_op,
159 BaseRuntimeGraph *runtime_graph, bool is_inplace)
160{
161 int num_inputs = cur_op->inputs()->size();
162 LUCI_INTERPRETER_CHECK(num_inputs > 0);
163
164 const auto input_index = cur_op->inputs()->operator[](0);
165 assert(input_index != -1);
166 const auto *t0 = runtime_graph->getCircleTensorByIndex(input_index);
167
168 switch (Tensor::element_type(t0))
169 {
170#ifndef DIS_FLOAT
171 case DataType::FLOAT32:
172 evalGeneric<float>(cur_op, runtime_graph, is_inplace);
173 break;
174#endif // DIS_FLOAT
175#ifndef DIS_QUANT
176 case DataType::S8:
177 evalGeneric<int8_t>(cur_op, runtime_graph, is_inplace);
178 break;
179#endif // DIS_QUANT
180 case DataType::S32:
181 evalGeneric<int32_t>(cur_op, runtime_graph, is_inplace);
182 break;
183 case DataType::S64:
184 evalGeneric<int64_t>(cur_op, runtime_graph, is_inplace);
185 break;
186 default:
187 assert(false && "Unsupported type.");
188 }
189}
