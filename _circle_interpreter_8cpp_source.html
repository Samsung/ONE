30void readDataFromFile(const std::string &filename, char *data, size_t data_size)
31{
32 std::ifstream fs(filename, std::ifstream::binary);
33 if (fs.fail())
34 throw std::runtime_error("Cannot open file \"" + filename + "\".\n");
35 if (fs.read(data, data_size).fail())
36 throw std::runtime_error("Input tensor size mismatches with \"" + filename + "\".\n");
37 if (fs.peek() != EOF)
38 throw std::runtime_error("Input tensor size mismatches with \"" + filename + "\".\n");
39}

41void writeDataToFile(const std::string &filename, const char *data, size_t data_size)
42{
43 std::ofstream fs(filename, std::ofstream::binary);
44 if (fs.fail())
45 throw std::runtime_error("Cannot open file \"" + filename + "\".\n");
46 if (fs.write(data, data_size).fail())
47 {
48 throw std::runtime_error("Failed to write data to file \"" + filename + "\".\n");
49 }
50}

52template <typename NodeT> size_t getTensorSize(const NodeT *node)
53{
54 uint32_t tensor_size = loco::size(node->dtype());
55 for (uint32_t i = 0; i < node->rank(); ++i)
56 tensor_size *= node->dim(i).value();
57 return tensor_size;
58}

74int entry(int argc, char **argv)
75{
76 arser::Arser arser("Interpreter driver for circle models");
77
78 arser::Helper::add_version(arser, print_version);
79
80 arser.add_argument("model_path").help("Circle model filepath");
81 arser.add_argument("input_prefix")
82 .help("Input data filepath for circle model. "
83 "n-th input data is read from ${input_prefix}n, "
84 "for example, Add.circle.input0, Add.circle.input1");
85 arser.add_argument("output_prefix")
86 .help("Output data filepath for circle model. "
87 "Output data is written in ${output_file}n, "
88 "for example, Add.circle.output0");
89
90 try
91 {
92 arser.parse(argc, argv);
93 }
94 catch (const std::runtime_error &err)
95 {
96 std::cout << err.what() << std::endl;
97 std::cout << arser;
98 return EXIT_FAILURE;
99 }
100
101 const auto filename = arser.get<std::string>("model_path");
102 const auto input_prefix = arser.get<std::string>("input_prefix");
103 const auto output_prefix = arser.get<std::string>("output_prefix");
104
105 // Load model from the file
106 luci::ImporterEx importer;
107 std::unique_ptr<luci::Module> module = importer.importVerifyModule(filename);
108 if (module == nullptr)
109 {
110 std::cerr << "ERROR: Failed to load '" << filename << "'" << std::endl;
111 return EXIT_FAILURE;
112 }
113
114 // Create interpreter.
115 luci_interpreter::Interpreter interpreter(module.get());
116
117 // Set input.
118 // Data for n'th input is read from ${input_prefix}n
119 // (ex: Add.circle.input0, Add.circle.input1 ..)
120 const auto input_nodes = loco::input_nodes(module->graph());
121 for (int32_t i = 0; i < input_nodes.size(); i++)
122 {
123 const auto *input_node = loco::must_cast<const luci::CircleInput *>(input_nodes[i]);
124 std::vector<char> input_data(getTensorSize(input_node));
125 readDataFromFile(std::string(input_prefix) + std::to_string(i), input_data.data(),
126 input_data.size());
127 interpreter.writeInputTensor(input_node, input_data.data(), input_data.size());
128 }
129
130 // Do inference.
131 interpreter.interpret();
132
133 // Get output.
134 const auto output_nodes = loco::output_nodes(module->graph());
135 for (int i = 0; i < module->graph()->outputs()->size(); i++)
136 {
137 const auto *output_node = loco::must_cast<const luci::CircleOutput *>(output_nodes[i]);
138 std::vector<char> output_data(getTensorSize(output_node));
139 interpreter.readOutputTensor(output_node, output_data.data(), output_data.size());
140
141 // Output data is written in ${output_file}n
142 // (ex: Add.circle.output0)
143 writeDataToFile(std::string(output_prefix) + std::to_string(i), output_data.data(),
144 output_data.size());
145 }
146 return EXIT_SUCCESS;
147}
