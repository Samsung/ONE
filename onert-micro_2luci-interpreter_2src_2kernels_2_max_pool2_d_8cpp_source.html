35void MaxPool2D::configure()
36{
37 LUCI_INTERPRETER_CHECK(input()->element_type() == output()->element_type());
38 assert(input()->shape().num_dims() == 4);
39 const Shape &input_shape = input()->shape();
40 const int32_t batches = input_shape.dim(0);
41 const int32_t input_height = input_shape.dim(1);
42 const int32_t input_width = input_shape.dim(2);
43 const int32_t depth = input_shape.dim(3);
44
45 const int32_t output_height =
46 computeOutputSize(_params.padding, input_height, _params.filter_height, _params.stride_height);
47 const int32_t output_width =
48 computeOutputSize(_params.padding, input_width, _params.filter_width, _params.stride_width);
49
50 _padding_height =
51 computePadding(_params.stride_height, 1, input_height, _params.filter_height, output_height);
52 _padding_width =
53 computePadding(_params.stride_width, 1, input_width, _params.filter_width, output_width);
54
55 // TODO: enable it only if kernel with dynamic shapes
56 output()->resize({batches, output_height, output_width, depth});
57 if (input()->element_type() == DataType::U8)
58 {
59 LUCI_INTERPRETER_CHECK(std::abs(output()->scale() - input()->scale()) <= 1.0e-6);
60 LUCI_INTERPRETER_CHECK(output()->zero_point() == input()->zero_point());
61 }
62 else if (input()->element_type() == DataType::S16)
63 {
64 LUCI_INTERPRETER_CHECK(std::abs(output()->scale() - input()->scale()) <= 1.0e-6);
65 LUCI_INTERPRETER_CHECK(input()->zero_point() == 0 && output()->zero_point() == 0);
66 }
67}

69void MaxPool2D::execute() const
70{
71 switch (input()->element_type())
72 {
73 case DataType::FLOAT32:
74 evalFloat();
75 break;
76 case DataType::U8:
77 evalQuantized();
78 break;
79 case DataType::S16:
80 evalSInt16();
81 break;
82 default:
83 assert(false && "Unsupported type.");
84 }
85}

87void MaxPool2D::evalFloat() const
88{

90 float activation_max{};
91 calculateActivationRange(_params.activation, &activation_min, &activation_max);
92
93 tflite::PoolParams params{};
94 params.padding_values.height = _padding_height;
95 params.padding_values.width = _padding_width;
96 params.stride_height = _params.stride_height;
97 params.stride_width = _params.stride_width;
98 params.filter_height = _params.filter_height;
99 params.filter_width = _params.filter_width;
100 params.float_activation_min = activation_min;
101 params.float_activation_max = activation_max;
102
103 tflite::reference_ops::MaxPool(params, getTensorShape(input()), getTensorData<float>(input()),
104 getTensorShape(output()), getTensorData<float>(output()));
