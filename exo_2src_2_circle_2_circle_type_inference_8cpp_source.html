40circle::TensorType translateLocoTypeToCircle(loco::DataType dtype)
41{
42 switch (dtype)
43 {
44 case loco::DataType::U8:
45 return circle::TensorType_UINT8;
46 // case loco::DataType::U16: unsupported
47 // case loco::DataType::U32: unsupported
48 // case loco::DataType::U64: unsupported
49 case loco::DataType::S8:
50 return circle::TensorType_INT8;
51 case loco::DataType::S16:
52 return circle::TensorType_INT16;
53 case loco::DataType::S32:
54 return circle::TensorType_INT32;
55 case loco::DataType::S64:
56 return circle::TensorType_INT64;
57 case loco::DataType::FLOAT16:
58 return circle::TensorType_FLOAT16;
59 case loco::DataType::FLOAT32:
60 return circle::TensorType_FLOAT32;
61 // case loco::DataType::FLOAT64: unsupported
62 default:
63 break;
64 }
65
66 INTERNAL_EXN_V("Invalid loco dtype", oops::to_uint32(dtype));
67}
