122// test options
123TEST(SUPPORT_NNC, verify_cl_options)
124{
125 // create command line
126 const char *argv[] = {
127 "CLTest", // program name
128 // string options
129 "-m", "multiopt_value", // second name for option with several names
130 "--single", "single_value", // option with single name
131 "-several_separators:SOME_VALUE1,SOME_VALUE2", // test option with several separators
132 "--one_separarot=AAA_VALUE", // test option whit one separator
133 "-default_val_opt", // test option with default value
134 "--optional_opt", "/home/guest/tmp", // test optional option
135 "-valid_opt", "value2", // test options with defined values
136 // integer options
137 "-neg_val", "-42", // test negative value for integer option
138 "-default_neg_val", // test integer option with default value
139 "-pos_val", "33", // test positive value for integer option
140 // char options
141 "-char-opt", "b", "-dash_opt", "-",
142 // bool options
143 "-bool_opt=false", "-bool-opt2",
144 // vector of strings options
145 "-vec_opt1", "1", "c", "222", "ABC", "857", "-vec_opt2", "--vec_opt_with_vals", "abc", "123",
146 "xxx", "abc", "xxx",
147 // grouped options
148 "-group_opt1", "-group_opt2", "abc", "-group_opt3", "11", nullptr};
149 int argc = (sizeof(argv) / sizeof(argv[0])) - 1;
150
151 // It must be failed if option is not passed and other options are in the same group
152 argv[argc - 5] = "-m"; // disable -group_opt1
153 ASSERT_DEATH(CommandLine::getParser()->parseCommandLine(argc, argv), "");
154 argv[argc - 5] = "-group_opt1"; // enable -group_opt1
155
156 // test when mandatory grouped option is not passed. It must be OK if options being from the same
157 // group are missed
158 argv[argc - 4] = "-m"; // disable -group_opt2
159 argv[argc - 2] = "-m"; // disable -group_opt3
160 CommandLine::getParser()->parseCommandLine(argc, argv);
161 argv[argc - 4] = "-group_opt2"; // enable -group_opt2
162 argv[argc - 2] = "-group_opt3"; // enable -group_opt3
163
164 // parse command line
165 CommandLine::getParser()->parseCommandLine(argc, argv);
166
167 // here we put value from options
168 std::string tmp_string = SMultiOpt;
169 int32_t tmp_sint = NNegOpt;
170 char tmp_char = CharOpt;
171 bool tmp_bool = BoolOpt;
172 std::vector<std::string> tmp_vec = VecStrOpt1;
173
174 //
175 // string options
176 //
177 // check option with several names
178 ASSERT_EQ(SMultiOpt, "multiopt_value");
179 ASSERT_EQ(tmp_string, "multiopt_value");
180 ASSERT_EQ(tmp_string, SMultiOpt);
181
182 // check option with single name
183 tmp_string = SSingleOpt;
184 ASSERT_EQ(SSingleOpt, "single_value");
185 ASSERT_EQ(tmp_string, "single_value");
186 ASSERT_EQ(SSingleOpt, tmp_string);
187
188 // check option with separators
189 ASSERT_EQ(SSeveralSepOpt, "SOME_VALUE1,SOME_VALUE2");
190
191 // check option with one separator
192 ASSERT_EQ(SOneSepOpt, "AAA_VALUE");
193
194 // check option with default value
195 ASSERT_EQ(SDefaultOpt, "DEFAULT_VALUE");
196
197 // check optional option
198 ASSERT_EQ(SOptionalOpt, "/home/guest/tmp");
199
200 //
201 // integer options
202 //
203 // check option with valid values
204 ASSERT_EQ(SValidValsOpt, "value2");
205
206 // check option with negative value
207 ASSERT_EQ(NNegOpt, -42);
208 ASSERT_EQ(tmp_sint, -42);
209 ASSERT_EQ(tmp_sint, NNegOpt);
210
211 // check integer option with default value
212 tmp_sint = NDefaultNegOpt;
213 ASSERT_EQ(NDefaultNegOpt, -33);
214 ASSERT_EQ(tmp_sint, -33);
215 ASSERT_EQ(NDefaultNegOpt, tmp_sint);
216
217 // check integer option with positive value
218 ASSERT_EQ(NPosOpt, 33u);
219
220 //
221 // char options
222 //
223 ASSERT_EQ(CharOpt, 'b');
224 ASSERT_EQ(tmp_char, 'b');
225 ASSERT_EQ(tmp_char, CharOpt);
226
227 tmp_char = DashOpt;
228 ASSERT_EQ(DashOpt, '-');
229 ASSERT_EQ(tmp_char, '-');
230 ASSERT_EQ(DashOpt, tmp_char);
231
232 //
233 // bool options
234 //
235 ASSERT_EQ(BoolOpt, false);
236 ASSERT_EQ(tmp_bool, false);
237 ASSERT_EQ(tmp_bool, BoolOpt);
238
239 tmp_bool = BoolOpt2;
240 ASSERT_EQ(BoolOpt2, true);
241 ASSERT_EQ(tmp_bool, true);
242 ASSERT_EQ(BoolOpt2, tmp_bool);
243
244 //
245 // vector of strings options
246 //
247 ASSERT_EQ(tmp_vec, VecStrOpt1);
248 ASSERT_EQ(VecStrOpt1[0], "1");
249 ASSERT_EQ(tmp_vec[1], "c");
250 ASSERT_EQ(VecStrOpt1[2], "222");
251 ASSERT_EQ(tmp_vec[3], "ABC");
252 ASSERT_EQ(VecStrOpt1[4], "857");
253
254 tmp_vec = VecStrOpt2;
255 ASSERT_EQ(VecStrOpt2, tmp_vec);
256 ASSERT_TRUE(VecStrOpt2.empty());
257
258 ASSERT_EQ(VecStrOptWithVals[0], "abc");
259 ASSERT_EQ(VecStrOptWithVals[1], "123");
260 ASSERT_EQ(VecStrOptWithVals[2], "xxx");
261 ASSERT_EQ(VecStrOptWithVals[3], "abc");
262 ASSERT_EQ(VecStrOptWithVals[4], "xxx");
263
264 //
265 // grouped options
266 //
267 ASSERT_TRUE(GroupOpt1.isGrouped() && GroupOpt2.isGrouped() && GroupOpt3.isGrouped());
268 ASSERT_EQ(GroupOpt1.getGroup(), GroupOpt2.getGroup());
269 ASSERT_NE(GroupOpt2.getGroup(), GroupOpt3.getGroup());
270 ASSERT_EQ(GroupOpt3.getGroupName(), "onnx");
271}
