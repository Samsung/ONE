45std::vector<std::string> split_into_vector(const std::string &str, const char &delim)
46{
47 std::vector<std::string> ret;
48 std::istringstream is(str);
49 for (std::string item; std::getline(is, item, delim);)
50 {
51 ret.push_back(item);
52 }
53
54 // Remove empty string
55 ret.erase(std::remove_if(ret.begin(), ret.end(), [](const std::string &s) { return s.empty(); }),
56 ret.end());
57
58 return ret;
59}

79std::unique_ptr<loco::Graph> make_graph(const std::vector<const luci::CircleNode *> nodes)
80{
81 auto graph = loco::make_graph();
82
83 luci::CloneContext ctx;
84 // clone nodes
85 for (const auto &n : nodes)
86 {
87 auto clone = luci::clone_node(n, graph.get());
88 ctx.emplace(n, clone);
89 }
90 // set graph input
91 for (const auto &n : nodes)
92 {
93 for (uint32_t i = 0; i < n->arity(); i++)
94 {
95 auto arg = n->arg(i);
96 auto input_node = dynamic_cast<luci::CircleNode *>(arg);
97 auto ctx_it = ctx.find(input_node);
98 // check if the node already has been cloned
99 if (ctx_it != ctx.end())
100 continue;
101 // the node isn't graph input if it is an other node's input
102 if (std::find(nodes.begin(), nodes.end(), arg) != nodes.end())
103 continue;
104 auto circle_const = dynamic_cast<luci::CircleConst *>(arg);
105 if (circle_const != nullptr)
106 {
107 auto clone = luci::clone_node(circle_const, graph.get());
108 ctx.emplace(circle_const, clone);
109 }
110 else
111 {
112 // circle input
113 auto circle_input = graph->nodes()->create<luci::CircleInput>();
114 input_node = dynamic_cast<luci::CircleNode *>(arg);
115 luci::copy_common_attributes(input_node, circle_input);
116 ctx.emplace(input_node, circle_input);
117 // graph input
118 auto graph_input = graph->inputs()->create();
119 graph_input->name(circle_input->name());
120 graph_input->dtype(circle_input->dtype());
121 // graph input shape
122 auto input_shape = std::make_unique<loco::TensorShape>();
123 input_shape->rank(circle_input->rank());
124 for (uint32_t i = 0; i < circle_input->rank(); i++)
125 {
126 if (circle_input->dim(i).known())
127 {
128 circle_input->dim(i).set(circle_input->dim(i).value());
129 }
130 }
131 graph_input->shape(std::move(input_shape));
132
133 circle_input->index(graph_input->index());
134 }
135 }
136 }
137 // set graph output
138 for (const auto &n : nodes)
139 {
140 auto outputs = loco::succs(n);
141 for (const auto &o : outputs)
142 {
143 // the node isn't graph output if it is an other node's output
144 if (std::find(nodes.begin(), nodes.end(), o) != nodes.end())
145 continue;
146 // circle output
147 auto circle_output = graph->nodes()->create<luci::CircleOutput>();
148 auto output_node = dynamic_cast<luci::CircleNode *>(o);
149 luci::copy_common_attributes(output_node, circle_output);
150 // connect to cloned output node
151 circle_output->from(ctx.find(n)->second);
152 // graph output
153 auto graph_output = graph->outputs()->create();
154 graph_output->name(output_node->name());
155 graph_output->dtype(output_node->dtype());
156 // graph output shape
157 auto output_shape = std::make_unique<loco::TensorShape>();
158 output_shape->rank(circle_output->rank());
159 for (uint32_t i = 0; i < circle_output->rank(); i++)
160 {
161 if (circle_output->dim(i).known())
162 {
163 circle_output->dim(i).set(circle_output->dim(i).value());
164 }
165 }
166 graph_output->shape(std::move(output_shape));
167
168 circle_output->index(graph_output->index());
169 }
170 }
171 // connect nodes
172 for (const auto &n : nodes)
173 {
174 luci::clone_connect(n, ctx);
175 }
176
177 return graph;
178}

193template <>
194std::vector<const luci::CircleNode *>
195OpSelector::select_by<SelectType::ID>(const std::vector<std::string> &comma_tokens)
196{
197 std::vector<uint32_t> by_id;
198
199 for (const auto &comma_token : comma_tokens)
200 {
201 auto dash_tokens = ::split_into_vector(comma_token, '-');
202 if (not::is_number(dash_tokens))
203 {
204 throw std::runtime_error{
205 "ERROR: To select operator by id, please use these args: [0-9], '-', ','"};
206 }
207
208 // Convert string into integer
209 std::vector<uint32_t> int_tokens;
210 try
211 {
212 std::transform(dash_tokens.begin(), dash_tokens.end(), std::back_inserter(int_tokens),
213 [](const std::string &str) { return static_cast<uint32_t>(std::stoi(str)); });
214 }
215 catch (const std::out_of_range &)
216 {
217 // Uf input is big integer like '123467891234', stoi throws this exception.
218 throw std::runtime_error{"ERROR: Argument is out of range."};
219 }
220 catch (...)
221 {
222 throw std::runtime_error{"ERROR: Unknown error"};
223 }
224
225 switch (int_tokens.size())
226 {
227 case 0: // inputs like "-"
228 {
229 throw std::runtime_error{"ERROR: Nothing was entered"};
230 }
231 case 1: // inputs like "1", "2"
232 {
233 by_id.push_back(int_tokens.at(0));
234 break;
235 }
236 case 2: // inputs like "1-2", "11-50"
237 {
238 for (uint32_t i = int_tokens.at(0); i <= int_tokens.at(1); i++)
239 {
240 by_id.push_back(i);
241 }
242 break;
243 }
244 default: // inputs like "1-2-3"
245 {
246 throw std::runtime_error{"ERROR: Too many '-' in str."};
247 }
248 }
249 }
250
251 loco::Graph *graph = _module->graph(0);
252 std::vector<const luci::CircleNode *> selected_nodes;
253
254 for (auto node : loco::all_nodes(graph))
255 {
256 auto cnode = loco::must_cast<const luci::CircleNode *>(node);
257
258 try
259 {
260 auto node_id = luci::get_node_id(cnode);
261 for (auto selected_id : by_id)
262 {
263 if (selected_id == node_id)
264 {
265 selected_nodes.emplace_back(cnode);
266 }
267 }
268 }
269 catch (const std::runtime_error &)
270 {
271 continue;
272 }
273 }
274
275 return selected_nodes;
276}

278template <>
279std::vector<const luci::CircleNode *>
280OpSelector::select_by<SelectType::NAME>(const std::vector<std::string> &tokens)
281{
282 loco::Graph *graph = _module->graph(0);
283 std::vector<const luci::CircleNode *> selected_nodes;
284
285 for (auto node : loco::all_nodes(graph))
286 {
287 auto cnode = loco::must_cast<const luci::CircleNode *>(node);
288 std::string node_name = cnode->name();
289
290 for (auto selected_name : tokens)
291 if (selected_name.compare(node_name) == 0) // find the selected name
292 selected_nodes.emplace_back(cnode);
293 }
294
295 return selected_nodes;
296}

298template <SelectType SELECT_TYPE>
299std::unique_ptr<luci::Module> OpSelector::select_by(const std::string &str)
300{
301 auto colon_tokens = ::split_into_vector(str, ',');
302 if (colon_tokens.empty())
303 {
304 throw std::runtime_error{"ERROR: Nothing was entered."};
305 }
306
307 assert(_module->size() == 1);
308
309 auto selected_nodes = select_by<SELECT_TYPE>(colon_tokens);
310
311 auto new_module = std::make_unique<luci::Module>();
312 new_module->add(::make_graph(selected_nodes));
313
314 return new_module;
315}
