45std::vector<std::string> split_into_vector(const std::string &str, const char &delim)
46{
47 std::vector<std::string> ret;
48 std::istringstream is(str);
49 for (std::string item; std::getline(is, item, delim);)
50 {
51 ret.push_back(item);
52 }
53
54 // Remove empty string
55 ret.erase(std::remove_if(ret.begin(), ret.end(), [](const std::string &s) { return s.empty(); }),
56 ret.end());
57
58 return ret;
59}

61bool is_number(const std::string &s)
62{
63 return !s.empty() && std::find_if(s.begin(), s.end(),
64 [](unsigned char c) { return !std::isdigit(c); }) == s.end();

106std::unique_ptr<loco::Graph> make_graph(const std::vector<const luci::CircleNode *> nodes)
107{
108 auto graph = loco::make_graph();
109
110 luci::CloneContext ctx;
111 // clone nodes
112 for (const auto &n : nodes)
113 {
114 auto clone = luci::clone_node(n, graph.get());
115 ctx.emplace(n, clone);
116 }
117 // set graph input
118 for (const auto &n : nodes)
119 {
120 for (uint32_t i = 0; i < n->arity(); i++)
121 {
122 auto arg = n->arg(i);
123 auto input_node = dynamic_cast<luci::CircleNode *>(arg);
124 auto ctx_it = ctx.find(input_node);
125 // check if the node already has been cloned
126 if (ctx_it != ctx.end())
127 continue;
128 // the node isn't graph input if it is an other node's input
129 if (std::find(nodes.begin(), nodes.end(), arg) != nodes.end())
130 continue;
131 // do not set graph input if the node is CircleOutputExclude.
132 auto circle_output_exclude = dynamic_cast<luci::CircleOutputExclude *>(arg);
133 if (circle_output_exclude)
134 {
135 auto clone = luci::clone_node(circle_output_exclude, graph.get());
136 ctx.emplace(circle_output_exclude, clone);
137 continue;
138 }
139 auto circle_const = dynamic_cast<luci::CircleConst *>(arg);
140 if (circle_const != nullptr)
141 {
142 auto clone = luci::clone_node(circle_const, graph.get());
143 ctx.emplace(circle_const, clone);
144 }
145 else
146 {
147 // circle input
148 auto circle_input = graph->nodes()->create<luci::CircleInput>();
149 input_node = dynamic_cast<luci::CircleNode *>(arg);
150 if (not input_node)
151 {
152 throw std::runtime_error{"ERROR: Invalid graph"};
153 }
154 luci::copy_common_attributes(input_node, circle_input);
155 ctx.emplace(input_node, circle_input);
156 // graph input
157 auto graph_input = graph->inputs()->create();
158 graph_input->name(circle_input->name());
159 graph_input->dtype(circle_input->dtype());
160 // graph input shape
161 auto input_shape = std::make_unique<loco::TensorShape>();
162 input_shape->rank(circle_input->rank());
163 for (uint32_t i = 0; i < circle_input->rank(); i++)
164 {
165 if (circle_input->dim(i).known())
166 {
167 circle_input->dim(i).set(circle_input->dim(i).value());
168 }
169 }
170 graph_input->shape(std::move(input_shape));
171
172 circle_input->index(graph_input->index());
173 }
174 }
175 }
176
177 const auto original_graph = nodes.at(0)->graph();
178 const auto original_outputs = loco::output_nodes(const_cast<loco::Graph *>(original_graph));
179
180 // set graph output
181 for (auto &n : nodes)
182 {
183 auto outputs = loco::succs(n);
184 bool beingUsed = false;
185 for (const auto &o : outputs)
186 {
187 if (std::find(nodes.begin(), nodes.end(), o) != nodes.end())
188 {
189 beingUsed = true;
190 break;
191 }
192 }
193
194 bool originalOutput = false;
195 for (const auto &o : outputs)
196 {
197 if (std::find(original_outputs.begin(), original_outputs.end(), o) != original_outputs.end())
198 {
199 originalOutput = true;
200 break;
201 }
202 }
203
204 // the node isn't graph output if it is an other node's output
205 if (beingUsed and not originalOutput)
206 continue;
207
208 IsMultiOutputNode multiout_visitor;
209 bool isMultiOut = n->accept(&multiout_visitor);
210 for (auto &o : outputs)
211 {
212 const luci::CircleNode *output_node = nullptr;
213 if (isMultiOut)
214 {
215 output_node = dynamic_cast<const luci::CircleNode *>(o);
216 if (not output_node)
217 {
218 throw std::runtime_error{"ERROR: Invalid graph"};
219 }
220 }
221 else
222 {
223 output_node = n;
224 }
225 // circle output
226 auto circle_output = graph->nodes()->create<luci::CircleOutput>();
227 luci::copy_common_attributes(output_node, circle_output);
228 // connect to cloned output node
229 circle_output->from(ctx.find(output_node)->second);
230 // graph output
231 auto graph_output = graph->outputs()->create();
232 graph_output->name(output_node->name());
233 graph_output->dtype(output_node->dtype());
234 // graph output shape
235 auto output_shape = std::make_unique<loco::TensorShape>();
236 output_shape->rank(circle_output->rank());
237 for (uint32_t i = 0; i < output_shape->rank(); i++)
238 {
239 if (circle_output->dim(i).known())
240 {
241 output_shape->dim(i).set(circle_output->dim(i).value());
242 }
243 }
244 graph_output->shape(std::move(output_shape));
245
246 circle_output->index(graph_output->index());
247 if (not isMultiOut)
248 break;
249 }
250 }
251 // connect nodes
252 for (const auto &n : nodes)
253 {
254 luci::clone_connect(n, ctx);
255 }
256
257 return graph;
258}

270template <>
271std::vector<const luci::CircleNode *>
272OpSelector::select_by<SelectType::ID>(const std::vector<std::string> &comma_tokens)
273{
274 std::vector<uint32_t> by_id;
275
276 for (const auto &comma_token : comma_tokens)
277 {
278 auto dash_tokens = ::split_into_vector(comma_token, '-');
279 if (not ::is_number(dash_tokens))
280 {
281 throw std::runtime_error{
282 "ERROR: To select operator by id, please use these args: [0-9], '-', ','"};
283 }
284
285 // Convert string into integer
286 std::vector<uint32_t> int_tokens;
287 try
288 {
289 std::transform(dash_tokens.begin(), dash_tokens.end(), std::back_inserter(int_tokens),
290 [](const std::string &str) { return static_cast<uint32_t>(std::stoi(str)); });
291 }
292 catch (const std::out_of_range &)
293 {
294 // Uf input is big integer like '123467891234', stoi throws this exception.
295 throw std::runtime_error{"ERROR: Argument is out of range."};
296 }
297 catch (...)
298 {
299 throw std::runtime_error{"ERROR: Unknown error"};
300 }
301
302 switch (int_tokens.size())
303 {
304 case 0: // inputs like "-"
305 {
306 throw std::runtime_error{"ERROR: Nothing was entered"};
307 }
308 case 1: // inputs like "1", "2"
309 {
310 by_id.push_back(int_tokens.at(0));
311 break;
312 }
313 case 2: // inputs like "1-2", "11-50"
314 {
315 for (uint32_t i = int_tokens.at(0); i <= int_tokens.at(1); i++)
316 {
317 by_id.push_back(i);
318 }
319 break;
320 }
321 default: // inputs like "1-2-3"
322 {
323 throw std::runtime_error{"ERROR: Too many '-' in str."};
324 }
325 }
326 }
327
328 loco::Graph *graph = _module->graph(0);
329 std::vector<const luci::CircleNode *> selected_nodes;
330
331 for (auto node : loco::all_nodes(graph))
332 {
333 auto cnode = loco::must_cast<const luci::CircleNode *>(node);
334
335 try
336 {
337 auto node_id = luci::get_node_id(cnode);
338 for (auto selected_id : by_id)
339 {
340 if (selected_id == node_id)
341 {
342 selected_nodes.emplace_back(cnode);
343 }
344 }
345 }
346 catch (const std::runtime_error &)
347 {
348 continue;
349 }
350 }
351
352 return selected_nodes;
353}

355template <>
356std::vector<const luci::CircleNode *>
357OpSelector::select_by<SelectType::NAME>(const std::vector<std::string> &tokens)
358{
359 loco::Graph *graph = _module->graph(0);
360 std::vector<const luci::CircleNode *> selected_nodes;
361
362 for (auto node : loco::all_nodes(graph))
363 {
364 auto cnode = loco::must_cast<const luci::CircleNode *>(node);
365 std::string node_name = cnode->name();
366
367 for (const auto &selected_name : tokens)
368 if (selected_name.compare(node_name) == 0) // find the selected name
369 selected_nodes.emplace_back(cnode);
370 }
371
372 return selected_nodes;
373}

375template <SelectType SELECT_TYPE>
376std::unique_ptr<luci::Module> OpSelector::select_by(const std::string &str)
377{
378 auto colon_tokens = ::split_into_vector(str, ',');
379 if (colon_tokens.empty())
380 {
381 throw std::runtime_error{"ERROR: Nothing was entered."};
382 }
383
384 auto selected_nodes = select_by<SELECT_TYPE>(colon_tokens);
385
386 // multiout node should be considered
387 IsMultiOutputNode multiout_visitor;
388 std::vector<const luci::CircleNode *> output_nodes;
389 for (const auto &node : selected_nodes)
390 {
391 if (node->accept(&multiout_visitor))
392 {
393 auto outputs = loco::succs(node);
394 for (auto &o : outputs)
395 {
396 output_nodes.push_back(dynamic_cast<luci::CircleNode *>(o));
397 }
398 }
399 }
400 selected_nodes.insert(selected_nodes.end(), output_nodes.begin(), output_nodes.end());
401
402 // TODO support two or more subgraphs
403 for (const auto &n : selected_nodes)
404 {
405 IsMultiGraphNode multigraph_visitor;
406 bool isMultiGraph = n->accept(&multigraph_visitor);
407 if (isMultiGraph)
408 {
409 throw std::runtime_error{"ERROR: If or While operator can't be selected."};
410 }
411 }
412
413 auto new_module = std::make_unique<luci::Module>();
414 new_module->add(::make_graph(selected_nodes));
415
416 return new_module;
417}
