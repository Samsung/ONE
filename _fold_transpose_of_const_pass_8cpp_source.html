37locoex::TFLTranspose *as_candidate(loco::Node *node)
38{
39 auto transpose = dynamic_cast<locoex::TFLTranspose *>(node);
40 if (not transpose)
41 return nullptr;
42
43 // Only accept Constant input of Transpose
44 if (not dynamic_cast<locoex::TFLConst *>(transpose->a()))
45 return nullptr;
46
47 // Only accept Constant permutation of Transpose
48 if (not dynamic_cast<locoex::TFLConst *>(transpose->perm()))
49 return nullptr;
50
51 return transpose;
52}

55{
56 nncc::core::ADT::tensor::Shape ret;

58 ret.resize(node->rank());
59 for (uint32_t axis = 0; axis < node->rank(); ++axis)
60 {

63

67void fold_transpose_of_const(locoex::TFLTranspose *transpose)
68{
69 const loco::DataType FLOAT32 = loco::DataType::FLOAT32;
70 const loco::DataType S32 = loco::DataType::S32;
71
72 auto const_orig = dynamic_cast<locoex::TFLConst *>(transpose->a());
73 auto perm = dynamic_cast<locoex::TFLConst *>(transpose->perm());
74
75 // Exceptions
76 {
77 EXO_ASSERT(const_orig, "Only support for Transpose-Const pair");
78 // TODO support other data types
79 if (const_orig->dtype() != FLOAT32)
80 INTERNAL_EXN_V("NYI for this data type", oops::to_uint32(const_orig->dtype()));
81
82 EXO_ASSERT(perm, "Only support for constant permutation for Transpose");
83 // TODO support other data types
84 if (perm->dtype() != S32)
85 INTERNAL_EXN_V("NYI for this data type", oops::to_uint32(perm->dtype()));
86
87 auto okay = [&]() {
88 if (perm->rank() != 1)
89 return false;
90 if (perm->dim(0).value() != const_orig->rank())
91 return false;
92 return true;
93 };
94 if (not okay())
95 INTERNAL_EXN("Input and permutation for Transpose is not congruent");
96 }
97
98 uint32_t rank = const_orig->rank();
99
100 // TFLConst to replace
101 auto const_new = transpose->graph()->nodes()->create<locoex::TFLConst>();
102
103 const_new->dtype(FLOAT32);
104 const_new->rank(rank);
105 const_new->size<FLOAT32>(const_orig->size<FLOAT32>());
106 for (uint32_t axis = 0; axis < rank; ++axis)
107 const_new->dim(axis) = const_orig->dim(perm->at<S32>(axis)).value();
108
109 // TODO remove dependency to angkor
110 auto shape_orig = angkor_shape(const_orig);
111 auto shape_new = angkor_shape(const_new);
112
113 nncc::core::ADT::tensor::LexicalLayout l;
114 nncc::core::ADT::tensor::IndexEnumerator e{shape_new};
115
116 for (; e.valid(); e.advance())
117 {
118 loco::TensorIndex index_new = e.current();
119 loco::TensorIndex index_orig;
120
121 // Set original index from matching new index
122 index_orig.resize(rank);
123 for (uint32_t axis = 0; axis < rank; ++axis)
124 index_orig.at(perm->at<S32>(axis)) = index_new.at(axis);
125
126 const_new->at<FLOAT32>(l.offset(shape_new, index_new)) =
127 const_orig->at<FLOAT32>(l.offset(shape_orig, index_orig));
128 }
129
130 // replace
131 loco::replace(transpose).with(const_new);
132}

139bool FoldTransposeOfConstPass::run(loco::Graph *g)
140{
141 bool changed = false;
142 for (auto node : loco::active_nodes(loco::output_nodes(g)))
143 {
144 if (auto transpose = as_candidate(node))
145 {
146 fold_transpose_of_const(transpose);
147 changed = true;
148 }
149 }
150
151 return changed;
152}
