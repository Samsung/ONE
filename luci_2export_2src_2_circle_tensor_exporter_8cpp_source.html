111struct NoOpDetector final : public luci::CircleNodeMutableVisitor<bool>
112{
113 // Input is Virtual but does produce a Tensor
114 // Output is Virtual that does not produce any Tensor
115 bool visit(luci::CircleOutput *) final { return true; }
116 bool visit(luci::CircleOutputExclude *) final { return true; }
117
118 // Return false by default
119 bool visit(luci::CircleNode *) final { return false; }

122void allocateCircleTensorInfo(CircleNode *node, CircleTensorContext &ctx)
123{
124 LOGGER(l);
125
126 auto tensor_index = static_cast<CircleTensorIndex>(ctx.size());
127 // TODO Use Graph-level metadata for Input & Output
128 std::string tensor_name = node->name();
129 // NOTE tensor_name maybe empty. this assertion will alert when this happens.
130 // currently we require tensor should have a name.
131 // TODO if this breaks, fix the cause or permit empty tensor_name.
132 assert(!tensor_name.empty());
133 if (ctx.exist(tensor_name))
134 {
135 // NOTE this should assign unique name for a Tensor.
136 tensor_name = tensor_name + "_" + std::to_string(tensor_index);
137 assert(!ctx.exist(tensor_name));
138 }
139 INFO(l) << "[luci] Tensor for " << tensor_name << ": " << tensor_index << std::endl;
140
141 CircleTensorInfo tensor_info;
142
143 tensor_info.name(tensor_name);
144 tensor_info.dtype(to_circle_tensortype(node->dtype()));
145 if (node->shape_status() == ShapeStatus::VALID)
146 tensor_info.shape(to_shape_description(node));
147 tensor_info.shape_status(node->shape_status());
148
149 tensor_info.content(dynamic_cast<luci::CircleConst *>(node));
150 tensor_info.quantparam(node->quantparam());
151 tensor_info.sparsityparam(node->sparsityparam());
152
153 tensor_info.is_variable(dynamic_cast<luci::CircleVariable *>(node) != nullptr);
154
155 set_tensor_index(node, tensor_index);
156
157 ctx.emplace_back(tensor_info);
158}

161{
162public:
163 MultiOutputDetector(CircleTensorContext &ctx) : _ctx(ctx) {}
164
165private:
166 void store_outputs(luci::CircleNode *node, uint32_t count)
167 {
168 auto outs = loco::succs(node);
169 assert(outs.size() == count);
170 (void)count; // for unused variable error in release build
171 for (auto out : outs)
172 {
173 auto circle_out = loco::must_cast<luci::CircleNode *>(out);
174 allocateCircleTensorInfo(circle_out, _ctx);
175 }
176 set_tensor_index(node, -1);
177 }

179public:

272void allocateCircleTensor(CircleNode *node, CircleTensorContext &ctx)
273{
274 if (node == nullptr)
275 throw std::runtime_error("allocateCIrcleTensor Failed : node is nullptr");
276
277 auto isNoOp = [](loco::Node *node) {
278 if (auto circle_node = dynamic_cast<luci::CircleNode *>(node))
279 {
280 NoOpDetector d;
281 return circle_node->accept(&d);
282 }
283 return false;
284 };
285
286 if (isNoOp(node))
287 {
288 set_tensor_index(node, -1);
289 return;
290 }
291
292 // TODO revise this when loco supports multiple outputs
293 // NOTE this will store all virtual output tensors and skip for the real node
294 if (auto circle_node = dynamic_cast<luci::CircleNode *>(node))
295 {
296 MultiOutputDetector d(ctx);
297 if (circle_node->accept(&d))
298 return;
299 }
300
301 allocateCircleTensorInfo(node, ctx);
302}

309flatbuffers::Offset<Vector<int32_t>> encodeShape(FlatBufferBuilder &builder,
310 const ShapeDescription &shape)
311{
312 assert(shape._rank_known && "unknown number of dimensions is not supported");
313
314 std::vector<int32_t> encoded_shape;
315 encoded_shape.resize(shape._dims.size());
316 for (uint32_t i = 0; i < shape._dims.size(); ++i)
317 encoded_shape.at(i) = shape._dims.at(i) == -1 ? 1 : shape._dims.at(i);
318
319 return builder.CreateVector(encoded_shape);
320}
321
322flatbuffers::Offset<Vector<int32_t>> encodeShapeSignature(FlatBufferBuilder &builder,
323 const ShapeDescription &shape)
324{
325 assert(shape._rank_known && "unknown number of dimensions is not supported");
326
327 // shape_signature is set if and only if at least one of dimensions are unknown.
328 for (uint32_t i = 0; i < shape._dims.size(); ++i)
329 if (shape._dims.at(i) == -1)
330 return builder.CreateVector(shape._dims);
331
332 return flatbuffers::Offset<Vector<int32_t>>();
333}

346template <loco::DataType DT>
347flatbuffers::Offset<circle::Buffer> encodeOpBufferByDType(FlatBufferBuilder &builder,
348 luci::CircleConst *c)
349{
350 using NativeType = typename loco::DataTypeImpl<DT>::Type;
351
352 std::vector<NativeType> raw_data;
353 const uint32_t size = c->size<DT>();
354 raw_data.reserve(size);
355 for (uint32_t i = 0; i < size; ++i)
356 {
357 raw_data.push_back(c->at<DT>(i));
358 }
359 const size_t raw_size = size * sizeof(NativeType);
360 auto array_offset = builder.CreateVector(reinterpret_cast<uint8_t *>(raw_data.data()), raw_size);
361 return CreateBuffer(builder, array_offset);
362}
363
364template <>
365flatbuffers::Offset<circle::Buffer>
366encodeOpBufferByDType<loco::DataType::STRING>(FlatBufferBuilder &builder, luci::CircleConst *c)
367{
368 const uint32_t count = c->size<loco::DataType::STRING>();
369 uint32_t raw_size = sizeof(int32_t) * (count + 2);
370 for (uint32_t i = 0; i < count; ++i)
371 {
372 auto &value = c->at<loco::DataType::STRING>(i);
373 raw_size += value.length();
374 }
375
376 // serialize string data
377 // int32_t count
378 // int32_t offsets[count + 1]
379 // string values[count]
380 std::vector<uint8_t> raw_data;
381 raw_data.reserve(raw_size);
382
383 auto *i32d = reinterpret_cast<int32_t *>(raw_data.data());
384 int32_t start = sizeof(int32_t) * (count + 2);
385 int32_t offset = start;
386 std::vector<int32_t> offsets;
387
388 *i32d++ = count;
389 *i32d++ = start;
390 offsets.push_back(start);
391 for (uint32_t i = 0; i < count; ++i)
392 {
393 auto &value = c->at<loco::DataType::STRING>(i);
394 offset += value.length();
395 *i32d++ = offset;
396 offsets.push_back(offset);
397 }
398
399 auto *data = reinterpret_cast<uint8_t *>(i32d);
400 for (uint32_t i = 0; i < count; ++i)
401 {
402 int32_t length = offsets[i + 1] - offsets[i];
403 auto &value = c->at<loco::DataType::STRING>(i);
404 memcpy(data, value.c_str(), length);
405 data += length;
406 }
407
408 auto array_offset = builder.CreateVector(reinterpret_cast<uint8_t *>(raw_data.data()), raw_size);
409 return CreateBuffer(builder, array_offset);
410}

412template <>
413flatbuffers::Offset<circle::Buffer> encodeOpBuffer(FlatBufferBuilder &builder, luci::CircleConst *c)
414{
415 switch (c->dtype())
416 {
417 case loco::DataType::FLOAT32:
418 return encodeOpBufferByDType<loco::DataType::FLOAT32>(builder, c);
419 case loco::DataType::S8:
420 return encodeOpBufferByDType<loco::DataType::S8>(builder, c);
421 case loco::DataType::S16:
422 return encodeOpBufferByDType<loco::DataType::S16>(builder, c);
423 case loco::DataType::S32:
424 return encodeOpBufferByDType<loco::DataType::S32>(builder, c);
425 case loco::DataType::S64:
426 return encodeOpBufferByDType<loco::DataType::S64>(builder, c);
427 case loco::DataType::U8:
428 return encodeOpBufferByDType<loco::DataType::U8>(builder, c);
429 case loco::DataType::BOOL:
430 return encodeOpBufferByDType<loco::DataType::BOOL>(builder, c);
431 case loco::DataType::STRING:
432 return encodeOpBufferByDType<loco::DataType::STRING>(builder, c);
433 default:
434 break;
435 }
436
437 // NOTE loco::DataType::FLOAT16 is added but we do not export this type
438 // as backends currently don't support this type.
439 // currently this is supported only for "Tensor(Float16) - Dequantize"
440 // sequence so that after 'fold_dequantize' option this Tensor is
441 // converted to FLOAT32.
442
443 INTERNAL_EXN_V("Unsupported datatype", oops::to_uint32(c->dtype()));
444}

446flatbuffers::Offset<circle::QuantizationParameters>
447encodeQuantizationParameters(FlatBufferBuilder &builder, luci::CircleQuantParam *quantparam)
448{
449 if (quantparam == nullptr)
450 return 0;
451
452 flatbuffers::Offset<flatbuffers::Vector<float>> min;
453 flatbuffers::Offset<flatbuffers::Vector<float>> max;
454 flatbuffers::Offset<flatbuffers::Vector<float>> scale;
455 flatbuffers::Offset<flatbuffers::Vector<int64_t>> zero_point;
456 if (quantparam->min.size() && quantparam->max.size())
457 {
458 min = builder.CreateVector(quantparam->min);
459 max = builder.CreateVector(quantparam->max);
460 }
461 if (quantparam->scale.size() && quantparam->zerop.size())
462 {
463 scale = builder.CreateVector(quantparam->scale);
464 zero_point = builder.CreateVector(quantparam->zerop);
465 }
466 // Note: QuantizationDetails is not supported
467 return circle::CreateQuantizationParameters(builder, min, max, scale, zero_point,
468 circle::QuantizationDetails::QuantizationDetails_NONE,
469 0, quantparam->quantized_dimension);
470}

472flatbuffers::Offset<circle::SparsityParameters>
473encodeSparsityParameters(FlatBufferBuilder &builder, luci::SparsityParam *sparsityparam)
474{
475 if (sparsityparam == nullptr)
476 return 0;
477
478 std::vector<flatbuffers::Offset<circle::DimensionMetadata>> dim_metadata_vec;
479 auto luci_dim_metadata = sparsityparam->dim_metadata;
480 for (auto it : luci_dim_metadata)
481 {
482 // array_segments
483 auto circle_array_segments = to_circle_sparse_index_vector(builder, it.array_segments());
484 auto circle_array_segments_type =
485 to_circle_sparse_index_vector_type(it.array_segments().type());
486
487 // array_indices
488 auto circle_array_indices = to_circle_sparse_index_vector(builder, it.array_indices());
489 auto circle_array_indices_type = to_circle_sparse_index_vector_type(it.array_indices().type());
490 auto dim_metadata = circle::CreateDimensionMetadata(
491 builder, to_circle_dimensiontype(it.format()), it.dense_size(), circle_array_segments_type,
492 circle_array_segments, circle_array_indices_type, circle_array_indices);
493 dim_metadata_vec.emplace_back(dim_metadata);
494 }
495
496 return circle::CreateSparsityParametersDirect(builder, &sparsityparam->traversal_order,
497 &sparsityparam->block_map, &dim_metadata_vec);
498}

500template <loco::DataType DT> bool has_same_elements(luci::CircleConst *lhs, luci::CircleConst *rhs)
501{
502 assert(lhs->dtype() == DT);
503 assert(rhs->dtype() == DT);
504 assert(lhs->size<DT>() == rhs->size<DT>());
505
506 for (uint32_t i = 0; i < lhs->size<DT>(); ++i)
507 if (lhs->at<DT>(i) != rhs->at<DT>(i))
508 return false;
509 return true;
510}
511
512bool has_same_values(luci::CircleConst *lhs, luci::CircleConst *rhs)
513{
514 if (lhs->dtype() != rhs->dtype())
515 return false;
516
517 if (lhs->rank() != rhs->rank())
518 return false;
519
520 for (uint32_t i = 0; i < lhs->rank(); ++i)
521 if (!(lhs->dim(i) == rhs->dim(i)))
522 return false;
523
524 switch (lhs->dtype())
525 {
526 case loco::DataType::FLOAT32:
527 return has_same_elements<loco::DataType::FLOAT32>(lhs, rhs);
528
529 case loco::DataType::S8:
530 return has_same_elements<loco::DataType::S8>(lhs, rhs);
531
532 case loco::DataType::S16:
533 return has_same_elements<loco::DataType::S16>(lhs, rhs);
534
535 case loco::DataType::S32:
536 return has_same_elements<loco::DataType::S32>(lhs, rhs);
537
538 case loco::DataType::S64:
539 return has_same_elements<loco::DataType::S64>(lhs, rhs);
540
541 case loco::DataType::U8:
542 return has_same_elements<loco::DataType::U8>(lhs, rhs);
543
544 case loco::DataType::BOOL:
545 return has_same_elements<loco::DataType::BOOL>(lhs, rhs);
546
547 default:
548 break;
549 }
550
551 return false;
552}

554uint32_t get_buffer_id(FlatBufferBuilder &builder, SerializedModelData &md, luci::CircleConst *node)
555{
556 if (node != nullptr)
557 {
558 // When buffer with same values is found, use the buffer id.
559 for (auto key_value : md._cached_buffer_id)
560 {
561 if (has_same_values(key_value.first, node))
562 return key_value.second;
563 }
564
565 // When buffer with same values is not found, generate new buffer
566 auto buffer = encodeOpBuffer(builder, node);
567
568 auto buffer_id = static_cast<uint32_t>(md._buffers.size());
569 md._buffers.push_back(buffer);
570
571 // Cache the newly generated buffer id
572 md._cached_buffer_id.insert({node, buffer_id});
573
574 return buffer_id;
575 }
576 else
577 {
578 // When there is no CircleConst, there is nothing to cache.
579 // So return new buffer id.
580 auto buffer = encodeOpBuffer(builder);
581
582 auto buffer_id = static_cast<uint32_t>(md._buffers.size());
583 md._buffers.push_back(buffer);
584
585 return buffer_id;
586 }

589void exportOpDefinedTensor(const CircleTensorInfo &info, FlatBufferBuilder &builder,
590 SerializedModelData &md, SerializedGraphData &gd)
591{
592 // Create and register output tensor shape
593 flatbuffers::Offset<Vector<int32_t>> shape_offset;
594 flatbuffers::Offset<Vector<int32_t>> shape_signature_offset;
595 if (info.shape_status() == ShapeStatus::VALID)
596 {
597 shape_offset = encodeShape(builder, info.shape());
598 shape_signature_offset = encodeShapeSignature(builder, info.shape());
599 }
600
601 auto quantparam = encodeQuantizationParameters(builder, info.quantparam());
602
603 auto sparsityparam = encodeSparsityParameters(builder, info.sparsityparam());
604
605 auto buffer_id = get_buffer_id(builder, md, info.content());
606
607 auto name_offset = builder.CreateString(info.name());
608
609 auto is_variable = info.is_variable();
610
611 auto tensor_offset = CreateTensor(builder, shape_offset, info.dtype(), buffer_id, name_offset,
612 quantparam, is_variable, sparsityparam, shape_signature_offset);
613 gd._tensors.push_back(tensor_offset);
614}

633void exportOpDefinedTensors(loco::Graph *g, FlatBufferBuilder &builder, SerializedModelData &md,
634 SerializedGraphData &gd)
635{
636 CircleTensorContext tensor_ctx;
637
638 // NOTE There may exist dangle CircleInput that is not visited with postorder_traversal()
639 // All dangle CircleOutput should be visited by postorder_traversal()
640 auto nodes = g->nodes();
641 for (uint32_t n = 0; n < nodes->size(); ++n)
642 {
643 auto node = dynamic_cast<luci::CircleInput *>(nodes->at(n));
644 if (node != nullptr)
645 allocateCircleTensor(node, tensor_ctx);
646 }
647
648 for (auto node : loco::postorder_traversal(loco::output_nodes(g)))
649 {
650 CircleNode *circle_node = loco::must_cast<luci::CircleNode *>(node);
651 if (dynamic_cast<const luci::CircleInput *>(circle_node) != nullptr)
652 continue;
653 allocateCircleTensor(circle_node, tensor_ctx);
654 }
655
656 for (const auto &tensor_info : tensor_ctx)
657 {
658 exportOpDefinedTensor(tensor_info, builder, md, gd);
659 }
660}
