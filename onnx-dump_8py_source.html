26def _data_type_str(data_type):
27 return TensorProto.DataType.Name(data_type)
28

30def _get_attribute_value(attr):
31 if attr.type == AttributeProto.TENSOR:
32 return "{}, {}".format(
33 _data_type_str(attr.t.data_type), numpy_helper.to_array(attr.t))
34 if attr.type == AttributeProto.GRAPH:
35 # TODO revise when graph node is available
36 return "<graph>"
37 if attr.type == AttributeProto.TENSORS:
38 # TODO revise to see contents
39 return "<tensors>..."
40 if attr.type == AttributeProto.GRAPHS:
41 # TODO revise when graph node is available
42 return "<graphs>..."
43 return helper.get_attribute_value(attr)

46def _dump_header(onnx_model):
47 print("[General] -----------------------------")
48 print("IR version =", onnx_model.ir_version)
49 print("Producer =", onnx_model.producer_name, onnx_model.producer_version)
50 print("")

53def _dump_operators(onnx_model):
54 opcodes_dict = dict()
55 for node in onnx_model.graph.node:
56 if node.op_type in opcodes_dict:
57 opcodes_dict[node.op_type] = opcodes_dict[node.op_type] + 1
58 else:
59 opcodes_dict[node.op_type] = 1
60
61 print("[Operators] ---------------------------")
62 for opcode_key in opcodes_dict:
63 print("{:>5} {}".format(opcodes_dict[opcode_key], opcode_key))
64
65 print("")
66

68def _dump_initializers(onnx_model):
69 print("[Initializers] ------------------------")
70 for initializer in onnx_model.graph.initializer:
71 init_name = '"{}"'.format(initializer.name)
72 dtstr = _data_type_str(initializer.data_type)
73 print('{:<15} {} {}'.format(init_name, dtstr, initializer.dims))
74
75 print("")

78def _dump_nodes(onnx_model):
79 print("[Nodes] -------------------------------")
80
81 for node in onnx_model.graph.node:
82 print('{0}("{1}")'.format(node.op_type, node.name))
83
84 attribute = ''
85 for attr in node.attribute:
86 if attribute != '':
87 attribute += ', '
88 attribute += "{}: {}".format(attr.name, _get_attribute_value(attr))
89
90 if attribute != '':
91 print(' A {0}'.format(attribute))
92
93 for inp in node.input:
94 print(' I "{0}"'.format(inp))
95 for out in node.output:
96 print(' O "{0}"'.format(out))
97
98 print("")
99
100
101def _dump_inputoutputs(onnx_model):
102 print("[Graph Input/Output]-------------------")
103 for mod_input in onnx_model.graph.input:
104 io_name = '"{}"'.format(mod_input.name)
105 dtstr = _data_type_str(mod_input.type.tensor_type.elem_type)
106 shape = mod_input.type.tensor_type.shape
107 input_shape = [dim.dim_value for dim in shape.dim]
108 print(' I: {:<15} {} {}'.format(io_name, dtstr, input_shape))
109
110 for mod_output in onnx_model.graph.output:
111 io_name = '"{}"'.format(mod_output.name)
112 dtstr = _data_type_str(mod_output.type.tensor_type.elem_type)
113 shape = mod_output.type.tensor_type.shape
114 output_shape = [dim.dim_value for dim in shape.dim]
115 print(' O: {:<15} {} {}'.format(io_name, dtstr, output_shape))
116
117 print("")

120def _dump_graph(onnx_model):
121 _dump_header(onnx_model)
122 _dump_operators(onnx_model)
123 _dump_initializers(onnx_model)
124 _dump_nodes(onnx_model)
125 _dump_inputoutputs(onnx_model)
