38bool has_one_wildcard_dim(const moco::TFConst *node)
39{
40 assert(node->dtype() == loco::DataType::S32);
41 assert(node->rank() == 1);
42
43 auto len = node->dim(0).value();
44 assert(len > 0);
45
46 // Must have one and only wildcard dimension(-1)
47 uint32_t count_wildcard_dim = 0;
48 for (uint32_t i = 0; i < len; ++i)
49 {
50 auto dim = node->at<loco::DataType::S32>(i);
51 if (dim == -1)
52 count_wildcard_dim++;
53 else
54 assert(dim >= 1);
55 }
56
57 assert(count_wildcard_dim <= 1 &&
58 "Invalid Reshape: there should be none or only one wildcard dimension");
59 return count_wildcard_dim;
60}

62uint32_t volume(const loco::TensorShape &shape)
63{
64 uint32_t ret = 1;
65 auto rank = shape.rank();
66 for (uint32_t axis = 0; axis < rank; ++axis)
67 {
68 ret *= shape.dim(axis).value();
69 }
70 return ret;

73void deduce_and_fix_wildcard_dim(moco::TFConst *node, const loco::NodeShape &tensor_input_shape)
74{
75 assert(has_one_wildcard_dim(node));
76
77 assert(tensor_input_shape.domain() == loco::Domain::Tensor);
78 auto shape = tensor_input_shape.as<loco::TensorShape>();
79
80 auto len = node->dim(0).value();
81 uint32_t wildcard_index = std::numeric_limits<uint32_t>::max();
82 uint32_t product_of_non_wildcard_dims = 1;
83
84 // Deduce
85 for (uint32_t i = 0; i < len; ++i)
86 {
87 auto dim = node->at<loco::DataType::S32>(i);
88 if (dim == -1)
89 {
90 wildcard_index = i;
91 }
92 else
93 {
94 product_of_non_wildcard_dims *= dim;
95 }
96 }
97 assert(wildcard_index != std::numeric_limits<uint32_t>::max());
98
99 // Fix
100 assert(volume(shape) % product_of_non_wildcard_dims == 0);
101 node->at<loco::DataType::S32>(wildcard_index) = volume(shape) / product_of_non_wildcard_dims;
102}

113bool resolve_wildcard_dim(moco::TFReshape *reshape)
114{
115 // Check conditions (WHEN)
116 auto const_shape_input = dynamic_cast<moco::TFConst *>(reshape->shape());
117 if (!const_shape_input)
118 return false;
119
120 if (!has_one_wildcard_dim(const_shape_input))
121 return false;
122
123 auto tensor_input_shape = moco::node_shape(reshape->tensor());
124 if (tensor_input_shape.domain() == loco::Domain::Unknown)
125 return false;
126
127 // Deduce (DO)
128 deduce_and_fix_wildcard_dim(const_shape_input, tensor_input_shape);
129
130 return true;
131}
