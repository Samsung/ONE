37 params.dilation_height_factor = dilation_height_factor;
38 params.dilation_width_factor = dilation_width_factor;

40 assert(input->quantization() != nullptr); // Fix caller
41 assert(filter->quantization() != nullptr); // Fix caller
42 assert(output->quantization() != nullptr); // Fix caller
43
44 const auto *input_scales = input->quantization()->scale();
45 const auto *filter_scales = filter->quantization()->scale();
46 const auto *output_scales = output->quantization()->scale();
47
48 assert(input_scales != nullptr); // Fix caller
49 assert(filter_scales != nullptr); // Fix caller
50 assert(output_scales != nullptr); // Fix caller
51
52 assert(input_scales->size() != 0); // Fix caller
53 assert(filter_scales->size() != 0); // Fix caller
54 assert(output_scales->size() != 0); // Fix caller
55
56 const auto input_zero_points = input->quantization()->zero_point();
57 const auto filter_zero_points = filter->quantization()->zero_point();
58 const auto output_zero_points = output->quantization()->zero_point();
59
60 assert(input_zero_points != nullptr); // Fix caller
61 assert(filter_zero_points != nullptr); // Fix caller
62 assert(output_zero_points != nullptr); // Fix caller

71
72 const auto output_scale = output_scales->operator[](0);
73
74 int32_t activation_min{};
75 int32_t activation_max{};
76 OMStatus status = execute::calculateActivationRangeQuantized(
77 options->fused_activation_function(), static_cast<int32_t>(output_zp), output_scale,
78 output->type(), &activation_min, &activation_max);
79 assert(status == Ok);
80 if (status != Ok)
81 return status;
82
83 // The kernel expects input and filter zero points to be negated.
84 params.input_offset = -static_cast<int32_t>(input_zp); // Note the '-'.
85 params.weights_offset = -static_cast<int32_t>(filter_zp); // Note the '-'.
86 params.output_offset = static_cast<int32_t>(output_zp);
87 params.quantized_activation_min = activation_min;
88 params.quantized_activation_max = activation_max;
89
90 assert(filter_scales->size() > 1); // Support only channel-wise quantization
91 // Channel-wise quantization
92 const auto input_scale = input_scales->operator[](0);
93 const std::vector<double> effective_output_scale =
94 execute::getQuantizedConvolutionMultiplers(input_scale, filter_scales, output_scale);
95
96 size_t n = effective_output_scale.size();
97 params.per_channel_output_shift.resize(n);
98 params.per_channel_output_multiplier.resize(n);
99 for (size_t i = 0; i < n; ++i)
100 {
101 execute::quantizeMultiplier(effective_output_scale[i], &params.per_channel_output_multiplier[i],
102 &params.per_channel_output_shift[i]);
103 }
104
105 return Ok;
106}
