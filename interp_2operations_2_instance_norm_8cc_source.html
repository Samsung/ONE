32void prepareInstanceNorm(ExecEnv *env, const ir::Operation &node)
33{
34 const auto &instancenorm_node =
35 nnfw::misc::polymorphic_downcast<const ir::operation::InstanceNorm &>(node);
36
37 const auto input_index = node.getInputs().at(instancenorm_node.INPUT);
38 const auto output_index = node.getOutputs().at(0);
39 const auto input_tensor = env->tensorAt(input_index);
40
41 if (input_tensor->getShape().rank() != 4)
42 {
43 throw std::runtime_error{"Interp(InstanceNorm): Input should be 4D-tensor"};
44 }
45
46 // Output shape should be same with input
47 env->allocateIfNeeded(output_index, input_tensor->tensorInfo());
48
49 auto output_tensor = env->tensorAt(output_index);
50 UNUSED_RELEASE(output_tensor);
51
52 // Handle same ifm & ofm data type only
53 assert(input_tensor->data_type() == output_tensor->data_type());
54 assert(input_tensor->tensorInfo().shape() == output_tensor->tensorInfo().shape());
55}

63void invoke(const ITensor *input_tensor, const ITensor *gamma_tensor, const ITensor *beta_tensor,
64 const ITensor *output_tensor, const ir::operation::InstanceNorm::Param &param)
65{
66 // Calculate
67 float activation_min, activation_max;
68 calculateActivationRange(param.activation, &activation_min, &activation_max);
69
70 nnfw::cker::InstanceNormParams cker_param;
71 cker_param.epsilon = param.epsilon;
72 cker_param.float_activation_min = activation_min;
73 cker_param.float_activation_max = activation_max;
74
75 const auto cker_input_shape = convertShape(input_tensor->tensorInfo().shape());
76 const auto cker_gamma_shape = convertShape(gamma_tensor->tensorInfo().shape());
77 const auto cker_beta_shape = convertShape(beta_tensor->tensorInfo().shape());
78 const auto cker_output_shape = convertShape(output_tensor->tensorInfo().shape());
79 const float *input_ptr = reinterpret_cast<const float *>(input_tensor->bufferRO());
80 const float *gamma_ptr = reinterpret_cast<const float *>(gamma_tensor->bufferRO());
81 const float *beta_ptr = reinterpret_cast<const float *>(beta_tensor->bufferRO());
82 float *output_ptr = reinterpret_cast<float *>(output_tensor->buffer());
83
84 nnfw::cker::InstanceNorm(cker_param, cker_input_shape, input_ptr, cker_gamma_shape, gamma_ptr,
85 cker_beta_shape, beta_ptr, cker_output_shape, output_ptr);
86}

88void invokeInstanceNorm(const ExecEnv *env, const ir::Operation &node)
89{
90 const auto &instancenorm_node =
91 nnfw::misc::polymorphic_downcast<const ir::operation::InstanceNorm &>(node);
92
93 const auto input_index = node.getInputs().at(instancenorm_node.INPUT);
94 const auto gamma_index = node.getInputs().at(instancenorm_node.GAMMA);
95 const auto beta_index = node.getInputs().at(instancenorm_node.BETA);
96 const auto out_index = node.getOutputs().at(0);
97 const auto input_tensor = env->tensorAt(input_index);
98 const auto gamma_tensor = env->tensorAt(gamma_index);
99 const auto beta_tensor = env->tensorAt(beta_index);
100 const auto out_tensor = env->tensorAt(out_index);
101 const auto data_type = input_tensor->data_type();
102
103 if (data_type == ir::DataType::FLOAT32)
104 {
105 invoke(input_tensor, gamma_tensor, beta_tensor, out_tensor, instancenorm_node.param());
106 }
107 else
108 {
109 throw std::runtime_error{"NYI: Unsupported data type"};
110 }
111}
