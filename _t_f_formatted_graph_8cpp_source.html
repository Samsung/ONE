33std::string opname(uint32_t opnum)
34{
35 static std::string prefix{"tf."};
36
37 switch (static_cast<moco::TFOpcode>(opnum))
38 {
39#define TENSORFLOW_NODE(OPCODE, CLASS) \
40 case moco::TFOpcode::OPCODE: \
41 return prefix + #OPCODE;
42#include <moco/IR/TFNodes.lst>
43#undef TENSORFLOW_NODE
44 default:
45 break;
46 };
47
48 return prefix + "Invalid";
49}

123bool TFNodeSummaryBuilderBase::build(const loco::Node *node, locop::NodeSummary &s) const
124{
125 if (node->dialect() != TFDialect::get())
126 return false;
127
128#define TENSORFLOW_NODE(OPCODE, CLASS) \
129 if (dynamic_cast<const CLASS *>(node)) \
130 { \
131 s.opname(opname(node->opnum())); \
132 return summary(dynamic_cast<const CLASS *>(node), s); \
133 }
134#include <moco/IR/TFNodes.lst>
135#undef TENSORFLOW_NODE
136
137 return false;
138}

182bool TFNodeSummaryBuilder::summary(const TFConst *node, locop::NodeSummary &s) const
183{
184 std::ostringstream ss;
185
186 auto dtype = node->dtype();
187 switch (dtype)
188 {
189 case loco::DataType::S32:
190 ss << node->size<loco::DataType::S32>();
191 break;
192 case loco::DataType::FLOAT32:
193 ss << node->size<loco::DataType::FLOAT32>();
194 break;
195 default:
196 INTERNAL_EXN_V("Unsupported data type", node->name());
197 }
198 s.args().append("size", ss.str());
199 s.state(locop::NodeSummary::State::PartiallyKnown);
200 return true;
201}

379bool MocoNodeSummaryBuilder::build(const loco::Node *node, locop::NodeSummary &s) const
380{
381 if (locop::CanonicalNodeSummaryBuilder(_tbl).build(node, s))
382 {
383 return true;
384 }
