24 * Run backward graph to calculate gradients
25 */
26OMStatus OMBackpropExecute::runBackward(const OMConfig &config, OMBackpropExecuteArgs &args,
27 core::memory::OMRuntimeAllocator &allocator)
28{
29 OMStatus status = Ok;
30
31 core::OMRuntimeContext &context = args.backward_context;
32 core::OMRuntimeStorage &forward_storage = args.forward_storage;
33 core::OMRuntimeStorage &backward_storage = args.backward_storage;
34
35 const core::reader::CircleOperators *operators = context.getCircleOperators();
36
37 const auto num_operators = operators->size();
38 const auto *op_codes = context.getCircleOpcodes();
39
40 uint32_t num_train_layers = config.training_context.num_of_train_layers == 0
41 ? num_operators
42 : config.training_context.num_of_train_layers;
43 uint32_t last_node_pos = std::min(num_operators, num_train_layers);
44
45 for (uint32_t i = 0; i < last_node_pos; ++i)
46 {
47 uint32_t cur_op_index = num_operators - i - 1;
48 auto *cur_op = operators->operator[](cur_op_index);
49
50 status = allocator.allocate(i, &context, &backward_storage);
51
52 if (status != Ok)
53 return status;
54
55 core::OMBuilderID builder_id = core::OMBuilderID::Size;
56 const circle::Operator *op = operators->operator[](cur_op_index);
57 uint32_t index = op->opcode_index();
58
59 assert(index < op_codes->size());
60
61 const auto opcode = op_codes->operator[](index);
62
63 status = core::getBuilderId(opcode, builder_id);
64
65 assert(status == Ok);
66 if (status != Ok)
67 return status;
68
69 args.kernel_index = cur_op_index;
70
71 if (i == last_node_pos - 1)
72 args.is_last_layer = true;
73
74 // Calculate gradients
75 KernelTrainFunc *train_func = nullptr;
76 if (size_t(builder_id) < size_t(core::OMBuilderID::BuiltinOperatorsSize))
77 {
78 // Builtin operator
79 status = kernel_builtin_train.getKernelTrainFunc(builder_id, &train_func);
80 }
81 else
82 {
83 assert(false && "Unsupported kernel type for training");
84 return UnsupportedOp;
85 }
86
87 assert(train_func != nullptr);
88
89 if (status != Ok)
90 return status;
91
92 status = train_func(args);
93
94 assert(status == Ok);
95
96 if (status != Ok)
97 return status;
98
99 // Deallocate tensors data in backward storage
100 status = allocator.deallocate(i, &backward_storage);
101 if (status != Ok)
102 return status;
103
104 // Deallocate tensors data in forward storage
105 status = allocator.deallocate(i, &forward_storage);
106 }
107
108 return status;
109}
