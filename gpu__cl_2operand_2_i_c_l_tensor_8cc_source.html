46void ICLTensor::writeConvertInit(tflite::gpu::TensorObjectConverterBuilder *converter_builder,
47 std::shared_ptr<tflite::gpu::cl::Environment> environment)
48{
49 _environment = environment;
50 TensorObjectDef input_def;
51 input_def.dimensions.b = handle()->Batch();
52 input_def.dimensions.h = handle()->Height();
53 input_def.dimensions.w = handle()->Width();
54 input_def.dimensions.c = handle()->Channels();
55 input_def.object_def.data_layout = DataLayout::BHWC;
56 input_def.object_def.data_type = DataType::FLOAT32;
57 input_def.object_def.object_type = ObjectType::CPU_MEMORY;
58 input_def.object_def.user_provided = true;
59
60 TensorObjectDef permute_def = input_def;
61 permute_def.object_def.object_type = ToObjectType(handle()->GetStorageType());
62
63 auto dims = permute_def.dimensions;
64 const BHWC shape(dims.b, dims.h, dims.w, dims.c);
65 const TensorDescriptor desc{
66 permute_def.object_def.data_type,
67 ToTensorStorageType(permute_def.object_def.object_type, permute_def.object_def.data_layout),
68 Layout::BHWC};
69 if (!AllocateTensorMemory(_environment->context(), shape, desc, &_cl_memory).ok())
70 {
71 throw std::runtime_error("Failed to AllocateTensorMemory");
72 }
73
74 TensorObjectDef output_def = permute_def;
75 output_def.object_def.data_layout = ToDataLayout(handle()->GetStorageType());
76 output_def.object_def.data_type = handle()->GetDataType();
77 input_def.object_def.user_provided = false;
78
79 if (!converter_builder->MakeConverter(input_def, permute_def, &_converter_to).ok())
80 {
81 throw std::runtime_error("Failed to make converter_to");
82 }
83 if (!converter_builder->MakeConverter(permute_def, output_def, &_converter_from).ok())
84 {
85 throw std::runtime_error("Failed to make converter_from");
86 }
87}

132void ICLTensor::enqueueWriteBuffer(const void *ptr, bool blocking)
133{
134 TensorObject input_obj = MakeReadableCpuMemory(
135 absl::MakeSpan(static_cast<const float *>(ptr), _info._shape.DimensionsProduct()));
136
137 TensorObject output_obj;
138
139 TensorObject permute_obj;
140 if (ToObjectType(handle()->GetStorageType()) == ObjectType::OPENCL_TEXTURE)
141 {
142 permute_obj = OpenClTexture{_cl_memory.memory()};
143 }
144 else
145 {
146 permute_obj = OpenClBuffer{_cl_memory.memory()};
147 }
148
149 if (handle()->GetStorageType() == TensorStorageType::BUFFER)
150 {
151 output_obj = OpenClBuffer{handle()->GetMemoryPtr()};
152 }
153 else if (handle()->GetStorageType() == TensorStorageType::IMAGE_BUFFER)
154 {
155 output_obj = OpenClBuffer{handle()->GetMemoryPtrForWriting()};
156 }
157 else
158 {
159 output_obj = OpenClTexture{handle()->GetMemoryPtr()};
160 }
161
162 if (!_converter_to->Convert(input_obj, permute_obj).ok())
163 {
164 throw std::runtime_error("Failed to write cl buffer from cpu memory");
165 }
166
167 if (blocking && !_environment->queue()->WaitForCompletion().ok())
168 {
169 throw std::runtime_error("Failed to WaitForCompletion");
170 }
171
172 if (!_converter_from->Convert(permute_obj, output_obj).ok())
173 {
174 throw std::runtime_error("Failed to change layout");
175 }
176}

178void ICLTensor::enqueueReadBuffer(void *ptr, bool blocking)
179{
180 TensorObject input_obj;
181
182 if (handle()->GetStorageType() == TensorStorageType::BUFFER)
183 {
184 input_obj = OpenClBuffer{handle()->GetMemoryPtr()};
185 }
186 else if (handle()->GetStorageType() == TensorStorageType::IMAGE_BUFFER)
187 {
188 input_obj = OpenClBuffer{handle()->GetMemoryPtrForWriting()};
189 }
190 else
191 {
192 input_obj = OpenClTexture{handle()->GetMemoryPtr()};
193 }
194
195 TensorObject permute_obj;
196 if (ToObjectType(handle()->GetStorageType()) == ObjectType::OPENCL_TEXTURE)
197 {
198 permute_obj = OpenClTexture{_cl_memory.memory()};
199 }
200 else
201 {
202 permute_obj = OpenClBuffer{_cl_memory.memory()};
203 }
204
205 TensorObject output_obj =
206 MakeCpuMemory(absl::MakeSpan(static_cast<float *>(ptr), _info._shape.DimensionsProduct()));
207
208 if (!_converter_from->Convert(input_obj, permute_obj).ok())
209 {
210 throw std::runtime_error("Failed to change layout");
211 }
212 if (!_converter_to->Convert(permute_obj, output_obj).ok())
213 {
214 throw std::runtime_error("Failed to read cl buffer");
215 }
216
217 if (blocking && !_environment->queue()->WaitForCompletion().ok())
218 {
219 throw std::runtime_error("Failed to WaitForCompletion");
220 }
221}
