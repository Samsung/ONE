28template <FusedActivationFunctionType Ac>
29void FullyConnected(const float *input_data, const Dims<4> &input_dims, const float *weights_data,
30 const Dims<4> &weights_dims, const float *bias_data, const Dims<4> &bias_dims,
31 float *output_data, const Dims<4> &output_dims)
32{
33 // TODO(b/62193649): this convoluted shape computation (determining
34 // input_rows from the weights_dims, then MapAsMatrixWithGivenNumberOfRows)
35 // is because the current --variable_batch hack consists in overwriting the
36 // 3rd dimension with the runtime batch size, as we don't keep track for each
37 // array of which dimension is the batch dimension in it.
38 // When that is fixed, this should become:
39 // const auto input_matrix_map =
40 // MapAsMatrixWithFirstDimAsRows(input_data, input_dims);
41 const int input_rows = ArraySize(weights_dims, 0);
42 const auto input_matrix_map =
43 MapAsMatrixWithGivenNumberOfRows(input_data, input_dims, input_rows);
44 const auto filter_matrix_map = MapAsMatrixWithFirstDimAsRows(weights_data, weights_dims);
45 auto output_matrix_map = MapAsMatrixWithFirstDimAsRows(output_data, output_dims);
46
47 Gemm(filter_matrix_map.transpose(), input_matrix_map, &output_matrix_map);
48 AddBiasAndEvalActivationFunction<Ac>(bias_data, bias_dims, output_data, output_dims);
49}
