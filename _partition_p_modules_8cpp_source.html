36void add_graph_input(loco::Graph *graph, luci::CircleInput *input_node)
37{
38 assert(graph != nullptr);
39 assert(input_node != nullptr);
40
41 auto graph_input = graph->inputs()->create();
42 graph_input->name(input_node->name());
43
44 // Set GraphInputOutputIndex for graph
45 input_node->index(graph_input->index());
46
47 // Data type
48 graph_input->dtype(input_node->dtype());
49
50 // Shape of GraphInput
51 auto input_shape = std::make_unique<loco::TensorShape>();
52 input_shape->rank(input_node->rank());
53 for (uint32_t r = 0; r < input_node->rank(); ++r)
54 {
55 if (input_node->dim(r).known())
56 input_shape->dim(r).set(input_node->dim(r).value());
57 }
58 graph_input->shape(std::move(input_shape));
59}

89std::unique_ptr<loco::Graph> clone_graph(loco::Graph *graph_org, luci::CloneContext &clonectx)
90{
91 auto graph = loco::make_graph();
92 auto graph_clone = graph.get();
93 auto &graph_name = graph_org->name();
94
95 graph_clone->name(graph_name);
96
97 // clone inputs
98 auto inputs = graph_org->inputs();
99 assert(inputs);
100 for (uint32_t n = 0; n < inputs->size(); ++n)
101 {
102 auto input_org = luci::input_node(graph_org, n);
103 assert(input_org != nullptr);
104
105 auto *input_clone = graph_clone->nodes()->create<luci::CircleInput>();
106 luci::copy_common_attributes(input_org, input_clone);
107
108 add_graph_input(graph_clone, input_clone);
109 clonectx.emplace(input_org, input_clone);
110 }
111
112 // clone nodes
113 auto nodes = graph_org->nodes();
114 for (uint32_t n = 0; n < nodes->size(); ++n)
115 {
116 auto node = nodes->at(n);
117
118 // skip for CircleInput, CircleOutput
119 if (dynamic_cast<luci::CircleInput *>(node) != nullptr)
120 continue;
121 if (dynamic_cast<luci::CircleOutput *>(node) != nullptr)
122 continue;
123
124 auto node_org = loco::must_cast<luci::CircleNode *>(node);
125 assert(clonectx.find(node_org) == clonectx.end());
126
127 auto *node_clone = clone_node(node_org, graph_clone);
128 clonectx.emplace(node_org, node_clone);
129 }
130
131 // connect nodes
132 for (uint32_t n = 0; n < nodes->size(); ++n)
133 {
134 auto node = nodes->at(n);
135
136 // skip for CircleInput, CircleOutput
137 if (dynamic_cast<luci::CircleInput *>(node) != nullptr)
138 continue;
139 if (dynamic_cast<luci::CircleOutput *>(node) != nullptr)
140 continue;
141
142 auto node_org = loco::must_cast<luci::CircleNode *>(node);
143 clone_connect(node_org, clonectx);
144 }
145
146 // clone outputs
147 for (uint32_t n = 0; n < graph_org->outputs()->size(); ++n)
148 {
149 auto output_org = luci::output_node(graph_org, n);
150 assert(output_org != nullptr);
151
152 auto *output_clone = graph_clone->nodes()->create<luci::CircleOutput>();
153 luci::copy_common_attributes(output_org, output_clone);
154 // note: we don't add output_clone to clonectx.
155 // logically, output is not used as an input to any other nodes.
156 auto output_from = loco::must_cast<luci::CircleNode *>(output_org->from());
157 auto it = clonectx.find(output_from);
158 assert(it != clonectx.end());
159 output_clone->from(it->second);
160
161 add_graph_output(graph_clone, output_clone);
162 }
163
164 return graph;
165}

167void clone_recursive_subgraphs(luci::PartedModule &pm, loco::Graph *graph,
168 const luci::CloneContext &clonectx)
169{
170 auto nodes = graph->nodes();
171 for (uint32_t n = 0; n < nodes->size(); ++n)
172 {
173 {
174 auto if_node = dynamic_cast<luci::CircleIf *>(nodes->at(n));
175 if (if_node != nullptr)
176 {
177 clone_ifnode_subgraphs(pm, if_node, clonectx);
178 }
179 }
180 {
181 auto while_node = dynamic_cast<luci::CircleWhile *>(nodes->at(n));
182 if (while_node != nullptr)
183 {
184 clone_whilenode_subgraphs(pm, while_node, clonectx);
185 }
186 }
187 // TODO handle others
188 }

191void clone_ifnode_subgraphs(luci::PartedModule &pm, const luci::CircleIf *if_node,
192 const luci::CloneContext &clonectx)
193{
194 assert(if_node != nullptr);
195
196 auto it = clonectx.find(if_node);
197 assert(it != clonectx.end());
198 auto if_clone = loco::must_cast<luci::CircleIf *>(it->second);
199
200 luci::CloneContext then_clonectx;
201 luci::CloneContext else_clonectx;
202
203 auto then_graph = if_node->then_graph();
204 auto else_graph = if_node->else_graph();
205
206 auto then_clone = clone_graph(then_graph, then_clonectx);
207 auto else_clone = clone_graph(else_graph, else_clonectx);
208 if_clone->then_graph(then_clone.get());
209 if_clone->else_graph(else_clone.get());
210
211 pm.module->add(std::move(then_clone));
212 int32_t then_index = pm.module->size() - 1;
213 pm.module->add(std::move(else_clone));
214 int32_t else_index = pm.module->size() - 1;
215 if_clone->then_branch(then_index);
216 if_clone->else_branch(else_index);
217
218 // do recursive copy subgraphs of CircleIf if there are any,
219 // inside then_graph or else_graph.
220 clone_recursive_subgraphs(pm, then_graph, then_clonectx);
221 clone_recursive_subgraphs(pm, else_graph, else_clonectx);
222}

224void clone_whilenode_subgraphs(luci::PartedModule &pm, const luci::CircleWhile *while_node,
225 const luci::CloneContext &clonectx)
226{
227 assert(while_node != nullptr);
228
229 auto it = clonectx.find(while_node);
230 assert(it != clonectx.end());
231 auto while_clone = loco::must_cast<luci::CircleWhile *>(it->second);
232
233 luci::CloneContext cond_clonectx;
234 luci::CloneContext body_clonectx;
235
236 auto cond_graph = while_node->cond_graph();
237 auto body_graph = while_node->body_graph();
238
239 auto cond_clone = clone_graph(cond_graph, cond_clonectx);
240 auto body_clone = clone_graph(body_graph, body_clonectx);
241 while_clone->cond_graph(cond_clone.get());
242 while_clone->body_graph(body_clone.get());
243
244 pm.module->add(std::move(cond_clone));
245 int32_t cond_index = pm.module->size() - 1;
246 pm.module->add(std::move(body_clone));
247 int32_t body_index = pm.module->size() - 1;
248 while_clone->cond_branch(cond_index);
249 while_clone->body_branch(body_index);
250
251 // do recursive copy subgraphs of CircleWhile if there are any,
252 // inside cond_graph or body_graph.
253 clone_recursive_subgraphs(pm, cond_graph, cond_clonectx);
254 clone_recursive_subgraphs(pm, body_graph, body_clonectx);

260void build_graph(luci::PartedModule &pm, loco::Graph *graph, const luci::PGroup *pgroup)
261{
262 LOGGER(l);
263
264 luci::CloneContext clonectx;
265
266 // add input node(s)
267 for (auto *input : pgroup->inputs)
268 {
269 auto *input_clone = graph->nodes()->create<luci::CircleInput>();
270 luci::copy_common_attributes(input, input_clone);
271
272 add_graph_input(graph, input_clone);
273 clonectx.emplace(input, input_clone);
274
275 INFO(l) << "MAP: "
276 << " input(" << input << ") -> " << input_clone << "(" << input_clone->name() << ")";
277 }
278
279 // add CircleConst for inputs
280 for (auto &pnode : pgroup->pnodes)
281 {
282 auto node = pnode->node;
283 uint32_t arity = node->arity();
284 for (uint32_t a = 0; a < arity; ++a)
285 {
286 auto in_a_const = dynamic_cast<luci::CircleConst *>(node->arg(a));
287 if (in_a_const != nullptr)
288 {
289 auto it = clonectx.find(in_a_const);
290 if (it == clonectx.end())
291 {
292 auto *clone = clone_node(in_a_const, graph);
293 clonectx.emplace(in_a_const, clone);
294
295 INFO(l) << "MAP: "
296 << " const(" << in_a_const << ") -> " << clone << "(" << clone->name() << ")";
297 }
298 }
299 }
300 }
301
302 // add nodes
303 for (auto &pnode : pgroup->pnodes)
304 {
305 auto *clone = clone_node(pnode->node, graph);
306 clonectx.emplace(pnode->node, clone);
307
308 INFO(l) << "MAP: "
309 << " node(" << pnode->node << ") -> " << clone << "(" << clone->name() << ")";
310 }
311 // connect nodes
312 for (auto &pnode : pgroup->pnodes)
313 {
314 clone_connect(pnode->node, clonectx);
315 }
316
317 // add output node(s)
318 for (auto *output : pgroup->outputs)
319 {
320 auto *output_clone = graph->nodes()->create<luci::CircleOutput>();
321 luci::copy_common_attributes(output, output_clone);
322 // note: we don't add output_clone to clonectx.
323 // logically, output is not used as an input to any other nodes.
324
325 auto it = clonectx.find(output);
326 assert(it != clonectx.end());
327 output_clone->from(it->second);
328
329 add_graph_output(graph, output_clone);
330
331 INFO(l) << "MAP: "
332 << "output(" << output << ") -> " << output_clone << "(" << output_clone->name() << ")"
333 << ": from " << it->second << "(" << it->second->name() << ")";
334 }
335
336 // TODO relocate this if needed
337 // subgraphs for IF/WHILE/... nodes
338 for (auto &pnode : pgroup->pnodes)
339 {
340 {
341 auto if_node = dynamic_cast<const luci::CircleIf *>(pnode->node);
342 if (if_node != nullptr)
343 {
344 clone_ifnode_subgraphs(pm, if_node, clonectx);
345 }
346 }
347 {
348 auto while_node = dynamic_cast<const luci::CircleWhile *>(pnode->node);
349 if (while_node != nullptr)
350 {
351 clone_whilenode_subgraphs(pm, while_node, clonectx);
352 }
353 }
354 // TODO handle others
355 }

358std::string make_name(const luci::PGroup *pgroup)
359{
360 auto &first_pnode = *pgroup->pnodes.begin();
361 auto *first_node = first_pnode->node;
362 std::string name = first_node->graph()->name();
363 name = name + "_" + pgroup->group;
364 return name;

375luci::PartedModules produce_pmodules(const luci::PGroups *pgroups)
376{
377 LOGGER(l);
378
379 luci::PartedModules pms;
380
381 for (auto &pgroup : pgroups->pgroups)
382 {
383 luci::PartedModule pm;
384 pm.module = std::make_unique<luci::Module>();
385 pm.group = pgroup->group;
386
387 // the main graph for this module
388 auto graph = loco::make_graph();
389 auto graph_ptr = graph.get();
390
391 auto graph_name = make_name(pgroup.get());
392 graph->name(graph_name);
393
394 // Add main graph so that other subgraphs can be added inside build_graph
395 pm.module->add(std::move(graph));
396
397 INFO(l) << "--- Partition Graph build----------------------";
398 INFO(l) << "--- name: " << graph_name;
399 build_graph(pm, graph_ptr, pgroup.get());
400
401 pms.pmodules.emplace_back(std::move(pm));
402 }
403
404 return pms;
405}
