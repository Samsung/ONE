51// Controls whether profiling is enabled or disabled and collects profiles.
52// TFLite is used on platforms that don't have posix threads, so the profiler is
53// kept as simple as possible. It is designed to be used only on a single
54// thread.
55//
56// Profiles are collected using Scoped*Profile objects that begin and end a
57// profile event.
58// An example usage is shown in the example below:
59//
60// Say Worker class has a DoWork method and we are interested in profiling
61// the overall execution time for DoWork and time spent in Task1 and Task2
62// functions.
63//
64// class Worker {
65// public:
66// void DoWork() {
67// ScopedProfile(&controller, "DoWork");
68// Task1();
69// Task2();
70// .....
71// }
72//
73// void Task1() {
74// ScopedProfile(&controller, "Task1");
75// ....
76// }
77//
78// void Task2() {
79// ScopedProfile(&controller, "Task2");
80// }
81//
82// Profiler profiler;
83// }
84//
85// We instrument the functions that need to be profiled.
86//
87// Profile can be collected by enable profiling and then getting profile
88// events.
89//
90// void ProfileWorker() {
91// Worker worker;
92// worker.profiler.EnableProfiling();
93// worker.DoWork();
94// worker.profiler.DisableProfiling();
95// // Profiling is complete, extract profiles.
96// auto profile_events = worker.profiler.GetProfiles();
97// }
98//
99//
100class Profiler {
101 public:
102 Profiler() : buffer_(1024, false) {}
103
104 void StartProfiling() { buffer_.SetEnabled(true); }
105 void StopProfiling() { buffer_.SetEnabled(false); }
106 void Reset() { buffer_.Reset(); }
107 std::vector<const ProfileEvent*> GetProfileEvents() {
108 std::vector<const ProfileEvent*> profile_events;
109 profile_events.reserve(buffer_.Size());
110 for (size_t i = 0; i < buffer_.Size(); i++) {
111 profile_events.push_back(buffer_.At(i));
112 }
113 return profile_events;
114 }
115
116 private:
117 friend class ScopedProfile;
118 friend class ScopedOperatorProfile;
119 ProfileBuffer* GetProfileBuffer() { return &buffer_; }
120 ProfileBuffer buffer_;
121};

123class ScopedProfile {
124 public:
125 // Adds a profile event to profile that begins with the construction
126 // of object and ends when the object goes out of scope.
127 // The lifetime of tag should be at least the lifetime of profiler.
128
129 ScopedProfile(Profiler* profiler, const char* tag)
130 : buffer_(nullptr), event_handle_(0) {
131 if (profiler) {
132 buffer_ = profiler->GetProfileBuffer();
133 event_handle_ =
134 buffer_->BeginEvent(tag, ProfileEvent::EventType::DEFAULT, 0);
135 }
136 }
137 ~ScopedProfile() {
138 if (buffer_) {
139 buffer_->EndEvent(event_handle_);
140 }
141 }
142
143 private:
144 ProfileBuffer* buffer_;
145 int32_t event_handle_;
146};
