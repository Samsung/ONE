190NNFW_STATUS nnfw_query_info_u32(nnfw_session *session, NNFW_INFO_ID id, uint32_t *val)
191{
192 (void)session;
193 switch (id)
194 {
195 case NNFW_INFO_ID_VERSION:
196 if (val)
197 {
198 *val = NNFW_VERSION;
199 return NNFW_STATUS_NO_ERROR;
200 }
201 break;
202 default:
203 return NNFW_STATUS_ERROR;
204 }
205 // It should not be reached.
206 return NNFW_STATUS_ERROR;
207}
