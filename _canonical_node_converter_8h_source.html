46template <typename CanonicalType>
47bool CanonicalNodeConverter<CanonicalType>::run(loco::Graph *graph)
48{
49 auto active_nodes = loco::active_nodes(loco::output_nodes(graph));
50 bool changed = false;
51
52 for (auto node : active_nodes)
53 {
54 // TODO Generalize this to all loco dialects
55 if (node->dialect() == loco::CanonicalDialect::get())
56 {
57 auto the_node = dynamic_cast<CanonicalType *>(node);
58 if (the_node != nullptr)
59 {
60 if (convert(the_node))
61 changed = true;
62 }
63 }
64 }
65
66 return changed;
67}
