36template <typename ConstNode> bool is_fixed_shape_input(ConstNode *const_shape_input)
37{
38 if (const_shape_input == nullptr)
39 return false;
40
41 // Shape input should be integer tensor of rank 1, e.g. [2, 3, 4] or [3, -1]
42 // TODO Support other possible data types, e.g. S64
43 assert(const_shape_input->dtype() == loco::DataType::S32);
44 assert(const_shape_input->rank() == 1);
45
46 auto shape_rank = const_shape_input->dim(0).value();
47 assert(shape_rank > 0);
48
49 for (uint32_t axis = 0; axis < shape_rank; ++axis)
50 {
51 auto shape_dim = const_shape_input->template at<loco::DataType::S32>(axis);
52 if (shape_dim == -1)
53 {
54 // has wildcard dimension, i.e. dynamic reshape
55 return false;
56 }
57 if (!(shape_dim >= 1))
58 {
59 throw oops::UserExn("New shape of Reshape has invalid dimension");
60 }
61 }
62 return true;
63}

66bool canonicalize_reshape(loco::Graph *graph, moco::TFReshape *node)
67{
68 LOGGER(l);
69 INFO(l) << "TFNodeCanonicalize TFReshape begin";
70
99 // create loco node to replace
100 auto fixed_reshape = graph->nodes()->create<loco::FixedReshape>();
101
102 // Supports 2 cases for Reshape's shape input:
103 // TF-dialect TFConst or Canonical ConstGen
104 loco::Node *shape_input = node->shape();
105 auto tfconst_shape_input = dynamic_cast<moco::TFConst *>(shape_input);
106 auto constgen_shape_input = dynamic_cast<loco::ConstGen *>(shape_input);
107
108 if (tfconst_shape_input)
109 {
110 // Only support fixed reshape
111 // TODO support dynamic reshape
112 if (!(is_fixed_shape_input(tfconst_shape_input)))
113 {
114 throw oops::UserExn("Supports only fixed reshape", node->name());
115 }
116
117 auto rank = tfconst_shape_input->dim(0).value();
118 fixed_reshape->rank(rank);
119 for (uint32_t axis = 0; axis < rank; ++axis)
120 {
121 fixed_reshape->dim(axis) = tfconst_shape_input->at<loco::DataType::S32>(axis);
122 }
123 }
124 else if (constgen_shape_input)
125 {
126 // ditto
127 if (!(is_fixed_shape_input(constgen_shape_input)))
128 {
129 throw oops::UserExn("Supports only fixed reshape", node->name());
130 }
131
132 auto rank = constgen_shape_input->dim(0).value();
133 fixed_reshape->rank(rank);
134 for (uint32_t axis = 0; axis < rank; ++axis)
135 {
136 fixed_reshape->dim(axis) = constgen_shape_input->at<loco::DataType::S32>(axis);
137 }
138 }
139 else
140 {
141 // TODO support dynamic reshape from not const node
142 throw oops::UserExn("Supports only const node as input shape", node->name());
143 }
144
145 // replace
146 auto in = node->tensor();
147 fixed_reshape->input(in);
148
149 replace(node).with(fixed_reshape);
150
151 INFO(l) << "TFNodeCanonicalize TFReshape done";
152
153 return true;
154}
