87void CPPCodeGenerator::materializeModelParams(ostream &out, const Serializer &s)
88{
89 using namespace params;
90
91 // First form a dump header
92 char header[HEADER_LEN];
93 uint32_t version = s.getFormatVersion();
94 uint32_t hash = s.getModelHash();
95 static_assert(VERSION_LEN == sizeof(version), "version length mismatch");
96 static_assert(HASH_LEN == sizeof(hash), "hash length mismatch");
97 memcpy(header, MAGIC, MAGIC_LEN);
98 memcpy(header + MAGIC_LEN, &version, VERSION_LEN);
99 memcpy(header + MAGIC_LEN + VERSION_LEN, &hash, HASH_LEN);
100
101 out.write(header, HEADER_LEN);
102 if (out.fail())
103 throw runtime_error("Failed to write model parameters header");
104 auto &params = s.getBuffer();
105 out.write(params.data(), params.size());
106 if (out.fail())
107 throw runtime_error("Failed to write model Parameters");
108}

110void CPPCodeGenerator::run(mir::Graph *graph)
111{
112 assert(graph);
113
114 // visit and analyze graph
115 ModelAnalyzer ma;
116 ma.analyze(graph);
117 // serialize parameters
118 Serializer serializer;
119 serializer.serialize(ma.getInferenceSequence());
120 // rename tensors for specific backend language
121 formatTensorNames(ma);
122
123 fs::create_directory(_output_dir);
124
125 const string base_path = _output_dir + "/" + _artifact_name;
126 const string header_path = base_path + ".h";
127 const string code_path = base_path + ".cpp";
128 const string params_path = base_path + ".params";
129
130 // Print header
131 auto header_stream = getStream(header_path);
132 materializeHeader(*header_stream, ma);
133 header_stream.reset();
134
135 // Print code
136 auto code_stream = getStream(code_path);
137 materializeCode(*code_stream, ma, serializer);
138 code_stream.reset();
139
140 // Print model parameters
141 auto model_stream = getStream(params_path);
142 materializeModelParams(*model_stream, serializer);
143 model_stream.reset();
144}

150void CPPCodeGenerator::formatTensorNames(const ModelAnalyzer &ma)
151{
152 using TensorType = TensorDescriptor::Type;
153
154 int tmp_tensors = 0;
155 for (const TensorDescriptor &td : ma.getTensors())
156 {
157 string formatted_name;
158 if (td.name.empty())
159 {
160 assert(td.type == TensorType::temporary);
161 formatted_name = "Tensor_" + to_string(tmp_tensors++);
162 }
163 else
164 {
165 if (td.type != TensorType::temporary)
166 formatted_name.append("_");
167 formatted_name.append(td.name);
168 for (char &c : formatted_name)
169 {
170 if (!isalnum(c))
171 c = '_';
172 }
173 }
174 _formattedTensors.push_back(move(formatted_name));
175 }
176}

185void CPPCodeGenerator::materializeHeader(ostream &out, const ModelAnalyzer &ma)
186{
187 string class_name = ma.getModelName() + "Model";
188
189 out.write(cpp_header_types, sizeof(cpp_header_types));
190 out << "class " << class_name
191 << "\n"
192 "{\n"
193 "public:\n"
194 " "
195 << class_name
196 << "(const std::string& parametersPath);\n"
197 " ~"
198 << class_name << "();\n";
199 // generate input setters
200 if (ma.getInputs().size() == 1)
201 out << " bool setInput(const Tensor& inputs);\n";
202 for (const size_t inId : ma.getInputs())
203 {
204 const string &tName = _formattedTensors[inId];
205 out << " bool set" << tName << "(const Tensor& t);\n";
206 }
207 // generate output getters
208 if (ma.getOutputs().size() == 1)
209 {
210 out << " std::shared_ptr<Tensor> getOutput();\n";
211 }
212 for (const size_t out_id : ma.getPersistentTensors())
213 {
214 const string &tensor_name = _formattedTensors[out_id];
215 out << " std::shared_ptr<Tensor> get" << tensor_name << "();\n";
216 }
217 out << " void doInference();\n\n"
218 "private:\n"
219 " "
220 << class_name
221 << "() = delete;\n"
222 " "
223 << class_name << "(const " << class_name
224 << "& orig) = delete;\n"
225 " "
226 << class_name << "& operator=(const " << class_name << "& orig) = delete;\n";
227 // generate input/output tensors
228 for (const size_t in_tensor_id : ma.getInputs())
229 {
230 const string &tName = _formattedTensors[in_tensor_id];
231 out << " Tensor " << tName << ";\n";
232 }
233 for (const size_t out_tensor_id : ma.getPersistentTensors())
234 {
235 const string &tName = _formattedTensors[out_tensor_id];
236 out << " std::shared_ptr<Tensor> " << tName << ";\n";
237 }
238 // pointer to NN parameters
239 out << " char* _parameters;\n";
240 out << " size_t _paramSize;\n";
241 out << "};\n";
242}

275void CPPCodeGenerator::printSetter(ostream &out, const string &class_name,
276 const string &setter_name, const TensorDescriptor &td)
277{
278
279 const string &var_name = _formattedTensors[td.id];
280 out << "bool " << class_name << "::set" << setter_name
281 << "(const Tensor& t)\n"
282 "{\n";
283 // need to insert input correctness check
284 const mir::Shape expected = td.shape;
285 int rank = expected.rank();
286 if (rank != 0)
287 {
288 out << " "
289 << "if (t.getShape().getDims() != " << td.shape.rank() << ") return false;\n";
290 for (int i = 0; i < rank; ++i)
291 out << " "
292 << "if (t.getShape()[" << i << "] != " << expected.dim(i) << ") return false;\n";
293 }
294 out << " " << var_name
295 << " = t;\n"
296 " return true;\n"
297 "}\n\n";

314void CPPCodeGenerator::materializeCall(ostream &out, const ModelAnalyzer &ma,
315 const sir::CallFunction *call)
316{
317 assert(call != nullptr);
318 if (call->mirOp->getType() == mir::Operation::Type::input)
319 return;
320 // materialize call
321 out << " " << call->funcName << "(";
322 const auto &prev_nodes = call->mirOp->getInputs();
323 const auto &out_tensors = call->outputs;
324 vector<string> args;
325 args.reserve(prev_nodes.size() + out_tensors.size() + 1);
326 // gather output arguments
327 gatherOperationArguments(ma, call->outputs, args);
328 // parameters offset
329 args.push_back("_parameters + " + to_string(params::HEADER_LEN + call->paramStartOffset));
330 // gather input arguments
331 gatherOperationArguments(ma, call->inputs, args);
332 // put arguments into stream
333 printOperationArgs(out, args);
334 out << ");\n";
335}

369void CPPCodeGenerator::materializeInferenceSequence(ostream &out, const ModelAnalyzer &ma)
370{
371
372 // Allocate temporary(im2col) tensor
373 out << " Tensor " << _formattedTensors[ma.getTempTID()] << "(Shape{" << ma.getMaxTemporarySize()
374 << "});\n";
375
376 for (const unique_ptr<Action> &action : ma.getInferenceSequence())
377 {
378 Action *ptr = action.get();
379 switch (action->type)
380 {
381 case Action::Type::callFunction:
382 materializeCall(out, ma, dynamic_cast<const sir::CallFunction *>(ptr));
383 break;
384 case Action::Type::transposeTensor:
385 materializeTranspose(out, ma, dynamic_cast<const sir::TransposeTensor *>(ptr));
386 break;
387 case Action::Type::createTmp:
388 materializeConstructor(out, ma, dynamic_cast<const sir::CreateTmp *>(ptr));
389 break;
390 case Action::Type::destroyTmp:
391 materializeDestructor(out, ma, dynamic_cast<const sir::DestroyTmp *>(ptr));
392 break;
393 default:
394 assert(false && "unexpected action type");
395 }
396 }
397}

403void CPPCodeGenerator::materializeCode(ostream &out, const ModelAnalyzer &ma, const Serializer &s)
404{
405 string class_name = ma.getModelName() + "Model";
406
407 out << "#include \"" << _artifact_name << ".h\"\n";
408
409 // put operations from tflite
410 out.write(eigen, sizeof(eigen));
411
412 out.write(CommonData, sizeof(CommonData));
413
414 out.write(cpp_common_funcs, sizeof(cpp_common_funcs));
415 out.write(cpp_capped_relu, sizeof(cpp_capped_relu));
416 out.write(cpp_concat, sizeof(cpp_concat));
417 out.write(cpp_conv, sizeof(cpp_conv));
418 out.write(cpp_depthwise_conv, sizeof(cpp_depthwise_conv));
419 out.write(cpp_fully_connected, sizeof(cpp_fully_connected));
420 out.write(cpp_resize, sizeof(cpp_resize));
421 out.write(cpp_sigmoid, sizeof(cpp_sigmoid));
422 out.write(cpp_pool, sizeof(cpp_pool));
423 out.write(cpp_relu, sizeof(cpp_relu));
424 out.write(cpp_reduce, sizeof(cpp_reduce));
425 out.write(cpp_softmax, sizeof(cpp_softmax));
426 out.write(cpp_slice, sizeof(cpp_slice));
427 out.write(cpp_elementwise, sizeof(cpp_elementwise));
428 out.write(cpp_elu, sizeof(cpp_elu));
429 out.write(cpp_tanh, sizeof(cpp_tanh));
430 out.write(cpp_pad, sizeof(cpp_pad));
431 out.write(cpp_sqrt, sizeof(cpp_sqrt));
432 out.write(cpp_conv_transpose, sizeof(cpp_conv_transpose));
433 out.write(cpp_transpose, sizeof(cpp_transpose));
434 out.write(cpp_gather, sizeof(cpp_gather));
435 out.write(cpp_broadcast, sizeof(cpp_broadcast));
436 // Operations calls into all of the above
437 out.write(cpp_operations, sizeof(cpp_operations));
438 // Below call into operations
439 out.write(cpp_leaky_relu, sizeof(cpp_leaky_relu));
440
441 // gen NN constructor
442 out << class_name << "::" << class_name
443 << "(const string& parametersPath)\n"
444 "{\n"
445 " readParameters(_parameters, _paramSize, parametersPath, "
446 << s.getFormatVersion() << ", " << s.getModelHash()
447 << ");\n"
448 "}\n\n";
449 // gen NN destructor
450 out << class_name << "::~" << class_name
451 << "()\n"
452 "{\n"
453 " releaseParameters(_parameters, _paramSize);\n"
454 "}\n\n";
455 // generate input setters
456 // generate main setter if network has only one
457 const auto &inputs = ma.getInputs();
458 const auto &tensors = ma.getTensors();
459 if (inputs.size() == 1)
460 {
461 const TensorDescriptor &td = tensors[inputs[0]];
462 printSetter(out, class_name, "Input", td);
463 }
464 // generate setters by names
465 for (size_t input_tensor_id : inputs)
466 {
467 const string &input_tensor_name = _formattedTensors[input_tensor_id];
468 const TensorDescriptor &td = tensors[input_tensor_id];
469 printSetter(out, class_name, input_tensor_name, td);
470 }
471
472 // gen output getters
473 // generate main getter if network has only one
474 const auto &outputs = ma.getOutputs();
475 if (outputs.size() == 1)
476 {
477 const TensorDescriptor &td = tensors[outputs[0]];
478 printGetter(out, class_name, "Output", td);
479 }
480 for (size_t output_tensor_id : ma.getPersistentTensors())
481 {
482 const string &output_tensor_name = _formattedTensors[output_tensor_id];
483 const TensorDescriptor &td = tensors[output_tensor_id];
484 printGetter(out, class_name, output_tensor_name, td);
485 }
486 out << "void " << class_name
487 << "::doInference()\n"
488 "{\n";
489 for (size_t output_tensor_id : ma.getPersistentTensors())
490 {
491 const string &output_tensor_name = _formattedTensors[output_tensor_id];
492 out << " " << output_tensor_name << ".reset(new Tensor());\n";
493 }
494
495 // gen inference sequence
496 materializeInferenceSequence(out, ma);
497 out << "}";
498}
