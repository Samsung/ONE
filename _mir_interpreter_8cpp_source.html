67void MIRInterpreter::setTensor(const Operation::Output *output, TensorVariant tensor)
68{
69 const auto result = _tensors.emplace(output, std::move(tensor));
70 if (!result.second)
71 {
72 const std::string &name = output->getName();
73 throw std::runtime_error("Attempt to overwrite data for tensor \"" + name + "\".");
74 }
75}

77TensorVariant &MIRInterpreter::allocateTensor(const Operation::Output *output)
78{
79 const auto result = _tensors.emplace(output, output->getType());
80 if (!result.second)
81 {
82 const std::string &name = output->getName();
83 throw std::runtime_error("Attempt to overwrite data for tensor \"" + name + "\".");
84 }
85 return result.first->second;
86}

88const TensorVariant &MIRInterpreter::getTensor(const Operation::Output *output) const
89{
90 const auto it = _tensors.find(output);
91 if (it == _tensors.end())
92 {
93 const std::string &name = output->getName();
94 throw std::runtime_error("Can't find data for tensor \"" + name + "\".");
95 }
96 return it->second;
97}

99std::vector<std::reference_wrapper<const TensorVariant>>
100MIRInterpreter::getInputTensors(const Operation &op)
101{
102 std::vector<std::reference_wrapper<const TensorVariant>> tensors;
103 for (const Operation::Output *input : op.getInputs())
104 {
105 tensors.emplace_back(getTensor(input));
106 }
107 return tensors;
108}

110std::vector<std::reference_wrapper<TensorVariant>>
111MIRInterpreter::allocateOutputTensors(const Operation &op)
112{
113 std::vector<std::reference_wrapper<TensorVariant>> tensors;
114 for (const Operation::Output &output : op.getOutputs())
115 {
116 tensors.emplace_back(allocateTensor(&output));
117 }
118 return tensors;
119}

272void MIRInterpreter::visit(ops::ResizeOp &op)
273{
274 // TODO support types other than float32
275 auto inputs = getInputTensors(op);
276 assert(inputs[0].get().getElementType() == mir::DataType::FLOAT32);
277 auto outputs = allocateOutputTensors(op);
278
279 Tensor<float> input(inputs[0]);
280 assert(op.getMode() == ops::ResizeOp::ResizeMethod::nearestNeighbor);
281
282 auto scales = op.getScales();
283 Fill(outputs[0], [&scales, &input](const Index &id) {
284 Index in_idx;
285 in_idx.resize(4);
286 for (int i = 0; i < input.getShape().rank(); i++)
287 {
288 in_idx.at(i) = static_cast<int>(floorf(id.at(i) / scales[i]));
289 }
290 return input.at(in_idx);
291 });
292}
