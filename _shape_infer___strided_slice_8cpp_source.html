95inline int Clamp(const int32_t v, const int32_t lo, const int32_t hi)
96{
97 LUCI_ASSERT(!(hi < lo), "Clamp hi < lo");
98 if (hi < v)
99 return hi;
100 if (v < lo)
101 return lo;
102 return v;

105// Return the index for the first element along that axis. This index will be a
106// positive integer between [0, axis_size - 1] that can be used to index
107// directly into the data.
108inline int32_t StartForAxis(const StridedSliceParams &params, const loco::TensorShape &input_shape,
109 uint32_t axis)
110{
111 const auto begin_mask = params.begin_mask;
112 const auto *start_indices = params.start_indices;
113 const auto *strides = params.strides;
114 const int32_t axis_size = static_cast<int>(input_shape.dim(axis).value());
115 if (axis_size == 0)
116 {
117 return 0;
118 }
119 // Begin with the specified index.
120 int32_t start = start_indices[axis];
121
122 // begin_mask override
123 if (begin_mask & (1 << axis))
124 {
125 if (strides[axis] > 0)
126 {
127 // Forward iteration - use the first element. These values will get
128 // clamped below (Note: We could have set them to 0 and axis_size-1, but
129 // use lowest() and max() to maintain symmetry with StopForAxis())
130 start = std::numeric_limits<int32_t>::lowest();
131 }
132 else
133 {
134 // Backward iteration - use the last element.
135 start = std::numeric_limits<int32_t>::max();
136 }
137 }
138
139 // Handle negative indices
140 if (start < 0)
141 {
142 start += axis_size;
143 }
144
145 // Clamping
146 if (strides[axis] > 0)
147 {
148 // Forward iteration
149 start = Clamp(start, 0, axis_size);
150 }
151 else
152 {
153 // Backward iteration
154 start = Clamp(start, -1, axis_size - 1);
155 }
156
157 return start;
158}
159
160// Return the "real" index for the end of iteration along that axis. This is an
161// "end" in the traditional C sense, in that it points to one past the last
162// element. ie. So if you were iterating through all elements of a 1D array of
163// size 4, this function would return 4 as the stop, because it is one past the
164// "real" indices of 0, 1, 2 & 3.
165inline int32_t StopForAxis(const StridedSliceParams &params, const loco::TensorShape &input_shape,
166 int32_t axis, int32_t start_for_axis)
167{
168 const auto end_mask = params.end_mask;
169 const auto shrink_axis_mask = params.shrink_axis_mask;
170 const auto *stop_indices = params.stop_indices;
171 const auto *strides = params.strides;
172 const int32_t axis_size = static_cast<int32_t>(input_shape.dim(axis).value());
173 if (axis_size == 0)
174 {
175 return 0;
176 }
177
178 // Begin with the specified index
179 const bool shrink_axis = shrink_axis_mask & (1 << axis);
180 int32_t stop = stop_indices[axis];
181
182 // When shrinking an axis, the end position does not matter (and can be
183 // incorrect when negative indexing is used, see Issue #19260). Always use
184 // start_for_axis + 1 to generate a length 1 slice, since start_for_axis has
185 // already been adjusted for negative indices.
186 if (shrink_axis)
187 {
188 return start_for_axis + 1;
189 }
190
191 // end_mask override
192 if (end_mask & (1 << axis))
193 {
194 if (strides[axis] > 0)
195 {
196 // Forward iteration - use the last element. These values will get
197 // clamped below
198 stop = std::numeric_limits<int32_t>::max();
199 }
200 else
201 {
202 // Backward iteration - use the first element.
203 stop = std::numeric_limits<int32_t>::lowest();
204 }
205 }
206
207 // Handle negative indices
208 if (stop < 0)
209 {
210 stop += axis_size;
211 }
212
213 // Clamping
214 // Because the end index points one past the last element, we need slightly
215 // different clamping ranges depending on the direction.
216 if (strides[axis] > 0)
217 {
218 // Forward iteration
219 stop = Clamp(stop, 0, axis_size);
220 }
221 else
222 {
223 // Backward iteration
224 stop = Clamp(stop, -1, axis_size - 1);
225 }
226
227 return stop;
228}

230StridedSliceParams BuildStridedSliceParams(StridedSliceContext *op_context)
231{
232 StridedSliceParams op_params;
233
234 // The ellipsis_mask and new_axis_mask in op_params are not used. Those masks
235 // are processed here to update begin_mask, end_mask and the index range.
236 op_params.begin_mask = 0;
237 op_params.ellipsis_mask = 0;
238 op_params.end_mask = 0;
239 op_params.new_axis_mask = 0;
240 op_params.shrink_axis_mask = 0;
241
242 // Count indexes where the new_axis_mask is set but the ellipsis_mask is not.
243 loco::TensorShape begin_shape = luci::shape_get(op_context->begin).as<loco::TensorShape>();
244 const uint32_t begin_count = begin_shape.dim(0).value();
245 uint32_t num_add_axis = 0;
246 for (uint32_t i = 0; i < begin_count; ++i)
247 {
248 if (!((1 << i) & op_context->params.ellipsis_mask) &&
249 ((1 << i) & op_context->params.new_axis_mask))
250 {
251 num_add_axis++;
252 }
253 }
254
255 // Calculate the dims of input after adding new axises.
256 const uint32_t effective_dims = op_context->input_dims + num_add_axis;
257
258 // If begin, end and strides are not fully provided, it means Ellipsis should
259 // be expanded to multiple dimensions (Ex: for spec [Ellipsis, 2] on a 3D
260 // input, the Ellipsis should be applied for the first 2 dimensions). Besides,
261 // If the new_axis_mask and the ellipsis_mask are set at the same index, the
262 // new_axis_mask will have no effect.
263 int32_t effective_ellipsis_mask = 0, effective_new_axis_mask = 0;
264 uint32_t ellipsis_start_idx = effective_dims, expanded_ellipsis = 0;
265 for (uint32_t i = 0; i < effective_dims;)
266 {
267 if ((1 << i) & op_context->params.ellipsis_mask)
268 {
269 ellipsis_start_idx = i;
270 uint32_t ellipsis_end_idx =
271 std::max(i + 1, std::min(i + 1 + num_add_axis + op_context->input_dims - begin_count,
272 effective_dims));
273 expanded_ellipsis = ellipsis_end_idx - ellipsis_start_idx - 1;
274
275 // Set bit for effective_ellipsis_mask.
276 for (; i < ellipsis_end_idx; ++i)
277 {
278 effective_ellipsis_mask |= (1 << i);
279 }
280 continue;
281 }
282
283 if ((1 << (i - expanded_ellipsis)) & op_context->params.new_axis_mask)
284 {
285 effective_new_axis_mask |= (1 << i);
286 }
287 ++i;
288 }
289
290 // Calculate effective_input_shape and its corresponding begin, end, strides.
291 loco::TensorShape input_shape = luci::shape_get(op_context->input).as<loco::TensorShape>();
292 uint32_t added_ellipsis = 0, added_axises = 0;
293 op_context->effective_input_shape.rank(effective_dims);
294
295 for (uint32_t i = 0; i < effective_dims; ++i)
296 {
297 if ((1 << i) & effective_ellipsis_mask)
298 {
299 // If ellipsis_mask, set the begin_mask and end_mask at that index.
300 added_ellipsis = std::max(0u, i - ellipsis_start_idx);
301 op_params.begin_mask |= (1 << i);
302 op_params.end_mask |= (1 << i);
303 op_params.strides[i] = 1;
304 op_context->effective_input_shape.dim(i) = input_shape.dim(i - added_axises);
305 }
306 else if ((1 << i) & effective_new_axis_mask)
307 {
308 // If new_axis_mask is set, it is equivalent to adding a new dim of 1 to
309 // input tensor. Store added shape to effective_input_shape.
310 op_params.start_indices[i] = 0;
311 op_params.stop_indices[i] = 1;
312 op_params.strides[i] = 1;
313 op_context->effective_input_shape.dim(i) = loco::Dimension(1);
314 added_axises++;
315 }
316 else if (i >= begin_count + expanded_ellipsis)
317 {
318 op_params.start_indices[i] = 0;
319 op_params.stop_indices[i] = 0;
320 op_params.strides[i] = 1;
321 op_params.begin_mask |= (1 << i);
322 op_params.end_mask |= (1 << i);
323 op_context->effective_input_shape.dim(i) = input_shape.dim(i - added_axises);
324 }
325 else
326 {
327 const uint32_t orig_idx = i - added_ellipsis;
328 op_params.start_indices[i] = op_context->begin->at<S32>(orig_idx);
329 op_params.stop_indices[i] = op_context->end->at<S32>(orig_idx);
330 op_params.strides[i] = op_context->strides->at<S32>(orig_idx);
331 if (op_context->params.begin_mask & (1 << orig_idx))
332 {
333 op_params.begin_mask |= (1 << i);
334 }
335 if (op_context->params.end_mask & (1 << orig_idx))
336 {
337 op_params.end_mask |= (1 << i);
338 }
339 if (op_context->params.shrink_axis_mask & (1 << orig_idx))
340 {
341 op_params.shrink_axis_mask |= (1 << i);
342 }
343 op_context->effective_input_shape.dim(i) = input_shape.dim(i - added_axises);
344 }
345 }
346 op_params.start_indices_count = effective_dims;
347 op_params.stop_indices_count = effective_dims;
348 op_params.strides_count = effective_dims;
349
350 return op_params;
351}

358loco::TensorShape infer_output_shape(const CircleStridedSlice *node)
359{
360 loco::TensorShape output_shape;
361
362 auto input_node = loco::must_cast<luci::CircleNode *>(node->input());
363
364 auto begin_node = dynamic_cast<luci::CircleConst *>(node->begin());
365 auto end_node = dynamic_cast<luci::CircleConst *>(node->end());
366 auto strides_node = dynamic_cast<luci::CircleConst *>(node->strides());
367 if (begin_node == nullptr || end_node == nullptr || strides_node == nullptr)
368 {
369 INTERNAL_EXN("StridedSlice begin/end/strides nodes are not Constant");
370 }
371
372 LUCI_ASSERT(begin_node->dtype() == S32, "Only support S32 for begin_node");
373 LUCI_ASSERT(end_node->dtype() == S32, "Only support S32 for end_node");
374 LUCI_ASSERT(strides_node->dtype() == S32, "Only support S32 for strides_node");
375
376 LUCI_ASSERT(begin_node->rank() == 1, "Only support rank 1 for begin_node");
377 LUCI_ASSERT(end_node->rank() == 1, "Only support rank 1 for end_node");
378 LUCI_ASSERT(strides_node->rank() == 1, "Only support rank 1 for strides_node");
379
380 loco::TensorShape input_shape = luci::shape_get(input_node).as<loco::TensorShape>();
381
382 assert(begin_node->size<S32>() <= input_shape.rank());
383 assert(end_node->size<S32>() <= input_shape.rank());
384 assert(strides_node->size<S32>() <= input_shape.rank());
385
386 StridedSliceContext op_context(node);
387 auto op_params = BuildStridedSliceParams(&op_context);
388 auto effective_input_shape = op_context.effective_input_shape;
389 std::vector<int32_t> output_shape_vector;
390
391 for (int32_t idx = effective_input_shape.rank() - 1; idx >= 0; --idx)
392 {
393 int32_t stride = op_params.strides[idx];
394 LUCI_ASSERT(stride != 0, "stride value has to be non-zero");
395
396 int32_t begin = StartForAxis(op_params, effective_input_shape, idx);
397 int32_t end = StopForAxis(op_params, effective_input_shape, idx, begin);
398
399 // When shrinking an axis, the end position does not matter (and can be
400 // incorrect when negative indexing is used, see Issue #19260). Always use
401 // begin + 1 to generate a length 1 slice, since begin has
402 // already been adjusted for negative indices by GetBeginValueAtIndex.
403 const bool shrink_axis = op_params.shrink_axis_mask & (1 << idx);
404 if (shrink_axis)
405 {
406 end = begin + 1;
407 }
408
409 // This is valid for both positive and negative strides
410 int32_t dim_shape = std::ceil((end - begin) / static_cast<float>(stride));
411 dim_shape = dim_shape < 0 ? 0 : dim_shape;
412 if (!shrink_axis)
413 {
414 output_shape_vector.push_back(dim_shape);
415 }
416 }
417
418 auto shape_size = output_shape_vector.size();
419 output_shape.rank(shape_size);
420 for (uint32_t idx = 0; idx < shape_size; ++idx)
421 {
422 // reverse copy
423 output_shape.dim(idx) = output_shape_vector.at(shape_size - 1u - idx);
424 }
425
426 return output_shape;
427}
